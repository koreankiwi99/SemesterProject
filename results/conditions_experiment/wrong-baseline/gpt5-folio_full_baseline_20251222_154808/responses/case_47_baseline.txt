  ∀ p : Person, Student p → WantsLongVacation p → Favorite p = Season.Summer

  Favorite Mia = Season.Spring ∨ Favorite Mia = Season.Summer ∨
  Favorite Mia = Season.Fall ∨ Favorite Mia = Season.Winter := by
  cases hx : Favorite Mia with
