Premises: The Picuris Mountains are a mountain range in New Mexico or Texas.
Juan de Onate visited the Picuris Mountains.
The Harding Pegmatite Mine, located in the Picuris Mountains, was donated.
There are no mountain ranges in texas that have mines that have been donated.
Conclusion: Juan de Onate visited a mountain range in New Mexico.

axiom Mine : Type
axiom Person : Type

axiom Picuris : MountainRange
axiom HardingPegmatiteMine : Mine
axiom Juan_de_Onate : Person

axiom InNewMexico : MountainRange → Prop
axiom InTexas : MountainRange → Prop
axiom LocatedIn : Mine → MountainRange → Prop
axiom Donated : Mine → Prop

-- Premises
axiom premise1 : InNewMexico Picuris ∨ InTexas Picuris
axiom premise2 : Visited Juan_de_Onate Picuris
axiom premise3a : LocatedIn HardingPegmatiteMine Picuris
axiom premise3b : Donated HardingPegmatiteMine
axiom premise4 : ∀ r : MountainRange, InTexas r → ¬ ∃ m : Mine, LocatedIn m r ∧ Donated m

-- Conclusion: Juan de Onate visited a mountain range in New Mexico.
theorem goal : ∃ r : MountainRange, InNewMexico r ∧ Visited Juan_de_Onate r := by
  -- There exists a donated mine located in Picuris
  have exMine : ∃ m : Mine, LocatedIn m Picuris ∧ Donated m := by
    exact ⟨HardingPegmatiteMine, And.intro premise3a premise3b⟩
  -- Therefore Picuris cannot be in Texas (would contradict premise4)
  have notTX : ¬ InTexas Picuris := by
    intro hTX
    have noMine : ¬ ∃ m : Mine, LocatedIn m Picuris ∧ Donated m := premise4 Picuris hTX
    exact noMine exMine
  -- From the disjunction, deduce Picuris is in New Mexico
  have inNM : InNewMexico Picuris :=
    Or.elim premise1
      (fun hnm => hnm)
      (fun htx => False.elim (notTX htx))
  -- Conclude existence of a mountain range in New Mexico visited by Juan de Onate
  exact ⟨Picuris, And.intro inNM premise2⟩
