Premises: Susan flies to LGA airport.
The departure and arrival can not be at the same airport.
John flies from LGA airport.
Conclusion: Susan flies from LGA airport.

-- We formalize persons, airports, and flying relations.
axiom Person : Type
axiom Airport : Type

axiom Susan : Person
axiom John : Person
axiom LGA : Airport

axiom FliesFrom : Person → Airport → Prop
axiom FliesTo   : Person → Airport → Prop

-- Premises:
-- 1) Susan flies to LGA airport.
axiom susan_to_LGA : FliesTo Susan LGA
-- 2) The departure and arrival cannot be at the same airport (for any person and airport).
axiom not_from_and_to_same_airport : ∀ p a, ¬ (FliesFrom p a ∧ FliesTo p a)
-- 3) John flies from LGA airport.
axiom john_from_LGA : FliesFrom John LGA

-- Conclusion to analyze (as a theorem to attempt to prove):
theorem goal : FliesFrom Susan LGA := by
  -- This goal is not derivable from the given premises.
  -- In fact, from premises we can prove ¬ FliesFrom Susan LGA (see lemma below),
  -- so this theorem is unprovable without contradicting the premises.
  sorry

-- We can in fact derive the negation of the conclusion from the premises:
theorem not_susan_from_LGA : ¬ FliesFrom Susan LGA := by
  intro h_from
  -- If Susan flew from LGA and to LGA, that contradicts the "not same airport" axiom.
  have h_and : FliesFrom Susan LGA ∧ FliesTo Susan LGA := And.intro h_from susan_to_LGA
  exact (not_from_and_to_same_airport Susan LGA) h_and

-- Equivalently, the conclusion implies False:
theorem goal_contradicts_premises : FliesFrom Susan LGA → False := by
  exact not_susan_from_LGA
