Premises: Susan flies to LGA airport.
The departure and arrival can not be at the same airport.
John flies from LGA airport.
Conclusion: John flies to LGA airport.

axiom Airport : Type

-- Constants

axiom FliesTo   : Person → Airport → Prop

-- Premises

axiom no_same_airport : ∀ (p : Person) (a : Airport), ¬ (FliesFrom p a ∧ FliesTo p a)

-- A derived fact: from the premises, John does not fly to LGA.

  intro hTo

  have hContra := no_same_airport John LGA
  exact hContra hAnd

-- Conclusion to analyze: John flies to LGA airport.

  sorry
