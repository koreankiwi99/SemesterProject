-- Premise 1: Events are either happy or sad (mutually exclusive and exhaustive).
axiom premise1 : ∀ e : Event, (Happy e ∧ ¬ Sad e) ∨ (Sad e ∧ ¬ Happy e)

-- Premise 2: At least one event is happy.
axiom premise2 : ∃ e : Event, Happy e

-- Goal: The conclusion "All events are sad" is false.
theorem goal : ¬ (∀ e : Event, Sad e) := by
  intro allSad
  obtain ⟨e, he⟩ := premise2
  have h := premise1 e
  cases h with
  | inl h1 =>
      -- h1 : Happy e ∧ ¬Sad e
      have hn : ¬ Sad e := h1.right
      have hs : Sad e := allSad e
      exact hn hs
  | inr h2 =>
      -- h2 : Sad e ∧ ¬Happy e contradicts he : Happy e
      exact h2.right he
