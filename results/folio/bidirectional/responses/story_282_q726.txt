Premises: All squares are four-sided.
All four-sided things are shapes. 
Conclusion: All squares are shapes.

axiom Entity : Type
axiom Square : Entity → Prop
axiom FourSided : Entity → Prop
axiom Shape : Entity → Prop

-- Premises
axiom all_squares_four_sided : ∀ x : Entity, Square x → FourSided x
axiom all_four_sided_shapes : ∀ x : Entity, FourSided x → Shape x

-- Conclusion: All squares are shapes.
theorem all_squares_shapes : ∀ x : Entity, Square x → Shape x := by
  intro x
  intro hx
  have h4 : FourSided x := all_squares_four_sided x hx
  have hs : Shape x := all_four_sided_shapes x h4
  exact hs
