Premises: All growth stocks are bought to earn profits from rapid price appreciation.
If the stock price is bought to earn profits from rapid price appreciation, then it is not suitable for a retirement fund.
Some stocks are growth stocks.
All mature stocks are suitable for a retirement fund.
KO is a mature stock.
Conclusion: If KO is a growth stock or bought to earn profits from rapid price appreciation, then KO is neither a stock nor is its price volatile.

PROVE TRUE ATTEMPT
======================================================================

axiom Entity : Type
axiom KO : Entity

-- Predicates
axiom Growth : Entity → Prop
axiom BoughtRapid : Entity → Prop
axiom Suitable : Entity → Prop
axiom Mature : Entity → Prop
axiom IsStock : Entity → Prop
axiom PriceVolatile : Entity → Prop

-- Premises
axiom growth_implies_bought : ∀ x, Growth x → BoughtRapid x
axiom bought_implies_not_suitable : ∀ x, BoughtRapid x → ¬ Suitable x
axiom some_stock_growth : ∃ x, IsStock x ∧ Growth x
axiom mature_implies_suitable : ∀ x, Mature x → Suitable x
axiom ko_mature : Mature KO

-- Conclusion to prove
theorem goal :
  (Growth KO ∨ BoughtRapid KO) → (¬ IsStock KO ∧ ¬ PriceVolatile KO) := by
  intro h
  have h_suitable : Suitable KO := mature_implies_suitable KO ko_mature
  have not_growth : ¬ Growth KO := by
    intro g
    have br : BoughtRapid KO := growth_implies_bought KO g
    have ns : ¬ Suitable KO := bought_implies_not_suitable KO br
    exact ns h_suitable
  have not_bought : ¬ BoughtRapid KO := by
    intro br
    have ns : ¬ Suitable KO := bought_implies_not_suitable KO br
    exact ns h_suitable
  have contra : False := by
    cases h with
    | inl g  => exact not_growth g
    | inr br => exact not_bought br
  exact False.elim contra

-- In fact, from the premises we can prove the original conclusion (via explosion),
