Premises: No one nice to animals is also mean to animals.
Some grumpy people are mean to animals.
All animal lovers are nice to animals.
All pet owners love animals.
Tom is a pet owner.
Conclusion: If Tom is not both a grumpy person and mean to animals, then Tom is neither a grumpy person nor an animal lover.

axiom Person : Type
axiom Tom : Person

axiom NiceToAnimals : Person → Prop
axiom MeanToAnimals : Person → Prop
axiom Grumpy : Person → Prop
axiom AnimalLover : Person → Prop
axiom PetOwner : Person → Prop

-- Premises
-- 1. No one nice to animals is also mean to animals.
axiom P1 : ∀ x : Person, NiceToAnimals x → ¬ MeanToAnimals x
-- 2. Some grumpy people are mean to animals.
axiom P2 : ∃ x : Person, Grumpy x ∧ MeanToAnimals x
-- 3. All animal lovers are nice to animals.
axiom P3 : ∀ x : Person, AnimalLover x → NiceToAnimals x
-- 4. All pet owners love animals.
axiom P4 : ∀ x : Person, PetOwner x → AnimalLover x
-- 5. Tom is a pet owner.
axiom P5 : PetOwner Tom

-- Useful derived facts about Tom
lemma Tom_animallover : AnimalLover Tom := by
  exact P4 Tom P5

lemma Tom_nice : NiceToAnimals Tom := by
  exact P3 Tom Tom_animallover

lemma Tom_not_mean : ¬ MeanToAnimals Tom := by
  exact P1 Tom Tom_nice

-- Conclusion to analyze (as a theorem)
-- If Tom is not both a grumpy person and mean to animals, then Tom is neither a grumpy person nor an animal lover.
theorem conclusion :
  (¬ (Grumpy Tom ∧ MeanToAnimals Tom)) → (¬ Grumpy Tom ∧ ¬ AnimalLover Tom) := by
  -- This conclusion does not follow from the premises; see the refutation below.
  sorry

-- Refutation: the conclusion contradicts the premises
theorem not_conclusion :
  ¬ ((¬ (Grumpy Tom ∧ MeanToAnimals Tom)) → (¬ Grumpy Tom ∧ ¬ AnimalLover Tom)) := by
  intro h
  -- Antecedent holds, because Tom is not mean to animals
  have ant : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) := by
    intro hgm
    exact Tom_not_mean hgm.right
  -- Apply the implication to get the consequent, which contradicts that Tom is an animal lover
  have cons := h ant
  have notAl : ¬ AnimalLover Tom := cons.right
  have al : AnimalLover Tom := Tom_animallover
  exact notAl al
