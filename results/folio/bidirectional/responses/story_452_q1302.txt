Premises: All aliens are extraterrestrials.
If someone is from Mars, then they are an alien.
No extraterrestrials are human.
All highly intelligent beings from Earth are humans.
Marvin is a highly intelligent being.
Marvin is either from Earth and from Mars, or he is from neither.
If Marvin is not from Earth, then Marvin is an extraterrestrial.
Conclusion: Marvin is neither a human nor from Mars.

PROVE TRUE ATTEMPT
======================================================================
Lean Code:
axiom Being : Type
axiom Alien : Being → Prop
axiom Extraterrestrial : Being → Prop
axiom Human : Being → Prop
axiom FromMars : Being → Prop
axiom FromEarth : Being → Prop
axiom HighlyIntelligent : Being → Prop
axiom Marvin : Being

-- Premises
axiom ax1 : ∀ x : Being, Alien x → Extraterrestrial x
axiom ax2 : ∀ x : Being, FromMars x → Alien x
axiom ax3 : ∀ x : Being, Extraterrestrial x → ¬ Human x
axiom ax4 : ∀ x : Being, HighlyIntelligent x ∧ FromEarth x → Human x
axiom ax5 : HighlyIntelligent Marvin
axiom ax6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom ax7 : ¬ FromEarth Marvin → Extraterrestrial Marvin

-- Conclusion: Marvin is neither a human nor from Mars.
theorem goal : ¬ Human Marvin ∧ ¬ FromMars Marvin := by
  cases ax6 with
  | inl hBoth =>
    have hFM : FromMars Marvin := hBoth.right
    have hFE : FromEarth Marvin := hBoth.left
    have hAlien : Alien Marvin := ax2 Marvin hFM
    have hExtra : Extraterrestrial Marvin := ax1 Marvin hAlien
    have hNotHuman : ¬ Human Marvin := ax3 Marvin hExtra
    have hHuman : Human Marvin := ax4 Marvin ⟨ax5, hFE⟩
    have hFalse : False := hNotHuman hHuman
    exact False.elim hFalse
  | inr hNeither =>
    have hNotE : ¬ FromEarth Marvin := hNeither.left
    have hNotM : ¬ FromMars Marvin := hNeither.right
    have hExtra : Extraterrestrial Marvin := ax7 hNotE
    have hNotHuman : ¬ Human Marvin := ax3 Marvin hExtra
    exact And.intro hNotHuman hNotM
