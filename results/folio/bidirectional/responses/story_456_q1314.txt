Premises: Some professional basketball players are not American nationals.
All professional basketball players can do jump shots.
If someone can jump shots, they leap straight into the air.
If someone leaps straight into the air, they activate their leg muscles. 
Yuri does not activate his leg muscles.
Conclusion: Yuri is not an American professional basketball player.

axiom Person : Type

-- Predicates
axiom ProfessionalBasketballPlayer : Person → Prop
axiom AmericanNational : Person → Prop
axiom CanDoJumpShots : Person → Prop
axiom LeapsStraightIntoTheAir : Person → Prop
axiom ActivatesLegMuscles : Person → Prop

-- Premises

axiom all_PBP_can_jump_shots : ∀ x : Person, ProfessionalBasketballPlayer x → CanDoJumpShots x
axiom jump_shots_implies_leap : ∀ x : Person, CanDoJumpShots x → LeapsStraightIntoTheAir x
axiom leap_implies_activate : ∀ x : Person, LeapsStraightIntoTheAir x → ActivatesLegMuscles x
axiom yuri_not_activate : ¬ ActivatesLegMuscles Yuri

-- Conclusion: Yuri is not an American professional basketball player.
theorem Yuri_not_American_PBP : ¬ (AmericanNational Yuri ∧ ProfessionalBasketballPlayer Yuri) := by
  intro h
  have hPBP : ProfessionalBasketballPlayer Yuri := h.right
  have hJump : CanDoJumpShots Yuri := all_PBP_can_jump_shots _ hPBP
  have hLeap : LeapsStraightIntoTheAir Yuri := jump_shots_implies_leap _ hJump
  have hAct : ActivatesLegMuscles Yuri := leap_implies_activate _ hLeap
  exact yuri_not_activate hAct
