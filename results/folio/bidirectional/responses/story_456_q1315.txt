-- 4. If someone leaps straight into the air, they activate their leg muscles.

-- Useful derived lemmas from the premises

  intro hx
  have hj : CanDoJumpShots x := all_pro_can_jump x hx
  have hl : LeapsStraightUp x := can_jump_implies_leap x hj
  exact leap_implies_activate x hl

  exact yuri_not_activate hact

-- This shows the given conclusion contradicts the premises
theorem contradicts_conclusion : ¬ (American Yuri ∧ ProBasketballPlayer Yuri) := by
