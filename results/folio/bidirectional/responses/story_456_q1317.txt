axiom pro_can_jump : ∀ x : Person, Pro x → JumpShot x
axiom jump_implies_leap : ∀ x : Person, JumpShot x → LeapStraightUp x
axiom leap_implies_activate : ∀ x : Person, LeapStraightUp x → ActivateLeg x
axiom yuri_not_activate : ¬ActivateLeg Yuri

-- We prove the negation of the conclusion:
-- Conclusion: If Yuri is not an American professional basketball player, then Yuri is a professional basketball player.
-- Formalized as: (¬(American Yuri ∧ Pro Yuri)) → Pro Yuri
-- Negation to prove: ¬((¬(American Yuri ∧ Pro Yuri)) → Pro Yuri)
theorem negate_conclusion : ¬((¬(American Yuri ∧ Pro Yuri)) → Pro Yuri) := by
  -- From the premises, show that if Yuri were a pro, he'd activate leg muscles.
  have pro_to_activate : Pro Yuri → ActivateLeg Yuri := by
    intro hPro
    have hJump : JumpShot Yuri := pro_can_jump Yuri hPro
    have hLeap : LeapStraightUp Yuri := jump_implies_leap Yuri hJump
    exact leap_implies_activate Yuri hLeap
  -- Therefore, since Yuri does not activate leg muscles, Yuri is not a pro.
  have not_pro : ¬Pro Yuri := by
    intro hPro
    have hAct : ActivateLeg Yuri := pro_to_activate hPro
    exact yuri_not_activate hAct
  -- From ¬Pro Yuri, infer ¬(American Yuri ∧ Pro Yuri).
  have not_A_and_P : ¬(American Yuri ∧ Pro Yuri) := by
    intro hAnd
    exact not_pro hAnd.right
  -- Use these to refute the implication.
  intro hImp
  have hPro : Pro Yuri := hImp not_A_and_P
  exact not_pro hPro
