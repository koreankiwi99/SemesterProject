axiom Terricolous : Entity → Prop

-- Premises
axiom premise1 : ∀ x : Entity, (Animal x ∧ Barks x) → ¬Likes Jane x
axiom premise2 : ∀ x : Entity, Dog x → (Animal x ∧ Barks x)
axiom premise3 : ∀ x : Entity, (Animal x ∧ Jumps x) → Likes Jane x
axiom premise4 : ∀ x : Entity, (Animal x ∧ HasLegs x) → Jumps x
axiom premise5 : ∀ x : Entity, (Animal x ∧ Terricolous x) → HasLegs x

axiom premise7 : (¬Jumps KiKi ∧ ¬HasLegs KiKi) → (Terricolous KiKi ∨ HasLegs KiKi)

-- We prove the negation of the conclusion: not (KiKi is a barking dog)
theorem neg_conclusion : ¬(Dog KiKi ∧ Barks KiKi) := by
  classical
  intro h
  have hDog : Dog KiKi := h.left
  have hBark : Barks KiKi := h.right
  -- From "Jane does not like any animals that bark"
  have hNotLike : ¬Likes Jane KiKi := by
    apply premise1
    exact And.intro premise6 hBark
  -- We will derive that Jane likes KiKi, contradicting hNotLike
  have hLike : Likes Jane KiKi := by
    -- It suffices to show that KiKi jumps (since Animal KiKi holds)
    by_cases hL : HasLegs KiKi
    · -- If KiKi has legs, then it jumps, hence Jane likes KiKi
      have hJ : Jumps KiKi := premise4 KiKi (And.intro premise6 hL)
      exact premise3 KiKi (And.intro premise6 hJ)
    · -- If KiKi does not have legs, split on whether it jumps
      by_cases hJ : Jumps KiKi
      · exact premise3 KiKi (And.intro premise6 hJ)
      · -- Impossible branch: from neither jumps nor has legs, deduce contradiction

        have hDisj : Terricolous KiKi ∨ HasLegs KiKi := premise7 hConj
        have hFalse : False := by
          cases hDisj with
          | inl hT =>
            have hLegs : HasLegs KiKi := premise5 KiKi (And.intro premise6 hT)
            exact hL hLegs
          | inr hLegs =>
            exact hL hLegs
        exact False.elim hFalse
  exact hNotLike hLike
