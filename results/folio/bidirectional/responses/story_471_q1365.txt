-- We prove the negation of the conclusion:
-- ¬((Cute Rockie ∧ Calm Rockie) → (Skittish Rockie ∧ Turtle Rockie))
theorem negate_conclusion :
  ¬ ((Cute Rockie ∧ Calm Rockie) → (Skittish Rockie ∧ Turtle Rockie)) := by
  intro h
  -- From p6
  have spottedR : Spotted Rockie := p6.left
  have calmR : Calm Rockie := p6.right
  -- Rockie cannot be a squirrel (would force skittish, contradicting calm)
  have notSquirrelR : ¬ Squirrel Rockie := by
    intro sq
    have sk : Skittish Rockie := p5 Rockie ⟨sq, spottedR⟩
    have ncalm : ¬ Calm Rockie := p4 Rockie sk
    exact ncalm calmR
  -- From "only spotted are rabbits or squirrels" and not-squirrel, get rabbit
  have rs : Rabbit Rockie ∨ Squirrel Rockie := p3 Rockie spottedR
  have rabbitR : Rabbit Rockie :=
    Or.elim rs (fun r => r) (fun sq => False.elim (notSquirrelR sq))
  -- Rabbits that are spotted are cute
  have cuteR : Cute Rockie := p1 Rockie ⟨rabbitR, spottedR⟩
  -- Build antecedent
  have antecedent : Cute Rockie ∧ Calm Rockie := And.intro cuteR calmR
  -- Apply h to get the consequent
  have consequent : Skittish Rockie ∧ Turtle Rockie := h antecedent
  -- But Skittish Rockie contradicts Calm Rockie via p4
  have notSkittishR : ¬ Skittish Rockie := by
    intro sk
    have ncalm : ¬ Calm Rockie := p4 Rockie sk
    exact ncalm calmR
  exact notSkittishR consequent.left
