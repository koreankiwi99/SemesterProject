Premises: Animals are either invertebrates or vertebrates.
All animals with backbones reproduce by male-and-female mating. 
All vertebrate animals have a backbone.
All bees do not reproduce by male-and-female mating.
All queen bees are bees.
Harry is a bee.
Conclusion: If Harry is either both a vertebrate and an animal with a backbone, or neither a vertebrate nor an animal with a backbone, then Harry is neither an invertebrate nor a queen bee.

-- Conclusion to analyze
theorem goal :

        intro _

        intro _
        exact False.elim contra
      exact And.intro notInv notQB
  | inr hnb =>
      -- From this branch alone, the desired conclusion does not follow from the given premises.

-- Premises
axiom premise1 : ∀ x : Animal, Invertebrate x ∨ Vertebrate x
axiom premise2 : ∀ x : Animal, Backbone x → ReproMF x
axiom premise3 : ∀ x : Animal, Vertebrate x → Backbone x
axiom premise4 : ∀ x : Animal, Bee x → ¬ReproMF x
axiom premise5 : ∀ x : Animal, QueenBee x → Bee x
axiom premise6 : Bee Harry

-- Negation of the conclusion:
-- ¬( ((Vertebrate Harry ∧ Backbone Harry) ∨ (¬Vertebrate Harry ∧ ¬Backbone Harry))
--     → (¬Invertebrate Harry ∧ ¬QueenBee Harry) )
theorem neg_conclusion :
  ¬( ((Vertebrate Harry ∧ Backbone Harry) ∨ (¬Vertebrate Harry ∧ ¬Backbone Harry))
      → (¬Invertebrate Harry ∧ ¬QueenBee Harry) ) := by
  intro h
  -- From Bee Harry and backbone→reproduceMF, get ¬Backbone Harry
  have hNotMF : ¬ReproMF Harry := by
    exact premise4 Harry premise6
  have hNotBack : ¬Backbone Harry := by
    intro hBack
    have hMF : ReproMF Harry := premise2 Harry hBack
    exact hNotMF hMF
  -- From Vertebrate→Backbone and ¬Backbone, get ¬Vertebrate Harry
  have hNotVert : ¬Vertebrate Harry := by
    intro hVert
    have hBack : Backbone Harry := premise3 Harry hVert
    exact hNotBack hBack
  -- The antecedent holds via the right disjunct
  have hAnte :
      (Vertebrate Harry ∧ Backbone Harry) ∨ (¬Vertebrate Harry ∧ ¬Backbone Harry) :=
    Or.inr ⟨hNotVert, hNotBack⟩
  -- Apply the implication to get the consequent
  have hCons : ¬Invertebrate Harry ∧ ¬QueenBee Harry := h hAnte
  have nInv : ¬Invertebrate Harry := hCons.left
  -- From premise1 and ¬Vertebrate, deduce Invertebrate Harry
  have invOrVert : Invertebrate Harry ∨ Vertebrate Harry := premise1 Harry
  have hInv : Invertebrate Harry := by
    cases invOrVert with
    | inl inv => exact inv
    | inr v   => exact False.elim (hNotVert v)
  -- Contradiction
  exact nInv hInv
