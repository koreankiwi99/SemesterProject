<lean>
axiom Person : Type

axiom PerformsOften : Person → Prop
axiom AttendsEngaged : Person → Prop
axiom InactiveDisinterested : Person → Prop
axiom Chaperones : Person → Prop
axiom Student : Person → Prop
axiom YoungChildOrTeen : Person → Prop
axiom WishFurther : Person → Prop

axiom Bonnie : Person

-- People who perform often are engaged
axiom A1 : ∀ p, PerformsOften p → AttendsEngaged p

-- Everyone either performs often or is inactive
axiom A2 : ∀ p, PerformsOften p ∨ InactiveDisinterested p

-- Chaperones are not students
axiom A3 : ∀ p, Chaperones p → ¬ Student p

-- Inactive members chaperone
axiom A4 : ∀ p, InactiveDisinterested p → Chaperones p

-- Young/teen with wish is student
axiom A5 : ∀ p, YoungChildOrTeen p → WishFurther p → Student p

-- Bonnie's engagement/student status: equivalent to (AttendsEngaged ↔ Student)
axiom A6 : (AttendsEngaged Bonnie ∧ Student Bonnie) ∨ (¬ AttendsEngaged Bonnie ∧ ¬ Student Bonnie)

-- The statement to prove: Antecedent → Consequent
theorem bonnie_statement : 
    ((YoungChildOrTeen Bonnie ∧ WishFurther Bonnie ∧ Chaperones Bonnie) ∨ 
    (¬ YoungChildOrTeen Bonnie ∧ ¬ WishFurther Bonnie)) →
    (Student Bonnie ∨ InactiveDisinterested Bonnie) := by
  intro hAntecedent
  -- Case split on Bonnie's status from A6
  cases A6 with
  | inl hCase1 => 
    -- Case 1: AttendsEngaged and Student
    left; exact hCase1.right
  | inr hCase2 => 
    -- Case 2: Not AttendsEngaged and not Student
    -- Show Bonnie is inactive
    have noPerform : ¬ PerformsOften Bonnie := by
      intro hPerform
      have hEngaged := A1 Bonnie hPerform
      exact hCase2.left hEngaged
    -- By A2, Bonnie must be inactive
    cases A2 Bonnie with
    | inl hPerform => contradiction
    | inr hInactive => right; exact hInactive

-- Since we proved the theorem, the statement is true
</lean>

ANSWER: True
