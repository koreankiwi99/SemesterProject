axiom FromBakedByMelissa : obj → Prop

-- Context axioms
axiom A1 : ∀ x, BakedSweet x → ¬ Spicy x                     -- No baked sweets are spicy
axiom A2 : ∀ x, Cupcake x → BakedSweet x                     -- All cupcakes are baked sweets
axiom A3 : ∀ x, MalaHotpot x → Spicy x                       -- All mala hotpots are spicy
axiom A4 : ∀ x, FromBakedByMelissa x → Cupcake x             -- All products from Baked by Melissa are cupcakes
axiom A5 : ∀ x, DriedThaiChilies x → (Spicy x ∨ MalaHotpot x ∨ ¬ BakedSweet x)  -- Dried Thai chilies are spicy or mala hotpots or not baked sweets

-- Theorem: Dried Thai chilies are not products of Baked by Melissa
theorem thm : ∀ x, DriedThaiChilies x → ¬ FromBakedByMelissa x := by
  intro x h_dtc   -- Assume x is a Dried Thai chili
  by_contra h_fbbm  -- Assume, for contradiction, that x is from Baked by Melissa
  -- From A4 and A2, x must be a baked sweet
  have h_cup : Cupcake x := A4 x h_fbbm
  have h_bs : BakedSweet x := A2 x h_cup
  -- From A5, either (Spicy x) or (MalaHotpot x) or (¬BakedSweet x)
  have h_disj : Spicy x ∨ MalaHotpot x ∨ ¬ BakedSweet x := A5 x h_dtc
  -- Since we know h_bs, the third disjunct is false
  cases h_disj with
  | inl sp => 
    -- Spicy case: contradicts A1 that baked sweets aren't spicy
    have n_sp : ¬ Spicy x := A1 x h_bs
    contradiction
  | inr rest => 
    cases rest with
    | inl mh => 
      -- MalaHotpot case: leads to Spicy, which also contradicts
      have sp : Spicy x := A3 x mh
      have n_sp : ¬ Spicy x := A1 x h_bs
      contradiction
    | inr n_bs => 
      -- Not baked sweet case: contradicts h_bs
      contradiction
-- Since all cases lead to contradiction, the assumption must be false

2. If a mala hotpot (thus spicy), same contradiction  
3. If not a baked sweet, contradicting that Baked by Melissa products are baked sweets  
Therefore, the statement "Dried Thai chilies are not products of Baked by Melissa" is true.
