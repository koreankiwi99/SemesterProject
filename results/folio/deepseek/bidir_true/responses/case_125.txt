<lean>
axiom Pet : Type
axiom Rock : Pet
axiom Mammal : Pet → Prop
axiom Animal : Pet → Prop
axiom Monkey : Pet → Prop
axiom Bird : Pet → Prop
axiom Fly : Pet → Prop
axiom Breathe : Pet → Prop
axiom Wings : Pet → Prop

axiom A1 : ∀ (p : Pet), Mammal p → Animal p
axiom A2 : ∀ (p : Pet), Monkey p → Mammal p
axiom A3 : ∀ (p : Pet), Monkey p ∨ Bird p
axiom A4 : ∀ (p : Pet), Bird p → Fly p
axiom A5 : ∀ (p : Pet), Animal p → Breathe p
axiom A6 : ∀ (p : Pet), Fly p → Wings p
axiom A7 : Fly Rock ∨ Bird Rock ∨ ¬ Breathe Rock

theorem rock_has_wings : Wings Rock := by
  apply A6  -- To show Wings Rock, we need Fly Rock
  -- Case analysis on A7
  cases A7 with
  | inl fly_rock => exact fly_rock
  | inr rest =>
    cases rest with
    | inl bird_rock => exact A4 Rock bird_rock
    | inr not_breathe =>
      -- Show Rock must be a bird (since monkey leads to contradiction)
      have pet_type := A3 Rock
      cases pet_type with
      | inl monkey_rock =>
        have mammal_rock := A2 Rock monkey_rock
        have animal_rock := A1 Rock mammal_rock
        have breathe_rock := A5 Rock animal_rock
        contradiction
      | inr bird_rock => exact A4 Rock bird_rock
</lean>

ANSWER: True
