axiom obj : Type
axiom HeinrichSchmidt : obj
axiom GermanPolitician : obj → Prop
axiom PrussianStateParliamentMember : obj → Prop
axiom NaziReichstagMember : obj → Prop

axiom fact1 : GermanPolitician HeinrichSchmidt
axiom fact2 : PrussianStateParliamentMember HeinrichSchmidt
axiom fact3 : NaziReichstagMember HeinrichSchmidt

    GermanPolitician x ∧ 
    PrussianStateParliamentMember x ∧ 
    NaziReichstagMember x := by
  apply Exists.intro HeinrichSchmidt
  constructor
  · exact fact1
  · constructor
    · exact fact2
    · exact fact3
