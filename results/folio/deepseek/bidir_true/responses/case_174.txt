<lean>
axiom Person : Type
axiom Taylor : Person

-- Predicates
axiom EatsSaladsRegularly : Person → Prop
axiom HealthConscious : Person → Prop
axiom GrewUpHealthConscious : Person → Prop
axiom FulfillsNutrition : Person → Prop
axiom DisregardsWellBeing : Person → Prop
axiom VisitsGymDaily : Person → Prop

-- Axioms
axiom axiom1 : ∀ p : Person, EatsSaladsRegularly p → HealthConscious p
axiom axiom2 : ∀ p : Person, GrewUpHealthConscious p → EatsSaladsRegularly p
axiom axiom3 : ∀ p : Person, FulfillsNutrition p → GrewUpHealthConscious p
axiom axiom4 : ∀ p : Person, DisregardsWellBeing p → ¬ HealthConscious p
axiom axiom5 : ∀ p : Person, VisitsGymDaily p → FulfillsNutrition p
axiom axiom6 : (GrewUpHealthConscious Taylor ∧ DisregardsWellBeing Taylor) ∨ (¬ GrewUpHealthConscious Taylor ∧ ¬ DisregardsWellBeing Taylor)

-- Statement to prove: Taylor neither grew up health-conscious nor visits gym daily.
theorem taylor_case : ¬ GrewUpHealthConscious Taylor ∧ ¬ VisitsGymDaily Taylor := by
  cases axiom6 with
  | inl case1 =>
      let ⟨h_grew, h_disregard⟩ := case1
      have h_health_conscious : HealthConscious Taylor := axiom1 Taylor (axiom2 Taylor h_grew)
      have h_not_health : ¬ HealthConscious Taylor := axiom4 Taylor h_disregard
      exact False.elim (h_not_health h_health_conscious)
  | inr case2 =>
      let ⟨not_grew, not_disregard⟩ := case2
      constructor
      · exact not_grew
      · intro h_gym
        have h_nutrition := axiom5 Taylor h_gym
        have h_grew' := axiom3 Taylor h_nutrition
        exact not_grew h_grew'
</lean>

ANSWER: True
