<lean>
axiom obj : Type
axiom Marvin : obj

axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom FromMars : obj → Prop
axiom Human : obj → Prop
axiom HighlyIntelligent : obj → Prop
axiom FromEarth : obj → Prop

axiom A1 : ∀ x, Alien x → Extraterrestrial x
axiom A2 : ∀ x, FromMars x → Alien x
axiom A3 : ∀ x, Extraterrestrial x → ¬ Human x
axiom A4 : ∀ x, HighlyIntelligent x → FromEarth x → Human x
axiom A5 : HighlyIntelligent Marvin
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom A7 : ¬ FromEarth Marvin → Extraterrestrial Marvin

theorem Marvin_neither : ¬ Human Marvin ∧ ¬ FromMars Marvin := by
  cases A6 with
  | inl h =>
      have ⟨hEarth, hMars⟩ := h
      have humanMarvin : Human Marvin := A4 Marvin A5 hEarth
      have alienMarvin : Alien Marvin := A2 Marvin hMars
      have extraterrestrialMarvin : Extraterrestrial Marvin := A1 Marvin alienMarvin
      have notHumanMarvin := A3 Marvin extraterrestrialMarvin
      contradiction
  | inr h =>
      have ⟨notEarth, notMars⟩ := h
      have extraterrestrialMarvin : Extraterrestrial Marvin := A7 notEarth
      have notHumanMarvin := A3 Marvin extraterrestrialMarvin
      exact ⟨notHumanMarvin, notMars⟩
</lean>

ANSWER: True
