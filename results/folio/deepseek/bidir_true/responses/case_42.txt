<lean>
axiom obj : Type
axiom Coco : obj
axiom RankedHigh : obj → Prop
axiom MostActive : obj → Prop
axiom LostToIga : obj → Prop
axiom Female : obj → Prop
axiom Male : obj → Prop
axiom AtRG2022 : obj → Prop
axiom LostToRafa : obj → Prop

axiom A1 : ∀ x, RankedHigh x → MostActive x
axiom A2 : ∀ x, LostToIga x → RankedHigh x
axiom A3 : ∀ x, (Female x ∧ AtRG2022 x) → LostToIga x
axiom A4 : ∀ x, AtRG2022 x → Female x ∨ Male x
axiom A5 : ∀ x, (Male x ∧ AtRG2022 x) → LostToRafa x
axiom A6 : (RankedHigh Coco ∨ LostToRafa Coco) → ¬ (Male Coco ∧ AtRG2022 Coco)
axiom A7 : AtRG2022 Coco

theorem coco_most_active : MostActive Coco := by
  have hRG := A7
  have gender := A4 Coco hRG
  cases gender with
  | inl female =>
    have hLostIga := A3 Coco ⟨female, hRG⟩
    have hRankHigh := A2 Coco hLostIga
    exact A1 Coco hRankHigh
  | inr male =>
    have hLostRafa := A5 Coco ⟨male, hRG⟩
    have hOr : RankedHigh Coco ∨ LostToRafa Coco := Or.inr hLostRafa
    have notMaleRG := A6 hOr
    have maleRG : Male Coco ∧ AtRG2022 Coco := ⟨male, hRG⟩
    contradiction
</lean>

ANSWER: True
