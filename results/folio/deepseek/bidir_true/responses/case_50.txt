<lean>
axiom obj : Type
axiom NineteenEightyFour : obj
axiom Digital : obj → Prop
axiom Analog : obj → Prop
axiom PrintedText : obj → Prop
axiom StreamingService : obj → Prop
axiom HardcoverBook : obj → Prop

-- No digital media are analog
axiom no_digital_analog : ∀ x : obj, Digital x → ¬ Analog x

-- Every printed text is analog media
axiom printed_analog : ∀ x : obj, PrintedText x → Analog x

-- All streaming services are digital media
axiom streaming_digital : ∀ x : obj, StreamingService x → Digital x

-- If hardcover book then printed text
axiom hardcover_printed : ∀ x : obj, HardcoverBook x → PrintedText x

-- If 1984 is streaming service then it's hardcover book
axiom conditional : StreamingService NineteenEightyFour → HardcoverBook NineteenEightyFour

-- Proof that 1984 is not a streaming service
theorem not_streaming : ¬ StreamingService NineteenEightyFour := by
  intro h -- Assume StreamingService NineteenEightyFour
  have h_digital := streaming_digital NineteenEightyFour h
  have h_hardcover := conditional h
  have h_printed := hardcover_printed NineteenEightyFour h_hardcover
  have h_analog := printed_analog NineteenEightyFour h_printed
  exact no_digital_analog NineteenEightyFour h_digital h_analog
</lean>

ANSWER: True
