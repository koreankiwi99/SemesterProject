<lean>
axiom obj : Type
axiom Ted : obj
axiom Pet : obj → Prop
axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop

axiom AllBrownSwissAreCows : ∀ x : obj, BrownSwissCattle x → Cow x
axiom SomePetsAreBrownSwiss : ∃ x : obj, Pet x ∧ BrownSwissCattle x
axiom AllCowsAreDomesticated : ∀ x : obj, Cow x → Domesticated x
axiom AlligatorsNotDomesticated : ∀ x : obj, Alligator x → ¬Domesticated x
axiom TedIsAlligator : Alligator Ted

theorem disprove_statement : ¬(Pet Ted ∧ BrownSwissCattle Ted) :=
  fun h : Pet Ted ∧ BrownSwissCattle Ted =>
  have brownTed : BrownSwissCattle Ted := h.2
  have cowTed : Cow Ted := AllBrownSwissAreCows Ted brownTed
  have domTed : Domesticated Ted := AllCowsAreDomesticated Ted cowTed
  have notDomTed : ¬Domesticated Ted := AlligatorsNotDomesticated Ted TedIsAlligator
  show False from notDomTed domTed
</lean>

ANSWER: False
