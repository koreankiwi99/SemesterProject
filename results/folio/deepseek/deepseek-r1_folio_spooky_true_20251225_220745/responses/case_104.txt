<lean>
axiom obj : Type
axiom Tom : obj
axiom GREApplicant : obj → Prop
axiom ProvesEconomicHardship : obj → Prop
axiom FinancialAid : obj → Prop
axiom LivesInSingleParentFamily : obj → Prop
axiom HasFewResources : obj → Prop

-- ETS provides financial aid to those GRE applicants who prove economic hardship.
axiom R1 : ∀ x : obj, GREApplicant x ∧ ProvesEconomicHardship x → FinancialAid x

-- Those living in single-parent families or having few resources can prove economic hardship.
axiom R2 : ∀ x : obj, LivesInSingleParentFamily x ∨ HasFewResources x → ProvesEconomicHardship x

-- Tom lives in a single-parent family.
axiom T1 : LivesInSingleParentFamily Tom

-- Tom has few resources available to them.
axiom T2 : HasFewResources Tom

-- Tom is applying to take the GRE test.
axiom T3 : GREApplicant Tom

-- Prove that ETS provides financial aid to Tom
theorem ets_aid_tom : FinancialAid Tom := by
  apply R1 Tom  -- Need to show GREApplicant Tom ∧ ProvesEconomicHardship Tom
  constructor
  · exact T3  -- GREApplicant Tom
  · apply R2 Tom  -- Need to show LivesInSingleParentFamily Tom ∨ HasFewResources Tom
    left  -- Use left disjunct (single-parent family)
    exact T1

-- Alternative proof using right disjunct:
theorem ets_aid_tom' : FinancialAid Tom := by
  apply R1 Tom
  constructor
  · exact T3
  · apply R2 Tom
    right  -- Use right disjunct (few resources)
    exact T2
</lean>

ANSWER: True
