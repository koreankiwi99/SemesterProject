3. From ¬J(Yuri) and "All professional basketball players can do jump shots" (∀x, P(x) → J(x)), we derive ¬P(Yuri) via contraposition.
4. Since ¬P(Yuri) (Yuri is not a professional basketball player), it follows that Yuri cannot be an American professional basketball player (¬(P(Yuri) ∧ A(Yuri))).

axiom obj : Type
axiom Yuri : obj

axiom P : obj → Prop   -- Professional basketball player
axiom A : obj → Prop   -- American national
axiom J : obj → Prop   -- Can do jump shots
axiom L : obj → Prop   -- Leaps straight into the air
axiom M : obj → Prop   -- Activates leg muscles

-- Premises
axiom some_not_American : ∃ x, P x ∧ ¬ A x
axiom all_can_jump : ∀ x, P x → J x
axiom jump_to_leap : ∀ x, J x → L x
axiom leap_to_muscle : ∀ x, L x → M x
axiom yuri_no_muscle : ¬ M Yuri

-- Prove Yuri is not an American professional basketball player
theorem yuri_not_american : ¬ (P Yuri ∧ A Yuri) :=
  fun ⟨hP, _⟩ =>

    have hL : L Yuri := jump_to_leap Yuri hJ
