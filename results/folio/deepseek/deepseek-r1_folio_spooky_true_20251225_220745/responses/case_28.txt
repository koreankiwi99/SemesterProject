Thus, Marvin is neither Human nor from Mars.

axiom obj : Type
axiom Marvin : obj
axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom FromMars : obj → Prop
axiom Human : obj → Prop
axiom HighlyIntelligent : obj → Prop
axiom FromEarth : obj → Prop

axiom all_aliens_are_extraterrestrial : ∀ x, Alien x → Extraterrestrial x
axiom from_mars_are_alien : ∀ x, FromMars x → Alien x
axiom no_extraterrestrial_human : ∀ x, Extraterrestrial x → ¬ Human x
axiom intelligent_earthlings_human : ∀ x, (HighlyIntelligent x ∧ FromEarth x) → Human x
axiom marvin_intelligent : HighlyIntelligent Marvin
axiom marvin_origins : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom not_earth_implies_extraterrestrial : ¬ FromEarth Marvin → Extraterrestrial Marvin

theorem marvin_neither : ¬ Human Marvin ∧ ¬ FromMars Marvin := by
  cases marvin_origins with
  | inl h =>
      have ⟨from_earth, from_mars⟩ := h
      have human_marvin : Human Marvin := intelligent_earthlings_human Marvin ⟨marvin_intelligent, from_earth⟩
      have alien_marvin : Alien Marvin := from_mars_are_alien Marvin from_mars
      have et_marvin : Extraterrestrial Marvin := all_aliens_are_extraterrestrial Marvin alien_marvin
      exact False.elim (no_extraterrestrial_human Marvin et_marvin human_marvin)
  | inr h =>
      have ⟨not_earth, not_mars⟩ := h
      have et_marvin : Extraterrestrial Marvin := not_earth_implies_extraterrestrial not_earth
      have not_human : ¬ Human Marvin := no_extraterrestrial_human Marvin et_marvin
