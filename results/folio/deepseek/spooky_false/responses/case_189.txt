<lean>
axiom obj : Type
axiom Yuri : obj
axiom ProfessionalBasketballPlayer : obj → Prop
axiom CanDoJumpShots : obj → Prop
axiom LeapStraightIntoAir : obj → Prop
axiom ActivateLegMuscles : obj → Prop
axiom AmericanNational : obj → Prop

axiom T1 : ∃ x, ProfessionalBasketballPlayer x ∧ ¬ AmericanNational x
axiom T2 : ∀ x, ProfessionalBasketballPlayer x → CanDoJumpShots x
axiom T3 : ∀ x, CanDoJumpShots x → LeapStraightIntoAir x
axiom T4 : ∀ x, LeapStraightIntoAir x → ActivateLegMuscles x
axiom T5 : ¬ ActivateLegMuscles Yuri

theorem disprove_conclusion : ¬ (ProfessionalBasketballPlayer Yuri ∧ AmericanNational Yuri) := 
by
  intro ⟨hPlayer, _⟩
  have jump := T2 _ hPlayer
  have leap := T3 _ jump
  have activate := T4 _ leap
  exact T5 activate
</lean>

ANSWER: False
