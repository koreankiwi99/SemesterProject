constant Turtle : obj → Prop

axiom P2 : ∃ x, Turtle x ∧ Spotted x

axiom P4 : ∀ x, Skittish x → ¬ Calm x

  have ⟨spotted, calm⟩ := P6
  have h1 : Squirrel Rockie ∧ Spotted Rockie := ⟨h, spotted⟩

  have not_calm := P4 Rockie skittish
  contradiction

  cases P3 Rockie spotted with
