axiom obj : Type
axiom Taylor : obj

axiom EatsSaladsRegularly : obj → Prop
axiom HealthConscious : obj → Prop
axiom GrewUpHealthConsciousHome : obj → Prop
axiom FulfillsNutrition : obj → Prop
axiom DisregardsPhysicalWellBeing : obj → Prop
axiom VisitsGymDaily : obj → Prop

axiom A1 : ∀ p : obj, EatsSaladsRegularly p → HealthConscious p
axiom A2 : ∀ p : obj, GrewUpHealthConsciousHome p → EatsSaladsRegularly p
axiom A3 : ∀ p : obj, FulfillsNutrition p → GrewUpHealthConsciousHome p
axiom A4 : ∀ p : obj, DisregardsPhysicalWellBeing p → ¬ HealthConscious p
axiom A5 : ∀ p : obj, VisitsGymDaily p → FulfillsNutrition p
axiom A6 : (GrewUpHealthConsciousHome Taylor ∧ DisregardsPhysicalWellBeing Taylor) ∨ 
           (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardsPhysicalWellBeing Taylor)

theorem taylor_case : ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ VisitsGymDaily Taylor := by
  have no_contradiction : ¬ (GrewUpHealthConsciousHome Taylor ∧ DisregardsPhysicalWellBeing Taylor) := by
    intro ⟨hHome, hDisregard⟩
    have hSalad := A2 Taylor hHome
    have hConscious := A1 Taylor hSalad
    exact A4 Taylor hDisregard hConscious
  cases A6 with
  | inl h => exact False.elim (no_contradiction h)
  | inr h => 
    have not_home := h.1
    have not_gym : ¬ VisitsGymDaily Taylor := by
      intro hGym
      have hNutrition := A5 Taylor hGym
      have hHome := A3 Taylor hNutrition
      contradiction
    exact ⟨not_home, not_gym⟩
