<lean>
axiom obj : Type
axiom Picuris : obj
axiom Juan : obj

axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Visited : obj → obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- The Picuris Mountains are a mountain range in New Mexico or Texas.
axiom A1 : MountainRange Picuris
axiom A2 : InNewMexico Picuris ∨ InTexas Picuris

-- Juan de Onate visited the Picuris Mountains.
axiom A3 : Visited Juan Picuris

-- There exists a mine in the Picuris Mountains that was donated (the Harding Pegmatite Mine).
axiom A4 : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m

-- There are no mountain ranges in Texas that have mines that have been donated.
axiom A5 : ∀ r : obj, MountainRange r ∧ InTexas r → ¬ (∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m)

-- Claim: Juan de Onate visited a mountain range in New Mexico.
theorem visited_NM : ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ Visited Juan r := by
  have not_texas : ¬ InTexas Picuris := by
    intro htex
    have h := A5 Picuris (And.intro A1 htex)
    exact h A4
  have hnm : InNewMexico Picuris :=
    Or.elim A2 (fun nm => nm) (fun tx => False.elim (not_texas tx))
  exact Exists.intro Picuris (And.intro A1 (And.intro hnm A3))
</lean>

ANSWER: True
