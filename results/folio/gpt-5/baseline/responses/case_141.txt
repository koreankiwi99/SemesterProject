axiom R3 : ∀ x : obj, RatedGA x → Appropriate x

      exact R1 Hachi H1 happ

  intro hev
  have hnot : ¬ CanWatch Hachi := R2 Hachi H1 hev
  exact hnot can_watch_Hachi
