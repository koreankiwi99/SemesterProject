-- A person either has US citizenship or Taiwanese citizenship.
axiom A3 : ∀ x : Person, USCitizen x ∨ TaiwaneseCitizen x
-- No Russian Federation officials hold Taiwanese citizenship.
axiom A4 : ∀ x : Person, RFOfficial x → ¬ TaiwaneseCitizen x
-- Vladimir neither holds Taiwanese citizenship nor is he a manager at Gazprom.
axiom A5a : ¬ TaiwaneseCitizen Vladimir
axiom A5b : ¬ GazpromManager Vladimir
-- Ekaterina can register to vote in the US, or she is a Russian federation official.

-- Useful consequence: Taiwanese citizenship implies not an RF official.
theorem taiwanese_not_rf : ∀ x : Person, TaiwaneseCitizen x → ¬ RFOfficial x := by
  intro x ht
  intro hr
  exact (A4 x hr) ht

-- From the exclusivity-or axiom and Vladimir not being Taiwanese, we can deduce he is a US citizen.

  have hnt : ¬ TaiwaneseCitizen Vladimir := A5a
  cases h with

  | inr ht => exact False.elim (hnt ht)

  exact A2 Vladimir vladimir_US
