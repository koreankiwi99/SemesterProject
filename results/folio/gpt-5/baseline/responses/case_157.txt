<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom UnivPT : obj → Prop
axiom TakingDB : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorksLab : obj → Prop

-- None of the students who currently work in the library are from the CS department.
axiom R1 : ∀ x : obj, Student x → WorksLibrary x → ¬ FromCS x
-- All of the students with part-time jobs offered by the university currently work in the library.
axiom R2 : ∀ x : obj, Student x → UnivPT x → WorksLibrary x
-- All of the students who are taking the database course are from the CS department.
axiom R3 : ∀ x : obj, Student x → TakingDB x → FromCS x
-- All of the students taking a class with Professor David are taking the database course.
axiom R4 : ∀ x : obj, Student x → TakingWithDavid x → TakingDB x
-- All students working in the lab are taking a class with Professor David.
axiom R5 : ∀ x : obj, Student x → WorksLab x → TakingWithDavid x

-- James is a student working in the lab.
axiom F1 : Student James
axiom F2 : WorksLab James

-- James doesn't work in the lab or have a part-time job offered by the university. (parsed as ¬WorksLab James ∨ ¬UnivPT James)
axiom F7 : ¬ WorksLab James ∨ ¬ UnivPT James

theorem james_no_univ_pt : ¬ UnivPT James := by
  have hwd : TakingWithDavid James := R5 James F1 F2
  have hdb : TakingDB James := R4 James F1 hwd
  have hcs : FromCS James := R3 James F1 hdb
  intro hpt
  have hlib : WorksLibrary James := R2 James F1 hpt
  have hncs : ¬ FromCS James := R1 James F1 hlib
  exact hncs hcs
</lean>

ANSWER: True
