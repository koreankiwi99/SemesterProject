<lean>
axiom obj : Type
axiom James : obj

axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom UnivPartTime : obj → Prop
axiom TakesDB : obj → Prop
axiom TakesWithDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- None who work in the library are from CS
axiom A1 : ∀ x : obj, WorksInLibrary x → ¬ FromCS x
-- All with university part-time jobs work in the library
axiom A2 : ∀ x : obj, UnivPartTime x → WorksInLibrary x
-- All taking the database course are from CS
axiom A3 : ∀ x : obj, TakesDB x → FromCS x
-- All taking a class with Professor David are taking the database course
axiom A4 : ∀ x : obj, TakesWithDavid x → TakesDB x
-- All working in the lab are taking a class with Professor David
axiom A5 : ∀ x : obj, WorksInLab x → TakesWithDavid x
-- James works in the lab
axiom A6 : WorksInLab James
-- James doesn't work in the lab or have a university part-time job
axiom A7 : ¬ (WorksInLab James ∨ UnivPartTime James)

theorem James_takes_database : TakesDB James := by
  have hPD : TakesWithDavid James := A5 James A6
  exact A4 James hPD
</lean>

ANSWER: True
