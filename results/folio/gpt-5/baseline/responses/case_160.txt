<lean>
axiom obj : Type
axiom George : obj

axiom LikesMusic : obj → Prop
axiom WantsCompose : obj → Prop
axiom HasAccessProgram : obj → Prop
axiom CanCompose : obj → Prop
axiom WillCompose : obj → Prop

axiom A1 : LikesMusic George → WantsCompose George
axiom A2 : HasAccessProgram George → CanCompose George
axiom A3 : CanCompose George ∧ WantsCompose George → WillCompose George

theorem george_will_compose :
  (LikesMusic George ∧ HasAccessProgram George) → WillCompose George :=
by
  intro h
  have lm : LikesMusic George := And.left h
  have ha : HasAccessProgram George := And.right h
  have w : WantsCompose George := A1 lm
  have c : CanCompose George := A2 ha
  exact A3 ⟨c, w⟩
</lean>

ANSWER: True
