<lean>
axiom obj : Type

axiom AirsMon8pm : obj → Prop
axiom GivesRoses : obj → Prop
axiom EpisodeBachelor : obj → Prop
axiom PortraysRealPeople : obj → Prop
axiom RealityTV : obj → Prop
axiom Show : obj → Prop

axiom BreakingBad : obj

-- Some show airing at 8 pm on Monday gives out roses on TV.
axiom ex_monday8_roses : ∃ s : obj, AirsMon8pm s ∧ GivesRoses s
-- If a show gives out roses on TV, then the show is an episode of The Bachelor.
axiom roses_implies_bachelor : ∀ x : obj, GivesRoses x → EpisodeBachelor x
-- The Bachelor portrays the lives of real people (modeled as: its episodes do).
axiom bachelor_portrays_real : ∀ x : obj, EpisodeBachelor x → PortraysRealPeople x
-- All shows portraying the lives of real people are reality TV shows.
axiom portrays_implies_reality : ∀ x : obj, PortraysRealPeople x → RealityTV x
-- Breaking Bad is a show.
axiom breakingbad_is_show : Show BreakingBad
-- Breaking Bad is not a reality TV show.
axiom breakingbad_not_reality : ¬RealityTV BreakingBad

-- Question: If roses are given out during Breaking Bad, then it is on Monday at 8 pm.
theorem breakingbad_roses_implies_monday8 : GivesRoses BreakingBad → AirsMon8pm BreakingBad := by
  intro hroses
  have hbachelor : EpisodeBachelor BreakingBad := roses_implies_bachelor BreakingBad hroses
  have hrealpeople : PortraysRealPeople BreakingBad := bachelor_portrays_real BreakingBad hbachelor
  have hreality : RealityTV BreakingBad := portrays_implies_reality BreakingBad hrealpeople
  have hfalse : False := breakingbad_not_reality hreality
  exact False.elim hfalse
</lean>

ANSWER: True
