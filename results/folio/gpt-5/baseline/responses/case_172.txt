-- All people who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : obj, EatsSaladsRegularly x → VeryConscious x
-- All people who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : obj, GrewUpHC x → EatsSaladsRegularly x
-- All people who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : obj, FulfillsDailyIntake x → GrewUpHC x
-- All people who disregard their physical well-being are not very conscious about their health and eating habits.
axiom A4 : ∀ x : obj, DisregardsPhysicalWellBeing x → ¬ VeryConscious x
-- If people visit the gym at least once a day, then they always fulfill their daily nutritional intakes.
axiom A5 : ∀ x : obj, VisitsGymDaily x → FulfillsDailyIntake x
-- Taylor either grew up in a health-conscious childhood home and disregards her physical well-being, or she did neither.
axiom A6 : (GrewUpHC Taylor ∧ DisregardsPhysicalWellBeing Taylor) ∨ (¬ GrewUpHC Taylor ∧ ¬ DisregardsPhysicalWellBeing Taylor)

-- The left disjunct leads to a contradiction, so it cannot hold.
theorem left_impossible : ¬ (GrewUpHC Taylor ∧ DisregardsPhysicalWellBeing Taylor) := by
  intro h
  have hG : GrewUpHC Taylor := And.left h
  have hD : DisregardsPhysicalWellBeing Taylor := And.right h
  have hESR : EatsSaladsRegularly Taylor := A2 Taylor hG
  have hVC : VeryConscious Taylor := A1 Taylor hESR
  have hNotVC : ¬ VeryConscious Taylor := A4 Taylor hD
  exact hNotVC hVC

-- Therefore the right disjunct holds: Taylor did neither.
theorem right_case : ¬ GrewUpHC Taylor ∧ ¬ DisregardsPhysicalWellBeing Taylor :=
  Or.elim A6 (fun hleft => False.elim (left_impossible hleft)) (fun hright => hright)

-- Consequences (do not settle whether Taylor eats salads regularly)
