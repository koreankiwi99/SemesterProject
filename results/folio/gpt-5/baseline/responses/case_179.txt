<lean>
axiom Person : Type
axiom Airport : Type

axiom Susan : Person
axiom John : Person
axiom LGA : Airport

axiom FliesFrom : Person → Airport → Prop
axiom FliesTo : Person → Airport → Prop

-- The departure and arrival cannot be at the same airport
axiom dep_arr_not_same : ∀ p a b, FliesFrom p a → FliesTo p b → a ≠ b

-- Facts from the context
axiom susan_to_LGA : FliesTo Susan LGA
axiom john_from_LGA : FliesFrom John LGA

-- Prove that Susan does not fly from LGA
theorem susan_not_from_LGA : ¬ FliesFrom Susan LGA :=
by
  intro h
  have hneq := dep_arr_not_same Susan LGA LGA h susan_to_LGA
  exact hneq rfl
</lean>
