<lean>
axiom obj : Type
axiom Yuri : obj

axiom American : obj → Prop
axiom Pro : obj → Prop
axiom JumpShot : obj → Prop
axiom Leap : obj → Prop
axiom Activate : obj → Prop

-- Some professional basketball players are not American nationals.
axiom some_nonamerican_pro : ∃ x : obj, Pro x ∧ ¬ American x
-- All professional basketball players can do jump shots.
axiom pro_jump : ∀ x : obj, Pro x → JumpShot x
-- If someone can jump shots, they leap straight into the air.
axiom jump_leap : ∀ x : obj, JumpShot x → Leap x
-- If someone leaps straight into the air, they activate their leg muscles.
axiom leap_activate : ∀ x : obj, Leap x → Activate x
-- Yuri does not activate his leg muscles.
axiom yuri_not_activate : ¬ Activate Yuri

-- Conclusion: Yuri is not an American professional basketball player.
theorem yuri_not_american_pro : ¬ (American Yuri ∧ Pro Yuri) :=
by
  intro hy
  have hypro : Pro Yuri := And.right hy
  have hjump : JumpShot Yuri := pro_jump Yuri hypro
  have hleap : Leap Yuri := jump_leap Yuri hjump
  have hact : Activate Yuri := leap_activate Yuri hleap
  exact yuri_not_activate hact
</lean>

ANSWER: True
