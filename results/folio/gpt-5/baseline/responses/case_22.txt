<lean>
axiom Show : Type
axiom StrangerThings : Show
axiom BlackMirror : Show

axiom Popular : Show → Prop
axiom NetflixShow : Show → Prop
axiom BingeKaren : Show → Prop
axiom DownloadKaren : Show → Prop
axiom ShareWithLisa : Show → Prop

-- "Stranger Things" is a popular Netflix show.
axiom ST_popular : Popular StrangerThings
axiom ST_netflix : NetflixShow StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom R_popular_netflix_to_binge : ∀ s : Show, NetflixShow s ∧ Popular s → BingeKaren s

-- If and only if Karen binge-watches a Netflix show, she will download it. (equivalence on Netflix shows)
axiom R_binge_iff_download : ∀ s : Show, NetflixShow s → (BingeKaren s ↔ DownloadKaren s)

-- Karen does not download "Black Mirror."
axiom BM_not_download : ¬ DownloadKaren BlackMirror

-- "Black Mirror" is a Netflix show.
axiom BM_netflix : NetflixShow BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom R_binge_share : ∀ s : Show, NetflixShow s ∧ BingeKaren s → ShareWithLisa s

-- From the above, we can show "Black Mirror" is not popular.
theorem BM_not_popular : ¬ Popular BlackMirror := by
  intro hpop
  have hiff := R_binge_iff_download BlackMirror BM_netflix
  have hb2d : BingeKaren BlackMirror → DownloadKaren BlackMirror := Iff.mp hiff
  have hnotb : ¬ BingeKaren BlackMirror := by
    intro hb
    exact BM_not_download (hb2d hb)
  have hbinge : BingeKaren BlackMirror := by
    apply R_popular_netflix_to_binge
    exact And.intro BM_netflix hpop
  exact hnotb hbinge
</lean>

ANSWER: False
