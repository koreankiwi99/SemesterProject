<lean>
axiom obj : Type
axiom Coco : obj

axiom RankedHighWTA : obj → Prop
axiom ActiveMajor : obj → Prop
axiom LostToIgaRG2022 : obj → Prop
axiom LostToRafaRG2022 : obj → Prop
axiom FemaleRG2022 : obj → Prop
axiom MaleRG2022 : obj → Prop
axiom AtRG2022 : obj → Prop

-- If someone is ranked highly by the WTA, then they are among the most active major tennis players.
axiom R1 : ∀ x : obj, RankedHighWTA x → ActiveMajor x
-- Everyone who lost to Iga Świątek at Roland Garros 2022 is ranked highly by the WTA.
axiom R2 : ∀ x : obj, LostToIgaRG2022 x → RankedHighWTA x
-- All female tennis players at Roland Garros 2022 lost to Iga Świątek.
axiom R3 : ∀ x : obj, FemaleRG2022 x → LostToIgaRG2022 x
-- All tennis players at Roland Garros 2022 are either female or male.
axiom R4 : ∀ x : obj, AtRG2022 x → FemaleRG2022 x ∨ MaleRG2022 x
-- All male tennis players at Roland Garros 2022 lost to Rafael Nadal.
axiom R5 : ∀ x : obj, MaleRG2022 x → LostToRafaRG2022 x
-- If Coco Gauff is ranked highly by the WTA or lost to Rafael Nadal, then Coco is not a male tennis player at Roland Garros 2022.
axiom R6 : (RankedHighWTA Coco ∨ LostToRafaRG2022 Coco) → ¬ MaleRG2022 Coco
-- Coco Gauff is at Roland Garros 2022.
axiom R7 : AtRG2022 Coco

theorem coco_active_major : ActiveMajor Coco := by
  have hdisj : FemaleRG2022 Coco ∨ MaleRG2022 Coco := R4 Coco R7
  cases hdisj with
  | inl hf =>
      have hL : LostToIgaRG2022 Coco := R3 Coco hf
      have hr : RankedHighWTA Coco := R2 Coco hL
      exact R1 Coco hr
  | inr hm =>
      have hLR : LostToRafaRG2022 Coco := R5 Coco hm
      have hnotm : ¬ MaleRG2022 Coco := R6 (Or.inr hLR)
      have contra : False := hnotm hm
      exact False.elim contra
</lean>

ANSWER: True
