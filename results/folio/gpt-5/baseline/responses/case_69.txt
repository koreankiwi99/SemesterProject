<lean>
axiom Person : Type
axiom Show : Type

axiom Badults : Show

axiom MemberPappys : Person → Prop
axiom StarredIn : Person → Show → Prop
axiom PilotedOnBBCTwo : Show → Prop
axiom PilotedOnBBCThree : Show → Prop

-- Context axioms
-- "Badults is a British Sitcom series starring members of Pappy's."
axiom A1 : ∃ p : Person, MemberPappys p ∧ StarredIn p Badults
-- "Badults was piloted in July 2013 on BBC Three."
axiom A2 : PilotedOnBBCThree Badults

-- The claim: "No members of Pappy's have starred in a show piloting on BBC Two or BBC Three."
def NoMembersStarredInPilotedOnTwoOrThree : Prop :=
  ¬ ∃ p : Person, ∃ s : Show,
      MemberPappys p ∧ StarredIn p s ∧ (PilotedOnBBCTwo s ∨ PilotedOnBBCThree s)

-- We prove the negation of the claim by exhibiting a member starring in Badults, which piloted on BBC Three.
theorem not_NoMembersStarredInPilotedOnTwoOrThree :
  ¬ NoMembersStarredInPilotedOnTwoOrThree := by
  intro h
  have ex : ∃ p : Person, ∃ s : Show,
      MemberPappys p ∧ StarredIn p s ∧ (PilotedOnBBCTwo s ∨ PilotedOnBBCThree s) := by
    rcases A1 with ⟨p, hp⟩
    rcases hp with ⟨hpMem, hpStar⟩
    exact ⟨p, Badults, ⟨hpMem, ⟨hpStar, Or.inr A2⟩⟩⟩
  exact h ex
</lean>

ANSWER: False
