<lean>
axiom obj : Type

-- Predicates
axiom AtMS : obj → Prop                          -- at Mary's school
axiom Works : obj → Prop                         -- works in student jobs on campus
axiom Needs : obj → Prop                         -- needs to earn money to help pay for college tuition
axiom OrdersTakeout : obj → Prop                 -- orders takeout frequently in college
axiom EnjoysDining : obj → Prop                  -- enjoys the dining hall meals and recipes
axiom NotPicky : obj → Prop                      -- is not a picky eater
axiom SpendsTimeDiningHalls : obj → Prop         -- spends a lot of time eating/catching up with friends in dining halls

axiom Hannah : obj

-- Rules from the context
axiom R1 : ∀ x : obj, AtMS x → Works x → Needs x
axiom R2 : ∀ x : obj, AtMS x → OrdersTakeout x → Works x
axiom R3 : ∀ x : obj, AtMS x → (OrdersTakeout x ∨ EnjoysDining x)
axiom R4 : ∀ x : obj, AtMS x → EnjoysDining x → NotPicky x
axiom R5 : ∀ x : obj, AtMS x → EnjoysDining x → SpendsTimeDiningHalls x

-- Facts about Hannah
axiom H_at : AtMS Hannah
axiom H_fact : Works Hannah ∧ (Needs Hannah → (NotPicky Hannah ∧ ¬Needs Hannah))

-- Target statement: Hannah needs to earn money to help pay for her college tuition.
theorem hannah_needs : Needs Hannah := by
  have hW : Works Hannah := And.left H_fact
  exact R1 Hannah H_at hW

-- For reference, the context also implies ¬Needs Hannah (inconsistency).
theorem hannah_not_needs : ¬ Needs Hannah := by
  have hImp : Needs Hannah → (NotPicky Hannah ∧ ¬ Needs Hannah) := And.right H_fact
  intro hN
  have hNotN : ¬ Needs Hannah := And.right (hImp hN)
  exact hNotN hN
</lean>

ANSWER: True
