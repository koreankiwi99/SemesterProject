axiom obj : Type
axiom DTC : obj  -- Dried Thai chilies

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom BBMProduct : obj → Prop

-- No baked sweets are spicy.
axiom no_baked_sweets_spicy : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom cupcakes_are_baked_sweets : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom mala_hotpots_are_spicy : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom bbm_products_are_cupcakes : ∀ x : obj, BBMProduct x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom DTC_disj : Spicy DTC ∨ MalaHotpot DTC ∨ ¬ BakedSweet DTC

theorem not_cupcake_DTC : ¬ Cupcake DTC := by
  intro c
  have b : BakedSweet DTC := cupcakes_are_baked_sweets DTC c
  have ns : ¬ Spicy DTC := no_baked_sweets_spicy DTC b
  cases DTC_disj with
  | inl s_or_m =>
      cases s_or_m with
      | inl s =>
          exact ns s
      | inr m =>
          have s : Spicy DTC := mala_hotpots_are_spicy DTC m
          exact ns s
  | inr nb =>
      exact nb b

theorem not_statement : ¬ (Cupcake DTC ∧ BBMProduct DTC) := by
  intro h
  have c : Cupcake DTC := h.left
  exact not_cupcake_DTC c
