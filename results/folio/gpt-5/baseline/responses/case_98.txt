axiom obj : Type
axiom Harry : obj

-- Predicates
axiom Animal : obj → Prop
axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom Mating : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop

-- Axioms from the context
axiom A1 : ∀ x : obj, Animal x → Invertebrate x ∨ Vertebrate x
axiom A2 : ∀ x : obj, Animal x → Backbone x → Mating x
axiom A3 : ∀ x : obj, Vertebrate x → Animal x
axiom A4 : ∀ x : obj, Vertebrate x → Backbone x
axiom A5 : ∀ x : obj, Bee x → ¬ Mating x
axiom A6 : ∀ x : obj, QueenBee x → Bee x
axiom A7 : Bee Harry

-- A derived fact: bees cannot be vertebrates (given the axioms)
theorem bee_not_vertebrate : ∀ x : obj, Bee x → ¬ Vertebrate x :=
by
  intro x hBee hVert
  have hAnimal : Animal x := A3 x hVert
  have hBack : Backbone x := A4 x hVert
  have hMate : Mating x := A2 x hAnimal hBack
  have hNotMate : ¬ Mating x := A5 x hBee
  exact hNotMate hMate

theorem harry_not_vertebrate : ¬ Vertebrate Harry :=
by
  have h := bee_not_vertebrate Harry A7
  exact h
