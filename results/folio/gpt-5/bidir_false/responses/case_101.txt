<lean>
axiom Person : Type
axiom Luke : Person

axiom MultipleBirth : Person → Prop
axiom SibsBornTogether : Person → Prop
axiom ComplainSibs : Person → Prop
axiom SpendTimeWithSiblings : Person → Prop
axiom LiveAtHome : Person → Prop
axiom LiveWithStrangers : Person → Prop

axiom A1 : ∀ x : Person, MultipleBirth x → SpendTimeWithSiblings x
axiom A2 : ∀ x : Person, SibsBornTogether x → MultipleBirth x
axiom A3 : ∀ x : Person, ComplainSibs x → SibsBornTogether x
axiom A4 : ∀ x : Person, LiveAtHome x → ¬LiveWithStrangers x
axiom A5 : ∀ x : Person, SpendTimeWithSiblings x → LiveAtHome x

axiom A6 : (MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨ (¬MultipleBirth Luke ∧ ¬LiveWithStrangers Luke)

-- From A1, A5, A4, we can show that no one can be both in a multiple birth and live with strangers
theorem no_mb_and_lws : ∀ x : Person, ¬(MultipleBirth x ∧ LiveWithStrangers x) := by
  intro x
  intro h
  have mb : MultipleBirth x := And.left h
  have lws : LiveWithStrangers x := And.right h
  have st : SpendTimeWithSiblings x := A1 x mb
  have lah : LiveAtHome x := A5 x st
  have nlws : ¬LiveWithStrangers x := A4 x lah
  exact nlws lws

-- Therefore, for Luke, the right disjunct must hold
theorem luke_not_mb_and_not_lws : ¬MultipleBirth Luke ∧ ¬LiveWithStrangers Luke := by
  have hdisj := A6
  have hnot := no_mb_and_lws Luke
  cases hdisj with
  | inl h =>
      exact False.elim (hnot h)
  | inr h =>
      exact h
