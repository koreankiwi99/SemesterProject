axiom A3 : ∀ x : obj, Clean x → Tidy x

-- Derivations showing the statement is actually true

theorem peter_not_spill_and_clean : ¬ (Spills Peter ∧ Clean Peter) := by
  intro h
  have hs : Spills Peter := h.left
  have hc : Clean Peter := h.right
  have ht : Tidy Peter := A3 Peter hc
  have hnt : ¬ Tidy Peter := A1 Peter hs
  exact hnt ht

theorem peter_not_spill_and_not_clean : ¬ Spills Peter ∧ ¬ Clean Peter := by
  cases A6 with
  | inl hsc =>
      have contra := peter_not_spill_and_clean hsc
      exact False.elim contra
  | inr hnc =>
      exact hnc

theorem peter_not_spills : ¬ Spills Peter := by
  exact (peter_not_spill_and_not_clean).left

theorem peter_not_clean : ¬ Clean Peter := by
  exact (peter_not_spill_and_not_clean).right

theorem peter_not_clumsy_foodie_out : ¬ ClumsyFoodieOut Peter := by
  intro hG
  have hs : Spills Peter := A2 Peter hG
  exact peter_not_spills hs

theorem peter_not_values : ¬ ValuesOrder Peter := by
  intro hv
  have hc : Clean Peter := A4 Peter hv
  exact peter_not_clean hc

theorem peter_not_family_prior : ¬ FamilyPrior Peter := by
  intro hf
  have hv : ValuesOrder Peter := A5 Peter hf
  exact peter_not_values hv
