<lean>
axiom obj : Type

-- Entities
axiom PicurisMountains : obj
axiom HardingPegmatiteMine : obj
axiom JuanDeOnate : obj

-- Predicates
axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

-- Facts from the context
axiom T1 : MountainRange PicurisMountains
axiom T2 : InNewMexico PicurisMountains ∨ InTexas PicurisMountains
axiom T3 : LocatedIn HardingPegmatiteMine PicurisMountains
axiom T4 : Donated HardingPegmatiteMine
axiom T5 : Visited JuanDeOnate PicurisMountains

-- No Texas mountain ranges have mines that have been donated
axiom R1 : ∀ m r : obj, InTexas r → LocatedIn m r → Donated m → False

-- Prove the negation of the statement:
-- "The Harding Pegmatite Mine is not located in a mountain range in New Mexico."
-- Negation: ∃ r, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingPegmatiteMine r
theorem mine_in_nm_mountain_range :
  ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingPegmatiteMine r := by
  have not_in_texas_picuris : ¬InTexas PicurisMountains :=
    fun htx => R1 HardingPegmatiteMine PicurisMountains htx T3 T4
  have picuris_in_nm : InNewMexico PicurisMountains := by
    cases T2 with
    | inl hnm => exact hnm
    | inr htx =>
        have contra : False := not_in_texas_picuris htx
        exact False.elim contra
  exact Exists.intro PicurisMountains (And.intro T1 (And.intro picuris_in_nm T3))
</lean>

ANSWER: False
