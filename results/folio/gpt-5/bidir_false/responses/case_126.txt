axiom obj : Type
axiom Rock : obj

axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

axiom A1 : ∀ x : obj, Pet x ∧ Mammal x → Animal x
axiom A2 : ∀ x : obj, Monkey x → Mammal x
axiom A3 : ∀ x : obj, Pet x → Monkey x ∨ Bird x
axiom A4 : ∀ x : obj, Bird x ∧ Pet x → CanFly x
axiom A5 : ∀ x : obj, Pet x ∧ Animal x → CanBreathe x
axiom A6 : ∀ x : obj, Pet x ∧ CanFly x → HasWings x

axiom A7 : Pet Rock
axiom A8 : CanFly Rock ∨ Bird Rock ∨ ¬CanBreathe Rock

theorem rock_has_wings : HasWings Rock := by
  cases A8 with
  | inl hFly =>
    apply A6 Rock
    exact And.intro A7 hFly
  | inr h =>
    cases h with
    | inl hBird =>
      have hFly : CanFly Rock := A4 Rock (And.intro hBird A7)
      apply A6 Rock
      exact And.intro A7 hFly
    | inr hNotBreathe =>
      have hMB : Monkey Rock ∨ Bird Rock := A3 Rock A7
      cases hMB with
      | inl hMonkey =>
        have hMammal : Mammal Rock := A2 Rock hMonkey
        have hAnimal : Animal Rock := A1 Rock (And.intro A7 hMammal)
        have hBreathe : CanBreathe Rock := A5 Rock (And.intro A7 hAnimal)
        have contra : False := hNotBreathe hBreathe
        exact False.elim contra
      | inr hBird2 =>
        have hFly : CanFly Rock := A4 Rock (And.intro hBird2 A7)
        apply A6 Rock
        exact And.intro A7 hFly
