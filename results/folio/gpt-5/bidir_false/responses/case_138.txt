<lean>
axiom obj : Type
axiom MLAlgo : obj → Prop
axiom Supervised : obj → Prop
axiom Unsupervised : obj → Prop
axiom Reinforcement : obj → Prop
axiom Used : obj → Prop
axiom RequiresLabels : obj → Prop

-- Categorization of ML algorithms
axiom Cat : ∀ x : obj, MLAlgo x → (Supervised x ∨ Unsupervised x ∨ Reinforcement x)
-- Unsupervised algorithms do not require labeled data
axiom UnsupNoLabels : ∀ x : obj, Unsupervised x → ¬RequiresLabels x
-- The SOTA model is trained with ML algorithms
axiom UsedIsML : ∀ x : obj, Used x → MLAlgo x
-- Reinforcement learning is not used to train the SOTA model
axiom ReinforceNotUsed : ∀ x : obj, Reinforcement x → ¬Used x
-- Algorithms used for training the text summarization model require labeled data
axiom UsedNeedsLabels : ∀ x : obj, Used x → RequiresLabels x

-- Prove that no unsupervised learning algorithm is used to train the SOTA model
theorem no_unsupervised_used : ¬ (∃ x : obj, Used x ∧ Unsupervised x) := by
  intro h
  apply Exists.elim h
  intro x hx
  have req : RequiresLabels x := UsedNeedsLabels x hx.left
  have notreq : ¬RequiresLabels x := UnsupNoLabels x hx.right
  exact notreq req
</lean>
