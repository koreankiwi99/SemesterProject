axiom HeckCattle : obj → Prop
axiom Aurochs : obj → Prop
axiom BredBack : obj → Prop
axiom ArtificialSelection : obj → Prop
axiom ResembleExtinct : obj → Prop

-- Breeding back is a form of artificial selection
axiom R1 : ∀ x : obj, BredBack x → ArtificialSelection x

-- Heck cattle are animals
axiom T1 : ∀ x : obj, HeckCattle x → Animal x

-- Aurochs are animals
axiom T2 : ∀ x : obj, Aurochs x → Animal x

-- Some animals to be bred back resemble extinct animals
axiom T3 : ∃ x : obj, Animal x ∧ BredBack x ∧ ResembleExtinct x

-- Heck cattle were bred back
axiom T4 : ∃ x : obj, HeckCattle x ∧ BredBack x

-- From the premises we can prove the original statement (existence)
theorem some_heck_artificial : ∃ x : obj, HeckCattle x ∧ ArtificialSelection x :=
by

  intro x hx
  have hxHeck : HeckCattle x := And.left hx
  have hxBB : BredBack x := And.right hx
  have hxAS : ArtificialSelection x := R1 x hxBB
  exact Exists.intro x (And.intro hxHeck hxAS)
