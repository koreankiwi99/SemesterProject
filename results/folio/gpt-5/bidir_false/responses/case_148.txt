axiom obj : Type
axiom Ekaterina : obj
axiom Vladimir : obj

axiom RegisterUS : obj → Prop
axiom Participate2024 : obj → Prop
axiom USCitizen : obj → Prop
axiom TaiwaneseCitizen : obj → Prop
axiom RFOfficial : obj → Prop
axiom GazpromManager : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, RegisterUS x → Participate2024 x
axiom P2 : ∀ x : obj, USCitizen x → RegisterUS x
axiom P3 : ∀ x : obj, USCitizen x ∨ TaiwaneseCitizen x
axiom P4 : ∀ x : obj, RFOfficial x → ¬TaiwaneseCitizen x
axiom P5 : ¬TaiwaneseCitizen Vladimir ∧ ¬GazpromManager Vladimir
axiom P6 : RegisterUS Ekaterina ∨ RFOfficial Ekaterina

-- Prove the negation of the statement: i.e., prove RegisterUS Ekaterina ∧ Participate2024 Vladimir
theorem negate_statement : RegisterUS Ekaterina ∧ Participate2024 Vladimir := by
  -- Vladimir participates
  have hV_part : Participate2024 Vladimir := by
    have hUSorTW := P3 Vladimir
    cases hUSorTW with
    | inl hUS =>
        have hReg : RegisterUS Vladimir := P2 Vladimir hUS
        exact P1 Vladimir hReg
    | inr hTW =>
        have hNotTW : ¬TaiwaneseCitizen Vladimir := And.left P5
        exact False.elim (hNotTW hTW)
  -- Ekaterina can register
  have hE_reg : RegisterUS Ekaterina := by
    cases P6 with
    | inl hReg => exact hReg
    | inr hRF =>
        have hNotTW : ¬TaiwaneseCitizen Ekaterina := P4 Ekaterina hRF
        have hUSorTW := P3 Ekaterina
        cases hUSorTW with
        | inl hUS => exact P2 Ekaterina hUS
        | inr hTW => exact False.elim (hNotTW hTW)
  exact And.intro hE_reg hV_part
