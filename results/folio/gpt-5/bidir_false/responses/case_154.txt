<lean>
axiom obj : Type
axiom mixture : obj

axiom Hydrocarbon : obj → Prop
axiom Organic : obj → Prop
axiom ChemicalCompound : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom OneElement : obj → Prop
axiom Alkane : obj → Prop

-- All hydrocarbons are organic compounds.
axiom A1 : ∀ x : obj, Hydrocarbon x → Organic x
-- All alkanes are hydrocarbons.
axiom A2 : ∀ x : obj, Alkane x → Hydrocarbon x
-- All organic compounds are chemical compounds.
axiom A3 : ∀ x : obj, Organic x → ChemicalCompound x
-- All organic compounds contain carbon.
axiom A4 : ∀ x : obj, Organic x → ContainsCarbon x
-- No chemical compounds contain only one element.
axiom A5 : ∀ x : obj, ChemicalCompound x → ¬ OneElement x
-- The mixture either both is a chemical compound and contains only one element, or neither.
axiom A6 : (ChemicalCompound mixture ∧ OneElement mixture) ∨ (¬ChemicalCompound mixture ∧ ¬OneElement mixture)

-- From A5 and A6, the mixture is not a chemical compound.
theorem mixture_not_CC : ¬ChemicalCompound mixture := by
  intro hCC
  cases A6 with
  | inl h =>
    have hOne : OneElement mixture := And.right h
    have hNotOne : ¬OneElement mixture := A5 mixture hCC
    exact hNotOne hOne
  | inr h =>
    have hNotCC : ¬ChemicalCompound mixture := And.left h
    exact hNotCC hCC

-- Hence the mixture is not an alkane (since alkanes imply chemical compounds).
theorem mixture_not_alkane : ¬Alkane mixture := by
  intro hAlk
  have hHydro : Hydrocarbon mixture := A2 mixture hAlk
  have hOrg : Organic mixture := A1 mixture hHydro
  have hCC : ChemicalCompound mixture := A3 mixture hOrg
  exact mixture_not_CC hCC

-- The original statement actually holds unconditionally (hence its negation is not provable).
theorem statement_holds :
  (OneElement mixture ∨ ContainsCarbon mixture) → (¬ChemicalCompound mixture ∧ ¬Alkane mixture) := by
  intro _
  exact And.intro mixture_not_CC mixture_not_alkane
