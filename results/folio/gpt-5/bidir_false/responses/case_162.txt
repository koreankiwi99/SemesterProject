<lean>
axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

axiom no_hawk_swims : ∀ x : obj, Hawk x → ¬ Swims x
axiom some_bird_is_hawk : ∃ x : obj, Bird x ∧ Hawk x

theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swims x) := by
  intro hAll
  apply Exists.elim some_bird_is_hawk
  intro a ha
  have hB : Bird a := And.left ha
  have hH : Hawk a := And.right ha
  have hS : Swims a := hAll a hB
  have hNS : ¬ Swims a := no_hawk_swims a hH
  exact hNS hS
</lean>

ANSWER: False
