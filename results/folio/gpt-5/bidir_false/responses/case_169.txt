axiom obj : Type

-- Predicates
axiom Deadly : obj → Prop
axiom LowSurvival : obj → Prop
axiom Severe : obj → Prop
axiom BileDuct : obj → Prop
axiom Cholangiocarcinoma : obj → Prop

-- Entities
axiom MildFlu : obj
axiom Colorectal : obj

-- Premises
axiom R1 : ∀ x : obj, Deadly x → LowSurvival x                -- Deadly ⇒ LowSurvival
axiom R2 : ∀ x : obj, Severe x → Deadly x                     -- Severe ⇒ Deadly
axiom R3 : ∀ x : obj, BileDuct x → Severe x                   -- BileDuct ⇒ Severe
axiom R4 : ∀ x : obj, Cholangiocarcinoma x → BileDuct x       -- Cholangiocarcinoma ⇒ BileDuct
axiom T1 : LowSurvival MildFlu                                -- MildFlu has low survival
axiom T2 : ¬(BileDuct Colorectal ∧ LowSurvival Colorectal)    -- Not both BileDuct and LowSurvival for Colorectal

-- A useful derived fact: from BileDuct one gets LowSurvival
theorem bileduct_implies_low : ∀ x : obj, BileDuct x → LowSurvival x :=
  fun x hx => R1 x (R2 x (R3 x hx))

-- Hence, we can show Colorectal is not a bile duct cancer
theorem not_bileduct_colorectal : ¬BileDuct Colorectal :=
  fun hBD => T2 (And.intro hBD (bileduct_implies_low Colorectal hBD))
