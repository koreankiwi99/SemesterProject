<lean>
axiom Person : Type
axiom Airport : Type

axiom Susan : Person
axiom John : Person
axiom LGA : Airport

axiom From : Person → Airport → Prop
axiom To : Person → Airport → Prop

-- Context axioms
axiom A1 : To Susan LGA
axiom A2 : From John LGA
axiom A3 : ∀ p : Person, ∀ a : Airport, ¬(From p a ∧ To p a)

theorem susan_not_from_LGA : ¬From Susan LGA := by
  intro hFrom
  have hconj : From Susan LGA ∧ To Susan LGA := And.intro hFrom A1
  exact (A3 Susan LGA) hconj
</lean>

ANSWER: False
