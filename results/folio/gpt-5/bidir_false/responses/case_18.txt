axiom obj : Type
axiom Rockie : obj

axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom NearCampus : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

-- All rabbits that can be spotted near the campus are cute.
axiom A1 : ∀ x : obj, Rabbit x ∧ NearCampus x → Cute x
-- Some turtles can be spotted near the campus.
axiom A2 : ∃ x : obj, Turtle x ∧ NearCampus x
-- The only animals that can be spotted near the campus are rabbits and squirrels.
axiom A3 : ∀ x : obj, NearCampus x → (Rabbit x ∨ Squirrel x)
-- If something is skittish, then it is not calm.
axiom A4 : ∀ x : obj, Skittish x → ¬Calm x
-- All the squirrels that can be spotted near the campus are skittish.
axiom A5 : ∀ x : obj, Squirrel x ∧ NearCampus x → Skittish x
-- Rockie can be spotted near the campus, and it is calm.
axiom A6 : NearCampus Rockie ∧ Calm Rockie

-- Derive that Rockie is cute (hence the statement "Rockie is a turtle or cute" holds)
theorem Rockie_cute : Cute Rockie := by
  have nearR : NearCampus Rockie := And.left A6
  have calmR : Calm Rockie := And.right A6
  have notSquirrel : ¬Squirrel Rockie := by
    intro hs
    have sk : Skittish Rockie := A5 Rockie (And.intro hs nearR)
    have ncalm : ¬Calm Rockie := A4 Rockie sk
    exact ncalm calmR
  have hrabbit : Rabbit Rockie := by
    have disj := A3 Rockie nearR
    cases disj with
    | inl hr => exact hr
    | inr hs => exact False.elim (notSquirrel hs)
  exact A1 Rockie (And.intro hrabbit nearR)

theorem target_true : Turtle Rockie ∨ Cute Rockie := Or.inr Rockie_cute
