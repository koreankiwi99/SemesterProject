<lean>
axiom obj : Type
axiom Yuri : obj

axiom American : obj → Prop
axiom Pro : obj → Prop
axiom Jump : obj → Prop
axiom Leap : obj → Prop
axiom Activate : obj → Prop

-- Some professional basketball players are not American nationals.
axiom ExNonAmericanPro : ∃ x : obj, Pro x ∧ ¬American x
-- All professional basketball players can do jump shots.
axiom AllProJump : ∀ x : obj, Pro x → Jump x
-- If someone can jump shots, they leap straight into the air.
axiom JumpToLeap : ∀ x : obj, Jump x → Leap x
-- If someone leaps straight into the air, they activate their leg muscles.
axiom LeapToActivate : ∀ x : obj, Leap x → Activate x
-- Yuri does not activate his leg muscles.
axiom YuriNotActivate : ¬Activate Yuri

theorem not_pro_yuri : ¬Pro Yuri :=
  fun hPro =>
    YuriNotActivate (LeapToActivate Yuri (JumpToLeap Yuri (AllProJump Yuri hPro)))

theorem not_leap_yuri : ¬Leap Yuri :=
  fun hL =>
    YuriNotActivate (LeapToActivate Yuri hL)

-- Negation of the statement:
-- ¬((¬Leap Yuri) → (American Yuri ∧ Pro Yuri))
theorem neg_statement : ¬((¬Leap Yuri) → (American Yuri ∧ Pro Yuri)) := by
  intro h
  have hnotL : ¬Leap Yuri := not_leap_yuri
  have hAP : American Yuri ∧ Pro Yuri := h hnotL
  have hPro : Pro Yuri := And.right hAP
  exact not_pro_yuri hPro
