<lean>
axiom Event : Type
axiom Happy : Event → Prop
axiom Sad : Event → Prop

-- "Events are either happy or sad" (exclusive or: not both)
axiom xor : ∀ e : Event, (Happy e ∨ Sad e) ∧ ¬(Happy e ∧ Sad e)

-- "At least one event is happy"
axiom existsHappy : ∃ e : Event, Happy e

-- We prove that it is not the case that all events are sad.
theorem not_all_sad : ¬(∀ e : Event, Sad e) :=
by
  intro hAllSad
  obtain ⟨w, hwHappy⟩ := existsHappy
  have hwSad : Sad w := hAllSad w
  have hNoBoth : ¬(Happy w ∧ Sad w) := (xor w).2
  exact hNoBoth ⟨hwHappy, hwSad⟩
</lean>

ANSWER: False
