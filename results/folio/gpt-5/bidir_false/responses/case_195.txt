axiom obj : Type
axiom Jane : obj
axiom KiKi : obj

axiom Animal : obj → Prop
axiom Bark : obj → Prop
axiom Dog : obj → Prop
axiom Jumps : obj → Prop
axiom Legs : obj → Prop
axiom Terricolous : obj → Prop
axiom Likes : obj → obj → Prop

-- Jane does not like any animals that bark.
axiom A1 : ∀ x : obj, Animal x ∧ Bark x → ¬Likes Jane x
-- All dogs are animals that bark.
axiom A2 : ∀ x : obj, Dog x → Animal x ∧ Bark x
-- Jane likes all animals that jump.
axiom A3 : ∀ x : obj, Animal x ∧ Jumps x → Likes Jane x
-- If an animal has legs, then it jumps.
axiom A4 : ∀ x : obj, Animal x ∧ Legs x → Jumps x
-- If an animal is terricolous, then it has legs.
axiom A5 : ∀ x : obj, Animal x ∧ Terricolous x → Legs x
-- KiKi is an animal.
axiom A6 : Animal KiKi
-- If KiKi neither jumps nor has legs, then KiKi is terricolous or has legs.
axiom A7 : (¬Jumps KiKi ∧ ¬Legs KiKi) → (Terricolous KiKi ∨ Legs KiKi)

-- Some derived facts (do not yield ¬Terricolous KiKi)

theorem animals_that_jump_not_bark : ∀ x : obj, Animal x ∧ Jumps x → ¬Bark x := by
  intro x hx
  intro hb
  have hlike : Likes Jane x := A3 x hx
  have hnot : ¬Likes Jane x := A1 x ⟨hx.left, hb⟩
  exact hnot hlike

theorem terricolous_animals_jump : ∀ x : obj, Animal x ∧ Terricolous x → Jumps x := by
  intro x hx
  have hlegs : Legs x := A5 x hx
  exact A4 x ⟨hx.left, hlegs⟩

theorem dogs_not_terr : ∀ x : obj, Dog x → ¬Terricolous x := by
  intro x hdog
  intro hterr
  have hab : Animal x ∧ Bark x := A2 x hdog
  have hjump : Jumps x := terricolous_animals_jump x ⟨hab.left, hterr⟩
  have nbark : ¬Bark x := animals_that_jump_not_bark x ⟨hab.left, hjump⟩
  exact nbark hab.right

theorem terr_kiki_not_bark : Terricolous KiKi → ¬Bark KiKi := by
  intro hterr
  have hjump : Jumps KiKi := terricolous_animals_jump KiKi ⟨A6, hterr⟩
  exact animals_that_jump_not_bark KiKi ⟨A6, hjump⟩
