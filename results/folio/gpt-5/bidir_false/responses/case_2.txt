<lean>
axiom obj : Type

axiom InClub : obj → Prop
axiom PerformOften : obj → Prop
axiom AttendEngaged : obj → Prop
axiom Inactive : obj → Prop
axiom Chaperone : obj → Prop
axiom Student : obj → Prop
axiom YoungTeen : obj → Prop
axiom WishFurther : obj → Prop

axiom Bonnie : obj

-- 1. Club members who perform often attend and are very engaged.
axiom A1 : ∀ x : obj, InClub x ∧ PerformOften x → AttendEngaged x
-- 2. Club members either perform often or are inactive/disinterested.
axiom A2 : ∀ x : obj, InClub x → (PerformOften x ∨ Inactive x)
-- 3. Club members who chaperone dances are not students.
axiom A3 : ∀ x : obj, InClub x ∧ Chaperone x → ¬Student x
-- 4. All inactive/disinterested club members chaperone dances.
axiom A4 : ∀ x : obj, InClub x ∧ Inactive x → Chaperone x
-- 5. All young/teen club members who wish to further academics are students.
axiom A5 : ∀ x : obj, InClub x ∧ YoungTeen x ∧ WishFurther x → Student x
-- 6. Bonnie is in the club and either both attends/engaged and is a student,
--    or neither attends/engaged nor is a student.
axiom A6 : InClub Bonnie ∧ ((AttendEngaged Bonnie ∧ Student Bonnie) ∨ (¬AttendEngaged Bonnie ∧ ¬Student Bonnie))

theorem negate_statement :
  ¬((Chaperone Bonnie ∨ (¬Chaperone Bonnie → PerformOften Bonnie)) →
    ((YoungTeen Bonnie ∧ WishFurther Bonnie) ∧ Inactive Bonnie)) := by
  intro h
  have bon_in : InClub Bonnie := And.left A6
  -- Prove the antecedent: right disjunct (¬Chaperone → PerformOften)
  have imp1 : (¬Chaperone Bonnie → PerformOften Bonnie) := by
    intro hnot
    have disj := A2 Bonnie bon_in
    cases disj with
    | inl hp => exact hp
    | inr hinact =>
      have chap : Chaperone Bonnie := A4 Bonnie (And.intro bon_in hinact)
      exact False.elim (hnot chap)
  have P : (Chaperone Bonnie ∨ (¬Chaperone Bonnie → PerformOften Bonnie)) := Or.inr imp1
  -- Prove negation of the consequent
  have notQ : ¬((YoungTeen Bonnie ∧ WishFurther Bonnie) ∧ Inactive Bonnie) := by
    intro q
    have ytw : YoungTeen Bonnie ∧ WishFurther Bonnie := And.left q
    have hinact : Inactive Bonnie := And.right q
    have stu : Student Bonnie :=
      A5 Bonnie (And.intro bon_in (And.intro (And.left ytw) (And.right ytw)))
    have chap : Chaperone Bonnie := A4 Bonnie (And.intro bon_in hinact)
    have nstu : ¬Student Bonnie := A3 Bonnie (And.intro bon_in chap)
    exact nstu stu
  have Q := h P
  exact notQ Q
</lean>

ANSWER: False
