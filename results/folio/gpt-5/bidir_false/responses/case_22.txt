<lean>
axiom obj : Type
axiom StrangerThings : obj
axiom BlackMirror : obj

axiom Netflix : obj → Prop
axiom Popular : obj → Prop
axiom Binge : obj → Prop
axiom Download : obj → Prop
axiom ShareWithLisa : obj → Prop

-- Context facts
axiom ST_netflix : Netflix StrangerThings
axiom ST_popular : Popular StrangerThings

-- Rules
axiom R_popular_implies_binge : ∀ x : obj, Netflix x → Popular x → Binge x
axiom R_binge_iff_download : ∀ x : obj, Netflix x → (Binge x ↔ Download x)
axiom BM_not_download : ¬Download BlackMirror
axiom BM_netflix : Netflix BlackMirror
axiom R_binge_share_lisa : ∀ x : obj, Netflix x → Binge x → ShareWithLisa x

-- Prove the negation of the statement: "Black Mirror" is popular.
theorem bm_not_popular : ¬Popular BlackMirror := by
  intro hpop
  have hbinge : Binge BlackMirror := R_popular_implies_binge BlackMirror BM_netflix hpop
  have hiff : Binge BlackMirror ↔ Download BlackMirror := R_binge_iff_download BlackMirror BM_netflix
  have hdl : Download BlackMirror := (Iff.mp hiff) hbinge
  exact BM_not_download hdl
</lean>

ANSWER: False
