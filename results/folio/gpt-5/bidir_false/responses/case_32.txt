<lean>
axiom obj : Type
axiom At : obj → Prop
axiom GS : obj → Prop
axiom ONA : obj → Prop
axiom ProTennis : obj → Prop
axiom Athlete : obj → Prop
axiom Celebrity : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom InTaxHaven : obj → Prop

axiom R1 : ∀ x : obj, At x → (GS x ∨ ONA x)
axiom R2 : ∀ x : obj, At x ∧ GS x → ProTennis x
axiom R3 : ∀ x : obj, At x ∧ ONA x → Celebrity x
axiom R4 : ∀ x : obj, At x ∧ ProTennis x → Athlete x
axiom R5 : ∀ x : obj, At x ∧ Celebrity x → WellPaid x
axiom R6 : ∀ x : obj, At x ∧ Athlete x → Famous x
axiom R7 : ∀ x : obj, At x ∧ WellPaid x → InTaxHaven x

axiom Djokovic : obj
axiom AtDjokovic : At Djokovic
axiom Special : Famous Djokovic ∧ Athlete Djokovic → WellPaid Djokovic

theorem djokovic_in_tax_haven : InTaxHaven Djokovic := by
  have hdisj : GS Djokovic ∨ ONA Djokovic := R1 Djokovic AtDjokovic
  cases hdisj with
  | inl hgs =>
      have hpro : ProTennis Djokovic := R2 Djokovic ⟨AtDjokovic, hgs⟩
      have hath : Athlete Djokovic := R4 Djokovic ⟨AtDjokovic, hpro⟩
      have hfam : Famous Djokovic := R6 Djokovic ⟨AtDjokovic, hath⟩
      have hwp : WellPaid Djokovic := Special ⟨hfam, hath⟩
      exact R7 Djokovic ⟨AtDjokovic, hwp⟩
  | inr hona =>
      have hcel : Celebrity Djokovic := R3 Djokovic ⟨AtDjokovic, hona⟩
      have hwp : WellPaid Djokovic := R5 Djokovic ⟨AtDjokovic, hcel⟩
      exact R7 Djokovic ⟨AtDjokovic, hwp⟩
</lean>
