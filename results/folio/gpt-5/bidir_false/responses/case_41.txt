axiom obj : Type

axiom Kelly : obj → Prop
axiom Brutalist : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop
axiom ByMax : obj → Prop
axiom Adored : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Zaha x ∧ Adored x → Interesting x
axiom A2 : ∀ x : obj, Brutalist x ∧ Adored x → ¬Interesting x
axiom A3 : ∀ x : obj, Adored x → (Zaha x ∨ Kelly x)
axiom A4 : ∀ x : obj, Kelly x ∧ Adored x → Evocative x
axiom A5 : ∀ x : obj, Kelly x ∧ Adored x → Dreamy x
axiom A6 : ∀ x : obj, ByMax x ∧ Adored x ∧ Interesting x → Brutalist x ∧ Evocative x

-- Derivable helper: any design by Max that he adores is both evocative and dreamy
theorem adored_byMax_ev_dr : ∀ x : obj, ByMax x ∧ Adored x → Evocative x ∧ Dreamy x := by
  intro x
  intro h
  have hBy : ByMax x := And.left h
  have hAd : Adored x := And.right h
  have hZorK : Zaha x ∨ Kelly x := A3 x hAd
  cases hZorK with
  | inl hZ =>
      have hi : Interesting x := A1 x (And.intro hZ hAd)
      have hBE : Brutalist x ∧ Evocative x := A6 x (And.intro hBy (And.intro hAd hi))
      have hni : ¬Interesting x := A2 x (And.intro hBE.left hAd)
      have contra : False := hni hi
      exact And.intro hBE.right (False.elim contra)
  | inr hK =>
      have ev : Evocative x := A4 x (And.intro hK hAd)
      have dr : Dreamy x := A5 x (And.intro hK hAd)
      exact And.intro ev dr
