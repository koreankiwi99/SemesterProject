<lean>
axiom obj : Type
axiom NineteenEightyFour : obj

axiom DigitalMedia : obj → Prop
axiom AnalogMedia : obj → Prop
axiom PrintedText : obj → Prop
axiom StreamingService : obj → Prop
axiom HardcoverBook : obj → Prop

-- No digital media are analog.
axiom R1 : ∀ x : obj, DigitalMedia x → ¬AnalogMedia x
-- Every printed text is analog media.
axiom R2 : ∀ x : obj, PrintedText x → AnalogMedia x
-- All streaming services are digital media.
axiom R3 : ∀ x : obj, StreamingService x → DigitalMedia x
-- If an object is a hardcover book, then it is printed text.
axiom R4 : ∀ x : obj, HardcoverBook x → PrintedText x
-- If 1984 is a streaming service, then 1984 is a hardcover book.
axiom R5 : StreamingService NineteenEightyFour → HardcoverBook NineteenEightyFour

theorem not_streaming_1984 : ¬StreamingService NineteenEightyFour :=
by
  intro hS
  have hH : HardcoverBook NineteenEightyFour := R5 hS
  have hP : PrintedText NineteenEightyFour := R4 NineteenEightyFour hH
  have hA : AnalogMedia NineteenEightyFour := R2 NineteenEightyFour hP
  have hD : DigitalMedia NineteenEightyFour := R3 NineteenEightyFour hS
  have hNotA : ¬AnalogMedia NineteenEightyFour := R1 NineteenEightyFour hD
  exact hNotA hA
</lean>
