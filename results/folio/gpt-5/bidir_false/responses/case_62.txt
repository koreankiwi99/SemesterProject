<lean>
axiom obj : Type
axiom Pet : obj → Prop
axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop
axiom Ted : obj

-- All Brown Swiss cattle are cows.
axiom A1 : ∀ x : obj, BrownSwissCattle x → Cow x
-- Some pets are Brown Swiss cattle.
axiom A2 : ∃ x : obj, Pet x ∧ BrownSwissCattle x
-- All cows are domesticated animals.
axiom A3 : ∀ x : obj, Cow x → Domesticated x
-- Alligators are not domesticated animals.
axiom A4 : ∀ x : obj, Alligator x → ¬ Domesticated x
-- Ted is an alligator.
axiom A5 : Alligator Ted

-- From A4 and A5 we can derive that Ted is not domesticated.
theorem not_dom_Ted : ¬Domesticated Ted := A4 Ted A5

-- From A2, A1, and A3 we can derive that some pet is domesticated, but this does not involve Ted.
theorem exists_domesticated_pet : ∃ x : obj, Pet x ∧ Domesticated x := by
  rcases A2 with ⟨x, hx⟩
  rcases hx with ⟨hpet, hbs⟩
  have hcow : Cow x := A1 x hbs
  have hdom : Domesticated x := A3 x hcow
  exact Exists.intro x ⟨hpet, hdom⟩

-- There is no given axiom linking being a pet to being domesticated or to being an alligator,
-- so we cannot derive ¬Pet Ted from the premises.
