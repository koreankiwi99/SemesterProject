<lean>
axiom obj : Type
axiom Hannah : obj

axiom At : obj → Prop
axiom Work : obj → Prop
axiom Needs : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom NotPicky : obj → Prop
axiom SpendTime : obj → Prop

-- Rules from the context
axiom R1 : ∀ x : obj, At x → Work x → Needs x
axiom R2 : ∀ x : obj, At x → Order x → Work x
axiom R3 : ∀ x : obj, At x → (Order x ∨ Enjoy x)
axiom R4 : ∀ x : obj, At x → Enjoy x → NotPicky x
axiom R5 : ∀ x : obj, At x → Enjoy x → SpendTime x

-- Facts about Hannah
axiom H_at : At Hannah
axiom H_work : Work Hannah
axiom H_cond : Needs Hannah → (NotPicky Hannah ∧ ¬Needs Hannah)

-- Prove the negation of the statement:
-- "Hannah is at Mary's school and she is not a picky eater and spends a lot of her time eating and catching up with friends in the campus dining halls."
theorem negate_statement : ¬(At Hannah ∧ NotPicky Hannah ∧ SpendTime Hannah) := by
  intro h
  have hAt : At Hannah := h.left
  have hWork : Work Hannah := H_work
  have hNeeds : Needs Hannah := R1 Hannah hAt hWork
  have hNotNeeds : ¬Needs Hannah := (H_cond hNeeds).right
  exact hNotNeeds hNeeds
</lean>

ANSWER: False
