axiom Deposit : obj → Money
axiom LE : Money → Money → Prop  -- ≤

axiom GT : Money → Money → Prop  -- >

-- Contextual facts
-- Pets are allowed in some managed buildings.
axiom C1 : ∃ b : obj, Managed b ∧ AllowsPets b
-- A deposit is required to rent an apartment in a managed building.
axiom C2 : ∀ p b, Managed b ∧ WillRent p b → DepositRequired b
-- The security deposit is at least the monthly rent at a managed building.
axiom C3 : ∀ b, Managed b → GE (Deposit b) (Rent b)
-- Fluffy is Tom's cat.
axiom C4 : Cat Fluffy
-- Cats are pets.
axiom C5 : ∀ x, Cat x → Pet x
-- The Olive Garden is a managed building.
axiom C6 : Managed OliveGarden
-- The monthly rent at the Olive Garden is $2000.
axiom C7 : Rent OliveGarden = M2000
-- $2000 is more than $1500.
axiom C8 : GT M2000 M1500
-- If a managed building allows pets, then people are allowed to move in with a pet.
axiom C9 : ∀ p pet b, Managed b ∧ AllowsPets b ∧ Pet pet → AllowedMoveInWith p pet b
-- Tom will rent an apartment in a managed building if he is allowed to move in with Fluffy,
-- and the security deposit is no more than $1500.
axiom C10 : ∀ b, Managed b ∧ AllowedMoveInWith Tom Fluffy b ∧ LE (Deposit b) M1500 → WillRent Tom b

-- Order interaction axioms (to reason about amounts)
axiom ge_gt_trans : ∀ x y z, GE x y → GT y z → GT x z
axiom gt_not_le : ∀ x y, GT x y → ¬ LE x y

-- A derived fact: Fluffy is a pet.

-- From the deposit ≥ rent at OG and 2000 > 1500, we can derive that the deposit at OG is not ≤ 1500.

  have hge' : GE (Deposit OliveGarden) M2000 := by
    simpa [hrent] using hge
  have hgt : GT (Deposit OliveGarden) M1500 := ge_gt_trans (Deposit OliveGarden) M2000 M1500 hge' C8
  exact gt_not_le (Deposit OliveGarden) M1500 hgt
