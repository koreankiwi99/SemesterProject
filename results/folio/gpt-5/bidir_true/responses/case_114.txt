axiom Taller : obj → obj → Prop        -- Taller x y means: x is taller than y

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlockShooting x y

-- Michael is a man who is taller than everyone else in his class.

axiom M_taller_than_classmates : ∀ y : obj, InMichaelClass y → y ≠ Michael → Taller Michael y

-- Transitivity of "taller than"
axiom taller_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom P_facts : Man Peter ∧ Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom block_if_no_jump : ∀ p : obj, ¬ JumpsWhenShooting p → CanBlockShooting Michael p

-- Michael cannot block Windy's shooting.
axiom not_block_windy : ¬ CanBlockShooting Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom jumpers_are_great : ∀ x : obj, JumpsWhenShooting x → GreatShooter x

-- Derivable fact: Peter is taller than any classmate of Michael distinct from Michael.

  ∀ y : obj, InMichaelClass y → y ≠ Michael → Taller Peter y :=
