<lean>
axiom obj : Type

-- Predicates
axiom Animal : obj → Prop
axiom Heck : obj → Prop
axiom Aurochs : obj → Prop
axiom BredBack : obj → Prop
axiom ArtificialSelection : obj → Prop
axiom ToBeBredBack : obj → Prop
axiom ExtinctAnimal : obj → Prop
axiom Resembles : obj → obj → Prop

-- Context axioms
-- Breeding back is a form of artificial selection
axiom R1 : ∀ x : obj, BredBack x → ArtificialSelection x

-- Heck cattle were bred back (existence of such individuals)
axiom T1 : ∃ x : obj, Heck x ∧ BredBack x

-- Heck cattle are animals
axiom T2 : ∀ x : obj, Heck x → Animal x

-- Aurochs are animals
axiom T3 : ∀ x : obj, Aurochs x → Animal x

-- Some animals to be bred back resemble extinct animals
axiom T4 : ∃ x : obj, Animal x ∧ ToBeBredBack x ∧ (∃ y : obj, ExtinctAnimal y ∧ Resembles x y)

-- Goal: Some Heck cattle are artificially selected
theorem some_heck_artificially_selected : ∃ x : obj, Heck x ∧ ArtificialSelection x := by
  rcases T1 with ⟨x, hx⟩
  rcases hx with ⟨hxHeck, hxBred⟩
  have hxArt : ArtificialSelection x := R1 x hxBred
  exact ⟨x, And.intro hxHeck hxArt⟩
</lean>

ANSWER: True
