axiom Organic : obj → Prop
axiom Alkane : obj → Prop
axiom ChemicalCompound : obj → Prop
axiom ContainsCarbon : obj → Prop

-- All hydrocarbons are organic compounds.
axiom A1 : ∀ x : obj, Hydrocarbon x → Organic x
-- All alkanes are hydrocarbons.
axiom A2 : ∀ x : obj, Alkane x → Hydrocarbon x
-- All organic compounds are chemical compounds.
axiom A3 : ∀ x : obj, Organic x → ChemicalCompound x
-- All organic compounds contain carbon.
axiom A4 : ∀ x : obj, Organic x → ContainsCarbon x
-- No chemical compounds contain only one element.
axiom A5 : ∀ x : obj, ChemicalCompound x → ¬ OnlyOneElement x
-- The mixture either both is a chemical compound and contains only one element, or neither.
axiom A6 : (ChemicalCompound Mixture ∧ OnlyOneElement Mixture) ∨ (¬ ChemicalCompound Mixture ∧ ¬ OnlyOneElement Mixture)

-- From A5 and A6, we can derive that the mixture is not a chemical compound and does not contain only one element.
theorem not_chemical_compound_mixture : ¬ ChemicalCompound Mixture := by
  have h := A6
  cases h with
  | inl hboth =>
      have hfalse : False := by
        have hnot := A5 Mixture hboth.left
        exact hnot hboth.right
      exact False.elim hfalse
  | inr hneither =>
      exact hneither.left

theorem not_only_one_element_mixture : ¬ OnlyOneElement Mixture := by
  have h := A6
  cases h with
  | inl hboth =>
      have hfalse : False := by
        have hnot := A5 Mixture hboth.left
        exact hnot hboth.right
      exact False.elim hfalse
  | inr hneither =>
      exact hneither.right

-- If the mixture were an alkane, it would be a chemical compound, contradicting not_chemical_compound_mixture.

  intro hAlk
  have hHydro : Hydrocarbon Mixture := A2 Mixture hAlk
  have hOrg : Organic Mixture := A1 Mixture hHydro
  have hChem : ChemicalCompound Mixture := A3 Mixture hOrg
