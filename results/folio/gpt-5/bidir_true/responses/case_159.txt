<lean>
axiom obj : Type
axiom James : obj

axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom UniPTJob : obj → Prop
axiom TakesDatabase : obj → Prop
axiom TakesWithDavid : obj → Prop
axiom WorksLab : obj → Prop

axiom A1 : ∀ x : obj, WorksLibrary x → ¬ FromCS x
axiom A2 : ∀ x : obj, UniPTJob x → WorksLibrary x
axiom A3 : ∀ x : obj, TakesDatabase x → FromCS x
axiom A4 : ∀ x : obj, TakesWithDavid x → TakesDatabase x
axiom A5 : ∀ x : obj, WorksLab x → TakesWithDavid x
axiom A6 : WorksLab James
axiom A7 : ¬ WorksLab James ∧ ¬ UniPTJob James

theorem james_takes_database : TakesDatabase James :=
  let h1 : TakesWithDavid James := A5 James A6
  A4 James h1
</lean>

ANSWER: True
