axiom Person : Type
axiom Taylor : Person

axiom EatSalad : Person → Prop
axiom VeryConscious : Person → Prop
axiom GrewUpHC : Person → Prop
axiom DisregardPB : Person → Prop
axiom FulfillDaily : Person → Prop
axiom VisitGymDaily : Person → Prop

-- All who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : Person, EatSalad x → VeryConscious x
-- All who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : Person, GrewUpHC x → EatSalad x
-- All who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : Person, FulfillDaily x → GrewUpHC x
-- All who disregard their physical well-being are not very conscious.
axiom A4 : ∀ x : Person, DisregardPB x → ¬ VeryConscious x
-- If people visit the gym at least once a day, then they fulfill their daily nutritional intakes.
axiom A5 : ∀ x : Person, VisitGymDaily x → FulfillDaily x
-- Taylor either (grew up HC and disregards) or (neither).
axiom A6 : (GrewUpHC Taylor ∧ DisregardPB Taylor) ∨ (¬ GrewUpHC Taylor ∧ ¬ DisregardPB Taylor)

-- From the premises, we can actually derive that Taylor does not visit the gym daily.
theorem taylor_not_visit : ¬ VisitGymDaily Taylor := by
  -- First, show the first disjunct in A6 is impossible.
  have hnotA : ¬ (GrewUpHC Taylor ∧ DisregardPB Taylor) := by
    intro hA
    have g : GrewUpHC Taylor := hA.left
    have d : DisregardPB Taylor := hA.right
    have e : EatSalad Taylor := A2 Taylor g
    have v : VeryConscious Taylor := A1 Taylor e
    have nv : ¬ VeryConscious Taylor := A4 Taylor d
    exact nv v
  -- Hence the second disjunct must hold: ¬GrewUpHC ∧ ¬DisregardPB.
  have hdisj := A6
  have hB : (¬ GrewUpHC Taylor ∧ ¬ DisregardPB Taylor) := by
    apply Or.elim hdisj
    · intro hA; exact False.elim (hnotA hA)
    · intro hB; exact hB
  have notG : ¬ GrewUpHC Taylor := hB.left
  -- Contraposition of A3 gives ¬GrewUpHC → ¬FulfillDaily.
  have notF : ¬ FulfillDaily Taylor := by
    intro f
    have g : GrewUpHC Taylor := A3 Taylor f
    exact notG g
  -- Contraposition of A5 gives ¬FulfillDaily → ¬VisitGymDaily.
  intro vgym
  exact notF (A5 Taylor vgym)
