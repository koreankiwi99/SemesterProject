axiom Horse : obj → Prop
axiom HasHooves : obj → Prop
axiom all_horses_have_hooves : ∀ x : obj, Horse x → HasHooves x
axiom no_humans_have_hooves : ∀ x : obj, Human x → ¬ HasHooves x

-- Target statement: ∃ x, Human x ∧ Horse x
-- This does not follow; in fact we can prove its negation.
theorem no_human_is_horse : ¬ ∃ x : obj, Human x ∧ Horse x := by
  intro h
  have contra_for : ∀ x : obj, Human x ∧ Horse x → False := fun x hx =>
    have hx_hooves : HasHooves x := all_horses_have_hooves x hx.right
    have hx_no_hooves : ¬ HasHooves x := no_humans_have_hooves x hx.left
    show False from hx_no_hooves hx_hooves
  exact Exists.elim h (fun x hx => contra_for x hx)
