<lean>
axiom obj : Type
axiom Yuri : obj

axiom Pro : obj → Prop
axiom American : obj → Prop
axiom Jump : obj → Prop
axiom Leap : obj → Prop
axiom Activate : obj → Prop

-- Some professional basketball players are not American nationals.
axiom E1 : ∃ x : obj, Pro x ∧ ¬ American x
-- All professional basketball players can do jump shots.
axiom R1 : ∀ x : obj, Pro x → Jump x
-- If someone can jump shots, they leap straight into the air.
axiom R2 : ∀ x : obj, Jump x → Leap x
-- If someone leaps straight into the air, they activate their leg muscles.
axiom R3 : ∀ x : obj, Leap x → Activate x
-- Yuri does not activate his leg muscles.
axiom T2 : ¬ Activate Yuri

theorem yuri_not_american_pro : ¬ (American Yuri ∧ Pro Yuri) := by
  intro h
  have hpro : Pro Yuri := And.right h
  have hyact : Activate Yuri := R3 Yuri (R2 Yuri (R1 Yuri hpro))
  exact T2 hyact
</lean>

ANSWER: True
