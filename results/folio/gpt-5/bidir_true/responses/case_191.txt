axiom obj : Type
axiom Yuri : obj

axiom American : obj → Prop
axiom Pro : obj → Prop
axiom CanJump : obj → Prop
axiom LeapStraight : obj → Prop
axiom ActivateLegs : obj → Prop

axiom someProNotAmerican : ∃ x : obj, Pro x ∧ ¬ American x
axiom allProCanJump : ∀ x : obj, Pro x → CanJump x
axiom jumpImpliesLeap : ∀ x : obj, CanJump x → LeapStraight x
axiom leapImpliesActivate : ∀ x : obj, LeapStraight x → ActivateLegs x
axiom Yuri_not_activate : ¬ ActivateLegs Yuri

-- Derived: professionals activate their leg muscles
lemma pro_implies_activate : ∀ x : obj, Pro x → ActivateLegs x := by
  intro x
  intro hx
  have hj : CanJump x := by exact allProCanJump x hx
  have hl : LeapStraight x := by exact jumpImpliesLeap x hj
  have hm : ActivateLegs x := by exact leapImpliesActivate x hl
  exact hm

-- Hence Yuri is not a professional basketball player
lemma not_pro_Yuri : ¬ Pro Yuri := by
  intro hpro
  have hm : ActivateLegs Yuri := by
    exact pro_implies_activate Yuri hpro
  exact Yuri_not_activate hm

-- From this, the antecedent is true: ¬(American Yuri ∧ Pro Yuri)
lemma antecedent_true : ¬ (American Yuri ∧ Pro Yuri) := by
  intro h
  cases h with
  | intro hAm hPro =>
    exact not_pro_Yuri hPro

-- The queried statement does not follow; in fact its negation is provable
theorem negate_statement : ¬ ((¬ (American Yuri ∧ Pro Yuri)) → Pro Yuri) := by
  intro hImp
  have hPro : Pro Yuri := by
    exact hImp antecedent_true
  exact not_pro_Yuri hPro
