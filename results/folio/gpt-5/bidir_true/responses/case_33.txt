<lean>
axiom obj : Type
axiom DiamondMine : obj
axiom RoderickStrong : obj
axiom Stable : obj → Prop
axiom Leads : obj → obj → Prop
axiom T1 : Stable DiamondMine
axiom T2 : Leads RoderickStrong DiamondMine

theorem strong_leads_a_stable : ∃ s : obj, Stable s ∧ Leads RoderickStrong s := by
  apply Exists.intro DiamondMine
  apply And.intro
  · exact T1
  · exact T2
</lean>

ANSWER: True
