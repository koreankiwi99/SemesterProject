<lean>
axiom obj : Type
axiom YaleUniversity : obj
axiom NewHaven : obj

axiom PrivateIvyLeagueResearchUniversity : obj → Prop
axiom MovedTo : obj → obj → Prop

axiom T1 : PrivateIvyLeagueResearchUniversity YaleUniversity
axiom T2 : MovedTo YaleUniversity NewHaven

theorem exists_private_ivy_research_uni_moved_to_new_haven :
  ∃ x : obj, PrivateIvyLeagueResearchUniversity x ∧ MovedTo x NewHaven := by
  refine Exists.intro YaleUniversity ?_
  exact And.intro T1 T2
</lean>

ANSWER: True
