<lean>
axiom Series : Type
axiom Person : Type
axiom Title : Type

axiom Badults : Series
axiom AndrewCollins : Person
axiom TSD : Title  -- "The Secret Dude Society"

axiom WorkingTitle : Series → Title → Prop
axiom ScriptEditor : Person → Series → Prop

-- From the context:
axiom WT_Badults : WorkingTitle Badults TSD
axiom SE_Badults : ScriptEditor AndrewCollins Badults

-- Claim: Andrew Collins was the script editor for a series with the working title "The Secret Dude Society".
theorem claim : ∃ s : Series, WorkingTitle s TSD ∧ ScriptEditor AndrewCollins s := by
  apply Exists.intro Badults
  constructor
  · exact WT_Badults
  · exact SE_Badults
</lean>

ANSWER: True
