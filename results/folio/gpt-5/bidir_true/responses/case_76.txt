<lean>
axiom obj : Type
axiom Hannah : obj

axiom AtMS : obj → Prop
axiom Work : obj → Prop
axiom NeedMoney : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom SpendTime : obj → Prop
axiom Picky : obj → Prop

axiom R1 : ∀ x : obj, AtMS x ∧ Work x → NeedMoney x
axiom R2 : ∀ x : obj, AtMS x ∧ OrderTakeout x → Work x
axiom R3 : ∀ x : obj, AtMS x → (OrderTakeout x ∨ EnjoyDining x)
axiom R4 : ∀ x : obj, AtMS x ∧ EnjoyDining x → ¬ Picky x
axiom R5 : ∀ x : obj, AtMS x ∧ EnjoyDining x → SpendTime x

axiom T1 : AtMS Hannah
axiom T2 : Work Hannah ∧ (NeedMoney Hannah → (¬ Picky Hannah ∧ ¬ NeedMoney Hannah))

theorem hannah_conclusion : AtMS Hannah ∧ ¬ Picky Hannah ∧ SpendTime Hannah := by
  have hAt : AtMS Hannah := T1
  have hWork : Work Hannah := And.left T2
  have hNeed : NeedMoney Hannah := R1 Hannah (And.intro hAt hWork)
  have hcond : NeedMoney Hannah → (¬ Picky Hannah ∧ ¬ NeedMoney Hannah) := And.right T2
  have hnp_and_notnm : ¬ Picky Hannah ∧ ¬ NeedMoney Hannah := hcond hNeed
  have hNotPicky : ¬ Picky Hannah := And.left hnp_and_notnm
  have hNotNeed : ¬ NeedMoney Hannah := And.right hnp_and_notnm
  have fals : False := hNotNeed hNeed
  have hSpend : SpendTime Hannah := False.elim fals
  exact And.intro hAt (And.intro hNotPicky hSpend)
</lean>

ANSWER: True
