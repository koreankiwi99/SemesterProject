<lean>
axiom obj : Type

-- Entities
axiom SASC : obj                     -- The Surprising Adventures of Sir Digby Chicken Caesar (the movie)
axiom Digby : obj                    -- Sir Digby
axiom DigbyNemesis : obj             -- Sir Digby's nemesis

-- Predicates
axiom SuperheroMovie : obj → Prop
axiom GoodGuy : obj → Prop
axiom BadGuy : obj → Prop
axiom InMovie : obj → obj → Prop     -- InMovie character movie
axiom NamedAfter : obj → obj → Prop  -- NamedAfter movie character
axiom Fights : obj → obj → Prop
axiom Wins : obj → Prop

-- Axioms from the context
axiom A1 : ∀ m : obj, SuperheroMovie m → ∀ x : obj, (InMovie x m ∧ GoodGuy x) → Wins x
axiom A2 : SuperheroMovie SASC
axiom A3 : ∀ x y : obj, GoodGuy x ∧ BadGuy y → Fights x y ∧ Fights y x
axiom A4 : Fights Digby DigbyNemesis
axiom A5 : ∀ m c : obj, SuperheroMovie m → NamedAfter m c → GoodGuy c
axiom A6 : NamedAfter SASC Digby
axiom A7 : ∀ x y : obj, Fights x y → Wins x → ¬ Wins y
axiom A8 : ∀ m c : obj, SuperheroMovie m → NamedAfter m c → InMovie c m

-- Goal: Sir Digby’s nemesis does not win.
theorem nemesis_not_win : ¬ Wins DigbyNemesis := by
  have hGood : GoodGuy Digby := A5 SASC Digby A2 A6
  have hIn : InMovie Digby SASC := A8 SASC Digby A2 A6
  have hWinDigby : Wins Digby := A1 SASC A2 Digby (And.intro hIn hGood)
  have hFight : Fights Digby DigbyNemesis := A4
  exact A7 Digby DigbyNemesis hFight hWinDigby
</lean>

ANSWER: True
