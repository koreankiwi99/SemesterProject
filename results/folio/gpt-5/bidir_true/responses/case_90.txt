<lean>
axiom Task : Type
axiom MT : Task

axiom NLPTask : Task → Prop
axiom Generation : Task → Prop
axiom Understanding : Task → Prop
axiom OutputTextSeq : Task → Prop

-- A natural language processing task can be categorized as either generation or understanding
axiom A1 : ∀ t : Task, NLPTask t → (Generation t ∨ Understanding t)
-- If the output is a text sequence, then it is a language generation task
axiom A2 : ∀ t : Task, OutputTextSeq t → Generation t
-- Machine translation is an NLP task
axiom A3 : NLPTask MT
-- The output of machine translation is a text sequence
axiom A4 : OutputTextSeq MT

theorem MT_is_generation : Generation MT := A2 MT A4
</lean>

ANSWER: True
