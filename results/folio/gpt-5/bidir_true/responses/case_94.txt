<lean>
axiom obj : Type
axiom Managed : obj → Prop
axiom OliveGarden : obj
axiom MonthlyRent : obj → Nat
axiom SecurityDeposit : obj → Nat

axiom OG_managed : Managed OliveGarden
axiom OG_rent : MonthlyRent OliveGarden = 2000
axiom DepositRule : ∀ b : obj, Managed b → (SecurityDeposit b = MonthlyRent b ∨ SecurityDeposit b > MonthlyRent b)

theorem deposit_olive_garden_eq2000_or_more : SecurityDeposit OliveGarden = 2000 ∨ SecurityDeposit OliveGarden > 2000 :=
by
  have h := DepositRule OliveGarden OG_managed
  cases h with
  | inl hEq =>
      left
      simpa [OG_rent] using hEq
  | inr hGt =>
      right
      simpa [OG_rent] using hGt
</lean>

ANSWER: True
