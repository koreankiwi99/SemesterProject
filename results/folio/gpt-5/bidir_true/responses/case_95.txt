axiom LegalEntity : obj → Prop

axiom A2 : ∀ x : obj, Company x → BusinessOrganization x

axiom A4 : ∀ x : obj, LegalEntity x → CreatedUnderLaw x
axiom A5 : ∀ x : obj, LegalEntity x → LegalObligations x
axiom A6 : CreatedUnderLaw HWBC → ¬ PrivateCompany HWBC

-- We can derive that HWBC is not a private company
theorem not_private_company : ¬ PrivateCompany HWBC :=
by
  intro hpc
  have hco : Company HWBC := A3 HWBC hpc
  have hbo : BusinessOrganization HWBC := A2 HWBC hco
  have hle : LegalEntity HWBC := A1 HWBC hbo
  have hcul : CreatedUnderLaw HWBC := A4 HWBC hle
  have hnpc : ¬ PrivateCompany HWBC := A6 hcul
  exact hnpc hpc
