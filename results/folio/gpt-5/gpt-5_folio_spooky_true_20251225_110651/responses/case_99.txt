<lean>
axiom obj : Type

-- Predicates over animals (domain is animals)
axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom MFRepro : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop

-- Constant for Harry
axiom Harry : obj

-- Axioms from the context
axiom A1 : ∀ x : obj, Invertebrate x ∨ Vertebrate x
axiom A2 : ∀ x : obj, Backbone x → MFRepro x
axiom A3 : ∀ x : obj, Vertebrate x → Backbone x
axiom A4 : ∀ x : obj, Bee x → ¬ MFRepro x
axiom A5 : ∀ x : obj, QueenBee x → Bee x
axiom A6 : Bee Harry

theorem harry_invertebrate_or_queenbee : Invertebrate Harry ∨ QueenBee Harry := by
  have hNotMF : ¬ MFRepro Harry := A4 Harry A6
  have hNotV : ¬ Vertebrate Harry := fun hv => hNotMF (A2 Harry (A3 Harry hv))
  have hInv : Invertebrate Harry := Or.resolve_right (A1 Harry) hNotV
  exact Or.inl hInv
</lean>

ANSWER: True
