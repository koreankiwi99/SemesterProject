axiom VeryConscious : Person → Prop
axiom GrewUpHealthHome : Person → Prop
axiom FulfillDailyNutrition : Person → Prop
axiom Disregard : Person → Prop
axiom VisitGymDaily : Person → Prop

-- Premises
axiom A1 : ∀ x : Person, EatSalad x → VeryConscious x
axiom A2 : ∀ x : Person, GrewUpHealthHome x → EatSalad x
axiom A3 : ∀ x : Person, FulfillDailyNutrition x → GrewUpHealthHome x
axiom A4 : ∀ x : Person, Disregard x → ¬ VeryConscious x
axiom A5 : ∀ x : Person, VisitGymDaily x → FulfillDailyNutrition x
axiom A6 : (GrewUpHealthHome Taylor ∧ Disregard Taylor) ∨ (¬ GrewUpHealthHome Taylor ∧ ¬ Disregard Taylor)

theorem not_G_and_D_Taylor : ¬ (GrewUpHealthHome Taylor ∧ Disregard Taylor) := by
  intro h
  have hG : GrewUpHealthHome Taylor := h.left
  have hD : Disregard Taylor := h.right
  have hEat : EatSalad Taylor := A2 Taylor hG
  have hV : VeryConscious Taylor := A1 Taylor hEat
  have hNV : ¬ VeryConscious Taylor := A4 Taylor hD
  exact hNV hV

theorem notG_and_notD_Taylor : ¬ GrewUpHealthHome Taylor ∧ ¬ Disregard Taylor := by
  have hdisj := A6
  have hcontra := not_G_and_D_Taylor
  apply Or.elim hdisj
  · intro hgd
    have : False := hcontra hgd
    exact False.elim this
  · intro h
    exact h

theorem taylor_not_visit : ¬ VisitGymDaily Taylor := by
  intro hvisit
  have hful : FulfillDailyNutrition Taylor := A5 Taylor hvisit
  have hG : GrewUpHealthHome Taylor := A3 Taylor hful
  have hNotG : ¬ GrewUpHealthHome Taylor := (notG_and_notD_Taylor).left
  exact hNotG hG
