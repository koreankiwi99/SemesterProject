<lean>
axiom obj : Type
axiom EatsSaladsRegularly : obj → Prop
axiom VeryConscious : obj → Prop
axiom GrewUpHC : obj → Prop
axiom FulfillsDailyIntakes : obj → Prop
axiom DisregardsWellBeing : obj → Prop
axiom VisitsGymDaily : obj → Prop

axiom A1 : ∀ x : obj, EatsSaladsRegularly x → VeryConscious x
axiom A2 : ∀ x : obj, GrewUpHC x → EatsSaladsRegularly x
axiom A3 : ∀ x : obj, FulfillsDailyIntakes x → GrewUpHC x
axiom A4 : ∀ x : obj, DisregardsWellBeing x → ¬ VeryConscious x
axiom A5 : ∀ x : obj, VisitsGymDaily x → FulfillsDailyIntakes x

axiom Taylor : obj
axiom A6 : (GrewUpHC Taylor ∧ DisregardsWellBeing Taylor) ∨ (¬ GrewUpHC Taylor ∧ ¬ DisregardsWellBeing Taylor)

theorem taylor_neither_grew_up_nor_visits_gym : ¬ GrewUpHC Taylor ∧ ¬ VisitsGymDaily Taylor := by
  have hG_to_VC : ∀ x : obj, GrewUpHC x → VeryConscious x := by
    intro x
    intro hx
    exact A1 x (A2 x hx)
  have hnot_left : ¬ (GrewUpHC Taylor ∧ DisregardsWellBeing Taylor) := by
    intro hgd
    have vc : VeryConscious Taylor := hG_to_VC Taylor hgd.left
    have nvc : ¬ VeryConscious Taylor := A4 Taylor hgd.right
    exact nvc vc
  have hRight : (¬ GrewUpHC Taylor ∧ ¬ DisregardsWellBeing Taylor) := by
    cases A6 with
    | inl hgd => exact False.elim (hnot_left hgd)
    | inr h => exact h
  have hNotG : ¬ GrewUpHC Taylor := hRight.left
  have hV_to_G : ∀ x : obj, VisitsGymDaily x → GrewUpHC x := by
    intro x
    intro hv
    exact A3 x (A5 x hv)
  have hNotV : ¬ VisitsGymDaily Taylor := by
    intro hv
    exact hNotG (hV_to_G Taylor hv)
  exact And.intro hNotG hNotV
</lean>

ANSWER: True
