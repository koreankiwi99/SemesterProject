<lean>
axiom obj : Type

axiom Marvin : obj

axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromMars : obj → Prop
axiom FromEarth : obj → Prop
axiom HighlyIntelligent : obj → Prop

-- 1. All aliens are extraterrestrials.
axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x
-- 2. If someone is from Mars, then they are an alien.
axiom A2 : ∀ x : obj, FromMars x → Alien x
-- 3. No extraterrestrials are human.
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬ Human x
-- 4. All highly intelligent beings from Earth are humans.
axiom A4 : ∀ x : obj, (HighlyIntelligent x ∧ FromEarth x) → Human x
-- 5. Marvin is a highly intelligent being.
axiom A5 : HighlyIntelligent Marvin
-- 6. Marvin is either from Earth and from Mars, or he is from neither.
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
-- 7. If Marvin is not from Earth, then Marvin is an extraterrestrial.
axiom A7 : ¬ FromEarth Marvin → Extraterrestrial Marvin

-- From being both Earth and Mars we derive a contradiction.
theorem not_bothEarthMars : ¬ (FromEarth Marvin ∧ FromMars Marvin) := by
  intro h
  have hE : FromEarth Marvin := And.left h
  have hM : FromMars Marvin := And.right h
  have hAlien : Alien Marvin := A2 Marvin hM
  have hExtra : Extraterrestrial Marvin := A1 Marvin hAlien
  have hHuman : Human Marvin := A4 Marvin (And.intro A5 hE)
  have hNotHuman : ¬ Human Marvin := A3 Marvin hExtra
  exact hNotHuman hHuman

-- Therefore the "neither" branch of A6 must hold, yielding not from Earth and not from Mars.
theorem not_from_earth : ¬ FromEarth Marvin := by
  have hneither : ¬ FromEarth Marvin ∧ ¬ FromMars Marvin := by
    cases A6 with
    | inl h => exact False.elim (not_bothEarthMars h)
    | inr h => exact h
  exact And.left hneither

theorem not_from_mars : ¬ FromMars Marvin := by
  have hneither : ¬ FromEarth Marvin ∧ ¬ FromMars Marvin := by
    cases A6 with
    | inl h => exact False.elim (not_bothEarthMars h)
    | inr h => exact h
  exact And.right hneither

-- Hence Marvin is extraterrestrial and not human.
theorem extraterrestrial_marvin : Extraterrestrial Marvin := by
  apply A7
  exact not_from_earth

theorem not_human_marvin : ¬ Human Marvin := by
  exact A3 Marvin extraterrestrial_marvin
