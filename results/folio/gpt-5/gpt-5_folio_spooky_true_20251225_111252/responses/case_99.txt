<lean>
axiom obj : Type
axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom MF : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop
axiom Harry : obj

-- Premises
axiom A1 : ∀ x : obj, Invertebrate x ∨ Vertebrate x
axiom A2 : ∀ x : obj, Backbone x → MF x
axiom A3 : ∀ x : obj, Vertebrate x → Backbone x
axiom A4 : ∀ x : obj, Bee x → ¬ MF x
axiom A5 : ∀ x : obj, QueenBee x → Bee x
axiom A6 : Bee Harry

theorem harry_invertebrate_or_queenbee : Invertebrate Harry ∨ QueenBee Harry := by
  have hNotMF : ¬ MF Harry := A4 Harry A6
  have v2mf : Vertebrate Harry → MF Harry := by
    intro hv
    exact A2 Harry (A3 Harry hv)
  have disj : Invertebrate Harry ∨ Vertebrate Harry := A1 Harry
  cases disj with
  | inl hi =>
      exact Or.inl hi
  | inr hv =>
      have : False := hNotMF (v2mf hv)
      exact Or.inl (False.elim this)
</lean>

ANSWER: True
