axiom Student : obj → Prop

axiom A1 : ∀ x : obj, Club x ∧ PerformOften x → AttendEngaged x
axiom A2 : ∀ x : obj, Club x → PerformOften x ∨ InactiveDisinterested x
axiom A3 : ∀ x : obj, Club x ∧ Chaperone x → ¬ Student x
axiom A4 : ∀ x : obj, Club x ∧ InactiveDisinterested x → Chaperone x
axiom A5 : ∀ x : obj, Club x ∧ (YoungChild x ∨ Teenager x) ∧ WishesFurther x → Student x

axiom Bonnie : obj
axiom Bonnie_in : Club Bonnie
axiom Bonnie_xnor : (AttendEngaged Bonnie ∧ Student Bonnie) ∨ (¬AttendEngaged Bonnie ∧ ¬Student Bonnie)

theorem bonnie_perform_implies_student : PerformOften Bonnie → Student Bonnie :=
by
  intro hperf
  have hae : AttendEngaged Bonnie := A1 Bonnie (And.intro Bonnie_in hperf)
  cases Bonnie_xnor with
  | inl h =>
    exact h.right
  | inr h =>
    have contra : False := h.left hae
    exact False.elim contra

theorem bonnie_inactive_implies_not_student : InactiveDisinterested Bonnie → ¬ Student Bonnie :=
by
  intro hinact
  have hchap : Chaperone Bonnie := A4 Bonnie (And.intro Bonnie_in hinact)
  have hnotS : ¬ Student Bonnie := A3 Bonnie (And.intro Bonnie_in hchap)
  exact hnotS

theorem bonnie_perform_and_inactive_contradiction :
  PerformOften Bonnie → InactiveDisinterested Bonnie → False :=
by
  intro hperf hinact
  have s : Student Bonnie := bonnie_perform_implies_student hperf
