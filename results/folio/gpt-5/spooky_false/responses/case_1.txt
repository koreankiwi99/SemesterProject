axiom PerformOften : obj → Prop
axiom Engaged : obj → Prop

axiom Student : obj → Prop
axiom Young : obj → Prop
axiom Teen : obj → Prop
axiom Wish : obj → Prop

axiom Bonnie : obj

-- Premises
axiom P1 : ∀ x : obj, Club x ∧ PerformOften x → Engaged x
axiom P2 : ∀ x : obj, Club x → PerformOften x ∨ Inactive x
axiom P3 : ∀ x : obj, Club x ∧ Chaperone x → ¬ Student x
axiom P4 : ∀ x : obj, Club x ∧ Inactive x → Chaperone x
axiom P5 : ∀ x : obj, ((Young x ∨ Teen x) ∧ Club x ∧ Wish x) → Student x
axiom P6 : Club Bonnie ∧ ((Engaged Bonnie ∧ Student Bonnie) ∨ (¬Engaged Bonnie ∧ ¬Student Bonnie))

-- Useful lemma: Bonnie is a student or inactive
theorem bonnie_student_or_inactive : Student Bonnie ∨ Inactive Bonnie := by
  have hC : Club Bonnie := And.left P6
  have hPI : PerformOften Bonnie ∨ Inactive Bonnie := P2 Bonnie hC
  cases hPI with
  | inl hP =>
      have hE : Engaged Bonnie := P1 Bonnie (And.intro hC hP)
      have hdisj : (Engaged Bonnie ∧ Student Bonnie) ∨ (¬Engaged Bonnie ∧ ¬Student Bonnie) := And.right P6
      cases hdisj with
      | inl hES =>
          exact Or.inl hES.right
      | inr hNot =>
          have hFalse : False := hNot.left hE
          exact Or.inl (False.elim hFalse)
  | inr hI =>
      exact Or.inr hI

-- The given statement (implication) holds
theorem statement_true :
  ( (((Young Bonnie ∨ Teen Bonnie) ∧ Wish Bonnie) ∧ Chaperone Bonnie) ∨ ¬((Young Bonnie ∨ Teen Bonnie) ∧ Wish Bonnie) )
  → (Student Bonnie ∨ Inactive Bonnie) := by
  intro _
  exact bonnie_student_or_inactive
