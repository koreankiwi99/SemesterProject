axiom obj : Type

axiom Picuris : obj
axiom Juan : obj

axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Visited : obj → obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Premises
axiom PM_is_mr : MountainRange Picuris
axiom PM_in_NM_or_TX : InNewMexico Picuris ∨ InTexas Picuris
axiom Juan_visited_PM : Visited Juan Picuris
axiom exists_donated_mine_in_PM : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m
axiom no_TX_mountain_ranges_with_donated_mines :
  ∀ r : obj, (MountainRange r ∧ InTexas r) → ¬ (∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m)

-- From the donated mine in Picuris and the Texas prohibition, Picuris is not in Texas
theorem PM_not_in_TX : ¬InTexas Picuris := by
  intro htx
  have hpair : MountainRange Picuris ∧ InTexas Picuris := And.intro PM_is_mr htx
  have hno := no_TX_mountain_ranges_with_donated_mines Picuris hpair
  exact hno exists_donated_mine_in_PM

-- Using the disjunction and not in Texas, Picuris is in New Mexico
theorem PM_in_NM : InNewMexico Picuris := by
  have h := PM_in_NM_or_TX
  have hnot := PM_not_in_TX
  cases h with
  | inl hnm => exact hnm
  | inr htx => exact False.elim (hnot htx)

-- Therefore, Juan de Onate visited a mountain range in New Mexico
theorem Juan_visited_mr_in_NM : ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ Visited Juan r := by
  exists Picuris
  apply And.intro
  · exact PM_is_mr
  · apply And.intro
    · exact PM_in_NM
    · exact Juan_visited_PM
