<lean>
axiom obj : Type
axiom Picuris : obj
axiom HardingMine : obj
axiom Juan : obj

axiom MountainRange : obj → Prop
axiom InNM : obj → Prop
axiom InTX : obj → Prop
axiom Visited : obj → obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Premises
axiom P1 : MountainRange Picuris
axiom P2 : InNM Picuris ∨ InTX Picuris
axiom P3 : Visited Juan Picuris
axiom P4 : Mine HardingMine
axiom P5 : LocatedIn HardingMine Picuris
axiom P6 : Donated HardingMine
axiom P7 : ∀ r : obj, MountainRange r ∧ InTX r → ¬∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m

-- Picuris cannot be in Texas, otherwise we'd have a donated mine in a Texas mountain range.
theorem Picuris_not_in_TX : ¬InTX Picuris := by
  intro htx
  have hpair : MountainRange Picuris ∧ InTX Picuris := And.intro P1 htx
  have hno : ¬∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m := P7 Picuris hpair
  have hex : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m := by
    apply Exists.intro HardingMine
    apply And.intro P4
    apply And.intro P5
    exact P6
  exact hno hex

-- Negation of: "Juan de Onate visited a mountain range in Texas" (here, the visited range is Picuris).
theorem not_juan_visited_a_texas_mountain_range : ¬(Visited Juan Picuris ∧ InTX Picuris) := by
  intro h
  have htx : InTX Picuris := And.right h
  exact Picuris_not_in_TX htx
</lean>
