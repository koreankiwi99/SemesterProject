<lean>
axiom obj : Type
axiom Picuris : obj
axiom Harding : obj
axiom Juan : obj

axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Mine : obj → Prop
axiom Visited : obj → obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Premises
axiom P1 : MountainRange Picuris
axiom P2 : InNewMexico Picuris ∨ InTexas Picuris
axiom P3 : Visited Juan Picuris
axiom P4 : Mine Harding
axiom P5 : LocatedIn Harding Picuris
axiom P6 : Donated Harding
axiom P7 : ∀ mr : obj, MountainRange mr → InTexas mr → ¬∃ m : obj, Mine m ∧ LocatedIn m mr ∧ Donated m

-- Negation of the statement: the Harding Pegmatite Mine is located in a mountain range in New Mexico.
theorem harding_located_in_nm_range : ∃ mr : obj, MountainRange mr ∧ InNewMexico mr ∧ LocatedIn Harding mr := by
  have exDonated : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m := by
    exact Exists.intro Harding (And.intro P4 (And.intro P5 P6))
  have notTexPicuris : ¬ InTexas Picuris := by
    intro hTex
    have hNo := P7 Picuris P1 hTex
    exact hNo exDonated
  have hNM : InNewMexico Picuris := Or.resolve_right P2 notTexPicuris
  exact Exists.intro Picuris (And.intro P1 (And.intro hNM P5))
</lean>

ANSWER: False
