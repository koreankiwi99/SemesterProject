axiom P2_taller_class : ∀ x : obj, InClass x → Taller Michael x

-- Taller is transitive.
axiom P3_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man taller than Michael.
axiom P4_man_Peter : Man Peter
axiom P4_taller : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom P5_M_blocks_if_not_jump : ∀ p : obj, ¬Jumps p → CanBlock Michael p

-- Michael cannot block Windy's shooting.
axiom P6_not_block_Windy : ¬CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom P7_jumpers_great : ∀ x : obj, Shooter x → Jumps x → Great x

-- Windy is a shooter (since we refer to Windy's shooting).
axiom P8_Windy_shooter : Shooter Windy

-- From P5 and P6 we can conclude Windy jumps.
theorem windy_jumps : Jumps Windy := by
  have h : ¬Jumps Windy → CanBlock Michael Windy := P5_M_blocks_if_not_jump Windy
  by_contra hj
  have hb : CanBlock Michael Windy := h hj
  exact P6_not_block_Windy hb

-- Therefore, Windy is a great shooter.
theorem windy_is_great : Great Windy := by
  have hj : Jumps Windy := windy_jumps
  exact P7_jumpers_great Windy P8_Windy_shooter hj
