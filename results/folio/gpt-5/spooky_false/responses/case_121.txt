axiom obj : Type

-- Entities and predicates
axiom JohnCar : obj
axiom Tesla3 : obj → Prop
axiom TransportMulti : obj → Prop
axiom OneSeater : obj → Prop
axiom SingleSeatEV : obj → Prop
axiom SoloEV : obj → Prop

-- Premises from the textual context
axiom P1 : ∀ x : obj, TransportMulti x → ¬ OneSeater x
axiom P2 : ∀ x : obj, Tesla3 x → TransportMulti x
axiom P3 : ∀ x : obj, SingleSeatEV x → OneSeater x
axiom P4 : ∀ x : obj, SoloEV x → SingleSeatEV x
axiom P5 : ¬ (Tesla3 JohnCar) → (¬ (Tesla3 JohnCar) ∨ SingleSeatEV JohnCar)

-- Useful derived lemma: Tesla3 implies not OneSeater
theorem Tesla3_not_OneSeater : ∀ x : obj, Tesla3 x → ¬ OneSeater x := by
  intro x
  intro hxT
  exact P1 x (P2 x hxT)

-- If John's car were a single-seat EV, then it couldn't be a Tesla3
theorem SingleSeatEV_John_implies_not_Tesla3 : SingleSeatEV JohnCar → ¬ Tesla3 JohnCar := by
  intro hS
  intro hT
  have hOne : OneSeater JohnCar := P3 JohnCar hS
  have hNotOne : ¬ OneSeater JohnCar := Tesla3_not_OneSeater JohnCar hT
  exact hNotOne hOne

-- Similarly, if John's car were a SoloEV, then it couldn't be a Tesla3
theorem SoloEV_John_implies_not_Tesla3 : SoloEV JohnCar → ¬ Tesla3 JohnCar := by
  intro hSolo
  have hS : SingleSeatEV JohnCar := P4 JohnCar hSolo
  exact SingleSeatEV_John_implies_not_Tesla3 hS
