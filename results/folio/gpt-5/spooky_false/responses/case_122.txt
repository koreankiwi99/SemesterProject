axiom obj : Type
axiom JohnsCar : obj

axiom CanTransportMultiplePassengers : obj → Prop
axiom OneSeater : obj → Prop
axiom TeslaModel3 : obj → Prop
axiom SingleSeatElectricCar : obj → Prop
axiom SoloEV : obj → Prop

-- If something can transport multiple passengers, then they are not one-seaters.
axiom A1 : ∀ x : obj, CanTransportMultiplePassengers x → ¬ OneSeater x
-- All Tesla Model 3 can transport multiple passengers.
axiom A2 : ∀ x : obj, TeslaModel3 x → CanTransportMultiplePassengers x
-- All single-seat electric cars are one-seaters.
axiom A3 : ∀ x : obj, SingleSeatElectricCar x → OneSeater x
-- All Solo EV cars are single-seat electric cars.
axiom A4 : ∀ x : obj, SoloEV x → SingleSeatElectricCar x
-- If John's car is not a Tesla Model 3, then John's car is not a Tesla Model 3 or a single-seat electric car.
axiom A5 : ¬ TeslaModel3 JohnsCar → (¬ TeslaModel3 JohnsCar ∨ SingleSeatElectricCar JohnsCar)

-- A derivable consequence: If John's car is a Solo EV, then it is not a Tesla Model 3.
theorem not_tesla_of_solo : SoloEV JohnsCar → ¬ TeslaModel3 JohnsCar := by
  intro hSolo
  intro hTesla
  have hSSE : SingleSeatElectricCar JohnsCar := A4 JohnsCar hSolo
  have hOne : OneSeater JohnsCar := A3 JohnsCar hSSE
  have hMulti : CanTransportMultiplePassengers JohnsCar := A2 JohnsCar hTesla
  have hNotOne : ¬ OneSeater JohnsCar := A1 JohnsCar hMulti
  exact hNotOne hOne
