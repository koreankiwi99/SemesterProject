axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

-- Constant
axiom Rock : obj

-- Premises
axiom P1 : ∀ x : obj, Pet x ∧ Mammal x → Animal x
axiom P2 : ∀ x : obj, Monkey x → Mammal x
axiom P3 : ∀ x : obj, Pet x → (Monkey x ∨ Bird x)
axiom P4 : ∀ x : obj, Pet x ∧ Bird x → CanFly x
axiom P5 : ∀ x : obj, Animal x ∧ Pet x → CanBreathe x
axiom P6 : ∀ x : obj, Pet x ∧ CanFly x → HasWings x
axiom P7 : Pet Rock
axiom P8 : CanFly Rock ∨ Bird Rock ∨ ¬CanBreathe Rock

-- Useful derived fact: if Rock cannot breathe, then Rock is not a monkey
theorem not_monkey_of_not_breathe : ¬CanBreathe Rock → ¬Monkey Rock :=
by
  intro hnb
  intro hmonk
  have hmamm : Mammal Rock := P2 Rock hmonk
  have hanimal : Animal Rock := by
    have hpm : Pet Rock ∧ Mammal Rock := And.intro P7 hmamm
    exact P1 Rock hpm
  have hbreath : CanBreathe Rock := by
    have hap : Animal Rock ∧ Pet Rock := And.intro hanimal P7
    exact P5 Rock hap
  exact hnb hbreath
