axiom obj : Type
axiom John : obj

axiom MakesPlans : obj → Prop
axiom Engaged : obj → Prop
axiom InvitesOthers : obj → Prop
axiom WellAttended : obj → Prop
axiom LargeFamily : obj → Prop
axiom InvitesFriends : obj → Prop

-- All people who make wedding plans are engaged.
axiom A1 : ∀ x : obj, MakesPlans x → Engaged x
-- All people who invite others to come to their ceremony make wedding plans.
axiom A2 : ∀ x : obj, InvitesOthers x → MakesPlans x
-- Anyone who has a well-attended wedding invites others to come to their ceremony.
axiom A3 : ∀ x : obj, WellAttended x → InvitesOthers x
-- Some people that have well-attended weddings have larger families.
axiom A4 : ∃ x : obj, WellAttended x ∧ LargeFamily x
-- If John is engaged, then not (John invites friends to their ceremony and makes wedding plans).
axiom A5 : Engaged John → ¬ (InvitesFriends John ∧ MakesPlans John)
-- If John has a larger family, then either John has a well-attended wedding or invites friends to their ceremony.
axiom A6 : LargeFamily John → (WellAttended John ∨ InvitesFriends John)

-- Some derived consequences (do not yield LargeFamily John).
theorem WellAttended_implies_MakesPlans (x : obj) : WellAttended x → MakesPlans x := by
  intro hw
  exact A2 x (A3 x hw)

theorem WellAttended_implies_Engaged (x : obj) : WellAttended x → Engaged x := by
  intro hw
  exact A1 x (WellAttended_implies_MakesPlans x hw)

theorem John_WellAttended_implies_not_InviteFriends : WellAttended John → ¬ InvitesFriends John := by
  intro hw
  have mp : MakesPlans John := WellAttended_implies_MakesPlans John hw
  have eng : Engaged John := A1 John mp
  have h := A5 eng
  intro ifr
  apply h
  exact And.intro ifr mp
