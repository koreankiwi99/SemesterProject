<lean>
axiom obj : Type
axiom Sucks : obj → Prop
axiom Plunger : obj → Prop
axiom Vacuum : obj → Prop
axiom Vampire : obj → Prop
axiom HouseholdAppliance : obj → Prop

axiom Space : obj
axiom Duster : obj

-- Premises
axiom P1 : ∀ x : obj, Plunger x → Sucks x          -- Plungers suck.
axiom P2 : ∀ x : obj, Vacuum x → Sucks x           -- Vacuums suck.
axiom P3 : ∀ x : obj, Vampire x → Sucks x          -- Vampires suck.
axiom P4 : Vacuum Space                             -- Space is a vacuum.
axiom P5 : HouseholdAppliance Duster                -- A duster is a household appliance ...
axiom P6 : ¬ Sucks Duster                           -- ... that doesn't suck.

-- Negation of the statement "If something is a household appliance, it sucks."
theorem not_all_household_appliances_suck : ¬ (∀ x : obj, HouseholdAppliance x → Sucks x) := by
  intro h
  have hD : Sucks Duster := h Duster P5
  exact P6 hD
</lean>

ANSWER: False
