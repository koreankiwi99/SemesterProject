axiom obj : Type

-- Predicates
axiom Film : obj → Prop
axiom AppropriateAllAges : obj → Prop
axiom CanWatchWithoutGuidance : obj → Prop
axiom ContainsExcessiveEroticViolent : obj → Prop
axiom RatedGA : obj → Prop
axiom FamilyFriendlyAnimated : obj → Prop
axiom InFrozenSeries : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, AppropriateAllAges x → CanWatchWithoutGuidance x
axiom A2 : ∀ x : obj, ContainsExcessiveEroticViolent x → ¬ CanWatchWithoutGuidance x
axiom A3 : ∀ x : obj, RatedGA x → AppropriateAllAges x
axiom A4 : ∀ x : obj, FamilyFriendlyAnimated x → RatedGA x
axiom A5 : ∀ x : obj, InFrozenSeries x → FamilyFriendlyAnimated x

-- Entity
axiom Hachi : obj
axiom Hachi_is_film : Film Hachi
axiom Hachi_ff_or_app : FamilyFriendlyAnimated Hachi ∨ AppropriateAllAges Hachi

-- Derivable facts
theorem hachi_can_watch : CanWatchWithoutGuidance Hachi :=
by
  cases Hachi_ff_or_app with
  | inl hff =>
      have hGA : RatedGA Hachi := A4 Hachi hff
      have happ : AppropriateAllAges Hachi := A3 Hachi hGA
      exact A1 Hachi happ
  | inr happ =>
      exact A1 Hachi happ

theorem hachi_not_contains_excessive : ¬ ContainsExcessiveEroticViolent Hachi :=
by
  intro hce
  have hno : ¬ CanWatchWithoutGuidance Hachi := A2 Hachi hce
  have hw : CanWatchWithoutGuidance Hachi := hachi_can_watch
  exact hno hw
