axiom Vladimir : obj
axiom Ekaterina : obj

axiom USCitizen : obj → Prop
axiom TaiwanCitizen : obj → Prop
axiom RussianOfficial : obj → Prop
axiom RegisterUS : obj → Prop
axiom Participate2024 : obj → Prop
axiom ManagerGazprom : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, RegisterUS x → Participate2024 x
axiom P2 : ∀ x : obj, USCitizen x → RegisterUS x
axiom P3 : ∀ x : obj, USCitizen x ∨ TaiwanCitizen x
axiom P4 : ∀ x : obj, RussianOfficial x → ¬TaiwanCitizen x
axiom P5a : ¬TaiwanCitizen Vladimir
axiom P5b : ¬ManagerGazprom Vladimir
axiom P6 : RegisterUS Ekaterina ∨ RussianOfficial Ekaterina

-- Some derivable consequences (do not yield RussianOfficial Vladimir)
theorem vladimir_US : USCitizen Vladimir := by
  have h := P3 Vladimir
  cases h with
  | inl hu =>
      exact hu
  | inr ht =>
      have f : False := P5a ht
      exact False.elim f

theorem vladimir_register : RegisterUS Vladimir := by
  apply P2
  exact vladimir_US

theorem vladimir_participate : Participate2024 Vladimir := by
  apply P1
  exact vladimir_register

theorem ekaterina_register : RegisterUS Ekaterina := by
  cases P6 with
  | inl hr =>
      exact hr
  | inr ho =>
      have notTw : ¬TaiwanCitizen Ekaterina := P4 Ekaterina ho
      have disj := P3 Ekaterina
      cases disj with
      | inl hu =>
          exact P2 Ekaterina hu
      | inr ht =>
          have f : False := notTw ht
          exact False.elim f
