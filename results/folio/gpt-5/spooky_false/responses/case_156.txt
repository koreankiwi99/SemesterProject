<lean>
axiom obj : Type

axiom James : obj

axiom WorksInLibrary : obj → Prop
axiom PartTimeJob : obj → Prop
axiom FromCS : obj → Prop
axiom TakesDatabase : obj → Prop
axiom WithProfessorDavid : obj → Prop
axiom WorksInLab : obj → Prop
axiom Student : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, WorksInLibrary x → ¬ FromCS x
axiom P2 : ∀ x : obj, PartTimeJob x → WorksInLibrary x
axiom P3 : ∀ x : obj, TakesDatabase x → FromCS x
axiom P4 : ∀ x : obj, WithProfessorDavid x → TakesDatabase x
axiom P5 : ∀ x : obj, WorksInLab x → WithProfessorDavid x
axiom P6 : WorksInLab James

theorem james_not_part_time : ¬ PartTimeJob James :=
by
  intro hpt
  have hlib : WorksInLibrary James := P2 James hpt
  have hnotcs : ¬ FromCS James := P1 James hlib
  have hdavid : WithProfessorDavid James := P5 James P6
  have hdb : TakesDatabase James := P4 James hdavid
  have hcs : FromCS James := P3 James hdb
  exact hnotcs hcs
</lean>

ANSWER: False
