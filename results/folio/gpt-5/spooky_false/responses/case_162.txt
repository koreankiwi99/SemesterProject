<lean>
axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

axiom Hawk_never_swims : ∀ x : obj, Hawk x → ¬ Swims x
axiom Some_birds_are_hawks : ∃ x : obj, Bird x ∧ Hawk x

theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swims x) := by
  intro hAll
  cases Some_birds_are_hawks with
  | intro x hx =>
    cases hx with
    | intro hxBird hxHawk =>
      have hxSwims : Swims x := hAll x hxBird
      have hxNotSwims : ¬ Swims x := Hawk_never_swims x hxHawk
      exact hxNotSwims hxSwims
</lean>

ANSWER: False
