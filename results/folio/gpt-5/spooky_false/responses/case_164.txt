axiom obj : Type

-- Predicates
axiom Show : obj → Prop
axiom AirsMon8 : obj → Prop
axiom GivesRosesOnTV : obj → Prop
axiom Bachelor : obj → Prop        -- "x is an episode of The Bachelor"
axiom PortraysReal : obj → Prop
axiom RealityTV : obj → Prop

-- Constants
axiom BreakingBad : obj

-- Premises from the text
axiom E1 : ∃ x : obj, Show x ∧ AirsMon8 x ∧ GivesRosesOnTV x
axiom R1 : ∀ x : obj, GivesRosesOnTV x → Bachelor x
axiom R2 : ∀ x : obj, Bachelor x → PortraysReal x                    -- "The Bachelor portrays the lives of real people"
axiom R3 : ∀ x : obj, PortraysReal x → RealityTV x                   -- All shows portraying lives of real people are reality TV
axiom T1 : Show BreakingBad
axiom T2 : ¬RealityTV BreakingBad

-- Negation of the statement to evaluate:
-- "Breaking Bad is a TV show in which roses are given out on Monday at 8 pm."
-- Formalized as: Show BreakingBad ∧ AirsMon8 BreakingBad ∧ GivesRosesOnTV BreakingBad
theorem not_stmt : ¬(Show BreakingBad ∧ AirsMon8 BreakingBad ∧ GivesRosesOnTV BreakingBad) := by
  intro h
  have hRoses : GivesRosesOnTV BreakingBad := h.right.right
  have hBachelor : Bachelor BreakingBad := R1 BreakingBad hRoses
  have hReal : PortraysReal BreakingBad := R2 BreakingBad hBachelor
  have hReality : RealityTV BreakingBad := R3 BreakingBad hReal
  exact T2 hReality
</lean>
