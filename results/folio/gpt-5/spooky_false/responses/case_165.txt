axiom AirsMon8pm : Show → Prop
axiom Bachelor : Show → Prop

-- Some show airing at 8 pm on Monday gives out roses on TV.

-- If a show gives out roses on TV, then the show is an episode of The Bachelor.
axiom roses_implies_bachelor : ∀ s : Show, GivesRoses s → Bachelor s

-- The Bachelor portrays the lives of real people (as a schema over its episodes).
axiom bachelor_portrays_real : ∀ s : Show, Bachelor s → PortraysReal s

-- All shows portraying the lives of real people are reality TV shows.
axiom real_implies_realityTV : ∀ s : Show, PortraysReal s → RealityTV s

-- Breaking Bad is not a reality TV show.
axiom BB_not_reality : ¬RealityTV BreakingBad

  have hr : PortraysReal BreakingBad := bachelor_portrays_real BreakingBad hb
  have hrt : RealityTV BreakingBad := real_implies_realityTV BreakingBad hr
  exact BB_not_reality hrt
