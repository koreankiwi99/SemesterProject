axiom InPrussianStateParliament : obj → Prop
axiom InNaziReichstag : obj → Prop

axiom HS_in_Prussian : InPrussianStateParliament HeinrichSchmidt
axiom HS_in_Reichstag : InNaziReichstag HeinrichSchmidt

  ∃ x : obj, GermanPolitician x ∧ InPrussianStateParliament x ∧ InNaziReichstag x := by
  refine Exists.intro HeinrichSchmidt ?_
  refine And.intro HS_is_GermanPolitician ?_
  refine And.intro HS_in_Prussian HS_in_Reichstag

-- Therefore, the negation of the statement implies False.
theorem negation_implies_False :
  (¬ ∃ x : obj, GermanPolitician x ∧ InPrussianStateParliament x ∧ InNaziReichstag x) → False := by
  intro hneg
  exact hneg exists_politician_in_both
