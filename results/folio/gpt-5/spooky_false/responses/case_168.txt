<lean>
axiom obj : Type
axiom HeinrichSchmidt : obj

axiom Politician : obj → Prop
axiom German : obj → Prop
axiom MemberPrussianStateParliament : obj → Prop
axiom MemberNaziReichstag : obj → Prop

-- Contextual facts
axiom HS_politician : Politician HeinrichSchmidt
axiom HS_german : German HeinrichSchmidt
axiom HS_prussian : MemberPrussianStateParliament HeinrichSchmidt
axiom HS_nazi : MemberNaziReichstag HeinrichSchmidt

-- Statement to evaluate: No politicians are part of the Nazi Reichstag
axiom NoPoliticiansInNaziReichstag : ∀ x : obj, Politician x → ¬ MemberNaziReichstag x

theorem contradiction : False := by
  have h := NoPoliticiansInNaziReichstag HeinrichSchmidt HS_politician
  exact h HS_nazi
