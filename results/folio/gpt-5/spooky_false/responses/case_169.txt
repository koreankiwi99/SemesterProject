-- Derived: Severe implies LowSurvival
theorem severe_implies_low : ∀ x : obj, Severe x → LowSurvival x := by
  intro x
  intro hx
  have hdead : Deadly x := P2 x hx
  exact P1 x hdead

  have hL : LowSurvival Colorectal := severe_implies_low Colorectal hS
  exact P6 (And.intro hB hL)
