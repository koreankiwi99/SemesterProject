<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop
axiom NearCampus : obj → Prop
axiom Rockie : obj

axiom A1 : ∀ x : obj, Rabbit x ∧ NearCampus x → Cute x
axiom A2 : ∃ x : obj, Turtle x ∧ NearCampus x
axiom A3 : ∀ x : obj, NearCampus x → Rabbit x ∧ Squirrel x
axiom A4 : ∀ x : obj, Skittish x → ¬Calm x
axiom A5 : ∀ x : obj, Squirrel x ∧ NearCampus x → Skittish x
axiom A6 : NearCampus Rockie ∧ Calm Rockie

theorem rockie_is_turtle : Turtle Rockie := by
  have hN : NearCampus Rockie := And.left A6
  have hC : Calm Rockie := And.right A6
  have hRS : Rabbit Rockie ∧ Squirrel Rockie := A3 Rockie hN
  have hS : Squirrel Rockie := And.right hRS
  have hSk : Skittish Rockie := A5 Rockie (And.intro hS hN)
  have hNotC : ¬Calm Rockie := A4 Rockie hSk
  have hFalse : False := hNotC hC
  exact False.elim hFalse
</lean>
