<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatSaladReg : obj → Prop
axiom VeryConscious : obj → Prop
axiom GrewHealthHome : obj → Prop
axiom FulfillDailyIntake : obj → Prop
axiom DisregardWellbeing : obj → Prop
axiom VisitGymDaily : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, EatSaladReg x → VeryConscious x
axiom A2 : ∀ x : obj, GrewHealthHome x → EatSaladReg x
axiom A3 : ∀ x : obj, FulfillDailyIntake x → GrewHealthHome x
axiom A4 : ∀ x : obj, DisregardWellbeing x → ¬VeryConscious x
axiom A5 : ∀ x : obj, VisitGymDaily x → FulfillDailyIntake x

-- Taylor either (grew up health-conscious and disregards well-being) or neither
axiom Taylor_either :
  (GrewHealthHome Taylor ∧ DisregardWellbeing Taylor) ∨
  (¬GrewHealthHome Taylor ∧ ¬DisregardWellbeing Taylor)

theorem taylor_not_visit : ¬VisitGymDaily Taylor := by
  intro hGym
  have hFul : FulfillDailyIntake Taylor := A5 Taylor hGym
  have hGrew : GrewHealthHome Taylor := A3 Taylor hFul
  have hEat : EatSaladReg Taylor := A2 Taylor hGrew
  have hVC : VeryConscious Taylor := A1 Taylor hEat
  cases Taylor_either with
  | inl h =>
      have hDis : DisregardWellbeing Taylor := h.right
      have hNotVC : ¬VeryConscious Taylor := A4 Taylor hDis
      exact hNotVC hVC
  | inr h =>
      have hNotGrew : ¬GrewHealthHome Taylor := h.left
      exact hNotGrew hGrew
</lean>

ANSWER: False
