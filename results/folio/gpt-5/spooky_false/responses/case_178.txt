<lean>
axiom obj : Type
axiom Human : obj → Prop
axiom Horse : obj → Prop
axiom Hooves : obj → Prop

axiom All_horses_have_hooves : ∀ x : obj, Horse x → Hooves x
axiom No_humans_have_hooves : ∀ x : obj, Human x → ¬Hooves x

-- Statement: Some humans are horses: ∃ x, Human x ∧ Horse x
-- We prove its negation.
theorem no_human_is_horse : ¬ (∃ x : obj, Human x ∧ Horse x) := by
  intro h
  rcases h with ⟨x, hx⟩
  rcases hx with ⟨hHuman, hHorse⟩
  have hHooves : Hooves x := All_horses_have_hooves x hHorse
  have hNoHooves : ¬ Hooves x := No_humans_have_hooves x hHuman
  exact hNoHooves hHooves
</lean>

ANSWER: False
