<lean>
axiom Person : Type
axiom Airport : Type

axiom Susan : Person
axiom John : Person
axiom LGA : Airport

axiom Dep : Person → Airport
axiom Arr : Person → Airport

-- Premises
axiom Susan_to_LGA : Arr Susan = LGA
axiom John_from_LGA : Dep John = LGA
axiom dep_ne_arr : ∀ p : Person, Dep p ≠ Arr p

-- Negation of the statement "Susan flies from LGA airport."
-- i.e., prove ¬ (Dep Susan = LGA)
theorem not_Susan_from_LGA : ¬ (Dep Susan = LGA) := by
  intro h
  have hneq := dep_ne_arr Susan
  have heq : Dep Susan = Arr Susan := Eq.trans h Susan_to_LGA.symm
  exact hneq heq
</lean>
