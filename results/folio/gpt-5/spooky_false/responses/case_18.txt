<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop
axiom Near : obj → Prop
axiom Cute : obj → Prop

-- 1. All rabbits that can be spotted near the campus are cute.
axiom A1 : ∀ x : obj, (Rabbit x ∧ Near x) → Cute x
-- 2. Some turtles can be spotted near the campus.
axiom A2 : ∃ x : obj, Turtle x ∧ Near x
-- 3. The only animals that can be spotted near the campus are rabbits and squirrels.
axiom A3 : ∀ x : obj, Near x → (Rabbit x ∨ Squirrel x)
-- 4. If something is skittish, then it is not calm.
axiom A4 : ∀ x : obj, Skittish x → ¬Calm x
-- 5. All the squirrels that can be spotted near the campus are skittish.
axiom A5 : ∀ x : obj, (Squirrel x ∧ Near x) → Skittish x
-- 6. Rockie can be spotted near the campus, and it is calm.
axiom Rockie : obj
axiom A6 : Near Rockie ∧ Calm Rockie

theorem not_squirrel_Rockie : ¬ Squirrel Rockie := by
  intro hSq
  have hNear : Near Rockie := A6.left
  have hCalm : Calm Rockie := A6.right
  have hSk : Skittish Rockie := A5 Rockie (And.intro hSq hNear)
  have hNotCalm : ¬ Calm Rockie := A4 Rockie hSk
  exact hNotCalm hCalm

theorem rockie_cute_or_turtle : Turtle Rockie ∨ Cute Rockie := by
  have hNear : Near Rockie := A6.left
  have hRS : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie hNear
  have hRabbit : Rabbit Rockie := by
    cases hRS with
    | inl hr => exact hr
    | inr hs =>
      have hNS : ¬ Squirrel Rockie := not_squirrel_Rockie
      have : False := hNS hs
      exact False.elim this
  have hCute : Cute Rockie := A1 Rockie (And.intro hRabbit hNear)
  exact Or.inr hCute
