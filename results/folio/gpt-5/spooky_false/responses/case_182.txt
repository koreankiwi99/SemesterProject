-- Tom is a pet owner.
axiom Tom_owner : Owner Tom

theorem tom_not_mean : ¬ Mean Tom := by
  have hNice : Nice Tom := by
    have hL : Lover Tom := owners_love Tom Tom_owner
    exact lovers_nice Tom hL
  exact no_nice_mean Tom hNice

-- Statement to evaluate: Tom is not both a grumpy person and mean to animals.
theorem tom_not_both : ¬ (Grumpy Tom ∧ Mean Tom) := by
  intro h
  exact tom_not_mean h.right
