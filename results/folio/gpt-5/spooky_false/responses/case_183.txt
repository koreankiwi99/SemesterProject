<lean>
axiom obj : Type
axiom NiceToAnimals : obj → Prop
axiom MeanToAnimals : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop
axiom Tom : obj

-- Premises
axiom P1 : ∀ x : obj, NiceToAnimals x → ¬ MeanToAnimals x
axiom P2 : ∃ x : obj, Grumpy x ∧ MeanToAnimals x
axiom P3 : ∀ x : obj, AnimalLover x → NiceToAnimals x
axiom P4 : ∀ x : obj, PetOwner x → AnimalLover x
axiom P5 : PetOwner Tom

-- Negation of the given statement: show the implication is false
theorem statement_is_false :
  ¬ ( (¬ (Grumpy Tom ∧ MeanToAnimals Tom)) → (¬ Grumpy Tom ∧ ¬ AnimalLover Tom) ) := by
  intro h
  have hLover : AnimalLover Tom := P4 Tom P5
  have hNice : NiceToAnimals Tom := P3 Tom hLover
  have hNotMean : ¬ MeanToAnimals Tom := P1 Tom hNice
  have hP : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) := by
    intro hgm
    have hMean : MeanToAnimals Tom := And.right hgm
    exact hNotMean hMean
  have hConseq : ¬ Grumpy Tom ∧ ¬ AnimalLover Tom := h hP
  have hNotLover : ¬ AnimalLover Tom := And.right hConseq
  exact hNotLover hLover
</lean>

ANSWER: False
