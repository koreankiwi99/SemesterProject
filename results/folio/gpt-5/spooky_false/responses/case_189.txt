<lean>
axiom obj : Type
axiom Yuri : obj

axiom ProBasketballPlayer : obj → Prop
axiom American : obj → Prop
axiom CanDoJumpShots : obj → Prop
axiom LeapsStraightIntoTheAir : obj → Prop
axiom ActivatesLegMuscles : obj → Prop

axiom some_pro_not_american : ∃ x : obj, ProBasketballPlayer x ∧ ¬American x
axiom all_pro_can_jump : ∀ x : obj, ProBasketballPlayer x → CanDoJumpShots x
axiom jump_implies_leap : ∀ x : obj, CanDoJumpShots x → LeapsStraightIntoTheAir x
axiom leap_implies_activate : ∀ x : obj, LeapsStraightIntoTheAir x → ActivatesLegMuscles x
axiom yuri_not_activate : ¬ActivatesLegMuscles Yuri

theorem not_yuri_american_pro : ¬(American Yuri ∧ ProBasketballPlayer Yuri) :=
by
  intro h
  have hpro : ProBasketballPlayer Yuri := And.right h
  have hjump : CanDoJumpShots Yuri := all_pro_can_jump Yuri hpro
  have hleap : LeapsStraightIntoTheAir Yuri := jump_implies_leap Yuri hjump
  have hact : ActivatesLegMuscles Yuri := leap_implies_activate Yuri hleap
  exact yuri_not_activate hact
</lean>

ANSWER: False
