lemma rockie_not_squirrel : ¬ Squirrel Rockie :=

  have hSk : Skittish Rockie := P5 Rockie (And.intro hSq hNear)

  exact hNotCalm (And.right P6)

  exact P1 Rockie (And.intro rockie_rabbit hNear)

theorem statement_true : (¬ (Turtle Rockie ∧ Squirrel Rockie)) → (Cute Rockie ∨ Skittish Rockie) :=
