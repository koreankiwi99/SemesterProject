<lean>
axiom obj : Type
axiom Yuri : obj
axiom American : obj → Prop
axiom Pro : obj → Prop -- Professional basketball player
axiom CanJumpShot : obj → Prop
axiom LeapsStraight : obj → Prop
axiom ActivatesLegMuscles : obj → Prop

axiom some_non_american_pro : ∃ x : obj, Pro x ∧ ¬American x
axiom pros_can_jump : ∀ x : obj, Pro x → CanJumpShot x
axiom jumpers_leap_straight : ∀ x : obj, CanJumpShot x → LeapsStraight x
axiom leapers_activate_legs : ∀ x : obj, LeapsStraight x → ActivatesLegMuscles x
axiom yuri_not_activate : ¬ActivatesLegMuscles Yuri

theorem negate_statement : ¬( (¬(American Yuri ∧ Pro Yuri)) → Pro Yuri ) := by
  -- From the chain of implications, any professional activates leg muscles
  have pro_activates : ∀ x : obj, Pro x → ActivatesLegMuscles x := by
    intro x
    intro hx
    have hj : CanJumpShot x := pros_can_jump x hx
    have hl : LeapsStraight x := jumpers_leap_straight x hj
    exact leapers_activate_legs x hl
  -- Hence Yuri is not a professional
  have yuri_not_pro : ¬Pro Yuri := by
    intro hy
    have act : ActivatesLegMuscles Yuri := pro_activates Yuri hy
    exact yuri_not_activate act
  -- Now negate the statement
  intro h
  -- Antecedent is true because Yuri is not a professional
  have ant : ¬(American Yuri ∧ Pro Yuri) := by
    intro hp
    exact yuri_not_pro hp.right
  -- Apply the implication to get a contradiction
  have py : Pro Yuri := h ant
  exact yuri_not_pro py
</lean>
