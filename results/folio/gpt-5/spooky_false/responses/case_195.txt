-- Jane does not like any animals that bark.
axiom H1 : ∀ x : obj, (Animal x ∧ Barks x) → ¬LikesJane x
-- All dogs are animals that bark.
axiom H2 : ∀ x : obj, Dog x → (Animal x ∧ Barks x)
-- Jane likes all animals that jump.
axiom H3 : ∀ x : obj, (Animal x ∧ Jumps x) → LikesJane x
-- If an animal has legs, then it jumps.
axiom H4 : ∀ x : obj, (Animal x ∧ Legs x) → Jumps x
-- If an animal is terricolous, then it has legs.
axiom H5 : ∀ x : obj, (Animal x ∧ Terricolous x) → Legs x
-- KiKi is an animal.
axiom H6 : Animal KiKi
-- If KiKi neither jumps nor has legs, then KiKi is terricolous or has legs.
axiom H7 : (¬Jumps KiKi ∧ ¬Legs KiKi) → (Terricolous KiKi ∨ Legs KiKi)

-- A derived fact: If KiKi is terricolous, then KiKi does not bark.
theorem terricolous_implies_not_barks :
  Terricolous KiKi → ¬Barks KiKi := by
  intro t
  have l : Legs KiKi := by
    have h : Animal KiKi ∧ Terricolous KiKi := And.intro H6 t
    exact H5 KiKi h
  have j : Jumps KiKi := by
    have h : Animal KiKi ∧ Legs KiKi := And.intro H6 l
    exact H4 KiKi h
  have like : LikesJane KiKi := by
    have h : Animal KiKi ∧ Jumps KiKi := And.intro H6 j
    exact H3 KiKi h
  intro bk
  have not_like : ¬LikesJane KiKi := by
    have h : Animal KiKi ∧ Barks KiKi := And.intro H6 bk
    exact H1 KiKi h
  exact not_like like
