<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom Spotted : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop
axiom Rockie : obj

-- Premises
axiom R1 : ∀ x : obj, Rabbit x ∧ Spotted x → Cute x
axiom R2 : ∃ x : obj, Turtle x ∧ Spotted x
axiom R3 : ∀ x : obj, Spotted x → Rabbit x ∨ Squirrel x
axiom R4 : ∀ x : obj, Skittish x → ¬Calm x
axiom R5 : ∀ x : obj, Squirrel x ∧ Spotted x → Skittish x
axiom R6 : Spotted Rockie ∧ Calm Rockie

-- Statement to evaluate
def stmt : Prop := (Cute Rockie ∧ Calm Rockie) → (Skittish Rockie ∧ Turtle Rockie)

-- We prove the negation of the statement
theorem negate_stmt : ¬stmt := by
  intro h
  have hSp : Spotted Rockie := And.left R6
  have hCalm : Calm Rockie := And.right R6
  have hDisj : Rabbit Rockie ∨ Squirrel Rockie := R3 Rockie hSp
  have hRab : Rabbit Rockie := by
    cases hDisj with
    | inl hr => exact hr
    | inr hs =>
      have hSk : Skittish Rockie := R5 Rockie ⟨hs, hSp⟩
      have hNotCalm : ¬Calm Rockie := R4 Rockie hSk
      exact False.elim (hNotCalm hCalm)
  have hCute : Cute Rockie := R1 Rockie ⟨hRab, hSp⟩
  have hQ : Skittish Rockie ∧ Turtle Rockie := h ⟨hCute, hCalm⟩
  have hSk : Skittish Rockie := And.left hQ
  have hNotCalm : ¬Calm Rockie := R4 Rockie hSk
  exact hNotCalm hCalm
</lean>

ANSWER: False
