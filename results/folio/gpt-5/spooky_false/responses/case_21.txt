-- If a Netflix show is popular, Karen will binge-watch it.

  ∀ x : obj, (Netflix x ∧ Popular x) → BingeKaren x

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom binge_iff_download_on_netflix :
  ∀ x : obj, Netflix x → (BingeKaren x ↔ DownloadKaren x)

-- Karen does not download Black Mirror.
axiom not_download_BM : ¬DownloadKaren BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom share_if_binge_on_netflix :
  ∀ x : obj, Netflix x → BingeKaren x → ShareWithLisa x

-- From the premises, Karen will share "Stranger Things" with Lisa.

      And.intro ST_netflix ST_popular
    exact popular_netflix_implies_binge StrangerThings hNP
  exact share_if_binge_on_netflix StrangerThings ST_netflix hBingeST
