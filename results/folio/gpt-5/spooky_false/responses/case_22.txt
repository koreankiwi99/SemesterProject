axiom ST : obj

axiom Popular : obj → Prop
axiom Binge : obj → Prop        -- "Karen binge-watches it"
axiom Download : obj → Prop     -- "Karen downloads it"
axiom ShareWithLisa : obj → Prop

-- "Stranger Things" is a popular Netflix show.
axiom ST_netflix : NetflixShow ST
axiom ST_popular : Popular ST

-- If a Netflix show is popular, Karen will binge-watch it.
axiom binge_of_popular_netflix : ∀ x : obj, NetflixShow x → Popular x → Binge x

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom binge_iff_download_on_netflix : ∀ x : obj, NetflixShow x → (Binge x ↔ Download x)

-- Karen does not download "Black Mirror."
axiom not_download_BM : ¬ Download BM

-- "Black Mirror" is a Netflix show.
axiom BM_netflix : NetflixShow BM

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom share_if_binge_netflix : ∀ x : obj, NetflixShow x → Binge x → ShareWithLisa x

-- Target: ¬ Popular BM
theorem not_popular_BM : ¬ Popular BM := by
  intro hpop
  have hbinge : Binge BM := binge_of_popular_netflix BM BM_netflix hpop
  have hiff : Binge BM ↔ Download BM := binge_iff_download_on_netflix BM BM_netflix
  have hdl : Download BM := (Iff.mp hiff) hbinge
  exact not_download_BM hdl
