axiom Show : Type
axiom StrangerThings : Show
axiom BlackMirror : Show

axiom Netflix : Show → Prop
axiom Popular : Show → Prop
axiom Binge : Show → Prop
axiom Download : Show → Prop
axiom ShareWithLisa : Show → Prop

-- "Stranger Things" is a popular Netflix show.
axiom ST_popular : Popular StrangerThings
axiom ST_netflix : Netflix StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom rule_popular_binge : ∀ s : Show, Netflix s → Popular s → Binge s

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom rule_binge_iff_download : ∀ s : Show, Netflix s → (Binge s ↔ Download s)

-- Karen does not download "Black Mirror."
axiom BM_not_download : ¬ Download BlackMirror

-- "Black Mirror" is a Netflix show.
axiom BM_netflix : Netflix BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom rule_binge_share : ∀ s : Show, Netflix s → Binge s → ShareWithLisa s

-- Derivable facts
theorem ST_binge : Binge StrangerThings := by
  apply rule_popular_binge StrangerThings
  · exact ST_netflix
  · exact ST_popular

theorem ST_share : ShareWithLisa StrangerThings := by
  apply rule_binge_share StrangerThings
  · exact ST_netflix
  · exact ST_binge

-- From ¬Download and (Binge ↔ Download) on a Netflix show, we get ¬Binge.
theorem BM_not_binge : ¬ Binge BlackMirror := by
  intro hB
  have hiff : Binge BlackMirror ↔ Download BlackMirror :=
    rule_binge_iff_download BlackMirror BM_netflix
  have hBD : Binge BlackMirror → Download BlackMirror := Iff.mp hiff
  have hD : Download BlackMirror := hBD hB
  exact BM_not_download hD
