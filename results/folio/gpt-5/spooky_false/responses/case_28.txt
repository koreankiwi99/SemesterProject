axiom obj : Type

axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom HighlyIntelligent : obj → Prop
axiom FromEarth : obj → Prop
axiom FromMars : obj → Prop

axiom Marvin : obj

-- Premises
axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x
axiom A2 : ∀ x : obj, FromMars x → Alien x
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬Human x
axiom A4 : ∀ x : obj, HighlyIntelligent x ∧ FromEarth x → Human x
axiom A5 : HighlyIntelligent Marvin
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬FromEarth Marvin ∧ ¬FromMars Marvin)
axiom A7 : ¬FromEarth Marvin → Extraterrestrial Marvin

-- From the premises, E ∧ M leads to a contradiction (Human and ¬Human), so not both.
theorem not_both : ¬(FromEarth Marvin ∧ FromMars Marvin) := by
  intro h
  have he : FromEarth Marvin := And.left h
  have hm : FromMars Marvin := And.right h
  have ha : Alien Marvin := A2 Marvin hm
  have het : Extraterrestrial Marvin := A1 Marvin ha
  have nh : ¬Human Marvin := A3 Marvin het
  have hi : HighlyIntelligent Marvin := A5
  have hh : Human Marvin := A4 Marvin (And.intro hi he)
  exact nh hh

-- Using the exclusive-style disjunction, deduce neither Earth nor Mars.
theorem neither : ¬FromEarth Marvin ∧ ¬FromMars Marvin := by
  have d := A6
  cases d with
  | inl h =>
      apply False.elim
      exact not_both h
  | inr h =>
      exact h

-- From not Earth, deduce extraterrestrial, hence not human.
theorem marvin_not_human : ¬Human Marvin := by
  have nE : ¬FromEarth Marvin := And.left neither
  have et : Extraterrestrial Marvin := A7 nE
  exact A3 Marvin et

-- Therefore, Marvin is neither a human nor from Mars.
theorem statement_true : ¬Human Marvin ∧ ¬FromMars Marvin := by
  apply And.intro
  · exact marvin_not_human
  · exact And.right neither
