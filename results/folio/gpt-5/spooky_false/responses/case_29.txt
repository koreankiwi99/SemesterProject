<lean>
axiom obj : Type
axiom Marvin : obj

axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromMars : obj → Prop
axiom FromEarth : obj → Prop
axiom HighlyIntelligent : obj → Prop

axiom A1_all_aliens_extraterrestrials : ∀ x : obj, Alien x → Extraterrestrial x
axiom A2_fromMars_alien : ∀ x : obj, FromMars x → Alien x
axiom A3_no_extraterrestrial_human : ∀ x : obj, Extraterrestrial x → ¬Human x
axiom A4_hi_fromEarth_human : ∀ x : obj, HighlyIntelligent x ∧ FromEarth x → Human x
axiom A5_Marvin_hi : HighlyIntelligent Marvin
axiom A6_Marvin_either_both_or_neither : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬FromEarth Marvin ∧ ¬FromMars Marvin)
axiom A7_if_not_fromEarth_then_extraterrestrial : ¬FromEarth Marvin → Extraterrestrial Marvin

theorem not_imp_statement : ¬(¬FromMars Marvin → Human Marvin) := by
  intro hImp
  have hdisj := A6_Marvin_either_both_or_neither
  cases hdisj with
  | inl hboth =>
      have hMars : FromMars Marvin := hboth.right
      have hAlien : Alien Marvin := A2_fromMars_alien Marvin hMars
      have hExtra : Extraterrestrial Marvin := A1_all_aliens_extraterrestrials Marvin hAlien
      have hNotHuman : ¬Human Marvin := A3_no_extraterrestrial_human Marvin hExtra
      have hHuman : Human Marvin := A4_hi_fromEarth_human Marvin ⟨A5_Marvin_hi, hboth.left⟩
      exact hNotHuman hHuman
  | inr hneither =>
      have hNotEarth : ¬FromEarth Marvin := hneither.left
      have hNotMars : ¬FromMars Marvin := hneither.right
      have hExtra : Extraterrestrial Marvin := A7_if_not_fromEarth_then_extraterrestrial hNotEarth
      have hNotHuman : ¬Human Marvin := A3_no_extraterrestrial_human Marvin hExtra
      have hHuman : Human Marvin := hImp hNotMars
      exact hNotHuman hHuman
</lean>

ANSWER: False
