<lean>
axiom obj : Type
axiom Djokovic : obj

axiom AtMixer : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom Famous : obj → Prop
axiom WellPaid : obj → Prop
axiom LivesInTaxHaven : obj → Prop

-- Everyone at the mixer is a Grand Slam champion or an Oscar-nominated actor.
axiom P1 : ∀ x : obj, AtMixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)
-- Every Grand Slam champion at the mixer is a professional tennis player.
axiom P2 : ∀ x : obj, AtMixer x → GrandSlamChampion x → ProTennisPlayer x
-- All Oscar-nominated actors at the mixer are celebrities.
axiom P3 : ∀ x : obj, AtMixer x → OscarNominatedActor x → Celebrity x
-- All professional tennis players at the mixer are athletes.
axiom P4 : ∀ x : obj, AtMixer x → ProTennisPlayer x → Athlete x
-- If a person at the mixer is a celebrity, then they are well paid.
axiom P5 : ∀ x : obj, AtMixer x → Celebrity x → WellPaid x
-- If a person at the mixer is an athlete, then they are famous.
axiom P6 : ∀ x : obj, AtMixer x → Athlete x → Famous x
-- All well-paid people at the mixer live in tax havens.
axiom P7 : ∀ x : obj, AtMixer x → WellPaid x → LivesInTaxHaven x

-- Djokovic is at the mixer
axiom P8a : AtMixer Djokovic
-- If Djokovic is a famous athlete, then Djokovic is well-paid.
axiom P8b : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

-- Derivable consequences (not the target negation)
theorem wellPaid_djokovic : WellPaid Djokovic := by
  have h0 : AtMixer Djokovic := P8a
  have h1 : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := P1 Djokovic h0
  cases h1 with
  | inl hgsc =>
    have hpt : ProTennisPlayer Djokovic := P2 Djokovic h0 hgsc
    have ha : Athlete Djokovic := P4 Djokovic h0 hpt
    have hf : Famous Djokovic := P6 Djokovic h0 ha
    exact P8b ⟨hf, ha⟩
  | inr hoa =>
    have hc : Celebrity Djokovic := P3 Djokovic h0 hoa
    exact P5 Djokovic h0 hc

theorem taxhaven_djokovic : LivesInTaxHaven Djokovic := by
  apply P7 Djokovic P8a
  exact wellPaid_djokovic
