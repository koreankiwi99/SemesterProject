<lean>
axiom obj : Type
axiom AtMixer : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProfessionalTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom LivesInTaxHaven : obj → Prop

axiom Djokovic : obj

-- Premises
axiom P1 : ∀ x : obj, AtMixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)
axiom P2 : ∀ x : obj, AtMixer x → GrandSlamChampion x → ProfessionalTennisPlayer x
axiom P3 : ∀ x : obj, AtMixer x → OscarNominatedActor x → Celebrity x
axiom P4 : ∀ x : obj, AtMixer x → ProfessionalTennisPlayer x → Athlete x
axiom P5 : ∀ x : obj, AtMixer x → Celebrity x → WellPaid x
axiom P6 : ∀ x : obj, AtMixer x → Athlete x → Famous x
axiom P7 : ∀ x : obj, AtMixer x → WellPaid x → LivesInTaxHaven x
axiom P8_at : AtMixer Djokovic
axiom P8_cond : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

-- Derive that Djokovic lives in a tax haven
theorem djokovic_tax_haven : LivesInTaxHaven Djokovic :=
by
  have hAt : AtMixer Djokovic := P8_at
  have hCase : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := P1 Djokovic hAt
  cases hCase with
  | inl hGS =>
      have hPro : ProfessionalTennisPlayer Djokovic := P2 Djokovic hAt hGS
      have hAth : Athlete Djokovic := P4 Djokovic hAt hPro
      have hFam : Famous Djokovic := P6 Djokovic hAt hAth
      have hWP : WellPaid Djokovic := P8_cond (And.intro hFam hAth)
      exact P7 Djokovic hAt hWP
  | inr hOsc =>
      have hCeleb : Celebrity Djokovic := P3 Djokovic hAt hOsc
      have hWP : WellPaid Djokovic := P5 Djokovic hAt hCeleb
      exact P7 Djokovic hAt hWP
</lean>
