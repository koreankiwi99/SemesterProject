<lean>
axiom obj : Type
axiom AtMixer : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom LivesInTaxHaven : obj → Prop

axiom everyone_at_mixer_is_gsc_or_ona : ∀ x : obj, AtMixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)
axiom gsc_at_mixer_are_ptp : ∀ x : obj, AtMixer x → GrandSlamChampion x → ProTennisPlayer x
axiom ona_at_mixer_are_celeb : ∀ x : obj, AtMixer x → OscarNominatedActor x → Celebrity x
axiom ptp_at_mixer_are_athletes : ∀ x : obj, AtMixer x → ProTennisPlayer x → Athlete x
axiom celeb_at_mixer_are_wellpaid : ∀ x : obj, AtMixer x → Celebrity x → WellPaid x
axiom athletes_at_mixer_are_famous : ∀ x : obj, AtMixer x → Athlete x → Famous x
axiom wellpaid_at_mixer_live_in_tax_havens : ∀ x : obj, AtMixer x → WellPaid x → LivesInTaxHaven x

axiom Djokovic : obj
axiom djokovic_at_mixer : AtMixer Djokovic
axiom djokovic_famous_athlete_wellpaid : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

theorem djokovic_lives_in_tax_haven : LivesInTaxHaven Djokovic := by
  have h := everyone_at_mixer_is_gsc_or_ona Djokovic djokovic_at_mixer
  cases h with
  | inl hGSC =>
    have hPTP : ProTennisPlayer Djokovic := gsc_at_mixer_are_ptp Djokovic djokovic_at_mixer hGSC
    have hAth : Athlete Djokovic := ptp_at_mixer_are_athletes Djokovic djokovic_at_mixer hPTP
    have hFam : Famous Djokovic := athletes_at_mixer_are_famous Djokovic djokovic_at_mixer hAth
    have hWP : WellPaid Djokovic := djokovic_famous_athlete_wellpaid ⟨hFam, hAth⟩
    exact wellpaid_at_mixer_live_in_tax_havens Djokovic djokovic_at_mixer hWP
  | inr hONA =>
    have hCeleb : Celebrity Djokovic := ona_at_mixer_are_celeb Djokovic djokovic_at_mixer hONA
    have hWP : WellPaid Djokovic := celeb_at_mixer_are_wellpaid Djokovic djokovic_at_mixer hCeleb
    exact wellpaid_at_mixer_live_in_tax_havens Djokovic djokovic_at_mixer hWP
</lean>

ANSWER: False
