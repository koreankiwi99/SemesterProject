<lean>
axiom obj : Type

axiom DesignByMax : obj → Prop
axiom MaxAdores : obj → Prop
axiom ZahaHadidStyle : obj → Prop
axiom KellyWearstlerStyle : obj → Prop
axiom Interesting : obj → Prop
axiom Brutalist : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom A1 : ∀ x : obj, ZahaHadidStyle x ∧ MaxAdores x → Interesting x
-- No brutalist buildings that Max adores have interesting geometries.
axiom A2 : ∀ x : obj, Brutalist x ∧ MaxAdores x → ¬Interesting x
-- Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom A3 : ∀ x : obj, MaxAdores x → ZahaHadidStyle x ∨ KellyWearstlerStyle x
-- All of Kelly Wearstler's design styles that Max adores are evocative.
axiom A4 : ∀ x : obj, KellyWearstlerStyle x ∧ MaxAdores x → Evocative x
-- All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom A5 : ∀ x : obj, KellyWearstlerStyle x ∧ MaxAdores x → Dreamy x
-- If a design by Max that he adores has interesting geometries, then it is a brutalist building and evocative.
axiom A6 : ∀ x : obj, DesignByMax x ∧ MaxAdores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- From the premises, we can show there is no design by Max that he adores and that has interesting geometries.
theorem no_max_adored_interesting_design :
  ∀ x : obj, DesignByMax x ∧ MaxAdores x ∧ Interesting x → False := by
  intro x
  intro h
  have hDM : DesignByMax x := And.left h
  have hAIG : MaxAdores x ∧ Interesting x := And.right h
  have hA : MaxAdores x := And.left hAIG
  have hIG : Interesting x := And.right hAIG
  have hBE : Brutalist x ∧ Evocative x := A6 x (And.intro hDM (And.intro hA hIG))
  have hB : Brutalist x := And.left hBE
  have hNotIG : ¬Interesting x := A2 x (And.intro hB hA)
  exact hNotIG hIG

-- Also, any Zaha Hadid style that Max adores cannot be brutalist.
theorem adored_ZH_not_brutalist :
  ∀ x : obj, ZahaHadidStyle x ∧ MaxAdores x → ¬Brutalist x := by
  intro x
  intro h
  intro hBru
  have hIG : Interesting x := A1 x h
  have hNotIG : ¬Interesting x := A2 x (And.intro hBru (And.right h))
  exact hNotIG hIG
