-- Derived: There is no design by Max that he adores which is interesting.
theorem no_adored_interesting_designByMax : ∀ x : obj, DesignByMax x → Adores x → Interesting x → False := by
  intro x dx ax ix
  have hBE : Brutalist x ∧ Evocative x := by
    have h := T6 x
    have hin : DesignByMax x ∧ Adores x ∧ Interesting x := And.intro dx (And.intro ax ix)
    exact h hin

  have hnotI : ¬ Interesting x := by
    have hbai : Brutalist x ∧ Adores x := And.intro hB ax
    exact (T2 x) hbai
  exact hnotI ix

  intro ix
  exact no_adored_interesting_designByMax x dx ax ix
