<lean>
axiom obj : Type

-- Individuals and predicates
axiom Coco : obj
axiom AtRG : obj → Prop
axiom Female : obj → Prop
axiom Male : obj → Prop
axiom LostIga : obj → Prop
axiom LostRafa : obj → Prop
axiom RankedHigh : obj → Prop
axiom ActiveMajor : obj → Prop

-- Premises
-- If someone is ranked highly by the WTA, then they are one of the most active players in major tennis.
axiom P1 : ∀ x : obj, RankedHigh x → ActiveMajor x
-- Everyone who lost to Iga Świątek at Roland Garros 2022 is ranked highly by the WTA.
axiom P2 : ∀ x : obj, LostIga x → RankedHigh x
-- All female tennis players at Roland Garros 2022 lost to Iga.
axiom P3 : ∀ x : obj, AtRG x → Female x → LostIga x
-- All tennis players at Roland Garros 2022 are either female or male.
axiom P4 : ∀ x : obj, AtRG x → (Female x ∨ Male x)
-- All male tennis players at Roland Garros in 2022 lost to Rafael Nadal.
axiom P5 : ∀ x : obj, AtRG x → Male x → LostRafa x
-- If Coco Gauff is ranked highly by the WTA or lost to Rafael Nadal, then Coco Gauff is not a male tennis player at Roland Garros 2022.
axiom P6 : (RankedHigh Coco ∨ LostRafa Coco) → ¬(Male Coco ∧ AtRG Coco)
-- Coco Gauff is at Roland Garros 2022.
axiom CocoAtRG : AtRG Coco

-- Some derived consequences (showing what follows from the premises)

theorem coco_not_male : ¬ Male Coco := by
  have hdisj : Female Coco ∨ Male Coco := P4 Coco CocoAtRG
  cases hdisj with
  | inl hfem =>
      have hLostIga : LostIga Coco := P3 Coco CocoAtRG hfem
      have hRank : RankedHigh Coco := P2 Coco hLostIga
      have hnot : ¬ (Male Coco ∧ AtRG Coco) := P6 (Or.inl hRank)
      exact (fun hm => hnot ⟨hm, CocoAtRG⟩)
  | inr hmale =>
      have hLost : LostRafa Coco := P5 Coco CocoAtRG hmale
      have hnot : ¬ (Male Coco ∧ AtRG Coco) := P6 (Or.inr hLost)
      exact False.elim (hnot ⟨hmale, CocoAtRG⟩)

theorem coco_female : Female Coco := by
  have hdisj : Female Coco ∨ Male Coco := P4 Coco CocoAtRG
  have hnm : ¬ Male Coco := coco_not_male
  cases hdisj with
  | inl hf => exact hf
  | inr hm => exact False.elim (hnm hm)

theorem coco_ranked : RankedHigh Coco := by
  have hf : Female Coco := coco_female
  have hIga : LostIga Coco := P3 Coco CocoAtRG hf
  exact P2 Coco hIga
