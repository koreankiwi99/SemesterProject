axiom obj : Type
axiom Cat : obj → Prop
axiom Mammal : obj → Prop
axiom Pet : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Cat x → Mammal x
axiom A2 : ∃ x : obj, Pet x ∧ ¬Mammal x

-- From the premises we can derive that some pet is not a cat.
-- This does not contradict "No pets are cats"; it only shows at least one pet is not a cat.
theorem some_pet_not_cat : ∃ x : obj, Pet x ∧ ¬Cat x := by
  apply Exists.elim A2
  intro x hx
  have hxPet : Pet x := And.left hx
  have hxNotMammal : ¬Mammal x := And.right hx
  have hxNotCat : ¬Cat x := by
    intro hxCat
    have hxMammal : Mammal x := A1 x hxCat
    exact hxNotMammal hxMammal
  exact Exists.intro x (And.intro hxPet hxNotCat)
