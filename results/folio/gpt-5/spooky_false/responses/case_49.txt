axiom P1 : ∀ x : obj, DigitalMedia x → ¬AnalogMedia x
axiom P2 : ∀ x : obj, PrintedText x → AnalogMedia x
axiom P3 : ∀ x : obj, StreamingService x → DigitalMedia x
axiom P4 : ∀ x : obj, HardcoverBook x → PrintedText x
axiom P5 : StreamingService NineteenEightyFour → HardcoverBook NineteenEightyFour

theorem not_streaming_1984 : ¬StreamingService NineteenEightyFour := by
  intro hS
  have hD : DigitalMedia NineteenEightyFour := P3 NineteenEightyFour hS
  have hHC : HardcoverBook NineteenEightyFour := P5 hS
  have hPT : PrintedText NineteenEightyFour := P4 NineteenEightyFour hHC
  have hA : AnalogMedia NineteenEightyFour := P2 NineteenEightyFour hPT
  have hNotA : ¬AnalogMedia NineteenEightyFour := P1 NineteenEightyFour hD
  exact hNotA hA
