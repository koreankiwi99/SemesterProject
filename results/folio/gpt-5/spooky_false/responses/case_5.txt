<lean>
axiom obj : Type

-- Individuals
axiom James : obj

-- Predicates
axiom Employee : obj → Prop
axiom Manager : obj → Prop
axiom InOtherCountries : obj → Prop
axiom RemoteFromHome : obj → Prop
axiom LunchAtHome : obj → Prop
axiom LunchInCompanyBuilding : obj → Prop
axiom SchedulesMeetingWithCustomers : obj → Prop
axiom GoToCompanyBuildingToday : obj → Prop
axiom AppearInCompanyToday : obj → Prop

-- Premises
-- 1. All employees who schedule a meeting with their customers will go to the company building today.
axiom P1 : ∀ x : obj, Employee x → SchedulesMeetingWithCustomers x → GoToCompanyBuildingToday x

-- 2. Everyone who has lunch in the company building schedules meetings with their customers.
axiom P2 : ∀ x : obj, LunchInCompanyBuilding x → SchedulesMeetingWithCustomers x

-- 3. Employees have lunch either in the company building or at home.
axiom P3 : ∀ x : obj, Employee x → (LunchInCompanyBuilding x ∨ LunchAtHome x)

-- 4. If an employee has lunch at home, they are working remotely from home.
axiom P4 : ∀ x : obj, Employee x → LunchAtHome x → RemoteFromHome x

-- 5. All employees who are in other countries work remotely from home.
axiom P5 : ∀ x : obj, Employee x → InOtherCountries x → RemoteFromHome x

-- 6. No managers work remotely from home.
axiom P6 : ∀ x : obj, Manager x → ¬RemoteFromHome x

-- 7. James will appear in the company today iff he is a manager.
axiom P7 : AppearInCompanyToday James ↔ Manager James

-- A small derived fact (for illustration): if James works remotely, then he is not a manager.
theorem remote_implies_not_manager_for_James : RemoteFromHome James → ¬Manager James := by
  intro r
  intro m
  have h := P6 James m
  exact h r

-- The statement to evaluate (as a single proposition):
-- If James is either a manager or in other countries, then James does not either has lunch at home or work remotely from home.
-- If James either has lunch at home or works remotely from home, then he is neither a manager nor does he work in other countries.
def Statement : Prop :=
  ((Manager James ∨ InOtherCountries James) → ¬(LunchAtHome James ∨ RemoteFromHome James)) ∧
  ((LunchAtHome James ∨ RemoteFromHome James) → (¬Manager James ∧ ¬InOtherCountries James))
</lean>
