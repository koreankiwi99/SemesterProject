axiom obj : Type
axiom Robert : obj
axiom BayernMunchen : obj

axiom Striker : obj → Prop
axiom SoccerPlayer : obj → Prop
axiom PlaysFor : obj → obj → Prop
axiom Leaves : obj → obj → Prop

-- Premises
axiom P1 : Striker Robert
axiom P2 : ∀ x : obj, Striker x → SoccerPlayer x
axiom P3 : Leaves Robert BayernMunchen
axiom R1 : ∀ x t : obj, SoccerPlayer x → Leaves x t → ¬PlaysFor x t

-- Conclusion to refute: PlaysFor Robert BayernMunchen
-- We prove its negation.
theorem robert_not_play_for_bayern : ¬PlaysFor Robert BayernMunchen := by
  have hsp : SoccerPlayer Robert := P2 Robert P1
  exact R1 Robert BayernMunchen hsp P3
