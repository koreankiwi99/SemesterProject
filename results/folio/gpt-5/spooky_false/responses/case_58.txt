-- Predicates
axiom City : obj → Prop
axiom State : obj → Prop
axiom In : obj → obj → Prop

def SameState (x y : obj) : Prop := ∃ s : obj, State s ∧ In x s ∧ In y s

axiom C_StPierre : City StPierre
axiom StPierre_not_in_Montana : ¬ In StPierre Montana

-- Any city in the same state as Butte is not in the same state as St Pierre
axiom any_city_in_Butte_not_in_StPierre :
  ∀ c : obj, City c → SameState c Butte → ¬ SameState c StPierre

-- Uniqueness of state for cities except the listed exceptions
axiom unique_state_except_exceptions :
  ∀ c s1 s2, City c → State s1 → State s2 → In c s1 → In c s2 →
    (c = Bristol ∨ c = Texarkana ∨ c = Texhoma ∨ c = UnionCity ∨ s1 = s2)

-- A derived fact (not the asked conclusion): Butte and St Pierre are not in the same state
theorem not_same_state_Butte_StPierre : ¬ SameState Butte StPierre := by
  have hb : SameState Butte Butte := by
    refine Exists.intro Montana ?_
    have hS : State Montana := S_Montana
    have hI : In Butte Montana := In_Butte_Montana
    exact And.intro hS (And.intro hI hI)
  exact any_city_in_Butte_not_in_StPierre Butte C_Butte hb
