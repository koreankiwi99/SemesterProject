axiom InUS : obj → Prop
axiom F1 : obj → Prop
axiom J1 : obj → Prop
axiom WantsWorkUS : obj → Prop
axiom NeedsCPT : obj → Prop
axiom NeedsOPT : obj → Prop

-- International students in the US have either an F1 visa or a J1 visa.
axiom visa_either : ∀ x : obj, InternationalStudent x ∧ InUS x → F1 x ∨ J1 x

-- An international student in the US with an F1 visa needs to apply for CPT or OPT if the student wants to work in the US.
axiom f1_work_rule :
  ∀ x : obj, InternationalStudent x ∧ InUS x ∧ F1 x ∧ WantsWorkUS x → NeedsCPT x ∨ NeedsOPT x

-- Mike is an international student.
axiom Mike : obj
axiom Mike_international : InternationalStudent Mike

-- Mike needs to apply for CPT if he wants to work in the US.
axiom Mike_CPT_if_wants : WantsWorkUS Mike → NeedsCPT Mike

-- A derived fact: if Mike is in the US, has F1, and wants to work in the US, then he needs CPT or OPT.
theorem Mike_CPT_or_OPT_if_F1_inUS_and_wants :
  InUS Mike → F1 Mike → WantsWorkUS Mike → NeedsCPT Mike ∨ NeedsOPT Mike := by
  intro hUS hF1 hW
  have hIntl : InternationalStudent Mike := Mike_international
  have hconj : InternationalStudent Mike ∧ InUS Mike ∧ F1 Mike ∧ WantsWorkUS Mike :=
    And.intro hIntl (And.intro hUS (And.intro hF1 hW))
  exact f1_work_rule Mike hconj

-- From Mike's specific rule, if he wants to work in the US, then he needs CPT or OPT (in fact CPT).
theorem Mike_CPT_or_OPT_if_wants : WantsWorkUS Mike → NeedsCPT Mike ∨ NeedsOPT Mike := by
  intro hW
  have hCPT : NeedsCPT Mike := Mike_CPT_if_wants hW
  exact Or.inl hCPT
