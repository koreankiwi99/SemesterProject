axiom obj : Type
axiom Pet : obj → Prop
axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop
axiom Ted : obj

-- Premises
axiom P1 : ∀ x : obj, BrownSwissCattle x → Cow x
axiom P2 : ∃ x : obj, Pet x ∧ BrownSwissCattle x
axiom P3 : ∀ x : obj, Cow x → Domesticated x
axiom P4 : ∀ x : obj, Alligator x → ¬Domesticated x
axiom P5 : Alligator Ted

-- Derived facts
theorem exists_domesticated_pet : ∃ x : obj, Pet x ∧ Domesticated x := by
  apply Exists.elim P2
  intro x hx
  have hCow : Cow x := P1 x (And.right hx)
  have hDom : Domesticated x := P3 x hCow
  exact Exists.intro x (And.intro (And.left hx) hDom)

theorem Ted_not_domesticated : ¬Domesticated Ted := by
  exact P4 Ted P5

theorem alligator_not_cow : ∀ x : obj, Alligator x → ¬Cow x := by
  intro x hx
  intro hCow
  have hDom : Domesticated x := P3 x hCow
  have hNot : ¬Domesticated x := P4 x hx
  exact hNot hDom

theorem alligator_not_brownSwiss : ∀ x : obj, Alligator x → ¬BrownSwissCattle x := by
  intro x hx
  intro hB
  have hCow : Cow x := P1 x hB
  exact (alligator_not_cow x hx) hCow

theorem Ted_not_brownSwiss : ¬BrownSwissCattle Ted := by
  have hA : Alligator Ted := P5
  intro hB
  have hCow : Cow Ted := P1 Ted hB
  exact (alligator_not_cow Ted hA) hCow
