<lean>
axiom obj : Type

axiom Ted : obj

axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Domesticated : obj → Prop
axiom Pet : obj → Prop
axiom Alligator : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, BrownSwissCattle x → Cow x
axiom P2 : ∃ x : obj, Pet x ∧ BrownSwissCattle x
axiom P3 : ∀ x : obj, Cow x → Domesticated x
axiom P4 : ∀ x : obj, Alligator x → ¬Domesticated x
axiom P5 : Alligator Ted

-- We prove the negation of the statement: ¬(Pet Ted ∧ BrownSwissCattle Ted)
theorem not_ted_pet_and_bsc : ¬(Pet Ted ∧ BrownSwissCattle Ted) := by
  intro h
  have hBSC : BrownSwissCattle Ted := h.right
  have hCow : Cow Ted := P1 Ted hBSC
  have hDom : Domesticated Ted := P3 Ted hCow
  have hNotDom : ¬Domesticated Ted := P4 Ted P5
  exact hNotDom hDom
</lean>

ANSWER: False
