<lean>
axiom Person : Type
axiom Show : Type

axiom Badults : Show
axiom PappysMember : Person → Prop
axiom StarredIn : Person → Show → Prop
axiom PilotedOnBBC2 : Show → Prop
axiom PilotedOnBBC3 : Show → Prop

-- From the text: "starring members of Pappy's" (choose a specific member who starred in Badults)
axiom someMember : Person
axiom someMember_isPappy : PappysMember someMember
axiom someMember_starred_in_Badults : StarredIn someMember Badults

-- From the text: "Badults was piloted ... on BBC Three."
axiom Badults_piloted_BBC3 : PilotedOnBBC3 Badults

-- The statement to evaluate:
-- "No members of Pappy's have starred in a show piloting on BBC Two or BBC Three."
def Statement : Prop :=
  ∀ p : Person, PappysMember p → ∀ s : Show, StarredIn p s → ¬ (PilotedOnBBC2 s ∨ PilotedOnBBC3 s)

-- We prove the negation of the statement from the premises.
theorem statement_false : ¬ Statement := by
  intro h
  have h_for_badults := h someMember someMember_isPappy Badults someMember_starred_in_Badults
  have disj : PilotedOnBBC2 Badults ∨ PilotedOnBBC3 Badults := Or.inr Badults_piloted_BBC3
  exact h_for_badults disj
</lean>

ANSWER: False
