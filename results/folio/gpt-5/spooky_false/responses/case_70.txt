axiom Growth : obj → Prop

axiom BoughtRapid : obj → Prop

axiom P1 : ∀ x : obj, Growth x → BoughtRapid x
axiom P2 : ∀ x : obj, BoughtRapid x → ¬Suitable x
axiom P3 : ∃ x : obj, Stock x ∧ Growth x
axiom P4 : ∀ x : obj, Mature x → Suitable x
axiom P5 : Mature KO

-- Derived facts

  apply P4
  exact P5

theorem exists_stock_not_suitable : ∃ x : obj, Stock x ∧ ¬Suitable x := by
  apply Exists.elim P3
  intro x hx
  have hxS : Stock x := And.left hx
  have hxG : Growth x := And.right hx
  have hBR : BoughtRapid x := P1 x hxG
  have hNotS : ¬Suitable x := P2 x hBR
  exact Exists.intro x (And.intro hxS hNotS)

theorem mature_implies_not_growth : ∀ x : obj, Mature x → ¬Growth x := by
  intro x hxM
  have hxS : Suitable x := P4 x hxM
  intro hxG
  have hxBR : BoughtRapid x := P1 x hxG
  have hxNotS : ¬Suitable x := P2 x hxBR
  exact hxNotS hxS

theorem KO_not_growth : ¬Growth KO := by
  have h : Mature KO → ¬Growth KO := mature_implies_not_growth KO
  exact h P5
