<lean>
axiom obj : Type
axiom Hannah : obj

axiom At : obj → Prop
axiom Work : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom Picky : obj → Prop
axiom SpendALot : obj → Prop
axiom NeedsMoney : obj → Prop

axiom R1 : ∀ x : obj, (At x ∧ Work x) → NeedsMoney x
axiom R2 : ∀ x : obj, (At x ∧ OrderTakeout x) → Work x
axiom R3 : ∀ x : obj, At x → (OrderTakeout x ∨ EnjoyDining x)
axiom R4 : ∀ x : obj, (At x ∧ EnjoyDining x) → ¬Picky x
axiom R5 : ∀ x : obj, (At x ∧ EnjoyDining x) → SpendALot x

axiom TAt : At Hannah
axiom TWork : Work Hannah
axiom TSpecial : NeedsMoney Hannah → (¬Picky Hannah ∧ ¬NeedsMoney Hannah)

def Statement : Prop := At Hannah ∧ (¬Picky Hannah ∨ (Picky Hannah → SpendALot Hannah))

theorem contradiction : False := by
  have hNeed : NeedsMoney Hannah := R1 Hannah (And.intro TAt TWork)
  have hcontra : ¬NeedsMoney Hannah := And.right (TSpecial hNeed)
  exact hcontra hNeed

theorem negate_statement : ¬Statement := by
  intro _
  exact False.elim contradiction
</lean>
