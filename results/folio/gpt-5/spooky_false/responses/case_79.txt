axiom A5 : Spicy DTC ∨ MalaHotpot DTC ∨ ¬ BakedSweet DTC

-- From the premises we can derive that Dried Thai chilies are not products of Baked by Melissa.

  have hc : Cupcake DTC := A4 DTC hbbm
  have hb : BakedSweet DTC := A2 DTC hc
  have hnsp : ¬ Spicy DTC := A1 DTC hb
  have hdisj := A5
  cases hdisj with
  | inl hs => exact hnsp hs
  | inr h2 =>
