axiom obj : Type

-- Entities
axiom MovieS : obj                    -- The Surprising Adventures of Sir Digby Chicken Caesar (the movie)
axiom SirDigby : obj
axiom Nemesis : obj

-- Predicates/relations
axiom SuperheroMovie : obj → Prop
axiom NamedAfter : obj → obj → Prop          -- NamedAfter movie character
axiom InMovie : obj → obj → Prop             -- InMovie character movie
axiom Good : obj → Prop
axiom Bad : obj → Prop
axiom Fights : obj → obj → Prop
axiom Wins : obj → Prop

-- Premises
axiom win_in_superhero_movies :
  ∀ m : obj, SuperheroMovie m → ∀ c : obj, InMovie c m → Good c → Wins c
axiom movie_is_superhero : SuperheroMovie MovieS
axiom fights_good_bad : ∀ x y : obj, Good x → Bad y → Fights x y
axiom fights_bad_good : ∀ x y : obj, Bad x → Good y → Fights x y
axiom digby_fights_nemesis : Fights SirDigby Nemesis
axiom named_good : ∀ m c : obj, SuperheroMovie m → NamedAfter m c → Good c
axiom named_after : NamedAfter MovieS SirDigby
axiom win_fight_exclusion : ∀ x y : obj, Fights x y → Wins x → ¬Wins y
axiom named_in_movie : ∀ m c : obj, SuperheroMovie m → NamedAfter m c → InMovie c m

-- Derivation: Sir Digby's nemesis does not win
theorem nemesis_not_win : ¬Wins Nemesis := by
  have good_sd : Good SirDigby := named_good MovieS SirDigby movie_is_superhero named_after
  have in_sd : InMovie SirDigby MovieS := named_in_movie MovieS SirDigby movie_is_superhero named_after
  have wins_sd : Wins SirDigby := win_in_superhero_movies MovieS movie_is_superhero SirDigby in_sd good_sd
  exact win_fight_exclusion SirDigby Nemesis digby_fights_nemesis wins_sd
</lean>
