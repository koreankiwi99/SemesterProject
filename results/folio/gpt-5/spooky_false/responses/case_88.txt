axiom Book : obj → Prop
axiom Reads : obj → obj → Prop
axiom GainsKnowledge : obj → Prop
axiom SmarterThanBefore : obj → Prop

-- Premises
axiom P_read_gain : ∀ x b, Person x → Book b → Reads x b → GainsKnowledge x
axiom P_gain_smart : ∀ x, GainsKnowledge x → SmarterThanBefore x

-- Entities and facts
axiom Harry : obj
axiom Walden : obj
axiom Harry_is_person : Person Harry
axiom Walden_is_book : Book Walden
axiom Harry_reads_Walden : Reads Harry Walden

-- From the premises, Harry is indeed smarter than before.
theorem harry_smarter : SmarterThanBefore Harry := by
  have hk : GainsKnowledge Harry := P_read_gain Harry Walden Harry_is_person Walden_is_book Harry_reads_Walden
  exact P_gain_smart Harry hk

-- Therefore, attempting to prove the negation leads to a contradiction:
theorem not_harry_smarter_impossible : (¬ SmarterThanBefore Harry) → False := by
  intro hneg
  exact hneg harry_smarter
