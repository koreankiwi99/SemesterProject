<lean>
axiom Person : Type
axiom Book : Type

axiom Reads : Person → Book → Prop
axiom GainsKnowledge : Person → Prop
axiom Smarter : Person → Prop

axiom read_imp_knowledge : ∀ p : Person, ∀ b : Book, Reads p b → GainsKnowledge p
axiom knowledge_imp_smarter : ∀ p : Person, GainsKnowledge p → Smarter p

axiom Harry : Person
axiom Walden : Book
axiom harry_read_walden : Reads Harry Walden

-- From the premises, we can derive that Harry gained knowledge and became smarter.
theorem harry_gained : GainsKnowledge Harry :=
  read_imp_knowledge Harry Walden harry_read_walden

theorem harry_smarter : Smarter Harry :=
  knowledge_imp_smarter Harry harry_gained

-- Hence, there exists a smarter person who has gained knowledge (Harry).
theorem exists_smarter_who_gained : ∃ p : Person, Smarter p ∧ GainsKnowledge p :=
  Exists.intro Harry (And.intro harry_smarter harry_gained)
