axiom Task : Type
axiom MachineTranslation : Task
axiom NLP : Task → Prop
axiom Generation : Task → Prop
axiom Understanding : Task → Prop
axiom OutputText : Task → Prop

-- Categorization as an exclusive choice between generation and understanding
axiom Cat : ∀ t : Task, NLP t → (Generation t ∧ ¬Understanding t) ∨ (Understanding t ∧ ¬Generation t)

-- If the output of an NLP task is a text sequence, then it is a language generation task
axiom OutImpliesGen : ∀ t : Task, NLP t → OutputText t → Generation t

axiom MT_is_NLP : NLP MachineTranslation
axiom MT_output_text : OutputText MachineTranslation

theorem mt_not_understanding : ¬Understanding MachineTranslation := by
  have hG : Generation MachineTranslation := OutImpliesGen MachineTranslation MT_is_NLP MT_output_text
  have hxor := Cat MachineTranslation MT_is_NLP
  cases hxor with
  | inl h =>
      exact h.right
  | inr h =>
      exact False.elim (h.right hG)
