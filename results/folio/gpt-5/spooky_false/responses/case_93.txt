<lean>
axiom Person : Type
axiom Animal : Type
axiom Building : Type

axiom Tom : Person
axiom Fluffy : Animal
axiom OliveGarden : Building

axiom Managed : Building → Prop
axiom AllowsPets : Building → Prop
axiom Cat : Animal → Prop
axiom Pet : Animal → Prop
axiom AllowedMoveIn : Person → Animal → Building → Prop
axiom WillRent : Person → Building → Prop

-- Some managed building allows pets (existential; does not specify The Olive Garden)
axiom some_managed_allows_pets : ∃ b : Building, Managed b ∧ AllowsPets b

-- A deposit is required to rent an apartment in a managed building
axiom DepositRequired : Building → Prop
axiom deposit_required_to_rent : ∀ p b, Managed b → WillRent p b → DepositRequired b

-- Security deposit and monthly rent
axiom SecurityDeposit : Building → Nat
axiom MonthlyRent : Building → Nat
axiom dep_ge_rent : ∀ b, Managed b → SecurityDeposit b ≥ MonthlyRent b

-- Fluffy is Tom's cat; cats are pets
axiom fluffy_cat : Cat Fluffy
axiom cats_are_pets : ∀ a, Cat a → Pet a

-- Managed status of The Olive Garden
axiom og_managed : Managed OliveGarden

-- The monthly rent at The Olive Garden is $2000
axiom rent_og_2000 : MonthlyRent OliveGarden = 2000

-- $2000 is more than $1500
axiom two_thousand_gt_one_thousand_five_hundred : 2000 > 1500

-- From the numeric facts (and "deposit ≥ rent"), we record that the deposit at OG is more than $1500
axiom dep_og_gt_1500 : SecurityDeposit OliveGarden > 1500

-- Tom will rent an apartment in a managed building if he is allowed to move in with Fluffy,
-- and the security deposit is no more than $1500
axiom rents_if_allowed_and_deposit_small :
  ∀ b, Managed b → AllowedMoveIn Tom Fluffy b → SecurityDeposit b ≤ 1500 → WillRent Tom b

-- If a managed building allows pets, then people are allowed to move in with a pet
axiom allows_pets_implies_allowed_movein :
  ∀ b, Managed b → AllowsPets b → ∀ p a, Pet a → AllowedMoveIn p a b

theorem fluffy_is_pet : Pet Fluffy := cats_are_pets Fluffy fluffy_cat
