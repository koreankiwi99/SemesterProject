axiom LegalEntity : obj → Prop

axiom HasLegalObligations : obj → Prop

axiom AllCompaniesAreBusiness : ∀ x : obj, Company x → BusinessOrg x

axiom AllLegalAreCreatedUnderLaw : ∀ x : obj, LegalEntity x → CreatedUnderLaw x

  intro hpriv

  have hleg : LegalEntity HarvardWeeklyBookClub := AllBusinessAreLegal HarvardWeeklyBookClub hbus

  have hnotpriv : ¬PrivateCompany HarvardWeeklyBookClub := HWB_created_under_law_implies_not_private hcul
  exact hnotpriv hpriv
