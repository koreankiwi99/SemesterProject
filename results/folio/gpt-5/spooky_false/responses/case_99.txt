axiom obj : Type

-- Predicates
axiom Animal : obj → Prop
axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom ReproMF : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop

-- Constant individual

-- Premises
axiom P1 : ∀ x : obj, Animal x → (Invertebrate x ∨ Vertebrate x)                  -- Animals are either invertebrates or vertebrates.
axiom P2 : ∀ x : obj, (Animal x ∧ Backbone x) → ReproMF x                         -- All animals with backbones reproduce by male-and-female mating.
axiom P3 : ∀ x : obj, Vertebrate x → Backbone x                                   -- All vertebrate animals have a backbone.
axiom P4 : ∀ x : obj, Bee x → ¬ReproMF x                                          -- All bees do not reproduce by male-and-female mating.
axiom P5 : ∀ x : obj, QueenBee x → Bee x                                          -- All queen bees are bees.
axiom P6 : Bee Harry                                                               -- Harry is a bee.

-- The statement to evaluate
def Statement : Prop := Invertebrate Harry ∨ QueenBee Harry

-- A helpful consequence: if Harry is an animal, then he is an invertebrate (cannot be a vertebrate, given P4 and P6).
theorem harry_invertebrate_if_animal : Animal Harry → Invertebrate Harry := by
  intro hAnimal
  have hbee : Bee Harry := P6
  have hdisj : Invertebrate Harry ∨ Vertebrate Harry := P1 Harry hAnimal
  cases hdisj with
  | inl hInv => exact hInv
  | inr hVer =>
      have hBack : Backbone Harry := P3 Harry hVer
      have hRepro : ReproMF Harry := P2 Harry ⟨hAnimal, hBack⟩
      have hNotRepro : ¬ReproMF Harry := P4 Harry hbee
      exact False.elim (hNotRepro hRepro)
