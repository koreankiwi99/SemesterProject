<lean>
axiom obj : Type

axiom InClub : obj → Prop
axiom PerformsOften : obj → Prop
axiom AttendsEvents : obj → Prop
axiom Engaged : obj → Prop
axiom InactiveDisinterested : obj → Prop
axiom ChaperoneDances : obj → Prop
axiom StudentAttendsSchool : obj → Prop
axiom YoungChild : obj → Prop
axiom Teenager : obj → Prop
axiom WishFurtherAcademic : obj → Prop

axiom Bonnie : obj

-- 1. Club members who perform often attend and are engaged with school events.
axiom A1 : ∀ x : obj, InClub x ∧ PerformsOften x → AttendsEvents x ∧ Engaged x

-- 2. Club members either perform often or are inactive and disinterested.
axiom A2 : ∀ x : obj, InClub x → PerformsOften x ∨ InactiveDisinterested x

-- 3. Club members who chaperone dances are not students at the school.
axiom A3 : ∀ x : obj, InClub x ∧ ChaperoneDances x → ¬ StudentAttendsSchool x

-- 4. Inactive and disinterested club members chaperone dances.
axiom A4 : ∀ x : obj, InClub x ∧ InactiveDisinterested x → ChaperoneDances x

-- 5. Young children or teenagers in the club who wish to further academics are students.
axiom A5 : ∀ x : obj, InClub x ∧ (YoungChild x ∨ Teenager x) ∧ WishFurtherAcademic x → StudentAttendsSchool x

-- 6. Bonnie is in the club.
axiom A6 : InClub Bonnie

-- 7. Bonnie either (attends and is engaged and is a student) or (is not both attending-and-engaged and is not a student).
axiom A7 :
  (AttendsEvents Bonnie ∧ Engaged Bonnie ∧ StudentAttendsSchool Bonnie) ∨
  ((¬ (AttendsEvents Bonnie ∧ Engaged Bonnie)) ∧ ¬ StudentAttendsSchool Bonnie)

-- Useful derived facts

theorem inactive_implies_not_student (x : obj) (hc : InClub x) (hi : InactiveDisinterested x) : ¬ StudentAttendsSchool x := by
  have hch : ChaperoneDances x := A4 x (And.intro hc hi)
  exact A3 x (And.intro hc hch)

theorem not_AE_implies_not_perform (x : obj) (hc : InClub x) (hnae : ¬ (AttendsEvents x ∧ Engaged x)) : ¬ PerformsOften x := by
  intro hp
  have hAE : AttendsEvents x ∧ Engaged x := A1 x (And.intro hc hp)
  exact hnae hAE

-- From membership, Bonnie either performs often or is inactive/disinterested.
theorem bonnie_performs_or_inactive : PerformsOften Bonnie ∨ InactiveDisinterested Bonnie := A2 Bonnie A6
