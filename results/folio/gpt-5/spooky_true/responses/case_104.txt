<lean>
axiom obj : Type
axiom Tom : obj

axiom GREApplicant : obj → Prop
axiom ProvesHardship : obj → Prop
axiom SingleParentFamily : obj → Prop
axiom FewResources : obj → Prop
axiom ETSProvidesAidTo : obj → Prop

axiom AidRule : ∀ x : obj, GREApplicant x → ProvesHardship x → ETSProvidesAidTo x
axiom HardshipFromSingleParent : ∀ x : obj, SingleParentFamily x → ProvesHardship x
axiom HardshipFromFewResources : ∀ x : obj, FewResources x → ProvesHardship x

axiom TomSingleParent : SingleParentFamily Tom
axiom TomFewResources : FewResources Tom
axiom TomApplicant : GREApplicant Tom

theorem ets_provides_financial_aid_to_Tom : ETSProvidesAidTo Tom := by
  have h : ProvesHardship Tom := HardshipFromSingleParent Tom TomSingleParent
  exact AidRule Tom TomApplicant h
</lean>

ANSWER: True
