-- Rules from the text
-- ETS provides financial aid to those GRE applicants who prove economic hardship.
axiom aid_rule :
  ∀ x : obj, ApplyingGRE x → ProvesEconHardship x → FinancialAidProvidedBy ETS x

-- Those living in single-parent families or having few resources can prove economic hardship.

  ∀ x : obj, (LivesSingleParent x ∨ FewResources x) → ProvesEconHardship x

  have h : LivesSingleParent Tom ∨ FewResources Tom := Or.inl tom_single
  exact prove_rule Tom h

  exact aid_rule Tom tom_apply tom_econ

  exact Exists.intro ETS tom_aid

-- The statement to evaluate:
-- "No one taking the GRE test has financial aid provided to them by something."
def Statement : Prop := ∀ x : obj, TakingGRE x → ¬ ∃ y : obj, FinancialAidProvidedBy y x
