axiom Person : Type
axiom Spill : Person → Prop
axiom Tidy : Person → Prop
axiom Cleanly : Person → Prop
axiom ValuesOrderSpotlessness : Person → Prop
axiom FamilyPrioritizesOrderSpotlessness : Person → Prop
axiom ClumsyFoodie : Person → Prop
axiom GoesOutFrequentlyForNewRestaurants : Person → Prop

-- If people spill a lot of food on their clothing, then they are not notably tidy.
axiom A1 : ∀ x : Person, Spill x → ¬ Tidy x

-- All clumsy foodies who go out frequently to find new food restaurants to try spill a lot of food on their clothing.
axiom A2 : ∀ x : Person, (ClumsyFoodie x ∧ GoesOutFrequentlyForNewRestaurants x) → Spill x

-- If people are cleanly, then they are notably tidy.
axiom A3 : ∀ x : Person, Cleanly x → Tidy x

-- If people value order and spotlessness, then they are cleanly.
axiom A4 : ∀ x : Person, ValuesOrderSpotlessness x → Cleanly x

-- All people whose families prioritize order and spotlessness themselves value order and spotlessness.
axiom A5 : ∀ x : Person, FamilyPrioritizesOrderSpotlessness x → ValuesOrderSpotlessness x

axiom Peter : Person

-- Peter is either (spills ∧ cleanly) or (¬spills ∧ ¬cleanly).
axiom Peter_disj : (Spill Peter ∧ Cleanly Peter) ∨ (¬ Spill Peter ∧ ¬ Cleanly Peter)

-- From being cleanly we can derive being notably tidy.
theorem tidy_of_cleanly (x : Person) (hc : Cleanly x) : Tidy x := by
  exact A3 x hc

-- If the first disjunct holds for Peter, then Peter is notably tidy.
theorem peter_tidy_if_first_case : (Spill Peter ∧ Cleanly Peter) → Tidy Peter := by
  intro h
  exact A3 Peter h.right

-- From the given disjunction we can at least derive: Peter is tidy or Peter is not cleanly.
theorem peter_tidy_or_not_cleanly : Tidy Peter ∨ ¬ Cleanly Peter := by
  cases Peter_disj with
  | inl h =>
      apply Or.inl
      exact A3 Peter h.right
  | inr h =>
      apply Or.inr
      exact h.right
