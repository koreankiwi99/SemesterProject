axiom obj : Type
axiom Evergreen : obj → Prop
axiom Fir : obj → Prop
axiom Worship : obj → Prop

-- Premises
axiom All_fir_are_evergreen : ∀ x : obj, Fir x → Evergreen x
axiom Some_worship_are_fir : ∃ x : obj, Worship x ∧ Fir x

-- From the premises we can derive that some evergreens are objects of worship
-- (but not that some evergreens are not objects of worship).
theorem some_evergreens_are_worship : ∃ x : obj, Evergreen x ∧ Worship x := by
  rcases Some_worship_are_fir with ⟨x, hx⟩
  rcases hx with ⟨hxW, hxF⟩
  have hxE : Evergreen x := All_fir_are_evergreen x hxF
  exact Exists.intro x ⟨hxE, hxW⟩
