<lean>
axiom obj : Type

axiom Picuris : obj
axiom Harding : obj
axiom Juan : obj

axiom MountainRange : obj → Prop
axiom InNM : obj → Prop
axiom InTX : obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

axiom Picuris_is_mtn : MountainRange Picuris
axiom Picuris_in_nm_or_tx : InNM Picuris ∨ InTX Picuris
axiom Juan_visited_Picuris : Visited Juan Picuris
axiom Harding_is_mine : Mine Harding
axiom Harding_located_in_Picuris : LocatedIn Harding Picuris
axiom Harding_was_donated : Donated Harding

-- No mountain range in Texas has a mine located in it that was donated
axiom no_tx_mtn_with_donated_mine :
  ∀ r : obj, MountainRange r → InTX r → ∀ m : obj, Mine m → LocatedIn m r → Donated m → False

theorem statement : ∃ r : obj, MountainRange r ∧ InNM r ∧ Visited Juan r := by
  have not_in_tx : ¬ InTX Picuris := by
    intro htx
    have h := no_tx_mtn_with_donated_mine Picuris Picuris_is_mtn htx
    have contra := h Harding Harding_is_mine Harding_located_in_Picuris Harding_was_donated
    exact contra
  have in_nm : InNM Picuris := by
    cases Picuris_in_nm_or_tx with
    | inl hnm => exact hnm
    | inr htx => exact False.elim (not_in_tx htx)
  exact Exists.intro Picuris (And.intro Picuris_is_mtn (And.intro in_nm Juan_visited_Picuris))
</lean>

ANSWER: True
