  intro htx
  exact no_tx_mtn_with_donated_mine Picuris Harding R_picuris htx harding_is_mine harding_in_picuris harding_donated

-- Therefore, by the disjunction, Picuris is in New Mexico

  cases picuris_in_nm_or_tx with
  | inl hnm => exact hnm
  | inr htx =>
    have contra := picuris_not_in_tx htx
    exact False.elim contra

-- Hence, Harding is located in a mountain range in New Mexico (namely, Picuris)

  ∃ r, MountainRange r ∧ InNM r ∧ LocatedIn Harding r := by

  apply And.intro picuris_in_nm
  exact harding_in_picuris
