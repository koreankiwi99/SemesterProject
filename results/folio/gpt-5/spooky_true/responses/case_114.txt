axiom CanBlock : obj → obj → Prop
axiom InMClass : obj → Prop
axiom Jumps : obj → Prop
axiom GreatShooter : obj → Prop

axiom Windy : obj

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom A1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom A2m : Man Michael
axiom A2 : ∀ y : obj, InMClass y → Taller Michael y

-- Transitivity of taller
axiom A3 : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom A4m : Man Peter
axiom A4 : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom A5 : ∀ p : obj, ¬ Jumps p → CanBlock Michael p

-- Michael cannot block Windy's shooting.
axiom A6 : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom A7 : ∀ s : obj, Jumps s → GreatShooter s

-- Useful consequences we can prove:

-- Peter is taller than everyone in Michael's class.
theorem peter_taller_than_all_in_michaels_class :
  ∀ x : obj, InMClass x → Taller Peter x := by
  intro x hx
  have hm_gt_x : Taller Michael x := A2 x hx
  have hp_gt_m : Taller Peter Michael := A4
  exact A3 Peter Michael x hp_gt_m hm_gt_x

-- From "cannot block" and "if not jumping then can block", we infer Windy jumps (classically).
theorem windy_jumps : Jumps Windy := by
  classical
  by_contra hnot
  have hblock : CanBlock Michael Windy := A5 Windy hnot
  exact A6 hblock

-- Therefore Windy is a great shooter.
theorem windy_great : GreatShooter Windy := by
  have hj : Jumps Windy := windy_jumps
  exact A7 Windy hj
