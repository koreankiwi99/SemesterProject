axiom obj : Type

-- Entities
axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

-- Predicates and relations
axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom CanBlock : obj → obj → Prop            -- CanBlock x y: x can block y's shooting
axiom Jumps : obj → Prop                     -- Jumps x: x can jump when shooting
axiom Great : obj → Prop                     -- Great x: x is a great shooter
axiom InMichaelClass : obj → Prop

-- Axioms from the text
axiom A_taller_blocks : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y
axiom A_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

axiom A_man_michael : Man Michael
axiom A_michael_taller_class : ∀ y : obj, InMichaelClass y → Taller Michael y

axiom A_man_peter : Man Peter
axiom A_peter_taller_michael : Taller Peter Michael

axiom A_michael_blocks_nonjump : ∀ p : obj, ¬ Jumps p → CanBlock Michael p
axiom A_michael_cannot_block_windy : ¬ CanBlock Michael Windy

axiom A_jumpers_are_great : ∀ p : obj, Jumps p → Great p

open Classical

theorem windy_great : Great Windy := by
  -- From "Michael can block anyone who does not jump" and "Michael cannot block Windy",
  -- deduce that Windy must jump (classically).
  have hnnJ : ¬¬ Jumps Windy := by
    intro hnot
    have hblock : CanBlock Michael Windy := A_michael_blocks_nonjump Windy hnot
    exact A_michael_cannot_block_windy hblock
  have hJ : Jumps Windy := by
    by_contra hnot
    exact hnnJ hnot
  -- Jumpers are great shooters
  exact A_jumpers_are_great Windy hJ
</lean>

ANSWER: True
