-- If a man is taller than another man, the taller one can block the other's shooting.
axiom P1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.

axiom P2t : ∀ y : obj, InMClass y → Taller Michael y

-- Transitivity of "taller than".
axiom P3 : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom P4m : Man Peter
axiom P4t : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom P5 : ∀ y : obj, ¬ Jumps y → CanBlock Michael y

-- Michael cannot block Windy's shooting.
axiom P6 : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom P7 : ∀ y : obj, Shooter y ∧ Jumps y → Great y

-- From "Windy's shooting" we model that Windy is a shooter.
axiom P8 : Shooter Windy

open Classical

-- A derivable fact: Windy must be able to jump when shooting (otherwise Michael would block).
theorem windy_jumps : Jumps Windy :=
by
  have himp : ¬ Jumps Windy → CanBlock Michael Windy := P5 Windy
  have hnot : ¬ CanBlock Michael Windy := P6
  by_contra hnj
  have hcb : CanBlock Michael Windy := himp hnj
  exact hnot hcb

-- Hence Windy is a great shooter (not used for the target conclusion, but derivable).
theorem windy_great : Great Windy :=
by
  apply P7 Windy
  apply And.intro
  · exact P8
  · exact windy_jumps
