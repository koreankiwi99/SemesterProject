<lean>
axiom obj : Type
axiom Zelda : obj

axiom JapaneseCompany : obj → Prop
axiom Created : obj → obj → Prop
axiom Top10 : obj → Prop
axiom SoldOver1M : obj → Prop

-- Premise 1: A Japanese game company created the game the Legend of Zelda.
axiom P1 : ∃ c : obj, JapaneseCompany c ∧ Created c Zelda

-- Premise 2: All games on the Top 10 list are made by Japanese game companies.
axiom P2 : ∀ g : obj, Top10 g → ∃ c : obj, JapaneseCompany c ∧ Created c g

-- Premise 3: If a game sells more than one million copies, then it will be included in the Top 10 list.
axiom P3 : ∀ g : obj, SoldOver1M g → Top10 g

-- Premise 4: The Legend of Zelda sold more than one million copies.
axiom P4 : SoldOver1M Zelda

-- Conclusion: The Legend of Zelda is on the Top 10 list.
theorem zelda_on_top10 : Top10 Zelda := by
  exact P3 Zelda P4
</lean>

ANSWER: True
