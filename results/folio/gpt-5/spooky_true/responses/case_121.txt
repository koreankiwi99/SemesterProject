axiom obj : Type
axiom JohnCar : obj

axiom TransportMultiplePassengers : obj → Prop
axiom OneSeater : obj → Prop
axiom TeslaModel3 : obj → Prop
axiom SingleSeatElectricCar : obj → Prop
axiom SoloEV : obj → Prop

-- If something can transport multiple passengers, then they are not one-seaters.
axiom A1 : ∀ x : obj, TransportMultiplePassengers x → ¬ OneSeater x
-- All Tesla Model 3 can transport multiple passengers.
axiom A2 : ∀ x : obj, TeslaModel3 x → TransportMultiplePassengers x
-- All single-seat electric cars are one-seaters.
axiom A3 : ∀ x : obj, SingleSeatElectricCar x → OneSeater x
-- All Solo EV cars are single-seat electric cars.
axiom A4 : ∀ x : obj, SoloEV x → SingleSeatElectricCar x
-- If John's car is not a Tesla Model 3, then John's car is not a Tesla Model 3 or a single-seat electric car.
axiom A5 : ¬ TeslaModel3 JohnCar → (¬ TeslaModel3 JohnCar ∨ SingleSeatElectricCar JohnCar)

-- A derived consequence: any Tesla Model 3 is not a one-seater.
theorem tesla_not_one_seater : ∀ x : obj, TeslaModel3 x → ¬ OneSeater x := by
  intro x hx
  have htmp : TransportMultiplePassengers x := A2 x hx
  have hno : ¬ OneSeater x := A1 x htmp
  exact hno

-- In particular, if John's car were a Tesla Model 3, it would not be a one-seater.
theorem john_tesla_implies_not_one : TeslaModel3 JohnCar → ¬ OneSeater JohnCar := by
  intro h
  exact tesla_not_one_seater JohnCar h
