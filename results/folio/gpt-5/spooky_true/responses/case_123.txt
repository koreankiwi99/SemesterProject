-- Derivable lemma: Solo EV implies not Tesla
theorem SoloEV_implies_not_Tesla : ∀ x : obj, SoloEV x → ¬ TeslaModel3 x := by
  intro x hSolo
  have hSSE : SingleSeatElectricCar x := R4 x hSolo
  have h1s : OneSeater x := R3 x hSSE
  intro hTesla
  have hnot1 : ¬ OneSeater x := R1 x (R2 x hTesla)
  exact hnot1 h1s

theorem SoloEV_John_implies_not_Tesla : SoloEV JohnCar → ¬ TeslaModel3 JohnCar := by
  exact SoloEV_implies_not_Tesla JohnCar
