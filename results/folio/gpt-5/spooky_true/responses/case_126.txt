axiom obj : Type
axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom Fly : obj → Prop
axiom Breathe : obj → Prop
axiom Wings : obj → Prop
axiom Rock : obj

-- All of Peter's pets that are mammals are also animals.
axiom A1 : ∀ x : obj, Pet x ∧ Mammal x → Animal x
-- All monkeys are mammals.
axiom A2 : ∀ x : obj, Monkey x → Mammal x
-- Peter's pets are all either monkeys or birds.
axiom A3 : ∀ x : obj, Pet x → Monkey x ∨ Bird x
-- Peter's birds can fly.
axiom A4 : ∀ x : obj, Pet x ∧ Bird x → Fly x
-- All animals that are Peter's pets can breathe.
axiom A5 : ∀ x : obj, Pet x ∧ Animal x → Breathe x
-- If Peter's pet can fly, then it has wings.
axiom A6 : ∀ x : obj, Pet x ∧ Fly x → Wings x
-- Rock is Peter's pet.
axiom A7 : Pet Rock
-- Rock can fly, or Rock is a bird, or Rock cannot breathe.
axiom A8 : Fly Rock ∨ (Bird Rock ∨ ¬Breathe Rock)

theorem wings_Rock : Wings Rock := by
  have pPet : Pet Rock := A7
  have h8 : Fly Rock ∨ (Bird Rock ∨ ¬Breathe Rock) := A8
  cases h8 with
  | inl hFly =>
      exact A6 Rock ⟨pPet, hFly⟩
  | inr hrest =>
      cases hrest with
      | inl hBird =>
          have hFly2 : Fly Rock := A4 Rock ⟨pPet, hBird⟩
          exact A6 Rock ⟨pPet, hFly2⟩
      | inr hNotB =>
          have hMB : Monkey Rock ∨ Bird Rock := A3 Rock pPet
          cases hMB with
          | inl hMonk =>
              have hMam : Mammal Rock := A2 Rock hMonk
              have hAni : Animal Rock := A1 Rock ⟨pPet, hMam⟩
              have hBre : Breathe Rock := A5 Rock ⟨pPet, hAni⟩
              have f : False := hNotB hBre
              exact False.elim f
          | inr hBird2 =>
              have hFly2 : Fly Rock := A4 Rock ⟨pPet, hBird2⟩
              exact A6 Rock ⟨pPet, hFly2⟩

theorem not_wings_Rock_contradiction : (¬Wings Rock) → False := by
  intro h
  exact h wings_Rock
