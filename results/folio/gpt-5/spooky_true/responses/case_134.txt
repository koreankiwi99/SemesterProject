<lean>
axiom obj : Type

axiom Sucks : obj → Prop
axiom Vacuum : obj → Prop
axiom Plunger : obj → Prop
axiom Vampire : obj → Prop
axiom Appliance : obj → Prop

axiom Space : obj
axiom Duster : obj

-- Premises
axiom P1 : ∀ x : obj, Plunger x → Sucks x
axiom P2 : ∀ x : obj, Vacuum x → Sucks x
axiom P3 : ∀ x : obj, Vampire x → Sucks x
axiom P4 : Vacuum Space
axiom P5 : Appliance Duster
axiom P6 : ¬ Sucks Duster

-- Conclusion: Space sucks.
theorem space_sucks : Sucks Space := by
  have hv : Vacuum Space := P4
  exact P2 Space hv
</lean>

ANSWER: True
