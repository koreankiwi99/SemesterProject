<lean>
axiom Algo : Type
axiom Model : Type
axiom SOTA : Model

axiom UsedToTrain : Algo → Model → Prop
axiom Unsupervised : Algo → Prop
axiom Supervised : Algo → Prop
axiom Reinforcement : Algo → Prop
axiom RequiresLabeled : Algo → Prop

-- Unsupervised learning algorithms do not require labeled data.
axiom A1 : ∀ a : Algo, Unsupervised a → ¬ RequiresLabeled a

-- The state-of-the-art text summarization model is trained with machine learning algorithms (existence).
axiom TrainExists : ∃ a : Algo, UsedToTrain a SOTA

-- Reinforcement learning is not used to train the SOTA model.
axiom NoRLUse : ∀ a : Algo, Reinforcement a → ¬ UsedToTrain a SOTA

-- The algorithm used to train text summarization models requires labeled data.
axiom NeedsLabelsForTS : ∀ a : Algo, UsedToTrain a SOTA → RequiresLabeled a

-- Therefore, no unsupervised algorithm is used to train the SOTA model.
theorem no_unsupervised_used : ¬ ∃ a : Algo, Unsupervised a ∧ UsedToTrain a SOTA :=
by
  intro h
  cases h with
  | intro a ha =>
    cases ha with
    | intro hua htrain =>
      have hreq : RequiresLabeled a := NeedsLabelsForTS a htrain
      have hnot : ¬ RequiresLabeled a := A1 a hua
      exact hnot hreq
