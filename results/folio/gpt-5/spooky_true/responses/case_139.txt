<lean>
axiom obj : Type

-- Predicates
axiom Film : obj → Prop
axiom AppropriateAllAges : obj → Prop
axiom ChildrenCanWatchWithoutGuidance : obj → Prop
axiom ExcessiveEroticViolentContent : obj → Prop
axiom RatedGeneralAudience : obj → Prop
axiom FamilyFriendlyAnimated : obj → Prop
axiom FrozenSeries : obj → Prop

-- Constants
axiom Hachi : obj

-- Premises
axiom P1 : ∀ x : obj, AppropriateAllAges x → ChildrenCanWatchWithoutGuidance x
axiom P2 : ∀ x : obj, ExcessiveEroticViolentContent x → ¬ ChildrenCanWatchWithoutGuidance x
axiom P3 : ∀ x : obj, RatedGeneralAudience x → AppropriateAllAges x
axiom P4 : ∀ x : obj, FamilyFriendlyAnimated x → RatedGeneralAudience x
axiom P5 : ∀ x : obj, FrozenSeries x → FamilyFriendlyAnimated x
axiom Hachi_is_film : Film Hachi
axiom Hachi_FFA_or_Appropriate : FamilyFriendlyAnimated Hachi ∨ AppropriateAllAges Hachi

-- Some consequences we can prove
theorem hachi_children_can_watch : ChildrenCanWatchWithoutGuidance Hachi := by
  cases Hachi_FFA_or_Appropriate with
  | inl hffa =>
      have hga : RatedGeneralAudience Hachi := P4 Hachi hffa
      have happ : AppropriateAllAges Hachi := P3 Hachi hga
      exact P1 Hachi happ
  | inr happ =>
      exact P1 Hachi happ

theorem hachi_not_excessive : ¬ ExcessiveEroticViolentContent Hachi := by
  have hc : ChildrenCanWatchWithoutGuidance Hachi := hachi_children_can_watch
  intro hex
  have hncw : ¬ ChildrenCanWatchWithoutGuidance Hachi := P2 Hachi hex
  exact hncw hc

theorem hachi_ratedGA_or_appropriate : RatedGeneralAudience Hachi ∨ AppropriateAllAges Hachi := by
  cases Hachi_FFA_or_Appropriate with
  | inl hffa =>
      exact Or.inl (P4 Hachi hffa)
  | inr happ =>
      exact Or.inr happ
