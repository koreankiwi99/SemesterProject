axiom CanWatchWithoutGuidance : obj → Prop
axiom ExcessiveEroticViolentContent : obj → Prop
axiom RatedGeneralAudience : obj → Prop
axiom FamilyFriendlyAnimated : obj → Prop
axiom FrozenSeries : obj → Prop

axiom Hachi : obj

-- Premises
axiom P1 : ∀ x : obj, Appropriate x → CanWatchWithoutGuidance x
axiom P2 : ∀ x : obj, ExcessiveEroticViolentContent x → ¬ CanWatchWithoutGuidance x

axiom P7 : FamilyFriendlyAnimated Hachi ∨ Appropriate Hachi

-- Useful derived facts
theorem hachi_can_watch : CanWatchWithoutGuidance Hachi := by
  cases P7 with
  | inl hffa =>
      exact P1 Hachi (P3 Hachi (P4 Hachi hffa))
  | inr happ =>
      exact P1 Hachi happ

theorem hachi_not_excessive : ¬ ExcessiveEroticViolentContent Hachi := by
  intro hev
  have hnot := P2 Hachi hev
  exact hnot hachi_can_watch
