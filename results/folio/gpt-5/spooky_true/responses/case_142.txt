<lean>
axiom obj : Type

axiom Animal : obj → Prop
axiom HeckCattle : obj → Prop
axiom Aurochs : obj → Prop
axiom Extinct : obj → Prop
axiom ArtificiallySelected : obj → Prop
axiom BredBack : obj → Prop
axiom Resemble : obj → obj → Prop

-- Breeding back is a form of artificial selection
axiom R1 : ∀ x : obj, BredBack x → ArtificiallySelected x

-- Heck cattle were bred back (existence)
axiom R2 : ∃ x : obj, HeckCattle x ∧ BredBack x

-- Heck cattle are animals
axiom R3 : ∀ x : obj, HeckCattle x → Animal x

-- Aurochs are animals
axiom R4 : ∀ x : obj, Aurochs x → Animal x

-- Some animals to be bred back resemble extinct animals
axiom R5 : ∃ x y : obj, Animal x ∧ BredBack x ∧ Extinct y ∧ Resemble x y

theorem some_heck_cattle_artificially_selected : ∃ x : obj, HeckCattle x ∧ ArtificiallySelected x := by
  rcases R2 with ⟨x, hxHC, hxBB⟩
  have hxAS : ArtificiallySelected x := R1 x hxBB
  exact ⟨x, hxHC, hxAS⟩
</lean>

ANSWER: True
