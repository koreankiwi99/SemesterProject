<lean>
axiom obj : Type

axiom HeckCattle : obj
axiom Aurochs : obj

axiom Animal : obj → Prop
axiom ToBeBredBack : obj → Prop
axiom Resemble : obj → obj → Prop
axiom Extinct : obj → Prop

-- Given facts
axiom HC_animal : Animal HeckCattle
axiom Au_animal : Animal Aurochs
axiom HC_resemble_Au : Resemble HeckCattle Aurochs
axiom HC_toBeBredBack : ToBeBredBack HeckCattle

-- Interpreting that animals to be bred back resemble extinct animals
axiom rule_resemble_extinct : ∀ x y : obj, ToBeBredBack x → Resemble x y → Extinct y

theorem aurochs_extinct : Extinct Aurochs :=
  rule_resemble_extinct HeckCattle Aurochs HC_toBeBredBack HC_resemble_Au
</lean>

ANSWER: True
