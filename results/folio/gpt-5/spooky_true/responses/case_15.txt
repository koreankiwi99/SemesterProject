<lean>
axiom obj : Type
axiom Square : obj → Prop
axiom FourSided : obj → Prop
axiom Shape : obj → Prop

-- All squares are four-sided
axiom A1 : ∀ x : obj, Square x → FourSided x
-- All four-sided things are shapes
axiom A2 : ∀ x : obj, FourSided x → Shape x

-- Therefore, all squares are shapes
theorem squares_are_shapes : ∀ x : obj, Square x → Shape x := by
  intro x hx
  exact A2 x (A1 x hx)
</lean>

ANSWER: True
