<lean>
axiom obj : Type
axiom Human : obj → Prop
axiom Greek : obj → Prop
axiom Mortal : obj → Prop

-- Premises
axiom all_humans_mortal : ∀ x : obj, Human x → Mortal x
axiom all_greeks_human : ∀ x : obj, Greek x → Human x

-- Derivable consequence: all Greeks are mortal
theorem all_greeks_mortal : ∀ x : obj, Greek x → Mortal x := by
  intro x
  intro gx
  apply all_humans_mortal x
  apply all_greeks_human x
  exact gx

-- From existence of a Greek, we can derive existence of a mortal Greek.
-- But without assuming existence of a Greek, we cannot prove ∃ x, Greek x ∧ Mortal x.
theorem some_greek_mortal_of_exists : (∃ x : obj, Greek x) → ∃ x : obj, Greek x ∧ Mortal x := by
  intro h
  cases h with
  | intro g gg =>
    refine ⟨g, ?_⟩
    apply And.intro
    · exact gg
    · apply all_humans_mortal g
      apply all_greeks_human g
      exact gg
