<lean>
axiom obj : Type
axiom Student : obj → Prop
axiom WorkLib : obj → Prop
axiom FromCS : obj → Prop
axiom PT : obj → Prop
axiom TakingDB : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorkLab : obj → Prop

axiom James : obj

-- Premises
-- None of the students who currently work in the library are from the CS department.
axiom P1 : ∀ x : obj, Student x ∧ WorkLib x → ¬ FromCS x
-- All of the students with part-time jobs offered by the university currently work in the library.
axiom P2 : ∀ x : obj, Student x ∧ PT x → WorkLib x
-- All of the students who are taking the database course are from the CS department.
axiom P3 : ∀ x : obj, Student x ∧ TakingDB x → FromCS x
-- All of the students taking a class with Professor David are taking the database course.
axiom P4 : ∀ x : obj, Student x ∧ TakingWithDavid x → TakingDB x
-- All students working in the lab are taking a class with Professor David.
axiom P5 : ∀ x : obj, Student x ∧ WorkLab x → TakingWithDavid x
-- James is a student working in the lab.
axiom P6 : Student James ∧ WorkLab James
-- James doesn't work in the lab or have a part-time job offered by the university.
-- Interpreted as: not (WorkLab James or PT James)
axiom P7 : ¬ (WorkLab James ∨ PT James)

theorem james_has_PT : PT James := by
  have hLab : WorkLab James := And.right P6
  have hOr : WorkLab James ∨ PT James := Or.inl hLab
  have hFalse : False := P7 hOr
  exact False.elim hFalse
</lean>

ANSWER: True
