<lean>
axiom Person : Type
axiom Student : Person → Prop
axiom WorksLibrary : Person → Prop
axiom WorksLab : Person → Prop
axiom FromCS : Person → Prop
axiom HasPT : Person → Prop
axiom TakingDB : Person → Prop
axiom WithDavid : Person → Prop

-- Premises
axiom P1 : ∀ x : Person, Student x ∧ WorksLibrary x → ¬ FromCS x
axiom P2 : ∀ x : Person, Student x ∧ HasPT x → WorksLibrary x
axiom P3 : ∀ x : Person, Student x ∧ TakingDB x → FromCS x
axiom P4 : ∀ x : Person, Student x ∧ WithDavid x → TakingDB x
axiom P5 : ∀ x : Person, Student x ∧ WorksLab x → WithDavid x

axiom James : Person
axiom JamesStudent : Student James
axiom JamesLab : WorksLab James
axiom James_not_lab_or_pt : ¬ (WorksLab James ∨ HasPT James)  -- included but not used

-- Goal: James does not have a part-time job offered by the university.
theorem james_no_pt : ¬ HasPT James :=
by
  intro hPT
  have hLib : WorksLibrary James :=
    P2 James (And.intro JamesStudent hPT)
  have hWithDavid : WithDavid James :=
    P5 James (And.intro JamesStudent JamesLab)
  have hDB : TakingDB James :=
    P4 James (And.intro JamesStudent hWithDavid)
  have hCS : FromCS James :=
    P3 James (And.intro JamesStudent hDB)
  have hNotCS : ¬ FromCS James :=
    P1 James (And.intro JamesStudent hLib)
  exact hNotCS hCS
</lean>

ANSWER: True
