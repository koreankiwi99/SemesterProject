<lean>
axiom LikesMusic : Prop
axiom WantsCompose : Prop
axiom HasAccessProgram : Prop
axiom CanCompose : Prop
axiom WillCompose : Prop

axiom R1 : LikesMusic → WantsCompose
axiom R2 : HasAccessProgram → CanCompose
axiom R3 : CanCompose ∧ WantsCompose → WillCompose

theorem george_will_compose :
  (LikesMusic ∧ HasAccessProgram) → WillCompose :=
by
  intro h
  have hL : LikesMusic := And.left h
  have hA : HasAccessProgram := And.right h
  have hW : WantsCompose := R1 hL
  have hC : CanCompose := R2 hA
  have hCW : CanCompose ∧ WantsCompose := And.intro hC hW
  exact R3 hCW
</lean>

ANSWER: True
