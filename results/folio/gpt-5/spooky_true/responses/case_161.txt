  intro hnotwill
  have h : ¬ (CanCompose George ∧ WantsCompose George) := by
    intro hAnd
    apply hnotwill
    exact P3 hAnd

  by_cases hc : CanCompose George

      intro hw
      apply h
      exact And.intro hc hw
    exact Or.inr this
