axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swim : obj → Prop

-- Premises
axiom hawk_never_swims : ∀ x : obj, Hawk x → ¬ Swim x
axiom some_bird_is_hawk : ∃ x : obj, Bird x ∧ Hawk x

-- From the premises, we can derive that not all birds swim.
theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swim x) := by
  intro h_all
  obtain ⟨x, hx⟩ := some_bird_is_hawk
  have hx_bird : Bird x := And.left hx
  have hx_hawk : Hawk x := And.right hx
  have not_swim_x : ¬ Swim x := hawk_never_swims x hx_hawk
  have swim_x : Swim x := h_all x hx_bird
  exact not_swim_x swim_x

-- Also, there exists a bird that does not swim.
theorem exists_bird_not_swim : ∃ x : obj, Bird x ∧ ¬ Swim x := by
  obtain ⟨x, hx⟩ := some_bird_is_hawk
  have hx_bird : Bird x := And.left hx
  have hx_hawk : Hawk x := And.right hx
  have not_swim_x : ¬ Swim x := hawk_never_swims x hx_hawk
  exact Exists.intro x (And.intro hx_bird not_swim_x)
