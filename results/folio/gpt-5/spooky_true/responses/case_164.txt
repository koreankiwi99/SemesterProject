axiom AirsAt8pmMonday : obj → Prop
axiom GivesOutRosesOnTV : obj → Prop
axiom EpisodeOfTheBachelor : obj → Prop
axiom PortraysRealPeople : obj → Prop
axiom RealityTV : obj → Prop

axiom BreakingBad : obj

-- Premises
axiom P1 : ∃ s : obj, AirsAt8pmMonday s ∧ GivesOutRosesOnTV s
axiom P2 : ∀ s : obj, GivesOutRosesOnTV s → EpisodeOfTheBachelor s
axiom P3 : ∀ s : obj, EpisodeOfTheBachelor s → PortraysRealPeople s
axiom P4 : ∀ s : obj, PortraysRealPeople s → RealityTV s
axiom P5 : Show BreakingBad
axiom P6 : ¬ RealityTV BreakingBad

-- From P2-P4 and P6, we can show Breaking Bad does not give out roses on TV
theorem not_gives_BB : ¬ GivesOutRosesOnTV BreakingBad := by
  intro h
  have hE : EpisodeOfTheBachelor BreakingBad := P2 BreakingBad h
  have hP : PortraysRealPeople BreakingBad := P3 BreakingBad hE
  have hR : RealityTV BreakingBad := P4 BreakingBad hP
  exact P6 hR

def Statement : Prop := Show BreakingBad ∧ AirsAt8pmMonday BreakingBad ∧ GivesOutRosesOnTV BreakingBad

theorem not_Statement : ¬ Statement := by
  intro h
  have hG : GivesOutRosesOnTV BreakingBad := And.right (And.right h)
  exact not_gives_BB hG
