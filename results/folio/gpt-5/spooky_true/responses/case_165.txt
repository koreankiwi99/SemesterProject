<lean>
axiom obj : Type
axiom Show : obj → Prop
axiom GivesRoses : obj → Prop
axiom Bachelor : obj → Prop
axiom PortraysReal : obj → Prop
axiom RealityTV : obj → Prop
axiom AirsMon8 : obj → Prop
axiom BreakingBad : obj

-- Some show airing at 8 pm on Monday gives out roses on TV.
axiom A1 : ∃ s : obj, AirsMon8 s ∧ GivesRoses s
-- If a show gives out roses on TV, then the show is an episode of The Bachelor.
axiom A2 : ∀ s : obj, GivesRoses s → Bachelor s
-- The Bachelor portrays the lives of real people. (Episodes inherit this property.)
axiom A3 : ∀ s : obj, Bachelor s → PortraysReal s
-- All shows portraying the lives of real people are reality TV shows.
axiom A4 : ∀ s : obj, PortraysReal s → RealityTV s
-- Breaking Bad is a show.
axiom A5 : Show BreakingBad
-- Breaking Bad is not a reality TV show.
axiom A6 : ¬ RealityTV BreakingBad

-- Statement: If roses are given out during Breaking Bad, then it is on Monday at 8 pm.
theorem breakingbad_roses_implies_monday8 : GivesRoses BreakingBad → AirsMon8 BreakingBad := by
  intro hroses
  have hb : Bachelor BreakingBad := A2 BreakingBad hroses
  have hpr : PortraysReal BreakingBad := A3 BreakingBad hb
  have hrtv : RealityTV BreakingBad := A4 BreakingBad hpr
  have contra : False := A6 hrtv
  exact False.elim contra
</lean>

ANSWER: True
