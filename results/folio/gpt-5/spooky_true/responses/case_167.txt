<lean>
axiom obj : Type

axiom HeinrichSchmidt : obj
axiom GermanPolitician : obj → Prop
axiom InPrussianStateParliament : obj → Prop
axiom InNaziReichstag : obj → Prop

axiom HS_isGermanPolitician : GermanPolitician HeinrichSchmidt
axiom HS_inPSP : InPrussianStateParliament HeinrichSchmidt
axiom HS_inNR : InNaziReichstag HeinrichSchmidt

theorem some_german_politician_in_both :
  ∃ x : obj, GermanPolitician x ∧ InPrussianStateParliament x ∧ InNaziReichstag x :=
by
  apply Exists.intro HeinrichSchmidt
  apply And.intro HS_isGermanPolitician
  apply And.intro HS_inPSP HS_inNR
</lean>

ANSWER: True
