axiom obj : Type

axiom DeadlyDisease : obj → Prop
axiom LowSurvivalRate : obj → Prop
axiom SevereCancer : obj → Prop
axiom BileDuctCancer : obj → Prop
axiom Cholangiocarcinoma : obj → Prop
axiom MildFlu : obj → Prop

axiom ColorectalCancer : obj

-- Premises
axiom A1 : ∀ x : obj, DeadlyDisease x → LowSurvivalRate x
axiom A2 : ∀ x : obj, SevereCancer x → DeadlyDisease x
axiom A3 : ∀ x : obj, BileDuctCancer x → SevereCancer x
axiom A4 : ∀ x : obj, Cholangiocarcinoma x → BileDuctCancer x
axiom A5 : ∀ x : obj, MildFlu x → LowSurvivalRate x
axiom A6 : ¬ (BileDuctCancer ColorectalCancer ∧ LowSurvivalRate ColorectalCancer)

-- From the premises we can derive that any bile duct cancer has a low survival rate
theorem bile_duct_implies_low : ∀ x : obj, BileDuctCancer x → LowSurvivalRate x := by
  intro x
  intro hb
  have hs : SevereCancer x := A3 x hb
  have hd : DeadlyDisease x := A2 x hs
  exact A1 x hd

-- Hence colorectal cancer is not a bile duct cancer
theorem not_bile_colorectal : ¬ BileDuctCancer ColorectalCancer := by
  intro hb
  have hl : LowSurvivalRate ColorectalCancer := bile_duct_implies_low ColorectalCancer hb
  exact A6 (And.intro hb hl)

-- Therefore colorectal cancer is not a cholangiocarcinoma (since those are bile duct cancers)
theorem not_cholangio_colorectal : ¬ Cholangiocarcinoma ColorectalCancer := by
  intro hchol
  have hb : BileDuctCancer ColorectalCancer := A4 ColorectalCancer hchol
  exact not_bile_colorectal hb

-- This is false given the premises.

     ∨ (Cholangiocarcinoma ColorectalCancer ∧ MildFlu ColorectalCancer ∧ BileDuctCancer ColorectalCancer)) := by
  intro h
  cases h with

    exact not_cholangio_colorectal hchol
