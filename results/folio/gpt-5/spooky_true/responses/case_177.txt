<lean>
axiom obj : Type

-- Entities
axiom UnitedStates : obj
axiom Tokyo : obj

-- Predicates/relations
axiom SportingEvent : obj → Prop
axiom SummerOlympics : obj → Prop
axiom HeldInLastSOG : obj → Prop              -- The last Summer Olympics was in this city
axiom WonMostMedalsInCity : obj → obj → Prop  -- nation → city
axiom WonMostMedalsInLastSOG : obj → Prop     -- nation

-- Premises
axiom P1 : ∀ e : obj, SummerOlympics e → SportingEvent e
axiom P2 : HeldInLastSOG Tokyo
axiom P3 : WonMostMedalsInCity UnitedStates Tokyo
axiom Bridge :
  ∀ n c : obj,
    HeldInLastSOG c →
    WonMostMedalsInCity n c →
    WonMostMedalsInLastSOG n

-- Conclusion
theorem us_won_most_medals_last_sog : WonMostMedalsInLastSOG UnitedStates := by
  exact Bridge UnitedStates Tokyo P2 P3
</lean>

ANSWER: True
