axiom obj : Type
axiom Human : obj → Prop
axiom Horse : obj → Prop
axiom Hooves : obj → Prop

-- All horses have hooves.
axiom Hhorses_hooves : ∀ x : obj, Horse x → Hooves x
-- No humans have hooves.
axiom Hhumans_no_hooves : ∀ x : obj, Human x → ¬ Hooves x

-- From the premises, no human can be a horse.
theorem no_human_is_horse : ¬ ∃ x : obj, Human x ∧ Horse x := by
  intro h
  cases h with
  | intro x hx =>
    have hxHuman : Human x := And.left hx
    have hxHorse : Horse x := And.right hx
    have hxHooves : Hooves x := Hhorses_hooves x hxHorse
    have hxNoHooves : ¬ Hooves x := Hhumans_no_hooves x hxHuman
    exact hxNoHooves hxHooves
