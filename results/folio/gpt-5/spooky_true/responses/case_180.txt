axiom Airport : Type

axiom FliesTo : Person → Airport → Prop

-- The departure and arrival cannot be at the same airport
axiom sep_same_airport : ∀ p a, FliesFrom p a → FliesTo p a → False

-- Susan flies to LGA airport
axiom susan_to_LGA : FliesTo Susan LGA

-- John flies from LGA airport
axiom john_from_LGA : FliesFrom John LGA

-- From the constraint, John cannot also fly to LGA
theorem not_john_to_LGA : ¬ FliesTo John LGA := by
  intro h
  have hcontra := sep_same_airport John LGA john_from_LGA h
  exact hcontra
