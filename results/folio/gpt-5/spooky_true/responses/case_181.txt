by
  intro x hx
  have hAL : AnimalLover x := P4 x hx
  have hNice : NiceToAnimals x := P3 x hAL
  exact P1 x hNice

by
  have hAL : AnimalLover Tom := P4 Tom P5
  exact P3 Tom hAL

theorem tom_not_mean : ¬ MeanToAnimals Tom :=
by
  have hNice : NiceToAnimals Tom := tom_nice
