<lean>
axiom obj : Type
axiom Tom : obj

axiom Nice : obj → Prop
axiom Mean : obj → Prop
axiom Grumpy : obj → Prop
axiom LovesAnimals : obj → Prop
axiom PetOwner : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Nice x → ¬ Mean x
axiom P2 : ∃ x : obj, Grumpy x ∧ Mean x
axiom P3 : ∀ x : obj, LovesAnimals x → Nice x
axiom P4 : ∀ x : obj, PetOwner x → LovesAnimals x
axiom P5 : PetOwner Tom

-- Conclusion: Tom is not both grumpy and mean to animals.
theorem tom_not_grumpy_and_mean : ¬ (Grumpy Tom ∧ Mean Tom) :=
by
  have hNotMean : ¬ Mean Tom := P1 Tom (P3 Tom (P4 Tom P5))
  intro h
  exact hNotMean h.right
</lean>

ANSWER: True
