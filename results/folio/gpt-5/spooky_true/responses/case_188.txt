<lean>
axiom obj : Type
axiom Yuri : obj

axiom American : obj → Prop
axiom ProBB : obj → Prop
axiom CanJump : obj → Prop
axiom Leap : obj → Prop
axiom ActivateLegs : obj → Prop

-- Some professional basketball players are not American nationals.
axiom some_nonAmerican : ∃ x : obj, ProBB x ∧ ¬ American x

-- All professional basketball players can do jump shots.
axiom all_pro_canJump : ∀ x : obj, ProBB x → CanJump x

-- If someone can jump shots, they leap straight into the air.
axiom canJump_implies_leap : ∀ x : obj, CanJump x → Leap x

-- If someone leaps straight into the air, they activate their leg muscles.
axiom leap_implies_activate : ∀ x : obj, Leap x → ActivateLegs x

-- Yuri does not activate his leg muscles.
axiom yuri_not_activate : ¬ ActivateLegs Yuri

theorem pro_implies_activate : ∀ x : obj, ProBB x → ActivateLegs x := by
  intro x
  intro hPro
  have hJump : CanJump x := all_pro_canJump x hPro
  have hLeap : Leap x := canJump_implies_leap x hJump
  have hAct : ActivateLegs x := leap_implies_activate x hLeap
  exact hAct

theorem yuri_not_pro : ¬ ProBB Yuri := by
  intro hPro
  have hAct : ActivateLegs Yuri := pro_implies_activate Yuri hPro
  exact yuri_not_activate hAct

theorem yuri_not_american_professional : ¬ (American Yuri ∧ ProBB Yuri) := by
  intro h
  cases h with
  | intro hAm hPro =>
    exact yuri_not_pro hPro
</lean>

ANSWER: True
