-- Premise 2: At least one event is happy.
axiom P2 : ∃ x : obj, Happy x

-- The queried statement "All events are sad" does not follow; in fact, we can refute it.

  have h := P1 e

  have heSad : Sad e := hAllSad e
  exact hNotBoth (And.intro heHappy heSad)
