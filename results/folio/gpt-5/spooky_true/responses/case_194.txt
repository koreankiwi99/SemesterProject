axiom Descampe : obj

axiom FromBelgium : obj → Prop
axiom OnLeaderboard : obj → Prop
axiom Participated : obj → Prop

-- Premises
axiom A1 : Winner Steinhauer
axiom A2 : Participated Steinhauer
-- "One six-way tie was on the leaderboard, and one person in the six-way tie was from Belgium."
-- We formalize the relevant part as existence of a Belgian in the six-way tie.
axiom A3 : ∃ x : obj, InSixWayTie x ∧ FromBelgium x
-- "Descampe is from Belgium and is on the leaderboard of the 1992 du Maurier Classic."
axiom A4a : FromBelgium Descampe
axiom A4b : OnLeaderboard Descampe
-- "All people on the leaderboard of the 1992 du Maurier Classic participated in the 1992 du Maurier Classic."
axiom A5 : ∀ x : obj, OnLeaderboard x → Participated x
-- "One six-way tie was on the leaderboard" (tie members are on the leaderboard)
axiom A6 : ∀ x : obj, InSixWayTie x → OnLeaderboard x

-- A derived fact (not the target): there exists a Belgian in the six-way tie who participated.
theorem belgian_tie_member_participated : ∃ x : obj, InSixWayTie x ∧ FromBelgium x ∧ Participated x := by
  rcases A3 with ⟨w, hwTie, hwBel⟩
  have hwOL : OnLeaderboard w := A6 w hwTie
  have hwPart : Participated w := A5 w hwOL
  exact ⟨w, hwTie, hwBel, hwPart⟩
