  have hN : Near Rockie := And.left A6
  have hCal : Calm Rockie := And.right A6

  have notS : ¬ Squirrel Rockie := by
    intro hS
    have hSk : Skittish Rockie := A5 Rockie (And.intro hS hN)
    have hNotCal : ¬ Calm Rockie := A4 Rockie hSk
