<lean>
axiom obj : Type

-- Entities
axiom AiltonSilva : obj
axiom Ailton : obj
axiom Braga : obj
axiom Nautico : obj
axiom Fluminense : obj

-- Predicates/Relations
axiom FootballClub : obj → Prop
axiom LoanedOutTo : obj → obj → Prop

-- Facts from the context
axiom alias_AiltonSilva_Ailton : AiltonSilva = Ailton
axiom loan_Ailton_Braga : LoanedOutTo Ailton Braga
axiom club_Braga : FootballClub Braga
axiom club_Nautico : FootballClub Nautico
axiom club_Fluminense : FootballClub Fluminense

-- Goal: Ailton Silva was loaned out to a football club (existential statement)
theorem AiltonSilva_loaned_to_a_football_club : ∃ c : obj, FootballClub c ∧ LoanedOutTo AiltonSilva c := by
  have hloan : LoanedOutTo AiltonSilva Braga := by
    simpa [alias_AiltonSilva_Ailton] using loan_Ailton_Braga
  exact ⟨Braga, ⟨club_Braga, hloan⟩⟩
</lean>

ANSWER: True
