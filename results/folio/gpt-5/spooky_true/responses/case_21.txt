<lean>
axiom obj : Type
axiom ST : obj
axiom BM : obj

axiom NetflixShow : obj → Prop
axiom Popular : obj → Prop
axiom Binge : obj → Prop            -- Karen binge-watches the show
axiom Download : obj → Prop         -- Karen downloads the show
axiom ShareWithLisa : obj → Prop    -- Karen shares the show with Lisa

-- Premises
axiom ST_netflix : NetflixShow ST
axiom ST_popular : Popular ST

axiom binge_if_popular_netflix : ∀ x : obj, NetflixShow x → Popular x → Binge x
axiom binge_iff_download_on_netflix : ∀ x : obj, NetflixShow x → (Binge x ↔ Download x)
axiom BM_not_download : ¬ Download BM
axiom BM_netflix : NetflixShow BM
axiom share_if_binge_netflix : ∀ x : obj, NetflixShow x → Binge x → ShareWithLisa x

-- Goal: Karen will share "Stranger Things" with Lisa.
theorem karen_shares_stranger_things : ShareWithLisa ST := by
  have hBinge : Binge ST := binge_if_popular_netflix ST ST_netflix ST_popular
  exact share_if_binge_netflix ST ST_netflix hBinge
</lean>

ANSWER: True
