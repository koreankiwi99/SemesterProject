<lean>
axiom obj : Type
axiom Event : Type

axiom Beijing : obj
axiom Hosted : obj → Event → Prop

axiom Summer2008Olympics : Event
axiom WinterOlympics : Event

-- From the text: "Beijing hosted the 2008 Summer Olympics and 2008 Summer Paralympics Games."
axiom H1 : Hosted Beijing Summer2008Olympics

-- From the text: "Beijing has hosted the Summer and Winter Olympics and the Summer and Winter Paralympics."
axiom H2 : Hosted Beijing WinterOlympics

theorem beijing_hosted_2008_and_winter :
  Hosted Beijing Summer2008Olympics ∧ Hosted Beijing WinterOlympics := by
  constructor
  · exact H1
  · exact H2
</lean>

ANSWER: True
