<lean>
axiom obj : Type
axiom James : obj

axiom Employee : obj → Prop
axiom Manager : obj → Prop
axiom SchedulesMeetingWithCustomers : obj → Prop
axiom InCompanyToday : obj → Prop
axiom LunchInCompany : obj → Prop
axiom LunchAtHome : obj → Prop
axiom RemoteFromHome : obj → Prop
axiom InOtherCountries : obj → Prop

-- 1. Employees who schedule a meeting with their customers will go to the company building today.
axiom A1 : ∀ x : obj, Employee x → SchedulesMeetingWithCustomers x → InCompanyToday x
-- 2. Everyone who has lunch in the company building schedules meetings with their customers.
axiom A2 : ∀ x : obj, LunchInCompany x → SchedulesMeetingWithCustomers x
-- 3. Employees have lunch either in the company building or at home.
axiom A3 : ∀ x : obj, Employee x → (LunchInCompany x ∨ LunchAtHome x)
-- 4. If an employee has lunch at home, they are working remotely from home.
axiom A4 : ∀ x : obj, Employee x → LunchAtHome x → RemoteFromHome x
-- 5. All employees who are in other countries work remotely from home.
axiom A5 : ∀ x : obj, Employee x → InOtherCountries x → RemoteFromHome x
-- 6. No managers work remotely from home.
axiom A6 : ∀ x : obj, Manager x → ¬ RemoteFromHome x
-- 7. James will appear in the company today iff he is a manager.
axiom A7 : InCompanyToday James ↔ Manager James

-- Derived: any employee who is a manager must have lunch in the company (cannot have lunch at home, which would force remote work).
theorem manager_employee_lunch_in_company :
  ∀ x : obj, Employee x → Manager x → LunchInCompany x := by
  intro x hxEmp hxMan
  have hdisj : LunchInCompany x ∨ LunchAtHome x := A3 x hxEmp
  cases hdisj with
  | inl hLunchCompany =>
      exact hLunchCompany
  | inr hLunchHome =>
      have hRemote : RemoteFromHome x := A4 x hxEmp hLunchHome
      have hNotRemote : ¬ RemoteFromHome x := A6 x hxMan
      exact False.elim (hNotRemote hRemote)
</lean>
