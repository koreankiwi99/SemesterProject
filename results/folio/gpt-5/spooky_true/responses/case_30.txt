axiom AtMixer : obj → Prop
axiom GrandSlamChampion : obj → Prop

axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom LivesInTaxHaven : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, AtMixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)
axiom P2 : ∀ x : obj, AtMixer x → GrandSlamChampion x → ProfessionalTennisPlayer x
axiom P3 : ∀ x : obj, AtMixer x → OscarNominatedActor x → Celebrity x
axiom P4 : ∀ x : obj, AtMixer x → ProfessionalTennisPlayer x → Athlete x
axiom P5 : ∀ x : obj, AtMixer x → Celebrity x → WellPaid x
axiom P6 : ∀ x : obj, AtMixer x → Athlete x → Famous x
axiom P7 : ∀ x : obj, AtMixer x → WellPaid x → LivesInTaxHaven x

-- Djokovic-specific facts
axiom D_at : AtMixer Djokovic
axiom D_rule : Famous Djokovic ∧ Athlete Djokovic → WellPaid Djokovic

-- Some derivable consequences (but not the target conclusion)
theorem Djokovic_wellpaid_if_GSC : GrandSlamChampion Djokovic → WellPaid Djokovic :=
fun hGSC => by
  have hPTP : ProfessionalTennisPlayer Djokovic := P2 Djokovic D_at hGSC
  have hAth : Athlete Djokovic := P4 Djokovic D_at hPTP
  have hFam : Famous Djokovic := P6 Djokovic D_at hAth
  exact D_rule (And.intro hFam hAth)

theorem Djokovic_taxhaven_if_GSC : GrandSlamChampion Djokovic → LivesInTaxHaven Djokovic :=
fun hGSC => by
  have hWP : WellPaid Djokovic := Djokovic_wellpaid_if_GSC hGSC
  exact P7 Djokovic D_at hWP

-- From the general disjunction we only get: GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic
theorem Djokovic_disj : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic :=
