<lean>
axiom obj : Type

axiom At : obj → Prop
axiom GSC : obj → Prop
axiom ONA : obj → Prop
axiom PTP : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom LivesTaxHaven : obj → Prop

axiom Djokovic : obj

-- Premises
axiom A1 : ∀ x : obj, At x → (GSC x ∨ ONA x)
axiom A2 : ∀ x : obj, At x → GSC x → PTP x
axiom A3 : ∀ x : obj, At x → ONA x → Celebrity x
axiom A4 : ∀ x : obj, At x → PTP x → Athlete x
axiom A5 : ∀ x : obj, At x → Celebrity x → WellPaid x
axiom A6 : ∀ x : obj, At x → Athlete x → Famous x
axiom A7 : ∀ x : obj, At x → WellPaid x → LivesTaxHaven x
axiom A8_At : At Djokovic
axiom A8_cond : Famous Djokovic ∧ Athlete Djokovic → WellPaid Djokovic

theorem djokovic_tax_haven : LivesTaxHaven Djokovic := by
  have atD : At Djokovic := A8_At
  have disj : GSC Djokovic ∨ ONA Djokovic := A1 Djokovic atD
  cases disj with
  | inl gsc =>
    have ptp : PTP Djokovic := A2 Djokovic atD gsc
    have ath : Athlete Djokovic := A4 Djokovic atD ptp
    have fam : Famous Djokovic := A6 Djokovic atD ath
    have wp : WellPaid Djokovic := A8_cond ⟨fam, ath⟩
    exact A7 Djokovic atD wp
  | inr ona =>
    have celeb : Celebrity Djokovic := A3 Djokovic atD ona
    have wp : WellPaid Djokovic := A5 Djokovic atD celeb
    exact A7 Djokovic atD wp
</lean>

ANSWER: True
