axiom obj : Type
axiom At : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom LivesInTaxHaven : obj → Prop

axiom Djokovic : obj

-- Everyone at the mixer is a Grand Slam champion or an Oscar-nominated actor.
axiom A1 : ∀ x : obj, At x → (GrandSlamChampion x ∨ OscarNominatedActor x)
-- Every Grand Slam champion at the mixer is a professional tennis player.
axiom A2 : ∀ x : obj, At x ∧ GrandSlamChampion x → ProTennisPlayer x
-- All Oscar-nominated actors at the mixer are celebrities.
axiom A3 : ∀ x : obj, At x ∧ OscarNominatedActor x → Celebrity x
-- All professional tennis players at the mixer are athletes.
axiom A4 : ∀ x : obj, At x ∧ ProTennisPlayer x → Athlete x
-- If a person at the mixer is a celebrity, then they are well paid.
axiom A5 : ∀ x : obj, At x ∧ Celebrity x → WellPaid x
-- If a person at the mixer is an athlete, then they are famous.
axiom A6 : ∀ x : obj, At x ∧ Athlete x → Famous x
-- All well-paid people at the mixer live in tax havens.
axiom A7 : ∀ x : obj, At x ∧ WellPaid x → LivesInTaxHaven x
-- Djokovic is at the mixer
axiom A8 : At Djokovic
-- If Djokovic is a famous athlete, then Djokovic is well-paid.
axiom A9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

-- From the premises, Djokovic lives in a tax haven.
theorem djokovic_lives_in_tax_haven : LivesInTaxHaven Djokovic := by
  have hAt : At Djokovic := A8
  have hdisj : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := A1 Djokovic hAt
  cases hdisj with
  | inl hGS =>
      have hPT : ProTennisPlayer Djokovic := A2 Djokovic ⟨hAt, hGS⟩
      have hAth : Athlete Djokovic := A4 Djokovic ⟨hAt, hPT⟩
      have hFam : Famous Djokovic := A6 Djokovic ⟨hAt, hAth⟩
      have hWP : WellPaid Djokovic := A9 ⟨hFam, hAth⟩
      exact A7 Djokovic ⟨hAt, hWP⟩
  | inr hOscar =>
      have hCel : Celebrity Djokovic := A3 Djokovic ⟨hAt, hOscar⟩
      have hWP : WellPaid Djokovic := A5 Djokovic ⟨hAt, hCel⟩
      exact A7 Djokovic ⟨hAt, hWP⟩
