<lean>
axiom obj : Type

-- Entities
axiom Symphony9 : obj
axiom ViennaMusicSociety : obj
axiom Beethoven : obj

-- Predicates and relations
axiom MusicPiece : obj → Prop
axiom Orchestra : obj → Prop
axiom Premiered : obj → obj → Prop
axiom Leads : obj → obj → Prop
axiom Composer : obj → Prop
axiom Conductor : obj → Prop
axiom Wrote : obj → obj → Prop

-- Facts from the context
axiom sym9_is_music : MusicPiece Symphony9
axiom vms_premiered_sym9 : Premiered ViennaMusicSociety Symphony9
axiom vms_is_orchestra : Orchestra ViennaMusicSociety
axiom beethoven_leads_vms : Leads Beethoven ViennaMusicSociety

-- Optional general rule from the context (not needed for the main proof)
axiom orchestras_led_by_conductors :
  ∀ (o l : obj), Orchestra o ∧ Leads l o → Conductor l

-- Optional specific fact from the context (not needed for the main proof)
axiom beethoven_wrote_sym9 : Wrote Beethoven Symphony9

-- Target: Some orchestras premiered music pieces.
theorem some_orchestra_premiered :
  ∃ (o m : obj), Orchestra o ∧ MusicPiece m ∧ Premiered o m := by
  refine Exists.intro ViennaMusicSociety ?_
  refine Exists.intro Symphony9 ?_
  exact And.intro vms_is_orchestra (And.intro sym9_is_music vms_premiered_sym9)
</lean>

ANSWER: True
