axiom obj : Type

-- Predicates
axiom ByMax : obj → Prop
axiom Adores : obj → Prop
axiom InterestingGeom : obj → Prop
axiom Brutalist : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop
axiom ZahaStyle : obj → Prop
axiom KellyStyle : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, ZahaStyle x ∧ Adores x → InterestingGeom x
axiom P2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ InterestingGeom x
axiom P3 : ∀ x : obj, Adores x → ZahaStyle x ∨ KellyStyle x
axiom P4 : ∀ x : obj, KellyStyle x ∧ Adores x → Evocative x
axiom P5 : ∀ x : obj, KellyStyle x ∧ Adores x → Dreamy x
axiom P6 : ∀ x : obj, ByMax x ∧ Adores x ∧ InterestingGeom x → Brutalist x ∧ Evocative x

-- A consequence we can derive (but it does not yield the target existential conclusion):
-- No design by Max that he adores can have interesting geometries.
theorem no_bymax_adored_with_IG : ∀ x : obj, ByMax x ∧ Adores x ∧ InterestingGeom x → False := by
  intro x h
  have hb : Brutalist x := And.left (P6 x h)
  have hA : Adores x := And.left (And.right h)
  have hnotIG : ¬ InterestingGeom x := P2 x (And.intro hb hA)
  exact hnotIG (And.right (And.right h))

-- Note: From the given premises, we cannot derive that there exists some object that is both a design by Max and a brutalist building:
-- we cannot prove ∃ x, ByMax x ∧ Brutalist x.
