<lean>
axiom obj : Type

-- Predicates over objects (which may be design styles or designs)
axiom ZahaStyle : obj → Prop
axiom KellyStyle : obj → Prop
axiom Brutalist : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop
axiom DesignByMax : obj → Prop
axiom AdoresMax : obj → Prop

-- Premises
-- 1. All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom P1 : ∀ x : obj, ZahaStyle x ∧ AdoresMax x → Interesting x

-- 2. No brutalist buildings that Max adores have interesting geometries.
axiom P2 : ∀ x : obj, Brutalist x ∧ AdoresMax x → ¬ Interesting x

-- 3. Every style that Max adores is either Zaha or Kelly.
axiom P3 : ∀ x : obj, AdoresMax x → ZahaStyle x ∨ KellyStyle x

-- 4. All Kelly Wearstler styles that Max adores are evocative.
axiom P4 : ∀ x : obj, KellyStyle x ∧ AdoresMax x → Evocative x

-- 5. All Kelly Wearstler styles that Max adores are dreamy.
axiom P5 : ∀ x : obj, KellyStyle x ∧ AdoresMax x → Dreamy x

-- 6. If a design by Max that he adores has interesting geometries, then it is brutalist and evocative.
axiom P6 : ∀ x : obj, DesignByMax x ∧ AdoresMax x ∧ Interesting x → Brutalist x ∧ Evocative x

-- Useful consequences of the premises

-- No design by Max that he adores can be interesting (contradiction via P6 and P2).
theorem not_interesting_if_design_adores : ∀ x : obj, DesignByMax x ∧ AdoresMax x → ¬ Interesting x := by
  intro x hDA
  intro hI
  have hBE : Brutalist x ∧ Evocative x :=
    P6 x (And.intro hDA.left (And.intro hDA.right hI))
  have hB : Brutalist x := hBE.left
  have hNotI : ¬ Interesting x := P2 x (And.intro hB hDA.right)
  exact hNotI hI

-- Kelly styles adored by Max are both evocative and dreamy.
theorem kelly_adored_evocative_dreamy : ∀ x : obj, KellyStyle x ∧ AdoresMax x → Evocative x ∧ Dreamy x := by
  intro x h
  have hE : Evocative x := P4 x h
  have hD : Dreamy x := P5 x h
  exact And.intro hE hD
