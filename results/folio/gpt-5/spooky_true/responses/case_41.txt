<lean>
axiom obj : Type

axiom MaxDesign : obj → Prop
axiom Adores : obj → Prop
axiom ZahaStyle : obj → Prop
axiom KellyStyle : obj → Prop
axiom Brutalist : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

axiom P1 : ∀ x : obj, ZahaStyle x ∧ Adores x → Interesting x
axiom P2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ Interesting x
axiom P3 : ∀ x : obj, Adores x → ZahaStyle x ∨ KellyStyle x
axiom P4 : ∀ x : obj, KellyStyle x ∧ Adores x → Evocative x
axiom P5 : ∀ x : obj, KellyStyle x ∧ Adores x → Dreamy x
axiom P6 : ∀ x : obj, MaxDesign x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

theorem max_design_evocative_or_dreamy :
  ∀ x : obj, MaxDesign x ∧ Adores x → Evocative x ∨ Dreamy x :=
by
  intro x
  intro h
  have hM : MaxDesign x := And.left h
  have hA : Adores x := And.right h
  have hZA : ZahaStyle x ∨ KellyStyle x := P3 x hA
  cases hZA with
  | inl hZ =>
      have hI : Interesting x := P1 x (And.intro hZ hA)
      have hBE : Brutalist x ∧ Evocative x := P6 x (And.intro hM (And.intro hA hI))
      exact Or.inl (And.right hBE)
  | inr hK =>
      have hE : Evocative x := P4 x (And.intro hK hA)
      exact Or.inl hE
</lean>

ANSWER: True
