-- From the premises we can actually derive that Coco both lost to Iga and is active.
-- This shows the queried statement (that Coco is not both) does not follow.
theorem coco_not_male : ¬ MaleRG22 Coco := by
  intro hm
  have hln : LostToNadal Coco := P5 Coco hm
  have hdisj : Ranked Coco ∨ LostToNadal Coco := Or.inr hln
  exact (P6 hdisj) hm

theorem coco_female : FemaleRG22 Coco := by
  have h : FemaleRG22 Coco ∨ MaleRG22 Coco := P4 Coco P7
  cases h with
  | inl hf => exact hf
  | inr hm => exact False.elim (coco_not_male hm)

theorem coco_lost_iga_and_active : LostToIgaRG22 Coco ∧ Active Coco := by

  have hR : Ranked Coco := P2 Coco hI
  have hA : Active Coco := P1 Coco hR
