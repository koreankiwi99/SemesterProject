<lean>
inductive Season where
| Spring : Season
| Summer : Season
| Fall   : Season
| Winter : Season

axiom obj : Type
axiom Emma : obj
axiom Mia : obj
axiom James : obj

axiom Favorite : obj → Season
axiom WantsLongVacation : obj → Prop

-- All students who want a long vacation have summer as their favorite.
axiom R1 : ∀ x : obj, WantsLongVacation x → Favorite x = Season.Summer

-- Emma's favorite season is summer.
axiom E1 : Favorite Emma = Season.Summer

-- Mia's favorite season is not the same as Emma's.
axiom M1 : Favorite Mia ≠ Favorite Emma

-- James wants to have a long vacation.
axiom J1 : WantsLongVacation James

-- Conclusion: James's favorite season is summer.
theorem james_favorite_is_summer : Favorite James = Season.Summer := by
  exact R1 James J1
</lean>

ANSWER: True
