<lean>
axiom obj : Type
axiom NineteenEightyFour : obj

axiom Digital : obj → Prop
axiom Analog : obj → Prop
axiom PrintedText : obj → Prop
axiom StreamingService : obj → Prop
axiom HardcoverBook : obj → Prop

axiom H1 : ∀ x : obj, Digital x → ¬ Analog x
axiom H2 : ∀ x : obj, PrintedText x → Analog x
axiom H3 : ∀ x : obj, StreamingService x → Digital x
axiom H4 : ∀ x : obj, HardcoverBook x → PrintedText x
axiom H5 : StreamingService NineteenEightyFour → HardcoverBook NineteenEightyFour

theorem not_streaming_1984 : ¬ StreamingService NineteenEightyFour :=
  fun hS =>
    let hD := H3 NineteenEightyFour hS
    let hNotA := H1 NineteenEightyFour hD
    let hHB := H5 hS
    let hPT := H4 NineteenEightyFour hHB
    let hA := H2 NineteenEightyFour hPT
    show False from hNotA hA
</lean>

ANSWER: True
