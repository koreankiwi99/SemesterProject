<lean>
axiom obj : Type
axiom RL : obj
axiom BayernMunchen : obj

axiom Striker : obj → Prop
axiom SoccerPlayer : obj → Prop
axiom PlaysFor : obj → obj → Prop
axiom LeftTeam : obj → obj → Prop

-- Premises
axiom P1 : Striker RL
axiom P2 : ∀ x : obj, Striker x → SoccerPlayer x
axiom P3 : LeftTeam RL BayernMunchen
axiom P4 : ∀ x t : obj, LeftTeam x t → ¬ PlaysFor x t

-- Conclusion to prove: Robert Lewandowski is a soccer player.
theorem RL_is_soccer_player : SoccerPlayer RL := by
  exact P2 RL P1
</lean>

ANSWER: True
