axiom City : Type
axiom State : Type

-- Cities
axiom Billings : City
axiom Butte : City
axiom Helena : City
axiom Missoula : City

axiom StPierre : City

-- State
axiom Montana : State

-- Relation: a city is in a state
axiom InState : City → State → Prop

-- Same-state relation between cities
def SameState (c d : City) : Prop := ∃ s : State, InState c s ∧ InState d s

-- Premises
axiom P1 : InState Billings Montana
axiom P2a : InState Butte Montana
axiom P2b : InState Helena Montana
axiom P2c : InState Missoula Montana
axiom P3 : ∃ s : State, InState WhiteSulphurSprings s ∧ InState Butte s
axiom P4 : ¬ InState StPierre Montana

-- "Any city in Butte is not in St Pierre."
-- Interpreted as: any city that shares a state with Butte does not share a state with St Pierre.
axiom P5 : ∀ c : City, SameState c Butte → ¬ SameState c StPierre

-- Uniqueness of state for cities except listed exceptions
axiom UniqueStateExcept :
  ∀ c : City,
    c ≠ Bristol ∧ c ≠ Texarkana ∧ c ≠ Texhoma ∧ c ≠ UnionCity →
    ∀ s1 s2 : State, InState c s1 → InState c s2 → s1 = s2

-- From P5 and the fact that Butte shares a state with itself, we derive that
-- Butte and St Pierre are not in the same state.
theorem butte_not_same_stpierre : ¬ SameState Butte StPierre := by
  have hSame : SameState Butte Butte := by
    refine ⟨Montana, And.intro P2a P2a⟩
  exact P5 Butte hSame
