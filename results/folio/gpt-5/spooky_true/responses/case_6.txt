<lean>
axiom obj : Type

axiom Animal : obj → Prop
axiom Mammal : obj → Prop
axiom Human : obj → Prop

axiom MPXOccurs : obj → Prop
axiom HasMPX : obj → Prop

axiom Fever : obj → Prop
axiom Headache : obj → Prop
axiom MusclePains : obj → Prop
axiom Tired : obj → Prop
axiom HasFlu : obj → Prop

-- When the Monkeypox virus occurs in a being, it may get Monkeypox.
axiom P1 : ∀ x : obj, MPXOccurs x → HasMPX x
-- Monkeypox virus can occur in certain animals (existential reading).
axiom P2 : ∃ x : obj, Animal x ∧ MPXOccurs x
-- Humans are mammals.
axiom P3 : ∀ x : obj, Human x → Mammal x
-- Mammals are animals.
axiom P4 : ∀ x : obj, Mammal x → Animal x
-- Symptoms of Monkeypox include fever, headache, muscle pains, and tiredness.
axiom P5 : ∀ x : obj, HasMPX x → Fever x ∧ Headache x ∧ MusclePains x ∧ Tired x
-- People feel tired when they get the flu.
axiom P6 : ∀ x : obj, HasFlu x → Tired x

theorem there_is_an_animal : ∃ x : obj, Animal x := by
  have ⟨a, ha⟩ := P2
  exact ⟨a, And.left ha⟩
</lean>

ANSWER: True
