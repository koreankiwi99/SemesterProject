axiom Pet : obj → Prop
axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop

axiom Ted : obj

-- Premises
axiom A1 : ∀ x : obj, BrownSwissCattle x → Cow x                 -- All Brown Swiss cattle are cows.
axiom A2 : ∃ x : obj, Pet x ∧ BrownSwissCattle x                  -- Some pets are Brown Swiss cattle.
axiom A3 : ∀ x : obj, Cow x → Domesticated x                      -- All cows are domesticated animals.
axiom A4 : ∀ x : obj, Alligator x → ¬ Domesticated x              -- Alligators are not domesticated animals.
axiom A5 : Alligator Ted                                          -- Ted is an alligator.

-- Consequences: Ted cannot be a cow, hence cannot be Brown Swiss cattle.
theorem not_cow_Ted : ¬ Cow Ted := by
  intro hCow
  have hDom : Domesticated Ted := A3 Ted hCow
  have hNotDom : ¬ Domesticated Ted := A4 Ted A5
  exact hNotDom hDom

theorem not_BSC_Ted : ¬ BrownSwissCattle Ted := by
  intro hBSC
  have hCow : Cow Ted := A1 Ted hBSC
  exact not_cow_Ted hCow

-- Therefore, the target statement is impossible.
theorem not_target : ¬ (Pet Ted ∧ BrownSwissCattle Ted) := by
  intro h
  have hBSC : BrownSwissCattle Ted := And.right h
  exact not_BSC_Ted hBSC
