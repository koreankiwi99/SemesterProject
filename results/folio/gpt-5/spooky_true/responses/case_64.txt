<lean>
axiom obj : Type

axiom BrownSwiss : obj → Prop
axiom Cow : obj → Prop
axiom Pet : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop

axiom Ted : obj

-- All Brown Swiss cattle are cows.
axiom A1 : ∀ x : obj, BrownSwiss x → Cow x
-- All cows are domesticated animals.
axiom A2 : ∀ x : obj, Cow x → Domesticated x
-- Alligators are not domesticated animals.
axiom A3 : ∀ x : obj, Alligator x → ¬ Domesticated x
-- Ted is an alligator.
axiom A4 : Alligator Ted
-- Some pets are Brown Swiss Cattle.
axiom A5 : ∃ x : obj, Pet x ∧ BrownSwiss x

theorem ted_not_pet_if_brown_swiss : BrownSwiss Ted → ¬ Pet Ted := by
  intro hbs
  intro hpet
  have hc : Cow Ted := A1 Ted hbs
  have hd : Domesticated Ted := A2 Ted hc
  have hnd : ¬ Domesticated Ted := A3 Ted A4
  exact hnd hd
</lean>

ANSWER: True
