<lean>
axiom obj : Type
axiom Yale : obj

axiom PrivateIvyLeagueResearchUniversity : obj → Prop
axiom MovedToNewHaven : obj → Prop
axiom MovedToNewHavenIn : obj → Nat → Prop

axiom Yale_is_PIRU : PrivateIvyLeagueResearchUniversity Yale
axiom Yale_moved_1716 : MovedToNewHavenIn Yale 1716
axiom move_time_forget : ∀ x n, MovedToNewHavenIn x n → MovedToNewHaven x

theorem exists_private_ivy_moved :
  ∃ x, PrivateIvyLeagueResearchUniversity x ∧ MovedToNewHaven x := by
  have h1 : PrivateIvyLeagueResearchUniversity Yale := Yale_is_PIRU
  have h2 : MovedToNewHaven Yale := move_time_forget Yale 1716 Yale_moved_1716
  exact Exists.intro Yale (And.intro h1 h2)
</lean>

ANSWER: True
