<lean>
axiom Series : Type
axiom Person : Type

axiom Badults : Series
axiom AndrewCollins : Person

-- Predicates
axiom WorkingTitleSDS : Series → Prop
axiom ScriptEditor : Person → Series → Prop

-- Facts from the context
axiom WT_Badults : WorkingTitleSDS Badults
axiom SE_Collins_Badults : ScriptEditor AndrewCollins Badults

-- Claim: There exists a series with the working title "The Secret Dude Society"
-- for which Andrew Collins was the script editor.
theorem statement :
  ∃ s : Series, WorkingTitleSDS s ∧ ScriptEditor AndrewCollins s := by
  exact ⟨Badults, And.intro WT_Badults SE_Collins_Badults⟩
</lean>

ANSWER: True
