axiom obj : Type

axiom Stock : obj → Prop
axiom Growth : obj → Prop
axiom Mature : obj → Prop
axiom Suitable : obj → Prop
axiom RapidBought : obj → Prop

-- All growth stocks are bought to earn profits from rapid price appreciation.
axiom A1 : ∀ x : obj, Growth x → RapidBought x
-- If bought to earn rapid price appreciation, then not suitable for a retirement fund.
axiom A2 : ∀ x : obj, RapidBought x → ¬ Suitable x
-- Some stocks are growth stocks.
axiom A3 : ∃ x : obj, Stock x ∧ Growth x
-- All mature stocks are suitable for a retirement fund.
axiom A4 : ∀ x : obj, Mature x → Suitable x

-- KO is a mature stock.
axiom KO : obj
axiom KO_mature_stock : Mature KO ∧ Stock KO

lemma KO_mature : Mature KO := And.left KO_mature_stock
lemma KO_stock : Stock KO := And.right KO_mature_stock

-- KO cannot be a growth stock (since mature ⇒ suitable, while growth ⇒ rapid ⇒ not suitable).
theorem KO_not_growth : ¬ Growth KO :=
by
  intro hG
  have hR : RapidBought KO := A1 KO hG
  have hNotS : ¬ Suitable KO := A2 KO hR
  have hS : Suitable KO := A4 KO KO_mature
  exact hNotS hS

-- Therefore, KO cannot be both a stock and a growth stock.
theorem not_stock_and_growth_KO : ¬ (Stock KO ∧ Growth KO) :=
by
  intro h
  exact KO_not_growth (And.right h)
