<lean>
axiom obj : Type

axiom Hannah : obj

axiom AtMS : obj → Prop
axiom Works : obj → Prop
axiom Needs : obj → Prop
axiom Orders : obj → Prop
axiom Enjoys : obj → Prop
axiom Picky : obj → Prop
axiom SpendsTime : obj → Prop

axiom A1 : ∀ x : obj, AtMS x → Works x → Needs x
axiom A2 : ∀ x : obj, AtMS x → Orders x → Works x
axiom A3 : ∀ x : obj, AtMS x → (Orders x ∨ Enjoys x)
axiom A4 : ∀ x : obj, AtMS x → Enjoys x → ¬ Picky x
axiom A5 : ∀ x : obj, AtMS x → Enjoys x → SpendsTime x

axiom A6 : AtMS Hannah
axiom A7a : Works Hannah
axiom A7b : Needs Hannah → (¬ Picky Hannah ∧ ¬ Needs Hannah)

theorem hannah_needs : Needs Hannah := by
  exact A1 Hannah A6 A7a
</lean>

ANSWER: True
