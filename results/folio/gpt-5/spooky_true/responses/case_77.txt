<lean>
axiom obj : Type
axiom At : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom Spend : obj → Prop
axiom Picky : obj → Prop

axiom A1 : ∀ x : obj, At x → Work x → Need x
axiom A2 : ∀ x : obj, At x → Order x → Work x
axiom A3 : ∀ x : obj, At x → (Order x ∨ Enjoy x)
axiom A4 : ∀ x : obj, At x → Enjoy x → ¬ Picky x
axiom A5 : ∀ x : obj, At x → Enjoy x → Spend x

axiom Hannah : obj
axiom A6 : At Hannah
axiom A7_work : Work Hannah
axiom A7_imp : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah)

theorem statement :
  At Hannah ∧ (¬ Picky Hannah ∨ (Picky Hannah → Spend Hannah)) := by
  have needH : Need Hannah := A1 Hannah A6 A7_work
  have notPickyH : ¬ Picky Hannah := (A7_imp needH).left
  exact And.intro A6 (Or.inl notPickyH)
</lean>

ANSWER: True
