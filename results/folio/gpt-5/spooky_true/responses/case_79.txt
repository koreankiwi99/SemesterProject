<lean>
axiom obj : Type
axiom DriedThaiChilies : obj
axiom Spicy : obj → Prop
axiom BakedSweet : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- No baked sweets are spicy.
axiom no_baked_sweets_spicy : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom cupcakes_baked_sweets : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom mala_hotpots_spicy : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom bbm_cupcakes : ∀ x : obj, FromBBM x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom dtc_disj : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

theorem dtc_not_from_bbm : ¬ FromBBM DriedThaiChilies := by
  intro hbbm
  have hc : Cupcake DriedThaiChilies := bbm_cupcakes DriedThaiChilies hbbm
  have hb : BakedSweet DriedThaiChilies := cupcakes_baked_sweets DriedThaiChilies hc
  have hnsp : ¬ Spicy DriedThaiChilies := no_baked_sweets_spicy DriedThaiChilies hb
  cases dtc_disj with
  | inl hsp =>
      exact hnsp hsp
  | inr hrest =>
      cases hrest with
      | inl hmh =>
          have hsp' : Spicy DriedThaiChilies := mala_hotpots_spicy DriedThaiChilies hmh
          exact hnsp hsp'
      | inr hnb =>
          exact hnb hb
</lean>

ANSWER: True
