<lean>
axiom obj : Type
axiom Listed : obj → Prop
axiom NegRev : obj → Prop
axiom HighRate : obj → Prop
axiom Takeout : obj → Prop
axiom Popular : obj → Prop
axiom HPS : obj

axiom A1 : ∀ x : obj, Listed x → ¬ NegRev x
axiom A2 : ∀ x : obj, HighRate x → Listed x
axiom A3 : ∃ x : obj, ¬ Takeout x ∧ NegRev x
axiom A4 : ∀ x : obj, Popular x → HighRate x
axiom A5 : HighRate HPS ∨ Popular HPS

theorem hamden_statement : (Takeout HPS ∧ NegRev HPS) → (HighRate HPS ∧ ¬ Takeout HPS) :=
by
  intro h
  have high : HighRate HPS :=
    by
      cases A5 with
      | inl hr => exact hr
      | inr pop => exact A4 HPS pop
  have listed : Listed HPS := A2 HPS high
  have notNeg : ¬ NegRev HPS := A1 HPS listed
  have contra : False := notNeg h.right
  have notTO : ¬ Takeout HPS :=
    by
      intro ht
      exact False.elim contra
  exact And.intro high notTO
</lean>

ANSWER: True
