<lean>
axiom Restaurant : Type
axiom Hamden : Restaurant

axiom Listed : Restaurant → Prop
axiom ManyNeg : Restaurant → Prop
axiom RatingGT4 : Restaurant → Prop
axiom ProvideTakeOut : Restaurant → Prop
axiom PopularLocal : Restaurant → Prop

-- If a restaurant is listed in Yelp’s recommendations, then it has not received many negative reviews.
axiom A1 : ∀ x : Restaurant, Listed x → ¬ ManyNeg x
-- All restaurants with a rating greater than four are listed in Yelp’s recommendations.
axiom A2 : ∀ x : Restaurant, RatingGT4 x → Listed x
-- Some restaurants that do not provide take-out service receive many negative reviews.
axiom A3 : ∃ x : Restaurant, ¬ ProvideTakeOut x ∧ ManyNeg x
-- All restaurants popular among local residents have ratings greater than four.
axiom A4 : ∀ x : Restaurant, PopularLocal x → RatingGT4 x
-- The Hamden Plaza Subway store has a rating greater than four, or it is popular among local residents.
axiom A5 : RatingGT4 Hamden ∨ PopularLocal Hamden

theorem hamden_rating : RatingGT4 Hamden := by
  apply Or.elim A5
  · intro h; exact h
  · intro hpl; exact A4 Hamden hpl

theorem hamden_listed : Listed Hamden := A2 Hamden hamden_rating

theorem hamden_not_many_neg_reviews : ¬ ManyNeg Hamden := A1 Hamden hamden_listed
