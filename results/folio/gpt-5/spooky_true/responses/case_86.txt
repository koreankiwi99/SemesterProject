<lean>
axiom obj : Type

-- Entities
axiom SDCC : obj            -- The Surprising Adventures of Sir Digby Chicken Caesar (the movie)
axiom SirDigby : obj
axiom Nemesis : obj → obj

-- Predicates
axiom SuperheroMovie : obj → Prop
axiom NamedAfter : obj → obj → Prop
axiom InMovie : obj → obj → Prop
axiom Good : obj → Prop
axiom Fights : obj → obj → Prop
axiom Wins : obj → Prop

-- Premises
-- In superhero movies, the good guys always win (for characters in the movie)
axiom P1 : ∀ m x : obj, SuperheroMovie m → InMovie x m → Good x → Wins x
-- The Surprising Adventures of Sir Digby Chicken Caesar is a superhero movie.
axiom P2 : SuperheroMovie SDCC
-- Sir Digby fights his nemesis.
axiom P3 : Fights SirDigby (Nemesis SirDigby)
-- If a superhero movie is named after a character, that character is a good guy.
axiom P4 : ∀ m c : obj, SuperheroMovie m → NamedAfter m c → Good c
-- The Surprising Adventures of Sir Digby Chicken Caesar is named after Sir Digby.
axiom P5 : NamedAfter SDCC SirDigby
-- If somebody wins a fight, the person they are fighting does not win.
axiom P6 : ∀ x y : obj, Fights x y → Wins x → ¬ Wins y
-- If a superhero movie is named after a character, that character is in the movie.
axiom P7 : ∀ m c : obj, SuperheroMovie m → NamedAfter m c → InMovie c m

-- Goal: Sir Digby’s nemesis does not win.
theorem nemesis_does_not_win : ¬ Wins (Nemesis SirDigby) := by
  have g : Good SirDigby := P4 SDCC SirDigby P2 P5
  have i : InMovie SirDigby SDCC := P7 SDCC SirDigby P2 P5
  have w : Wins SirDigby := P1 SDCC SirDigby P2 i g
  exact P6 SirDigby (Nemesis SirDigby) P3 w
</lean>

ANSWER: True
