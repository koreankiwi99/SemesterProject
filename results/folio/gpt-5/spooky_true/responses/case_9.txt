<lean>
axiom obj : Type
axiom Tom : obj

axiom Wild : obj → Prop
axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Gould : obj → Prop
axiom Merriam : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop

-- Every wild turkey is one of the six listed types
axiom SixTypes :
  ∀ x : obj,
    Wild x →
    (Eastern x ∨ Osceola x ∨ Gould x ∨ Merriam x ∨ RioGrande x ∨ Ocellated x)

-- Facts about Tom
axiom Tom_not_Eastern : ¬ Eastern Tom
axiom Tom_not_Osceola : ¬ Osceola Tom
axiom Tom_not_Gould : ¬ Gould Tom
axiom Tom_not_Merriam : ¬ Merriam Tom
axiom Tom_not_RioGrande : ¬ RioGrande Tom
axiom Tom_Wild : Wild Tom

theorem Tom_is_Ocellated : Ocellated Tom :=
by
  have h := SixTypes Tom Tom_Wild
  cases h with
  | inl hE => exact False.elim (Tom_not_Eastern hE)
  | inr h1 =>
    cases h1 with
    | inl hO => exact False.elim (Tom_not_Osceola hO)
    | inr h2 =>
      cases h2 with
      | inl hG => exact False.elim (Tom_not_Gould hG)
      | inr h3 =>
        cases h3 with
        | inl hM => exact False.elim (Tom_not_Merriam hM)
        | inr h4 =>
          cases h4 with
          | inl hR => exact False.elim (Tom_not_RioGrande hR)
          | inr hOc => exact hOc
</lean>

ANSWER: True
