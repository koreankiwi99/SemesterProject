<lean>
axiom obj : Type
axiom MT : obj
axiom NLPTask : obj → Prop
axiom Generation : obj → Prop
axiom Understanding : obj → Prop
axiom OutputTextSeq : obj → Prop

-- A natural language processing task can be categorized as either a language generation task or a language understanding task.
axiom C1 : ∀ x : obj, NLPTask x → (Generation x ∨ Understanding x)
-- If the output of a natural language processing task is a text sequence, then it is a language generation task.
axiom C2 : ∀ x : obj, OutputTextSeq x → Generation x
-- Machine translation is a natural language processing task.
axiom C3 : NLPTask MT
-- The output of machine translation is a text sequence.
axiom C4 : OutputTextSeq MT

-- Statement to evaluate: Machine translation is a language generation task.
theorem mt_is_generation : Generation MT := C2 MT C4
</lean>

ANSWER: True
