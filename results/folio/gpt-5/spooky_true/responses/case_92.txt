axiom obj : Type

-- Entities
axiom Tom : obj
axiom Fluffy : obj
axiom OliveGarden : obj

-- Predicates and functions
axiom Managed : obj → Prop
axiom Pet : obj → Prop
axiom Cat : obj → Prop
axiom AllowsPets : obj → Prop
axiom AllowedMoveInWith : obj → obj → obj → Prop   -- building, person, pet
axiom WillRent : obj → obj → Prop                  -- building, person
axiom MonthlyRent : obj → Nat
axiom SecurityDeposit : obj → Nat

-- Facts
axiom some_managed_allows_pets : ∃ b : obj, Managed b ∧ AllowsPets b
axiom deposit_ge_rent : ∀ b : obj, Managed b → MonthlyRent b ≤ SecurityDeposit b
axiom fluffy_is_cat : Cat Fluffy
axiom cats_are_pets : ∀ x : obj, Cat x → Pet x
axiom olive_is_managed : Managed OliveGarden
axiom olive_rent_2000 : MonthlyRent OliveGarden = 2000
axiom two_thousand_gt_1500 : 1500 < 2000
axiom tom_rents_if : ∀ b : obj, Managed b → (AllowedMoveInWith b Tom Fluffy ∧ SecurityDeposit b ≤ 1500) → WillRent b Tom
axiom allows_pets_implies_allowed_move_in : ∀ b p a : obj, Managed b → AllowsPets b → Pet a → AllowedMoveInWith b p a

-- A derived predicate for convenience
def DepositAtMost1500 (b : obj) : Prop := SecurityDeposit b ≤ 1500

-- From the facts, the deposit at Olive Garden is strictly greater than 1500.
lemma sd_gt_1500_olive : 1500 < SecurityDeposit OliveGarden := by
  have h1 : 1500 < MonthlyRent OliveGarden := by
    have h : 1500 < 2000 := two_thousand_gt_1500
    simpa [olive_rent_2000] using h
  have h2 : MonthlyRent OliveGarden ≤ SecurityDeposit OliveGarden :=
    deposit_ge_rent OliveGarden olive_is_managed
  exact lt_of_lt_of_le h1 h2

lemma not_deposit_le_1500_olive : ¬ SecurityDeposit OliveGarden ≤ 1500 :=
  not_le_of_gt sd_gt_1500_olive

lemma not_deposit_at_most_1500_olive : ¬ DepositAtMost1500 OliveGarden := not_deposit_le_1500_olive

-- Therefore, Tom's condition for renting (which requires the deposit ≤ 1500) cannot be satisfied at Olive Garden.
lemma toms_condition_fails_at_olive :
  ¬ (AllowedMoveInWith OliveGarden Tom Fluffy ∧ SecurityDeposit OliveGarden ≤ 1500) := by
  intro h
  exact not_deposit_le_1500_olive h.right
