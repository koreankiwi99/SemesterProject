<lean>
axiom obj : Type
axiom Managed : obj → Prop
axiom OliveGarden : obj
axiom Rent : obj → Nat
axiom Deposit : obj → Nat

-- Policy: in any managed building, the security deposit is at least the monthly rent
axiom deposit_at_least_rent : ∀ b : obj, Managed b → Deposit b ≥ Rent b

-- Facts about the Olive Garden
axiom og_managed : Managed OliveGarden
axiom rent_og : Rent OliveGarden = 2000

-- Conclusion: the security deposit at the Olive Garden is $2000 or more
theorem security_deposit_og_ge2000 : Deposit OliveGarden ≥ 2000 :=
by
  have h : Deposit OliveGarden ≥ Rent OliveGarden := deposit_at_least_rent OliveGarden og_managed
  simpa [rent_og] using h
</lean>

ANSWER: True
