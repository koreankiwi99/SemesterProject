-- People in this club who perform in school talent shows often attend and are very engaged with school events.
axiom A1 : ∀ x : Person, InClub x ∧ Performs x → EngagedEvents x

-- People in this club either perform in school talent shows often or are inactive and disinterested community members.
axiom A2 : ∀ x : Person, InClub x → (Performs x ∨ InactiveDisinterested x)

-- People in this club who chaperone high school dances are not students who attend the school.
axiom A3 : ∀ x : Person, InClub x ∧ ChaperonesDances x → ¬ StudentAtSchool x

-- All people in this club who are inactive and disinterested members of their community chaperone high school dances.
axiom A4 : ∀ x : Person, InClub x ∧ InactiveDisinterested x → ChaperonesDances x

-- All young children and teenagers in this club who wish to further their academic careers and educational opportunities are students who attend the school.
axiom A5 : ∀ x : Person, InClub x ∧ YoungOrTeen x ∧ WishesAcademic x → StudentAtSchool x

-- Bonnie is in this club, and either (engaged with events and a student) or (not engaged with events and not a student).
axiom A6 : InClub Bonnie
axiom A7 : (EngagedEvents Bonnie ∧ StudentAtSchool Bonnie) ∨ (¬ EngagedEvents Bonnie ∧ ¬ StudentAtSchool Bonnie)

-- A derived fact: inactive club members are not students.
theorem inactive_implies_not_student {x : Person} (hc : InClub x) (hi : InactiveDisinterested x) : ¬ StudentAtSchool x := by
  have hchap : ChaperonesDances x := A4 x (And.intro hc hi)
  exact A3 x (And.intro hc hchap)
