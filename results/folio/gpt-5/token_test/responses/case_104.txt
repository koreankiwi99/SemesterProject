<lean>
axiom obj : Type
axiom Tom : obj

axiom GREApplicant : obj → Prop
axiom ProvesHardship : obj → Prop
axiom ProvidesAid : obj → Prop

axiom SingleParent : obj → Prop
axiom FewResources : obj → Prop

-- Facts about Tom
axiom Tom_single_parent : SingleParent Tom
axiom Tom_few_resources : FewResources Tom
axiom Tom_gre_applicant : GREApplicant Tom

-- Rules from the context
axiom AidRule : ∀ x : obj, GREApplicant x → ProvesHardship x → ProvidesAid x
axiom Hardship_from_single_parent : ∀ x : obj, SingleParent x → ProvesHardship x
axiom Hardship_from_few_resources : ∀ x : obj, FewResources x → ProvesHardship x

theorem ETS_provides_financial_aid_to_Tom : ProvidesAid Tom := by
  apply AidRule Tom
  · exact Tom_gre_applicant
  · exact Hardship_from_single_parent Tom Tom_single_parent
</lean>

ANSWER: True
