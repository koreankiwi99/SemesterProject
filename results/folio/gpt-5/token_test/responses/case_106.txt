axiom Person : Type
axiom Provider : Type

axiom Tom : Person
axiom ETS : Provider

axiom GREApplicant : Person → Prop
axiom EconomicHardship : Person → Prop
axiom SingleParentFamily : Person → Prop
axiom FewResources : Person → Prop
axiom ProvidesFA : Provider → Person → Prop

-- Rules from the context
axiom R_provide : ∀ x : Person, GREApplicant x → EconomicHardship x → ProvidesFA ETS x
axiom R_hardship : ∀ x : Person, (SingleParentFamily x ∨ FewResources x) → EconomicHardship x

-- Facts about Tom
axiom F_spf_Tom : SingleParentFamily Tom
axiom F_few_Tom : FewResources Tom
axiom F_app_Tom : GREApplicant Tom

lemma Tom_hardship : EconomicHardship Tom :=
  R_hardship Tom (Or.inl F_spf_Tom)

-- There exists a GRE applicant who has financial aid provided to them by something (ETS).
theorem exists_applicant_with_fa : ∃ x : Person, GREApplicant x ∧ ∃ p : Provider, ProvidesFA p x := by
  refine Exists.intro Tom ?_
  refine And.intro F_app_Tom ?_
  refine Exists.intro ETS ?_
  exact R_provide Tom F_app_Tom Tom_hardship

-- Therefore, it is not the case that no GRE applicant has financial aid from anything.
theorem not_no_applicant_has_fa :
  ¬ (∀ x : Person, GREApplicant x → ¬ ∃ p : Provider, ProvidesFA p x) := by
  intro h
  have hx := h Tom
  have hx' : GREApplicant Tom → ¬ ∃ p : Provider, ProvidesFA p Tom := hx
  have faTom : ∃ p : Provider, ProvidesFA p Tom := by
    refine Exists.intro ETS ?_
    exact R_provide Tom F_app_Tom Tom_hardship
  have neg := hx' F_app_Tom
  exact neg faTom
</lean>
