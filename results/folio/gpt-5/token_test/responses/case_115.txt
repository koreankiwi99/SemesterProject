<lean>
axiom Person : Type
axiom Michael : Person
axiom Peter : Person
axiom Windy : Person

axiom Man : Person → Prop
axiom Taller : Person → Person → Prop
axiom CanBlock : Person → Person → Prop
axiom JumpsWhenShooting : Person → Prop
axiom GreatShooter : Person → Prop
axiom InMichaelClass : Person → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R1 : ∀ x y, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom M1 : Man Michael
axiom M2 : ∀ x, InMichaelClass x → Taller Michael x

-- Transitivity of taller
axiom Ttrans : ∀ x y z, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom P1 : Man Peter
axiom P2 : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom Mblock_rule : ∀ p, ¬ JumpsWhenShooting p → CanBlock Michael p

-- Michael cannot block Windy's shooting.
axiom NotBlockWindy : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom JumpImpliesGreat : ∀ p, JumpsWhenShooting p → GreatShooter p

open Classical

theorem windy_great : GreatShooter Windy := by
  have hjump : JumpsWhenShooting Windy := by
    by_contra h
    have hb : CanBlock Michael Windy := Mblock_rule Windy h
    exact NotBlockWindy hb
  exact JumpImpliesGreat Windy hjump
</lean>

ANSWER: True

  have hjump : JumpsWhenShooting Windy := by
    by_cases h : JumpsWhenShooting Windy
