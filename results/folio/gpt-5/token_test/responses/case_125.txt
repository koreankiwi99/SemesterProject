<lean>
axiom obj : Type

axiom PeterPet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

axiom Rock : obj

-- All of Peter's pets that are mammals are also animals.
axiom A1 : ∀ x : obj, PeterPet x ∧ Mammal x → Animal x
-- All monkeys are mammals.
axiom A2 : ∀ x : obj, Monkey x → Mammal x
-- Peter's pets are all either monkeys or birds.
axiom A3 : ∀ x : obj, PeterPet x → Monkey x ∨ Bird x
-- Peter's birds can fly.
axiom A4 : ∀ x : obj, PeterPet x ∧ Bird x → CanFly x
-- All animals that are Peter's pets can breathe.
axiom A5 : ∀ x : obj, PeterPet x ∧ Animal x → CanBreathe x
-- If Peter's pet can fly, then it has wings.
axiom A6 : ∀ x : obj, PeterPet x ∧ CanFly x → HasWings x
-- Rock is Peter's pet.
axiom A7 : PeterPet Rock
-- Rock can fly, or Rock is a bird, or Rock cannot breathe.
axiom A8 : CanFly Rock ∨ Bird Rock ∨ ¬CanBreathe Rock

theorem rock_has_wings : HasWings Rock := by
  have petR : PeterPet Rock := A7
  cases A8 with
  | inl hFly =>
      exact A6 Rock ⟨petR, hFly⟩
  | inr h =>
      cases h with
      | inl hBird =>
          have hFly : CanFly Rock := A4 Rock ⟨petR, hBird⟩
          exact A6 Rock ⟨petR, hFly⟩
      | inr hNotBreathe =>
          have mb : Monkey Rock ∨ Bird Rock := A3 Rock petR
          cases mb with
          | inl hMonkey =>
              have hMammal : Mammal Rock := A2 Rock hMonkey
              have hAnimal : Animal Rock := A1 Rock ⟨petR, hMammal⟩
              have hBreathe : CanBreathe Rock := A5 Rock ⟨petR, hAnimal⟩
              exact False.elim (hNotBreathe hBreathe)
          | inr hBird2 =>
              have hFly : CanFly Rock := A4 Rock ⟨petR, hBird2⟩
              exact A6 Rock ⟨petR, hFly⟩
</lean>

ANSWER: True
