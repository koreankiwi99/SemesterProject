<lean>
axiom obj : Type
axiom Rock : obj

axiom PetersPet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

-- All of Peter's pets that are mammals are also animals.
axiom A1 : ∀ x : obj, PetersPet x ∧ Mammal x → Animal x
-- All monkeys are mammals.
axiom A2 : ∀ x : obj, Monkey x → Mammal x
-- Peter's pets are all either monkeys or birds.
axiom A3 : ∀ x : obj, PetersPet x → (Monkey x ∨ Bird x)
-- Peter's birds can fly.
axiom A4 : ∀ x : obj, PetersPet x ∧ Bird x → CanFly x
-- All animals that are Peter's pets can breathe.
axiom A5 : ∀ x : obj, PetersPet x ∧ Animal x → CanBreathe x
-- If Peter's pet can fly, then it has wings.
axiom A6 : ∀ x : obj, PetersPet x ∧ CanFly x → HasWings x
-- Rock is Peter's pet.
axiom A7 : PetersPet Rock
-- Rock can fly, or Rock is a bird, or Rock cannot breathe.
axiom A8 : CanFly Rock ∨ (Bird Rock ∨ ¬ CanBreathe Rock)

theorem rock_has_wings : HasWings Rock := by
  have hPet : PetersPet Rock := A7
  cases A8 with
  | inl hFly =>
      exact A6 Rock (And.intro hPet hFly)
  | inr h =>
      cases h with
      | inl hBird =>
          have hFly : CanFly Rock := A4 Rock (And.intro hPet hBird)
          exact A6 Rock (And.intro hPet hFly)
      | inr hNotBreath =>
          have hMB : Monkey Rock ∨ Bird Rock := A3 Rock hPet
          cases hMB with
          | inl hMon =>
              have hMam : Mammal Rock := A2 Rock hMon
              have hAni : Animal Rock := A1 Rock (And.intro hPet hMam)
              have hB : CanBreathe Rock := A5 Rock (And.intro hPet hAni)
              exact False.elim (hNotBreath hB)
          | inr hBird2 =>
              have hFly : CanFly Rock := A4 Rock (And.intro hPet hBird2)
              exact A6 Rock (And.intro hPet hFly)
</lean>
