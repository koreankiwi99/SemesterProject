<lean>
axiom obj : Type
axiom John : obj

axiom Engaged : obj → Prop
axiom MakesWeddingPlans : obj → Prop
axiom InvitesOthersToCeremony : obj → Prop
axiom WellAttendedWedding : obj → Prop
axiom LargerFamily : obj → Prop
axiom InvitesFriendsToCeremony : obj → Prop

-- 1. All people who make wedding plans are people who are engaged.
axiom A1 : ∀ x : obj, MakesWeddingPlans x → Engaged x

-- 2. All people who invite others to come to their ceremony make wedding plans.
axiom A2 : ∀ x : obj, InvitesOthersToCeremony x → MakesWeddingPlans x

-- 3. Anyone who has a well-attended wedding invites others to come to their ceremony.
axiom A3 : ∀ x : obj, WellAttendedWedding x → InvitesOthersToCeremony x

-- 4. Some people that have well-attended weddings have larger families.
axiom A4 : ∃ x : obj, WellAttendedWedding x ∧ LargerFamily x

-- 5. If John is engaged, then John does not (invite friends to their ceremony and make wedding plans).
axiom A5 : Engaged John → ¬(InvitesFriendsToCeremony John ∧ MakesWeddingPlans John)

-- 6. If John has a larger family, then either John has a well-attended wedding or invites friends to their ceremony.
axiom A6 : LargerFamily John → (WellAttendedWedding John ∨ InvitesFriendsToCeremony John)

-- Some derived consequences (not sufficient to prove or refute LargerFamily John)
theorem well_attended_implies_engaged (x : obj) : WellAttendedWedding x → Engaged x :=
  fun hw => A1 x (A2 x (A3 x hw))

theorem john_well_attended_implies_not_invite_friends :
  WellAttendedWedding John → ¬InvitesFriendsToCeremony John :=
  fun hw => by
    have hmp : MakesWeddingPlans John := A2 John (A3 John hw)
    have hej : Engaged John := A1 John hmp
    have hnot : ¬(InvitesFriendsToCeremony John ∧ MakesWeddingPlans John) := A5 hej
    intro hf
    exact hnot ⟨hf, hmp⟩
