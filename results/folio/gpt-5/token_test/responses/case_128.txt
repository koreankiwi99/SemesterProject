<lean>
axiom obj : Type
axiom John : obj

axiom MakesPlans : obj → Prop
axiom Engaged : obj → Prop
axiom InvitesOthers : obj → Prop
axiom WellAttended : obj → Prop
axiom LargerFamily : obj → Prop
axiom InvitesFriends : obj → Prop

-- 1. All who make wedding plans are engaged.
axiom A1 : ∀ x : obj, MakesPlans x → Engaged x
-- 2. All who invite others to their ceremony make wedding plans.
axiom A2 : ∀ x : obj, InvitesOthers x → MakesPlans x
-- 3. Anyone with a well-attended wedding invites others to their ceremony.
axiom A3 : ∀ x : obj, WellAttended x → InvitesOthers x
-- 4. Some people with well-attended weddings have larger families.
axiom A4 : ∃ x : obj, WellAttended x ∧ LargerFamily x
-- 5. If John is engaged, then John does not invite friends and makes wedding plans.
axiom A5 : Engaged John → (¬ InvitesFriends John ∧ MakesPlans John)
-- 6. If John has a larger family, then either John has a well-attended wedding or invites friends.
axiom A6 : LargerFamily John → (WellAttended John ∨ InvitesFriends John)

-- Derived: Well-attended implies engaged (for any person).
theorem wa_imp_engaged (x : obj) : WellAttended x → Engaged x :=
by
  intro hwa
  have hio : InvitesOthers x := A3 x hwa
  have hmp : MakesPlans x := A2 x hio
  exact A1 x hmp

-- Derived: If John has a well-attended wedding, then John does not invite friends.
theorem john_wa_implies_not_invitefriends : WellAttended John → ¬ InvitesFriends John :=
by
  intro hwa
  have heng : Engaged John := wa_imp_engaged John hwa
  have hpair : ¬ InvitesFriends John ∧ MakesPlans John := A5 heng
  exact hpair.left
