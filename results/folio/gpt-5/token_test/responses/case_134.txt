<lean>
axiom obj : Type

axiom Plunger : obj → Prop
axiom Vacuum : obj → Prop
axiom Vampire : obj → Prop
axiom HouseholdAppliance : obj → Prop
axiom Suck : obj → Prop

axiom Space : obj
axiom Duster : obj

-- Plungers suck
axiom A1 : ∀ x : obj, Plunger x → Suck x
-- Vacuums suck
axiom A2 : ∀ x : obj, Vacuum x → Suck x
-- Vampires suck
axiom A3 : ∀ x : obj, Vampire x → Suck x
-- Space is a vacuum
axiom A4 : Vacuum Space
-- A duster is a household appliance that doesn't suck
axiom A5 : HouseholdAppliance Duster
axiom A6 : ¬ Suck Duster

theorem space_sucks : Suck Space := by
  exact A2 Space A4
</lean>

ANSWER: True
