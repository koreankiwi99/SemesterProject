<lean>
axiom Alg : Type
axiom Model : Type
axiom SOTA : Model

axiom ML : Alg → Prop
axiom Supervised : Alg → Prop
axiom Unsupervised : Alg → Prop
axiom Reinforcement : Alg → Prop
axiom RequiresLabels : Alg → Prop
axiom UsedToTrain : Alg → Model → Prop

-- Machine Learning algorithms can be categorized as supervised, unsupervised, or reinforcement
axiom categorized : ∀ a : Alg, ML a → (Supervised a ∨ Unsupervised a ∨ Reinforcement a)

-- Unsupervised learning algorithms do not require labeled data
axiom unsup_no_labels : ∀ a : Alg, Unsupervised a → ¬ RequiresLabels a

-- The state-of-the-art text summarization model is trained with machine learning algorithms
axiom trained_with_ml : ∃ a : Alg, ML a ∧ UsedToTrain a SOTA

-- Reinforcement learning is not used to train the state-of-the-art text summarization model
axiom no_rl_used : ∀ a : Alg, Reinforcement a → ¬ UsedToTrain a SOTA

-- The Machine Learning algorithm for training text summarization models requires labeled data
axiom used_requires_labels : ∀ a : Alg, UsedToTrain a SOTA → RequiresLabels a

-- Claim to evaluate: "Unsupervised learning is used to train the state-of-the-art text summarization model."
-- This corresponds to: ∃ a, UsedToTrain a SOTA ∧ Unsupervised a
-- We prove its negation.
theorem no_unsupervised_used : ¬ (∃ a : Alg, UsedToTrain a SOTA ∧ Unsupervised a) := by
  intro h
  cases h with
  | intro a hpair =>
    have hused : UsedToTrain a SOTA := hpair.left
    have hunsup : Unsupervised a := hpair.right
    have hreq : RequiresLabels a := used_requires_labels a hused
    have hnot : ¬ RequiresLabels a := unsup_no_labels a hunsup
    exact hnot hreq
</lean>
