<lean>
axiom obj : Type
axiom Vladimir : obj
axiom Ekaterina : obj

axiom CanRegisterUS : obj → Prop
axiom CanParticipate2024 : obj → Prop
axiom HasUSCitizenship : obj → Prop
axiom HasTaiwanCitizenship : obj → Prop
axiom RussianOfficial : obj → Prop
axiom ManagerGazprom : obj → Prop

-- Everyone who can register to vote in the United States can participate in the 2024 US presidential election.
axiom A1 : ∀ x : obj, CanRegisterUS x → CanParticipate2024 x
-- If someone has United States citizenship, then they can register to vote in the United States.
axiom A2 : ∀ x : obj, HasUSCitizenship x → CanRegisterUS x
-- A person either has United States citizenship or Taiwanese citizenship.
axiom A3 : ∀ x : obj, HasUSCitizenship x ∨ HasTaiwanCitizenship x
-- No Russian Federation officials hold Taiwanese citizenship.
axiom A4 : ∀ x : obj, RussianOfficial x → ¬ HasTaiwanCitizenship x
-- Vladimir neither holds Taiwanese citizenship nor is he a manager at Gazprom.
axiom T_Vlad_not_TW : ¬ HasTaiwanCitizenship Vladimir
axiom T_Vlad_not_mgr : ¬ ManagerGazprom Vladimir
-- Ekaterina can register to vote in the United States, or she is a Russian federation official.
axiom T_Eka_reg_or_off : CanRegisterUS Ekaterina ∨ RussianOfficial Ekaterina

-- Goal: Ekaterina can participate in the 2024 US presidential election or is a manager at Gazprom.
theorem ekaterina_participate_or_manager :
  CanParticipate2024 Ekaterina ∨ ManagerGazprom Ekaterina := by
  cases T_Eka_reg_or_off with
  | inl hreg =>
      exact Or.inl (A1 Ekaterina hreg)
  | inr hoff =>
      have dis := A3 Ekaterina
      cases dis with
      | inl hUS =>
          have hreg2 := A2 Ekaterina hUS
          exact Or.inl (A1 Ekaterina hreg2)
      | inr hTW =>
          have hnotTW := A4 Ekaterina hoff
          have contra := hnotTW hTW
          have participate : CanParticipate2024 Ekaterina := False.elim contra
          exact Or.inl participate
</lean>

ANSWER: True
