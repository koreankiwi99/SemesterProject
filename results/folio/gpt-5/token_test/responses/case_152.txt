<lean>
axiom obj : Type
axiom mix : obj

axiom Hydrocarbon : obj → Prop
axiom Organic : obj → Prop
axiom Alkane : obj → Prop
axiom Chemical : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom OnlyOneElement : obj → Prop

-- All hydrocarbons are organic compounds.
axiom A1 : ∀ x : obj, Hydrocarbon x → Organic x
-- All alkanes are hydrocarbons.
axiom A2 : ∀ x : obj, Alkane x → Hydrocarbon x
-- All organic compounds are chemical compounds.
axiom A3 : ∀ x : obj, Organic x → Chemical x
-- All organic compounds contain carbon.
axiom A4 : ∀ x : obj, Organic x → ContainsCarbon x
-- No chemical compounds contain only one element.
axiom A5 : ∀ x : obj, Chemical x → ¬ OnlyOneElement x
-- The mixture either both is a chemical compound and contains only one element, or neither.
axiom A6 : (Chemical mix ∧ OnlyOneElement mix) ∨ (¬ Chemical mix ∧ ¬ OnlyOneElement mix)

-- From A5, the left disjunct in A6 is impossible.
theorem not_left_case : ¬(Chemical mix ∧ OnlyOneElement mix) := by
  intro h
  have hc : Chemical mix := And.left h
  have ho : OnlyOneElement mix := And.right h
  have hno : ¬ OnlyOneElement mix := A5 mix hc
  exact hno ho

-- Therefore, the right disjunct must hold: the mixture is neither chemical nor single-element.
theorem mix_not_chemical_and_not_one : ¬Chemical mix ∧ ¬OnlyOneElement mix := by
  have h := A6
  have hn := not_left_case
  cases h with
  | inl hl =>
      exact False.elim (hn hl)
  | inr hr =>
      exact hr

-- Consequently, the mixture is not organic (since organic → chemical).
theorem not_organic_mix : ¬Organic mix := by
  intro horg
  have hc : Chemical mix := A3 mix horg
  exact (And.left mix_not_chemical_and_not_one) hc

-- A4 only gives Organic x → ContainsCarbon x, and we have ¬Organic mix, which does not imply ¬ContainsCarbon mix.
-- Hence the truth of ContainsCarbon mix is underdetermined by the premises.
