<lean>
axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

-- A hawk never swims
axiom no_swim_hawk : ∀ x : obj, Hawk x → ¬ Swims x
-- Some birds are hawks
axiom some_birds_are_hawks : ∃ x : obj, Bird x ∧ Hawk x

-- Therefore, it is not the case that all birds swim
theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swims x) :=
by
  intro h
  cases some_birds_are_hawks with
  | intro w hw =>
    have hb : Bird w := And.left hw
    have hh : Hawk w := And.right hw
    have hnot : ¬ Swims w := no_swim_hawk w hh
    have hsw : Swims w := h w hb
    exact hnot hsw
</lean>

ANSWER: False
