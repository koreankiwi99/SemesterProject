<lean>
axiom obj : Type

axiom Show : obj → Prop
axiom AirsMon8pm : obj → Prop
axiom GivesRosesOnTV : obj → Prop
axiom EpisodeOfTheBachelor : obj → Prop
axiom PortraysRealLives : obj → Prop
axiom RealityTVShow : obj → Prop

axiom BreakingBad : obj

-- Context axioms
axiom some_monday8_rose_show :
  ∃ s : obj, Show s ∧ AirsMon8pm s ∧ GivesRosesOnTV s

axiom roses_implies_bachelor :
  ∀ x : obj, GivesRosesOnTV x → EpisodeOfTheBachelor x

axiom bachelor_portrays_real :
  ∀ x : obj, EpisodeOfTheBachelor x → PortraysRealLives x

axiom portrays_implies_reality :
  ∀ x : obj, PortraysRealLives x → RealityTVShow x

axiom breaking_bad_is_show : Show BreakingBad
axiom breaking_bad_not_reality : ¬ RealityTVShow BreakingBad

-- From the rules, any show that gives out roses is reality TV; Breaking Bad is not.
-- Hence Breaking Bad does not give out roses.
theorem breaking_bad_not_gives_roses : ¬ GivesRosesOnTV BreakingBad :=
by
  intro hrose
  have hbachelor : EpisodeOfTheBachelor BreakingBad :=
    roses_implies_bachelor BreakingBad hrose
  have hportray : PortraysRealLives BreakingBad :=
    bachelor_portrays_real BreakingBad hbachelor
  have hreality : RealityTVShow BreakingBad :=
    portrays_implies_reality BreakingBad hportray
  exact breaking_bad_not_reality hreality

-- Therefore the full conjunctive statement is false.
theorem breaking_bad_not_monday8_rose_show :
  ¬ (Show BreakingBad ∧ AirsMon8pm BreakingBad ∧ GivesRosesOnTV BreakingBad) :=
by
  intro h
  have hAG : AirsMon8pm BreakingBad ∧ GivesRosesOnTV BreakingBad := And.right h
  have hG : GivesRosesOnTV BreakingBad := And.right hAG
  exact breaking_bad_not_gives_roses hG
</lean>
