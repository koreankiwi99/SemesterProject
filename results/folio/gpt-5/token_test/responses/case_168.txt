axiom obj : Type
axiom HeinrichSchmidt : obj

axiom German : obj → Prop
axiom Politician : obj → Prop
axiom MemberPrussianStateParliament : obj → Prop
axiom MemberNaziReichstag : obj → Prop

axiom HS_is_politician : Politician HeinrichSchmidt
axiom HS_in_prussian : MemberPrussianStateParliament HeinrichSchmidt
axiom HS_in_reichstag : MemberNaziReichstag HeinrichSchmidt

-- The claim to evaluate: No politicians are part of the Nazi Reichstag
-- This is ∀ x, Politician x → ¬ MemberNaziReichstag x
-- We prove its negation by providing a counterexample (Heinrich Schmidt).
theorem not_no_politicians_in_reichstag :
  ¬ (∀ x : obj, Politician x → ¬ MemberNaziReichstag x) :=
by
  intro h
  have hn : ¬ MemberNaziReichstag HeinrichSchmidt := h HeinrichSchmidt HS_is_politician
  exact hn HS_in_reichstag
