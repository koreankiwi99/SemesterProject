<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatsSaladsRegularly : obj → Prop
axiom VeryConscious : obj → Prop
axiom GrewUpHealthConsciousHome : obj → Prop
axiom FulfillDailyNutritionalIntakes : obj → Prop
axiom DisregardPhysicalWellBeing : obj → Prop
axiom VisitGymOnceADay : obj → Prop

-- 1. All who eat salads regularly are very conscious
axiom A1 : ∀ x : obj, EatsSaladsRegularly x → VeryConscious x
-- 2. All who grew up in health-conscious homes eat salads regularly
axiom A2 : ∀ x : obj, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
-- 3. All who fulfill daily nutritional intakes grew up in health-conscious homes
axiom A3 : ∀ x : obj, FulfillDailyNutritionalIntakes x → GrewUpHealthConsciousHome x
-- 4. All who disregard physical well-being are not very conscious
axiom A4 : ∀ x : obj, DisregardPhysicalWellBeing x → ¬ VeryConscious x
-- 5. If visit gym once a day, then fulfill daily nutritional intakes
axiom A5 : ∀ x : obj, VisitGymOnceADay x → FulfillDailyNutritionalIntakes x
-- 6. Taylor either (grew up HC and disregards) or (neither)
axiom A6 : (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellBeing Taylor)
         ∨ (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardPhysicalWellBeing Taylor)

theorem taylor_not_visit_gym : ¬ VisitGymOnceADay Taylor := by
  -- Show the first disjunct in A6 is impossible
  have hNotBoth : ¬ (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellBeing Taylor) := by
    intro h
    have hHC : GrewUpHealthConsciousHome Taylor := And.left h
    have hDis : DisregardPhysicalWellBeing Taylor := And.right h
    have hEat : EatsSaladsRegularly Taylor := A2 Taylor hHC
    have hVC  : VeryConscious Taylor := A1 Taylor hEat
    have hNotVC : ¬ VeryConscious Taylor := A4 Taylor hDis
    exact hNotVC hVC
  -- From A6 and the impossibility above, conclude Taylor did neither
  have hNeither : ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardPhysicalWellBeing Taylor := by
    apply Or.elim A6
    · intro hBoth
      exact False.elim (hNotBoth hBoth)
    · intro h
      exact h
  have hNotHC : ¬ GrewUpHealthConsciousHome Taylor := And.left hNeither
  -- If Taylor visited the gym daily, that would imply HC, contradicting ¬HC
  intro hVG
  have hFul : FulfillDailyNutritionalIntakes Taylor := A5 Taylor hVG
  have hHC : GrewUpHealthConsciousHome Taylor := A3 Taylor hFul
  exact hNotHC hHC
</lean>
