<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatsSaladsRegularly : obj → Prop
axiom VeryConscious : obj → Prop
axiom GrewUpHealthConscious : obj → Prop
axiom FulfillsDailyNutrition : obj → Prop
axiom DisregardsPhysicalWellbeing : obj → Prop
axiom VisitsGymDaily : obj → Prop

-- All people who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : obj, EatsSaladsRegularly x → VeryConscious x
-- All people who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : obj, GrewUpHealthConscious x → EatsSaladsRegularly x
-- All people who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : obj, FulfillsDailyNutrition x → GrewUpHealthConscious x
-- All people who disregard their physical well-being are not very conscious about their health and eating habits.
axiom A4 : ∀ x : obj, DisregardsPhysicalWellbeing x → ¬ VeryConscious x
-- If people visit the gym at least once a day, then they always fulfill their daily nutritional intakes.
axiom A5 : ∀ x : obj, VisitsGymDaily x → FulfillsDailyNutrition x
-- Taylor either (grew up health-conscious ∧ disregards well-being) or (neither).
axiom A6 : (GrewUpHealthConscious Taylor ∧ DisregardsPhysicalWellbeing Taylor) ∨
            (¬ GrewUpHealthConscious Taylor ∧ ¬ DisregardsPhysicalWellbeing Taylor)

theorem taylor_neither_grew_nor_visits :
  ¬ GrewUpHealthConscious Taylor ∧ ¬ VisitsGymDaily Taylor := by
  -- From growing up health-conscious, Taylor would be very conscious.
  have hVfromG : ∀ x, GrewUpHealthConscious x → VeryConscious x :=
    fun x hg => A1 x (A2 x hg)
  -- Cannot have both grew up health-conscious and disregards physical well-being.
  have hnotGD : ¬ (GrewUpHealthConscious Taylor ∧ DisregardsPhysicalWellbeing Taylor) := by
    intro h
    have hg := h.left
    have hd := h.right
    have v : VeryConscious Taylor := hVfromG Taylor hg
    have nv : ¬ VeryConscious Taylor := A4 Taylor hd
    exact nv v
  -- From the exclusive-style disjunction, deduce neither.
  have hneitherGD : ¬ GrewUpHealthConscious Taylor ∧ ¬ DisregardsPhysicalWellbeing Taylor := by
    cases A6 with
    | inl hgd => exact False.elim (hnotGD hgd)
    | inr hneither => exact hneither
  have hnotG : ¬ GrewUpHealthConscious Taylor := hneitherGD.left
  -- Visiting gym daily would imply grew up health-conscious.
  have hGymImpG : VisitsGymDaily Taylor → GrewUpHealthConscious Taylor :=
    fun hvisit => A3 Taylor (A5 Taylor hvisit)
  -- Hence Taylor does not visit the gym daily.
  have hnotGym : ¬ VisitsGymDaily Taylor := by
    intro hvisit
    have hg : GrewUpHealthConscious Taylor := hGymImpG hvisit
    exact hnotG hg
  exact And.intro hnotG hnotGym
</lean>

ANSWER: True
