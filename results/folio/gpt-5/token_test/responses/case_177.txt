<lean>
axiom obj : Type
axiom UnitedStates : obj
axiom LastSummerOlympics : obj
axiom Tokyo : obj

-- Predicates
axiom WasIn : obj → obj → Prop        -- an event was in a location
axiom WonMostMedalsAt : obj → obj → Prop  -- a country won the most medals at an event or location

-- Facts from the context
axiom T1 : WasIn LastSummerOlympics Tokyo
axiom T2 : WonMostMedalsAt UnitedStates Tokyo

-- Bridging rule: if an event was in a location and someone won the most medals at that location,
-- then they won the most medals at that event.
axiom R1 : ∀ x e loc : obj, WasIn e loc → WonMostMedalsAt x loc → WonMostMedalsAt x e

theorem us_won_most_medals_last : WonMostMedalsAt UnitedStates LastSummerOlympics :=
  R1 UnitedStates LastSummerOlympics Tokyo T1 T2
</lean>

ANSWER: True
