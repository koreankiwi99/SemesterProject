<lean>
axiom Person : Type
axiom Airport : Type

axiom Susan : Person
axiom John : Person
axiom LGA : Airport

axiom FliesTo : Person → Airport → Prop
axiom FliesFrom : Person → Airport → Prop

-- Given facts
axiom T1 : FliesTo Susan LGA
axiom T2 : FliesFrom John LGA

-- Constraint: departure and arrival cannot be at the same airport
axiom R1 : ∀ p : Person, ∀ a : Airport, FliesTo p a → ¬ FliesFrom p a

-- From arriving at LGA and the constraint, Susan does not depart from LGA
theorem not_susan_from_lga : ¬ FliesFrom Susan LGA := R1 Susan LGA T1
</lean>
