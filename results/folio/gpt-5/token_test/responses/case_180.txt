<lean>
axiom Person : Type
axiom Airport : Type
axiom Susan : Person
axiom John : Person
axiom LGA : Airport

axiom FliesTo : Person → Airport → Prop
axiom FliesFrom : Person → Airport → Prop

-- Given facts
axiom Susan_to_LGA : FliesTo Susan LGA
axiom not_same_airport : ∀ p a, ¬ (FliesTo p a ∧ FliesFrom p a)
axiom John_from_LGA : FliesFrom John LGA

-- Prove that John does not fly to LGA
theorem john_not_to_LGA : ¬ FliesTo John LGA := by
  intro hto
  have hcontra : FliesTo John LGA ∧ FliesFrom John LGA := And.intro hto John_from_LGA
  exact (not_same_airport John LGA) hcontra
</lean>

ANSWER: False
