<lean>
axiom obj : Type
axiom Tom : obj

axiom Nice : obj → Prop
axiom Mean : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop

-- No one nice to animals is also mean to animals.
axiom A1 : ∀ x : obj, Nice x → ¬ Mean x
-- Some grumpy people are mean to animals.
axiom A2 : ∃ x : obj, Grumpy x ∧ Mean x
-- All animal lovers are nice to animals.
axiom A3 : ∀ x : obj, AnimalLover x → Nice x
-- All pet owners love animals.
axiom A4 : ∀ x : obj, PetOwner x → AnimalLover x
-- Tom is a pet owner.
axiom TTom : PetOwner Tom

-- We show the given statement is false by proving its negation.
theorem not_statement :
  ¬ ((¬ (Grumpy Tom ∧ Mean Tom)) → (¬ Grumpy Tom ∧ ¬ AnimalLover Tom)) := by
  -- Tom loves animals
  have loverTom : AnimalLover Tom := A4 Tom TTom
  -- Tom is nice to animals
  have niceTom : Nice Tom := A3 Tom loverTom
  -- Hence Tom is not mean to animals
  have notMeanTom : ¬ Mean Tom := A1 Tom niceTom
  -- Therefore, it's not the case that Tom is both grumpy and mean
  have ante : ¬ (Grumpy Tom ∧ Mean Tom) := by
    intro h
    exact notMeanTom h.right
  -- But the conclusion (¬Grumpy Tom ∧ ¬AnimalLover Tom) is false since Tom is an animal lover
  have notConseq : ¬ (¬ Grumpy Tom ∧ ¬ AnimalLover Tom) := by
    intro h
    exact h.right loverTom
  -- So the implication is false
  intro hImp
  exact notConseq (hImp ante)
</lean>

ANSWER: False
