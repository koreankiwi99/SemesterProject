<lean>
axiom obj : Type
axiom Yuri : obj

axiom American : obj → Prop
axiom ProBB : obj → Prop
axiom CanJumpShot : obj → Prop
axiom LeapStraight : obj → Prop
axiom ActivateLegMuscles : obj → Prop

axiom some_non_american_pro : ∃ x : obj, ProBB x ∧ ¬ American x
axiom all_pro_can_jump : ∀ x : obj, ProBB x → CanJumpShot x
axiom jump_implies_leap : ∀ x : obj, CanJumpShot x → LeapStraight x
axiom leap_implies_activate : ∀ x : obj, LeapStraight x → ActivateLegMuscles x
axiom yuri_not_activate : ¬ ActivateLegMuscles Yuri

theorem yuri_not_american_professional : ¬ (American Yuri ∧ ProBB Yuri) := by
  intro h
  cases h with
  | intro hAm hPro =>
    have hJump : CanJumpShot Yuri := all_pro_can_jump Yuri hPro
    have hLeap : LeapStraight Yuri := jump_implies_leap Yuri hJump
    have hAct : ActivateLegMuscles Yuri := leap_implies_activate Yuri hLeap
    exact yuri_not_activate hAct
</lean>

ANSWER: True
