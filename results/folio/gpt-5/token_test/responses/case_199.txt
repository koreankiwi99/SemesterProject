axiom FC_Nautico : FootballClub Nautico

-- We prove its negation.
theorem statement_false : ¬ (¬ ∃ c : obj, FootballClub c ∧ PlaysFor AiltonSilva c) := by
  intro h
  exact h exists_club_played_for
