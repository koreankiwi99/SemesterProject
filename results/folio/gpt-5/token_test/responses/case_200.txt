<lean>
axiom obj : Type
axiom Ailton : obj
axiom Braga : obj
axiom Nautico : obj
axiom Fluminense : obj

axiom FootballClub : obj → Prop
axiom LoanedTo : obj → obj → Prop
axiom PlaysFor : obj → obj → Prop

-- Given facts
axiom loaned_Ailton_Braga : LoanedTo Ailton Braga
axiom club_Braga : FootballClub Braga
axiom club_Nautico : FootballClub Nautico
axiom club_Fluminense : FootballClub Fluminense
axiom plays_Ailton_Nautico : PlaysFor Ailton Nautico

-- Negation of the claim: Ailton was loaned out to a football club
theorem ailton_loaned_to_a_football_club : ∃ c : obj, FootballClub c ∧ LoanedTo Ailton c :=
  Exists.intro Braga (And.intro club_Braga loaned_Ailton_Braga)
</lean>

ANSWER: False
