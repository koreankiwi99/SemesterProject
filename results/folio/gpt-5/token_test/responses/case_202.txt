axiom obj : Type

-- Entities
axiom AiltonSilva : obj
axiom Ailton : obj
axiom Braga : obj

axiom Fluminense : obj

-- Predicates/Relations
axiom FootballPlayer : obj → Prop
axiom BrazilianFootballer : obj → Prop
axiom PlaysFor : obj → obj → Prop
axiom LoanedTo : obj → obj → Prop
axiom FootballClub : obj → Prop

-- Facts from the context
axiom Alias : AiltonSilva = Ailton
axiom AiltonLoanedBraga : LoanedTo Ailton Braga
axiom AiltonIsFootballPlayer : FootballPlayer Ailton
axiom AiltonSilvaBrazilianFootballer : BrazilianFootballer AiltonSilva

-- Target: Ailton Silva was loaned out to a football club.
theorem ailtonSilva_loaned_to_fc : ∃ c, FootballClub c ∧ LoanedTo AiltonSilva c := by
  have h : LoanedTo AiltonSilva Braga := by simpa [Alias.symm] using AiltonLoanedBraga
  exact Exists.intro Braga (And.intro BragaClub h)
