<lean>
axiom Show : Type
axiom BlackMirror : Show
axiom StrangerThings : Show

axiom Popular : Show → Prop
axiom NetflixShow : Show → Prop
axiom BingeKaren : Show → Prop
axiom DownloadKaren : Show → Prop
axiom ShareWithLisa : Show → Prop

-- "Stranger Things" is a popular Netflix show.
axiom ST_popular : Popular StrangerThings
axiom ST_netflix : NetflixShow StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom binge_of_pop_netflix : ∀ x : Show, NetflixShow x ∧ Popular x → BingeKaren x

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom binge_iff_download_on_netflix : ∀ x : Show, NetflixShow x → (BingeKaren x ↔ DownloadKaren x)

-- Karen does not download "Black Mirror."
axiom not_download_BM : ¬ DownloadKaren BlackMirror

-- "Black Mirror" is a Netflix show.
axiom BM_netflix : NetflixShow BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom share_if_binge_netflix : ∀ x : Show, NetflixShow x → BingeKaren x → ShareWithLisa x

theorem black_mirror_not_popular : ¬ Popular BlackMirror := by
  intro hpop
  have hnet : NetflixShow BlackMirror := BM_netflix
  have hiff : BingeKaren BlackMirror ↔ DownloadKaren BlackMirror :=
    binge_iff_download_on_netflix BlackMirror hnet
  have hb2d : BingeKaren BlackMirror → DownloadKaren BlackMirror := Iff.mp hiff
  have hnotd : ¬ DownloadKaren BlackMirror := not_download_BM
  have hnotb : ¬ BingeKaren BlackMirror := by
    intro hb
    exact hnotd (hb2d hb)
  have hb : BingeKaren BlackMirror := by
    have hconj : NetflixShow BlackMirror ∧ Popular BlackMirror := And.intro hnet hpop
    exact binge_of_pop_netflix BlackMirror hconj
  exact hnotb hb
</lean>
