<lean>
axiom obj : Type
axiom Marvin : obj

axiom Alien : obj → Prop
axiom ET : obj → Prop
axiom Human : obj → Prop
axiom FromEarth : obj → Prop
axiom FromMars : obj → Prop
axiom HighlyIntelligent : obj → Prop

-- 1. All aliens are extraterrestrials.
axiom A1 : ∀ x : obj, Alien x → ET x
-- 2. If someone is from Mars, then they are an alien.
axiom A2 : ∀ x : obj, FromMars x → Alien x
-- 3. No extraterrestrials are human.
axiom A3 : ∀ x : obj, ET x → ¬ Human x
-- 4. All highly intelligent beings from Earth are humans.
axiom A4 : ∀ x : obj, (HighlyIntelligent x ∧ FromEarth x) → Human x
-- 5. Marvin is a highly intelligent being.
axiom A5 : HighlyIntelligent Marvin
-- 6. Marvin is either from Earth and from Mars, or he is from neither.
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
-- 7. If Marvin is not from Earth, then Marvin is an extraterrestrial.
axiom A7 : ¬ FromEarth Marvin → ET Marvin

-- The first disjunct leads to contradiction.
lemma first_contradicts : (FromEarth Marvin ∧ FromMars Marvin) → False := by
  intro h
  have hE : FromEarth Marvin := And.left h
  have hM : FromMars Marvin := And.right h
  have hAlien : Alien Marvin := A2 Marvin hM
  have hET : ET Marvin := A1 Marvin hAlien
  have hNotHuman : ¬ Human Marvin := A3 Marvin hET
  have hHuman : Human Marvin := A4 Marvin (And.intro A5 hE)
  exact hNotHuman hHuman

-- Hence the second disjunct holds: Marvin is from neither Earth nor Mars.
lemma second_holds : (¬ FromEarth Marvin ∧ ¬ FromMars Marvin) := by
  cases A6 with
  | inl hfirst => exact False.elim (first_contradicts hfirst)
  | inr hsecond => exact hsecond

-- From not being from Earth, Marvin is an extraterrestrial.
lemma et_marvin : ET Marvin := by
  have hNotEarth : ¬ FromEarth Marvin := And.left second_holds
  exact A7 hNotEarth

-- Cannot derive Alien Marvin or ¬ Alien Marvin from the given axioms.
-- We have FromMars Marvin is false, but that only blocks one sufficient condition for Alien,
-- and ET Marvin does not imply Alien Marvin.
</lean>
