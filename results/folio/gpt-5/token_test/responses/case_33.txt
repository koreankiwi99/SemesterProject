<lean>
axiom obj : Type

-- Entities
axiom DiamondMine : obj
axiom RoderickStrong : obj
axiom CreedBrothers : obj
axiom IvyNile : obj
axiom Imperium : obj

-- Predicates/Relations
axiom ProWrestlingStable : obj → Prop
axiom Leads : obj → obj → Prop
axiom Includes : obj → obj → Prop
axiom Feud : obj → obj → Prop

-- Given facts
axiom DM_is_stable : ProWrestlingStable DiamondMine
axiom Strong_leads_DM : Leads RoderickStrong DiamondMine
axiom DM_includes_CB : Includes DiamondMine CreedBrothers
axiom DM_includes_IN : Includes DiamondMine IvyNile
axiom Imperium_feuds_DM : Feud Imperium DiamondMine

-- Goal: Roderick Strong leads a professional wrestling stable
theorem strong_leads_a_pro_wrestling_stable : ∃ s : obj, ProWrestlingStable s ∧ Leads RoderickStrong s := by
  exists DiamondMine
  constructor
  · exact DM_is_stable
  · exact Strong_leads_DM
</lean>

ANSWER: True
