axiom Beethoven : obj
axiom Symphony9 : obj
axiom ViennaMusicSociety : obj

axiom MusicPiece : obj → Prop
axiom Composer : obj → Prop
axiom Writes : obj → obj → Prop
axiom Premiered : obj → obj → Prop
axiom Orchestra : obj → Prop
axiom Conductor : obj → Prop
axiom Leads : obj → obj → Prop

-- Symphony No. 9 is a music piece.
axiom S9_piece : MusicPiece Symphony9

-- Composers write music pieces (existential reading).
axiom composers_write : ∀ x : obj, Composer x → ∃ y : obj, MusicPiece y ∧ Writes x y

-- Beethoven wrote Symphony No. 9.
axiom beethoven_wrote_s9 : Writes Beethoven Symphony9

-- Vienna Music Society premiered Symphony No. 9.
axiom vms_premiered_s9 : Premiered ViennaMusicSociety Symphony9

-- Vienna Music Society is an orchestra.
axiom vms_is_orchestra : Orchestra ViennaMusicSociety

-- Beethoven leads the Vienna Music Society.
axiom beethoven_leads_vms : Leads Beethoven ViennaMusicSociety

-- Orchestras are led by conductors (if someone leads an orchestra, they are a conductor).
axiom leader_of_orchestra_is_conductor :
  ∀ x o : obj, Leads x o ∧ Orchestra o → Conductor x

-- From the last two facts, we can deduce Beethoven is a conductor.
theorem beethoven_is_conductor : Conductor Beethoven :=
by
  have h : Leads Beethoven ViennaMusicSociety ∧ Orchestra ViennaMusicSociety :=
    And.intro beethoven_leads_vms vms_is_orchestra
  exact leader_of_orchestra_is_conductor Beethoven ViennaMusicSociety h
