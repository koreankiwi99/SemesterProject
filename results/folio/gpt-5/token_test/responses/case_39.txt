axiom obj : Type

axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom ByMax : obj → Prop
axiom MaxAdores : obj → Prop
axiom Interesting : obj → Prop
axiom Brutalist : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom A1 : ∀ x : obj, Zaha x ∧ MaxAdores x → Interesting x
-- No brutalist buildings that Max adores have interesting geometries.
axiom A2 : ∀ x : obj, Brutalist x ∧ MaxAdores x → ¬ Interesting x
-- Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom A3 : ∀ x : obj, MaxAdores x → Zaha x ∨ Kelly x
-- All of Kelly Wearstler's design styles that Max adores are evocative.
axiom A4 : ∀ x : obj, Kelly x ∧ MaxAdores x → Evocative x
-- All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom A5 : ∀ x : obj, Kelly x ∧ MaxAdores x → Dreamy x
-- If a design by Max that he adores has interesting geometries, then it is a brutalist building and evocative.
axiom A6 : ∀ x : obj, ByMax x ∧ MaxAdores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- A useful consequence: there cannot be a design that is by Max, adored by Max, and a Zaha style.
lemma no_bymax_adored_zaha : ∀ x : obj, ByMax x ∧ MaxAdores x ∧ Zaha x → False :=
by
  intro x h
  have hb : ByMax x := h.left
  have hmz : MaxAdores x ∧ Zaha x := h.right
  have hm : MaxAdores x := hmz.left
  have hz : Zaha x := hmz.right
  have hi : Interesting x := A1 x (And.intro hz hm)
  have hbe : Brutalist x ∧ Evocative x := A6 x (And.intro hb (And.intro hm hi))
  have hbld : Brutalist x := hbe.left
  have hnoti : ¬ Interesting x := A2 x (And.intro hbld hm)
  exact hnoti hi

-- We cannot derive the existence of a design by Max that is a brutalist building,
-- nor can we refute it, from the given axioms.

</lean>
