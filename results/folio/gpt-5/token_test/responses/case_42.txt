<lean>
axiom obj : Type
axiom Coco : obj

axiom RankedWTA : obj → Prop
axiom MostActiveGS : obj → Prop
axiom LostToIgaRG2022 : obj → Prop
axiom LostToRafaRG2022 : obj → Prop
axiom FemaleRG2022 : obj → Prop
axiom MaleRG2022 : obj → Prop
axiom PlayerRG2022 : obj → Prop

axiom A1 : ∀ x : obj, RankedWTA x → MostActiveGS x
axiom A2 : ∀ x : obj, LostToIgaRG2022 x → RankedWTA x
axiom A3 : ∀ x : obj, FemaleRG2022 x → LostToIgaRG2022 x
axiom A4 : ∀ x : obj, PlayerRG2022 x → FemaleRG2022 x ∨ MaleRG2022 x
axiom A5 : ∀ x : obj, MaleRG2022 x → LostToRafaRG2022 x
axiom A6 : (RankedWTA Coco ∨ LostToRafaRG2022 Coco) → ¬ MaleRG2022 Coco
axiom A7 : PlayerRG2022 Coco

theorem coco_most_active : MostActiveGS Coco := by
  have hdisj : FemaleRG2022 Coco ∨ MaleRG2022 Coco := A4 Coco A7
  have hfemale : FemaleRG2022 Coco := by
    cases hdisj with
    | inl hf => exact hf
    | inr hm =>
        have hRafa : LostToRafaRG2022 Coco := A5 Coco hm
        have hNotMale : ¬ MaleRG2022 Coco := A6 (Or.inr hRafa)
        exact False.elim (hNotMale hm)
  have hLostIga : LostToIgaRG2022 Coco := A3 Coco hfemale
  have hRanked : RankedWTA Coco := A2 Coco hLostIga
  exact A1 Coco hRanked
</lean>

ANSWER: True
