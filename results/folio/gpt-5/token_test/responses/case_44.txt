<lean>
axiom obj : Type
axiom Coco : obj

axiom RankedHighlyWTA : obj → Prop
axiom MostActiveMajor : obj → Prop
axiom LostToIgaRG22 : obj → Prop
axiom FemaleRG22 : obj → Prop
axiom MaleRG22 : obj → Prop
axiom LostToNadalRG22 : obj → Prop
axiom AtRG22 : obj → Prop

-- If someone is ranked highly by the WTA, then they are one of the most active players in major tennis.
axiom A1 : ∀ x : obj, RankedHighlyWTA x → MostActiveMajor x
-- Everyone who lost to Iga Świątek at Roland Garros 2022 is ranked highly by the WTA.
axiom A2 : ∀ x : obj, LostToIgaRG22 x → RankedHighlyWTA x
-- All female tennis players at Roland Garros 2022 lost to Iga Świątek.
axiom A3 : ∀ x : obj, FemaleRG22 x → LostToIgaRG22 x
-- All tennis players at Roland Garros 2022 are either female or male.
axiom A4 : ∀ x : obj, AtRG22 x → (FemaleRG22 x ∨ MaleRG22 x)
-- All male tennis players at Roland Garros in 2022 lost to Rafael Nadal.
axiom A5 : ∀ x : obj, MaleRG22 x → LostToNadalRG22 x
-- If Coco Gauff is ranked highly by the WTA or lost to Rafael Nadal, then Coco Gauff is not a male tennis player at RG22.
axiom A6 : (RankedHighlyWTA Coco ∨ LostToNadalRG22 Coco) → ¬ MaleRG22 Coco
-- Coco Gauff is at Roland Garros 2022.
axiom A7 : AtRG22 Coco

-- We show the negation of the claim by proving Coco both lost to Iga and is one of the most active players.
theorem coco_both : LostToIgaRG22 Coco ∧ MostActiveMajor Coco := by
  have hFM : FemaleRG22 Coco ∨ MaleRG22 Coco := A4 Coco A7
  have hHRorN : RankedHighlyWTA Coco ∨ LostToNadalRG22 Coco := by
    cases hFM with
    | inl hF =>
        have hI : LostToIgaRG22 Coco := A3 Coco hF
        have hR : RankedHighlyWTA Coco := A2 Coco hI
        exact Or.inl hR
    | inr hM =>
        have hN : LostToNadalRG22 Coco := A5 Coco hM
        exact Or.inr hN
  have hNotM : ¬ MaleRG22 Coco := A6 hHRorN
  have hF : FemaleRG22 Coco := by
    cases hFM with
    | inl hF => exact hF
    | inr hM =>
        exact False.elim (hNotM hM)
  have hI : LostToIgaRG22 Coco := A3 Coco hF
  have hR : RankedHighlyWTA Coco := A2 Coco hI
  have hMact : MostActiveMajor Coco := A1 Coco hR
  exact And.intro hI hMact
</lean>

ANSWER: False
