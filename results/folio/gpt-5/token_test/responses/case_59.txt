<lean>
axiom obj : Type
axiom City : obj → Prop
axiom State : obj → Prop

axiom Billings : obj
axiom Montana : obj
axiom Butte : obj
axiom Helena : obj
axiom Missoula : obj
axiom WhiteSulphurSprings : obj
axiom StPierre : obj
axiom Bristol : obj
axiom Texarkana : obj
axiom Texhoma : obj
axiom UnionCity : obj

axiom InUSState : obj → obj → Prop

-- Billings is a city in the state of Montana in U.S.
axiom billings_in_montana : InUSState Billings Montana

-- The state of Montana includes the cities of Butte, Helena, and Missoula.
axiom butte_in_montana : InUSState Butte Montana
axiom helena_in_montana : InUSState Helena Montana
axiom missoula_in_montana : InUSState Missoula Montana

-- White Sulphur Springs and Butte are cities in the same state in U.S.
axiom same_state_wss_butte : ∀ s : obj, InUSState WhiteSulphurSprings s ↔ InUSState Butte s

-- The city of St Pierre is not in the state of Montana.
axiom stpierre_not_in_montana : ¬ InUSState StPierre Montana

-- Any city in Butte is not in St Pierre.
axiom in_butte_not_in_stpierre : ∀ c : obj, InUSState c Butte → ¬ InUSState c StPierre

-- A city can only be in one state in U.S. except for Bristol, Texarkana, Texhoma and Union City.
axiom unique_state_except : ∀ c s1 s2 : obj, InUSState c s1 → InUSState c s2 →
  s1 = s2 ∨ c = Bristol ∨ c = Texarkana ∨ c = Texhoma ∨ c = UnionCity

-- Goal: Montana is home to the city of Missoula.
theorem montana_home_to_missoula : InUSState Missoula Montana := missoula_in_montana
</lean>

ANSWER: True
