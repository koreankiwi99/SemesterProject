<lean>
axiom obj : Type
axiom KO : obj

axiom Stock : obj → Prop
axiom Growth : obj → Prop
axiom BoughtForRapidAppreciation : obj → Prop
axiom SuitableForRetirementFund : obj → Prop
axiom Mature : obj → Prop

-- All growth stocks are bought to earn profits from rapid price appreciation.
axiom A1 : ∀ x : obj, Growth x → BoughtForRapidAppreciation x
-- If a stock is bought to earn profits from rapid price appreciation, then it is not suitable for a retirement fund.
axiom A2 : ∀ x : obj, BoughtForRapidAppreciation x → ¬ SuitableForRetirementFund x
-- Some stocks are growth stocks.
axiom A3 : ∃ x : obj, Stock x ∧ Growth x
-- All mature stocks are suitable for a retirement fund.
axiom A4 : ∀ x : obj, Mature x → SuitableForRetirementFund x
-- KO is a mature stock.
axiom KO_stock : Stock KO
axiom KO_mature : Mature KO

theorem ko_is_stock : Stock KO := by
  exact KO_stock
</lean>

ANSWER: True
