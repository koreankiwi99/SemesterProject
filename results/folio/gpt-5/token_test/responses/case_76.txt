<lean>
axiom obj : Type

-- Entities
axiom Hannah : obj

-- Predicates
axiom AtSchool : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom Picky : obj → Prop
axiom SpendTimeDining : obj → Prop

-- Rules
axiom R1 : ∀ x : obj, AtSchool x ∧ Work x → Need x
axiom R2 : ∀ x : obj, AtSchool x ∧ OrderTakeout x → Work x
axiom R3 : ∀ x : obj, AtSchool x → (OrderTakeout x ∨ EnjoyDining x)
axiom R4 : ∀ x : obj, AtSchool x ∧ EnjoyDining x → ¬ Picky x
axiom R5 : ∀ x : obj, AtSchool x ∧ EnjoyDining x → SpendTimeDining x

-- Facts about Hannah
axiom H_school : AtSchool Hannah
axiom H_work : Work Hannah
axiom H_cond : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah)

theorem hannah_goal : AtSchool Hannah ∧ ¬ Picky Hannah ∧ SpendTimeDining Hannah := by
  have hs : AtSchool Hannah := H_school
  have hw : Work Hannah := H_work
  have hneed : Need Hannah := R1 Hannah ⟨hs, hw⟩
  have hcond : ¬ Picky Hannah ∧ ¬ Need Hannah := H_cond hneed
  have hnp : ¬ Picky Hannah := hcond.left
  have hnneed : ¬ Need Hannah := hcond.right
  have contra : False := hnneed hneed
  have hspend : SpendTimeDining Hannah := False.elim contra
  exact And.intro hs (And.intro hnp hspend)
</lean>

ANSWER: True
