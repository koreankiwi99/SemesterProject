<lean>
axiom Restaurant : Type
axiom H : Restaurant

axiom Listed : Restaurant → Prop
axiom Neg : Restaurant → Prop
axiom Takeout : Restaurant → Prop
axiom RatingGT4 : Restaurant → Prop
axiom Popular : Restaurant → Prop

-- 1. If listed in Yelp’s recommendations, then not many negative reviews.
axiom A1 : ∀ x : Restaurant, Listed x → ¬ Neg x
-- 2. All with rating > 4 are listed.
axiom A2 : ∀ x : Restaurant, RatingGT4 x → Listed x
-- 3. Some do not provide take-out and receive many negative reviews.
axiom A3 : ∃ x : Restaurant, ¬ Takeout x ∧ Neg x
-- 4. All popular among locals have ratings > 4.
axiom A4 : ∀ x : Restaurant, Popular x → RatingGT4 x
-- 5. H has rating > 4 or is popular among locals.
axiom A5 : RatingGT4 H ∨ Popular H

theorem hamden_takeout_or_notneg : Takeout H ∨ ¬ Neg H := by
  have RH : RatingGT4 H := by
    cases A5 with
    | inl r => exact r
    | inr p => exact A4 H p
  have LH : Listed H := A2 H RH
  have nn : ¬ Neg H := A1 H LH
  exact Or.inr nn
</lean>

ANSWER: True
