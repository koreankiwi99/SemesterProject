<lean>
axiom obj : Type
axiom Tom : obj

axiom WildTurkey : obj → Prop
axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Goulds : obj → Prop
axiom Merriams : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop

-- Tom's properties
axiom H_wt : WildTurkey Tom
axiom H_ne : ¬ Eastern Tom
axiom H_no : ¬ Osceola Tom
axiom H_ng : ¬ Goulds Tom
axiom H_nm : ¬ Merriams Tom
axiom H_nr : ¬ RioGrande Tom

-- Completeness of the six types for wild turkeys
axiom types_complete :
  ∀ x : obj, WildTurkey x →
    Eastern x ∨ Osceola x ∨ Goulds x ∨ Merriams x ∨ RioGrande x ∨ Ocellated x

theorem tom_is_ocellated : Ocellated Tom := by
  have h1 := types_complete Tom H_wt
  have h2 : Osceola Tom ∨ Goulds Tom ∨ Merriams Tom ∨ RioGrande Tom ∨ Ocellated Tom := by
    apply Or.elim h1
    · intro hE
      exact False.elim (H_ne hE)
    · intro hRest
      exact hRest
  have h3 : Goulds Tom ∨ Merriams Tom ∨ RioGrande Tom ∨ Ocellated Tom := by
    apply Or.elim h2
    · intro hO
      exact False.elim (H_no hO)
    · intro hRest
      exact hRest
  have h4 : Merriams Tom ∨ RioGrande Tom ∨ Ocellated Tom := by
    apply Or.elim h3
    · intro hG
      exact False.elim (H_ng hG)
    · intro hRest
      exact hRest
  have h5 : RioGrande Tom ∨ Ocellated Tom := by
    apply Or.elim h4
    · intro hM
      exact False.elim (H_nm hM)
    · intro hRest
      exact hRest
  have h6 : Ocellated Tom := by
    apply Or.elim h5
    · intro hR
      exact False.elim (H_nr hR)
    · intro hOc
      exact hOc
  exact h6
</lean>

ANSWER: True
