<lean>
axiom obj : Type

-- Entities
axiom OliveGarden : obj

-- Predicates and functions
axiom Managed : obj → Prop
axiom Rent : obj → Nat
axiom Deposit : obj → Nat

-- Facts from the context
axiom Olive_is_managed : Managed OliveGarden
axiom Rent_Olive_eq : Rent OliveGarden = 2000

-- Rule: At any managed building, the security deposit is at least the monthly rent
axiom deposit_ge_rent : ∀ b : obj, Managed b → Rent b ≤ Deposit b

-- Conclusion: The security deposit at the Olive Garden is $2000 or more
theorem olive_deposit_at_least_2000 : 2000 ≤ Deposit OliveGarden := by
  have h : Rent OliveGarden ≤ Deposit OliveGarden := deposit_ge_rent OliveGarden Olive_is_managed
  simpa [Rent_Olive_eq] using h
</lean>

ANSWER: True
