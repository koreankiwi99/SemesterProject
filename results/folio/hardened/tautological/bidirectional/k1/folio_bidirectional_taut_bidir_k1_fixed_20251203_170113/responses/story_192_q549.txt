-- International students in the US have either an F1 visa or a J1 visa.
axiom intl_has_visa : ∀ p : Person, InternationalStudentUS p → (VisaF1 p ∨ VisaJ1 p)

-- An international student in the US with an F1 visa needs to apply for CPT or OPT if the student wants to work in the US.
axiom f1_need_cpt_or_opt_if_work :
  ∀ p : Person, InternationalStudentUS p → VisaF1 p → (WantsWorkUS p → (NeedsCPT p ∨ NeedsOPT p))

-- Mike is an international student.

-- Mike needs to apply for CPT if he wants to work in the US.
axiom Mike_need_cpt_if_work : WantsWorkUS Mike → NeedsCPT Mike

-- Conclusion to analyze: Mike has a J1 visa.
theorem goal : VisaJ1 Mike := by
  have hv : VisaF1 Mike ∨ VisaJ1 Mike := intl_has_visa Mike Mike_is_intl
  cases hv with
  | inl hf1 =>
      -- From hf1 (Mike has F1), we can derive:
      -- If Mike wants to work, then NeedsCPT Mike ∨ NeedsOPT Mike (from f1_need_cpt_or_opt_if_work),
      -- and also NeedsCPT Mike (from Mike_need_cpt_if_work).
      -- These are consistent and do not yield a contradiction, so we cannot rule out F1.
      sorry
  | inr hj1 =>
      exact hj1
