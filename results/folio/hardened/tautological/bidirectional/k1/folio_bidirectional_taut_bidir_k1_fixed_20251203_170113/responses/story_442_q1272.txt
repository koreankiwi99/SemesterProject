/- Useful derived facts -/

by
  intro hBS
  have hCow : Cow x := all_BS_are_cows x hBS
  exact cows_domesticated x hCow

lemma not_BS_Ted : ¬ BrownSwissCattle Ted :=
by
  intro hBSTed
  have hDomTed : Domesticated Ted := BS_implies_domesticated Ted hBSTed
  have hNotDomTed : ¬ Domesticated Ted := alligators_not_domesticated Ted ted_is_alligator
  exact hNotDomTed hDomTed

-/

  exact not_BS_Ted h.right
