axiom premise2 : ∀ x : Being, FromMars x → Alien x                              -- If someone is from Mars, then they are an alien
axiom premise3 : ∀ x : Being, Extraterrestrial x → ¬Human x                     -- No extraterrestrials are human
axiom premise4 : ∀ x : Being, HighlyIntelligent x ∧ FromEarth x → Human x       -- All highly intelligent beings from Earth are humans
axiom premise5 : HighlyIntelligent Marvin                                       -- Marvin is a highly intelligent being
axiom premise6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬FromEarth Marvin ∧ ¬FromMars Marvin)
-- Marvin is either from Earth and from Mars, or from neither
axiom premise7 : ¬FromEarth Marvin → Extraterrestrial Marvin                    -- If Marvin is not from Earth, then he is an extraterrestrial

-- Goal: Prove the negation of the conclusion
-- Conclusion to analyze: If Marvin is not from Mars, then Marvin is a human.
-- We prove: ¬(¬FromMars Marvin → Human Marvin)
theorem negate_conclusion : ¬(¬FromMars Marvin → Human Marvin) := by
  intro hImp
  -- Split on Marvin's origin disjunction
  cases premise6 with
  | inl hEM =>
      -- Case: FromEarth Marvin ∧ FromMars Marvin
      have hE : FromEarth Marvin := hEM.left
      have hM : FromMars Marvin := hEM.right
      -- From HI and FromEarth, Marvin is Human
      have hHI : HighlyIntelligent Marvin := premise5
      have hHuman : Human Marvin := premise4 Marvin ⟨hHI, hE⟩
      -- From Mars -> Alien -> Extraterrestrial -> ¬Human, contradiction
      have hAlien : Alien Marvin := premise2 Marvin hM
      have hET : Extraterrestrial Marvin := premise1 Marvin hAlien
      have hNotHuman : ¬Human Marvin := premise3 Marvin hET
      exact hNotHuman hHuman
  | inr hneither =>
      -- Case: ¬FromEarth Marvin ∧ ¬FromMars Marvin
      have hNotE : ¬FromEarth Marvin := hneither.left
      have hNotM : ¬FromMars Marvin := hneither.right
      -- From the assumed implication, Human follows from ¬FromMars
      have hHuman : Human Marvin := hImp hNotM
      -- From ¬FromEarth, Marvin is ET, hence not Human, contradiction
      have hET : Extraterrestrial Marvin := premise7 hNotE
      have hNotHuman : ¬Human Marvin := premise3 Marvin hET
      exact hNotHuman hHuman
