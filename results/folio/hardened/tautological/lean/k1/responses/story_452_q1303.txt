lemma marsImpliesExtraterrestrial (x : Being) : FromMars x → Extraterrestrial x := by
  intro hfm

lemma not_both_earth_and_mars : ¬ (FromEarth Marvin ∧ FromMars Marvin) := by

  exact hNotHuman hHuman
