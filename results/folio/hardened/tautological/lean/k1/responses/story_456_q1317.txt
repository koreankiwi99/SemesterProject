-- Some professional basketball players are not American nationals.
axiom some_pro_not_american : ∃ x : Person, ProPlayer x ∧ ¬ American x

-- All professional basketball players can do jump shots.
axiom all_pro_can_jump : ∀ x : Person, ProPlayer x → CanJumpShot x

-- If someone can jump shots, they leap straight into the air.
axiom jump_implies_leap : ∀ x : Person, CanJumpShot x → LeapsStraight x

-- If someone leaps straight into the air, they activate their leg muscles.
axiom leap_implies_activate : ∀ x : Person, LeapsStraight x → ActivatesLegs x

-- Yuri does not activate his leg muscles.
axiom yuri_not_activate : ¬ ActivatesLegs Yuri

-- If Yuri is not an American professional basketball player, then Yuri is a professional basketball player.
axiom yuri_conditional : ¬ (American Yuri ∧ ProPlayer Yuri) → ProPlayer Yuri

-- From the premises, derive a contradiction.
theorem contradiction : False := by
  -- From the universal implications, any professional activates leg muscles.
  have pro_implies_activate : ∀ x : Person, ProPlayer x → ActivatesLegs x := by
    intro x hx
    have hj : CanJumpShot x := all_pro_can_jump x hx
    have hl : LeapsStraight x := jump_implies_leap x hj
    exact leap_implies_activate x hl
  -- Therefore, Yuri cannot be a professional.
  have not_pro_yuri : ¬ ProPlayer Yuri := by
    intro hy
    have act : ActivatesLegs Yuri := pro_implies_activate Yuri hy
    exact yuri_not_activate act
  -- Hence, it is not the case that Yuri is both American and a professional.
  have not_am_and_pro : ¬ (American Yuri ∧ ProPlayer Yuri) := by
    intro h
    exact not_pro_yuri h.right
  -- By the given conditional, Yuri must be a professional, contradiction.
  have pro_yuri : ProPlayer Yuri := yuri_conditional not_am_and_pro
  exact not_pro_yuri pro_yuri
