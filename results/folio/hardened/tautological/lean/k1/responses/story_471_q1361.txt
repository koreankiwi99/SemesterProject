-- If something is skittish, then it is not calm.
axiom A4 : ∀ x : Thing, Skittish x → ¬ Calm x
-- All the squirrels that can be spotted near the campus are skittish.
axiom A5 : ∀ x : Thing, (Squirrel x ∧ Spotted x) → Skittish x
-- Rockie can be spotted near the campus, and it is calm.
axiom A6 : Spotted Rockie ∧ Calm Rockie

-- From the premises we can derive that Rockie is not a squirrel.
theorem not_squirrel_Rockie : ¬ Squirrel Rockie := by
  intro hSqr
  have hSpot : Spotted Rockie := And.left A6
  have hSk : Skittish Rockie := A5 Rockie ⟨hSqr, hSpot⟩
  have hNotCalm : ¬ Calm Rockie := A4 Rockie hSk
  exact hNotCalm (And.right A6)

-- Hence Rockie must be a rabbit (given only rabbits or squirrels are spotted).
theorem rabbit_Rockie : Rabbit Rockie := by
  have hSpot : Spotted Rockie := And.left A6
  have h := A3 Rockie hSpot
  cases h with
  | inl hRabbit => exact hRabbit
  | inr hSquirrel =>
      exact (False.elim (not_squirrel_Rockie hSquirrel))

-- Consequently, Rockie is cute.
theorem cute_Rockie : Cute Rockie := by
  have hSpot : Spotted Rockie := And.left A6
  have hRab : Rabbit Rockie := rabbit_Rockie
  exact A1 Rockie ⟨hRab, hSpot⟩
