axiom ax_animal_partition : ∀ x, Animal x → (Invertebrate x ∨ Vertebrate x)

-- 2. All animals with backbones reproduce by male-and-female mating.
axiom ax_animals_with_backbones_reproduce :
  ∀ x, (Animal x ∧ HasBackbone x) → ReproByMaleFemaleMating x

-- 3. All vertebrate animals have a backbone.
axiom ax_vertebrate_animals_have_backbone :
  ∀ x, (Animal x ∧ Vertebrate x) → HasBackbone x

-- 4. All bees do not reproduce by male-and-female mating.
axiom ax_bees_not_reproduce_by_mfm :
  ∀ x, Bee x → ¬ ReproByMaleFemaleMating x

-- 5. All queen bees are bees.
axiom ax_qb_are_bees :
  ∀ x, QueenBee x → Bee x

-- 6. Harry is a bee.
axiom ax_harry_is_bee : Bee Harry

-- A useful derived lemma: if Harry is an animal, then Harry is an invertebrate.
lemma harry_invertebrate_if_animal : Animal Harry → Invertebrate Harry := by
  intro hAnimal
  have hPartition : Invertebrate Harry ∨ Vertebrate Harry :=
    ax_animal_partition Harry hAnimal
  have hNotRepro : ¬ ReproByMaleFemaleMating Harry :=
    ax_bees_not_reproduce_by_mfm Harry ax_harry_is_bee
  cases hPartition with
  | inl hInv =>
      exact hInv
  | inr hVert =>
      -- From Animal ∧ Vertebrate, Harry has a backbone
      have hBack : HasBackbone Harry :=
        ax_vertebrate_animals_have_backbone Harry ⟨hAnimal, hVert⟩
      -- From Animal ∧ Backbone, Harry reproduces by MFM
      have hRepro : ReproByMaleFemaleMating Harry :=
        ax_animals_with_backbones_reproduce Harry ⟨hAnimal, hBack⟩
      -- Contradiction with bees not reproducing by MFM
      exact False.elim (hNotRepro hRepro)

-- If we knew Harry is an animal, we could conclude the target disjunction.
theorem goal_of_animal : Animal Harry → (Invertebrate Harry ∨ QueenBee Harry) := by
  intro hAnimal
  have hInv : Invertebrate Harry := harry_invertebrate_if_animal hAnimal
  exact Or.inl hInv
