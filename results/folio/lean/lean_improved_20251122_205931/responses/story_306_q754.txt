Example ID: 754
Premises: If a restaurant is listed in Yelp’s recommendations, then the restaurant has not received many negative reviews.
All restaurants with a rating greater than four are listed in Yelp’s recommendations.
Some restaurants that do not provide take-out service receive many negative reviews.
All restaurants that are popular among local residents have ratings greater than four.
The Hamden Plaza Subway store has a rating greater than four, or it is popular among local residents.
Conclusion: If the Hamden Plaza Subway store provides take-out service and receives many negative reviews, then its rating is greater than 4 and it does not provide take-out service.

constant Popular : Restaurant → Prop

axiom p1 : ∀ r : Restaurant, Listed r → ¬ NegRev r
axiom p2 : ∀ r : Restaurant, RatingGT4 r → Listed r
axiom p3 : ∃ r : Restaurant, ¬ TakeOut r ∧ NegRev r
axiom p4 : ∀ r : Restaurant, Popular r → RatingGT4 r
axiom p5 : RatingGT4 HamdenPlazaSubway ∨ Popular HamdenPlazaSubway

theorem target :
  (TakeOut HamdenPlazaSubway ∧ NegRev HamdenPlazaSubway) →
  (RatingGT4 HamdenPlazaSubway ∧ ¬ TakeOut HamdenPlazaSubway) := by
  intro h
  -- Derive RatingGT4 for HPS from the disjunction and popularity implication
  have hr : RatingGT4 HamdenPlazaSubway := by
    cases p5 with
    | inl hr => exact hr
    | inr hp => exact p4 _ hp
  -- From rating > 4, derive listed, hence not many negative reviews
  have hnnr : ¬ NegRev HamdenPlazaSubway := by
    have hl : Listed HamdenPlazaSubway := p2 _ hr
    exact p1 _ hl
  -- The assumption gives NegRev, contradicting hnnr
  have contra : False := by
    exact hnnr (And.right h)
  -- Conclude the required conjunction using ex falso for ¬TakeOut
  apply And.intro
  · exact hr
  · exact False.elim contra
