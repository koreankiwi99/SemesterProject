Example ID: 1269
Premises: No one nice to animals is also mean to animals.
Some grumpy people are mean to animals.
All animal lovers are nice to animals.
All pet owners love animals.
Tom is a pet owner.
Conclusion: Tom is not both a grumpy person and mean to animals.

==================================================
<lean>
constant Person : Type

constant Nice : Person → Prop
constant Mean : Person → Prop
constant Grumpy : Person → Prop
constant Lover : Person → Prop
constant PetOwner : Person → Prop

-- Premises
axiom nice_not_mean : ∀ x : Person, Nice x → ¬ Mean x
axiom some_grumpy_mean : ∃ x : Person, Grumpy x ∧ Mean x
axiom lover_nice : ∀ x : Person, Lover x → Nice x
axiom owner_lover : ∀ x : Person, PetOwner x → Lover x

constant Tom : Person
axiom Tom_owner : PetOwner Tom

theorem Tom_not_grumpy_and_mean : ¬ (Grumpy Tom ∧ Mean Tom) := by
  have h1 : Lover Tom := owner_lover Tom Tom_owner
  have h2 : Nice Tom := lover_nice Tom h1
  have h3 : ¬ Mean Tom := nice_not_mean Tom h2
  intro h
  exact h3 h.right
