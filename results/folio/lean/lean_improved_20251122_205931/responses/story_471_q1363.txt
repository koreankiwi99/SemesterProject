Premises: All rabbits that can be spotted near the campus are cute. 
Some turtles can be spotted near the campus. 
The only animals that can be spotted near the campus are rabbits and squirrels.
If something is skittish, then it is not calm.
All the squirrels that can be spotted near the campus are skittish.
Rockie can be spotted near the campus, and it is calm.
Conclusion: Rockie is a turtle or cute.

constant Rabbit : U → Prop

constant Turtle : U → Prop

constant Calm : U → Prop

constant rockie : U

-- Premises
axiom a1 : ∀ x : U, Rabbit x ∧ Near x → Cute x
axiom a2 : ∃ x : U, Turtle x ∧ Near x
axiom a3 : ∀ x : U, Near x → (Rabbit x ∨ Squirrel x)
axiom a4 : ∀ x : U, Skittish x → ¬ Calm x
axiom a5 : ∀ x : U, Squirrel x ∧ Near x → Skittish x
axiom a6 : Near rockie ∧ Calm rockie

theorem goal : Turtle rockie ∨ Cute rockie :=
by
  have hNear : Near rockie := a6.left
  have hCalm : Calm rockie := a6.right
  have disj : Rabbit rockie ∨ Squirrel rockie := a3 rockie hNear
  have notSquirrel : ¬ Squirrel rockie := by
    intro hS
    have hSk : Skittish rockie := a5 rockie ⟨hS, hNear⟩
    have hNotCalm : ¬ Calm rockie := a4 rockie hSk
    exact hNotCalm hCalm
  have hCute : Cute rockie :=
    Or.elim disj
      (fun hR => a1 rockie ⟨hR, hNear⟩)
      (fun hS => False.elim (notSquirrel hS))
  exact Or.inr hCute
