Example ID: 1033
Premises: If something is a deadly disease, then it comes with a low survival rate. 
Severe cancers are deadly diseases.
Bile duct cancer is a severe form cancer. 
All Cholangiocarcinoma is bile duct cancer.
Mild flu comes with a low survival rate.
Colorectal cancer is not both a bile duct cancer and with a low survival rate.
Conclusion: If colorectal cancer is a kind of bile duct cancer or a form of Cholangiocarcinoma, then colorectal cancer is a kind of bile duct cancer and a kind of mild flu.

axiom mildFlu_implies_low : ∀ x : Disease, MildFlu x → LowSurvival x
axiom not_bileDuct_and_low_for_colorectal :
  ¬ (BileDuctCancer ColorectalCancer ∧ LowSurvival ColorectalCancer)

-- Target theorem
theorem target :
  (BileDuctCancer ColorectalCancer ∨ Cholangiocarcinoma ColorectalCancer) →
  (BileDuctCancer ColorectalCancer ∧ MildFlu ColorectalCancer) := by
  intro h
  have hb : BileDuctCancer ColorectalCancer := by
    cases h with
    | inl hb => exact hb
    | inr hc => exact cholangio_implies_bileDuct ColorectalCancer hc
  have hs : SevereCancer ColorectalCancer :=
    bileDuct_implies_severe ColorectalCancer hb
  have hd : Deadly ColorectalCancer :=
    severe_implies_deadly ColorectalCancer hs
  have hl : LowSurvival ColorectalCancer :=
    deadly_implies_low ColorectalCancer hd
  have contra : False :=
    not_bileDuct_and_low_for_colorectal ⟨hb, hl⟩
  exact False.elim contra
