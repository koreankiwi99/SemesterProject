Premises: All rabbits that can be spotted near the campus are cute. 
Some turtles can be spotted near the campus. 
The only animals that can be spotted near the campus are rabbits and squirrels.
If something is skittish, then it is not calm.
All the squirrels that can be spotted near the campus are skittish.
Rockie can be spotted near the campus, and it is calm.
Conclusion: Rockie is a turtle or cute.

axiom Turtle : Entity → Prop
axiom Squirrel : Entity → Prop

axiom Rockie : Entity

axiom all_rabbits_spotted_cute : ∀ x : Entity, (Rabbit x ∧ Spotted x) → Cute x
axiom some_turtles_spotted : ∃ x : Entity, Turtle x ∧ Spotted x
axiom only_spotted_are_rabbit_or_squirrel : ∀ x : Entity, Spotted x → (Rabbit x ∨ Squirrel x)
axiom skittish_not_calm : ∀ x : Entity, Skittish x → ¬ Calm x
axiom squirrels_spotted_skittish : ∀ x : Entity, (Squirrel x ∧ Spotted x) → Skittish x
axiom rockie_spotted_calm : Spotted Rockie ∧ Calm Rockie

theorem goal : Turtle Rockie ∨ Cute Rockie := by
  have h_spotted : Spotted Rockie := rockie_spotted_calm.left
  have h_calm : Calm Rockie := rockie_spotted_calm.right
  have h_rs : Rabbit Rockie ∨ Squirrel Rockie := only_spotted_are_rabbit_or_squirrel Rockie h_spotted
  have h_cute : Cute Rockie := by
    cases h_rs with
    | inl h_rab =>
        exact all_rabbits_spotted_cute Rockie ⟨h_rab, h_spotted⟩
    | inr h_sq =>
        have h_skit : Skittish Rockie := squirrels_spotted_skittish Rockie ⟨h_sq, h_spotted⟩
        have h_not_calm : ¬ Calm Rockie := skittish_not_calm Rockie h_skit
        exact False.elim (h_not_calm h_calm)
  exact Or.inr h_cute
