Premises: Everyone who can register to vote in the United States can participate in the 2024 United States presidential election.
If someone has United States citizenship, then they can register to vote in the United States.
A person either has United States citizenship or Taiwanese citizenship.
No Russian Federation officials hold Taiwanese citizenship. 
Vladimir neither holds Taiwanese citizenship nor is he a manager at Gazprom. 
Ekaterina she can register to vote in the United States, or she is a Russian federation official.
Conclusion: Ekaterina can participate in the 2024 United States presidential election or Vladimir is a manager at Gazprom.

axiom Person : Type
axiom Ekaterina : Person
axiom Vladimir : Person

axiom CanRegisterUS : Person → Prop
axiom CanParticipate2024 : Person → Prop
axiom USCitizenship : Person → Prop
axiom TaiwaneseCitizenship : Person → Prop
axiom RussianOfficial : Person → Prop

-- Premises
axiom reg_implies_part : ∀ x : Person, CanRegisterUS x → CanParticipate2024 x
axiom us_implies_reg : ∀ x : Person, USCitizenship x → CanRegisterUS x
axiom us_or_taiw : ∀ x : Person, USCitizenship x ∨ TaiwaneseCitizenship x
axiom no_official_taiw : ∀ x : Person, RussianOfficial x → ¬ TaiwaneseCitizenship x
axiom vladimir_not_taiw : ¬ TaiwaneseCitizenship Vladimir
axiom vladimir_not_manager : ¬ GazpromManager Vladimir
axiom ekaterina_reg_or_official : CanRegisterUS Ekaterina ∨ RussianOfficial Ekaterina

theorem goal : CanParticipate2024 Ekaterina ∨ GazpromManager Vladimir := by
  have hParticipateE : CanParticipate2024 Ekaterina := by
    cases ekaterina_reg_or_official with
    | inl hreg =>
        exact reg_implies_part Ekaterina hreg
    | inr hOff =>
        have hNotTW : ¬ TaiwaneseCitizenship Ekaterina := no_official_taiw Ekaterina hOff
        have hUSorTW : USCitizenship Ekaterina ∨ TaiwaneseCitizenship Ekaterina := us_or_taiw Ekaterina
        have hUS : USCitizenship Ekaterina := by
          cases hUSorTW with
          | inl hUS => exact hUS
          | inr hTW => exact False.elim (hNotTW hTW)
        have hReg : CanRegisterUS Ekaterina := us_implies_reg Ekaterina hUS
        exact reg_implies_part Ekaterina hReg
  exact Or.inl hParticipateE
