Juan de Onate visited the Picuris Mountains.

Conclusion: Juan de Onate visited a mountain range in Texas.

-- Premises:
-- 1. The Picuris Mountains are a mountain range in New Mexico or Texas.
axiom ax1 : MountainRange Picuris
axiom ax2 : InNewMexico Picuris ∨ InTexas Picuris

-- 2. Juan de Onate visited the Picuris Mountains.
axiom ax3 : Visited Juan Picuris

-- 3. The Harding Pegmatite Mine, located in the Picuris Mountains, was donated.
axiom ax4 : Mine HardingMine
axiom ax5 : LocatedIn HardingMine Picuris
axiom ax6 : Donated HardingMine

-- 4. There are no mountain ranges in Texas that have mines that have been donated.
axiom ax7 : ∀ r, (MountainRange r ∧ InTexas r) → ¬ (∃ m, Mine m ∧ LocatedIn m r ∧ Donated m)

-- A derived fact: Picuris is not in Texas (since it has a donated mine located in it).
theorem notInTexas_Picuris : ¬ InTexas Picuris := by
  intro hTX
  have hMTX : MountainRange Picuris ∧ InTexas Picuris := And.intro ax1 hTX
  have hNo : ¬ (∃ m, Mine m ∧ LocatedIn m Picuris ∧ Donated m) := ax7 Picuris hMTX
  have hEx : ∃ m, Mine m ∧ LocatedIn m Picuris ∧ Donated m :=
    Exists.intro HardingMine (And.intro ax4 (And.intro ax5 ax6))
  exact hNo hEx

-- Hence, from "In New Mexico or Texas" and not Texas, we get In New Mexico.
theorem inNewMexico_Picuris : InNewMexico Picuris := by
  cases ax2 with
  | inl hNM => exact hNM
  | inr hTX => exact False.elim (notInTexas_Picuris hTX)
