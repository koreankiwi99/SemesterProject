Premises: Machine Learning algorithms can be categorized as supervised learning, unsupervised learning, and reinforcement learning.
Unsupervised learning algorithms do not require labeled data.
The state-of-the-art text summarization model is trained with machine learning algorithms.
Reinforcement learning is not used to train the state-of-the-art text summarization model.
The Machine Learning algorithm for training text summarization models requires labeled data.
Conclusion: Supervised learning is used to train the state-of-the-art text summarization model.

universe u

constant Alg : Type u
constant Model : Type u

-- Predicates
constant ML : Alg → Prop
constant Supervised : Alg → Prop
constant Unsupervised : Alg → Prop
constant Reinforcement : Alg → Prop
constant RequiresLabels : Alg → Prop
constant UsedToTrain : Alg → Model → Prop
constant TSM : Model → Prop

-- Constant for the state-of-the-art text summarization model
constant SOTA : Model

-- Premises
axiom categorization :
  ∀ a : Alg, ML a → (Supervised a ∨ Unsupervised a ∨ Reinforcement a)

axiom unsupervised_no_labels :
  ∀ a : Alg, Unsupervised a → ¬ RequiresLabels a

axiom sota_is_tsm : TSM SOTA

axiom trained_with_ml :
  ∃ a : Alg, ML a ∧ UsedToTrain a SOTA

axiom no_rl_used :
  ∀ a : Alg, UsedToTrain a SOTA → ¬ Reinforcement a

axiom training_alg_requires_labels :
  ∀ a : Alg, ∀ m : Model, ML a → UsedToTrain a m → TSM m → RequiresLabels a

-- Conclusion: Supervised learning is used to train the SOTA text summarization model.
theorem supervised_used_for_sota :
  ∃ a : Alg, UsedToTrain a SOTA ∧ Supervised a :=
by
  obtain ⟨a, haML, haUsed⟩ := trained_with_ml
  have haReq : RequiresLabels a :=
    training_alg_requires_labels a SOTA haML haUsed sota_is_tsm
  have hNotUnsup : ¬ Unsupervised a :=
  by
    intro hU
    have hNoReq : ¬ RequiresLabels a := unsupervised_no_labels a hU
    exact hNoReq haReq
  have hNotRL : ¬ Reinforcement a := no_rl_used a haUsed
  have hcat := categorization a haML
  -- From the categorization and eliminating the other two options, get Supervised a
  have hSup : Supervised a :=
  by
    cases hcat with
    | inl hS => exact hS
    | inr h =>
      cases h with
      | inl hU => exact (False.elim (hNotUnsup hU))
      | inr hR => exact (False.elim (hNotRL hR))
  exact ⟨a, And.intro haUsed hSup⟩
