constant Person : Type

constant CanBlock : Person → Person → Prop
constant JumpsWhenShooting : Person → Prop
constant GreatShooter : Person → Prop

-- Premise: Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom block_if_not_jump : ∀ p : Person, ¬ JumpsWhenShooting p → CanBlock Michael p

-- Premise: Michael cannot block Windy's shooting.
axiom cannot_block_windy : ¬ CanBlock Michael Windy

-- Premise: Every shooter who can jump when shooting is a great shooter.
-- We model "can jump when shooting" directly as the antecedent.
axiom great_if_jump : ∀ p : Person, JumpsWhenShooting p → GreatShooter p

-- Conclusion: Windy is a great shooter.
theorem windy_is_great : GreatShooter Windy :=
by
  have hjump : JumpsWhenShooting Windy :=
    not_not.mp (mt (block_if_not_jump Windy) cannot_block_windy)
  exact great_if_jump Windy hjump
