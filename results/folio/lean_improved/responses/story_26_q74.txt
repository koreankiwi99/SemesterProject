Premises: Some mammals have teeth.
Platypuses have no teeth.
Platypuses are mammals. 
Humans have teeth.
Conclusion: Platypuses are mammals with no teeth.

constant U : Type u

-- Predicates
constant Mammal : U → Prop
constant HasTeeth : U → Prop
constant Platypus : U → Prop
constant Human : U → Prop

-- Given premises
axiom some_mammals_have_teeth : ∃ x : U, Mammal x ∧ HasTeeth x
axiom platypuses_have_no_teeth : ∀ x : U, Platypus x → ¬ HasTeeth x
axiom platypuses_are_mammals : ∀ x : U, Platypus x → Mammal x
axiom humans_have_teeth : ∀ x : U, Human x → HasTeeth x

-- Conclusion: Platypuses are mammals with no teeth
theorem platypuses_are_mammals_with_no_teeth :
  ∀ x : U, Platypus x → (Mammal x ∧ ¬ HasTeeth x) :=
by
  intro x hx
  have hm : Mammal x := (platypuses_are_mammals x) hx
  have hnt : ¬ HasTeeth x := (platypuses_have_no_teeth x) hx
  exact And.intro hm hnt
</lean>
