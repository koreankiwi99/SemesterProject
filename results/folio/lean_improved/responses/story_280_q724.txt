At least one event is happy. 
Conclusion: All events are sad.

constant Event : Type u
constant Happy Sad : Event → Prop

-- Premises
axiom happy_or_sad : ∀ e : Event, Happy e ∨ Sad e
axiom exists_happy : ∃ e : Event, Happy e
axiom all_sad : ∀ e : Event, Sad e

-- Conclusion: There exists an event that is both happy and sad
theorem exists_happy_and_sad : ∃ e : Event, Happy e ∧ Sad e :=
by
  rcases exists_happy with ⟨e, he⟩
  have hs : Sad e := all_sad e
  exact ⟨e, And.intro he hs⟩
