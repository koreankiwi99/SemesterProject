Premises: If people spill a lot of food on their clothing, then they are not notably tidy.
All clumsy foodies who go out frequently to find new food restaurants to try are people who spill a lot of food on their clothing.
If people are cleanly, then they are notably tidy.
If people value order and spotlessness, then they are cleanly.
All people whose families prioritize order and spotlessness themselves value order and spotlessness.
Peter is either a person who spills a lot of food on his clothing and is cleanly, or a person who does not spill a lot of food on his clothing and is not cleanly
Conclusion: Peter is either both a clumsy foodie who goes out frequently to find new food restaurants to try and issomeone whose family prioritizes order and spotlessness, or he neither is a clumsy foodie who goes out frequently to find new food restaurants to try nor does his family prioritize order and spotlessness.

constant Person : Type u

-- Predicates
constant Spills : Person → Prop
constant Tidy : Person → Prop
constant Cleanly : Person → Prop
constant ValuesOrder : Person → Prop
constant FamilyPrioritizesOrder : Person → Prop
constant ClumsyFoodieWhoGoesOut : Person → Prop

-- Individual
constant Peter : Person

-- Premises
axiom A1 : ∀ x : Person, Spills x → ¬ Tidy x
axiom A2 : ∀ x : Person, ClumsyFoodieWhoGoesOut x → Spills x
axiom A3 : ∀ x : Person, Cleanly x → Tidy x
axiom A4 : ∀ x : Person, ValuesOrder x → Cleanly x
axiom A5 : ∀ x : Person, FamilyPrioritizesOrder x → ValuesOrder x
axiom A6 : (Spills Peter ∧ Cleanly Peter) ∨ (¬ Spills Peter ∧ ¬ Cleanly Peter)

-- Conclusion to prove:
-- Peter is either both a clumsy foodie who goes out and from a family that prioritizes order,
-- or neither.
theorem Peter_both_or_neither :
  (ClumsyFoodieWhoGoesOut Peter ∧ FamilyPrioritizesOrder Peter) ∨
  (¬ ClumsyFoodieWhoGoesOut Peter ∧ ¬ FamilyPrioritizesOrder Peter) :=
by
  -- First, show Spills Peter ∧ Cleanly Peter is impossible (would imply Tidy and ¬Tidy).
  have h_not_SC : ¬ (Spills Peter ∧ Cleanly Peter) := by
    intro h
    have s : Spills Peter := h.left
    have c : Cleanly Peter := h.right
    have nt : ¬ Tidy Peter := (A1 Peter) s
    have t  : Tidy Peter := (A3 Peter) c
    exact nt t

  -- From the given disjunction, deduce ¬Spills Peter ∧ ¬Cleanly Peter.
  have h_ns_nc : ¬ Spills Peter ∧ ¬ Cleanly Peter := by
    cases A6 with
    | inl hsc => exact False.elim (h_not_SC hsc)
    | inr h    => exact h

  have ns : ¬ Spills Peter := h_ns_nc.left
  have nc : ¬ Cleanly Peter := h_ns_nc.right

  -- Derive ¬ClumsyFoodieWhoGoesOut Peter from ns and A2.
  have ng : ¬ ClumsyFoodieWhoGoesOut Peter := by
    intro g
    have s : Spills Peter := (A2 Peter) g
    exact ns s

  -- Derive ¬FamilyPrioritizesOrder Peter from nc and A4∘A5.
  have nf : ¬ FamilyPrioritizesOrder Peter := by
    intro f
    have v : ValuesOrder Peter := (A5 Peter) f
    have c : Cleanly Peter := (A4 Peter) v
    exact nc c

  -- Conclude the right disjunct.
  exact Or.inr ⟨ng, nf⟩
