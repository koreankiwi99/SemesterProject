Story ID: 363
Example ID: 966
Premises: All hydrocarbons are organic compounds .
All alkanes are hydrocarbons
All organic compounds are chemical compounds.
All organic compounds contain carbon.
No chemical compounds contain only one element.
The mixture either both is a chemical compound and contains only one element, or neither. 
Conclusion: If the mixture contains only one element or contains carbon, then the mixture is neither a chemical compound nor an alkane.

universe u

constant Thing : Type u

constant Hydrocarbon : Thing → Prop
constant Organic : Thing → Prop
constant Alkane : Thing → Prop
constant Chemical : Thing → Prop
constant ContainsCarbon : Thing → Prop
constant OnlyOneElement : Thing → Prop

constant mixture : Thing

axiom h1 : ∀ x : Thing, Hydrocarbon x → Organic x
axiom h2 : ∀ x : Thing, Alkane x → Hydrocarbon x
axiom h3 : ∀ x : Thing, Organic x → Chemical x
axiom h4 : ∀ x : Thing, Organic x → ContainsCarbon x
axiom h5 : ∀ x : Thing, Chemical x → ¬ OnlyOneElement x
axiom h6 : (Chemical mixture ∧ OnlyOneElement mixture) ∨ (¬ Chemical mixture ∧ ¬ OnlyOneElement mixture)

theorem result :
  (OnlyOneElement mixture ∨ ContainsCarbon mixture) → (¬ Chemical mixture ∧ ¬ Alkane mixture) := by
  intro _
  have hNotChem : ¬ Chemical mixture := by
    intro hChem
    have hNotOne : ¬ OnlyOneElement mixture := h5 _ hChem
    cases h6 with
    | inl hBoth =>
        exact hNotOne hBoth.right
    | inr hNeither =>
        exact hNeither.left hChem
  have hNotAlk : ¬ Alkane mixture := by
    intro hAlk
    have hChem : Chemical mixture := h3 _ (h1 _ (h2 _ hAlk))
    exact hNotChem hChem
  exact And.intro hNotChem hNotAlk
