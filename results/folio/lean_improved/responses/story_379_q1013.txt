Example ID: 1013
Premises: People who are born in a multiple birth with siblings spend a lot of time hanging out with and playing with their siblings.
If people have siblings who were born together, then they were born in a multiple birth.
If people complain often about annoying siblings, then they have siblings who were born together.
If people live at home, then they do not live with strangers.
If people spend a lot of time hanging out with and playing with their siblings, then they often live at home. 
Luke either is a baby born in a multiple birth and live with strangers, or is not a baby born in a multiple birth and does not live with strangers
Conclusion: Luke is neither born in a multiple birth nor complains often about annoying siblings.

constant Person : Type

constant BornInMultipleBirth : Person → Prop
constant SiblingsBornTogether : Person → Prop
constant ComplainOften : Person → Prop
constant SpendTimeWithSiblings : Person → Prop
constant LiveAtHome : Person → Prop
constant LiveWithStrangers : Person → Prop
constant Baby : Person → Prop
constant Luke : Person

-- Premises
axiom prem1 : ∀ x : Person, BornInMultipleBirth x → SpendTimeWithSiblings x
axiom prem2 : ∀ x : Person, SiblingsBornTogether x → BornInMultipleBirth x
axiom prem3 : ∀ x : Person, ComplainOften x → SiblingsBornTogether x
axiom prem4 : ∀ x : Person, LiveAtHome x → ¬ LiveWithStrangers x
axiom prem5 : ∀ x : Person, SpendTimeWithSiblings x → LiveAtHome x
axiom prem6 :
  (Baby Luke ∧ BornInMultipleBirth Luke ∧ LiveWithStrangers Luke) ∨
  (¬ Baby Luke ∧ ¬ BornInMultipleBirth Luke ∧ ¬ LiveWithStrangers Luke)

theorem goal : ¬ BornInMultipleBirth Luke ∧ ¬ ComplainOften Luke := by
  -- From being born in multiple birth to not living with strangers
  have born_imp_not_strangers :
      ∀ x : Person, BornInMultipleBirth x → ¬ LiveWithStrangers x := by
    intro x hB
    have hSpend := prem1 x hB
    have hHome := prem5 x hSpend
    exact prem4 x hHome

  -- Derive ¬ BornInMultipleBirth Luke from the disjunction
  have notBorn : ¬ BornInMultipleBirth Luke := by
    cases prem6 with
    | inl h =>
        -- h : Baby Luke ∧ BornInMultipleBirth Luke ∧ LiveWithStrangers Luke
        have hnot := born_imp_not_strangers Luke h.2.1
        exact False.elim (hnot h.2.2)
    | inr h =>
        -- h : ¬ Baby Luke ∧ ¬ BornInMultipleBirth Luke ∧ ¬ LiveWithStrangers Luke
        exact h.2.1

  -- ComplainOften implies BornInMultipleBirth
  have complain_imp_born :
      ∀ x : Person, ComplainOften x → BornInMultipleBirth x := by
    intro x hc
    exact prem2 x (prem3 x hc)

  -- Hence ¬ ComplainOften Luke
  have notComplain : ¬ ComplainOften Luke := by
    intro hc
    exact notBorn (complain_imp_born Luke hc)

  exact And.intro notBorn notComplain
</lean>
