Premises: Some professional basketball players are not American nationals.
All professional basketball players can do jump shots.
If someone can jump shots, they leap straight into the air.
If someone leaps straight into the air, they activate their leg muscles. 
Yuri does not activate his leg muscles.
Conclusion: Yuri is not an American professional basketball player.

==================================================
<lean>
universe u

constant Person : Type
constant Yuri : Person

-- Predicates
constant ProfessionalBasketballPlayer : Person → Prop
constant American : Person → Prop
constant CanJumpShots : Person → Prop
constant LeapsStraight : Person → Prop
constant ActivatesLegMuscles : Person → Prop

-- Premises
axiom somePlayersNotAmerican :
  ∃ x : Person, ProfessionalBasketballPlayer x ∧ ¬ American x
axiom allPlayersJump :
  ∀ x : Person, ProfessionalBasketballPlayer x → CanJumpShots x
axiom jumpsLeap :
  ∀ x : Person, CanJumpShots x → LeapsStraight x
axiom leapsActivate :
  ∀ x : Person, LeapsStraight x → ActivatesLegMuscles x
axiom yuriNotActivate :
  ¬ ActivatesLegMuscles Yuri

-- Conclusion: Yuri is not an American professional basketball player
theorem yuri_not_american_pro_player :
  ¬ (American Yuri ∧ ProfessionalBasketballPlayer Yuri) := by
  -- From the chain of implications, any pro player activates leg muscles
  have p_to_m : ProfessionalBasketballPlayer Yuri → ActivatesLegMuscles Yuri := by
    intro py
    have jy := allPlayersJump Yuri py
    have ly := jumpsLeap Yuri jy
    exact leapsActivate Yuri ly
  -- But Yuri does not activate leg muscles, hence he is not a pro player
  have notP : ¬ ProfessionalBasketballPlayer Yuri := by
    intro py
    exact yuriNotActivate (p_to_m py)
  -- Therefore, he cannot be both American and a pro player
  intro h
  exact notP h.right
</lean>
