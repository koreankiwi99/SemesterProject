Example ID: 1316
Premises: Some professional basketball players are not American nationals.
All professional basketball players can do jump shots.
If someone can jump shots, they leap straight into the air.
If someone leaps straight into the air, they activate their leg muscles. 
Yuri does not activate his leg muscles.
Conclusion: If Yuri does not leap straight into the air, then Yuri is an American professional basketball player.

Iteration 1
==================================================
<lean>
namespace Problem

constant Person : Type

constant ProfessionalBasketballPlayer : Person → Prop
constant AmericanNational : Person → Prop
constant CanDoJumpShots : Person → Prop
constant LeapsStraightIntoAir : Person → Prop
constant ActivatesLegMuscles : Person → Prop

constant Yuri : Person

-- Premises
axiom some_pro_not_american :
  ∃ x : Person, ProfessionalBasketballPlayer x ∧ ¬ AmericanNational x

axiom pros_can_jump :
  ∀ x : Person, ProfessionalBasketballPlayer x → CanDoJumpShots x

axiom can_jump_implies_leap :
  ∀ x : Person, CanDoJumpShots x → LeapsStraightIntoAir x

axiom leap_implies_activate :
  ∀ x : Person, LeapsStraightIntoAir x → ActivatesLegMuscles x

axiom yuri_not_activate :
  ¬ ActivatesLegMuscles Yuri

axiom if_yuri_not_leap_then_american_pro :
  ¬ LeapsStraightIntoAir Yuri → (AmericanNational Yuri ∧ ProfessionalBasketballPlayer Yuri)

-- Conclusion: Yuri is an American professional basketball player
theorem yuri_american_and_pro : AmericanNational Yuri ∧ ProfessionalBasketballPlayer Yuri := by
  -- From leap → activate and ¬activate, derive ¬leap (modus tollens)
  have yuri_not_leap : ¬ LeapsStraightIntoAir Yuri := by
    intro hleap
    have hact : ActivatesLegMuscles Yuri := (leap_implies_activate Yuri) hleap
    exact yuri_not_activate hact
  -- Apply the given conditional
  exact if_yuri_not_leap_then_american_pro yuri_not_leap

end Problem
</lean>
