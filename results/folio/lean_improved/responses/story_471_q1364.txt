Premises: All rabbits that can be spotted near the campus are cute. 
Some turtles can be spotted near the campus. 
The only animals that can be spotted near the campus are rabbits and squirrels.
If something is skittish, then it is not calm.
All the squirrels that can be spotted near the campus are skittish.
Rockie can be spotted near the campus, and it is calm.
Conclusion: If Rockie is not both a turtle and a squirrel, then Rockie is either cute or skittish.

constant Rabbit Turtle Squirrel Near Cute Skittish Calm : Entity → Prop

-- Premises
axiom ax1 : ∀ x : Entity, Rabbit x ∧ Near x → Cute x
axiom ax2 : ∃ x : Entity, Turtle x ∧ Near x
axiom ax3 : ∀ x : Entity, Near x → (Rabbit x ∨ Squirrel x)
axiom ax4 : ∀ x : Entity, Skittish x → ¬ Calm x
axiom ax5 : ∀ x : Entity, Squirrel x ∧ Near x → Skittish x
axiom ax6 : Near Rockie ∧ Calm Rockie

-- Conclusion to prove
theorem main : (¬ (Turtle Rockie ∧ Squirrel Rockie)) → (Cute Rockie ∨ Skittish Rockie) := by
  intro hnotboth
  have hNear : Near Rockie := ax6.left
  have hRS : Rabbit Rockie ∨ Squirrel Rockie := ax3 Rockie hNear
  cases hRS with
  | inl hR =>
      have hCute : Cute Rockie := by
        apply ax1 Rockie
        exact And.intro hR hNear
      exact Or.inl hCute
  | inr hS =>
      have hSk : Skittish Rockie := by
        apply ax5 Rockie
        exact And.intro hS hNear
      exact Or.inr hSk
