Premises: Everyone who can register to vote in the United States can participate in the 2024 United States presidential election.
If someone has United States citizenship, then they can register to vote in the United States.
A person either has United States citizenship or Taiwanese citizenship.
No Russian Federation officials hold Taiwanese citizenship. 
Vladimir neither holds Taiwanese citizenship nor is he a manager at Gazprom. 
Ekaterina she can register to vote in the United States, or she is a Russian federation official.
Conclusion: Ekaterina can participate in the 2024 United States presidential election or Vladimir is a manager at Gazprom.

constant Person : Type

constant CanRegisterUS : Person → Prop
constant CanParticipate2024 : Person → Prop
constant USCitizen : Person → Prop
constant TWCitizen : Person → Prop
constant RFOfficial : Person → Prop
constant ManagerGazprom : Person → Prop

-- Premises
axiom axRegToPart : ∀ x : Person, CanRegisterUS x → CanParticipate2024 x
axiom axUSCtoReg : ∀ x : Person, USCitizen x → CanRegisterUS x
axiom axUSorTW : ∀ x : Person, USCitizen x ∨ TWCitizen x
axiom axNoRFHaveTW : ∀ x : Person, RFOfficial x → ¬ TWCitizen x
axiom axVladNotTW : ¬ TWCitizen Vladimir
axiom axVladNotMgr : ¬ ManagerGazprom Vladimir
axiom axEkaRegOrRF : CanRegisterUS Ekaterina ∨ RFOfficial Ekaterina

theorem goal : CanParticipate2024 Ekaterina ∨ ManagerGazprom Vladimir := by
  cases axEkaRegOrRF with
  | inl hReg =>
      exact Or.inl (axRegToPart Ekaterina hReg)
  | inr hRF =>
      have hNotTW : ¬ TWCitizen Ekaterina := axNoRFHaveTW Ekaterina hRF
      have hUSorTW : USCitizen Ekaterina ∨ TWCitizen Ekaterina := axUSorTW Ekaterina
      have hUS : USCitizen Ekaterina := by
        cases hUSorTW with
        | inl hUS => exact hUS
        | inr hTW => exact False.elim (hNotTW hTW)
      have hReg' : CanRegisterUS Ekaterina := axUSCtoReg Ekaterina hUS
      have hPart : CanParticipate2024 Ekaterina := axRegToPart Ekaterina hReg'
      exact Or.inl hPart
