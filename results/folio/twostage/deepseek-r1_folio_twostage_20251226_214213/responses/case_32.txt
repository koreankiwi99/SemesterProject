-- Premises

axiom P2 : ∀ (p : Person), IsAtMixer p → IsGrandSlamChampion p → IsProfessionalTennisPlayer p
axiom P3 : ∀ (p : Person), IsAtMixer p → IsOscarNominatedActor p → IsCelebrity p
axiom P4 : ∀ (p : Person), IsAtMixer p → IsProfessionalTennisPlayer p → IsAthlete p
