<lean>
axiom Animal : Type
axiom Rockie : Animal

-- Predicates
axiom SpottedNearCampus : Animal → Prop
axiom Rabbit : Animal → Prop
axiom Turtle : Animal → Prop
axiom Squirrel : Animal → Prop
axiom Cute : Animal → Prop
axiom Skittish : Animal → Prop
axiom Calm : Animal → Prop

-- Premise 1: All spotted rabbits are cute
axiom A1 : ∀ a : Animal, Rabbit a ∧ SpottedNearCampus a → Cute a

-- Premise 2: Some spotted turtles exist (not directly used)
axiom A2 : ∃ a : Animal, Turtle a ∧ SpottedNearCampus a

-- Premise 3: Only rabbits/squirrels are spotted
axiom A3 : ∀ a : Animal, SpottedNearCampus a → Rabbit a ∨ Squirrel a

-- Premise 4: Skittish implies not calm
axiom A4 : ∀ a : Animal, Skittish a → ¬ Calm a

-- Premise 5: All spotted squirrels are skittish
axiom A5 : ∀ a : Animal, Squirrel a ∧ SpottedNearCampus a → Skittish a

-- Premise 6: Rockie is spotted and calm
axiom A6 : SpottedNearCampus Rockie ∧ Calm Rockie

-- Conclusion: If Rockie isn't both turtle and squirrel, then cute or skittish
theorem goal : ¬ (Turtle Rockie ∧ Squirrel Rockie) → (Cute Rockie ∨ Skittish Rockie) :=
by
  intro _
  have h_spot := A6.left
  have h_rabbit_or_squirrel := A3 Rockie h_spot
  cases h_rabbit_or_squirrel with
  | inl h_rabbit => 
      left
      exact A1 Rockie ⟨h_rabbit, h_spot⟩
  | inr h_squirrel => 
      right
      exact A5 Rockie ⟨h_squirrel, h_spot⟩
