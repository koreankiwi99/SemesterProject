    intro hAB
    have hR : ReproMF Harry := P2 Harry hAB
    have hNR : ¬ ReproMF Harry := P4 Harry P6
    exact hNR hR
  -- From the assumption, the left disjunct contradicts hNoAB, so we can extract ¬Vertebrate Harry.
  have hNotV : ¬ Vertebrate Harry := by
    cases h with
    | inl h1 =>
      -- Left disjunct gives Animal ∧ Backbone, contradicting hNoAB
      have : False := by
        exact hNoAB h1.right
      exact False.elim this
    | inr h2 =>
      exact h2.left

  -- Hence we derive a contradiction to close the proof (impossible in Lean without additional axioms).

    | inl h1 =>

      exact Or.inr this
