axiom obj : Type

axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

-- Premises
axiom P1 : MountainRange PicurisMountains
axiom P2 : InNewMexico PicurisMountains ∨ InTexas PicurisMountains
axiom P3 : Visited JuanDeOnate PicurisMountains
axiom P4 : Mine HardingPegmatiteMine
axiom P5 : LocatedIn HardingPegmatiteMine PicurisMountains
axiom P6 : Donated HardingPegmatiteMine

-- Conclusion to formalize
theorem goal : ∃ x : obj, MountainRange x ∧ InNewMexico x ∧ Visited JuanDeOnate x := by
  have hNM : InNewMexico PicurisMountains := by
    cases P2 with
    | inl h => exact h
    | inr hTX =>
        have hx : MountainRange PicurisMountains ∧ InTexas PicurisMountains := And.intro P1 hTX
        have hNo := P7 PicurisMountains hx
        have hEx : ∃ m : obj, Mine m ∧ LocatedIn m PicurisMountains ∧ Donated m :=
          Exists.intro HardingPegmatiteMine (And.intro P4 (And.intro P5 P6))
        exact False.elim (hNo hEx)
  exact Exists.intro PicurisMountains (And.intro P1 (And.intro hNM P3))
