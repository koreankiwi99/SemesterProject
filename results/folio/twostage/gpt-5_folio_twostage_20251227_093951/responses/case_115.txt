theorem goal : GreatShooter Windy :=
by
  classical
  have h : JumpsWhenShooting Windy ∨ ¬ JumpsWhenShooting Windy :=
    Classical.em (JumpsWhenShooting Windy)
  cases h with
  | inl hJ =>
      exact A7 Windy ⟨A8, hJ⟩
  | inr hnot =>
