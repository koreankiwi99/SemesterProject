<lean>
axiom obj : Type
axiom Mammal : obj → Prop
axiom HasTeeth : obj → Prop
axiom NoTeeth : obj → Prop
axiom Platypus : obj → Prop
axiom Human : obj → Prop

-- Premises
axiom A1 : ∃ x : obj, Mammal x ∧ HasTeeth x
axiom A2 : ∀ x : obj, Platypus x → NoTeeth x
axiom A3 : ∀ x : obj, Platypus x → Mammal x
axiom A4 : ∀ x : obj, Human x → HasTeeth x

-- Conclusion to formalize: Platypuses are mammals with no teeth.
theorem goal : ∀ x : obj, Platypus x → (Mammal x ∧ NoTeeth x) := by
  intro x
  intro hpl
  apply And.intro
  · exact A3 x hpl
  · exact A2 x hpl
</lean>
