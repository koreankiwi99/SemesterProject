-- Premise: Heck cattle were bred back (in the 1920s to resemble the aurochs).
axiom P2 : ∃ x : obj, HeckCattle x ∧ BredBack x

-- Premise: Heck cattle are animals.
axiom P3 : ∀ x : obj, HeckCattle x → Animal x

-- Premise: Aurochs are animals.

theorem goal : ∃ x : obj, HeckCattle x ∧ ArtificiallySelected x :=
by
  match P2 with
  | ⟨x, hx⟩ =>
    match hx with
    | And.intro hxHeck hxBred =>
