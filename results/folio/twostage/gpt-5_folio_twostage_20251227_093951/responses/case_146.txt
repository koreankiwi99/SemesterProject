--- Iteration 1 ---

<lean>
axiom obj : Type

-- Individuals
axiom Vladimir : obj
axiom Ekaterina : obj

-- Predicates
axiom CanRegisterToVoteInUS : obj → Prop
axiom CanParticipateIn2024USPresElection : obj → Prop
axiom HasUSCitizenship : obj → Prop
axiom HasTaiwaneseCitizenship : obj → Prop
axiom RussianFederationOfficial : obj → Prop
axiom ManagerAtGazprom : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, CanRegisterToVoteInUS x → CanParticipateIn2024USPresElection x
axiom P2 : ∀ x : obj, HasUSCitizenship x → CanRegisterToVoteInUS x
axiom P3 : ∀ x : obj, HasUSCitizenship x ∨ HasTaiwaneseCitizenship x
axiom P4 : ∀ x : obj, RussianFederationOfficial x → ¬ HasTaiwaneseCitizenship x
axiom P5 : ¬ HasTaiwaneseCitizenship Vladimir ∧ ¬ ManagerAtGazprom Vladimir
axiom P6 : CanRegisterToVoteInUS Ekaterina ∨ RussianFederationOfficial Ekaterina

-- Conclusion to formalize
theorem goal : CanParticipateIn2024USPresElection Ekaterina ∨ ManagerAtGazprom Ekaterina := by
  cases P6 with
  | inl hReg =>
      exact Or.inl (P1 Ekaterina hReg)
  | inr hRFO =>
      have hNoTW : ¬ HasTaiwaneseCitizenship Ekaterina := P4 Ekaterina hRFO
      have hUS : HasUSCitizenship Ekaterina := by
        have hUSorTW := P3 Ekaterina
        cases hUSorTW with
        | inl h => exact h
        | inr hTW => exact False.elim (hNoTW hTW)
      exact Or.inl (P1 Ekaterina (P2 Ekaterina hUS))
</lean>
