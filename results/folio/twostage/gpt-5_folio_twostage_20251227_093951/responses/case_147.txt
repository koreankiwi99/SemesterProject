<lean>
axiom obj : Type

-- Individuals
axiom Ekaterina : obj
axiom Vladimir : obj

axiom CanRegisterUS : obj → Prop
axiom CanParticipateUS2024 : obj → Prop
axiom USCitizen : obj → Prop
axiom TaiwaneseCitizen : obj → Prop
axiom RussianFederationOfficial : obj → Prop
axiom ManagerAtGazprom : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, CanRegisterUS x → CanParticipateUS2024 x
axiom P2 : ∀ x : obj, USCitizen x → CanRegisterUS x
axiom P3 : ∀ x : obj, USCitizen x ∨ TaiwaneseCitizen x
axiom P4 : ∀ x : obj, RussianFederationOfficial x → ¬ TaiwaneseCitizen x
axiom P5a : ¬ TaiwaneseCitizen Vladimir
axiom P5b : ¬ ManagerAtGazprom Vladimir
axiom P6 : CanRegisterUS Ekaterina ∨ RussianFederationOfficial Ekaterina

-- Conclusion to formalize
theorem goal : CanParticipateUS2024 Ekaterina ∨ ManagerAtGazprom Vladimir :=
by
  have hleft : CanParticipateUS2024 Ekaterina :=
    by
      cases P6 with
      | inl hreg =>
          exact P1 Ekaterina hreg
      | inr hoff =>
          have hnotTai : ¬ TaiwaneseCitizen Ekaterina := P4 Ekaterina hoff
          have hsum : USCitizen Ekaterina ∨ TaiwaneseCitizen Ekaterina := P3 Ekaterina
          have hUSC : USCitizen Ekaterina :=
            by
              cases hsum with
              | inl hu =>
                  exact hu
              | inr ht =>
                  exact False.elim (hnotTai ht)
          exact P1 Ekaterina (P2 Ekaterina hUSC)
  exact Or.inl hleft
</lean>
