--- Iteration 1 ---

<lean>
axiom obj : Type
axiom Student : obj → Prop
axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom HasUniPartTimeJob : obj → Prop
axiom TakesDatabaseCourse : obj → Prop
axiom TakesClassWithProfDavid : obj → Prop
axiom WorksLab : obj → Prop
axiom James : obj

-- None of the students who currently work in the library are from the CS department.
axiom A1 : ∀ x : obj, Student x ∧ WorksLibrary x → ¬ FromCS x

-- All of the students with part-time jobs offered by the university currently work in the library.
axiom A2 : ∀ x : obj, Student x ∧ HasUniPartTimeJob x → WorksLibrary x

-- All of the students who are taking the database course are from the CS department.
axiom A3 : ∀ x : obj, Student x ∧ TakesDatabaseCourse x → FromCS x

-- All of the students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, Student x ∧ TakesClassWithProfDavid x → TakesDatabaseCourse x

-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, Student x ∧ WorksLab x → TakesClassWithProfDavid x

-- James is a student working in the lab.
axiom A6 : Student James
axiom A7 : WorksLab James

-- James doesn't work in the lab or have a part-time job offered by the university.
axiom A8 : ¬ (WorksLab James ∨ HasUniPartTimeJob James)

-- Conclusion: James has a part-time job offered by the university.
theorem goal : HasUniPartTimeJob James := sorry
</lean>
