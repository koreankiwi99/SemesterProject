axiom LikesMusic : obj → Prop
axiom WantsToCompose : obj → Prop

axiom CanCompose : obj → Prop
axiom WillCompose : obj → Prop

axiom P1 : LikesMusic George → WantsToCompose George

theorem goal : LikesMusic George ∧ HasAccessToProgram George → WillCompose George := by
  intro h
  have hm : LikesMusic George := And.left h
  have ha : HasAccessToProgram George := And.right h
  have hw : WantsToCompose George := P1 hm
  have hc : CanCompose George := P2 ha
  exact P3 (And.intro hc hw)
