--- Iteration 1 ---

<lean>
axiom obj : Type

axiom Rabbit : obj → Prop
axiom Turtle : obj → Prop
axiom Squirrel : obj → Prop
axiom SpottedNearCampus : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

axiom Rockie : obj

-- All rabbits that can be spotted near the campus are cute.
axiom A1 : ∀ x : obj, Rabbit x ∧ SpottedNearCampus x → Cute x

-- Some turtles can be spotted near the campus.
axiom A2 : ∃ x : obj, Turtle x ∧ SpottedNearCampus x

-- The only animals that can be spotted near the campus are rabbits and squirrels.
axiom A3 : ∀ x : obj, SpottedNearCampus x → (Rabbit x ∨ Squirrel x)

-- If something is skittish, then it is not calm.
axiom A4 : ∀ x : obj, Skittish x → ¬ Calm x

-- All the squirrels that can be spotted near the campus are skittish.
axiom A5 : ∀ x : obj, Squirrel x ∧ SpottedNearCampus x → Skittish x

-- Rockie can be spotted near the campus, and it is calm.
axiom A6 : SpottedNearCampus Rockie ∧ Calm Rockie

-- Conclusion: Rockie is a turtle or cute.
theorem goal : Turtle Rockie ∨ Cute Rockie := by
  have hSC : SpottedNearCampus Rockie := And.left A6
  have hCalm : Calm Rockie := And.right A6
  have hRS : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie hSC
  have hCute : Cute Rockie := by
    cases hRS with
    | inl hR =>
        exact A1 Rockie (And.intro hR hSC)
    | inr hS =>
        have hSk : Skittish Rockie := A5 Rockie (And.intro hS hSC)
        have hNotCalm : ¬ Calm Rockie := A4 Rockie hSk
        exact False.elim (hNotCalm hCalm)
  exact Or.inr hCute
</lean>
ANSWER: True
