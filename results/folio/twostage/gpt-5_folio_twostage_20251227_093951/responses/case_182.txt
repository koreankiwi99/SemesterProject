<lean>
axiom obj : Type
axiom Tom : obj

axiom NiceToAnimals : obj → Prop
axiom MeanToAnimals : obj → Prop
axiom Grumpy : obj → Prop
axiom LovesAnimals : obj → Prop
axiom PetOwner : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, NiceToAnimals x → ¬ MeanToAnimals x
axiom P2 : ∃ x : obj, Grumpy x ∧ MeanToAnimals x
axiom P3 : ∀ x : obj, LovesAnimals x → NiceToAnimals x
axiom P4 : ∀ x : obj, PetOwner x → LovesAnimals x
axiom P5 : PetOwner Tom

-- Conclusion to formalize
theorem goal : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) :=
by
  have hLove : LovesAnimals Tom := P4 Tom P5
  have hNice : NiceToAnimals Tom := P3 Tom hLove
  have hNotMean : ¬ MeanToAnimals Tom := P1 Tom hNice
  intro h
  exact hNotMean h.right
</lean>
ANSWER: True
