--- Iteration 1 ---

<lean>
axiom obj : Type

axiom Rabbit : obj → Prop
axiom Turtle : obj → Prop
axiom Squirrel : obj → Prop
axiom SpottedNearCampus : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

axiom Rockie : obj

-- Premises
axiom P1 : ∀ x : obj, (Rabbit x ∧ SpottedNearCampus x) → Cute x
axiom P2 : ∃ x : obj, Turtle x ∧ SpottedNearCampus x
axiom P3 : ∀ x : obj, SpottedNearCampus x → (Rabbit x ∨ Squirrel x)
axiom P4 : ∀ x : obj, Skittish x → ¬ Calm x
axiom P5 : ∀ x : obj, (Squirrel x ∧ SpottedNearCampus x) → Skittish x
axiom P6 : SpottedNearCampus Rockie ∧ Calm Rockie

-- Conclusion
theorem goal : (¬ (Turtle Rockie ∧ Squirrel Rockie)) → (Cute Rockie ∨ Skittish Rockie) := by
  intro h
  have hSC : SpottedNearCampus Rockie := And.left P6
  have hRS : Rabbit Rockie ∨ Squirrel Rockie := P3 Rockie hSC
  cases hRS with
  | inl hR =>
      exact Or.inl (P1 Rockie (And.intro hR hSC))
  | inr hS =>
      exact Or.inr (P5 Rockie (And.intro hS hSC))
</lean>
ANSWER: True
