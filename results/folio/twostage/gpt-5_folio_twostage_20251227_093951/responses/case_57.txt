<lean>
axiom City : Type
axiom State : Type

axiom InState : City → State → Prop
axiom InUS_State : State → Prop

axiom Billings : City
axiom Butte : City
axiom Helena : City
axiom Missoula : City
axiom WhiteSulphurSprings : City
axiom StPierre : City
axiom Bristol : City
axiom Texarkana : City
axiom Texhoma : City
axiom UnionCity : City

axiom Montana : State

-- Billings is in the state of Montana in U.S.
axiom P1a : InState Billings Montana
axiom P1b : InUS_State Montana

-- The state of Montana includes the cities of Butte, Helena, and Missoula.
axiom P2a : InState Butte Montana
axiom P2b : InState Helena Montana
axiom P2c : InState Missoula Montana

-- White Sulphur Springs and Butte are cities in the same state in U.S.
axiom P3 : ∃ s : State, InUS_State s ∧ InState WhiteSulphurSprings s ∧ InState Butte s

-- The city of St Pierre is not in the state of Montana.
axiom P4 : ¬ InState StPierre Montana

-- Any state that contains Butte (in U.S.) does not contain St Pierre.
axiom P5 : ∀ s : State, InUS_State s → InState Butte s → ¬ InState StPierre s

-- A city can only be in one state in U.S. except for Bristol, Texarkana, Texhoma and Union City.
axiom P6 :
  ∀ c : City, c ≠ Bristol → c ≠ Texarkana → c ≠ Texhoma → c ≠ UnionCity →
  ∀ s1 s2 : State, InUS_State s1 → InUS_State s2 → InState c s1 → InState c s2 → s1 = s2

-- Conclusion: Butte and St Pierre are in the same state.
theorem goal : ∃ s : State, InState Butte s ∧ InState StPierre s := sorry
</lean>
