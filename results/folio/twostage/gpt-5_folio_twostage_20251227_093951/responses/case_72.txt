<lean>
axiom obj : Type
axiom Stock : obj → Prop
axiom Growth : obj → Prop
axiom Mature : obj → Prop
axiom BoughtForRapidProfit : obj → Prop
axiom SuitableForRetirement : obj → Prop
axiom PriceVolatile : obj → Prop
axiom KO : obj

-- Premises
axiom A1 : ∀ x : obj, Growth x → BoughtForRapidProfit x
axiom A2 : ∀ x : obj, BoughtForRapidProfit x → ¬ SuitableForRetirement x
axiom A3 : ∃ x : obj, Stock x ∧ Growth x
axiom A4 : ∀ x : obj, Mature x → SuitableForRetirement x
axiom A5 : Mature KO

-- Conclusion to formalize
theorem goal : (Growth KO ∨ BoughtForRapidProfit KO) → (¬ Stock KO ∧ ¬ PriceVolatile KO) := by
  intro h
  have hSFR : SuitableForRetirement KO := A4 KO A5
  have hNotB : ¬ BoughtForRapidProfit KO := by
    intro hb
    have hNSFR : ¬ SuitableForRetirement KO := A2 KO hb
    exact hNSFR hSFR
  have hNotG : ¬ Growth KO := by
    intro hg
    exact hNotB (A1 KO hg)
  have hFalse : False := by
    cases h with
    | inl hg => exact hNotG hg
    | inr hb => exact hNotB hb
  apply And.intro
  · intro hStock
    exact False.elim hFalse
  · intro hPV
    exact False.elim hFalse
</lean>
