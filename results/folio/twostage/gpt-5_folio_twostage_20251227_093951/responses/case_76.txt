<lean>
axiom obj : Type

-- Individuals
axiom Hannah : obj

-- Predicates
axiom AtMarysSchool : obj → Prop
axiom WorksStudentJobsOnCampus : obj → Prop
axiom NeedsMoneyForTuition : obj → Prop
axiom OrderTakeoutFrequentlyInCollege : obj → Prop
axiom EnjoyDiningHallMealsAndRecipes : obj → Prop
axiom SpendALotOfTimeEatingCatchingUpInCampusDiningHalls : obj → Prop
axiom PickyEater : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, AtMarysSchool x → WorksStudentJobsOnCampus x → NeedsMoneyForTuition x
axiom P2 : ∀ x : obj, AtMarysSchool x → OrderTakeoutFrequentlyInCollege x → WorksStudentJobsOnCampus x
axiom P3 : ∀ x : obj, AtMarysSchool x → (OrderTakeoutFrequentlyInCollege x ∨ EnjoyDiningHallMealsAndRecipes x)
axiom P4 : ∀ x : obj, AtMarysSchool x → EnjoyDiningHallMealsAndRecipes x → ¬ PickyEater x
axiom P5 : ∀ x : obj, AtMarysSchool x → EnjoyDiningHallMealsAndRecipes x → SpendALotOfTimeEatingCatchingUpInCampusDiningHalls x
axiom P6 : AtMarysSchool Hannah
axiom P7 : WorksStudentJobsOnCampus Hannah ∧ (NeedsMoneyForTuition Hannah → (¬ PickyEater Hannah ∧ ¬ NeedsMoneyForTuition Hannah))

-- Conclusion to formalize
theorem goal :
  AtMarysSchool Hannah ∧ ¬ PickyEater Hannah ∧ SpendALotOfTimeEatingCatchingUpInCampusDiningHalls Hannah := by
  have A : AtMarysSchool Hannah := P6
  have h7 := P7
  have imp : NeedsMoneyForTuition Hannah → (¬ PickyEater Hannah ∧ ¬ NeedsMoneyForTuition Hannah) := And.right h7
  have notOT : ¬ OrderTakeoutFrequentlyInCollege Hannah := by
    intro OT
    have W2 : WorksStudentJobsOnCampus Hannah := P2 Hannah A OT
    have Need2 : NeedsMoneyForTuition Hannah := P1 Hannah A W2
    have conj2 := imp Need2
    have notNeed2 : ¬ NeedsMoneyForTuition Hannah := And.right conj2
    exact notNeed2 Need2
  have disj : OrderTakeoutFrequentlyInCollege Hannah ∨ EnjoyDiningHallMealsAndRecipes Hannah := P3 Hannah A
  have enjoy : EnjoyDiningHallMealsAndRecipes Hannah := Or.resolve_left disj notOT
  have spend : SpendALotOfTimeEatingCatchingUpInCampusDiningHalls Hannah := P5 Hannah A enjoy
  have notPicky : ¬ PickyEater Hannah := P4 Hannah A enjoy
  exact And.intro A (And.intro notPicky spend)
</lean>
