<lean>
axiom obj : Type

-- Objects
axiom DriedThaiChilies : obj

-- Predicates
axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
axiom P2 : ∀ x : obj, Cupcake x → BakedSweet x
axiom P3 : ∀ x : obj, MalaHotpot x → Spicy x
axiom P4 : ∀ x : obj, FromBBM x → Cupcake x
axiom P5 : Spicy DriedThaiChilies ∨ (MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies)

-- Conclusion to formalize
theorem goal : ¬ FromBBM DriedThaiChilies := by
  intro hF
  have hC : Cupcake DriedThaiChilies := P4 DriedThaiChilies hF
  have hB : BakedSweet DriedThaiChilies := P2 DriedThaiChilies hC
  have hNotS : ¬ Spicy DriedThaiChilies := P1 DriedThaiChilies hB
  have hS : Spicy DriedThaiChilies := by
    apply Or.elim P5
    · intro hs; exact hs
    · intro h
      apply Or.elim h
      · intro hm; exact P3 DriedThaiChilies hm
      · intro hnb; exact False.elim (hnb hB)
  exact hNotS hS
</lean>
