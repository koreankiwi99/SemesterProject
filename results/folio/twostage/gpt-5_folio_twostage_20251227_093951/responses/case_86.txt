axiom person : Type
axiom movie : Type

axiom Digby : person
axiom SDCC : movie  -- The Surprising Adventures of Sir Digby Chicken Caesar

axiom SuperheroMovie : movie → Prop
axiom NamedAfter : movie → person → Prop
axiom InMovie : person → movie → Prop
axiom GoodGuy : person → Prop
axiom BadGuy : person → Prop
axiom Fights : person → person → Prop
axiom Wins : person → Prop
axiom Nemesis : person → person

-- In superhero movies, the good guys always win.
axiom P1 : ∀ m : movie, ∀ p : person, SuperheroMovie m → InMovie p m → GoodGuy p → Wins p

-- The Surprising Adventures of Sir Digby Chicken Caesar is a superhero movie.
axiom P2 : SuperheroMovie SDCC

-- Good guys fight bad guys and vice versa.
axiom P3 : ∀ g b : person, GoodGuy g → BadGuy b → (Fights g b ∧ Fights b g)

-- Sir Digby fights his nemesis.
axiom P4 : Fights Digby (Nemesis Digby)

-- If a superhero movie is named after a character, that character is a good guy.
axiom P5 : ∀ m : movie, ∀ c : person, SuperheroMovie m → NamedAfter m c → GoodGuy c

-- The Surprising Adventures of Sir Digby Chicken Caesar is named after Sir Digby.
axiom P6 : NamedAfter SDCC Digby

-- If somebody wins a fight, the person they are fighting does not win.
axiom P7 : ∀ p q : person, Fights p q → Wins p → ¬ Wins q

-- If a superhero movie is named after a character, that character is in the movie.
axiom P8 : ∀ m : movie, ∀ c : person, SuperheroMovie m → NamedAfter m c → InMovie c m

-- Conclusion: Sir Digby’s nemesis does not win.
theorem goal : ¬ Wins (Nemesis Digby) :=
by
  have hSM : SuperheroMovie SDCC := P2
  have hNamed : NamedAfter SDCC Digby := P6
  have hGood : GoodGuy Digby := P5 SDCC Digby hSM hNamed
  have hIn : InMovie Digby SDCC := P8 SDCC Digby hSM hNamed
  have hWin : Wins Digby := P1 SDCC Digby hSM hIn hGood
  exact P7 Digby (Nemesis Digby) P4 hWin
</lean>
