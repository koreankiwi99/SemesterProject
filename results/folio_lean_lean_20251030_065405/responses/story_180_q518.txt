Premises: Sam is doing a project.
A project is written either in C++ or Python.
If Sam does a project written in Python, he will not use a Mac.
Sam is using a Mac.
If Sam uses a Mac, he will play a song.
If a song is not titled "Perfect," Sam will never play it.
Conclusion: The project Sam is doing is written in C++.

axiom Sam : Person

axiom projectSam : Project
axiom Cpp : Project → Prop
axiom Python : Project → Prop

axiom UsesMac : Person → Prop

axiom Plays : Person → Song → Prop
axiom Perfect : Song → Prop

-- A project is written either in C++ or Python.
axiom proj_either : ∀ p : Project, Cpp p ∨ Python p

-- If Sam does a project written in Python, he will not use a Mac.
axiom py_implies_not_mac : Python projectSam → ¬ UsesMac Sam

-- Sam is using a Mac.
axiom mac_sam : UsesMac Sam

-- If Sam uses a Mac, he will play a song.
axiom mac_implies_play : UsesMac Sam → ∃ s : Song, Plays Sam s

-- If a song is not titled "Perfect," Sam will never play it.
axiom never_play_nonperfect : ∀ s : Song, ¬ Perfect s → ¬ Plays Sam s

theorem project_is_cpp : Cpp projectSam := by
  have disj := proj_either projectSam
  have not_py : ¬ Python projectSam := by
    intro hp
    have h_not_mac := py_implies_not_mac hp
    exact h_not_mac mac_sam
  exact Or.resolve_right disj not_py
