Lean translation and proof:

universe u

constant Thing : Type u

constant Square : Thing → Prop
constant FourSided : Thing → Prop
constant Shape : Thing → Prop

axiom allSquaresFourSided : ∀ x, Square x → FourSided x
axiom allFourSidedShapes : ∀ x, FourSided x → Shape x

theorem allSquaresAreShapes : ∀ x, Square x → Shape x := by
  intro x hx
  have hfs : FourSided x := allSquaresFourSided x hx
  exact allFourSidedShapes x hfs

True
