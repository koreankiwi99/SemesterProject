constant Alg : Type u

constants
  Used : Alg → Prop
  ML : Alg → Prop
  Supervised : Alg → Prop
  Unsupervised : Alg → Prop
  Reinforcement : Alg → Prop
  RequiresLabels : Alg → Prop

constant A : Alg

axioms
  usedA : Used A
  mlA : ML A
  unsup_no_labels : ∀ a, Unsupervised a → ¬ RequiresLabels a
  not_reinforcementA : ¬ Reinforcement A
  requires_labelsA : RequiresLabels A
  ml_categorized : ∀ a, ML a → (Supervised a ∨ Unsupervised a ∨ Reinforcement a)

theorem not_unsupervisedA : ¬ Unsupervised A :=
by
  intro hU
  have hnl : ¬ RequiresLabels A := unsup_no_labels A hU
  exact hnl requires_labelsA

theorem supervisedA : Supervised A :=
by

  have hU := not_unsupervisedA
  have hR := not_reinforcementA
  rcases h with hs | hur
  · exact hs
  · rcases hur with hu | hr
    · exact False.elim (hU hu)
    · exact False.elim (hR hr)

theorem Q1 : Used A ∧ Supervised A :=
And.intro usedA supervisedA

theorem Q2_neg : ¬ (Used A ∧ Unsupervised A) :=
by
  intro h
  have hU : Unsupervised A := And.right h
  exact not_unsupervisedA hU

end Reasoning
