  ∀ {x y : Person}, Man x → Man y → Taller x y → CanBlock x y

  ∀ {x y z : Person}, Taller x y → Taller y z → Taller x z

axiom A3_peter_facts : Man Peter ∧ Taller Peter Michael

axiom A4_michael_facts :
  Man Michael ∧ (∀ {y : Person}, InMClass y → y ≠ Michael → Taller Michael y)

axiom A5_michael_blocks_nonjumpers :
  ∀ {p : Person}, ¬ Jumps p → CanBlock Michael p

axiom A6_michael_cannot_block_windy :
  ¬ CanBlock Michael Windy

axiom A7_jumpers_are_great :
  ∀ {s : Person}, Jumps s → Great s

open Classical

-- From A5 and A6 we derive that Windy jumps, hence is great.
theorem windy_is_great : Great Windy := by
  -- First show Windy jumps by contrapositive of A5 and A6
  have windy_jumps : Jumps Windy := by
    have h : (¬ Jumps Windy) → CanBlock Michael Windy := A5
    exact by_contra (fun hNJ => A6 (h hNJ))
  -- Then apply that all who jump are great
  exact A7 windy_jumps
