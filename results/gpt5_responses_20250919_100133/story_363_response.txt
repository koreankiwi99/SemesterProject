Lean formalization:

universe u

open Classical

constant U : Type u

-- Predicates
constant Hydrocarbon : U → Prop
constant Organic : U → Prop
constant Chemical : U → Prop
constant Alkane : U → Prop
constant ContainsCarbon : U → Prop
constant OnlyOne : U → Prop

-- Axioms from the text
axiom H_h_o : ∀ x : U, Hydrocarbon x → Organic x                -- all hydrocarbons are organic
axiom H_a_h : ∀ x : U, Alkane x → Hydrocarbon x                 -- all alkanes are hydrocarbons
axiom H_o_c : ∀ x : U, Organic x → Chemical x                   -- all organic compounds are chemical compounds
axiom H_o_cc : ∀ x : U, Organic x → ContainsCarbon x            -- all organic compounds contain carbon
axiom H_c_not_one : ∀ x : U, Chemical x → ¬ OnlyOne x           -- no chemical compounds contain only one element

-- The mixture and its special condition
constant mixture : U
axiom mix_either :
  (Chemical mixture ∧ OnlyOne mixture) ∨ (¬ Chemical mixture ∧ ¬ OnlyOne mixture)
-- (equivalently: Chemical mixture ↔ OnlyOne mixture)

-- Useful derived facts
lemma mixture_not_chemical : ¬ Chemical mixture := by
  intro hC
  cases mix_either with
  | inl h =>
      have hno : ¬ OnlyOne mixture := H_c_not_one mixture h.left
      exact hno h.right
  | inr h =>
      exact h.left hC

lemma mixture_not_one : ¬ OnlyOne mixture := by
  intro h1
  cases mix_either with
  | inl h =>
      have hno : ¬ OnlyOne mixture := H_c_not_one mixture h.left
      exact hno h1
  | inr h =>
      exact h.right h1

lemma mixture_not_organic : ¬ Organic mixture := by
  intro hO
  have hC : Chemical mixture := H_o_c mixture hO
  exact mixture_not_chemical hC

lemma mixture_not_alkane : ¬ Alkane mixture := by
  intro hA
  have hH  : Hydrocarbon mixture := H_a_h mixture hA
  have hO  : Organic mixture     := H_h_o mixture hH
  have hC  : Chemical mixture    := H_o_c mixture hO
  exact mixture_not_chemical hC

-- Q2: The mixture is an alkane and contains carbon. This is false.
theorem Q2_false : ¬ (Alkane mixture ∧ ContainsCarbon mixture) := by
  intro h
  exact mixture_not_alkane h.left

-- Q3: If the mixture contains only one element or contains carbon,
--     then the mixture is neither a chemical compound nor an alkane. This is true.
theorem Q3_true :
  (OnlyOne mixture ∨ ContainsCarbon mixture) → (¬ Chemical mixture ∧ ¬ Alkane mixture) := by
  intro _
  exact And.intro mixture_not_chemical mixture_not_alkane
