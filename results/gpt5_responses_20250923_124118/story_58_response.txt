universe u

constant Person : Type
constant Book : Type

constant containsKnowledge : Book → Prop
constant gainsKnowledge : Person → Prop
constant smarter : Person → Prop
constant reads : Person → Book → Prop

constant Walden : Book

-- Axioms from the textual context
axiom books_have_knowledge : ∀ b : Book, containsKnowledge b
axiom reading_gives_knowledge : ∀ p : Person, ∀ b : Book, reads p b → gainsKnowledge p
