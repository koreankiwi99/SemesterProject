Premises: It costs $205 to take the GRE test, which is cheaper than $300.

Tom's dad has been out of work, and Tom has few resources available to them. 
Tom is applying to take the GRE test.

Lean 4 code:

universe u

constant Person : Type u
constant Entity : Type u

constant Tom : Person
constant ETS : Entity

constant gre_applicant : Person → Prop
constant provides : Entity → Person → Prop
constant proves_econ_hardship : Person → Prop
constant single_parent : Person → Prop
constant few_resources : Person → Prop

constant gre_cost : Nat

axiom policy :
  ∀ p : Person, gre_applicant p ∧ proves_econ_hardship p → provides ETS p

axiom econ_rule :
  ∀ p : Person, single_parent p ∨ few_resources p → proves_econ_hardship p

axiom tom_single : single_parent Tom
axiom tom_few : few_resources Tom
axiom tom_applicant : gre_applicant Tom

axiom cost_is_205 : gre_cost = 205
axiom cost_lt_300 : gre_cost < 300

theorem Q1_Tom_gets_aid : provides ETS Tom := by
  have hhard : proves_econ_hardship Tom := by
    exact econ_rule Tom (Or.inl tom_single)
  exact policy Tom ⟨tom_applicant, hhard⟩

theorem Q2_cost_below_300 : gre_cost < 300 := cost_lt_300

def Q3_stmt : Prop := ∀ p : Person, gre_applicant p → ∀ e : Entity, ¬ provides e p

theorem not_Q3_stmt : ¬ Q3_stmt := by
  intro H
  have hprov : provides ETS Tom := Q1_Tom_gets_aid
  have hga : gre_applicant Tom := tom_applicant
  have hno := H Tom hga ETS
  exact hno hprov

- Q1 is provable (True).
- Q2 is provable (True).
- Q3’s statement is refutable; we proved its negation (False).
