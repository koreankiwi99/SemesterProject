Context:
In academics, studying a topic in depth leads to expertise and authority on that subject. On the other hand, learning broadly exposes someone to diverse ideas and perspectives. However, students may opt to either study a single topic in depth or learn broadly across disciplines.

Question:
Does this entail that for Sarah, a dedicated student, either she gains expertise on a single topic or is exposed to diverse perspectives?

axiom Person : Type
axiom Topic : Type

axiom Student : Person → Prop
axiom Dedicated : Person → Prop

axiom Sarah : Person
axiom Sarah_student : Student Sarah
axiom Sarah_dedicated : Dedicated Sarah

axiom StudyInDepth : Person → Topic → Prop
axiom Expertise : Person → Topic → Prop
axiom Authority : Person → Topic → Prop
axiom LearnBroadly : Person → Prop
axiom DiversePerspectives : Person → Prop

-- Studying a topic in depth leads to expertise and authority on that subject.
axiom depth_leads : ∀ p t, StudyInDepth p t → Expertise p t ∧ Authority p t

-- Learning broadly exposes someone to diverse ideas and perspectives.
axiom broad_exposes : ∀ p, LearnBroadly p → DiversePerspectives p

-- Students may opt to either study a single topic in depth (exists a topic) or learn broadly.
axiom student_options : ∀ p, Student p → ((∃ t, StudyInDepth p t) ∨ LearnBroadly p)

theorem sarah_expertise_or_diverse : (∃ t, Expertise Sarah t) ∨ DiversePerspectives Sarah := by
  have h := student_options Sarah Sarah_student
  cases h with
  | inl hex =>
      cases hex with
      | intro t hstud =>
          have hEA := depth_leads Sarah t hstud
          have hE : Expertise Sarah t := And.left hEA
          exact Or.inl ⟨t, hE⟩
  | inr hb =>
      have hd := broad_exposes Sarah hb
      exact Or.inr hd
</lean>
