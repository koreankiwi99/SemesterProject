axiom Athlete : Type

axiom AdequateRest : Athlete → Prop
axiom PeakPerformance : Athlete → Prop

-- Coaches' rule: if someone trains with dedication, then if they get adequate rest, they perform at their peak.
axiom coaches_rule : ∀ a : Athlete, TrainsWithDedication a → (AdequateRest a → PeakPerformance a)

-- Facts about Simone
axiom simone_rested : AdequateRest Simone
axiom simone_not_peak : ¬ PeakPerformance Simone

-- Conclusion: Simone did not train with dedication.
theorem simone_did_not_train : ¬ TrainsWithDedication Simone := by
  intro hTrain
  have hImp : AdequateRest Simone → PeakPerformance Simone := coaches_rule Simone hTrain
  have hPeak : PeakPerformance Simone := hImp simone_rested
  exact simone_not_peak hPeak
