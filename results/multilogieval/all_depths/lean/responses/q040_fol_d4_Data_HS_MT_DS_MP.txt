Context:
Someone wanted to become an author and get their book published. They realized that if they wrote every day, they would complete writing their book. And if they completed their book, they could get it published. Either they wrote every day, or they took writing classes to improve their skills, or both. If someone takes writing classes, they gain valuable feedback on their work.

Question:
Mary did not get her book published. Did she gain valuable feedback on her work?

-- Premises
axiom h1 : ∀ x : Person, WroteEveryDay x → CompletedBook x
axiom h2 : ∀ x : Person, CompletedBook x → Published x
axiom h3 : ∀ x : Person, WroteEveryDay x ∨ TookClasses x
axiom h4 : ∀ x : Person, TookClasses x → GainedFeedback x

-- Fact about Mary
axiom not_pub_Mary : ¬ Published Mary

theorem mary_feedback : GainedFeedback Mary := by
  have hWED_to_Pub : WroteEveryDay Mary → Published Mary := by
    intro hw
    exact h2 Mary (h1 Mary hw)
  have hnotWED : ¬ WroteEveryDay Mary := by
    intro hw
    exact not_pub_Mary (hWED_to_Pub hw)
  have hdisj : WroteEveryDay Mary ∨ TookClasses Mary := h3 Mary
  have hclasses : TookClasses Mary :=
    Or.elim hdisj
      (fun hw => False.elim (hnotWED hw))
      (fun hc => hc)
  exact h4 Mary hclasses
