Question:
Can we conclude there was a grading error?

-- If there was a grading error, then Alex could not have actually gotten 90%.
axiom grading_error_implies_not90 : GradingError → ¬ Got90

-- Rule capturing that when two sources conflict, the one with better information determines the truth
axiom prefer_better_info_conflict_pos :
  ∀ p1 p2, BetterInfo p1 p2 → Said90 p1 → SaidNot90 p2 → Got90

-- From the conflict and better information, conclude Alex did get 90.
theorem got90_from_conflict : Got90 := by
  exact prefer_better_info_conflict_pos Teacher Principal better_info teacher_said principal_said

-- Therefore, there was no grading error.
theorem no_grading_error : ¬ GradingError := by
  intro h
  have hnot : ¬ Got90 := grading_error_implies_not90 h
  exact hnot got90_from_conflict
