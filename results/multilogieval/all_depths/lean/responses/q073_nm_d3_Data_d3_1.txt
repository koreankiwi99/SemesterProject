axiom enjoys_imp_excited : EnjoysJohn → ExcitedSarah

-- What we can prove: if Sarah does not get excited, then car 1 does not have high performance.
-- This follows by contraposition through the chain HighPerf car1 → EnjoysJohn → ExcitedSarah.
theorem not_excited_implies_not_hp1 : ¬ExcitedSarah → ¬HighPerf car1 := by
  intro hNotExcited
  intro hHP1
  have hEnjoys : EnjoysJohn := hp1_imp_enjoys hHP1
  have hExcited : ExcitedSarah := enjoys_imp_excited hEnjoys
  exact hNotExcited hExcited

-- We also note we can derive the "usual high performance" facts from the turbo facts,
-- but these do not yield actual high performance in classical logic.
theorem usually_hp_car1 : UsuallyHighPerf car1 := by
