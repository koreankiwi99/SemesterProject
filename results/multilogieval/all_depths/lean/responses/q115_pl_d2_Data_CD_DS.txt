axiom StudyHard : Prop

axiom CanceledExam : Prop
axiom Relaxed : Prop

axiom study_implies_A : StudyHard → GotA
axiom cancel_implies_relax : CanceledExam → Relaxed
axiom study_or_cancel : StudyHard ∨ CanceledExam

theorem notA_implies_relax : ¬ GotA → Relaxed := by
  intro notA
  have notStudy : ¬ StudyHard := by
    intro s
    exact notA (study_implies_A s)
  have canceled : CanceledExam := by
    cases study_or_cancel with
    | inl s => exact False.elim (notStudy s)
    | inr c => exact c
  exact cancel_implies_relax canceled
