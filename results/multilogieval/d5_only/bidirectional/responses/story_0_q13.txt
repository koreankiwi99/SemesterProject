Premises: The factory manager knew that if production increases, more workers need to be hired. The marketing director realized that if a new model is released, demand will be high. Either production increased, or demand for the new phone was low. The factory manager understood that if more workers were hired, wages would increase. If wages increased, applications to work at the factory would rise.
Conclusion: If the company released a new model phone, then did applications to work at the factory rise?

axiom ProductionIncreased : Prop
axiom MoreWorkersHired : Prop
axiom WagesIncreased : Prop
axiom ApplicationsRose : Prop
axiom ReleasedNewModel : Prop
axiom DemandHigh : Prop

-- Premises
-- 1. If production increases, more workers need to be hired.
axiom prem1 : ProductionIncreased → MoreWorkersHired
-- 2. If a new model is released, demand will be high.
axiom prem2 : ReleasedNewModel → DemandHigh
-- 3. Either production increased, or demand for the new phone was low (encoded as ¬ DemandHigh).
axiom prem3 : ProductionIncreased ∨ ¬ DemandHigh
-- 4. If more workers were hired, wages would increase.
axiom prem4 : MoreWorkersHired → WagesIncreased
-- 5. If wages increased, applications to work at the factory would rise.
axiom prem5 : WagesIncreased → ApplicationsRose

-- Conclusion: If the company released a new model phone, then applications to work at the factory rose.
theorem goal : ReleasedNewModel → ApplicationsRose := by
  intro hRel
  have hHigh : DemandHigh := prem2 hRel
  -- Use the disjunction: ProductionIncreased ∨ ¬ DemandHigh
  -- If ProductionIncreased, we can chain implications to ApplicationsRose.
  -- If ¬ DemandHigh, it contradicts hHigh, and from false we can derive ApplicationsRose.
  exact Or.elim prem3
    (fun hProd => prem5 (prem4 (prem1 hProd)))
    (fun hNotHigh => False.elim (hNotHigh hHigh))

axiom NewModelReleased : Prop

-- Premises

axiom premise2 : NewModelReleased → DemandHigh
axiom premise3 : ProductionIncreased ∨ DemandLow
axiom premise4 : MoreWorkersHired → WagesIncreased
axiom premise5 : WagesIncreased → ApplicationsRose
-- Natural incompatibility between low and high demand
axiom premise6 : DemandLow → ¬DemandHigh

-- Derived: If production increased, applications rose
lemma prod_to_apps : ProductionIncreased → ApplicationsRose := by
  intro p
  have hmw : MoreWorkersHired := premise1 p
  have hw : WagesIncreased := premise4 hmw
  exact premise5 hw

-- From the premises, the original conclusion actually follows
theorem conclusion_true : NewModelReleased → ApplicationsRose := by
  intro n
  have dh : DemandHigh := premise2 n
  have notLow : ¬DemandLow := by
    intro l
    have nh : ¬DemandHigh := premise6 l
    exact nh dh
  cases premise3 with
  | inl p =>
      exact prod_to_apps p
  | inr l =>
      exact False.elim (notLow l)

-- We attempt to prove the negation (that the conclusion is false)
