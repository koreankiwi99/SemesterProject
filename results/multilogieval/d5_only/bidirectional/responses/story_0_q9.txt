Premises: Jane has an exciting day coming up. If her friends invite her to a concert, she'll buy tickets to go with them. Also, if her grandmother is visiting from out of town, Jane will stay home to spend time with her. It seems either Jane will be invited to a concert or grandma will visit, maybe both. If Jane buys tickets, she'll have fun with her friends at the show. And if she has fun with her friends, that means she enjoyed the concert.
Conclusion: If Jane did not stay home with grandma, then did she enjoy the concert?

======================================================================
PROVE TRUE ATTEMPT
======================================================================
Lean Code:
axiom Invite : Prop              -- Jane is invited to a concert
axiom Visit : Prop               -- Grandma is visiting
axiom StayHome : Prop            -- Jane stays home with grandma
axiom BuyTickets : Prop          -- Jane buys tickets
axiom Fun : Prop                 -- Jane has fun with her friends
axiom Enjoy : Prop               -- Jane enjoyed the concert

-- Premises
axiom h1 : Invite → BuyTickets
axiom h2 : Visit → StayHome
axiom h3 : Invite ∨ Visit
axiom h4 : BuyTickets → Fun
axiom h5 : Fun → Enjoy

-- Conclusion: If Jane did not stay home with grandma, then she enjoyed the concert.
theorem goal : ¬StayHome → Enjoy := by
  intro notStay
  -- From Visit → StayHome and ¬StayHome, derive ¬Visit
  have notVisit : ¬Visit := by
    intro v
    exact notStay (h2 v)
  -- From Invite ∨ Visit and ¬Visit, derive Invite
  have inv : Invite := Or.elim h3 (fun i => i) (fun v => False.elim (notVisit v))
  -- Chain implications to Enjoy
  have bt : BuyTickets := h1 inv
  have fu : Fun := h4 bt
  have en : Enjoy := h5 fu
  exact en

Verification Success: True

axiom premise5 : FunWithFriends → EnjoyedConcert

-- The conclusion in natural language: ¬StayHome → EnjoyedConcert
-- We can actually prove the conclusion from the premises
theorem conclusion_true : ¬StayHome → EnjoyedConcert := by
  intro notStay
  -- From ¬StayHome, we can rule out GrandmaVisiting using premise2
  have notGrandma : ¬GrandmaVisiting := by
    intro g
    have sh : StayHome := premise2 g
    exact notStay sh
  -- From Invite ∨ GrandmaVisiting and ¬GrandmaVisiting, deduce Invite
  have inv : Invite := by
    cases premise3 with
    | inl i => exact i
    | inr g => exact False.elim (notGrandma g)
  -- Chain implications to EnjoyedConcert
  have t : BuyTickets := premise1 inv
  have f : FunWithFriends := premise4 t
  exact premise5 f

-- The negation of the conclusion would be: ¬(¬StayHome → EnjoyedConcert)
-- This negation is incompatible with the premises, as shown here.
theorem negation_is_false : ¬(¬StayHome → EnjoyedConcert) → False := by
  intro hneg
  exact hneg conclusion_true
