Context:
Someone wanted to become fluent in French. They realized that if they practiced speaking regularly, their vocabulary would improve. With more vocabulary, they found reading became easier over time. Therefore, if they practiced speaking, reading French became easier. Either they practiced speaking, or they watched French films, or they did both. If someone watches French films, it means they enjoy hearing the language. And if they enjoy hearing French, they listen to French music and podcasts too.

Question:
If Amy does not find reading French easier, does she listen to French music and podcasts?

axiom Person : Type
axiom Amy : Person

axiom Practiced : Person → Prop
axiom WatchedFilms : Person → Prop
axiom EnjoysHearing : Person → Prop
axiom ListensMusicPodcasts : Person → Prop
axiom ReadingEasier : Person → Prop
axiom MoreVocab : Person → Prop

-- Premises from the text
axiom practice_improves_vocab : ∀ x : Person, Practiced x → MoreVocab x
axiom vocab_improves_reading : ∀ x : Person, MoreVocab x → ReadingEasier x
axiom amy_practice_or_watch : Practiced Amy ∨ WatchedFilms Amy
axiom watch_implies_enjoy : ∀ x : Person, WatchedFilms x → EnjoysHearing x
axiom enjoy_implies_listen : ∀ x : Person, EnjoysHearing x → ListensMusicPodcasts x

-- Goal: If Amy does not find reading French easier, then she listens to French music and podcasts.
theorem Amy_listens_if_not_reading_easier :
  ¬ ReadingEasier Amy → ListensMusicPodcasts Amy := by
  intro hNotRead
  -- From ¬ReadingEasier Amy and (Practiced → ReadingEasier), get ¬Practiced Amy
  have hNotPractice : ¬ Practiced Amy := by
    intro hPractice
    have mv : MoreVocab Amy := practice_improves_vocab Amy hPractice
    have re : ReadingEasier Amy := vocab_improves_reading Amy mv
    exact hNotRead re
  -- From (Practiced Amy ∨ WatchedFilms Amy) and ¬Practiced Amy, deduce WatchedFilms Amy
  have hWatch : WatchedFilms Amy := by
    cases amy_practice_or_watch with
    | inl hP => exact False.elim (hNotPractice hP)
    | inr hW => exact hW
  -- Chain to listening
  have hEnjoy : EnjoysHearing Amy := watch_implies_enjoy Amy hWatch
  exact enjoy_implies_listen Amy hEnjoy
