Context:
Someone wanted to learn how to cook better meals at home. They learned that if they followed recipes carefully, the food would turn out well. Following recipes meant measuring ingredients precisely and using the proper cooking techniques. Therefore, if someone followed recipes carefully, they could cook good meals at home. Either they followed recipes carefully, or they experimented with ingredients spontaneously, or they tried both approaches. If someone experiments with ingredients spontaneously, it means they embrace creativity in the kitchen. And if they embrace creativity in the kitchen, it implies they enjoy the artistry of food preparation.

Question:
If Joey cannot cook good meals at home, does he enjoy the artistry of food preparation?

axiom Person : Type
axiom Joey : Person

axiom FollowRecipesCarefully : Person → Prop
axiom ExperimentSpontaneously : Person → Prop
axiom CookGoodMealsAtHome : Person → Prop

-- If someone follows recipes carefully, they can cook good meals at home.
axiom rule_recipes : ∀ x : Person, FollowRecipesCarefully x → CookGoodMealsAtHome x
-- If someone experiments spontaneously, they embrace creativity.
axiom rule_experiment_creativity : ∀ x : Person, ExperimentSpontaneously x → EmbraceCreativity x
-- If someone embraces creativity, they enjoy the artistry of food preparation.
axiom rule_creativity_artistry : ∀ x : Person, EmbraceCreativity x → EnjoyArtistry x

-- For Joey: either he follows recipes, or experiments spontaneously, or both.
axiom choice_joey :
  FollowRecipesCarefully Joey ∨ ExperimentSpontaneously Joey ∨ (FollowRecipesCarefully Joey ∧ ExperimentSpontaneously Joey)

theorem joey_enjoys_artistry_if_not_good :
  ¬ CookGoodMealsAtHome Joey → EnjoyArtistry Joey := by
  intro hNotGood
  -- From not good meals, deduce not following recipes carefully (by contrapositive reasoning).
  have hNotFRC : ¬ FollowRecipesCarefully Joey := by
    intro hFRC
    exact hNotGood (rule_recipes Joey hFRC)
  -- From the trichotomy and ¬FRC, deduce Joey experiments spontaneously.
  have hEIS : ExperimentSpontaneously Joey := by
    cases choice_joey with
    | inl hFRC =>
        exact False.elim (hNotFRC hFRC)
    | inr h =>
        cases h with
        | inl hEIS => exact hEIS
        | inr hBoth => exact hBoth.right
  -- Chain implications to conclude enjoyment of artistry.
  have hEC : EmbraceCreativity Joey := rule_experiment_creativity Joey hEIS
  exact rule_creativity_artistry Joey hEC
