theorem priya_icecream_if_snowstorm : Snowstorm → IceCream Priya := by
  intro hsnow
  have hcanc : SchoolCanceled := snow_implies_canceled hsnow
  -- From (StudiesHard Priya ∨ ¬SchoolCanceled) and SchoolCanceled, conclude StudiesHard Priya
