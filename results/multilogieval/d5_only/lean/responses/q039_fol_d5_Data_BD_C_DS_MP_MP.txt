Context:
If someone feeds their dog on time, the dog will be energetic. If they take the dog to the park, then they will play fetch together. Either the dog was fed on time, or they did not play fetch in the park. If the dog is energetic, it will run fast. If the dog runs fast, it will quickly fetch the ball.

Question:
If James took his dog to the park, did the dog quickly fetch the ball?

axiom James : Person

axiom Energetic : Person → Prop
axiom TookToPark : Person → Prop
axiom PlayFetch : Person → Prop
axiom RunsFast : Person → Prop
axiom QuicklyFetch : Person → Prop

axiom feed_on_time_then_energetic : ∀ p : Person, FeedsOnTime p → Energetic p
axiom park_then_play : ∀ p : Person, TookToPark p → PlayFetch p
axiom fed_or_not_play : ∀ p : Person, FeedsOnTime p ∨ ¬ PlayFetch p
axiom energetic_then_runfast : ∀ p : Person, Energetic p → RunsFast p
axiom runfast_then_quick : ∀ p : Person, RunsFast p → QuicklyFetch p

theorem James_quickly_fetch_if_took_to_park : TookToPark James → QuicklyFetch James := by
  intro hpark
  have hplay : PlayFetch James := (park_then_play James) hpark
  have hdisj : FeedsOnTime James ∨ ¬ PlayFetch James := fed_or_not_play James
  have hfeed : FeedsOnTime James := by
    cases hdisj with
    | inl hf => exact hf
    | inr hnp => exact False.elim (hnp hplay)
  have hener : Energetic James := (feed_on_time_then_energetic James) hfeed
  have hrun : RunsFast James := (energetic_then_runfast James) hener
  exact (runfast_then_quick James) hrun
