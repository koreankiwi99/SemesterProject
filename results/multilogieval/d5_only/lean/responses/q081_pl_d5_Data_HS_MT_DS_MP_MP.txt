Context:
Oliver decided he wanted to simplify his life starting with his living space. He realized if he cleaned his room regularly, it would stay tidy. And if his room stayed tidy, he would feel peaceful and relaxed there. Oliver either cleans regularly or declutters his belongings, or does both to get organized. If Oliver declutters, he naturally becomes more organized overall. And if he becomes more organized, his life feels simpler.

Question:
If Oliver does not feel peaceful in his living space, then does his life feel simpler?

axiom Clean : Prop
axiom Tidy : Prop
axiom Peaceful : Prop
axiom Declutter : Prop
axiom Organized : Prop
axiom Simpler : Prop

axiom clean_implies_tidy : Clean → Tidy
axiom tidy_implies_peaceful : Tidy → Peaceful
axiom clean_or_declutter : Clean ∨ Declutter
axiom declutter_implies_organized : Declutter → Organized
axiom organized_implies_simpler : Organized → Simpler

theorem life_simpler_if_not_peaceful : ¬ Peaceful → Simpler := by
  intro hnp
  have h_not_clean : ¬ Clean := by
    intro hC
    have hT : Tidy := clean_implies_tidy hC
    have hP : Peaceful := tidy_implies_peaceful hT
    exact hnp hP
  have hD : Declutter := by
    cases clean_or_declutter with
    | inl hC =>
      exact False.elim (h_not_clean hC)
    | inr hDeclutter =>
      exact hDeclutter
  have hOrg : Organized := declutter_implies_organized hD
  exact organized_implies_simpler hOrg
