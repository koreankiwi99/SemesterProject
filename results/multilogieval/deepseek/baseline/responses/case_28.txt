<lean>
axiom obj : Type
axiom Jane : obj
axiom Injured : obj → Prop
axiom SeeDoctor : obj → Prop
axiom RestInjury : obj → Prop
axiom R1 : ∀ x : obj, Injured x → SeeDoctor x
axiom R2 : ∀ x : obj, Injured x → RestInjury x
axiom Premise : ¬SeeDoctor Jane ∨ ¬RestInjury Jane

theorem jane_not_hurt : ¬Injured Jane := by
  intro h_jane_hurt
  have h_doctor := R1 Jane h_jane_hurt
  have h_rest := R2 Jane h_jane_hurt
  cases Premise with
  | inl no_doctor => contradiction
  | inr no_rest => contradiction
</lean>

ANSWER: No
