<lean>
axiom obj : Type
axiom James : obj
axiom Start : obj → Prop
axiom Motivated : obj → Prop
axiom GymCloses : obj → Prop
axiom FollowRoutine : obj → Prop
axiom Frustrated : obj → Prop
axiom Indulge : obj → Prop

axiom A1 : ∀ x : obj, Start x → Motivated x
axiom A2 : ∀ x : obj, GymCloses x → ¬ FollowRoutine x
axiom A3 : ∀ x : obj, Start x ∨ GymCloses x
axiom A4 : ∀ x : obj, ¬ FollowRoutine x → Frustrated x
axiom A5 : ∀ x : obj, Frustrated x → Indulge x

axiom T1 : ¬ Motivated James

theorem james_indulge : Indulge James := by
  -- Prove James didn't start (from T1 and A1)
  have no_start : ¬ Start James := 
    fun start : Start James => T1 (A1 James start)
  
  -- Resolve disjunction: Start ∨ GymCloses
  have start_or_gym := A3 James
  cases start_or_gym with
  | inl start => 
      -- James started → contradicts no_start
      exact False.elim (no_start start)
  | inr gym_closed =>
      -- Gym closed → can't follow routine
      have no_follow := A2 James gym_closed
      -- Thus frustrated
      have frustrated := A4 James no_follow
      -- Thus indulges
      exact A5 James frustrated
