axiom Person : Type
axiom Emma : Person

axiom SpeaksDaily : Person → Prop
axiom ListensPodcast : Person → Prop
axiom FluencyImproves : Person → Prop

axiom ListeningComprehension : Person → Prop

axiom A1 : ∀ p : Person, SpeaksDaily p → FluencyImproves p
axiom A2 : ∀ p : Person, FluencyImproves p → CanConverse p
axiom A3 : ∀ p : Person, SpeaksDaily p ∨ ListensPodcast p
axiom A4 : ∀ p : Person, ListensPodcast p → ListeningComprehension p

theorem answer : ¬ CanConverse Emma → ListeningComprehension Emma := by
  intro h_not_converse
  have not_fluent : ¬ FluencyImproves Emma := by
    intro h_fluent
    apply h_not_converse
    exact A2 Emma h_fluent
  have not_speaks : ¬ SpeaksDaily Emma := by
    intro h_speaks
    apply not_fluent
    exact A1 Emma h_speaks
