<lean>
axiom R : Prop  -- It rains today
axiom G : Prop  -- Softball game happens
axiom P : Prop  -- Swimming pool is open
axiom K : Prop  -- Kids go for a swim
axiom E : Prop  -- Players enjoy playing
axiom S : Prop  -- Players have smiles

axiom A1 : ¬R → G
axiom A2 : P → K
axiom A3 : ¬R ∨ P
axiom A4 : G → E
axiom A5 : E → S

theorem answer : ¬K → S := by
  intro hNK  -- Assume kids didn't swim
  have hNP : ¬P := fun hP => hNK (A2 hP)  -- Contrapositive of A2
  cases A3 with
  | inl hNR => 
    have hG : G := A1 hNR
    have hE : E := A4 hG
    exact A5 hE
  | inr hP => 
    contradiction  -- hP and hNP contradict
</lean>

ANSWER: Yes
