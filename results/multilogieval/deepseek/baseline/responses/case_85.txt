axiom Person : Type

axiom WatersAdequately : Person → Prop
axiom BloomsBrightly : Person → Prop
axiom AttractsBees : Person → Prop
axiom UsesOrganicFertilizer : Person → Prop

axiom A4 : ∀ p : Person, UsesOrganicFertilizer p → SoilHighQuality p

theorem answer : ¬AttractsBees John → SoilHighQuality John := by
  intro no_bees
  have cases := A3 John
  cases cases with
  | inl watered => 
      have : BloomsBrightly John := A1 John watered
      have : AttractsBees John := A2 John this
      contradiction
  | inr used_organic => 
      exact A4 John used_organic
