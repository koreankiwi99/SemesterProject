<lean>
axiom Student : Type
axiom Priya : Student
axiom AttendsRegularly : Student → Prop
axiom SpendSufficientTime : Student → Prop
axiom PerformWell : Student → Prop

-- Context: If attends regularly, then sufficient time revising implies performs well
axiom R1 : ∀ s : Student, AttendsRegularly s → (SpendSufficientTime s → PerformWell s)

-- Given: Priya spent sufficient time revising
axiom T1 : SpendSufficientTime Priya
-- Given: Priya did not perform well
axiom T2 : ¬ PerformWell Priya

-- Prove Priya did NOT attend lectures regularly
theorem not_attended : ¬ AttendsRegularly Priya := 
  fun hAttend : AttendsRegularly Priya =>
  have h1 : SpendSufficientTime Priya → PerformWell Priya := R1 Priya hAttend
  have h2 : PerformWell Priya := h1 T1
  show False from T2 h2
</lean>

ANSWER: No
