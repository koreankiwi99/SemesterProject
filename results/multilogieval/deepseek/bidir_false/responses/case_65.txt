<lean>
axiom obj : Type
axiom Bookshelf : obj
axiom Wood : obj
axiom HighQuality : obj → Prop
axiom PreciseJoints : obj → Prop
axiom Sturdy : obj → Prop
axiom LongLasting : obj → Prop

-- Context: If wood is high quality and joints are precise, then product is sturdy and long-lasting
axiom R1 : ∀ (w p : obj), HighQuality w → PreciseJoints p → Sturdy p ∧ LongLasting p

-- Premises: Joints were cut precisely for the bookshelf
axiom T1 : PreciseJoints Bookshelf
-- Premises: Bookshelf was shaky (not sturdy) and fell apart quickly (not long-lasting)
axiom T2 : ¬Sturdy Bookshelf
axiom T3 : ¬LongLasting Bookshelf

-- Prove the wood was NOT high quality
theorem wood_not_high_quality : ¬HighQuality Wood := by
  intro h
  have H : Sturdy Bookshelf ∧ LongLasting Bookshelf := R1 Wood Bookshelf h T1
  cases H with
  | intro s l =>
    exact T2 s   -- Contradiction from not sturdy
    -- Alternatively: exact T3 l   would also work
</lean>

ANSWER: No
