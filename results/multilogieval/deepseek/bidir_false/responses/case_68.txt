<lean>
axiom Student : Type
axiom Exam : Type
axiom Emma : Student
axiom Biology : Exam
axiom StudiesHard : Student → Prop
axiom EnoughSleep : Student → Exam → Prop
axiom AceTest : Student → Exam → Prop
axiom Belief : ∀ (s : Student) (e : Exam), StudiesHard s → EnoughSleep s e → AceTest s e
axiom SleepFact : EnoughSleep Emma Biology
axiom NotAce : ¬ AceTest Emma Biology

theorem emma_not_studied_hard : ¬ StudiesHard Emma :=
  fun h : StudiesHard Emma =>
    have ace : AceTest Emma Biology := Belief Emma Biology h SleepFact
    show False from NotAce ace
</lean>

ANSWER: No
