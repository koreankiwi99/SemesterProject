<lean>
axiom obj : Type
axiom Simone : obj
axiom TrainsWithDedication : obj → Prop
axiom GetsAdequateRest : obj → Prop
axiom PerformsAtPeak : obj → Prop

axiom A1 : ∀ x : obj, TrainsWithDedication x → GetsAdequateRest x → PerformsAtPeak x
axiom Fact1 : GetsAdequateRest Simone
axiom Fact2 : ¬PerformsAtPeak Simone

theorem no_dedication : ¬TrainsWithDedication Simone := by
  intro h
  -- Assume Simone trained with dedication
  have h1 : GetsAdequateRest Simone → PerformsAtPeak Simone := A1 Simone h
  -- Apply A1 to get R → P
  have h2 : PerformsAtPeak Simone := h1 Fact1
  -- Apply modus ponens with Fact1 to get P
  exact Fact2 h2
</lean>

ANSWER: No
