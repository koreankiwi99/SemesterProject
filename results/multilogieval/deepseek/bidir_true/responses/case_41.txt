<lean>
axiom W : Prop -- Priya waters plants daily
axiom D : Prop -- There was a drought
axiom R : Prop -- Water restrictions imposed
axiom E : Prop -- Plants got enough water
axiom G : Prop -- Plants grew well
axiom Died : Prop -- Plants died

axiom P1 : W → G
axiom P2 : D → R
axiom P3 : W ∨ D
axiom P4 : R → ¬E
axiom P5 : ¬E → Died

theorem answer : ¬G → Died := by
  intro h_neg_G
  have h_neg_W : ¬W := fun h_W => h_neg_G (P1 h_W)
  have h_D : D := by
    cases P3 with
    | inl h_W => contradiction
    | inr h_D => exact h_D
  have h_R : R := P2 h_D
  have h_neg_E : ¬E := P4 h_R
  exact P5 h_neg_E
</lean>

ANSWER: Yes
