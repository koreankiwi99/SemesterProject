<lean>
axiom obj : Type
axiom mary : obj
axiom P : obj → Prop  -- practiced skills often
axiom T : obj → Prop  -- technique improved
axiom S : obj → Prop  -- artwork stands out as high quality
axiom C : obj → Prop  -- took an art class
axiom L : obj → Prop  -- learned new methods and mediums

-- If practiced often, then technique improves
axiom A1 : ∀ x, P x → T x
-- If technique improves, then artwork stands out
axiom A2 : ∀ x, T x → S x
-- Either practiced often, or took art class (or both)
axiom A3 : ∀ x, P x ∨ C x
-- If took art class, then learned new methods
axiom A4 : ∀ x, C x → L x

-- Given that Mary's artwork did not stand out
axiom not_S_mary : ¬ S mary

theorem mary_learned : L mary := by
  -- Since ¬S mary and T x → S x, then ¬T mary
  have not_T_mary : ¬ T mary := fun hT : T mary => not_S_mary (A2 mary hT)
  -- Since ¬T mary and P x → T x, then ¬P mary
  have not_P_mary : ¬ P mary := fun hP : P mary => not_T_mary (A1 mary hP)
  -- From A3: P mary ∨ C mary. Since ¬P mary, must have C mary
  have C_mary : C mary := by cases A3 mary with
    | inl hP => exact False.elim (not_P_mary hP)
    | inr hC => exact hC
  -- Apply A4: C mary → L mary
  exact A4 mary C_mary
</lean>

ANSWER: Yes
