<lean>
axiom AddSalt : Prop
axiom Savory : Prop
axiom StirContinuously : Prop
axiom ThickenPerfectly : Prop
axiom Taste : Prop
axiom PrepareSalad : Prop

axiom rule1 : AddSalt → Savory
axiom rule2 : StirContinuously → ThickenPerfectly
axiom law : AddSalt ∨ ¬ThickenPerfectly
axiom rule3 : Taste → StirContinuously
axiom secret : Taste ∨ PrepareSalad

theorem salad_when_not_savory : ¬Savory → PrepareSalad :=
  fun h : ¬Savory =>
  have h1 : ¬AddSalt := by
    intro hAdd
    exact h (rule1 hAdd)
  have h2 : ¬ThickenPerfectly := by
    cases law with
    | inl hAdd => exact absurd hAdd h1
    | inr hNotThick => exact hNotThick
  have h3 : ¬StirContinuously := by
    intro hStir
    exact h2 (rule2 hStir)
  have h4 : ¬Taste := by
    intro hTaste
    exact h3 (rule3 hTaste)
  show PrepareSalad from
    Or.resolve_left secret h4
</lean>

ANSWER: Yes
