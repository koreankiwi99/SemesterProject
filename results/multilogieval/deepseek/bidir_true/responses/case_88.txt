<lean>
axiom obj : Type
axiom OvenTempCorrect : Prop
axiom BatterContainsEggs : Prop
axiom BakesProperly : Prop
axiom IsFluffy : Prop
axiom TastesDelicious : Prop
axiom GuestsHappy : Prop

-- If the oven temperature is set to 350 degrees, the cake will bake properly.
axiom A1 : OvenTempCorrect → BakesProperly
-- If the cake batter contains eggs, it will be fluffy when baked.
axiom A2 : BatterContainsEggs → IsFluffy
-- Either Sam set the oven temperature correctly, or the cake batter did not contain eggs.
axiom A3 : OvenTempCorrect ∨ ¬BatterContainsEggs
-- If a cake bakes properly, it will taste delicious.
axiom A4 : BakesProperly → TastesDelicious
-- If a cake tastes delicious, guests at the party will be happy.
axiom A5 : TastesDelicious → GuestsHappy

theorem guests_happy_if_eggs : BatterContainsEggs → GuestsHappy := by
  intro h_eggs
  -- From A3, since BatterContainsEggs is true, OvenTempCorrect must be true.
  have h_oven : OvenTempCorrect := by
    cases A3 with
    | inl h => exact h
    | inr h => contradiction -- Contradicts h_eggs
  -- Apply A1 to get BakesProperly.
  have h_bakes : BakesProperly := A1 h_oven
  -- Apply A4 to get TastesDelicious.
  have h_tastes : TastesDelicious := A4 h_bakes
  -- Apply A5 to get GuestsHappy.
  exact A5 h_tastes
</lean>

ANSWER: Yes
