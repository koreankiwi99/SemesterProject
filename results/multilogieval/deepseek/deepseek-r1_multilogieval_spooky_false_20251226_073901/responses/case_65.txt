<lean>
axiom obj : Type
axiom wood : obj
axiom joints : obj
axiom product : obj
axiom HighQuality : obj → Prop
axiom PreciseJoints : obj → Prop
axiom SturdyProduct : obj → Prop

-- Premise: If wood is high quality, then precise joints imply sturdy product
axiom A1 : HighQuality wood → (PreciseJoints joints → SturdyProduct product)

-- Given facts
axiom T1 : PreciseJoints joints
axiom T2 : ¬SturdyProduct product

-- Prove wood was NOT high quality
theorem not_high_quality : ¬HighQuality wood :=
  fun h : HighQuality wood =>
    have h1 : SturdyProduct product := A1 h T1
    show False from T2 h1
</lean>

ANSWER: No
