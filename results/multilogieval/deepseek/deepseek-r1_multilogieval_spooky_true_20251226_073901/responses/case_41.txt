<lean>
axiom WaterDaily : Prop
axiom Drought : Prop
axiom Restrictions : Prop
axiom EnoughWater : Prop
axiom GrowWell : Prop
axiom WiltAndDie : Prop

-- Premises
axiom P1 : WaterDaily → GrowWell
axiom P2 : Drought → Restrictions
axiom P3 : WaterDaily ∨ Drought
axiom P4 : Restrictions → ¬ EnoughWater
axiom P5 : ¬ EnoughWater → WiltAndDie

-- Given fact
axiom notGrowWell : ¬ GrowWell

-- We want to prove WiltAndDie
theorem plants_died : WiltAndDie := by
  -- Since plants didn't grow well, they weren't watered daily (contrapositive of P1)
  have notWaterDaily : ¬ WaterDaily := fun hWaterDaily => notGrowWell (P1 hWaterDaily)
  -- From disjunction P3 and ¬WaterDaily, we must have Drought
  have drought : Drought := by
    cases P3 with
    | inl water => contradiction
    | inr d => exact d
  -- From drought and P2, restrictions are imposed
  have restrictions : Restrictions := P2 drought
  -- From restrictions and P4, plants don't get enough water
  have notEnough : ¬ EnoughWater := P4 restrictions
  -- From notEnough and P5, plants wilt and die
  exact P5 notEnough
</lean>

ANSWER: Yes
