axiom obj : Type
axiom Priya : obj
axiom AttendsRegularly : obj → Prop
axiom StudiesDiligently : obj → Prop
axiom ExcelsInCourse : obj → Prop

axiom Rule : ∀ (x : obj), AttendsRegularly x → StudiesDiligently x → ExcelsInCourse x
axiom Diligent : StudiesDiligently Priya
axiom NotExcel : ¬ ExcelsInCourse Priya

theorem not_attended : ¬ AttendsRegularly Priya :=
  fun (hAttend : AttendsRegularly Priya) =>
    have hExcel : ExcelsInCourse Priya := Rule Priya hAttend Diligent
    NotExcel hExcel
</lean>
