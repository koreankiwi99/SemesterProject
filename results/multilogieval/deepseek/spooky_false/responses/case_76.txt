axiom ClearMind : Prop
axiom Organized : Prop
axiom ToDoList : Prop

axiom T1 : Weekend → Relax
axiom T2 : FinishTasks → ClearMind
axiom T3 : Weekend ∨ ¬ClearMind
axiom T4 : Organized → FinishTasks
axiom T5 : ToDoList → Organized
axiom notRelax : ¬Relax

theorem noToDoList : ¬ToDoList := by
  intro hToDo
  have hOrg : Organized := T5 hToDo
  have hFin : FinishTasks := T4 hOrg
  have hClear : ClearMind := T2 hFin
  have hNotWeekend : ¬Weekend := fun w => notRelax (T1 w)
  cases T3 with
  | inl w => exact hNotWeekend w
  | inr notClear => exact notClear hClear
