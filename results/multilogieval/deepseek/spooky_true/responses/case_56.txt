axiom obj : Type
axiom Emma : obj

axiom S : obj → Prop  -- Speaks daily with native speakers
axiom P : obj → Prop  -- Listens to educational podcasts
axiom F : obj → Prop  -- Fluency improves
axiom C : obj → Prop  -- Able to have real conversations
axiom L : obj → Prop  -- Gains valuable listening comprehension

axiom A1 : ∀ x, S x → F x
axiom A2 : ∀ x, F x → C x
axiom A3 : ∀ x, S x ∨ P x
axiom A4 : ∀ x, P x → L x

theorem T1 : ¬ C Emma → L Emma := by
  intro h_notC
  cases A3 Emma with
  | inl hS =>
      have hF := A1 Emma hS
      have hC := A2 Emma hF
      contradiction
  | inr hP =>
      exact A4 Emma hP
