<lean>
axiom Monsoon : Prop
axiom HeavyRains : Prop
axiom Waters : Prop
axiom GrowStrong : Prop
axiom BountifulHarvest : Prop
axiom GoodProfit : Prop

axiom P1 : Waters → GrowStrong
axiom P2 : Monsoon → HeavyRains
axiom P3 : Waters ∨ ¬ HeavyRains
axiom P4 : GrowStrong → BountifulHarvest
axiom P5 : BountifulHarvest → GoodProfit

theorem monsoon_implies_profit : Monsoon → GoodProfit := by
  intro hMonsoon
  have hRains : HeavyRains := P2 hMonsoon
  have hWaters : Waters := by
    cases P3
    case inl h => exact h
    case inr nh => exact False.elim (nh hRains)
  have hStrong : GrowStrong := P1 hWaters
  have hBountiful : BountifulHarvest := P4 hStrong
  exact P5 hBountiful
</lean>

ANSWER: Yes
