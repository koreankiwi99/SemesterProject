<lean>
axiom obj : Type
axiom Alex : obj

axiom StudiedHard : obj → Prop
axiom Participated : obj → Prop
axiom GoodGrades : obj → Prop
axiom MakeNewFriends : obj → Prop
axiom Invited : obj → Prop

axiom R1 : StudiedHard Alex → GoodGrades Alex
axiom R2 : Participated Alex → MakeNewFriends Alex
axiom D1 : StudiedHard Alex ∨ Participated Alex
axiom R3 : ∀ x : obj, MakeNewFriends x → Invited x
axiom NG : ¬ GoodGrades Alex

theorem alex_invited : Invited Alex := by
  have hnotSH : ¬ StudiedHard Alex := by
    intro hsh
    exact NG (R1 hsh)
  have hPart : Participated Alex :=
    Or.elim D1 (fun hs => False.elim (hnotSH hs)) (fun hp => hp)
  have hFriends : MakeNewFriends Alex := R2 hPart
  exact R3 Alex hFriends
</lean>

ANSWER: Yes
