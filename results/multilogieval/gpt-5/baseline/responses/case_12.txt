<lean>
axiom obj : Type
axiom James : obj

axiom Follow : obj → Prop
axiom Watch : obj → Prop
axiom DishesWell : obj → Prop
axiom Confidence : obj → Prop

-- If someone followed recipes precisely, their dishes turn out well.
axiom R1 : ∀ x : obj, Follow x → DishesWell x
-- If someone's dishes turn out well, they gain confidence in the kitchen.
axiom R2 : ∀ x : obj, DishesWell x → Confidence x
-- Either they followed recipes precisely or watched cooking shows (or both).
axiom R3 : ∀ x : obj, Follow x ∨ Watch x

theorem james_watches : (¬ Confidence James) → Watch James := by
  intro hnc
  have hw : Follow James ∨ Watch James := R3 James
  cases hw with
  | inl hf =>
      have hc : Confidence James := R2 James (R1 James hf)
      exact False.elim (hnc hc)
  | inr hwj =>
      exact hwj
</lean>

ANSWER: Yes
