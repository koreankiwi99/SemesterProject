<lean>
axiom person : Type
axiom Sam : person

axiom Impressed : person → Prop
axiom ExtraWork : person → Prop
axiom Experience : person → Prop
axiom Promotion : person → Prop
axiom LeadTeam : person → Prop

-- Rules from the context
axiom R1 : ∀ x : person, Impressed x → Promotion x
axiom R2 : ∀ x : person, ExtraWork x → Experience x
axiom R3 : Impressed Sam ∨ ExtraWork Sam
axiom R4 : ∀ x : person, Experience x → LeadTeam x

-- Given: Sam did not get a promotion
axiom NotProm_Sam : ¬ Promotion Sam

theorem sam_leads_team : LeadTeam Sam :=
by
  have not_imp : ¬ Impressed Sam :=
    fun hImp => NotProm_Sam (R1 Sam hImp)
  have hExtra : ExtraWork Sam :=
    Or.elim R3
      (fun hImp => False.elim (not_imp hImp))
      (fun hEx => hEx)
  have hExp : Experience Sam := R2 Sam hExtra
  exact R4 Sam hExp
</lean>

ANSWER: Yes
