<lean>
axiom A_scores_more : Prop
axiom A_wins : Prop
axiom B_fumbles : Prop
axiom B_loses : Prop

axiom R1 : A_scores_more → A_wins
axiom R2 : B_fumbles → B_loses
axiom R3 : A_scores_more ∨ B_fumbles

theorem b_not_lose_then_a_scores_more : ¬ B_loses → A_scores_more := by
  intro hnotL
  have hnotF : ¬ B_fumbles := fun hf => hnotL (R2 hf)
  have hA : A_scores_more :=
    Or.elim R3
      (fun hA => hA)
      (fun hf => False.elim (hnotF hf))
  exact hA
</lean>

ANSWER: Yes
