<lean>
axiom Person : Type
axiom Lucy : Person

axiom Hacked : Person → Prop
axiom ResetPassword : Person → Prop
axiom NotifyBank : Person → Prop

-- Context rules: if hacked, then must reset password and notify bank
axiom Reset_if_hacked : ∀ x : Person, Hacked x → ResetPassword x
axiom Notify_if_hacked : ∀ x : Person, Hacked x → NotifyBank x

-- Given: Either Lucy did not change her password or did not inform her bank
axiom Lucy_violation : ¬ ResetPassword Lucy ∨ ¬ NotifyBank Lucy

theorem lucy_not_hacked : ¬ Hacked Lucy := by
  intro h
  have r : ResetPassword Lucy := Reset_if_hacked Lucy h
  have n : NotifyBank Lucy := Notify_if_hacked Lucy h
  cases Lucy_violation with
  | inl nr => exact nr r
  | inr nn => exact nn n
</lean>

ANSWER: No
