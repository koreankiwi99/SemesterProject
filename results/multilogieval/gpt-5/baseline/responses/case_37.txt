axiom Prepared : Person → Prop

axiom read_implies_packed : ∀ x : Person, ReadBooks x → PackedEnough x
axiom disj : Nice ∨ ¬ PackedEnough Jake ∨ (Nice ∧ ¬ PackedEnough Jake)
axiom prepared_implies_read : ∀ x : Person, Prepared x → ReadBooks x

-- We cannot prove: ¬Relax Jake → Prepared Jake
-- But we can prove the opposite direction: ¬Relax Jake → ¬Prepared Jake
theorem not_relax_implies_not_prepared : ¬ Relax Jake → ¬ Prepared Jake := by
  intro hnotrelax
  intro hprep
  have hread : ReadBooks Jake := prepared_implies_read Jake hprep
  have hpack : PackedEnough Jake := read_implies_packed Jake hread
  have hnotnice : ¬ Nice := by
    intro hnice
    have hrelax : Relax Jake := nice_implies_relax Jake hnice
    exact hnotrelax hrelax
  have hnotpacked : ¬ PackedEnough Jake := by
    cases disj with
    | inl hnice =>
        exact False.elim (hnotnice hnice)
    | inr h2 =>
        cases h2 with
        | inl hnp => exact hnp
        | inr hboth =>
            exact False.elim (hnotnice hboth.left)
  exact hnotpacked hpack
