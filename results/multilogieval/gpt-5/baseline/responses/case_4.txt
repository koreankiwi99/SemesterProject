<lean>
axiom Marketing : Prop
axiom MoreCustomers : Prop
axiom SalesUp : Prop
axiom ProdCostsInc : Prop
axiom InvestMarketing_implies_MoreCustomers : Marketing → MoreCustomers
axiom MoreCustomers_implies_SalesUp : MoreCustomers → SalesUp
axiom Marketing_or_ProdCostsInc : Marketing ∨ ProdCostsInc
axiom Sales_not_increase : ¬ SalesUp

theorem production_costs_increased : ProdCostsInc := by
  have m_to_s : Marketing → SalesUp := fun m => MoreCustomers_implies_SalesUp (InvestMarketing_implies_MoreCustomers m)
  have not_m : ¬ Marketing := fun m => Sales_not_increase (m_to_s m)
  cases Marketing_or_ProdCostsInc with
  | inl m =>
    exact False.elim (not_m m)
  | inr p =>
    exact p
</lean>

ANSWER: Yes
