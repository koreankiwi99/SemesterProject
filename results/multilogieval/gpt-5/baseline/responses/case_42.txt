axiom Sunny : Prop
axiom WantPlayOutside : Prop
axiom PlayOutside : Prop
axiom SleepWell : Prop
axiom FluSeason : Prop
axiom ManySick : Prop

-- If it is sunny, children will want to play outside.
axiom R1 : Sunny → WantPlayOutside

-- If flu season begins, then many children will get sick.
axiom R2 : FluSeason → ManySick

-- Either it is sunny, or not many kids get the flu, or it was sunny and many kids did not get sick.
axiom D1 : Sunny ∨ ¬ ManySick ∨ (Sunny ∧ ¬ ManySick)

-- If children play outside, they will sleep well at night.
axiom R4 : PlayOutside → SleepWell

-- From flu season and the disjunction, we can derive that it is sunny.
theorem sunny_from_flu (hFlu : FluSeason) : Sunny :=
by
  have hm : ManySick := R2 hFlu
  cases D1 with
  | inl hs => exact hs
  | inr h =>
    cases h with
    | inl hnot =>
      have f : False := hnot hm
      exact False.elim f
    | inr hs_and =>
      have f : False := hs_and.right hm
      exact False.elim f
