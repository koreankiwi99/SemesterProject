<lean>
axiom obj : Type
axiom Casey : obj

axiom Volunteers : obj → Prop
axiom Helps : obj → Prop
axiom Adopts : obj → Prop
axiom ProvidesHome : obj → Prop
axiom Fulfilled : obj → Prop

axiom R1 : ∀ x : obj, Volunteers x → Helps x
axiom R2 : ∀ x : obj, Adopts x → ProvidesHome x
axiom R3 : ∀ x : obj, Volunteers x ∨ (¬ProvidesHome x ∨ (Volunteers x ∧ ¬ProvidesHome x))
axiom R4 : ∀ x : obj, Helps x → Fulfilled x

axiom TCasey : Adopts Casey

theorem casey_fulfilled : Fulfilled Casey := by
  have ph : ProvidesHome Casey := R2 Casey TCasey
  have disj : Volunteers Casey ∨ (¬ProvidesHome Casey ∨ (Volunteers Casey ∧ ¬ProvidesHome Casey)) := R3 Casey
  apply Or.elim disj
  · intro hV
    exact R4 Casey (R1 Casey hV)
  · intro hRight
    apply Or.elim hRight
    · intro hNotPH
      have contra : False := hNotPH ph
      exact False.elim contra
    · intro hAnd
      have hV : Volunteers Casey := And.left hAnd
      exact R4 Casey (R1 Casey hV)
</lean>

ANSWER: Yes
