<lean>
axiom Person : Type
axiom Mary : Person

axiom LearnedBotany : Person → Prop
axiom UnderstoodPlantNeeds : Person → Prop
axiom ThrivingPlants : Person → Prop
axiom SoughtAdvice : Person → Prop

axiom learn_imp_understand : ∀ x : Person, LearnedBotany x → UnderstoodPlantNeeds x
axiom understand_imp_thrive : ∀ x : Person, UnderstoodPlantNeeds x → ThrivingPlants x
axiom learn_or_seek : ∀ x : Person, LearnedBotany x ∨ SoughtAdvice x

axiom mary_not_thrive : ¬ ThrivingPlants Mary

theorem mary_sought_advice : SoughtAdvice Mary := by
  have hImp : LearnedBotany Mary → ThrivingPlants Mary :=
    fun hL => understand_imp_thrive Mary (learn_imp_understand Mary hL)
  have hNotLearn : ¬ LearnedBotany Mary := fun hL => mary_not_thrive (hImp hL)
  have disj : LearnedBotany Mary ∨ SoughtAdvice Mary := learn_or_seek Mary
  cases disj with
  | inl hL => exact False.elim (hNotLearn hL)
  | inr hS => exact hS
</lean>

ANSWER: Yes
