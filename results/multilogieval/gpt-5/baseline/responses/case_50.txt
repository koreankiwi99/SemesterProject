<lean>
axiom obj : Type
axiom Jesse : obj

-- Predicates
axiom SprintFreq : obj → Prop
axiom RunImp : obj → Prop
axiom Goal : obj → Prop
axiom Stretch : obj → Prop
axiom Flexible : obj → Prop
axiom LessProneInjury : obj → Prop

-- Axioms from the context
axiom A1 : ∀ x : obj, SprintFreq x → RunImp x
axiom A2 : ∀ x : obj, RunImp x → Goal x
axiom A3 : SprintFreq Jesse ∨ Stretch Jesse
axiom A4 : ∀ x : obj, Stretch x → (Flexible x ∧ LessProneInjury x)

-- Given: Improving running form and stamina did not happen for Jesse
axiom HnotRunImp : ¬ RunImp Jesse

theorem jesse_flexible_and_lessprone : Flexible Jesse ∧ LessProneInjury Jesse := by
  -- From ¬RunImp Jesse and (SprintFreq → RunImp), get ¬SprintFreq Jesse
  have h_not_sprint : ¬ SprintFreq Jesse := fun hs => HnotRunImp (A1 Jesse hs)
  -- From (SprintFreq Jesse ∨ Stretch Jesse) and ¬SprintFreq Jesse, deduce Stretch Jesse
  have h_stretch : Stretch Jesse :=
    Or.elim A3
      (fun hs => False.elim (h_not_sprint hs))
      (fun ht => ht)
  -- Apply stretching implication to obtain both properties
  exact A4 Jesse h_stretch
</lean>

ANSWER: Yes
