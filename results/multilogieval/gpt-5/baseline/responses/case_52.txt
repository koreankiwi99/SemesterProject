<lean>
axiom obj : Type
axiom James : obj

axiom ForecastRain : Prop
axiom BringUmbrella : obj → Prop
axiom GoOut : obj → Prop
axiom BigGame : Prop
axiom FriendsTogether : Prop
axiom HighViewership : Prop
axiom GoPlay : Prop

-- If the forecast predicts rain, then people will bring an umbrella when they go out.
axiom R1 : ForecastRain → ∀ x : obj, GoOut x → BringUmbrella x

-- If there is a big game on TV, then friends will get together to watch it.
axiom R2 : BigGame → FriendsTogether

-- Either rain is forecasted, or friends did not get together to watch the game,
-- or both rain is forecasted and friends did not get together.
axiom R3 : ForecastRain ∨ ¬ FriendsTogether ∨ (ForecastRain ∧ ¬ FriendsTogether)

-- If the TV viewership is high, then there is a big game on tv.
axiom R4 : HighViewership → BigGame

-- Either the TV viewership is high, or they all decide to go play out, or both.
axiom R5 : HighViewership ∨ GoPlay ∨ (HighViewership ∧ GoPlay)

-- James did not bring an umbrella.
axiom T : ¬ BringUmbrella James

-- Some derivable consequences (note these do not yield GoPlay)
theorem rain_of_highviewership : HighViewership → ForecastRain :=
by
  intro hv
  have bg : BigGame := R4 hv
  have ft : FriendsTogether := R2 bg
  have h3 : ForecastRain ∨ (¬FriendsTogether ∨ (ForecastRain ∧ ¬FriendsTogether)) := R3
  apply Or.elim h3
  · intro rain; exact rain
  · intro h
    apply Or.elim h
    · intro nft
      exact False.elim (nft ft)
    · intro both
      exact both.left

theorem not_goout_james_if_rain : ForecastRain → ¬ GoOut James :=
by
  intro rain
  have imp : GoOut James → BringUmbrella James := (R1 rain) James
  intro go
  have umb : BringUmbrella James := imp go
  exact T umb

theorem not_goout_james_if_highviewership : HighViewership → ¬ GoOut James :=
by
  intro hv
  exact not_goout_james_if_rain (rain_of_highviewership hv)
