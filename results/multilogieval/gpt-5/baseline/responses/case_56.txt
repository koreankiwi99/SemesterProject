<lean>
axiom Person : Type
axiom Emma : Person

axiom SpeaksDaily : Person → Prop
axiom FluencyImproves : Person → Prop
axiom VocabularyImproves : Person → Prop
axiom RealConversations : Person → Prop
axiom ListensPodcasts : Person → Prop
axiom GainsListening : Person → Prop

axiom R1 : ∀ x : Person, SpeaksDaily x → FluencyImproves x ∧ VocabularyImproves x
axiom R2 : ∀ x : Person, FluencyImproves x → RealConversations x
axiom R3 : ∀ x : Person, SpeaksDaily x ∨ ListensPodcasts x
axiom R4 : ∀ x : Person, ListensPodcasts x → GainsListening x

theorem emma_gains_listening :
  (¬ RealConversations Emma) → GainsListening Emma := by
  intro hnr
  have hdisj : SpeaksDaily Emma ∨ ListensPodcasts Emma := R3 Emma
  cases hdisj with
  | inl sd =>
      have hf : FluencyImproves Emma := And.left (R1 Emma sd)
      have hrc : RealConversations Emma := R2 Emma hf
      have : False := hnr hrc
      exact this.elim
  | inr lp =>
      exact R4 Emma lp
</lean>

ANSWER: Yes
