<lean>
axiom Person : Type
axiom John : Person

axiom Alarm : Person → Prop
axiom Ready : Person → Prop
axiom MissBus : Person → Prop
axiom Late : Person → Prop
axiom UpLate : Person → Prop
axiom BedLate : Person → Prop

-- If the alarm went off, then it was time to get ready.
axiom R_alarm_ready : ∀ x : Person, Alarm x → Ready x
-- If they missed the bus, then they'd be late for school.
axiom R_miss_late : ∀ x : Person, MissBus x → Late x
-- Either the alarm went off, or they weren't late for school.
axiom R_alarm_or_not_late : ∀ x : Person, Alarm x ∨ ¬ Late x
-- If they got up late, then they miss the bus.
axiom R_uplate_miss : ∀ x : Person, UpLate x → MissBus x
-- If they went to bed late, then they get up late.
axiom R_bedlate_uplate : ∀ x : Person, BedLate x → UpLate x

-- John knows it is not the time to get ready (treat as the fact ¬ Ready John)
axiom H_not_ready : ¬ Ready John

theorem john_not_bedlate : ¬ BedLate John := by
  intro hBed
  have hUp : UpLate John := R_bedlate_uplate John hBed
  have hMiss : MissBus John := R_uplate_miss John hUp
  have hLate : Late John := R_miss_late John hMiss
  have hNotAlarm : ¬ Alarm John := by
    intro hAlarm
    exact H_not_ready (R_alarm_ready John hAlarm)
  have hNotLate : ¬ Late John := by
    have hOr : Alarm John ∨ ¬ Late John := R_alarm_or_not_late John
    cases hOr with
    | inl hA =>
        have hFalse : False := hNotAlarm hA
        exact fun _ => False.elim hFalse
    | inr hNL =>
        exact hNL
  exact hNotLate hLate
</lean>
