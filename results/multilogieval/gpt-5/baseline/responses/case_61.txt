<lean>
axiom obj : Type
axiom Priya : obj
axiom Attend : obj → Prop
axiom Study : obj → Prop
axiom Excel : obj → Prop

-- If a student attends lectures regularly, then if they study diligently, they will excel.
axiom R1 : ∀ x : obj, Attend x → (Study x → Excel x)

-- Facts about Priya
axiom Priya_Studied : Study Priya
axiom Priya_Not_Excel : ¬ Excel Priya

-- Conclusion: Priya did not attend lectures regularly
theorem priya_not_attend : ¬ Attend Priya := by
  intro hAttend
  have hExcel : Excel Priya := (R1 Priya) hAttend Priya_Studied
  exact Priya_Not_Excel hExcel
</lean>

ANSWER: No
