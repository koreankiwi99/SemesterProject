axiom obj : Type
axiom Mark : obj

axiom MadeFriends : obj → Prop
axiom Enjoy : obj → Prop
axiom JoinClub : obj → Prop
axiom Busy : obj → Prop
axiom Outgoing : obj → Prop
axiom Introduced : obj → Prop

axiom A1 : ∀ x : obj, MadeFriends x → Enjoy x
axiom A2 : ∀ x : obj, JoinClub x → Busy x
axiom A3 : ∀ x : obj, MadeFriends x ∨ ¬ Busy x
axiom A4 : ∀ x : obj, Outgoing x → JoinClub x
axiom A5 : ∀ x : obj, Introduced x → Outgoing x

-- From the assumptions we can derive: if Mark did not enjoy, then Mark did not introduce himself.
theorem not_enjoy_implies_not_introduced_mark : ¬ Enjoy Mark → ¬ Introduced Mark :=
  fun hNotEnjoy =>
    fun hIntro =>
      have hOut : Outgoing Mark := A5 Mark hIntro
      have hJoin : JoinClub Mark := A4 Mark hOut
      have hBusy : Busy Mark := A2 Mark hJoin
      have hNotBusy : ¬ Busy Mark :=
        by
          have disj : MadeFriends Mark ∨ ¬ Busy Mark := A3 Mark
          cases disj with
          | inl hMF =>
              exact (fun _hB : Busy Mark => hNotEnjoy (A1 Mark hMF))
          | inr hNB => exact hNB
      show False from hNotBusy hBusy
