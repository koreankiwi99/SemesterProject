<lean>
axiom Person : Type
axiom Priya : Person
axiom Attend : Person → Prop
axiom Revise : Person → Prop
axiom PerformWell : Person → Prop

-- If someone attends lectures regularly, then if they revise sufficiently, they will perform well
axiom rule : ∀ x : Person, Attend x → Revise x → PerformWell x

-- Facts about Priya
axiom priya_revise : Revise Priya
axiom priya_not_well : ¬ PerformWell Priya

theorem priya_not_attend : ¬ Attend Priya := by
  intro hAttend
  have hImp : Revise Priya → PerformWell Priya := rule Priya hAttend
  have hWell : PerformWell Priya := hImp priya_revise
  exact priya_not_well hWell
</lean>

ANSWER: No
