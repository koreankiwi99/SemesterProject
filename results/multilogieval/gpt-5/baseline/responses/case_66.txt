<lean>
axiom obj : Type
axiom Gadget : obj
axiom Innovative : obj → Prop
axiom MarketedEff : obj → Prop
axiom CommerciallySuccessful : obj → Prop

-- General rule: if a product is innovative, then if it is marketed effectively, it will be commercially successful
axiom Rule : ∀ x : obj, Innovative x → (MarketedEff x → CommerciallySuccessful x)

-- Facts about Acme's gadget
axiom marketed_gadget : MarketedEff Gadget
axiom not_success_gadget : ¬CommerciallySuccessful Gadget

-- Conclusion: Acme's gadget was not innovative
theorem gadget_not_innovative : ¬Innovative Gadget := by
  intro hInnov
  have hSucc : CommerciallySuccessful Gadget := (Rule Gadget hInnov) marketed_gadget
  exact not_success_gadget hSucc
</lean>

ANSWER: No
