<lean>
axiom person : Type
axiom exam : Type

axiom Emma : person
axiom Biology : exam

axiom StudiesHard : person → Prop
axiom SleepBefore : person → exam → Prop
axiom PerformWell : person → exam → Prop
axiom Ace : person → exam → Prop

-- If someone studies hard, then if they get enough sleep before an exam,
-- they will perform well and ace the test.
axiom Belief :
  ∀ x : person, ∀ e : exam,
    StudiesHard x → SleepBefore x e → (PerformWell x e ∧ Ace x e)

axiom EmmaSlept : SleepBefore Emma Biology
axiom EmmaNotAce : ¬ Ace Emma Biology

theorem emma_not_study_hard : ¬ StudiesHard Emma :=
  fun hStudy => by
    have h := Belief Emma Biology hStudy EmmaSlept
    have hAce : Ace Emma Biology := And.right h
    exact EmmaNotAce hAce
</lean>

ANSWER: No
