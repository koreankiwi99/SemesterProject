<lean>
axiom obj : Type
axiom Lebron : obj

axiom TrainsHard : obj → Prop
axiom Focused : obj → Prop
axiom Best : obj → Prop

-- If a player trains hard, then if they remain focused, they will perform to the best of their ability.
axiom coaches_rule : ∀ x : obj, TrainsHard x → (Focused x → Best x)

axiom Lebron_focused : Focused Lebron
axiom Lebron_not_best : ¬ Best Lebron

theorem lebron_not_trains_hard : ¬ TrainsHard Lebron := by
  intro hTH
  have hBest : Best Lebron := (coaches_rule Lebron hTH) Lebron_focused
  exact Lebron_not_best hBest
</lean>

ANSWER: No
