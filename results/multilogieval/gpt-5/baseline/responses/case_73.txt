<lean>
axiom obj : Type
axiom Simone : obj

axiom TrainsWithDedication : obj → Prop
axiom AdequateRest : obj → Prop
axiom PeakOnGameDay : obj → Prop

-- If someone trains with dedication, then if they get adequate rest, they will perform at their peak on game day.
axiom CoachRule : ∀ x : obj, TrainsWithDedication x → AdequateRest x → PeakOnGameDay x

-- Facts about Simone
axiom Rest_Simone : AdequateRest Simone
axiom NotPeak_Simone : ¬ PeakOnGameDay Simone

theorem simone_not_train : ¬ TrainsWithDedication Simone := by
  intro hTrain
  apply NotPeak_Simone
  exact CoachRule Simone hTrain Rest_Simone
</lean>
