<lean>
axiom obj : Type
axiom Chef : obj
axiom UsedFresh : obj → Prop
axiom CookedComplicated : obj → Prop
axiom Delicious : obj → Prop
axiom TakesMoreTime : obj → Prop

-- If a chef uses fresh ingredients, their food will taste delicious
axiom R1 : ∀ x : obj, UsedFresh x → Delicious x
-- If a chef cooks a complicated recipe, it takes more time to prepare
axiom R2 : ∀ x : obj, CookedComplicated x → TakesMoreTime x
-- The chef used fresh ingredients
axiom T1 : UsedFresh Chef

theorem complicated_implies_delicious : CookedComplicated Chef → Delicious Chef :=
by
  have d : Delicious Chef := R1 Chef T1
  intro _
  exact d
</lean>

ANSWER: Yes
