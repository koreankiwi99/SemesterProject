<lean>
axiom Sunny : Prop
axiom KidsPlayPark : Prop
axiom TeamPracticedDaily : Prop
axiom TeamPrepared : Prop
axiom TeamConfident : Prop
axiom TeamHeadHigh : Prop

axiom R1 : TeamPracticedDaily → TeamPrepared
axiom R2 : Sunny → KidsPlayPark
axiom R3 : TeamPracticedDaily ∨ ¬ KidsPlayPark
axiom R4 : TeamPrepared → TeamConfident
axiom R5 : TeamConfident → TeamHeadHigh

theorem sunny_team_head_high : Sunny → TeamHeadHigh := by
  intro hsun
  have hkids : KidsPlayPark := R2 hsun
  have hpract : TeamPracticedDaily :=
    Or.elim R3 (fun hp => hp) (fun hnot => False.elim (hnot hkids))
  have hprep : TeamPrepared := R1 hpract
  have hconf : TeamConfident := R4 hprep
  exact R5 hconf
</lean>

ANSWER: Yes
