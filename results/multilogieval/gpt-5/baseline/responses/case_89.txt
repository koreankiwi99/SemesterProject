<lean>
axiom Monsoon : Prop
axiom HeavyRains : Prop
axiom Water : Prop
axiom Healthy : Prop
axiom Strong : Prop
axiom Bountiful : Prop
axiom Profit : Prop

axiom R1 : Water → Healthy ∧ Strong
axiom R2 : Monsoon → HeavyRains
axiom R3 : Water ∨ ¬HeavyRains
axiom R4 : Strong → Bountiful
axiom R5 : Bountiful → Profit

theorem monsoon_profit : Monsoon → Profit := by
  intro hm
  have hr : HeavyRains := R2 hm
  have w : Water := Or.elim R3 (fun hw => hw) (fun nhr => False.elim (nhr hr))
  have hs : Healthy ∧ Strong := R1 w
  have s : Strong := And.right hs
  have b : Bountiful := R4 s
  exact R5 b
</lean>

ANSWER: Yes
