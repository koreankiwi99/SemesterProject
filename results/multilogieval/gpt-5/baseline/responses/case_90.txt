<lean>
axiom Winning : Prop
axiom Rivalry : Prop
axiom Invested : Prop
axiom LoudEnergized : Prop
axiom Electric : Prop
axiom Unforgettable : Prop

axiom A1 : Winning → LoudEnergized
axiom A2 : Rivalry → Invested
axiom A3 : Winning ∨ Rivalry
axiom A4 : LoudEnergized → Electric
axiom A5 : Electric → Unforgettable

theorem electric_if_not_invested : ¬Invested → Electric := by
  intro hNotInv
  have hNotR : ¬Rivalry := by
    intro hr
    exact hNotInv (A2 hr)
  have hWin : Winning := by
    cases A3 with
    | inl hw => exact hw
    | inr hr =>
      have hFalse : False := hNotR hr
      exact False.elim hFalse
  have hLE : LoudEnergized := A1 hWin
  exact A4 hLE
</lean>

ANSWER: Yes
