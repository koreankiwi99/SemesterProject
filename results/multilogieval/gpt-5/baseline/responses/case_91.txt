<lean>
axiom Party : Prop
axiom Drums : Prop
axiom Noise : Prop
axiom Complaints : Prop
axiom PoliceCalled : Prop

axiom R1 : Party → Noise
axiom R2 : Drums → Noise
axiom E1 : Party ∨ Drums
axiom R3 : Party → Complaints
axiom R4 : Complaints → PoliceCalled

theorem no_noise_implies_police_called : ¬Noise → PoliceCalled := by
  intro hNoNoise
  have hNoise : Noise := Or.elim E1 (fun hP => R1 hP) (fun hD => R2 hD)
  have contra : False := hNoNoise hNoise
  exact False.elim contra
</lean>

ANSWER: Yes
