<lean>
axiom Water : Prop
axiom GrowWell : Prop
axiom PestsInvaded : Prop
axiom DamagedCrops : Prop
axiom BountifulHarvest : Prop
axiom GoodProfit : Prop

axiom R1 : Water → GrowWell
axiom R2 : PestsInvaded → DamagedCrops
axiom R3 : Water ∨ ¬DamagedCrops ∨ (Water ∧ DamagedCrops)
axiom R4 : GrowWell → BountifulHarvest
axiom R5 : BountifulHarvest → GoodProfit

theorem pests_profit : PestsInvaded → GoodProfit := by
  intro p
  have d : DamagedCrops := R2 p
  cases R3 with
  | inl w =>
      exact R5 (R4 (R1 w))
  | inr h =>
      cases h with
      | inl nd =>
          exact False.elim (nd d)
      | inr wd =>
          have w : Water := wd.left
          exact R5 (R4 (R1 w))
</lean>

ANSWER: Yes
