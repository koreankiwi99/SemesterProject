<lean>
axiom PracticedHard : Prop
axiom Prepared : Prop
axiom VeryHotOutside : Prop
axiom GamePostponed : Prop
axiom PlaysWell : Prop
axiom Wins : Prop

axiom R1 : PracticedHard → Prepared
axiom R2 : VeryHotOutside → GamePostponed
axiom R3 : PracticedHard ∨ ¬ GamePostponed
axiom R4 : Prepared → PlaysWell
axiom R5 : PlaysWell → Wins

theorem hot_implies_win : VeryHotOutside → Wins := by
  intro hhot
  have gp : GamePostponed := R2 hhot
  cases R3 with
  | inl hph =>
    exact R5 (R4 (R1 hph))
  | inr hngp =>
    have contra : False := hngp gp
    exact False.elim contra
</lean>

ANSWER: Yes
