<lean>
axiom Rain : Prop
axiom Saturated : Prop
axiom Watered : Prop
axiom Bloom : Prop
axiom Fragrance : Prop
axiom Butterflies : Prop

axiom R1 : Watered → Bloom
axiom R2 : Rain → Saturated
axiom R3 : Watered ∨ ¬Saturated
axiom R4 : Bloom → Fragrance
axiom R5 : Fragrance → Butterflies

theorem garden_attracts_from_rain : Rain → Butterflies := by
  intro hr
  have hs : Saturated := R2 hr
  have hw : Watered := Or.elim R3 (fun hw => hw) (fun hns => False.elim (hns hs))
  have hb : Bloom := R1 hw
  have hf : Fragrance := R4 hb
  exact R5 hf
</lean>

ANSWER: Yes
