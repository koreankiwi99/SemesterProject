-- If someone participates actively, they gain leadership skills.

-- If someone is a leader, they will be selected as class prefect.
axiom R3 : ∀ x : obj, Leader x → Prefect x

-- Given: the student did not excel academically.
axiom NotExcel : ¬Excel Student

-- From the premises and ¬Excel, we can in fact prove the student is class prefect.

    exact NotExcel (R1 Student hs)
  -- From (Study ∨ Participate) and ¬Study, derive Participate.

    cases E1 with
    | inl hs => exact False.elim (notStudy hs)
    | inr hp => exact hp
  -- Then Leader and hence Prefect.
  have leader : Leader Student := R2 Student part
  exact R3 Student leader
