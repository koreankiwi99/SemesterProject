<lean>
axiom person : Type
axiom James : person

axiom Practice : person → Prop
axiom Prepared : person → Prop
axiom Confident : person → Prop
axiom JoinClub : person → Prop

-- If they practice thoroughly, they feel prepared.
axiom A1 : ∀ x : person, Practice x → Prepared x
-- If they feel prepared, they can speak confidently.
axiom A2 : ∀ x : person, Prepared x → Confident x
-- Either they practiced thoroughly or they joined a public speaking club (or both).
axiom A3 : Practice James ∨ JoinClub James

-- Given: James cannot speak confidently in public.
axiom H1 : ¬Confident James

-- From ¬Confident and Prepared → Confident, derive ¬Prepared for James.
lemma not_prepared : ¬Prepared James := by
  intro hp
  have hc : Confident James := A2 James hp
  exact H1 hc

-- From ¬Prepared and Practice → Prepared, derive ¬Practice for James.
lemma not_practice : ¬Practice James := by
  intro hpr
  have hp : Prepared James := A1 James hpr
  exact not_prepared hp

-- From (Practice ∨ JoinClub) and ¬Practice, derive JoinClub.
theorem joined : JoinClub James := by
  cases A3 with
  | inl hp => exact False.elim (not_practice hp)
  | inr hc => exact hc

-- Therefore, we cannot prove ¬JoinClub James; in fact, we have JoinClub James.
