<lean>
axiom obj : Type
axiom Linda : obj

axiom Cold : obj → Prop
axiom Chill : obj → Prop
axiom AddLayers : obj → Prop
axiom Adjust : obj → Prop

-- If someone is cold, they should put on more layers.
axiom R1 : ∀ x : obj, Cold x → AddLayers x
-- If someone feels a chill, they should turn up the heat (adjust thermostat).
axiom R2 : ∀ x : obj, Chill x → Adjust x

-- Given: Linda is either not adding layers or not adjusting the thermostat.
axiom H : ¬AddLayers Linda ∨ ¬Adjust Linda

-- From not adding layers, we can derive not cold (by contrapositive reasoning via R1).
theorem not_cold_from_not_layers : ¬AddLayers Linda → ¬Cold Linda := by
  intro hnl
  intro hc
  have hadd : AddLayers Linda := R1 Linda hc
  exact hnl hadd

-- From not adjusting the thermostat, we can derive not feeling a chill (via R2).
theorem not_chill_from_not_adjust : ¬Adjust Linda → ¬Chill Linda := by
  intro hna
  intro hch
  have hadj : Adjust Linda := R2 Linda hch
  exact hna hadj

-- From the given disjunction, we can only conclude (¬Cold Linda) ∨ (¬Chill Linda),
-- which is insufficient to derive ¬Cold Linda outright.
theorem negs_disj : (¬Cold Linda) ∨ (¬Chill Linda) := by
  cases H with
  | inl hnl =>
      exact Or.inl (not_cold_from_not_layers hnl)
  | inr hna =>
      exact Or.inr (not_chill_from_not_adjust hna)
