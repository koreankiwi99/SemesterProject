<lean>
axiom obj : Type
axiom City : obj

axiom Drought : obj → Prop
axiom WaterRestrictions : obj → Prop
axiom Rationing : obj → Prop

-- If a city experiences a drought, authorities should implement water restrictions.
axiom R1 : ∀ x : obj, Drought x → WaterRestrictions x
-- Drought conditions require rationing water usage.
axiom R2 : ∀ x : obj, Drought x → Rationing x

-- Given statement: Either the city did not enforce water limits, or it did not ration water usage.
axiom H : ¬WaterRestrictions City ∨ ¬Rationing City

theorem city_not_in_drought : ¬Drought City := by
  cases H with
  | inl hwr =>
      exact fun d => hwr (R1 City d)
  | inr hr =>
      exact fun d => hr (R2 City d)
</lean>

ANSWER: No
