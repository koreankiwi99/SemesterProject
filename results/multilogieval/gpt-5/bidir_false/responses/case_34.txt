<lean>
axiom obj : Type
axiom DrLee : obj

axiom Professor : obj → Prop
axiom Collaborates : obj → Prop
axiom Publishes : obj → Prop
axiom Expert : obj → Prop
axiom Recognized : obj → Prop
axiom Curious : obj → Prop
axiom Asks : obj → Prop

-- Publishing research makes one an expert
axiom R1 : ∀ x : obj, Publishes x → Expert x
-- Collaborating with top scholars brings recognition
axiom R2 : ∀ x : obj, Collaborates x → Recognized x
-- Professors are neither considered experts nor recognized leaders
axiom A1 : ∀ x : obj, Professor x → ¬Expert x ∧ ¬Recognized x
-- Either research was not published or collaborations did not happen (for professors)
axiom A2 : ∀ x : obj, Professor x → (¬Publishes x ∨ ¬Collaborates x)
-- Being curious and asking questions leads to publishing original research
axiom R3 : ∀ x : obj, (Curious x ∧ Asks x) → Publishes x

-- Dr. Lee is a professor at the university
axiom ProfDL : Professor DrLee
-- Statement to evaluate: Dr. Lee collaborates with top scholars
axiom CollabDL : Collaborates DrLee

theorem drlee_not_curious_and_asks : ¬(Curious DrLee ∧ Asks DrLee) := by
  intro hCA
  have hpub : Publishes DrLee := R3 DrLee hCA
  have hnpub : ¬Publishes DrLee := by
    have hdisj : ¬Publishes DrLee ∨ ¬Collaborates DrLee := A2 DrLee ProfDL
    apply Or.elim hdisj
    · intro h
      exact h
    · intro hnc
      have : False := hnc CollabDL
      exact False.elim this
  exact hnpub hpub
</lean>
