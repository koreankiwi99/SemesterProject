<lean>
axiom obj : Type
axiom Priya : obj

axiom Student : obj → Prop
axiom Attend : obj → Prop
axiom Study : obj → Prop
axiom Excel : obj → Prop

-- If a student attends lectures regularly, then if they study diligently, they will excel.
axiom R : ∀ x : obj, Student x → Attend x → (Study x → Excel x)

-- Facts about Priya
axiom Priya_student : Student Priya
axiom Priya_studied : Study Priya
axiom Priya_not_excel : ¬ Excel Priya

theorem Priya_not_attend : ¬ Attend Priya := by
  intro hAttend
  have hImp : Study Priya → Excel Priya := R Priya Priya_student hAttend
  have hExcel : Excel Priya := hImp Priya_studied
  exact Priya_not_excel hExcel
</lean>

ANSWER: No
