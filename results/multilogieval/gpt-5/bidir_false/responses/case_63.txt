<lean>
axiom obj : Type
axiom Linda : obj
axiom PlantsResistant : obj → Prop
axiom WatersFeedsProperly : obj → Prop
axiom BountifulHarvest : obj → Prop

-- If someone plants disease-resistant varieties, then if they water and feed properly, they will have a bountiful harvest.
axiom R1 : ∀ x : obj, PlantsResistant x → (WatersFeedsProperly x → BountifulHarvest x)

-- Facts about Linda
axiom W_Linda : WatersFeedsProperly Linda
axiom not_B_Linda : ¬ BountifulHarvest Linda

theorem linda_not_plants_resistant : ¬ PlantsResistant Linda :=
by
  intro hP
  have hImp : WatersFeedsProperly Linda → BountifulHarvest Linda := R1 Linda hP
  have hB : BountifulHarvest Linda := hImp W_Linda
  exact not_B_Linda hB
</lean>

ANSWER: No
