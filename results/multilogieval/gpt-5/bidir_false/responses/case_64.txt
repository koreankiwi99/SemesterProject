<lean>
axiom obj : Type
axiom Priya : obj

axiom Attend : obj → Prop
axiom Revise : obj → Prop
axiom Perform : obj → Prop

-- Context: If a student attends lectures regularly, then if they revise sufficiently, they perform well.
axiom R1 : ∀ x : obj, Attend x → (Revise x → Perform x)

-- Given statements about Priya
axiom T_rev : Revise Priya
axiom Not_perf : ¬Perform Priya

theorem priya_not_attend : ¬Attend Priya := by
  intro hAttend
  have hImp : Revise Priya → Perform Priya := R1 Priya hAttend
  have hPerf : Perform Priya := hImp T_rev
  exact Not_perf hPerf
</lean>

ANSWER: No
