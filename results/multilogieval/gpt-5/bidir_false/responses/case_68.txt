<lean>
axiom Person : Type
axiom Exam : Type
axiom Emma : Person
axiom BioExam : Exam

axiom StudiesHard : Person → Prop
axiom EnoughSleepBefore : Person → Exam → Prop
axiom PerformsWell : Person → Exam → Prop
axiom Aces : Person → Exam → Prop

axiom Rule :
  ∀ x : Person, StudiesHard x →
  ∀ e : Exam, EnoughSleepBefore x e → PerformsWell x e ∧ Aces x e

axiom SleepEmma : EnoughSleepBefore Emma BioExam
axiom NotAceEmma : ¬Aces Emma BioExam

theorem emma_not_studyhard : ¬StudiesHard Emma :=
by
  intro hStudy
  have hAll :
    ∀ e : Exam, EnoughSleepBefore Emma e → PerformsWell Emma e ∧ Aces Emma e :=
    Rule Emma hStudy
  have hPair : PerformsWell Emma BioExam ∧ Aces Emma BioExam := hAll BioExam SleepEmma
  have hAce : Aces Emma BioExam := hPair.right
  exact NotAceEmma hAce
</lean>

ANSWER: No
