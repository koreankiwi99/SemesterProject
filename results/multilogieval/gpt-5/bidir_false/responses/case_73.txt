<lean>
axiom obj : Type
axiom Simone : obj

axiom Dedicated : obj → Prop
axiom Rest : obj → Prop
axiom Peak : obj → Prop

-- If someone trains with dedication, then if they get adequate rest, they will perform at their peak
axiom R1 : ∀ x : obj, Dedicated x → (Rest x → Peak x)

-- Facts about Simone
axiom Rest_Simone : Rest Simone
axiom NotPeak_Simone : ¬ Peak Simone

theorem simone_not_dedicated : ¬ Dedicated Simone := by
  intro hD
  have hImp : Rest Simone → Peak Simone := R1 Simone hD
  have hPeak : Peak Simone := hImp Rest_Simone
  exact NotPeak_Simone hPeak
</lean>
