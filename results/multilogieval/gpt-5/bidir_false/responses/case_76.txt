<lean>
axiom obj : Type
axiom Alex : obj

axiom WeekendComing : Prop
axiom Relax : obj → Prop
axiom ClearMind : obj → Prop
axiom Finished : obj → Prop
axiom Organized : obj → Prop
axiom TodoList : obj → Prop

-- If it is the weekend coming up, then they relax
axiom R1 : ∀ x : obj, WeekendComing → Relax x
-- If someone finishes tasks, then they have a clear mind
axiom R2 : ∀ x : obj, Finished x → ClearMind x
-- Either the weekend is coming, or Alex does not have a clear mind
axiom R3 : WeekendComing ∨ ¬ClearMind Alex
-- If someone is organized, then they finish their tasks
axiom R4 : ∀ x : obj, Organized x → Finished x
-- If someone made a to-do list, then they are organized
axiom R5 : ∀ x : obj, TodoList x → Organized x

-- From the question: Alex does not relax this weekend
axiom NR : ¬Relax Alex

theorem no_todo_for_Alex : ¬TodoList Alex :=
by
  intro hTodo
  have hOrg : Organized Alex := R5 Alex hTodo
  have hFin : Finished Alex := R4 Alex hOrg
  have hCM  : ClearMind Alex := R2 Alex hFin
  have hWk  : WeekendComing :=
    by
      cases R3 with
      | inl w => exact w
      | inr ncm => exact False.elim (ncm hCM)
  have hRelax : Relax Alex := R1 Alex hWk
  exact NR hRelax
</lean>

ANSWER: No
