<lean>
axiom obj : Type
axiom James : obj

axiom PracticedHard : obj → Prop
axiom PlayWell : obj → Prop
axiom PickedStart : obj → Prop
axiom Talented : obj → Prop
axiom NaturalTalent : obj → Prop
axiom TopYouth : obj → Prop

-- If someone practiced hard, then they would play well
axiom R1 : ∀ x : obj, PracticedHard x → PlayWell x
-- If someone is picked to start, then they are talented
axiom R2 : ∀ x : obj, PickedStart x → Talented x
-- Either they practiced hard, or they are not talented
axiom R3 : ∀ x : obj, PracticedHard x ∨ ¬Talented x
-- If someone has natural talent, then they get picked to start
axiom R4 : ∀ x : obj, NaturalTalent x → PickedStart x
-- If someone was a top player in youth leagues, then they have natural talent
axiom R5 : ∀ x : obj, TopYouth x → NaturalTalent x

-- From not playing well, we can derive not a top player in youth leagues
theorem not_youth_of_not_playwell : ¬PlayWell James → ¬TopYouth James := by
  intro hnpw
  have hnp : ¬PracticedHard James := by
    intro hp
    exact hnpw (R1 James hp)
  have hnt : ¬Talented James := by
    have dis := R3 James
    cases dis with
    | inl hp =>
        exact False.elim (hnp hp)
    | inr nt =>
        exact nt
  have hnps : ¬PickedStart James := by
    intro ps
    exact hnt (R2 James ps)
  have hnab : ¬NaturalTalent James := by
    intro nat
    exact hnps (R4 James nat)
  intro ty
  exact hnab (R5 James ty)
