<lean>
axiom Person : Type
axiom John : Person

axiom Rain : Prop
axiom Cold : Prop
axiom HotShower : Person → Prop
axiom Relax : Person → Prop
axiom WorkedHard : Person → Prop
axiom Sore : Person → Prop

-- Universal rules
axiom hot_relax : ∀ x : Person, HotShower x → Relax x
axiom cold_hot : ∀ x : Person, Cold → HotShower x
axiom rain_cold : Rain → Cold

-- Facts specific to John from the story
axiom hard_or_not_relax : WorkedHard John ∨ ¬Relax John
axiom hard_sore : WorkedHard John → Sore John

-- Useful derived lemmas
theorem rain_implies_relax : Rain → Relax John := by
  intro hR
  have hC : Cold := rain_cold hR
  have hH : HotShower John := cold_hot John hC
  exact hot_relax John hH

theorem rain_implies_hard : Rain → WorkedHard John := by
  intro hR
  have hRelax : Relax John := rain_implies_relax hR
  have hDisj : WorkedHard John ∨ ¬Relax John := hard_or_not_relax
  cases hDisj with
  | inl hHard =>
      exact hHard
  | inr hNotRelax =>
      have hFalse : False := hNotRelax hRelax
      exact False.elim hFalse

theorem rain_implies_sore : Rain → Sore John := by
  intro hR
  have hHard : WorkedHard John := rain_implies_hard hR
  exact hard_sore hHard

theorem not_sore_implies_not_rain : ¬Sore John → ¬Rain := by
  intro hNotSore
  intro hRain
  have hSore : Sore John := rain_implies_sore hRain
  exact hNotSore hSore
