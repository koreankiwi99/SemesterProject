<lean>
axiom obj : Type
axiom John : obj

axiom FinishedAhead : obj → Prop
axiom LongLunch : obj → Prop
axiom BringTreats : obj → Prop
axiom Recharged : obj → Prop
axiom ProductiveAfternoon : obj → Prop
axiom Birthday : Prop

-- If someone finishes ahead, they take a long lunch.
axiom R1 : ∀ x : obj, FinishedAhead x → LongLunch x
-- If it's a coworker's birthday, people (everyone) bring in treats.
axiom R2 : Birthday → ∀ x : obj, BringTreats x
-- Either someone finishes ahead, or it's a birthday (or both).
axiom R3 : (∃ x : obj, FinishedAhead x) ∨ Birthday
-- Long lunch ⇒ recharged.
axiom R4 : ∀ x : obj, LongLunch x → Recharged x
-- Recharged ⇒ very productive afternoon.
axiom R5 : ∀ x : obj, Recharged x → ProductiveAfternoon x

-- From "John did not bring treats" we can derive "not Birthday".
theorem not_birthday_of_not_bring_john : ¬BringTreats John → ¬Birthday :=
by
  intro hnb
  intro hb
  have hall : ∀ x : obj, BringTreats x := R2 hb
  have hj : BringTreats John := hall John
  exact hnb hj

-- If John finished ahead, then John had a very productive afternoon.
theorem productive_if_finished_john : FinishedAhead John → ProductiveAfternoon John :=
by
  intro hfa
  have hl : LongLunch John := R1 John hfa
  have hr : Recharged John := R4 John hl
  exact R5 John hr
