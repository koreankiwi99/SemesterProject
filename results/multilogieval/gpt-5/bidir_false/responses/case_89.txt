theorem daily_from_monsoon : Monsoon → WatersDaily := by
  intro m
  have hr : HeavyRains := A2 m
  have hnn : ¬¬HeavyRains := by
    intro nh
    exact nh hr
  exact Or.elim A3
    (fun wd => wd)
    (fun nhr => False.elim (hnn nhr))

  intro m
  have wd : WatersDaily := daily_from_monsoon m
  have w : Waters := A6 wd
  have s : Strong := A1 w
  have b : BountifulHarvest := A4 s
