<lean>
axiom Party : Type
axiom SamParty : Party

axiom DishBrought : Party → Prop
axiom EnoughFood : Party → Prop
axiom Birthday : Party → Prop
axiom Cake : Party → Prop
axiom IceCream : Party → Prop

-- If someone brought a dish to share, there will be enough food for everyone.
axiom R1 : ∀ p : Party, DishBrought p → EnoughFood p

-- If it is a birthday party, there will be cake and ice cream.
axiom R2 : ∀ p : Party, Birthday p → Cake p ∧ IceCream p

-- For the party in question, either someone brought a dish to share, or there was no birthday cake.
axiom R3 : DishBrought SamParty ∨ ¬Cake SamParty

-- It is Sam's birthday party.
axiom S1 : Birthday SamParty

theorem enough_food_for_everyone : EnoughFood SamParty := by
  cases R3 with
  | inl hDish =>
    exact R1 SamParty hDish
  | inr hNoCake =>
    have hCake : Cake SamParty := And.left (R2 SamParty S1)
    exact False.elim (hNoCake hCake)
