<lean>
axiom obj : Type
axiom Alex : obj

axiom Studied : obj → Prop
axiom Participated : obj → Prop
axiom GoodGrades : obj → Prop
axiom NewFriends : obj → Prop
axiom Invited : obj → Prop

-- If Alex studied hard, then Alex gets good grades
axiom R1 : Studied Alex → GoodGrades Alex
-- If Alex participated in extracurriculars, then Alex makes new friends
axiom R2 : Participated Alex → NewFriends Alex
-- If someone makes new friends, they get invited to social events (applied to Alex)
axiom R3 : NewFriends Alex → Invited Alex

-- Alex either studied hard or participated in extracurricular activities
axiom D : Studied Alex ∨ Participated Alex
-- Given: Alex did not get good grades
axiom NotGG : ¬ GoodGrades Alex

theorem alex_invited : Invited Alex := by
  cases D with
  | inl hS =>
    have gg : GoodGrades Alex := R1 hS
    have contra : False := NotGG gg
    exact False.elim contra
  | inr hP =>
    have nf : NewFriends Alex := R2 hP
    exact R3 nf
</lean>

ANSWER: Yes
