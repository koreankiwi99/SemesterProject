<lean>
axiom obj : Type
axiom Project : obj

axiom NiceDisplay : obj → Prop
axiom JudgesImpressed : obj → Prop
axiom ResearchThorough : obj → Prop
axiom ExpertsSerious : obj → Prop
axiom ReceivesAward : obj → Prop

axiom T1 : ∀ x : obj, NiceDisplay x → JudgesImpressed x
axiom T2 : ∀ x : obj, ResearchThorough x → ExpertsSerious x
axiom T3 : NiceDisplay Project ∨ ResearchThorough Project
axiom T4 : ∀ x : obj, ExpertsSerious x → ReceivesAward x
axiom T5 : ¬ JudgesImpressed Project

theorem project_awarded : ReceivesAward Project := by
  have hNotNice : ¬ NiceDisplay Project := by
    intro hNice
    have : JudgesImpressed Project := T1 Project hNice
    exact T5 this
  have hThorough : ResearchThorough Project := by
    cases T3 with
    | inl hNice =>
        exact False.elim (hNotNice hNice)
    | inr hTh =>
        exact hTh
  have hSerious : ExpertsSerious Project := T2 Project hThorough
  exact T4 Project hSerious
</lean>

ANSWER: Yes
