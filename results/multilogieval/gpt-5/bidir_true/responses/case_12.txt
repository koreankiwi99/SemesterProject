<lean>
axiom obj : Type
axiom James : obj

axiom FollowPrecise : obj → Prop
axiom WatchShows : obj → Prop
axiom DishesWell : obj → Prop
axiom GainConfidence : obj → Prop

-- If someone followed recipes precisely, their dishes turn out well
axiom A1 : ∀ x : obj, FollowPrecise x → DishesWell x
-- If someone's dishes turn out well, they gain confidence in the kitchen
axiom A2 : ∀ x : obj, DishesWell x → GainConfidence x
-- Either they followed recipes precisely or they watched cooking shows (or both)
axiom A3 : ∀ x : obj, FollowPrecise x ∨ WatchShows x

-- Given: James does not gain confidence in the kitchen
axiom H : ¬ GainConfidence James

theorem james_watches : WatchShows James :=
by
  -- From following precisely implies well, and well implies confidence, we get:
  have hFC : FollowPrecise James → GainConfidence James :=
    fun hF => A2 James (A1 James hF)
  -- Contraposition with H gives not FollowPrecise James
  have hNotF : ¬ FollowPrecise James := fun hF => H (hFC hF)
  -- From the disjunction and not FollowPrecise, conclude WatchShows
  have hDisj : FollowPrecise James ∨ WatchShows James := A3 James
  exact Or.elim hDisj
    (fun hF => False.elim (hNotF hF))
    (fun hW => hW)
</lean>

ANSWER: Yes
