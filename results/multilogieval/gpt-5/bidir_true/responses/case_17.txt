<lean>
axiom obj : Type
axiom James : obj

axiom Practice : obj → Prop
axiom Prepared : obj → Prop
axiom Confident : obj → Prop
axiom JoinClub : obj → Prop

axiom R1 : ∀ x : obj, Practice x → Prepared x
axiom R2 : ∀ x : obj, Prepared x → Confident x
axiom R3 : ∀ x : obj, Practice x ∨ JoinClub x

theorem james_joins_from_not_confident : (¬ Confident James) → JoinClub James := by
  intro hnot
  have hpc : Practice James → Confident James := by
    intro hp
    have hprep : Prepared James := R1 James hp
    exact R2 James hprep
  have hnprac : ¬ Practice James := by
    intro hp
    exact hnot (hpc hp)
  have hdisj : Practice James ∨ JoinClub James := R3 James
  cases hdisj with
  | inl hp =>
      exact False.elim (hnprac hp)
  | inr hj =>
      exact hj
</lean>

ANSWER: Yes
