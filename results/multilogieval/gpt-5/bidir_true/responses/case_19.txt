<lean>
axiom Service : Prop
axiom Reputation : Prop
axiom Events : Prop
axiom Traffic : Prop
axiom Revenue : Prop

-- If they offer great customer service, reputation grows.
axiom H1 : Service → Reputation
-- If they host author events, more customers will visit (traffic increases).
axiom H2 : Events → Traffic
-- The owner either provides excellent service or hosts events.
axiom H3 : Service ∨ Events
-- If customer traffic increases, revenue will improve.
axiom H4 : Traffic → Revenue
-- Given: service quality decreased (not excellent service).
axiom H5 : ¬ Service

theorem revenue_improved : Revenue := by
  have ev : Events := Or.resolve_left H3 H5
  have tr : Traffic := H2 ev
  exact H4 tr
</lean>

ANSWER: Yes
