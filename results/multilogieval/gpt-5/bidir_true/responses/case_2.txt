<lean>
axiom Rain : Prop
axiom Hot : Prop
axiom Sprinklers : Prop
axiom WateredNaturally : Prop

-- If it rains, the grass and plants are watered naturally.
axiom R1 : Rain → WateredNaturally
-- If it's very hot, children play in sprinklers.
axiom R2 : Hot → Sprinklers
-- Either it rained today, or the children did not play in sprinklers.
axiom R3 : Rain ∨ ¬ Sprinklers
-- Given: The weather was very hot.
axiom GivenHot : Hot

theorem watered_today : WateredNaturally := by
  have s : Sprinklers := R2 GivenHot
  have r : Rain := by
    cases R3 with
    | inl hR => exact hR
    | inr hNotS =>
      have f : False := hNotS s
      exact False.elim f
  exact R1 r
</lean>

ANSWER: Yes
