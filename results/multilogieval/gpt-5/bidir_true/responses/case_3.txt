<lean>
axiom MoreA : Prop
axiom WinA : Prop
axiom FumbleB : Prop
axiom LoseB : Prop

axiom A_scores_more_implies_A_wins : MoreA → WinA
axiom B_fumbles_implies_B_loses : FumbleB → LoseB
axiom Either_A_scores_more_or_B_fumbles : MoreA ∨ FumbleB

theorem statement : (¬ LoseB) → MoreA := by
  intro hNotLoseB
  cases Either_A_scores_more_or_B_fumbles with
  | inl hMoreA => exact hMoreA
  | inr hFumbleB =>
      have hLoseB : LoseB := B_fumbles_implies_B_loses hFumbleB
      exact False.elim (hNotLoseB hLoseB)
</lean>

ANSWER: Yes
