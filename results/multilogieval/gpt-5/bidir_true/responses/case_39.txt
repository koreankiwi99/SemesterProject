axiom Alex : obj

axiom Monday : Prop
axiom Quiz : Prop

axiom DoWell : obj → Prop        -- does well on quizzes

-- If it was Monday, then there would be a quiz.
axiom A1 : Monday → Quiz
-- If someone is good at math, they do well on quizzes.
axiom A2 : ∀ x : obj, Good x → DoWell x
-- If they do well on the quiz, then they studied hard.
axiom A3 : ∀ x : obj, DoWell x → StudiedHard x
-- Either it is Monday, or they (Alex) did not study hard, or it is Monday and they did not study hard.
axiom A4 : Monday ∨ ¬StudiedHard Alex ∨ (Monday ∧ ¬StudiedHard Alex)

-- From no quiz, deduce not Monday.
lemma notMonday_of_notQuiz : ¬Quiz → ¬Monday := by
  intro hNotQ
  intro hMon
  have hQ : Quiz := A1 hMon
  exact hNotQ hQ

-- Using the disjunction and not Monday, deduce Alex did not study hard.

  intro hNotMon
  have h := A4
  cases h with
  | inl hMon =>
      exact False.elim (hNotMon hMon)
  | inr hRest =>
      cases hRest with
      | inl hNotSH => exact hNotSH
      | inr hMonAndNotSH => exact hMonAndNotSH.right
