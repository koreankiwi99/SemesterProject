<lean>
axiom Marketing : Prop
axiom MoreCustomers : Prop
axiom SalesIncrease : Prop
axiom ProductionCostsIncreased : Prop

-- If they invested in marketing, they would gain more customers.
axiom R1 : Marketing → MoreCustomers
-- With more customers, sales numbers would go up.
axiom R2 : MoreCustomers → SalesIncrease
-- Either they focused budget on marketing, or production costs increased, or both.
axiom Or1 : Marketing ∨ ProductionCostsIncreased

theorem costs_increased_if_no_sales : (¬ SalesIncrease) → ProductionCostsIncreased := by
  intro hNotSales
  have hNotMarketing : ¬ Marketing := by
    intro hM
    apply hNotSales
    exact R2 (R1 hM)
  cases Or1 with
  | inl hM => exact False.elim (hNotMarketing hM)
  | inr hC => exact hC
</lean>

ANSWER: Yes
