<lean>
axiom Person : Type
axiom James : Person

axiom Started : Person → Prop
axiom Motivated : Person → Prop
axiom FollowRoutine : Person → Prop
axiom Frustrated : Person → Prop
axiom Indulged : Person → Prop
axiom GymClosed : Prop

axiom A1 : ∀ x : Person, Started x → Motivated x
axiom A2 : ∀ x : Person, GymClosed → ¬ FollowRoutine x
axiom A3 : Started James ∨ GymClosed
axiom A4 : ∀ x : Person, ¬ FollowRoutine x → Frustrated x
axiom A5 : ∀ x : Person, Frustrated x → Indulged x
axiom A6 : ¬ Motivated James

theorem james_indulged : Indulged James := by
  have hNotStarted : ¬ Started James := by
    intro hS
    have hM : Motivated James := A1 James hS
    exact A6 hM
  have hClosed : GymClosed := Or.resolve_left A3 hNotStarted
  have hNoFollow : ¬ FollowRoutine James := A2 James hClosed
  have hFrustrated : Frustrated James := A4 James hNoFollow
  exact A5 James hFrustrated
</lean>

ANSWER: Yes
