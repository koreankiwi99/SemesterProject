<lean>
axiom WaterDaily : Prop
axiom GrowWell : Prop
axiom Drought : Prop
axiom Restrictions : Prop
axiom LackWater : Prop
axiom WiltDie : Prop

-- If someone waters their plants daily, the plants will grow well.
axiom R1 : WaterDaily → GrowWell
-- If there is a drought, then water restrictions will be imposed.
axiom R2 : Drought → Restrictions
-- Either they watered the plants daily, or there was a drought (or both).
axiom R3 : WaterDaily ∨ Drought
-- If water restrictions were imposed, then the plants would not get enough water (lack water).
axiom R4 : Restrictions → LackWater
-- If the plants lacked water, they would wilt and die.
axiom R5 : LackWater → WiltDie

-- From "Priya's plants did not grow well", conclude "they wilted and died".
theorem wilt_from_not_grow : (¬ GrowWell) → WiltDie :=
  fun hNotGrow =>
    Or.elim R3
      (fun hWaterDaily => False.elim (hNotGrow (R1 hWaterDaily)))
      (fun hDrought => R5 (R4 (R2 hDrought)))
</lean>

ANSWER: Yes
