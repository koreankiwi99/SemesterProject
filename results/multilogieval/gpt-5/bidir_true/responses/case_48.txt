<lean>
axiom Flip : Prop
axiom Gasp : Prop
axiom High : Prop
axiom Clowns : Prop
axiom Laugh : Prop

axiom A1 : Flip → Gasp
axiom A2 : Gasp → High
axiom A3 : Flip ∨ Clowns
axiom A4 : Clowns → Laugh
axiom A5 : ¬ High

theorem youngest_kids_laughed : Laugh := by
  have notFlip : ¬ Flip := by
    intro hf
    have hg : Gasp := A1 hf
    have hh : High := A2 hg
    exact A5 hh
  have hclowns : Clowns := by
    cases A3 with
    | inl hf =>
      have : False := notFlip hf
      exact False.elim this
    | inr hc =>
      exact hc
  exact A4 hclowns
</lean>

ANSWER: Yes
