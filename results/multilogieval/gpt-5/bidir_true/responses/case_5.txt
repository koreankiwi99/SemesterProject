<lean>
axiom obj : Type
axiom Mary : obj

axiom LearnedBotany : obj → Prop
axiom UnderstoodNeeds : obj → Prop
axiom ThrivingPlants : obj → Prop
axiom SoughtAdvice : obj → Prop

-- If someone learned about botany, they understand plant needs
axiom R1 : ∀ x : obj, LearnedBotany x → UnderstoodNeeds x
-- If someone understands plant needs, their plants thrive
axiom R2 : ∀ x : obj, UnderstoodNeeds x → ThrivingPlants x
-- Either Mary learned about botany or she sought advice (or both)
axiom D1 : LearnedBotany Mary ∨ SoughtAdvice Mary

theorem mary_sought_advice (hnot : ¬ ThrivingPlants Mary) : SoughtAdvice Mary := by
  have hLB_to_TP : LearnedBotany Mary → ThrivingPlants Mary := by
    intro hlb
    have hUN : UnderstoodNeeds Mary := R1 Mary hlb
    exact R2 Mary hUN
  have hnotLB : ¬ LearnedBotany Mary := by
    intro hlb
    exact hnot (hLB_to_TP hlb)
  cases D1 with
  | inl hlb =>
      exact False.elim (hnotLB hlb)
  | inr hsa =>
      exact hsa
</lean>

ANSWER: Yes
