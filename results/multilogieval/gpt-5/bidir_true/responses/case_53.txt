<lean>
axiom obj : Type
axiom Mary : obj

axiom PracticedOften : obj → Prop
axiom TechniqueImproved : obj → Prop
axiom ArtworkHighQuality : obj → Prop
axiom TookArtClass : obj → Prop
axiom LearnedNewMethods : obj → Prop

-- If someone practiced often, their technique improved.
axiom R1 : ∀ x : obj, PracticedOften x → TechniqueImproved x
-- If someone's technique improved, their artwork stood out as high quality.
axiom R2 : ∀ x : obj, TechniqueImproved x → ArtworkHighQuality x
-- For Mary: either she practiced often or she took an art class (or both).
axiom D1 : PracticedOften Mary ∨ TookArtClass Mary
-- If someone takes an art class, they learn new methods and mediums.
axiom R3 : ∀ x : obj, TookArtClass x → LearnedNewMethods x

-- Given statement: Mary's artwork did not stand out as high quality.
axiom H_notHQ : ¬ ArtworkHighQuality Mary

theorem mary_learned_new_methods : LearnedNewMethods Mary := by
  -- From not high quality, infer not improved (modus tollens on R2)
  have hNotImproved : ¬ TechniqueImproved Mary := by
    intro hImp
    exact H_notHQ (R2 Mary hImp)
  -- From not improved, infer not practiced often (modus tollens on R1)
  have hNotPracticed : ¬ PracticedOften Mary := by
    intro hP
    exact hNotImproved (R1 Mary hP)
  -- Using the disjunction and not practiced, conclude she took an art class
  have hTookClass : TookArtClass Mary := by
    cases D1 with
    | inl hP => exact False.elim (hNotPracticed hP)
    | inr hT => exact hT
  -- From taking an art class, conclude she learned new methods and mediums
  exact R3 Mary hTookClass
</lean>

ANSWER: Yes
