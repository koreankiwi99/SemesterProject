<lean>
axiom Good : Prop
axiom Many : Prop
axiom Low : Prop
axiom BigStar : Prop
axiom Sequel : Prop

-- If the movie gets good reviews, then a lot of people will go see it.
axiom R1 : Good → Many

-- If a big star is in the movie, then it got good reviews.
axiom R4 : BigStar → Good

-- Either there is a big star in it, or it is a sequel, or both.
axiom R5 : BigStar ∨ Sequel ∨ (BigStar ∧ Sequel)

-- Either the movie got good reviews, or attendance was low, or both.
axiom R3 : Good ∨ Low ∨ (Good ∧ Low)

-- "A lot of people will go see it" implies attendance is not low.
axiom ManyNotLow : Many → ¬ Low

theorem low_implies_sequel : Low → Sequel := by
  intro hLow
  -- From low attendance, there cannot be a big star (since BigStar → Good → Many → ¬Low)
  have hNotStar : ¬ BigStar := by
    intro hStar
    have hGood : Good := R4 hStar
    have hMany : Many := R1 hGood
    have hNotLow : ¬ Low := ManyNotLow hMany
    exact hNotLow hLow
  -- Reduce the triple disjunction to BigStar ∨ Sequel
  have hOr : BigStar ∨ Sequel := by
    cases R5 with
    | inl hStar =>
        exact Or.inl hStar
    | inr hRest =>
        cases hRest with
        | inl hSeq =>
            exact Or.inr hSeq
        | inr hBoth =>
            exact Or.inl hBoth.left
  -- Resolve the disjunction with ¬BigStar to get Sequel
  exact Or.resolve_left hOr hNotStar
</lean>

ANSWER: Yes
