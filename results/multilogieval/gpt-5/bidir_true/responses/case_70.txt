axiom AdequateSunlight : obj → Prop
axiom AdequateWatering : obj → Prop
axiom GrowsLargePlentiful : obj → Prop

-- Gardeners' knowledge rule:
axiom Rule :
  ∀ x : obj, PlantedProperly x → (AdequateSunlight x ∧ AdequateWatering x → GrowsLargePlentiful x)

-- Facts about Farmer Joe:
axiom SunJoe : AdequateSunlight Joe
axiom WaterJoe : AdequateWatering Joe
axiom NotGrowJoe : ¬ GrowsLargePlentiful Joe

-- From the rule and facts, we can derive that Joe did NOT plant properly.
theorem not_planted_properly_Joe : ¬ PlantedProperly Joe := by
  intro hP
  have hImp : AdequateSunlight Joe ∧ AdequateWatering Joe → GrowsLargePlentiful Joe := Rule Joe hP
  have hGrow : GrowsLargePlentiful Joe := hImp ⟨SunJoe, WaterJoe⟩
  exact NotGrowJoe hGrow
