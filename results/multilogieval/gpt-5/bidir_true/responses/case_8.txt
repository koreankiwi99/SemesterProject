<lean>
axiom obj : Type
axiom Chef : obj
axiom Fresh : obj → Prop
axiom Delicious : obj → Prop
axiom Complicated : obj → Prop
axiom MoreTime : obj → Prop
axiom R1 : ∀ x : obj, Fresh x → Delicious x
axiom R2 : ∀ x : obj, Complicated x → MoreTime x
axiom T1 : Fresh Chef

theorem complicated_recipe_implies_delicious : Complicated Chef → Delicious Chef := by
  have hdel : Delicious Chef := R1 Chef T1
  intro _hcomp
  exact hdel
</lean>

ANSWER: Yes
