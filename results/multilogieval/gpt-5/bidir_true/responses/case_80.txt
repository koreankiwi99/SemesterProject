<lean>
axiom Rain : Prop
axiom Game : Prop
axiom PoolOpen : Prop
axiom KidsSwim : Prop
axiom EnjoyPlaying : Prop
axiom SmilesEnd : Prop

axiom A1 : ¬ Rain → Game
axiom A2 : PoolOpen → KidsSwim
axiom A3 : ¬ Rain ∨ PoolOpen
axiom A4 : Game → EnjoyPlaying
axiom A5 : EnjoyPlaying → SmilesEnd

theorem smiles_if_no_swim : (¬ KidsSwim) → SmilesEnd := by
  intro hNoSwim
  cases A3 with
  | inl hNoRain =>
      have hGame := A1 hNoRain
      have hEnjoy := A4 hGame
      exact A5 hEnjoy
  | inr hPoolOpen =>
      have hSwim := A2 hPoolOpen
      have hFalse : False := hNoSwim hSwim
      exact False.elim hFalse
</lean>

ANSWER: Yes
