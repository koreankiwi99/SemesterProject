<lean>
axiom Eggs : Prop
axiom Oven350 : Prop
axiom Proper : Prop
axiom Fluffy : Prop
axiom Delicious : Prop
axiom GuestsHappy : Prop

axiom R1 : Oven350 → Proper
axiom R2 : Eggs → Fluffy
axiom R3 : Oven350 ∨ ¬ Eggs
axiom R4 : Proper → Delicious
axiom R5 : Delicious → GuestsHappy

theorem eggs_implies_happy : Eggs → GuestsHappy := by
  intro e
  have h350 : Oven350 := by
    cases R3 with
    | inl h => exact h
    | inr hne =>
        have hFalse : False := hne e
        exact False.elim hFalse
  have hprop : Proper := R1 h350
  have hdel : Delicious := R4 hprop
  exact R5 hdel
</lean>

ANSWER: Yes
