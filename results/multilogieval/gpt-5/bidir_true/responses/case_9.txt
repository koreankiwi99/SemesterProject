<lean>
axiom Birthday : Prop
axiom Cake : Prop
axiom IceCream : Prop
axiom SomeoneBrought : Prop
axiom EnoughFood : Prop

axiom R1 : SomeoneBrought → EnoughFood
axiom R2 : Birthday → (Cake ∧ IceCream)
axiom R3 : SomeoneBrought ∨ ¬ Cake
axiom T_Birthday : Birthday

theorem enough_food : EnoughFood := by
  have hCake : Cake := (And.left (R2 T_Birthday))
  have hSB : SomeoneBrought := by
    cases R3 with
    | inl h => exact h
    | inr hnc =>
      have : False := hnc hCake
      exact False.elim this
  exact R1 hSB
</lean>

ANSWER: Yes
