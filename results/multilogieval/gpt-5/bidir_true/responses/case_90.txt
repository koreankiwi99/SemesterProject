<lean>
axiom Winning : Prop
axiom Rivalry : Prop
axiom Loud : Prop
axiom Invested : Prop
axiom Electric : Prop
axiom Unforgettable : Prop

axiom R1 : Winning → Loud
axiom R2 : Rivalry → Invested
axiom R3 : Loud → Electric
axiom R4 : Electric → Unforgettable
axiom T : Winning ∨ Rivalry

theorem electric_if_not_invested : (¬ Invested) → Electric := by
  intro hni
  exact Or.elim T
    (fun hW => R3 (R1 hW))
    (fun hR => False.elim (hni (R2 hR)))
</lean>

ANSWER: Yes
