<lean>
axiom Sub : Prop
axiom Pizza : Prop
axiom Hyper : Prop
axiom GoofOff : Prop
axiom LearnMuch : Prop
axiom Homework : Prop

axiom R1 : Sub → GoofOff
axiom R2 : Pizza → Hyper
axiom R3 : Sub ∨ Pizza
axiom R4 : GoofOff → ¬ LearnMuch
axiom R5 : ¬ LearnMuch → Homework

theorem no_hyper_homework : ¬ Hyper → Homework := by
  intro notH
  have notP : ¬ Pizza := by
    intro p
    have h : Hyper := R2 p
    exact notH h
  have s : Sub := by
    cases R3 with
    | inl s => exact s
    | inr p =>
      have f : False := notP p
      exact False.elim f
  have g : GoofOff := R1 s
  have nl : ¬ LearnMuch := R4 g
  exact R5 nl
</lean>

ANSWER: Yes
