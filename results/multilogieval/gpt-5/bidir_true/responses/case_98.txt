<lean>
axiom Practice : Prop
axiom Prepared : Prop
axiom PlayWell : Prop
axiom Win : Prop
axiom VeryHot : Prop
axiom Postponed : Prop

-- If the team practices hard, they will be prepared for the game.
axiom A1 : Practice → Prepared
-- If it is very hot outside, the game is postponed.
axiom A2 : VeryHot → Postponed
-- Either the team practiced hard, or the game was not postponed.
axiom A3 : Practice ∨ ¬ Postponed
-- If a team is prepared, they will play well.
axiom A4 : Prepared → PlayWell
-- When a team plays well, they will win.
axiom A5 : PlayWell → Win

theorem hot_implies_win : VeryHot → Win :=
by
  intro hHot
  have hPost : Postponed := A2 hHot
  have hPractice : Practice :=
    Or.elim A3
      (fun hp => hp)
      (fun hNotPost => False.elim (hNotPost hPost))
  have hPrepared : Prepared := A1 hPractice
  have hPlayWell : PlayWell := A4 hPrepared
  exact A5 hPlayWell
</lean>

ANSWER: Yes
