<lean>
axiom Watered : Prop
axiom Bloom : Prop
axiom HeavyRain : Prop
axiom Saturated : Prop
axiom Fragrance : Prop
axiom AttractButterflies : Prop

axiom R1 : Watered → Bloom
axiom R2 : HeavyRain → Saturated
axiom R3 : Watered ∨ ¬Saturated
axiom R4 : Bloom → Fragrance
axiom R5 : Fragrance → AttractButterflies

theorem heavy_rain_implies_butterflies : HeavyRain → AttractButterflies := by
  intro hR
  have hS : Saturated := R2 hR
  have hW : Watered :=
    Or.elim R3
      (fun hW => hW)
      (fun hNotS => False.elim (hNotS hS))
  have hB : Bloom := R1 hW
  have hF : Fragrance := R4 hB
  exact R5 hF
</lean>

ANSWER: Yes
