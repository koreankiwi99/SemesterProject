<lean>
axiom NiceDisplay : Prop
axiom JudgesImpressed : Prop
axiom ThoroughResearch : Prop
axiom ExpertsTakeSeriously : Prop
axiom Award : Prop

-- Premises
axiom P1 : NiceDisplay → JudgesImpressed
axiom P2 : ThoroughResearch → ExpertsTakeSeriously
axiom P3 : NiceDisplay ∨ ThoroughResearch
axiom P4 : ExpertsTakeSeriously → Award

-- Given: judges were not impressed with the display
axiom G1 : ¬JudgesImpressed

-- From not impressed and (NiceDisplay → JudgesImpressed), we get ¬NiceDisplay
theorem not_nice : ¬NiceDisplay := by
  intro h
  apply G1
  exact P1 h

-- From NiceDisplay ∨ ThoroughResearch and ¬NiceDisplay, we get ThoroughResearch
theorem have_TR : ThoroughResearch := by
  cases P3 with
  | inl hNice =>
      have f : False := not_nice hNice
      exact False.elim f
  | inr hTR =>
      exact hTR

-- Therefore, the project receives an award
theorem project_awarded : Award := by
  have tr : ThoroughResearch := have_TR
  have s : ExpertsTakeSeriously := P2 tr
  exact P4 s
