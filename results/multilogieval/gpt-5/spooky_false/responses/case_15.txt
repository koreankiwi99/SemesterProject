<lean>
axiom obj : Type
axiom Student : obj

axiom Study : obj → Prop
axiom Excel : obj → Prop
axiom Participate : obj → Prop
axiom Leader : obj → Prop
axiom Prefect : obj → Prop

-- If someone studies diligently, they will excel in academics.
axiom R1 : ∀ x : obj, Study x → Excel x
-- If someone participates actively, they will gain leadership skills.
axiom R2 : ∀ x : obj, Participate x → Leader x
-- The student either studies hard or participates actively.
axiom disj : Study Student ∨ Participate Student
-- If someone is a leader, they will be selected as class prefect.
axiom R3 : ∀ x : obj, Leader x → Prefect x
-- Given: the student did not excel academically.
axiom not_excel : ¬ Excel Student

-- From the premises and the given, the student is indeed a class prefect.
theorem student_prefect : Prefect Student := by
  -- First, the student did not study (otherwise they'd excel)
  have h_not_study : ¬ Study Student := by
    intro hS
    exact not_excel (R1 Student hS)
  -- Hence, from the disjunction, the student participated
  have h_part : Participate Student := Or.resolve_left disj h_not_study
  -- Participation implies leadership
  have h_leader : Leader Student := R2 Student h_part
  -- Leadership implies being class prefect
  exact R3 Student h_leader
</lean>
