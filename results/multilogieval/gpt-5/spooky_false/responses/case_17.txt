<lean>
axiom Person : Type
axiom James : Person

axiom Practiced : Person → Prop
axiom Prepared : Person → Prop
axiom Confident : Person → Prop
axiom JoinedClub : Person → Prop

-- If they practiced thoroughly, they feel more prepared.
axiom H1 : ∀ x : Person, Practiced x → Prepared x
-- If they felt prepared, they could speak confidently.
axiom H2 : ∀ x : Person, Prepared x → Confident x
-- Either they practiced thoroughly, or they joined a public speaking club, or both.
axiom H3 : ∀ x : Person, Practiced x ∨ JoinedClub x

-- Given: James cannot speak confidently in public.
axiom James_not_confident : ¬Confident James

-- From the premises, we can show James joined a public speaking club.
theorem james_joined_club : JoinedClub James := by
  have hPC : Practiced James → Confident James := by
    intro hP
    exact H2 James (H1 James hP)
  have hNotP : ¬Practiced James := by
    intro hP
    exact James_not_confident (hPC hP)
  have hOr : Practiced James ∨ JoinedClub James := H3 James
  cases hOr with
  | inl hP =>
      exact False.elim (hNotP hP)
  | inr hC =>
      exact hC
