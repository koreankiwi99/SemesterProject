<lean>
axiom obj : Type
axiom Sam : obj

axiom Impress : obj → Prop
axiom Promotion : obj → Prop
axiom Extra : obj → Prop
axiom Experience : obj → Prop
axiom Lead : obj → Prop

-- If someone impresses the manager with excellent work, they get a promotion.
axiom R1 : ∀ x : obj, Impress x → Promotion x
-- If someone takes on extra responsibilities, they gain valuable experience.
axiom R2 : ∀ x : obj, Extra x → Experience x
-- This employee (Sam) either impressed the manager or took on extra work.
axiom R3 : Impress Sam ∨ Extra Sam
-- If someone gains experience, they will be assigned to lead a new team.
axiom R4 : ∀ x : obj, Experience x → Lead x

-- Given: Sam did not get a promotion.
axiom NP : ¬Promotion Sam

-- From the premises we can derive that Sam is assigned to lead a new team.
theorem sam_assigned : Lead Sam := by
  have hNotImpress : ¬Impress Sam := by
    intro hI
    exact NP (R1 Sam hI)
  have hExtra : Extra Sam := by
    cases R3 with
    | inl hI =>
      exact False.elim (NP (R1 Sam hI))
    | inr hE =>
      exact hE
  have hExp : Experience Sam := R2 Sam hExtra
  exact R4 Sam hExp
</lean>
