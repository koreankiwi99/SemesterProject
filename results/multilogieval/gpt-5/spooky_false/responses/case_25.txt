axiom Emma : obj

axiom Perceptive : obj → Prop

axiom EffectivePresentation : obj → Prop
axiom ConvertsSales : obj → Prop

-- Building rapport leads to building trust.
axiom P2 : ∀ x : obj, BuildsRapport x → BuildsTrust x
-- Effective presentations convert more sales.
axiom P3 : ∀ x : obj, EffectivePresentation x → ConvertsSales x
-- Not every salesperson builds trust or converts sales.
axiom P4 : ∃ x : obj, Salesperson x ∧ ¬(BuildsTrust x ∨ ConvertsSales x)

-- Given: Emma gives effective sales presentations.
axiom Emma_effective : EffectivePresentation Emma

-- A consequence we can derive: Emma converts sales.
theorem Emma_converts : ConvertsSales Emma := by
  apply P3 Emma
  exact Emma_effective
