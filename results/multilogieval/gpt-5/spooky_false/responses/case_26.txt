<lean>
axiom Disappointed : Prop
axiom Demoralized : Prop
axiom LostImportantGame : Prop
axiom LostBigMatch : Prop
axiom LostSignificantGame : Prop

-- Premises from the textual context
axiom P1 : LostImportantGame → Disappointed
axiom P2 : LostBigMatch → Demoralized

-- Bridging synonym: a significant game is either important or big
axiom Sig_to_imp_or_big : LostSignificantGame → (LostImportantGame ∨ LostBigMatch)

-- From the statement to evaluate
axiom NotDisappointed : ¬Disappointed
axiom NotDemoralized : ¬Demoralized

theorem no_significant_loss : ¬LostSignificantGame := by
  intro hsig
  have h := Sig_to_imp_or_big hsig
  cases h with
  | inl hImp =>
      exact NotDisappointed (P1 hImp)
  | inr hBig =>
      exact NotDemoralized (P2 hBig)
</lean>

ANSWER: No
