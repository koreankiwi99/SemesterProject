<lean>
axiom obj : Type
axiom Jane : obj
axiom Injured : obj → Prop
axiom SawDoctor : obj → Prop
axiom RestedInjury : obj → Prop

-- Premises from the context
axiom P1 : ∀ x : obj, Injured x → SawDoctor x
axiom P2 : ∀ x : obj, Injured x → RestedInjury x

-- Given statement to evaluate (disjunction about Jane)
axiom D : ¬SawDoctor Jane ∨ ¬RestedInjury Jane

theorem jane_not_injured : ¬Injured Jane := by
  intro hInj
  have hDoc : SawDoctor Jane := P1 Jane hInj
  have hRest : RestedInjury Jane := P2 Jane hInj
  cases D with
  | inl hNotDoc =>
      exact hNotDoc hDoc
  | inr hNotRest =>
      exact hNotRest hRest
</lean>

ANSWER: No
