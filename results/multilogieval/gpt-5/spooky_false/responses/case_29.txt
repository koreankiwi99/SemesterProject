<lean>
axiom obj : Type

axiom Ty : obj

axiom SuspectsID : obj → Prop
axiom FraudSuspected : obj → Prop
axiom CheckedCredit : obj → Prop
axiom ContactBank : obj → Prop

-- If someone suspects identity theft, they should check their credit report.
axiom A1 : ∀ x : obj, SuspectsID x → CheckedCredit x
-- If fraud is suspected, they should contact their bank.
axiom A2 : ∀ x : obj, FraudSuspected x → ContactBank x
-- Identity theft is a kind of fraud: suspecting identity theft implies fraud is suspected.
axiom A3 : ∀ x : obj, SuspectsID x → FraudSuspected x

-- Either Ty did not verify his credit report or did not get in touch with his bank.
axiom Prem : ¬CheckedCredit Ty ∨ ¬ContactBank Ty

theorem ty_not_suspect : ¬SuspectsID Ty := by
  intro s
  have cc : CheckedCredit Ty := A1 Ty s
  have fs : FraudSuspected Ty := A3 Ty s
  have cb : ContactBank Ty := A2 Ty fs
  cases Prem with
  | inl ncc => exact ncc cc
  | inr ncb => exact ncb cb
</lean>

ANSWER: No
