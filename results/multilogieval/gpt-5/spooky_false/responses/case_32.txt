<lean>
axiom obj : Type
axiom Jane : obj

axiom Exercise : obj → Prop
axiom EatWell : obj → Prop
axiom Fit : obj → Prop
axiom Energetic : obj → Prop
axiom ValuesWellness : obj → Prop

-- Contextual knowledge
axiom Ex_implies_Fit : ∀ x : obj, Exercise x → Fit x
axiom EatWell_implies_Energy : ∀ x : obj, EatWell x → Energetic x
axiom Not_all_active_or_energetic : ¬ (∀ x : obj, Exercise x ∨ Energetic x)
axiom Wellness_implies_Exercise : ∀ x : obj, ValuesWellness x → Exercise x

-- Fact about Jane
axiom Jane_eats_well : EatWell Jane

open Classical

-- If someone exercises or eats well, then they are active or energetic
theorem or_monotone : ∀ x : obj, Exercise x ∨ EatWell x → Exercise x ∨ Energetic x := by
  intro x
  intro h
  cases h with
  | inl hx =>
      exact Or.inl hx
  | inr hw =>
      exact Or.inr (EatWell_implies_Energy x hw)

-- From (∀x, Exercise x ∨ EatWell x) infer (∀x, Exercise x ∨ Energetic x)
theorem all_or_imp : (∀ x : obj, Exercise x ∨ EatWell x) → (∀ x : obj, Exercise x ∨ Energetic x) := by
  intro h
  intro x
  have hx := h x
  exact or_monotone x hx

-- Therefore, not everyone exercises or eats well
theorem not_all_or_eatwell : ¬(∀ x : obj, Exercise x ∨ EatWell x) := by
  intro h
  apply Not_all_active_or_energetic
  exact all_or_imp h

-- Hence, there exists someone who does neither
theorem exists_counterexample : ∃ x : obj, ¬(Exercise x ∨ EatWell x) := by
  by_contra h
  have all_or : ∀ x : obj, Exercise x ∨ EatWell x := by
    intro x
    by_contra hx
    apply h
    exact Exists.intro x hx
  exact not_all_or_eatwell all_or

-- Stronger mayor conclusion: someone neither exercises nor eats well
theorem mayor_conclusion : ∃ x : obj, ¬Exercise x ∧ ¬EatWell x := by
  have h := exists_counterexample
  cases h with
  | intro x hx =>
      have hx1 : ¬Exercise x := by
        intro ex
        apply hx
        exact Or.inl ex
      have hx2 : ¬EatWell x := by
        intro ew
        apply hx
        exact Or.inr ew
      exact Exists.intro x (And.intro hx1 hx2)

-- From Jane eating well, she is energetic
theorem jane_energetic : Energetic Jane := by
  apply EatWell_implies_Energy
  exact Jane_eats_well
</lean>
