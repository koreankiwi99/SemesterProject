<lean>
axiom person : Type
axiom Lucy : person

axiom Hacked : person → Prop
axiom ResetPwd : person → Prop
axiom InformedBank : person → Prop

-- If an account is hacked, they reset their password
axiom rule_reset : ∀ x : person, Hacked x → ResetPwd x
-- If an account is hacked (unauthorized access), they inform their bank
axiom rule_inform : ∀ x : person, Hacked x → InformedBank x

-- Either Lucy did not change her password or she did not inform her bank
axiom lucy_disj : ¬ResetPwd Lucy ∨ ¬InformedBank Lucy

theorem lucy_not_hacked : ¬Hacked Lucy := by
  intro h
  have hr : ResetPwd Lucy := rule_reset Lucy h
  have hi : InformedBank Lucy := rule_inform Lucy h
  cases lucy_disj with
  | inl nReset => exact nReset hr
  | inr nInform => exact nInform hi
</lean>

ANSWER: No
