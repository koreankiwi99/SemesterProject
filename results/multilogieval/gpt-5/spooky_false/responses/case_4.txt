<lean>
axiom Marketing : Prop
axiom CustomersUp : Prop
axiom SalesUp : Prop
axiom ProdCostsInc : Prop

-- If they invested in marketing, they would gain more customers.
axiom R1 : Marketing → CustomersUp
-- With more customers, sales numbers would go up.
axiom R2 : CustomersUp → SalesUp
-- Either they focused budget on marketing, or production costs increased (or both).
axiom R3 : Marketing ∨ ProdCostsInc

theorem marketing_implies_sales : Marketing → SalesUp := by
  intro m
  have c : CustomersUp := R1 m
  exact R2 c

-- From "sales did not increase," we can conclude "not marketing" (contrapositive).
theorem not_marketing_of_not_sales : ¬SalesUp → ¬Marketing := by
  intro ns
  intro m
  apply ns
  exact marketing_implies_sales m

-- From "sales did not increase" and the disjunction (Marketing ∨ ProdCostsInc), deduce ProdCostsInc.
theorem production_costs_increased_of_no_sales (ns : ¬SalesUp) : ProdCostsInc := by
  cases R3 with
  | inl m =>
    have s : SalesUp := marketing_implies_sales m
    exact False.elim (ns s)
  | inr p => exact p
</lean>
