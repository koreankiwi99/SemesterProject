axiom Motivated : obj → Prop

axiom GymClosed : Prop

-- If someone starts a new workout routine, they will be motivated.
axiom R1 : ∀ x : obj, Started x → Motivated x
-- If the gym closes, then they can't follow their routine.
axiom R2 : ∀ x : obj, GymClosed → CantFollow x
-- If they couldn't follow their routine, they would feel frustrated.
axiom R3 : ∀ x : obj, CantFollow x → Frustrated x
-- If they felt frustrated, they might indulge in unhealthy snacks.
axiom R4 : ∀ x : obj, Frustrated x → MightSnack x

-- Either James started a new workout, or the gym was closed.
axiom disj : Started James ∨ GymClosed
-- James was not motivated to workout.
axiom not_motivated : ¬Motivated James

theorem not_started : ¬Started James := by
  intro hS
  have hM : Motivated James := R1 James hS
  exact not_motivated hM

theorem gym_closed : GymClosed := by
  exact Or.resolve_left disj not_started

theorem cant_follow_james : CantFollow James := by
  exact R2 James gym_closed

theorem frustrated_james : Frustrated James := by
  exact R3 James cant_follow_james

theorem might_snack_james : MightSnack James := by
  exact R4 James frustrated_james
