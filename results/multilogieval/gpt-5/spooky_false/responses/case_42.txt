<lean>
axiom Sunny : Prop
axiom WantPlayOutside : Prop
axiom PlayOutside : Prop
axiom SleepWellAtNight : Prop
axiom FluSeasonStarted : Prop
axiom ManyKidsSick : Prop

-- Premises
axiom P1 : Sunny → WantPlayOutside
axiom P2 : FluSeasonStarted → ManyKidsSick
axiom P3 : Sunny ∨ ¬ManyKidsSick ∨ (Sunny ∧ ¬ManyKidsSick)
axiom P4 : PlayOutside → SleepWellAtNight

-- From flu season starting, many kids get sick
theorem many_sick_from_flu (fs : FluSeasonStarted) : ManyKidsSick := P2 fs

-- Using the disjunction P3 and the fact that ¬ManyKidsSick is false when flu starts,
-- we can conclude it is Sunny.
theorem sunny_from_flu (fs : FluSeasonStarted) : Sunny := by
  have ms : ManyKidsSick := P2 fs
  cases P3 with
  | inl hSunny =>
    exact hSunny
  | inr h =>
    cases h with
    | inl hNotMany =>
      exact False.elim (hNotMany ms)
    | inr hSunnyAndNot =>
      exact hSunnyAndNot.left

-- From Sunny, children will want to play outside
theorem want_play_from_flu (fs : FluSeasonStarted) : WantPlayOutside :=
  P1 (sunny_from_flu fs)
</lean>
