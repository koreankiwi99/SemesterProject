axiom Read_imp_Ideas : ∀ x : person, ReadBlogs x → NewIdeas x
axiom not_R_Matt : ¬RedidLR Matt

theorem Matt_gets_new_ideas : NewIdeas Matt :=
by
  have not_skilled : ¬Skilled Matt :=
    fun hs => not_R_Matt (S_imp_R Matt hs)
  have not_watched : ¬Watched Matt :=
    fun hw => not_skilled (W_imp_S Matt hw)
  have read_matt : ReadBlogs Matt :=
    Or.elim disj
      (fun hw => False.elim (not_watched hw))
      (fun hr => hr)
  exact Read_imp_Ideas Matt read_matt
