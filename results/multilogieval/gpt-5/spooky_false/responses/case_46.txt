<lean>
axiom obj : Type
axiom Casey : obj

axiom Volunteers : obj → Prop
axiom Helps : obj → Prop
axiom Adopts : obj → Prop
axiom Home : obj → Prop
axiom Fulfillment : obj → Prop

-- If someone volunteers at a local shelter, they will help the community.
axiom R1 : ∀ x : obj, Volunteers x → Helps x
-- If someone adopts a pet from the shelter, then they provide a home to a pet in need.
axiom R2 : ∀ x : obj, Adopts x → Home x
-- Either someone volunteers at the shelter, or they do not provide a home to a pet in need,
-- or someone volunteered at the shelter, and they did not provide a home to a pet in need.
axiom R3 : ∀ x : obj, Volunteers x ∨ ¬Home x ∨ (Volunteers x ∧ ¬Home x)
-- If someone helps the community, then they feel a sense of fulfillment.
axiom R4 : ∀ x : obj, Helps x → Fulfillment x

theorem volunteers_of_home : ∀ x : obj, Home x → Volunteers x :=
by
  intro x hx
  have disj := R3 x
  cases disj with
  | inl v => exact v
  | inr h =>
    cases h with
    | inl nh => exact False.elim (nh hx)
    | inr both => exact both.left

-- From "Casey adopts a pet from the shelter", it entails that she feels a sense of fulfillment.
theorem casey_fulfillment_from_adopts : Adopts Casey → Fulfillment Casey :=
by
  intro hA
  have hH : Home Casey := R2 Casey hA
  have hV : Volunteers Casey := volunteers_of_home Casey hH
  have hHelps : Helps Casey := R1 Casey hV
  exact R4 Casey hHelps
