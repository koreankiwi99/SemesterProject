axiom LearnsNewSkills : obj → Prop

-- If the package arrives on time, people will be able to start a new project.
axiom P1 : PackageOnTime → ∀ x : obj, AbleToStartProject x
-- If the special order is in stock, they can pick up the materials they need.
axiom P2 : SpecialOrderInStock → ∀ x : obj, CanPickUp x
-- Either the package will arrive on time or the special order will be ready (maybe both).
axiom P3 : PackageOnTime ∨ SpecialOrderInStock
-- If someone starts his new project, they will learn some new skills.
axiom P4 : ∀ x : obj, StartsNewProject x → LearnsNewSkills x

-- Given: Max was unable to pick up the materials he needed.
axiom Max_unable_pickup : ¬ CanPickUp Max

-- Consequences we can derive
theorem not_special_order : ¬ SpecialOrderInStock := by
  intro hS
  have hpick : CanPickUp Max := (P2 hS) Max
  exact Max_unable_pickup hpick

  apply Or.elim h

    exact False.elim (Max_unable_pickup hpick)
