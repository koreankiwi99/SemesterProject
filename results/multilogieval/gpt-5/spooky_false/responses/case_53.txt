<lean>
axiom Person : Type
axiom Mary : Person

axiom PracticedOften : Person → Prop
axiom TechniqueImproved : Person → Prop
axiom HighQuality : Person → Prop
axiom TookArtClass : Person → Prop
axiom LearnedNewMethods : Person → Prop

-- If someone practices often, their technique improves.
axiom P1 : ∀ x : Person, PracticedOften x → TechniqueImproved x
-- If someone's technique improves, their artwork stands out as high quality.
axiom P2 : ∀ x : Person, TechniqueImproved x → HighQuality x
-- Either they practiced their skills often, or they took an art class, or both.
axiom P3 : ∀ x : Person, PracticedOften x ∨ TookArtClass x
-- If someone takes an art class, they learn new methods and mediums for creating art.
axiom P4 : ∀ x : Person, TookArtClass x → LearnedNewMethods x

-- Given: Mary's artwork did not stand out as high quality.
axiom Mary_not_high_quality : ¬HighQuality Mary

theorem mary_learned_new_methods : LearnedNewMethods Mary := by
  -- From ¬HighQuality Mary and (Practiced → Improved → HighQuality), conclude ¬Practiced Mary.
  have not_practice : ¬PracticedOften Mary := by
    intro hpr
    have hti : TechniqueImproved Mary := P1 Mary hpr
    have hhq : HighQuality Mary := P2 Mary hti
    exact Mary_not_high_quality hhq
  -- From Practiced ∨ TookClass and ¬Practiced, deduce TookClass.
  have disj : PracticedOften Mary ∨ TookArtClass Mary := P3 Mary
  have tookClass : TookArtClass Mary := by
    cases disj with
    | inl hpr => exact False.elim (not_practice hpr)
    | inr htc => exact htc
  -- From TookClass, conclude LearnedNewMethods.
  exact P4 Mary tookClass
</lean>
