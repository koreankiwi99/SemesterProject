<lean>
axiom obj : Type
axiom Emma : obj

axiom LearnBasicChords : obj → Prop
axiom LearnStrumming : obj → Prop
axiom PlaySimpleSongs : obj → Prop
axiom EntertainFriends : obj → Prop
axiom TakesLessons : obj → Prop
axiom MusicTheory : obj → Prop

-- If someone learned basic chords and strumming patterns, they could play simple songs.
axiom A1 : ∀ x : obj, LearnBasicChords x ∧ LearnStrumming x → PlaySimpleSongs x
-- If someone could play simple songs, they would be able to entertain friends with the guitar.
axiom A2 : ∀ x : obj, PlaySimpleSongs x → EntertainFriends x
-- Either they learned basic chords or they took guitar lessons (inclusive).
axiom A3 : ∀ x : obj, LearnBasicChords x ∨ TakesLessons x
-- If someone takes guitar lessons, they gain knowledge of music theory.
axiom A4 : ∀ x : obj, TakesLessons x → MusicTheory x

-- Emma is unable to entertain friends with the guitar.
axiom Emma_not_entertain : ¬EntertainFriends Emma

-- What we can derive from the premises:
theorem emma_not_play_simple : ¬PlaySimpleSongs Emma :=
by
  intro hps
  have hf : EntertainFriends Emma := A2 Emma hps
  exact Emma_not_entertain hf

theorem emma_not_both_chords_and_strumming : ¬(LearnBasicChords Emma ∧ LearnStrumming Emma) :=
by
  intro hboth
  have hps : PlaySimpleSongs Emma := A1 Emma hboth
  exact emma_not_play_simple hps
