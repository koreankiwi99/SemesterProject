<lean>
axiom obj : Type
axiom John : obj

axiom Alarm : Prop
axiom Time : Prop
axiom Late : obj → Prop
axiom MissBus : obj → Prop
axiom UpLate : obj → Prop
axiom BedLate : obj → Prop

-- Premises
axiom P1 : Alarm → Time
axiom P2 : ∀ x : obj, MissBus x → Late x
axiom P3 : Alarm ∨ ¬Late John
axiom P4 : ∀ x : obj, UpLate x → MissBus x
axiom P5 : ∀ x : obj, BedLate x → UpLate x

-- "John know it is not the time to get ready"
axiom NotTime : ¬Time

theorem john_not_bedlate : ¬BedLate John := by
  intro hBL
  have hNotAlarm : ¬Alarm := by
    intro hA
    have hT : Time := P1 hA
    exact NotTime hT
  have hNotLate : ¬Late John := by
    cases P3 with
    | inl hA =>
      exact False.elim (hNotAlarm hA)
    | inr hNL =>
      exact hNL
  have hNotMiss : ¬MissBus John := by
    intro hMB
    have hL : Late John := P2 John hMB
    exact hNotLate hL
  have hNotUp : ¬UpLate John := by
    intro hUL
    have hMB : MissBus John := P4 John hUL
    exact hNotMiss hMB
  have hUp : UpLate John := P5 John hBL
  exact hNotUp hUp
</lean>

ANSWER: No
