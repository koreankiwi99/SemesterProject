<lean>
axiom obj : Type
axiom Priya : obj
axiom Attend : obj → Prop
axiom Study : obj → Prop
axiom Excel : obj → Prop

-- If a student attends lectures regularly, then if they study diligently, they will excel.
axiom H : ∀ x : obj, Attend x → (Study x → Excel x)

-- Priya studied diligently.
axiom Priya_studied : Study Priya

-- Priya did not excel in the course.
axiom Priya_not_excel : ¬Excel Priya

theorem priya_did_not_attend : ¬Attend Priya := by
  intro hAttend
  have hStudyImp : Study Priya → Excel Priya := H Priya hAttend
  have hExcel : Excel Priya := hStudyImp Priya_studied
  exact Priya_not_excel hExcel
</lean>

ANSWER: No
