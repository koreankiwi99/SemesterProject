axiom A3 : ∀ x : obj, MadeFriends x ∨ ¬Busy x
-- If someone is outgoing, then they would join a club.

-- If someone introduced themselves to classmates, then they are outgoing.
axiom A5 : ∀ x : obj, Introduced x → Outgoing x

theorem introduced_busy : ∀ x : obj, Introduced x → Busy x := by
  intro x hx
  exact A2 x (A4 x (A5 x hx))

theorem introduced_friends : ∀ x : obj, Introduced x → MadeFriends x := by
  intro x hx
  have hb : Busy x := introduced_busy x hx
  have hdisj : MadeFriends x ∨ ¬Busy x := A3 x
  cases hdisj with
  | inl hf => exact hf
  | inr hnb => exact False.elim (hnb hb)

theorem introduced_enjoy : ∀ x : obj, Introduced x → Enjoy x := by
  intro x hx
  exact A1 x (introduced_friends x hx)

  intro x hne hx
  exact hne (introduced_enjoy x hx)
