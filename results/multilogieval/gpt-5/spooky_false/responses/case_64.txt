<lean>
axiom obj : Type
axiom Priya : obj
axiom Attend : obj → Prop
axiom Revise : obj → Prop
axiom Perform : obj → Prop

-- Context: If a student attends lectures, then if they revise sufficiently, they will perform well.
axiom A1 : ∀ x : obj, Attend x → (Revise x → Perform x)

-- Given statements about Priya
axiom Priya_revised : Revise Priya
axiom Priya_not_perform : ¬Perform Priya

-- Conclusion: Priya did not attend lectures regularly.
theorem Priya_not_attend : ¬Attend Priya :=
by
  intro hAttend
  have hImp : Revise Priya → Perform Priya := A1 Priya hAttend
  have hPerf : Perform Priya := hImp Priya_revised
  exact Priya_not_perform hPerf
</lean>

ANSWER: No
