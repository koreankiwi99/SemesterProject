<lean>
axiom obj : Type
axiom Bookshelf : obj

axiom HighQuality : obj → Prop
axiom PreciseJoints : obj → Prop
axiom Sturdy : obj → Prop
axiom LongLasting : obj → Prop

-- If the wood is high quality, then if the joints are precise, the product is sturdy and long-lasting.
axiom CarpentryRule : ∀ x : obj, HighQuality x → (PreciseJoints x → Sturdy x ∧ LongLasting x)

-- The carpenter cut the joints very precisely.
axiom Precise_Bookshelf : PreciseJoints Bookshelf
-- The bookshelf was shaky and fell apart quickly.
axiom NotSturdy_Bookshelf : ¬Sturdy Bookshelf
axiom NotLongLasting_Bookshelf : ¬LongLasting Bookshelf

theorem wood_not_high_quality : ¬HighQuality Bookshelf := by
  intro hq
  have h_imp : PreciseJoints Bookshelf → Sturdy Bookshelf ∧ LongLasting Bookshelf := CarpentryRule Bookshelf hq
  have h_and : Sturdy Bookshelf ∧ LongLasting Bookshelf := h_imp Precise_Bookshelf
  have h_sturdy : Sturdy Bookshelf := And.left h_and
  exact NotSturdy_Bookshelf h_sturdy
</lean>

ANSWER: No
