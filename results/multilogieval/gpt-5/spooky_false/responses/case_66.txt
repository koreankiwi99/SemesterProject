<lean>
axiom obj : Type
axiom Acme : obj

axiom Innovative : obj → Prop
axiom MarketedEff : obj → Prop
axiom Successful : obj → Prop

-- If a company develops an innovative product, then if they market it effectively, the product will be commercially successful.
axiom Rule : ∀ x : obj, Innovative x → (MarketedEff x → Successful x)

-- Acme marketed their new gadget effectively.
axiom Acme_marketed : MarketedEff Acme
-- Acme's gadget was not commercially successful.
axiom Acme_not_success : ¬Successful Acme

theorem acme_not_innovative : ¬Innovative Acme := by
  intro hInnov
  have hImp : MarketedEff Acme → Successful Acme := Rule Acme hInnov
  have hSucc : Successful Acme := hImp Acme_marketed
  exact Acme_not_success hSucc
</lean>

ANSWER: No
