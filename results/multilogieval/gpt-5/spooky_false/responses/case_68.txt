<lean>
axiom obj : Type
axiom Emma : obj

axiom StudyHard : obj → Prop
axiom EnoughSleepBeforeExam : obj → Prop
axiom PerformWell : obj → Prop
axiom AceTest : obj → Prop

-- Belief: If someone studies hard, then if they get enough sleep before an exam, they will perform well and ace the test.
axiom belief : ∀ x : obj, StudyHard x → (EnoughSleepBeforeExam x → (PerformWell x ∧ AceTest x))

-- Facts about Emma
axiom sleep_emma : EnoughSleepBeforeExam Emma
axiom not_ace_emma : ¬AceTest Emma

theorem emma_not_studyhard : ¬StudyHard Emma := by
  intro hS
  have hImp := belief Emma hS
  have hPA := hImp sleep_emma
  have hA : AceTest Emma := And.right hPA
  exact not_ace_emma hA
</lean>

ANSWER: No
