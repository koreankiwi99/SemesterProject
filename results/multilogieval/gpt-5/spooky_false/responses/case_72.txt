axiom Sam : Person

axiom Close : Prop
axiom StartCooking : Person → Prop
axiom GoShopping : Person → Prop
axiom MakeDinner : Person → Prop
axiom FridgeEmpty : Person → Prop
axiom PantryEmpty : Person → Prop

-- If it is close to dinnertime, then they should start cooking.
axiom R1 : Close → StartCooking Sam
-- If they go shopping, then they (will) make dinner.
axiom R2 : GoShopping Sam → MakeDinner Sam
-- Either it is close to dinnertime, or they won't make dinner.
axiom R3 : Close ∨ ¬MakeDinner Sam
-- If the fridge is empty, then they must go shopping.
axiom R4 : FridgeEmpty Sam → GoShopping Sam
-- If the pantry is empty, then the fridge is empty.
axiom R5 : PantryEmpty Sam → FridgeEmpty Sam

-- Some derived consequences from the premises
theorem pantry_empty_implies_make : PantryEmpty Sam → MakeDinner Sam :=
  fun pe => R2 (R4 (R5 pe))

theorem pantry_empty_implies_close : PantryEmpty Sam → Close :=
  fun pe => by
    have md : MakeDinner Sam := pantry_empty_implies_make pe
    exact Or.elim R3 (fun hClose => hClose) (fun hNotMD => False.elim (hNotMD md))

theorem pantry_empty_implies_start : PantryEmpty Sam → StartCooking Sam :=
  fun pe => R1 (pantry_empty_implies_close pe)
