<lean>
axiom obj : Type
axiom Simone : obj

axiom TrainsWithDedication : obj → Prop
axiom AdequateRest : obj → Prop
axiom PeakPerformance : obj → Prop

-- If someone trains with dedication, then if they get adequate rest, they will perform at their peak.
axiom CoachesRule : ∀ x : obj, TrainsWithDedication x → AdequateRest x → PeakPerformance x

-- Simone got plenty of rest before the big match.
axiom Rest_Simone : AdequateRest Simone

-- She did not play her best during the game.
axiom Not_Peak_Simone : ¬PeakPerformance Simone

theorem simone_did_not_train : ¬TrainsWithDedication Simone := by
  intro hTrain
  have hPeak : PeakPerformance Simone := CoachesRule Simone hTrain Rest_Simone
  exact Not_Peak_Simone hPeak
</lean>
