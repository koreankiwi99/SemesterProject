<lean>
axiom obj : Type
axiom Liz : obj

axiom PreventiveMaintenance : obj → Prop
axiom QualityParts : obj → Prop
axiom LongServiceLife : obj → Prop

-- Mechanics' claim: If preventive maintenance is done, then if quality parts are used, the car will have a long service life.
axiom MechanicsClaim : ∀ x : obj, PreventiveMaintenance x → (QualityParts x → LongServiceLife x)

-- Facts about Liz's car
axiom QP_Liz : QualityParts Liz
axiom NotLong_Liz : ¬LongServiceLife Liz

-- Conclusion: Preventive maintenance was not done on Liz's car.
theorem no_PM_Liz : ¬PreventiveMaintenance Liz := by
  intro hpm
  have hImp : QualityParts Liz → LongServiceLife Liz := MechanicsClaim Liz hpm
  have hL : LongServiceLife Liz := hImp QP_Liz
  exact NotLong_Liz hL
</lean>

ANSWER: No
