<lean>
axiom obj : Type
axiom Alex : obj

-- Predicates
axiom Weekend : Prop
axiom Relax : obj → Prop
axiom Finish : obj → Prop
axiom Clear : obj → Prop
axiom Organized : obj → Prop
axiom ToDoList : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Weekend → Relax x
axiom P2 : ∀ x : obj, Finish x → Clear x
axiom P3 : Weekend ∨ ¬Clear Alex
axiom P4 : ∀ x : obj, Organized x → Finish x
axiom P5 : ∀ x : obj, ToDoList x → Organized x

-- Conclusion: From ¬Relax Alex, derive ¬ToDoList Alex
theorem not_list_of_not_relax : ¬Relax Alex → ¬ToDoList Alex := by
  intro hNotRelax
  intro hList
  have hOrg : Organized Alex := P5 Alex hList
  have hFin : Finish Alex := P4 Alex hOrg
  have hClear : Clear Alex := P2 Alex hFin
  have hWeekend : Weekend := by
    cases P3 with
    | inl hW => exact hW
    | inr hNotClear =>
        exact False.elim (hNotClear hClear)
  have hRelax : Relax Alex := P1 Alex hWeekend
  exact hNotRelax hRelax
</lean>
