<lean>
axiom obj : Type
axiom Team : obj

axiom Sunny : Prop
axiom KidsAtPark : Prop

axiom PracticeDaily : obj → Prop
axiom Prepared : obj → Prop
axiom Confident : obj → Prop
axiom HeadUpHigh : obj → Prop

-- If the team practices every day, they will be prepared for the big game.
axiom H1 : PracticeDaily Team → Prepared Team
-- If it is sunny outside, children will play at the park.
axiom H2 : Sunny → KidsAtPark
-- Either the team practiced daily, or kids did not go to the park today.
axiom H3 : PracticeDaily Team ∨ ¬KidsAtPark
-- If a team feels prepared, then they will have confidence.
axiom H4 : Prepared Team → Confident Team
-- When people feel confident, they hold their head up high.
axiom H5 : ∀ x : obj, Confident x → HeadUpHigh x

-- From the premises we can derive: If it was sunny, the team held their heads up high.
theorem sunny_implies_team_head_up : Sunny → HeadUpHigh Team := by
  intro hSunny
  have hKids : KidsAtPark := H2 hSunny
  have hPractice : PracticeDaily Team := by
    cases H3 with
    | inl hp => exact hp
    | inr hnot =>
      have : False := hnot hKids
      exact False.elim this
  have hPrepared : Prepared Team := H1 hPractice
  have hConf : Confident Team := H4 hPrepared
  exact H5 Team hConf
</lean>
