<lean>
axiom obj : Type
axiom John : obj

axiom WatersAdequately : obj → Prop
axiom BloomsBrightly : obj → Prop
axiom AttractsMoreBeesButterflies : obj → Prop
axiom UsesOrganicFertilizers : obj → Prop
axiom SoilHighQuality : obj → Prop
axiom PlantsGrowHealthier : obj → Prop

-- If they water adequately, plants bloom brightly
axiom R1 : ∀ x : obj, WatersAdequately x → BloomsBrightly x
-- If plants bloom brightly, they attract more bees and butterflies
axiom R2 : ∀ x : obj, BloomsBrightly x → AttractsMoreBeesButterflies x
-- Either they water adequately or they use organic fertilizers (or both)
axiom R3 : ∀ x : obj, WatersAdequately x ∨ UsesOrganicFertilizers x
-- If they use organic fertilizers, soil quality improves (is high quality)
axiom R4 : ∀ x : obj, UsesOrganicFertilizers x → SoilHighQuality x
-- If soil quality improves, plants grow healthier
axiom R5 : ∀ x : obj, SoilHighQuality x → PlantsGrowHealthier x

theorem water_implies_attracts : ∀ x : obj, WatersAdequately x → AttractsMoreBeesButterflies x := by
  intro x
  intro hW
  have hB : BloomsBrightly x := R1 x hW
  exact R2 x hB

-- From the premises, the queried implication actually holds:
-- If John's garden does not attract more bees and butterflies, then his soil is high quality.
theorem if_not_attract_then_soil_high_quality :
  (¬AttractsMoreBeesButterflies John) → SoilHighQuality John := by
  intro hNotA
  have hDisj : WatersAdequately John ∨ UsesOrganicFertilizers John := R3 John
  cases hDisj with
  | inl hW =>
      have hA : AttractsMoreBeesButterflies John := water_implies_attracts John hW
      exact False.elim (hNotA hA)
  | inr hF =>
      exact R4 John hF
