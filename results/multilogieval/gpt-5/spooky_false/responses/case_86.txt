<lean>
axiom obj : Type
axiom Priya : obj

axiom StudiedHard : obj → Prop
axiom GoodGrades : obj → Prop
axiom ParentsProud : obj → Prop
axiom IceCream : obj → Prop

axiom Snowstorm : Prop
axiom BigSnowstorm : Prop
axiom SchoolCanceled : Prop

-- If someone studies hard, they get good grades.
axiom StudyImpliesGrades : ∀ x : obj, StudiedHard x → GoodGrades x
-- If there is a snowstorm, school is canceled.
axiom SnowstormImpliesCanceled : Snowstorm → SchoolCanceled
-- Either Priya studied hard, or school was not called off.
axiom PriyaStudiedOrNotCanceled : StudiedHard Priya ∨ ¬SchoolCanceled
-- If a student gets good grades, their parents are proud.
axiom GradesImpliesProud : ∀ x : obj, GoodGrades x → ParentsProud x
-- When parents are proud, they take their kids out for ice cream.
axiom ProudImpliesIceCream : ∀ x : obj, ParentsProud x → IceCream x

-- From a (regular) snowstorm, we can derive that Priya gets taken out for ice cream.
-- This uses: snowstorm → canceled; with canceled, the disjunction forces StudiedHard Priya;
-- then good grades → proud → ice cream.
theorem snowstorm_implies_icecream_for_Priya : Snowstorm → IceCream Priya := by
  intro hs
  have hc : SchoolCanceled := SnowstormImpliesCanceled hs
  have disj : StudiedHard Priya ∨ ¬SchoolCanceled := PriyaStudiedOrNotCanceled
  have hstudy : StudiedHard Priya := by
    cases disj with
    | inl h => exact h
    | inr hnot => exact False.elim (hnot hc)
  have hg : GoodGrades Priya := StudyImpliesGrades Priya hstudy
  have hp : ParentsProud Priya := GradesImpliesProud Priya hg
  exact ProudImpliesIceCream Priya hp
