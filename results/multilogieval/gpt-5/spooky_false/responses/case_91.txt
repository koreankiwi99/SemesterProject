<lean>
axiom Party : Prop
axiom Drums : Prop
axiom VeryNoisy : Prop
axiom NoiseUp : Prop
axiom Complain : Prop
axiom PoliceCalled : Prop

-- If there's a party, the house gets very noisy.
axiom party_causes_noisy : Party → VeryNoisy
-- If someone is practicing drums, the noise levels go up.
axiom drums_cause_noiseup : Drums → NoiseUp
-- Either there is a party or someone is playing drums (or both).
axiom party_or_drums : Party ∨ Drums
-- If the house gets noisy from a party (we model this as: if party), neighbors complain.
axiom party_noise_triggers_complaint : Party → Complain
-- If neighbors complain, the police are called to intervene.
axiom complaints_call_police : Complain → PoliceCalled

-- From the premises we can prove the statement: (¬NoiseUp) → PoliceCalled
theorem police_called_if_no_noiseup : (¬NoiseUp) → PoliceCalled := by
  intro hNoUp
  have hNotDrums : ¬Drums := fun hD => hNoUp (drums_cause_noiseup hD)
  have hParty : Party := by
    cases party_or_drums with
    | inl hP => exact hP
    | inr hD => exact False.elim (hNotDrums hD)
  have hComplain : Complain := party_noise_triggers_complaint hParty
  exact complaints_call_police hComplain
</lean>
