<lean>
axiom Person : Type
axiom Lauren : Person

axiom TrainedHard : Person → Prop
axiom RacesFaster : Person → Prop
axiom StretchedAfterRuns : Person → Prop
axiom RecoverySerious : Person → Prop
axiom FocusFlexibility : Person → Prop

-- If they trained hard weekly, they could improve their race times (run races faster).
axiom H1 : ∀ x : Person, TrainedHard x → RacesFaster x

-- Either they trained hard at sprinting, or they stretched regularly after runs (for Lauren).
axiom H2 : TrainedHard Lauren ∨ StretchedAfterRuns Lauren

-- If they stretched after runs, they took recovery seriously.
axiom H3 : ∀ x : Person, StretchedAfterRuns x → RecoverySerious x

-- If they took recovery seriously, they focused on flexibility to prevent injury.
axiom H4 : ∀ x : Person, RecoverySerious x → FocusFlexibility x

-- From not running races faster, Lauren focuses on flexibility to prevent injury.
theorem lauren_focus_from_not_faster : ¬RacesFaster Lauren → FocusFlexibility Lauren := by
  intro notFaster
  have notTH : ¬TrainedHard Lauren := by
    intro th
    exact notFaster (H1 Lauren th)
  cases H2 with
  | inl th =>
      exact False.elim (notTH th)
  | inr stretched =>
      exact H4 Lauren (H3 Lauren stretched)
