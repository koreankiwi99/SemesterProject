axiom obj : Type
axiom Mary : obj

-- Propositions about the world
axiom Clear : Prop
axiom GameOn : Prop
axiom FansCheer : Prop
axiom ExcitingNight : Prop

-- Predicates about people
axiom Finish : obj → Prop
axiom Meet : obj → Prop

-- Premises from the context
axiom R1 : Clear → GameOn
axiom R2 : ∀ x : obj, Finish x → Meet x
axiom R3 : Clear ∨ ∃ x : obj, Finish x
axiom R4 : GameOn → FansCheer
axiom R5 : FansCheer → ExcitingNight

-- Some consequences we can derive

-- If it clears up, then it is an exciting night at the ballpark.
theorem clear_implies_exciting : Clear → ExcitingNight := by
  intro hClear
  have hGame : GameOn := R1 hClear
  have hCheer : FansCheer := R4 hGame
  exact R5 hCheer

-- From "if someone finishes their errands, they'll meet friends for dinner",
-- we can contrapose for Mary: if Mary did not meet friends, then she did not finish errands.
theorem not_meet_Mary_implies_not_finish_Mary : ¬Meet Mary → ¬Finish Mary := by
  intro hNotMeet
  intro hFinish
  have hMeet : Meet Mary := R2 Mary hFinish
  exact hNotMeet hMeet
