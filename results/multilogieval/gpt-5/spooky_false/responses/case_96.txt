axiom ClassCanceled : Prop
axiom MayCancel : Prop

-- If someone studies hard for an exam, they will feel confident going into the test.
axiom H1 : ∀ p : Person, ∀ e : Exam, StudiedHard p e → Confident p e
-- If there is severe weather in the forecast, schools may cancel class that day. (modal/weakened)
axiom H2 : SevereWeather → MayCancel
-- Either Alex studied diligently for his biology exam, or class was not canceled due to the weather.
axiom H3 : (StudiedHard Alex Biology) ∨ (¬ClassCanceled)
-- If a student feels confident about the material, they will likely do well on the exam.
axiom H4 : ∀ p : Person, ∀ e : Exam, Confident p e → DidWell p e
-- If a student performs well on an exam, they will be satisfied with their hard work.
axiom H5 : ∀ p : Person, ∀ e : Exam, DidWell p e → Satisfied p e

-- Derived: studying hard for an exam leads to being satisfied with that exam.
theorem study_to_satisfied : ∀ p : Person, ∀ e : Exam, StudiedHard p e → Satisfied p e := by
  intro p e hs
  have hc : Confident p e := H1 p e hs
  have hw : DidWell p e := H4 p e hc
  exact H5 p e hw
