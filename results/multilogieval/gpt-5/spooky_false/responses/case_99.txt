axiom HeavyRain : Prop
axiom SoilSaturated : Prop
axiom Watered : Prop
axiom Bloom : Prop
axiom Fragrance : Prop
axiom Butterflies : Prop

-- If someone waters the garden regularly, the flowers will bloom brightly.
axiom H1 : Watered → Bloom
-- If it rains a lot (heavy rain), then the soil becomes saturated.
axiom H2 : HeavyRain → SoilSaturated
-- Either the garden is watered routinely, or the soil did not saturate.
axiom H3 : ¬SoilSaturated ∨ Watered
-- If the flowers bloom, they will release fragrance.
axiom H4 : Bloom → Fragrance
-- If there is fragrance, it will attract butterflies.
axiom H5 : Fragrance → Butterflies

-- From the premises we can prove: If there was heavy rain, the garden attracted butterflies.
theorem heavy_rain_implies_butterflies : HeavyRain → Butterflies := by
  intro hr
  have s : SoilSaturated := H2 hr
  have w : Watered := by
    cases H3 with
    | inl hns =>
      have : False := hns s
      exact False.elim this
    | inr hw =>
      exact hw
  have b : Bloom := H1 w
  have f : Fragrance := H4 b
  exact H5 f
