<lean>
axiom obj : Type
axiom Alex : obj

axiom Studied : obj → Prop
axiom GoodGrades : obj → Prop
axiom Participated : obj → Prop
axiom NewFriends : obj → Prop
axiom Invited : obj → Prop

axiom R1 : ∀ x : obj, Studied x → GoodGrades x
axiom R2 : ∀ x : obj, Participated x → NewFriends x
axiom R3 : Studied Alex ∨ Participated Alex
axiom R4 : ∀ x : obj, NewFriends x → Invited x
axiom HnotGG : ¬ GoodGrades Alex

theorem alex_invited : Invited Alex := by
  have hns : ¬ Studied Alex := fun hs => HnotGG (R1 Alex hs)
  cases R3 with
  | inl hs =>
      exact False.elim (hns hs)
  | inr hp =>
      have nf : NewFriends Alex := R2 Alex hp
      exact R4 Alex nf
</lean>

ANSWER: Yes
