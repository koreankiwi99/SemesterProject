<lean>
axiom NiceDisplay : Prop
axiom JudgesImpressed : Prop
axiom ThoroughResearch : Prop
axiom ExpertsSerious : Prop
axiom Award : Prop

axiom R1 : NiceDisplay → JudgesImpressed
axiom R2 : ThoroughResearch → ExpertsSerious
axiom R3 : NiceDisplay ∨ ThoroughResearch
axiom R4 : ExpertsSerious → Award
axiom notImpressed : ¬ JudgesImpressed

theorem project_awarded : Award :=
by
  have hNotD : ¬ NiceDisplay := fun hD => notImpressed (R1 hD)
  have hT : ThoroughResearch :=
    Or.elim R3
      (fun hD => False.elim (hNotD hD))
      (fun hT => hT)
  have hS : ExpertsSerious := R2 hT
  exact R4 hS
</lean>

ANSWER: Yes
