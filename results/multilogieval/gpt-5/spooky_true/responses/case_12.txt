<lean>
axiom obj : Type
axiom James : obj

axiom Chef : obj → Prop
axiom Follow : obj → Prop
axiom Watch : obj → Prop
axiom GoodDish : obj → Prop
axiom Confidence : obj → Prop

-- If a chef follows recipes precisely, their dishes turn out well.
axiom R1 : ∀ x : obj, Chef x → Follow x → GoodDish x
-- If a chef's dishes turn out well, they gain confidence in the kitchen.
axiom R2 : ∀ x : obj, Chef x → GoodDish x → Confidence x
-- Each chef either followed recipes precisely or watched cooking shows (or both).
axiom R3 : ∀ x : obj, Chef x → (Follow x ∨ Watch x)

-- James is a chef.
axiom ChefJames : Chef James

theorem james_watches_if_not_confident : (¬ Confidence James) → Watch James := by
  intro hNotConf
  have hChoice := R3 James ChefJames
  cases hChoice with
  | inl hFollow =>
      have hGood : GoodDish James := R1 James ChefJames hFollow
      have hConf : Confidence James := R2 James ChefJames hGood
      have : False := hNotConf hConf
      exact False.elim this
  | inr hWatch =>
      exact hWatch
</lean>

ANSWER: Yes
