<lean>
axiom obj : Type
axiom Alex : obj

axiom AdvertisesCleverly : obj → Prop
axiom OffersDiscounts : obj → Prop
axiom AttractsMoreCustomers : obj → Prop
axiom IncreasesSales : obj → Prop
axiom ExpandsBusiness : obj → Prop

axiom A1 : ∀ x : obj, AdvertisesCleverly x → AttractsMoreCustomers x
axiom A2 : ∀ x : obj, OffersDiscounts x → IncreasesSales x
axiom A3 : ∀ x : obj, AdvertisesCleverly x ∨ OffersDiscounts x
axiom A4 : ∀ x : obj, IncreasesSales x → ExpandsBusiness x

axiom H : ¬ AttractsMoreCustomers Alex

theorem alex_expands : ExpandsBusiness Alex := by
  have hdisj : AdvertisesCleverly Alex ∨ OffersDiscounts Alex := A3 Alex
  have hnotad : ¬ AdvertisesCleverly Alex := fun h => H (A1 Alex h)
  have hOff : OffersDiscounts Alex := by
    cases hdisj with
    | inl hAd =>
      exact False.elim (hnotad hAd)
    | inr hOf =>
      exact hOf
  have hInc : IncreasesSales Alex := A2 Alex hOff
  exact A4 Alex hInc
</lean>

ANSWER: Yes
