<lean>
axiom obj : Type
axiom Student : obj
axiom Study : obj → Prop
axiom Excel : obj → Prop
axiom Participate : obj → Prop
axiom Leader : obj → Prop
axiom Prefect : obj → Prop

axiom Study_implies_Excel : Study Student → Excel Student
axiom Participate_implies_Leader : Participate Student → Leader Student
axiom leader_to_prefect : ∀ x : obj, Leader x → Prefect x
axiom study_or_participate : Study Student ∨ Participate Student
axiom not_excel : ¬ Excel Student

theorem student_chosen_prefect : Prefect Student := by
  have not_study : ¬ Study Student := fun hs => not_excel (Study_implies_Excel hs)
  have hpart : Participate Student :=
    Or.elim study_or_participate
      (fun hs => False.elim (not_study hs))
      (fun hp => hp)
  have hleader : Leader Student := Participate_implies_Leader hpart
  exact leader_to_prefect Student hleader
</lean>

ANSWER: Yes
