<lean>
axiom obj : Type
axiom Sarah : obj

axiom Practice : obj → Prop
axiom TechniqueImproves : obj → Prop
axiom LearnFaster : obj → Prop
axiom ListenAttentively : obj → Prop

-- If they practiced guitar (scales/chords daily), their technique would improve.
axiom A1 : Practice Sarah → TechniqueImproves Sarah
-- If their technique improved, they could learn songs faster.
axiom A2 : TechniqueImproves Sarah → LearnFaster Sarah
-- Either they practiced regularly or they listened attentively (or both).
axiom A3 : Practice Sarah ∨ ListenAttentively Sarah

theorem sarah_listens : ¬ LearnFaster Sarah → ListenAttentively Sarah := by
  intro hnot
  have hPL : Practice Sarah → LearnFaster Sarah := by
    intro hP
    have hT : TechniqueImproves Sarah := A1 hP
    exact A2 hT
  have hNotP : ¬ Practice Sarah := by
    intro hP
    exact hnot (hPL hP)
  cases A3 with
  | inl hP =>
    have hFalse : False := hNotP hP
    exact False.elim hFalse
  | inr hL =>
    exact hL
</lean>

ANSWER: Yes
