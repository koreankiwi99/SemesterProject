<lean>
axiom obj : Type
axiom James : obj

axiom Practice : obj → Prop
axiom Prepared : obj → Prop
axiom Confident : obj → Prop
axiom JoinClub : obj → Prop

axiom A1 : ∀ x : obj, Practice x → Prepared x
axiom A2 : ∀ x : obj, Prepared x → Confident x
axiom A3 : ∀ x : obj, Practice x ∨ JoinClub x

theorem james_joins_if_not_confident :
  ¬ Confident James → JoinClub James :=
by
  intro hnc
  have hdisj : Practice James ∨ JoinClub James := A3 James
  have hnotprac : ¬ Practice James :=
    by
      intro hprac
      have hprep : Prepared James := A1 James hprac
      have hconf : Confident James := A2 James hprep
      exact hnc hconf
  cases hdisj with
  | inl hprac =>
      have : False := hnotprac hprac
      exact False.elim this
  | inr hjoin => exact hjoin
</lean>

ANSWER: Yes
