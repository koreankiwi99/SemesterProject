<lean>
axiom obj : Type
axiom Sam : obj

axiom Impressed : obj → Prop
axiom ExtraWork : obj → Prop
axiom Promotion : obj → Prop
axiom Experience : obj → Prop
axiom AssignedLead : obj → Prop

axiom R1 : ∀ x : obj, Impressed x → Promotion x
axiom R2 : ∀ x : obj, ExtraWork x → Experience x
axiom R3 : Impressed Sam ∨ ExtraWork Sam
axiom R4 : ∀ x : obj, Experience x → AssignedLead x

theorem Sam_assigned_of_not_promoted (hnp : ¬ Promotion Sam) : AssignedLead Sam :=
by
  have hnotI : ¬ Impressed Sam := fun hI => hnp (R1 Sam hI)
  have hExtra : ExtraWork Sam :=
    Or.elim R3
      (fun hI => False.elim (hnotI hI))
      (fun hE => hE)
  have hExp : Experience Sam := R2 Sam hExtra
  exact R4 Sam hExp
</lean>

ANSWER: Yes
