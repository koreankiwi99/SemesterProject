<lean>
axiom Rain : Prop
axiom Hot : Prop
axiom Sprinklers : Prop
axiom Watered : Prop

axiom rain_implies_watered : Rain → Watered
axiom hot_implies_sprinklers : Hot → Sprinklers
axiom rain_or_not_sprinklers : Rain ∨ ¬Sprinklers
axiom Hot_today : Hot

theorem watered_today : Watered := by
  have s : Sprinklers := hot_implies_sprinklers Hot_today
  have not_not_sprinklers : ¬¬Sprinklers := fun h => h s
  have r : Rain :=
    Or.elim rain_or_not_sprinklers
      (fun hr => hr)
      (fun hns => False.elim (not_not_sprinklers hns))
  exact rain_implies_watered r
</lean>

ANSWER: Yes
