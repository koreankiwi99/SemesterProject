<lean>
axiom obj : Type
axiom Mary : obj

axiom BelowFreezing : Prop
axiom SnowOnGround : Prop
axiom EnoughSnow : Prop

axiom Built : obj → Prop
axiom LovesWinter : obj → Prop

-- If the temperature dropped below freezing, then there would be snow on the ground.
axiom H1 : BelowFreezing → SnowOnGround

-- If Mary built a snowman, that meant there was enough snow.
axiom H2 : Built Mary → EnoughSnow

-- Either it was below freezing, or there was not enough snow, or it was below freezing and there was not enough snow.
axiom H3 : BelowFreezing ∨ ¬EnoughSnow ∨ (BelowFreezing ∧ ¬EnoughSnow)

-- If someone loves winter, then they build snowmen.
axiom H4 : ∀ x : obj, LovesWinter x → Built x

-- From no snow on the ground, we can derive Mary does not love winter.
theorem no_snow_implies_not_loves_Mary : ¬SnowOnGround → ¬LovesWinter Mary :=
by
  intro hNoSnow
  -- From no snow on ground and BelowFreezing → SnowOnGround, get ¬BelowFreezing
  have hNotBF : ¬BelowFreezing := by
    intro hBF
    exact hNoSnow (H1 hBF)
  -- Use the certainty disjunction to conclude ¬EnoughSnow
  have hNotEnough : ¬EnoughSnow := by
    cases H3 with
    | inl hBF =>
        exact False.elim (hNotBF hBF)
    | inr hrest =>
        cases hrest with
        | inl hNE =>
            exact hNE
        | inr hBFNE =>
            exact hBFNE.right
  -- From Built Mary → EnoughSnow and ¬EnoughSnow, conclude ¬Built Mary
  have hNotBuilt : ¬Built Mary := by
    intro hB
    exact hNotEnough (H2 hB)
  -- From LovesWinter Mary → Built Mary and ¬Built Mary, conclude ¬LovesWinter Mary
  exact fun hLoves => hNotBuilt (H4 Mary hLoves)
