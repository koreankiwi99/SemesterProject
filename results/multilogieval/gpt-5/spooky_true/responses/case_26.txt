<lean>
axiom obj : Type
axiom team : obj

-- Predicates
axiom LostImportant : obj → Prop
axiom LostBig : obj → Prop
axiom LostSignificant : obj → Prop
axiom Disappointed : obj → Prop
axiom Demoralized : obj → Prop

-- Contextual rules
axiom R1 : ∀ x : obj, LostImportant x → Disappointed x
axiom R2 : ∀ x : obj, LostBig x → Demoralized x

-- Synonym bridging: "significant game" implies both "important game" and "big match"
axiom Sig_to_Imp : ∀ x : obj, LostSignificant x → LostImportant x
axiom Sig_to_Big : ∀ x : obj, LostSignificant x → LostBig x

-- Statement to evaluate (assumed for the question): players are neither disappointed nor demoralized
axiom players_neither : ¬ Disappointed team ∧ ¬ Demoralized team

-- Conclusion: the team did not lose a significant game
theorem not_lost_significant : ¬ LostSignificant team := by
  intro hsig
  have himp : LostImportant team := Sig_to_Imp team hsig
  have hd : Disappointed team := R1 team himp
  have hnd : ¬ Disappointed team := And.left players_neither
  exact hnd hd
</lean>

ANSWER: Yes
