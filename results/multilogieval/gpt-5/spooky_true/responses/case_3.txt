<lean>
axiom A_score : Prop
axiom A_win : Prop
axiom B_fumble : Prop
axiom B_lose : Prop

axiom H1 : A_score → A_win
axiom H2 : B_fumble → B_lose
axiom H3 : A_score ∨ B_fumble

theorem statement : ¬ B_lose → A_score := by
  intro h_not_lose
  cases H3 with
  | inl hA => exact hA
  | inr hB =>
      exact False.elim (h_not_lose (H2 hB))
</lean>

ANSWER: Yes
