axiom HealthyDiet : obj → Prop
axiom Strong : obj → Prop

axiom Skillful : obj → Prop

axiom K3 : ∀ x : obj, Athlete x → (¬ Skillful x ∨ ¬ Strong x) -- athletes lack skill or are not strong
axiom K4 : ∀ x : obj, Dedicated x → ProperTraining x         -- dedication leads to proper training

-- Given fact about Mark
axiom Mark_healthy : HealthyDiet Mark
axiom Mark_athlete : Athlete Mark

-- From the context and the given, we can derive that Mark is not dedicated (contrary to the target).
theorem mark_not_dedicated : ¬ Dedicated Mark := by
  have strongM : Strong Mark := K2 Mark Mark_healthy
  have hdisj : ¬ Skillful Mark ∨ ¬ Strong Mark := K3 Mark Mark_athlete
  have notSkill : ¬ Skillful Mark := by
    cases hdisj with
    | inl ns => exact ns
    | inr nstr =>
        have contra : False := nstr strongM
        exact False.elim contra
  intro d
  have sk : Skillful Mark := K1 Mark (K4 Mark d)
  exact notSkill sk
