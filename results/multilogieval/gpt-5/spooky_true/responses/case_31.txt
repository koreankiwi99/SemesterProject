axiom Priya : person

axiom Suspects : person → Prop          -- suspects partner is cheating
axiom Talked : person → Prop            -- had an open conversation
axiom Reassessed : person → Prop        -- re-evaluated the relationship

-- Textual context as implications
axiom talk_rule : ∀ x : person, Suspects x → Talked x
axiom reassess_rule : ∀ x : person, Suspects x → Reassessed x

-- Given statement to evaluate
axiom disj : ¬ Talked Priya ∨ ¬ Reassessed Priya

-- From the disjunction and the rules, we can derive that Priya did NOT suspect infidelity
theorem priya_did_not_suspect : ¬ Suspects Priya := by
  cases disj with
  | inl hNotTalk =>
    intro hSus
    have hTalk : Talked Priya := talk_rule Priya hSus
    exact hNotTalk hTalk
  | inr hNotReassess =>
    intro hSus
    have hReassess : Reassessed Priya := reassess_rule Priya hSus
    exact hNotReassess hReassess
