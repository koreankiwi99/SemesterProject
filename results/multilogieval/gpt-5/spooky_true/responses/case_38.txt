<lean>
axiom obj : Type
axiom Mike : obj
axiom HasGas : obj → Prop
axiom CanDriveLong : obj → Prop
axiom SeeSights : obj → Prop
axiom ScenicRoute : obj → Prop
axiom LovesRoadTrips : obj → Prop

axiom gas_implies_long : ∀ x : obj, HasGas x → CanDriveLong x
axiom see_implies_scenic : ∀ x : obj, SeeSights x → ScenicRoute x
axiom disj_mike : HasGas Mike ∨ ¬ScenicRoute Mike ∨ (HasGas Mike ∧ ¬ScenicRoute Mike)
axiom loves_common : ∀ x : obj, LovesRoadTrips x → ScenicRoute x ∧ SeeSights x

theorem no_gas_of_not_long : ∀ x : obj, ¬CanDriveLong x → ¬HasGas x := by
  intro x
  intro hNotLong
  intro hGas
  have hLong : CanDriveLong x := gas_implies_long x hGas
  exact hNotLong hLong

theorem not_scenic_of_not_long_Mike : ¬CanDriveLong Mike → ¬ScenicRoute Mike := by
  intro hNotLong
  have hNotGas : ¬HasGas Mike := no_gas_of_not_long Mike hNotLong
  have h := disj_mike
  cases h with
  | inl hGas =>
    exfalso
    exact hNotGas hGas
  | inr h2 =>
    cases h2 with
    | inl hNotScenic =>
      exact hNotScenic
    | inr hAnd =>
      exact And.right hAnd

theorem not_sights_of_not_scenic : ∀ x : obj, ¬ScenicRoute x → ¬SeeSights x := by
  intro x
  intro hNotScenic
  intro hSee
  have hScenic : ScenicRoute x := see_implies_scenic x hSee
  exact hNotScenic hScenic

theorem not_long_implies_not_loves_Mike : ¬CanDriveLong Mike → ¬LovesRoadTrips Mike := by
  intro hNotLong
  have hNotScenic : ¬ScenicRoute Mike := not_scenic_of_not_long_Mike hNotLong
  intro hLoves
  have hSV : ScenicRoute Mike ∧ SeeSights Mike := loves_common Mike hLoves
  exact hNotScenic hSV.left
