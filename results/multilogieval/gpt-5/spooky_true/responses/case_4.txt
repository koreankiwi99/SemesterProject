<lean>
axiom Marketing : Prop
axiom MoreCustomers : Prop
axiom SalesIncreased : Prop
axiom ProductionCostsIncreased : Prop

-- If they invested in marketing, they would gain more customers.
axiom invest_imp_customers : Marketing → MoreCustomers
-- With more customers, sales numbers would go up.
axiom customers_imp_sales : MoreCustomers → SalesIncreased
-- Either they focused budget on marketing, or production costs increased (inclusive or).
axiom marketing_or_costs : Marketing ∨ ProductionCostsIncreased

theorem costs_increased_of_no_sales (h : ¬ SalesIncreased) : ProductionCostsIncreased := by
  -- From marketing, sales would increase.
  have m_imp_s : Marketing → SalesIncreased :=
    fun m => customers_imp_sales (invest_imp_customers m)
  -- Since sales did not increase, they did not do marketing.
  have not_m : ¬ Marketing := fun m => h (m_imp_s m)
  -- From (Marketing ∨ ProductionCostsIncreased) and ¬Marketing, conclude ProductionCostsIncreased.
  cases marketing_or_costs with
  | inl m =>
      exact False.elim (not_m m)
  | inr c =>
      exact c
</lean>

ANSWER: Yes
