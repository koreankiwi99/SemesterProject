<lean>
axiom obj : Type
axiom James : obj

axiom Start : obj → Prop
axiom Motivated : obj → Prop
axiom Follow : obj → Prop
axiom Frustrated : obj → Prop
axiom Snacks : obj → Prop
axiom GymClosed : Prop

-- If someone starts a new workout routine, they will be motivated.
axiom A1 : ∀ x : obj, Start x → Motivated x
-- If the gym closes, then they can't follow their routine.
axiom A2 : ∀ x : obj, GymClosed → ¬ Follow x
-- Either they started a new workout, or the gym was closed (or both).
axiom A3 : ∀ x : obj, Start x ∨ GymClosed
-- If they couldn't follow their routine, they would feel frustrated.
axiom A4 : ∀ x : obj, ¬ Follow x → Frustrated x
-- If they felt frustrated, they might indulge in some unhealthy snacks.
axiom A5 : ∀ x : obj, Frustrated x → Snacks x

-- James was not motivated to workout.
axiom notMotJames : ¬ Motivated James

theorem james_indulged : Snacks James := by
  have h := A3 James
  cases h with
  | inl hStart =>
      have hm : Motivated James := A1 James hStart
      have hf : False := notMotJames hm
      exact False.elim hf
  | inr hClosed =>
      have hnf : ¬ Follow James := A2 James hClosed
      have hf : Frustrated James := A4 James hnf
      exact A5 James hf
</lean>

ANSWER: Yes
