<lean>
axiom Person : Type
axiom Priya : Person

axiom WaterDaily : Person → Prop
axiom GrowWell : Person → Prop
axiom Drought : Prop
axiom WaterRestrictions : Prop
axiom LackedWater : Person → Prop
axiom WiltDie : Person → Prop

axiom R1 : ∀ p : Person, WaterDaily p → GrowWell p
axiom R2 : Drought → WaterRestrictions
axiom R3 : (WaterDaily Priya) ∨ Drought ∨ ((WaterDaily Priya) ∧ Drought)
axiom R4 : ∀ p : Person, WaterRestrictions → LackedWater p
axiom R5 : ∀ p : Person, LackedWater p → WiltDie p

theorem priya_wilt_and_die (hNotGW : ¬ GrowWell Priya) : WiltDie Priya := by
  have hDrought : Drought := by
    have h := R3
    apply Or.elim h
    · intro hWD
      have hGW : GrowWell Priya := R1 Priya hWD
      have : False := hNotGW hGW
      exact False.elim this
    · intro h'
      apply Or.elim h'
      · intro hD; exact hD
      · intro hBoth
        exact hBoth.right
  have hWR : WaterRestrictions := R2 hDrought
  have hLW : LackedWater Priya := R4 Priya hWR
  exact R5 Priya hLW
</lean>

ANSWER: Yes
