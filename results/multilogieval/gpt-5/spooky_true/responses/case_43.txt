<lean>
axiom obj : Type
axiom James : obj
axiom Subway : obj → Prop
axiom BusDowntown : obj → Prop
axiom OnTime : obj → Prop

-- If someone takes the subway, they'll get to work on time.
axiom Subway_implies_OnTime : ∀ x : obj, Subway x → OnTime x

-- Whether it's the subway or the bus: for any person, they take one or the other.
axiom Bus_or_Subway : ∀ x : obj, BusDowntown x ∨ Subway x

-- Given: James did not take the bus downtown.
axiom NotBusJames : ¬ BusDowntown James

theorem James_on_time : OnTime James := by
  have h := Bus_or_Subway James
  cases h with
  | inl hb =>
      exact False.elim (NotBusJames hb)
  | inr hs =>
      exact Subway_implies_OnTime James hs
</lean>

ANSWER: Yes
