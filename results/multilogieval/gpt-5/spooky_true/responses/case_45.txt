<lean>
axiom obj : Type
axiom Matt : obj
axiom Watch : obj → Prop
axiom Read : obj → Prop
axiom Skill : obj → Prop
axiom Redo : obj → Prop
axiom Ideas : obj → Prop

axiom A1 : ∀ x : obj, Watch x → Skill x
axiom A2 : ∀ x : obj, Skill x → Redo x
axiom A3 : Watch Matt ∨ Read Matt
axiom A4 : ∀ x : obj, Read x → Ideas x

theorem matt_gets_ideas : (¬Redo Matt) → Ideas Matt := by
  intro hnr
  have hnskill : ¬Skill Matt := by
    intro hskill
    have hredo : Redo Matt := A2 Matt hskill
    exact hnr hredo
  have hnwatch : ¬Watch Matt := by
    intro hwatch
    have hskill : Skill Matt := A1 Matt hwatch
    exact hnskill hskill
  cases A3 with
  | inl hwatch =>
    have hf : False := hnwatch hwatch
    exact False.elim hf
  | inr hread =>
    exact A4 Matt hread
</lean>

ANSWER: Yes
