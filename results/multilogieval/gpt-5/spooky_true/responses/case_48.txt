<lean>
axiom Flip : Prop
axiom Gasp : Prop
axiom High : Prop
axiom Clowns : Prop
axiom Laugh : Prop

axiom R1 : Flip → Gasp
axiom R2 : Gasp → High
axiom R3 : Flip ∨ Clowns
axiom R4 : Clowns → Laugh
axiom NotHigh : ¬ High

theorem youngest_kids_laughed : Laugh := by
  have hc : Clowns := by
    cases R3 with
    | inl hf =>
      have hi : High := R2 (R1 hf)
      exact False.elim (NotHigh hi)
    | inr hc => exact hc
  exact R4 hc
</lean>

ANSWER: Yes
