<lean>
axiom Person : Type
axiom Mary : Person

axiom Learned : Person → Prop
axiom Understand : Person → Prop
axiom Thrive : Person → Prop
axiom SoughtAdvice : Person → Prop

-- If someone learned botany, they understand plant needs.
axiom A1 : ∀ x : Person, Learned x → Understand x
-- If someone understands plant needs, their plants thrive.
axiom A2 : ∀ x : Person, Understand x → Thrive x
-- Either they learned botany or they sought advice (or both).
axiom A3 : ∀ x : Person, Learned x ∨ SoughtAdvice x

-- Mary's plants did not thrive.
axiom NotThriveMary : ¬ Thrive Mary

theorem mary_sought_advice : SoughtAdvice Mary := by
  -- From learning to thriving (composition of A1 and A2)
  have learn_to_thrive : Learned Mary → Thrive Mary :=
    fun hL => A2 Mary (A1 Mary hL)
  -- If Mary's plants did not thrive, then she did not learn botany.
  have not_learned : ¬ Learned Mary :=
    fun hL => NotThriveMary (learn_to_thrive hL)
  -- Use the disjunction: either learned or sought advice.
  have disj := A3 Mary
  cases disj with
  | inl hL =>
      -- Contradiction if she had learned; thus she must have sought advice.
      exact False.elim (not_learned hL)
  | inr hS =>
      exact hS
</lean>

ANSWER: Yes
