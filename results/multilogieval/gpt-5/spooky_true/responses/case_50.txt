<lean>
axiom obj : Type
axiom Jesse : obj

axiom Sprinting : obj → Prop
axiom Improved : obj → Prop
axiom GoalTime : obj → Prop
axiom Stretching : obj → Prop
axiom Flexible : obj → Prop
axiom LessInjury : obj → Prop

axiom A1 : ∀ x : obj, Sprinting x → Improved x
axiom A2 : ∀ x : obj, Improved x → GoalTime x
axiom A3 : Sprinting Jesse ∨ Stretching Jesse
axiom A4 : ∀ x : obj, Stretching x → (Flexible x ∧ LessInjury x)

axiom Hnot : ¬ Improved Jesse

theorem jesse_flexible_lessinjury : Flexible Jesse ∧ LessInjury Jesse := by
  have notS : ¬ Sprinting Jesse := by
    intro sj
    exact Hnot (A1 Jesse sj)
  have st : Stretching Jesse := by
    cases A3 with
    | inl sj => exact False.elim (notS sj)
    | inr st => exact st
  exact A4 Jesse st
</lean>

ANSWER: Yes
