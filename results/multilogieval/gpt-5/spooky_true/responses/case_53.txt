<lean>
axiom person : Type
axiom Mary : person

axiom Practice : person → Prop
axiom Improve : person → Prop
axiom StandOut : person → Prop
axiom ArtClass : person → Prop
axiom LearnNew : person → Prop

-- If someone practices often, their technique improves
axiom R1 : ∀ x : person, Practice x → Improve x
-- If someone's technique improves, their artwork stands out as high quality
axiom R2 : ∀ x : person, Improve x → StandOut x
-- If someone takes an art class, they learn new methods and mediums
axiom R3 : ∀ x : person, ArtClass x → LearnNew x
-- Either Mary practiced often or Mary took an art class (or both)
axiom E1 : Practice Mary ∨ ArtClass Mary

theorem mary_learned : ¬ StandOut Mary → LearnNew Mary := by
  intro hNotS
  -- From practice, Mary would stand out
  have hPS : Practice Mary → StandOut Mary := by
    intro hP
    have hI : Improve Mary := R1 Mary hP
    exact R2 Mary hI
  -- Therefore, if Mary does not stand out, she did not practice
  have hNotP : ¬ Practice Mary := by
    intro hP
    exact hNotS (hPS hP)
  -- From (Practice ∨ ArtClass) and ¬Practice, deduce ArtClass
  have hA : ArtClass Mary := by
    cases E1 with
    | inl hP =>
        have : False := hNotP hP
        exact False.elim this
    | inr hA => exact hA
  -- From ArtClass, conclude Mary learned new methods and mediums
  exact R3 Mary hA
</lean>

ANSWER: Yes
