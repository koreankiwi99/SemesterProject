axiom Pumped : Prop
axiom CheerLoudly : Prop
axiom PlayRival : Prop
axiom Booing : Prop

axiom R1 : ScoresEarly → Pumped
axiom R2 : Pumped → CheerLoudly
axiom R3 : ScoresEarly ∨ PlayRival
axiom R4 : Pumped → Booing

-- Given statement to evaluate:
axiom HnotCheer : ¬ CheerLoudly

theorem not_pumped : ¬ Pumped := by
  intro hp
  have hc : CheerLoudly := R2 hp
  exact HnotCheer hc

theorem not_scores : ¬ ScoresEarly := by
  intro hs
  have hp : Pumped := R1 hs
  exact not_pumped hp

theorem we_play_rival : PlayRival := by
  cases R3 with
  | inl hs =>
      exact False.elim (not_scores hs)
  | inr pr =>
      exact pr

-- From HnotCheer we derived ¬Pumped and hence ¬ScoresEarly, and thus PlayRival.
-- However, Booing only follows from Pumped (R4), which contradicts ¬Pumped.
-- Therefore Booing cannot be derived from these premises.
