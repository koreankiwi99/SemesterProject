<lean>
axiom GoodReviews : Prop
axiom ManyPeople : Prop
axiom LowAttendance : Prop
axiom BigStar : Prop
axiom Sequel : Prop

-- If the movie gets good reviews, then many people will go see it.
axiom A1 : GoodReviews → ManyPeople

-- If someone buys advance tickets, then they plan to see the movie. (Irrelevant to the proof.)
axiom Person : Type
axiom BuysAdvanceTickets : Person → Prop
axiom PlansToSee : Person → Prop
axiom A2 : ∀ p : Person, BuysAdvanceTickets p → PlansToSee p

-- Either the movie got good reviews, or not many people went to see it, or it got good reviews and attendance was low. (Not used.)
axiom A3 : GoodReviews ∨ ¬ManyPeople ∨ (GoodReviews ∧ LowAttendance)

-- If a big star is in the movie, then it got good reviews.
axiom A4 : BigStar → GoodReviews

-- Either there is a big star, or it is a sequel, or both.
axiom A5 : BigStar ∨ Sequel ∨ (BigStar ∧ Sequel)

-- Low attendance means not many people went.
axiom A6 : LowAttendance → ¬ManyPeople

theorem low_attendance_implies_sequel : LowAttendance → Sequel := fun hL => by
  have hnotmany : ¬ManyPeople := A6 hL
  cases A5 with
  | inl hBS =>
      have hm : ManyPeople := A1 (A4 hBS)
      exact False.elim (hnotmany hm)
  | inr h =>
      cases h with
      | inl hSeq => exact hSeq
      | inr hboth => exact hboth.right
</lean>

ANSWER: Yes
