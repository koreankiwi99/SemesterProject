<lean>
axiom Person : Type
axiom Emma : Person

axiom SpeaksDaily : Person → Prop
axiom ListensPodcasts : Person → Prop
axiom FluencyImproved : Person → Prop
axiom VocabImproved : Person → Prop
axiom RealConversations : Person → Prop
axiom ListeningComp : Person → Prop

axiom A1 : ∀ x : Person, SpeaksDaily x → (FluencyImproved x ∧ VocabImproved x)
axiom A2 : ∀ x : Person, FluencyImproved x → RealConversations x
axiom A3 : ∀ x : Person, SpeaksDaily x ∨ ListensPodcasts x
axiom A4 : ∀ x : Person, ListensPodcasts x → ListeningComp x

theorem emma_listeningcomp_of_not_realconversations :
  (¬ RealConversations Emma) → ListeningComp Emma := by
  intro hNotRC
  have disj := A3 Emma
  have sd_to_rc : SpeaksDaily Emma → RealConversations Emma := fun hSD => by
    have hFI := A1 Emma hSD
    have hFl := And.left hFI
    exact A2 Emma hFl
  cases disj with
  | inl hSD =>
      have hRC := sd_to_rc hSD
      have hFalse : False := hNotRC hRC
      exact False.elim hFalse
  | inr hLP =>
      exact A4 Emma hLP
</lean>

ANSWER: Yes
