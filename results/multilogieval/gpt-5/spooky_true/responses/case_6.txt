<lean>
axiom obj : Type
axiom Dallas : obj

-- Single predicate capturing the idea of heavy/excessive rain
axiom Rain : obj → Prop

-- Overflow of reservoirs/dams in a city
axiom Overflow : obj → Prop

-- Flooding in a city
axiom Flood : obj → Prop

-- If there is excessive/heavy rain, reservoirs/dams overflow
axiom R1 : ∀ x : obj, Rain x → Overflow x

-- If reservoirs overflow, flooding occurs
axiom R2 : ∀ x : obj, Overflow x → Flood x

-- Given: There is heavy rainfall in Dallas
axiom T1 : Rain Dallas

theorem flooding_in_Dallas : Flood Dallas := by
  have h1 : Overflow Dallas := R1 Dallas T1
  exact R2 Dallas h1
</lean>

ANSWER: Yes
