axiom not_success_gadget : ¬ Successful Gadget

-- From the context and facts, we can derive that Acme's gadget was not innovative

  have hImp : MarketedEffectively Gadget → Successful Gadget := R Gadget hInnov
  have hSucc : Successful Gadget := hImp M_gadget
  exact not_success_gadget hSucc
