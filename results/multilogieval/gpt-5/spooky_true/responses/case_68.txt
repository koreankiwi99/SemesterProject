<lean>
axiom obj : Type
axiom Emma : obj
axiom StudyHard : obj → Prop
axiom EnoughSleep : obj → Prop
axiom PerformWell : obj → Prop
axiom Ace : obj → Prop

-- Belief: if someone studies hard, then if they get enough sleep, they will perform well and ace the test
axiom Belief : ∀ x : obj, StudyHard x → (EnoughSleep x → (PerformWell x ∧ Ace x))

-- Facts about Emma
axiom Sleep_Emma : EnoughSleep Emma
axiom NotAce_Emma : ¬ Ace Emma

-- From the belief and facts, we can derive that Emma did not study hard.
theorem emma_not_study_hard : ¬ StudyHard Emma := by
  intro hStudy
  have hImp : EnoughSleep Emma → (PerformWell Emma ∧ Ace Emma) := Belief Emma hStudy
  have hPW_Ace : PerformWell Emma ∧ Ace Emma := hImp Sleep_Emma
  have hAce : Ace Emma := hPW_Ace.right
  exact NotAce_Emma hAce
</lean>
