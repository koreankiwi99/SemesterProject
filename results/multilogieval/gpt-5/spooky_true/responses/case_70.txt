<lean>
axiom obj : Type
axiom Joe : obj

axiom ProperlyPlanted : obj → Prop
axiom AdequateSunlight : obj → Prop
axiom AdequateWatering : obj → Prop
axiom GrowLargePlentiful : obj → Prop

-- Gardeners' rule: if planted properly, then with adequate sunlight and watering, tomatoes grow large and plentiful
axiom gardeners_rule :
  ∀ x : obj, ProperlyPlanted x → (AdequateSunlight x ∧ AdequateWatering x → GrowLargePlentiful x)

-- Facts about Farmer Joe
axiom JoeAdequateSun : AdequateSunlight Joe
axiom JoeAdequateWater : AdequateWatering Joe
axiom JoeNotGrow : ¬ GrowLargePlentiful Joe

-- From the rule and Joe's facts, we can derive that Joe did not plant properly
theorem joe_not_properly_planted : ¬ ProperlyPlanted Joe := by
  intro hP
  have himpl := gardeners_rule Joe hP
  have hconds : AdequateSunlight Joe ∧ AdequateWatering Joe := And.intro JoeAdequateSun JoeAdequateWater
  have hgrow : GrowLargePlentiful Joe := himpl hconds
  exact JoeNotGrow hgrow
</lean>
