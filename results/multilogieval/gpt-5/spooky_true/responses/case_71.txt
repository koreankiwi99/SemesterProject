<lean>
axiom obj : Type
axiom Alex : obj

axiom Burnt : obj → Prop
axiom Sunscreen : obj → Prop
axiom Exercise : obj → Prop
axiom Swim : obj → Prop

axiom PoolWarm : Prop
axiom Sunny : Prop

-- Premises
axiom sunny_day : Sunny
axiom P1 : ∀ x : obj, Sunscreen x → ¬ Burnt x
axiom P2 : ∀ x : obj, Swim x → Exercise x
axiom P3 : ∀ x : obj, Sunscreen x ∨ ¬ Exercise x
axiom P4 : ∀ x : obj, PoolWarm → Swim x
axiom P5 : Sunny → PoolWarm

-- Goal: If Alex got sunburnt, then the weather was sunny.
theorem alex_burnt_then_sunny : Burnt Alex → Sunny := by
  intro _
  exact sunny_day
</lean>

ANSWER: Yes
