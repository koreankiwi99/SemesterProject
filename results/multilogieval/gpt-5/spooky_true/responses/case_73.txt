axiom AdequateRest : person → Prop

axiom CoachRule : ∀ x : person, TrainsDed x → AdequateRest x → PeakPerformance x

-- Facts about Simone
axiom Rest_Simone : AdequateRest Simone
axiom NotPeak_Simone : ¬ PeakPerformance Simone

-- From the rule, Simone's rest, and her not performing at peak, we can derive that she did not train with dedication.
theorem not_trained_Simone : ¬ TrainsDed Simone := by
  intro hT
  have hP : PeakPerformance Simone := CoachRule Simone hT Rest_Simone
  exact NotPeak_Simone hP
