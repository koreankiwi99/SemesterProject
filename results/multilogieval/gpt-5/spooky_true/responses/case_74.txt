axiom WatchGame : obj → Prop

axiom FinishHW : obj → Prop
axiom PlanWell : obj → Prop
axiom FridayNight : Prop

axiom A1 : ∀ x : obj, Student x → (FridayNight → Party x)
axiom A2 : ∀ x : obj, FinishHW x → WatchGame x
axiom A3 : ∀ x : obj, Student x → (FridayNight ∨ ¬ WatchGame x)
axiom A4 : ∀ x : obj, Student x → (StudyEveryday x → FinishHW x)
axiom A5 : ∀ x : obj, PlanWell x → StudyEveryday x

axiom SJ : Student James
axiom NP : ¬ Party James

theorem not_planwell_james : ¬ PlanWell James := by
  have hFP : FridayNight → Party James := A1 James SJ
  have hNotFri : ¬ FridayNight := fun hF => NP (hFP hF)
  have hNWG : ¬ WatchGame James := by
    have hDisj : FridayNight ∨ ¬ WatchGame James := A3 James SJ
    cases hDisj with
    | inl hF => exact False.elim (hNotFri hF)
    | inr h => exact h
  have hPWtoWG : PlanWell James → WatchGame James := by
    intro hPW
    have hSD : StudyEveryday James := A5 James hPW
    have hFHW : FinishHW James := A4 James SJ hSD
    exact A2 James hFHW
  exact fun hPW => hNWG (hPWtoWG hPW)
