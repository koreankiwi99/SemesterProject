<lean>
axiom Car : Type
axiom Liz : Car

axiom PreventiveMaintenance : Car → Prop
axiom QualityParts : Car → Prop
axiom LongServiceLife : Car → Prop

-- Mechanics' statement: if preventive maintenance is done, then if quality parts are used, the car will have a long service life.
axiom rule : ∀ c : Car, PreventiveMaintenance c → (QualityParts c → LongServiceLife c)

-- Given facts about Liz's car
axiom quality_Liz : QualityParts Liz
axiom not_long_Liz : ¬ LongServiceLife Liz

-- From the rule and the given facts, we can derive that preventive maintenance was not done on Liz's car.
theorem not_preventive_Liz : ¬ PreventiveMaintenance Liz := by
  intro hPM
  have hQL : QualityParts Liz → LongServiceLife Liz := rule Liz hPM
  have hL : LongServiceLife Liz := hQL quality_Liz
  exact not_long_Liz hL
