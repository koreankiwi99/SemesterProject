axiom PracticedHard : obj → Prop
axiom PickedStart : obj → Prop
axiom Talented : obj → Prop
axiom NaturalAbility : obj → Prop
axiom TopYouth : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, PracticedHard x → PlaysWell x
axiom P2 : ∀ x : obj, PickedStart x → Talented x
axiom P3 : ∀ x : obj, PracticedHard x ∨ ¬ Talented x
axiom P4 : ∀ x : obj, NaturalAbility x → PickedStart x
axiom P5 : ∀ x : obj, TopYouth x → NaturalAbility x

-- Derivable consequence: if James did not play well, then he was not a top youth player
theorem not_playwell_not_topJames : ¬ PlaysWell James → ¬ TopYouth James := by
  intro notPW
  have notPract : ¬ PracticedHard James := by
    intro hP
    apply notPW
    exact P1 James hP
  have disj : PracticedHard James ∨ ¬ Talented James := P3 James
  have notTal : ¬ Talented James := by
    cases disj with
    | inl hP =>
        have f : False := notPract hP
        exact fun t => False.elim f
    | inr hNT =>
        exact hNT
  have notPick : ¬ PickedStart James := by
    intro hPick
    have tal : Talented James := P2 James hPick
    exact notTal tal
  have notNat : ¬ NaturalAbility James := by
    intro hNat
    have p : PickedStart James := P4 James hNat
    exact notPick p
  intro hTop
  have nat : NaturalAbility James := P5 James hTop
  exact notNat nat
