axiom Person : Type
axiom John : Person

axiom WorkHard : Person → Prop
axiom Sore : Person → Prop
axiom HotShower : Person → Prop
axiom Relax : Person → Prop

axiom Cold : Prop
axiom Rain : Prop

-- If someone worked out hard, they are sore the next day.
axiom H1 : ∀ x : Person, WorkHard x → Sore x
-- If someone takes a hot shower, their muscles will relax.
axiom H2 : ∀ x : Person, HotShower x → Relax x
-- Either they worked out hard, or their muscles did not relax.
axiom H3 : ∀ x : Person, WorkHard x ∨ ¬ Relax x
-- If the weather is cold, they take a hot shower.
axiom H4 : ∀ x : Person, Cold → HotShower x
-- If it rains, then the weather is cold.
axiom H5 : Rain → Cold

-- From "not sore today" we can derive "not rain" (so the queried implication to Rain does not hold).
theorem not_sore_implies_not_rain : ¬ Sore John → ¬ Rain := by
  intro hNotSore
  intro hRain
  have hCold : Cold := H5 hRain
  have hHot : HotShower John := H4 John hCold
  have hRelax : Relax John := H2 John hHot
  have hNotWork : ¬ WorkHard John := by
    intro hWork
    have hSore : Sore John := H1 John hWork
    exact hNotSore hSore
  have hDisj : WorkHard John ∨ ¬ Relax John := H3 John
  have hNotRelax : ¬ Relax John := by
    cases hDisj with
    | inl hW =>
      exact False.elim (hNotWork hW)
    | inr hNR =>
      exact hNR
  exact hNotRelax hRelax
