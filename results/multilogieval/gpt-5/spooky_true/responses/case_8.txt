<lean>
axiom obj : Type
axiom Chef : obj

axiom Fresh : obj → Prop
axiom Delicious : obj → Prop
axiom Complicated : obj → Prop
axiom MoreTime : obj → Prop

-- If someone uses fresh ingredients, the food will taste delicious.
axiom H1 : ∀ x : obj, Fresh x → Delicious x
-- If someone cooks a complicated recipe, it takes more time to prepare.
axiom H2 : ∀ x : obj, Complicated x → MoreTime x
-- The chef used fresh ingredients. (Interpreting the given "Either the chef used fresh ingredients.")
axiom H3 : Fresh Chef

theorem complicated_implies_delicious : Complicated Chef → Delicious Chef :=
by
  intro _
  exact H1 Chef H3
</lean>

ANSWER: Yes
