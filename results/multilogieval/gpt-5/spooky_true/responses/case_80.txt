<lean>
axiom Rain : Prop
axiom Game : Prop
axiom PoolOpen : Prop
axiom KidsSwim : Prop
axiom Enjoy : Prop
axiom Smiles : Prop

axiom A1 : ¬Rain → Game
axiom A2 : PoolOpen → KidsSwim
axiom A3 : ¬Rain ∨ PoolOpen
axiom A4 : Game → Enjoy
axiom A5 : Enjoy → Smiles

theorem players_smile_if_kids_did_not_swim : (¬KidsSwim) → Smiles := by
  intro hNoSwim
  have hNoOpen : ¬PoolOpen := by
    intro hOpen
    exact hNoSwim (A2 hOpen)
  have hNoRain : ¬Rain := by
    cases A3 with
    | inl hnr => exact hnr
    | inr hOpen => exact False.elim (hNoOpen hOpen)
  have hGame : Game := A1 hNoRain
  have hEnjoy : Enjoy := A4 hGame
  exact A5 hEnjoy
</lean>

ANSWER: Yes
