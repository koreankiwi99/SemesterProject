<lean>
axiom obj : Type
axiom John : obj

axiom Water : obj → Prop
axiom Bloom : obj → Prop
axiom Attract : obj → Prop
axiom Organic : obj → Prop
axiom SoilHigh : obj → Prop
axiom Healthy : obj → Prop

axiom A1 : ∀ x : obj, Water x → Bloom x
axiom A2 : ∀ x : obj, Bloom x → Attract x
axiom A3 : ∀ x : obj, Water x ∨ Organic x
axiom A4 : ∀ x : obj, Organic x → SoilHigh x
axiom A5 : ∀ x : obj, SoilHigh x → Healthy x

theorem john_soil_high_if_not_attracts :
  (¬ Attract John) → SoilHigh John := by
  intro hNotAttract
  have hNotBloom : ¬ Bloom John := by
    intro hBloom
    exact hNotAttract (A2 John hBloom)
  have hNotWater : ¬ Water John := by
    intro hWater
    exact hNotBloom (A1 John hWater)
  have hDisj : Water John ∨ Organic John := A3 John
  cases hDisj with
  | inl hWater =>
    exact False.elim (hNotWater hWater)
  | inr hOrg =>
    exact A4 John hOrg
</lean>

ANSWER: Yes
