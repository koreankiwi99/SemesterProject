<lean>
axiom obj : Type
axiom John : obj

axiom Early : obj → Prop
axiom LongLunch : obj → Prop
axiom Recharged : obj → Prop
axiom Productive : obj → Prop
axiom Birthday : Prop
axiom Treats : obj → Prop

-- If someone finishes early, they can take a long lunch
axiom R1 : ∀ x : obj, Early x → LongLunch x
-- If someone takes a long lunch, they'll come back recharged
axiom R2 : ∀ x : obj, LongLunch x → Recharged x
-- If someone is recharged, they'll have a very productive afternoon
axiom R3 : ∀ x : obj, Recharged x → Productive x
-- If it's a coworker's birthday, people (everyone) will bring in treats
axiom R4 : ∀ x : obj, Birthday → Treats x
-- Either projects get done early (for each person) or it's a birthday (global), or both
axiom EorB : ∀ x : obj, Early x ∨ Birthday

theorem john_productive_of_not_treats : (¬ Treats John) → Productive John :=
by
  intro hnot
  have hor : Early John ∨ Birthday := EorB John
  cases hor with
  | inl hE =>
      have hll : LongLunch John := R1 John hE
      have hr : Recharged John := R2 John hll
      exact R3 John hr
  | inr hB =>
      have ht : Treats John := R4 John hB
      exact False.elim (hnot ht)
</lean>

ANSWER: Yes
