<lean>
axiom Oven350 : Prop
axiom BakesProperly : Prop
axiom Eggs : Prop
axiom Fluffy : Prop
axiom Delicious : Prop
axiom GuestsHappy : Prop

axiom R1 : Oven350 → BakesProperly
axiom R2 : Eggs → Fluffy
axiom R3 : Oven350 ∨ ¬Eggs
axiom R4 : BakesProperly → Delicious
axiom R5 : Delicious → GuestsHappy

theorem eggs_implies_happy : Eggs → GuestsHappy := by
  intro hEggs
  cases R3 with
  | inl hOven =>
      have hProper : BakesProperly := R1 hOven
      have hDelicious : Delicious := R4 hProper
      exact R5 hDelicious
  | inr hNotEggs =>
      have hFalse : False := hNotEggs hEggs
      exact False.elim hFalse
</lean>

ANSWER: Yes
