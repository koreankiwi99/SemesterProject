<lean>
axiom Winning : Prop
axiom Rivalry : Prop
axiom Invested : Prop
axiom Loud : Prop
axiom Electric : Prop
axiom Unforgettable : Prop

axiom R1 : Winning → Loud
axiom R2 : Rivalry → Invested
axiom R3 : Winning ∨ Rivalry
axiom R4 : Loud → Electric
axiom R5 : Electric → Unforgettable

theorem atmosphere_electric_if_not_invested : (Not Invested) → Electric := by
  intro hNotI
  have hNotR : Not Rivalry := by
    intro hR
    exact hNotI (R2 hR)
  have hW : Winning := by
    cases R3 with
    | inl hW => exact hW
    | inr hR => exact False.elim (hNotR hR)
  have hL : Loud := R1 hW
  exact R4 hL
</lean>

ANSWER: Yes
