<lean>
axiom Party : Prop
axiom Drums : Prop
axiom Noise : Prop
axiom Complain : Prop
axiom Police : Prop

-- If there's a party, it's noisy
axiom party_noise : Party → Noise
-- If someone plays drums, noise levels go up
axiom drums_noise : Drums → Noise
-- Either there is a party or someone is playing drums (or both)
axiom party_or_drums : Party ∨ Drums
-- If noisy from a party (i.e., if party), neighbors complain
axiom party_complain : Party → Complain
-- If neighbors complain, police are called
axiom complain_police : Complain → Police

-- Statement to evaluate: If there was no noise level up, were the police called?
theorem no_noise_implies_police : ¬ Noise → Police := by
  intro hNotNoise
  -- From ¬Noise and Party → Noise, derive ¬Party
  have hNotParty : ¬ Party := by
    intro hp
    apply hNotNoise
    exact party_noise hp
  -- From ¬Noise and Drums → Noise, derive ¬Drums
  have hNotDrums : ¬ Drums := by
    intro hd
    apply hNotNoise
    exact drums_noise hd
  -- But we know Party ∨ Drums, contradiction
  have contra : False := by
    cases party_or_drums with
    | inl hp => exact hNotParty hp
    | inr hd => exact hNotDrums hd
  -- From false, anything follows
  exact False.elim contra
</lean>

ANSWER: Yes
