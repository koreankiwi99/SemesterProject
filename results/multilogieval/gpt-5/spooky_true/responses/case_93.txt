<lean>
axiom PracticeHard : Prop
axiom Prepared : Prop
axiom Injured : Prop
axiom CanPlay : Prop
axiom Wins : Prop
axiom Celebrate : Prop

axiom R1 : PracticeHard → Prepared
axiom R2 : Injured → ¬CanPlay
axiom R3 : PracticeHard ∨ ¬Injured
axiom R4 : Prepared → Wins
axiom R5 : Wins → Celebrate

theorem injured_team_celebrates : Injured → Celebrate := by
  intro hI
  have hOr : PracticeHard ∨ ¬Injured := R3
  apply Or.elim hOr
  intro hPH
  have hPrep : Prepared := R1 hPH
  have hWin : Wins := R4 hPrep
  exact R5 hWin
  intro hNotI
  have hFalse : False := hNotI hI
  exact False.elim hFalse
</lean>

ANSWER: Yes
