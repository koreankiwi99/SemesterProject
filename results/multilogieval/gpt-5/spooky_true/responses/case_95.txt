<lean>
axiom ClearWeather : Prop
axiom GameOn : Prop
axiom MaryFinishedErrands : Prop
axiom MaryMetFriendsForDinner : Prop
axiom FansCheer : Prop
axiom ExcitingNightAtBallpark : Prop

axiom W1 : ClearWeather → GameOn
axiom W2 : MaryFinishedErrands → MaryMetFriendsForDinner
axiom W3 : ClearWeather ∨ MaryFinishedErrands
axiom W4 : GameOn → FansCheer
axiom W5 : FansCheer → ExcitingNightAtBallpark

theorem exciting_if_mary_did_not_meet :
  (¬ MaryMetFriendsForDinner) → ExcitingNightAtBallpark := by
  intro hNotMeet
  cases W3 with
  | inl hClear =>
      exact W5 (W4 (W1 hClear))
  | inr hFinished =>
      have hMeet : MaryMetFriendsForDinner := W2 hFinished
      have contra : False := hNotMeet hMeet
      exact False.elim contra
</lean>

ANSWER: Yes
