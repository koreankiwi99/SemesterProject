axiom Sub : Prop
axiom Pizza : Prop
axiom Goof : Prop
axiom Hyper : Prop
axiom Learn : Prop
axiom Homework : Prop

axiom A1 : Sub → Goof
axiom A2 : Pizza → Hyper
axiom A3 : Sub ∨ Pizza
axiom A4 : Goof → ¬Learn
axiom A5 : ¬Learn → Homework

theorem homework_if_not_hyper : ¬Hyper → Homework := by
  intro notHyper
  have notPizza : ¬Pizza := by
    intro hpizza
    exact notHyper (A2 hpizza)
  have hSub : Sub := by
    cases A3 with
    | inl hS =>
        exact hS
    | inr hP =>
        exact False.elim (notPizza hP)
  have hGoof : Goof := A1 hSub
  have hNotLearn : ¬Learn := A4 hGoof
  exact A5 hNotLearn
</lean>
