<lean>
axiom Watered : Prop
axiom Saturated : Prop
axiom HeavyRain : Prop
axiom Bloom : Prop
axiom Fragrance : Prop
axiom Butterflies : Prop

axiom P1 : Watered → Bloom
axiom P2 : HeavyRain → Saturated
axiom P3 : Watered ∨ ¬ Saturated
axiom P4 : Bloom → Fragrance
axiom P5 : Fragrance → Butterflies

theorem heavy_rain_attracts_butterflies : HeavyRain → Butterflies :=
  fun hr =>
    have s : Saturated := P2 hr
    have w : Watered :=
      Or.elim P3
        (fun w => w)
        (fun ns => False.elim (ns s))
    have b : Bloom := P1 w
    have f : Fragrance := P4 b
    show Butterflies from P5 f
</lean>

ANSWER: Yes
