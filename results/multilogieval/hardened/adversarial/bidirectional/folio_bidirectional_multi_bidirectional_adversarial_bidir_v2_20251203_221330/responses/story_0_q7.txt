-- Premises:
-- 1. If it is the first day of summer, then they get to sleep in late.
axiom p1 : FirstDayOfSummer → SleepInLate
-- 2. If the family goes on vacation, then the kids have to pack their bags.
axiom p2 : Vacation → PackBags
-- 3. If they did well in their exams (treated here as having finished their exams), then the family will go on vacation.
axiom p3 : FinishedExams → Vacation
-- 4. Either it was the first day of summer, or they did not pack their bags, or it was the first day of summer and they did not pack their bags.
axiom p4 : FirstDayOfSummer ∨ ¬PackBags ∨ (FirstDayOfSummer ∧ ¬PackBags)

-- Conclusion: If the kids did not sleep in late, then they did not finish their exams.
theorem conclusion : ¬SleepInLate → ¬FinishedExams := by
  intro hNotSleep
  intro hFinished
  have hVac : Vacation := p3 hFinished
  have hPack : PackBags := p2 hVac
  have hNotFirst : ¬FirstDayOfSummer := by
    intro hFirst
    have hSleep : SleepInLate := p1 hFirst
    exact hNotSleep hSleep
  have hNotPack : ¬PackBags := by
    have hdisj := p4
    cases hdisj with
    | inl hFirst =>
        exact (fun _ => False.elim (hNotFirst hFirst))
    | inr rest =>
        cases rest with
        | inl hnp => exact hnp
        | inr hAnd =>
            exact hAnd.right
  exact hNotPack hPack
