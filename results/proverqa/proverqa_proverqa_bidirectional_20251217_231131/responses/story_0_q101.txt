    exact hx.2 ⟨a9, hek⟩

  -- From "collaborates or publishes", exclude the "collaborates" case via contradiction to get PublishesResearch Lilian
  have hPR : PublishesResearch Lilian := by
    have h := a7 Lilian
    cases h with
    | inl hCC =>
        -- If collaborates (and not mentoring), then advance field

          intro hboth
          exact a3 hboth.right
        have hAF : AdvanceField Lilian := a25 ⟨Or.inl hCC, hNotBoth⟩
        -- From curiosity and advance field, get DNT → IU → EK, contradicting ¬EK

        have hEK  : EnhancesKnowledge Lilian := a17 Lilian (Or.inl hIU)
        exact False.elim (hNotEK hEK)
    | inr hPR => exact hPR

  -- From publishing and not leading innovative projects, deduce breakthroughs
  have hMB : MakesBreakthroughs Lilian := by
    have h := a23 hPR
    cases h.1 with
    | inl hmb => exact hmb
    | inr hlip => exact False.elim (a12 hlip)
