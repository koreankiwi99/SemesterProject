  ((ProducesToxins Queenie ∧ ¬ AdaptsEnvironments Queenie) ∨ (AdaptsEnvironments Queenie ∧ ¬ ProducesToxins Queenie)) → EvadesImmune Queenie

-- 9. If Cora either produces toxins or adapts to environments (but not both), then she can evade the immune system.
axiom prem9 :
  ((ProducesToxins Cora ∧ ¬ AdaptsEnvironments Cora) ∨ (AdaptsEnvironments Cora ∧ ¬ ProducesToxins Cora)) → EvadesImmune Cora

-- 10. If Queenie can evade the immune system, then she can either multiply rapidly or dominate the gut microbiome, but not both.
axiom prem10 :
  EvadesImmune Queenie →
  ((MultiplyRapidly Queenie ∧ ¬ DominatesGutMicrobiome Queenie) ∨ (DominatesGutMicrobiome Queenie ∧ ¬ MultiplyRapidly Queenie))

-- 11. A parasite may either cause chronic disease or multiply rapidly, but not necessarily both.

-- 12. Queenie adapts to environments.

-- 13. If Queenie either disrupts digestion or infests her host (but not both), then she causes chronic disease.

  ((DisruptsDigestion Queenie ∧ ¬ InfestsHost Queenie) ∨ (InfestsHost Queenie ∧ ¬ DisruptsDigestion Queenie)) → CausesChronicDisease Queenie
