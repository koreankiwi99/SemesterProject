  -- Using that Mckenzie is an animal researcher, deduce she is not a veterinarian.
  have hNotVet : ¬ Veterinarian Mckenzie := by
    have hNotBoth := And.right hVA_AR
    have hAR : AnimalResearcher Mckenzie := ax19
    intro hVet
    exact hNotBoth ⟨hVet, hAR⟩

    intro hE
    exact hNotSciAndEcol ⟨ax6, hE⟩
  -- Therefore, Mckenzie is a conservationist.
  have hCons : Conservationist Mckenzie := by
    cases hCorE with
    | inl hC => exact hC
    | inr hE => exact False.elim (hNotEcol hE)
  -- From being a conservationist, (Vet ∨ WorksWithAnimals) and not both.
  have hVorW := And.left (ax11 hCons)
  -- Using not veterinarian, deduce works with animals.
  have hWWA : WorksWithAnimals Mckenzie := by
    cases hVorW with
    | inl hVet => exact False.elim (hNotVet hVet)
    | inr hW => exact hW
  -- Conclude: Mckenzie handles elephants.
  exact ax9 hWWA
