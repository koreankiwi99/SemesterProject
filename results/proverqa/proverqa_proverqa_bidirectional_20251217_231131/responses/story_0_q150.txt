-- Premises

-- 1. Monroe listens actively.

-- 2. Monroe is either diplomatic or impulsive, but not both.
axiom p2 : (Diplomatic Monroe ∨ Impulsive Monroe) ∧ ¬(Diplomatic Monroe ∧ Impulsive Monroe)

-- 3. If Monroe builds trust, then he understands cultures and thinks critically.
axiom p3 : BuildsTrust Monroe → (UnderstandsCultures Monroe ∧ ThinksCritically Monroe)

-- 4. Monroe does not think critically.
axiom p4 : ¬ ThinksCritically Monroe

-- 5. Monroe negotiates effectively.
axiom p5 : NegotiatesEffectively Monroe

-- 6. Anyone who is diplomatic or builds trust can resolve conflicts.
axiom p6 : ∀ x : Person, (Diplomatic x ∨ BuildsTrust x) → ResolvesConflicts x

-- 7. Monroe does not have emotional intelligence.
axiom p7 : ¬ EmotionalIntelligence Monroe

-- 8. Monroe does not find common ground.
axiom p8 : ¬ FindsCommonGround Monroe

-- 9. Everyone who listens actively either builds strong relationships or finds common ground (or both).
axiom p9 : ∀ x : Person, ListensActively x → (BuildsStrongRelationships x ∨ FindsCommonGround x)

-- 10. If Monroe has leadership skills, then he either resolves conflicts or negotiates effectively, but not both.
axiom p10 : LeadershipSkills Monroe → ((ResolvesConflicts Monroe ∨ NegotiatesEffectively Monroe) ∧ ¬(ResolvesConflicts Monroe ∧ NegotiatesEffectively Monroe))

-- 11. Monroe either speaks multiple languages or has emotional intelligence, but not both.
axiom p11 : (SpeaksMultipleLanguages Monroe ∨ EmotionalIntelligence Monroe) ∧ ¬(SpeaksMultipleLanguages Monroe ∧ EmotionalIntelligence Monroe)

-- 12. Monroe remains calm.
axiom p12 : RemainsCalm Monroe

-- 13. If Monroe remains calm and communicates clearly, then he has leadership skills.
axiom p13 : (RemainsCalm Monroe ∧ CommunicatesClearly Monroe) → LeadershipSkills Monroe

-- Conclusion to prove: Monroe is not impulsive.
theorem not_impulsive : ¬ Impulsive Monroe := by
  -- From (3) and (4), we can derive that Monroe does not build trust.
  have h_notBT : ¬ BuildsTrust Monroe := by
    intro hBT
    have hTC : ThinksCritically Monroe := (p3 hBT).right
    exact p4 hTC

  -- From (1), (9), and (8), we can derive that Monroe builds strong relationships.
  have hBSR_or_FG : BuildsStrongRelationships Monroe ∨ FindsCommonGround Monroe := p9 Monroe p1
  have _hBSR : BuildsStrongRelationships Monroe := by
    cases hBSR_or_FG with
    | inl h => exact h
    | inr h => exact False.elim (p8 h)

  -- Further progress toward ¬ Impulsive Monroe appears to be blocked by the given premises.
  -- In particular, the exclusivity between Diplomatic and Impulsive in (2) does not, together with
  -- the other premises, allow us to eliminate the Impulsive case.
  sorry
