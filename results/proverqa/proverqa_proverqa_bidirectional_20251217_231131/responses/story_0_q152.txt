Premises: If Sylvia studies electromagnetic waves and conducts innovative experiments, then she will make groundbreaking discoveries. Sylvia understands wave theory. If Sylvia explores new ideas, then she either collaborates with others or advances her field. If someone collects data and understands wave theory, then they can analyze signal patterns. Sylvia does not advance her field. Sylvia explores new ideas or conducts innovative experiments. Sylvia builds on existing research. If Sylvia has a PhD, then she either analyzes signal patterns or specializes in acoustics, but not both. Sylvia collects data. If someone has curiosity, then they explore new ideas and solve complex problems. If Sylvia either analyzes signal patterns or studies electromagnetic waves (but not both), then she pioneers new technologies. For Sylvia, pioneering new technologies and building on existing research are mutually exclusive. Sylvia does not collaborate with others.
Conclusion: Sylvia will make groundbreaking discoveries.

open Classical

axiom StudyEM : Prop
axiom InnovExp : Prop
axiom Groundbreaking : Prop
axiom UnderstandWave : Prop

axiom BuildOnResearch : Prop

axiom SolveComplex : Prop
axiom PioneerTech : Prop

axiom ax1 : (StudyEM ∧ InnovExp) → Groundbreaking
axiom ax2 : UnderstandWave
axiom ax3 : ExploreIdeas → (Collaborate ∨ AdvanceField)
axiom ax4 : (CollectData ∧ UnderstandWave) → AnalyzePatterns
axiom ax5 : ¬AdvanceField
axiom ax6 : ExploreIdeas ∨ InnovExp
axiom ax7 : BuildOnResearch
axiom ax8 : HasPhD → ((AnalyzePatterns ∧ ¬SpecAcoustics) ∨ (SpecAcoustics ∧ ¬AnalyzePatterns))
axiom ax9 : CollectData
axiom ax10 : Curious → (ExploreIdeas ∧ SolveComplex)
axiom ax11 : ((AnalyzePatterns ∧ ¬StudyEM) ∨ (StudyEM ∧ ¬AnalyzePatterns)) → PioneerTech
axiom ax12a : PioneerTech → ¬BuildOnResearch
axiom ax12b : BuildOnResearch → ¬PioneerTech
axiom ax13 : ¬Collaborate

theorem goal : Groundbreaking := by
  -- From data collection and understanding wave theory, Sylvia can analyze signal patterns
  have hAP : AnalyzePatterns := by
    exact ax4 ⟨ax9, ax2⟩
  -- From "explore ⇒ (collaborate ∨ advance)" and (¬collaborate ∧ ¬advance), conclude ¬explore
  have hNotEI : ¬ExploreIdeas := by
    intro hEI
    have h := ax3 hEI
    cases h with
    | inl hC =>
        exact ax13 hC
    | inr hAF =>
        exact ax5 hAF
  -- From (explore ∨ innovate) and ¬explore, get innovate
  have hIE : InnovExp := by
    cases ax6 with
    | inl hEI =>
        exact False.elim (hNotEI hEI)
    | inr hIE =>
        exact hIE
  -- From building on research and mutual exclusivity, get ¬pioneer
  have hNotPT : ¬PioneerTech := by
    exact ax12b ax7
  -- Therefore, the XOR condition (AP xor StudyEM) is false
  have hNotXor : ¬((AnalyzePatterns ∧ ¬StudyEM) ∨ (StudyEM ∧ ¬AnalyzePatterns)) := by
    intro hx
    exact hNotPT (ax11 hx)
  -- Using AP and ¬(AP ∧ ¬StudyEM), get ¬¬StudyEM
  have hNotNotSE : ¬¬StudyEM := by
    intro hNotSE
    exact hNotXor (Or.inl ⟨hAP, hNotSE⟩)
  -- Classical double negation elimination gives StudyEM
  have hSE : StudyEM := by
    exact not_not.mp hNotNotSE
  -- Conclude groundbreaking discoveries from StudyEM ∧ InnovExp
  exact ax1 ⟨hSE, hIE⟩
