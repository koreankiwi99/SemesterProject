   ¬ (UnderstandsPhysics Presley ∧ AppreciatesMathematics Presley))

-- Goal: Presley values artistic expression.
theorem goal : ValuesArtisticExpression Presley := by
  -- From enjoys gardening, Presley specializes in landscaping
  have hSL : SpecializesLandscaping Presley := by
    exact landscaping_from_outdoors_or_gardening (Or.inr presley_enjoys_gardening)
  -- From SL and exclusivity with DI, ¬DI
  have hNotDI : ¬ DesignsInteriors Presley := by
    intro hdi
    exact landscaping_di_not_both ⟨hSL, hdi⟩
  -- From DI ∨ BS and ¬DI, deduce BS
  have hBS : BuildsStructures Presley := by
    cases di_or_bs with
    | inl hdi => exact False.elim (hNotDI hdi)
    | inr hbs => exact hbs
  -- From BS, deduce UP
  have hUP : UnderstandsPhysics Presley := builds_implies_up hBS
  -- From arch or eng and ¬eng, deduce arch
  have hArch : Architect Presley := by
    cases arch_or_eng with
    | inl harch => exact harch
    | inr heng => exact False.elim (presley_not_eng heng)
  -- From arch, get (UP xor AM), i.e., (UP ∨ AM) ∧ ¬(UP ∧ AM)
  have hXor := architect_implies_up_xor_am hArch
  have hNotBoth : ¬ (UnderstandsPhysics Presley ∧ AppreciatesMathematics Presley) := hXor.right
  -- With UP and ¬(UP ∧ AM), deduce ¬AM
  have hNotAM : ¬ AppreciatesMathematics Presley := by
    intro ham
    exact hNotBoth ⟨hUP, ham⟩
  -- From AM ∨ VAE and ¬AM, deduce VAE
  cases am_or_vae with
  | inl ham => exact False.elim (hNotAM ham)
  | inr hvae => exact hvae
