-- Conclusion: Kyler is not charming.
theorem Kyler_not_charming : ¬ Charming Kyler := by
  intro hcharm
  -- From the "xor service/attracts -> runs" and not running + not service, deduce not attracts
  have hnotA : ¬ AttractsDevotedCustomers Kyler := by
    intro a
    have hx :
      ((ProvidesGoodService Kyler ∧ ¬ AttractsDevotedCustomers Kyler) ∨
       (AttractsDevotedCustomers Kyler ∧ ¬ ProvidesGoodService Kyler)) :=
      Or.inr ⟨a, not_service_Kyler⟩
    have run := xor_service_attracts_imp_runs hx
    exact not_runs_Kyler run
  -- From (Creative ∨ StandsOut) -> EarnsRec and not EarnsRec, deduce not StandsOut
  have hnotS : ¬ StandsOut Kyler := by
    intro s
    have er := creative_or_stands_imp_rec_Kyler (Or.inr s)
    exact not_rec_Kyler er
  -- From Unique -> StandsOut and not StandsOut, deduce not Unique
  have hnotD : ¬ DevelopsUniqueStyle Kyler := by
    intro d
    have s := unique_imp_standout Kyler d
    exact hnotS s
  -- From Acclaim -> (Unique ∨ Loyal) and not Unique and not Loyal, deduce not Acclaim
  have hnotG : ¬ GainsCriticsAcclaim Kyler := by
    intro g
    have disj := acclaim_imp_unique_or_loyal Kyler g
    cases disj with
    | inl d => exact hnotD d
    | inr l => exact not_buildsloyal_Kyler l
  -- Charming Kyler gives XOR on (Gains, Attracts), contradicting both negations
  have hx := charming_Kyler_imp_xor hcharm
  cases hx with
  | inl h =>
      exact hnotG h.left
  | inr h =>
      exact hnotA h.left
