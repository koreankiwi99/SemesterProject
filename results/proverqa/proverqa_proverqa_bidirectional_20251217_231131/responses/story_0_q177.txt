-- Darling thrives in the dark.
axiom ax15 : ThrivesInDark Darling

  (HarmfulToRaccoonHost Queenie ∨ BeneficialToRaccoonHost Queenie)
  ∧ ¬(HarmfulToRaccoonHost Queenie ∧ BeneficialToRaccoonHost Queenie)

  (CoexistsWithHost Bumpkin ∨ HarmfulToRaccoonHost Bumpkin)
  ∧ ¬(CoexistsWithHost Bumpkin ∧ HarmfulToRaccoonHost Bumpkin)

axiom ax21 : MultipliesQuickly Darling

theorem goal : BeneficialToRaccoonHost Queenie := by
  -- From thriving in the dark, Queenie responds to changes.
  have hResp : RespondsToChanges Queenie := (ax8 ax7).left
  -- From monitoring and responding, Queenie senses host status.
  have hSense : SensesHostStatus Queenie := by
    exact (ax20 Queenie) (And.intro ax5 hResp)
  -- From sensing and adapting, Queenie limits consumption.
  have hLimit : LimitsConsumption Queenie := ax9 (And.intro hSense ax10)
  -- Build the exclusive-or condition to trigger regulation:
  have hDisj : ControlsGrowth Queenie ∨ LimitsConsumption Queenie := Or.inr hLimit
  have hNotBoth : ¬(ControlsGrowth Queenie ∧ LimitsConsumption Queenie) := by
    intro h
    exact ax3 h.left
  have hReg : RegulatesFeeding Queenie := ax1 (And.intro hDisj hNotBoth)
  -- Regulation implies coexistence.
  have hCoexist : CoexistsWithHost Queenie := ax14 hReg
  -- From XOR of (coexist, harmful) and coexist true, deduce not harmful.
  have hNotHarmful : ¬ HarmfulToRaccoonHost Queenie := by
    have hNoBoth : ¬ (CoexistsWithHost Queenie ∧ HarmfulToRaccoonHost Queenie) := (And.right ax6)
    intro hH
    exact hNoBoth (And.intro hCoexist hH)
  -- From XOR of (harmful, beneficial) and not harmful, deduce beneficial.
  have hHB : (HarmfulToRaccoonHost Queenie ∨ BeneficialToRaccoonHost Queenie) := (And.left ax16)
  have hBen : BeneficialToRaccoonHost Queenie := by
    cases hHB with
    | inl hH =>
        exact False.elim (hNotHarmful hH)
    | inr hB =>
        exact hB
  exact hBen
