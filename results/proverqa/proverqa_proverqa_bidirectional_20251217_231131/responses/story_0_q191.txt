  (FormsSymbioticRelationshipsWithHost Queenie ∨ EvadesHostsDefenses Queenie) ∧
  ¬(FormsSymbioticRelationshipsWithHost Queenie ∧ EvadesHostsDefenses Queenie)

  HasFlagella Queenie ∧ CompetesForResources Queenie → RegulatesNutrientFlow Queenie

  ∀ x : Creature, InhabitsIntestine x → (ProducesEnzymes x ∨ FeedsOnNutrients x)

  ∀ x : Creature, DominatesEnvironment x → CanReproduceSuccessfully x

-- Conclusion to prove
theorem goal : ThrivesInHost Queenie := by
  -- Derive that Queenie can reproduce successfully from dominance
  have hRepro : CanReproduceSuccessfully Queenie :=
    dominance_implies_repro Queenie queenie_dominates
  -- From XOR, deduce she does not produce enzymes
