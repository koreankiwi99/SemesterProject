-- Goal: Juliette inspires others.
theorem goal : InspiresOthers Juliette := by
  -- From confidence, Juliette breaks rules
  have hConf : Confidence Juliette := a22
  have hBR : BreaksRules Juliette := (a11 hConf).left
  -- Juliette does not defy conventions

  -- Hence XOR(Defies, Breaks) holds for Juliette
  have hXor : Xor (DefiesConventions Juliette) (BreaksRules Juliette) := by
    refine And.intro (Or.inr hBR) ?_
    intro hBoth
    exact hNoDef hBoth.left
  -- Therefore, Juliette rebels against norms
  have hReb : RebelsAgainstNorms Juliette := a19 hXor

  have hFSA : FreeSpiritedArtist Juliette := by
    cases hDisj with
    | inl h => exact h
    | inr h => exact False.elim (a16 h)
  -- Free-spirited artists inspire others
  have hPair : ExpressesUniquePerspective Juliette ∧ InspiresOthers Juliette := a21 Juliette hFSA
  exact hPair.right
