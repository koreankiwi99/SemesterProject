axiom Person : Type
axiom Lillian : Person

axiom Litterer : Person → Prop
axiom DamagesSoil : Person → Prop

axiom NeglectsResponsibility : Person → Prop
axiom IgnoresWarnings : Person → Prop
axiom Conservationist : Person → Prop
axiom CausesEnvironmentalHarm : Person → Prop

-- Premises
-- 1. If Lillian is a litterer, then she either damages soil or pollutes water, but not both.
axiom premise1 :
  Litterer Lillian →
  ((DamagesSoil Lillian ∨ PollutesWater Lillian) ∧ ¬(DamagesSoil Lillian ∧ PollutesWater Lillian))

-- 2. Everyone who either disposes waste or damages soil (but not both) harms the ecosystem.
axiom premise2 :
  ∀ x : Person,
    ((DisposesWaste x ∨ DamagesSoil x) ∧ ¬(DisposesWaste x ∧ DamagesSoil x)) → HarmsEcosystem x

-- 3. Lillian does not dispose waste.
axiom premise3 : ¬DisposesWaste Lillian

-- 4. Lillian throws trash.
axiom premise4 : ThrowsTrash Lillian

-- 5. Lillian does not harm wildlife.
axiom premise5 : ¬HarmsWildlife Lillian

-- 6. If Lillian neglects nature, then she is either a litterer or ignores rules, but not both.
axiom premise6 :
  NeglectsNature Lillian →
  ((Litterer Lillian ∨ IgnoresRules Lillian) ∧ ¬(Litterer Lillian ∧ IgnoresRules Lillian))

-- 7. Lillian does not neglect her responsibility.
axiom premise7 : ¬NeglectsResponsibility Lillian

-- 8. If someone pollutes water and causes environmental harm, then they harm wildlife.
axiom premise8 :
  ∀ x : Person, (PollutesWater x ∧ CausesEnvironmentalHarm x) → HarmsWildlife x

-- 9. Anyone who throws trash is a litterer.
axiom premise9 :
  ∀ x : Person, ThrowsTrash x → Litterer x

-- 10. Lillian is either a conservationist or an environmental harm doer, but not both.
--     Here, we identify "environmental harm doer" with "causes environmental harm".
axiom premise10 :
  (Conservationist Lillian ∨ CausesEnvironmentalHarm Lillian) ∧
  ¬(Conservationist Lillian ∧ CausesEnvironmentalHarm Lillian)

-- 11. Lillian does not ignore warnings.
axiom premise11 : ¬IgnoresWarnings Lillian

-- 12. Whenever someone harms the ecosystem, they either ignore warnings or neglect their responsibility (or both).
axiom premise12 :
  ∀ x : Person, HarmsEcosystem x → (IgnoresWarnings x ∨ NeglectsResponsibility x)

-- Negation of the conclusion ("Lillian is not a conservationist") is: Lillian is a conservationist.
theorem goal : Conservationist Lillian := by
  -- Lillian is a litterer (throws trash → litterer)
  have hLit : Litterer Lillian := premise9 Lillian premise4
  -- From litterer, she either damages soil or pollutes water (but not both)
  have hds_pw_and : (DamagesSoil Lillian ∨ PollutesWater Lillian) ∧ ¬(DamagesSoil Lillian ∧ PollutesWater Lillian) :=
    premise1 hLit
  -- From "not ignore warnings" and "not neglect responsibility" and premise12, Lillian does not harm the ecosystem
  have hNotHE : ¬HarmsEcosystem Lillian := by
    intro he
    have hw_or_nr : IgnoresWarnings Lillian ∨ NeglectsResponsibility Lillian := premise12 Lillian he
    cases hw_or_nr with
    | inl hiw => exact premise11 hiw
    | inr hnr => exact premise7 hnr
  -- Case analysis on (damages soil ∨ pollutes water)
  cases hds_pw_and.left with
  | inl hDS =>
      -- Build the XOR (DisposesWaste ∨ DamagesSoil) ∧ ¬(DisposesWaste ∧ DamagesSoil) to use premise2
      have hXorDWDS : (DisposesWaste Lillian ∨ DamagesSoil Lillian) ∧ ¬(DisposesWaste Lillian ∧ DamagesSoil Lillian) := by
        apply And.intro
        · exact Or.inr hDS
        · intro hboth
          exact premise3 hboth.left
      have hHE : HarmsEcosystem Lillian := premise2 Lillian hXorDWDS
      have contra : False := hNotHE hHE
      exact False.elim contra
  | inr hPW =>
      -- From not harming wildlife and (pollutes water ∧ causes env harm → harms wildlife), deduce ¬causes env harm
      have hNotCEH : ¬CausesEnvironmentalHarm Lillian := by
        intro hCEH
        have hpair : PollutesWater Lillian ∧ CausesEnvironmentalHarm Lillian := And.intro hPW hCEH
        have hHW : HarmsWildlife Lillian := premise8 Lillian hpair
        exact premise5 hHW
      -- Use the XOR (Conservationist ∨ CausesEnvHarm) ∧ ¬(both) to conclude Conservationist
      have hx := premise10
      have hdisj := hx.left
      cases hdisj with
      | inl hCons => exact hCons
      | inr hCEH  => exact False.elim (hNotCEH hCEH)
