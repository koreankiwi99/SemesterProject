  have hNotKM : ¬ KnowsMasonry Penny := by
    intro hKM
    exact hNotBothKMES ⟨hKM, hES⟩

  have hAD : AttentiveDetail Penny := ax7
  have hSC : SkilledCraftswoman Penny := ax21 hAD
  have hX : (WorksIndependently Penny ∨ ExperiencedQuarryman Penny) ∧ ¬(WorksIndependently Penny ∧ ExperiencedQuarryman Penny) := ax20 hSC
  have hWI : WorksIndependently Penny := ax8
  have hNotEQ : ¬ ExperiencedQuarryman Penny := by
    intro hEQ
    exact (And.right hX) ⟨hWI, hEQ⟩
  -- From (WorksWithStone Penny ∨ ExperiencedQuarryman Penny) and ¬EQ, get WorksWithStone Penny
  have hWWSorEQ : WorksWithStone Penny ∨ ExperiencedQuarryman Penny := ax12
  have hWWS : WorksWithStone Penny := by
    cases hWWSorEQ with
    | inl h => exact h
    | inr hEQ => exact False.elim (hNotEQ hEQ)
  -- From working with stone, deduce KnowsMasonry or CreatesLastingStructures
  have hKMorCLS : KnowsMasonry Penny ∨ CreatesLastingStructures Penny := ax16 Penny hWWS
  -- Using ¬KM, deduce CLS
  have hCLS : CreatesLastingStructures Penny := by
    cases hKMorCLS with
    | inl hKM => exact False.elim (hNotKM hKM)
    | inr hC => exact hC
  -- Contradiction with ax5, conclude anything (in particular, EarnsRespect Penny)
  have hFalse : False := ax5 hCLS
  exact False.elim hFalse
