                            ¬(RemainsFocused Baylor ∧ GetsAnxious Baylor) :=
    baylor_faces_challenge_implies_focused_or_anxious_xor h_fc

    And.right h_focus_or_anx_and

    intro hanx
    exact h_not_both_focus_anx ⟨baylor_focused, hanx⟩
  -- From anxious/calm xor and not anxious, Baylor is calm

    And.left baylor_anxious_or_calm_xor
