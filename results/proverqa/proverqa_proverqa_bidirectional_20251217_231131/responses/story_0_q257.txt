Premises: If Callan has a unique design, then it either pioneers a discovery or breaks boundaries (or both). Any spacecraft that meets specifications or is well maintained is fully functional. Callan does not make history. Galilea is maintained. Galilea does not break boundaries. If Callan is either remarkable or has a unique design (but not both), then it makes history. If Galilea is either fully functional or completely damaged (but not both), then it is remarkable. Galilea either pioneers a discovery or gathers critical data, but not both. Kenji runs smoothly. Any spacecraft with advanced sensors gathers critical data. Galilea operates normally or is fully functional. Kenji has advanced sensors. All maintained spacecraft either operate normally or run smoothly.

  ((FullyFunctional Galilea ∨ CompletelyDamaged Galilea) ∧ ¬(FullyFunctional Galilea ∧ CompletelyDamaged Galilea)) → Remarkable Galilea

  ∀ x : Spacecraft, AdvancedSensors x → GathersCriticalData x

  ∀ x : Spacecraft, Maintained x → (OperatesNormally x ∨ RunsSmoothly x)

open Classical

theorem conclusion : CompletelyDamaged Galilea := by
  -- From advanced sensors, Galilea gathers critical data
  have hGD : GathersCriticalData Galilea :=
    advanced_sensors_implies_gathers Galilea Galilea_advanced_sensors
  -- From the XOR about pioneers/gathers and having gathers, deduce not pioneers
  have hNotBoth : ¬(Pioneers Galilea ∧ GathersCriticalData Galilea) :=
    And.right Galilea_pioneers_xor_gathers
  have hNotPioneers : ¬ Pioneers Galilea :=
    fun hP => hNotBoth ⟨hP, hGD⟩
  -- From UD → (P ∨ B), together with ¬P and ¬B, deduce ¬UD
  have hNotUD : ¬ UniqueDesign Galilea :=
    fun hUD =>
      match Galilea_UD_implies_P_or_B hUD with
      | Or.inl hP => hNotPioneers hP
      | Or.inr hB => Galilea_not_breaks hB
  -- From (XOR(R,U) → MH) and ¬MH, deduce ¬XOR(R,U)
  have hNotXorRU :
      ¬ ((Remarkable Galilea ∨ UniqueDesign Galilea) ∧ ¬(Remarkable Galilea ∧ UniqueDesign Galilea)) :=
    fun hx => Galilea_not_MH (Galilea_Xor_R_or_U_implies_MH hx)
  -- Using ¬UD and ¬XOR(R,U), deduce ¬Remarkable
  have hNotRemarkable : ¬ Remarkable Galilea :=
    fun hR =>
      let hx : (Remarkable Galilea ∨ UniqueDesign Galilea) ∧ ¬(Remarkable Galilea ∧ UniqueDesign Galilea) :=
        And.intro (Or.inl hR) (fun hAnd => hNotUD hAnd.right)
      hNotXorRU hx
  -- From Maintained and the rule, Galilea is FullyFunctional
  have hFF : FullyFunctional Galilea :=
    meets_or_maintained_implies_fully Galilea (Or.inr Galilea_maintained)
  -- From (XOR(FF,CD) → R) and ¬R, deduce ¬XOR(FF,CD)
  have hNotXorFC :
      ¬ ((FullyFunctional Galilea ∨ CompletelyDamaged Galilea) ∧ ¬(FullyFunctional Galilea ∧ CompletelyDamaged Galilea)) :=
    fun hx => hNotRemarkable (Galilea_Xor_FF_or_CD_implies_R hx)
  -- Use classical EM on CD; if not CD, build XOR with FF and contradict hNotXorFC
  have hCDor : CompletelyDamaged Galilea ∨ ¬ CompletelyDamaged Galilea := Classical.em (CompletelyDamaged Galilea)
  match hCDor with
  | Or.inl hCD => exact hCD
  | Or.inr hNotCD =>
      have hx1 : FullyFunctional Galilea ∨ CompletelyDamaged Galilea := Or.inl hFF
      have hx2 : ¬(FullyFunctional Galilea ∧ CompletelyDamaged Galilea) :=
        fun hAnd => hNotCD hAnd.right
      have contra : False := hNotXorFC ⟨hx1, hx2⟩
      exact False.elim contra
