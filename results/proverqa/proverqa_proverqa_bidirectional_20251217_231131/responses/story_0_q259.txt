  have not_rumba : ¬ PerformsRumba Aileen := by
    intro r
    exact h_not_both_br ⟨Aileen_ballet, r⟩
  -- Build XOR (rumba xor samba) to infer teaches dance
  have xor_RS :
      (PerformsRumba Aileen ∨ DancesSamba Aileen) ∧ ¬(PerformsRumba Aileen ∧ DancesSamba Aileen) := by
    refine And.intro (Or.inr Aileen_samba) ?_
    intro h
    exact not_rumba h.left

  have not_both_ws : ¬ (LovesWaltz Aileen ∧ LovesSalsa Aileen) := hWorS.right

    intro ls
    exact not_both_ws ⟨Aileen_lovesWaltz, ls⟩
  -- From teaching dance, get (salsa or enjoys) and not both; deduce enjoys tango
  have hLSorET := Aileen_teach_imp_salsa_or_enjoy teaches

    Or.elim or1 (fun ls => False.elim (not_ls ls)) (fun et => et)
  -- From Aileen's enjoys/dislikes XOR, conclude not dislikes
  have not_both_ed : ¬ (EnjoysTango Aileen ∧ DislikesTango Aileen) := Aileen_enjoys_or_dislikes_xor.right
  intro dt
  exact not_both_ed ⟨enjoys, dt⟩
