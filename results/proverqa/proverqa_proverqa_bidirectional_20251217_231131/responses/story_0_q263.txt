-- Goal: prove the negation of the conclusion "Queenie multiplies rapidly"

  -- From virulence and not colonizing, deduce domination
  have hVD : ((Colonizes Queenie ∧ ¬Dominates Queenie) ∨ (Dominates Queenie ∧ ¬Colonizes Queenie)) :=
