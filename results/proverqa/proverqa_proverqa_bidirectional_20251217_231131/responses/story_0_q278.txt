-- Anyone who fights injustice or values integrity will stand up for themselves.

  ∀ x : Person, (FightsInjustice x ∨ ValuesIntegrity x) → StandsUpForSelf x

-- Anyone who defends their rights or challenges wrongs will fight injustice.

  ∀ x : Person, (DefendsRights x ∨ ChallengesWrongs x) → FightsInjustice x

-- Isabella does not believe in fairness.

-- Isabella either supports others or challenges wrongs, but not both.

  (SupportsOthers Isabella ∧ ¬ ChallengesWrongs Isabella) ∨
  (ChallengesWrongs Isabella ∧ ¬ SupportsOthers Isabella)

-- Luca is principled.

-- Isabella does not seek a judicial review.
axiom Isa_not_seekJR : ¬ SeeksJudicialReview Isabella

-- Isabella supports others.
axiom Isa_supports : SupportsOthers Isabella

-- Goal: Isabella gains respect.
theorem goal : GainsRespect Isabella := by
  -- From accusation and the implication, Isabella either seeks JR or defends rights
  have hdisj : SeeksJudicialReview Isabella ∨ DefendsRights Isabella :=
    Isa_face_imp Isa_faces
  -- Using the fact that she does not seek JR, conclude she defends her rights
  have hdef : DefendsRights Isabella := by
    cases hdisj with
    | inl hSJR =>
        have : False := Isa_not_seekJR hSJR
        exact False.elim this
    | inr hDR => exact hDR
  -- From defending rights, she fights injustice
  have hfight : FightsInjustice Isabella := by
    have hdc : DefendsRights Isabella ∨ ChallengesWrongs Isabella := Or.inl hdef
    exact DefendOrChallenge_imp_fight Isabella hdc
  -- From fighting (or integrity), she stands up for herself
  have hstand : StandsUpForSelf Isabella :=
    FightOrIntegrity_imp_stand Isabella (Or.inl hfight)
  -- Standing up yields gaining respect (and protecting reputation)
  have hres : GainsRespect Isabella ∧ ProtectsReputation Isabella :=
    StandsUp_imp_respect Isabella hstand
  exact hres.left
