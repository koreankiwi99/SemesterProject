axiom Person : Type

-- Premises
-- 1. If Justin either analyzes clinical data or creates treatment plans (but not both), then he improves patient outcomes.
axiom prem1 :
  ((AnalyzesClinicalData Justin ∧ ¬ CreatesTreatmentPlans Justin) ∨
   (CreatesTreatmentPlans Justin ∧ ¬ AnalyzesClinicalData Justin)) →
  ImprovesPatientOutcomes Justin

-- 2. Anyone who pursues medical advancements or develops new procedures can improve patient outcomes.
axiom prem2 :
  ∀ x : Person, (PursuesMedicalAdvancements x ∨ DevelopsNewProcedures x) → ImprovesPatientOutcomes x

-- 3. Justin does not advance medical knowledge.
axiom prem3 : ¬ AdvancesMedicalKnowledge Justin

-- 4. If Justin is dedicated to research, then he can either improve patient outcomes or inspire future generations, but not both.
axiom prem4 :
  DedicatedToResearch Justin →
  ((ImprovesPatientOutcomes Justin ∧ ¬ InspiresFutureGenerations Justin) ∨
   (InspiresFutureGenerations Justin ∧ ¬ ImprovesPatientOutcomes Justin))

-- 5. Justin is curious.
axiom prem5 : Curious Justin

-- 6. If Aron has a medical background, then he either pursues medical advancements or collaborates with specialists, but not both.
axiom prem6 :
  MedicalBackground Aron →
  ((PursuesMedicalAdvancements Aron ∧ ¬ CollaboratesWithSpecialists Aron) ∨
   (CollaboratesWithSpecialists Aron ∧ ¬ PursuesMedicalAdvancements Aron))

-- 7. If Justin seeks knowledge and enjoys learning, then he is dedicated to research.
axiom prem7 :
  (SeeksKnowledge Justin ∧ EnjoysLearning Justin) → DedicatedToResearch Justin

-- 8. Justin values accuracy.
axiom prem8 : ValuesAccuracy Justin

-- 9. If Justin strives for excellence, then he either advances medical knowledge or develops new procedures (or both).
axiom prem9 :
  StrivesForExcellence Justin → (AdvancesMedicalKnowledge Justin ∨ DevelopsNewProcedures Justin)

-- 10. Justin collaborates with specialists.
axiom prem10 : CollaboratesWithSpecialists Justin

-- 11. Sincere collaborates with specialists.
axiom prem11 : CollaboratesWithSpecialists Sincere

-- 12. If Justin is curious, then he either has a medical background or works independently, but not both.
axiom prem12 :
  Curious Justin →
  ((MedicalBackground Justin ∧ ¬ WorksIndependently Justin) ∨
   (WorksIndependently Justin ∧ ¬ MedicalBackground Justin))

-- 13. Justin does not work independently.
axiom prem13 : ¬ WorksIndependently Justin

-- 14. If Justin has a degree, then he either has a medical background or attends conferences, but not both.
axiom prem14 :
  HasDegree Justin →
  ((MedicalBackground Justin ∧ ¬ AttendsConferences Justin) ∨
   (AttendsConferences Justin ∧ ¬ MedicalBackground Justin))

-- 15. Celine values accuracy.
axiom prem15 : ValuesAccuracy Celine

-- 16. Celine reads medical journals.
axiom prem16 : ReadsMedicalJournals Celine

-- 17. Celine sets high standards or strives for excellence.
axiom prem17 : SetsHighStandards Celine ∨ StrivesForExcellence Celine

-- 18. Justin reads medical journals.
axiom prem18 : ReadsMedicalJournals Justin

-- 19. If Justin has a medical background, then he either pursues medical advancements or collaborates with specialists, but not both.
axiom prem19 :
  MedicalBackground Justin →
  ((PursuesMedicalAdvancements Justin ∧ ¬ CollaboratesWithSpecialists Justin) ∨
   (CollaboratesWithSpecialists Justin ∧ ¬ PursuesMedicalAdvancements Justin))

-- 20. Justin sets high standards or strives for excellence.
axiom prem20 : SetsHighStandards Justin ∨ StrivesForExcellence Justin

-- 21. If Celine is curious, then he either has a medical background or works independently, but not both.
axiom prem21 :
  Curious Celine →
  ((MedicalBackground Celine ∧ ¬ WorksIndependently Celine) ∨
   (WorksIndependently Celine ∧ ¬ MedicalBackground Celine))

-- 22. Justin does not set high standards.
axiom prem22 : ¬ SetsHighStandards Justin

-- 23. Anyone who values accuracy and reads medical journals is dedicated to research.
axiom prem23 :
  ∀ x : Person, (ValuesAccuracy x ∧ ReadsMedicalJournals x) → DedicatedToResearch x

-- Conclusion: Justin does not inspire future generations.
theorem goal : ¬ InspiresFutureGenerations Justin := by
  -- From (SetsHigh ∨ Strives) and ¬SetsHigh, get Strives
  have hStrives : StrivesForExcellence Justin :=
    Or.elim prem20
      (fun hSH => False.elim (prem22 hSH))
      (fun hSE => hSE)
  -- From Strives → (Advances ∨ Develops) and ¬Advances, get Develops
  have hDev : DevelopsNewProcedures Justin := by
    have h := prem9 hStrives
    cases h with
    | inl hAdv => exact False.elim (prem3 hAdv)
    | inr hDev => exact hDev
  -- From (Pursues ∨ Develops) → Improves and Develops, get Improves
  have hImprove : ImprovesPatientOutcomes Justin :=
    prem2 Justin (Or.inr hDev)
  -- From ValuesAccuracy ∧ Reads → Dedicated, get Dedicated
  have hDedicated : DedicatedToResearch Justin :=
    prem23 Justin ⟨prem8, prem18⟩
  -- From Dedicated → XOR(Improves, Inspires), combine with Improves to get ¬Inspires
  have hXor := prem4 hDedicated
  cases hXor with
  | inl h => exact h.2
  | inr h => 
      -- h.2 : ¬ Improves, contradicts hImprove, hence ¬ Inspires
      exact (by intro _; exact h.2 hImprove)
