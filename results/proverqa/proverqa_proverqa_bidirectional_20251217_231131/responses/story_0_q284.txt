open Classical

theorem goal : RegainsHealth Muffin := by
  -- From hydration and not healthy, deduce not grooming
  have not_groom : ¬ GroomsSelf Muffin := by
    intro g
    have healthy : FeelsHealthy Muffin :=
      all_cats_hydrate_groom_imp_healthy Muffin ⟨muffin_stays_hydrated, g⟩
    exact muffin_not_feels_healthy healthy
  -- From not shiny and (vibrant or playful) -> shiny, deduce not vibrant
  have not_vibrant : ¬ LooksVibrant Muffin := by
    intro v
    have shiny : ShinyCoat Muffin := muffin_vibrant_or_playful_imp_shiny (Or.inl v)
    exact muffin_not_shiny shiny
  -- Deduce Muffin does not improve her appearance
  have not_improves : ¬ ImprovesAppearance Muffin := by
    intro imp
    have ante :
      (GroomsSelf Muffin ∧ ¬ ImprovesAppearance Muffin) ∨ (¬ GroomsSelf Muffin ∧ ImprovesAppearance Muffin) :=
      Or.inr ⟨not_groom, imp⟩
    have vib : LooksVibrant Muffin := muffin_groom_or_improve_xor_imp_vibrant ante
    exact not_vibrant vib
  -- Show Muffin loses weight (using RA and the XOR-implies-improves, plus not_improves)
  have lw : LosesWeight Muffin := by
    by_cases h : LosesWeight Muffin
    · exact h
    ·
      have ante :
        (LosesWeight Muffin ∧ ¬ ReducesAppetite Muffin) ∨ (¬ LosesWeight Muffin ∧ ReducesAppetite Muffin) :=
        Or.inr ⟨h, muffin_reduces_appetite⟩
      have improves : ImprovesAppearance Muffin := muffin_lw_xor_ra_imp_improves ante
      exact False.elim (not_improves improves)
  -- From losing weight and not increasing energy, deduce regular exercise
  have re : RegularExercise Muffin := by
    have ie_or_re : IncreasesEnergy Muffin ∨ RegularExercise Muffin :=
      muffin_lw_imp_ie_or_re lw
    cases ie_or_re with
    | inl hie => exact False.elim (muffin_not_increase_energy hie)
    | inr hre => exact hre
  -- Finally, use strict diet and regular exercise to conclude regain health
  have reg : RegainsHealth Muffin :=
    strict_diet_and_exercise_imp_regain Muffin ⟨muffin_strict_diet, re⟩
  exact reg
