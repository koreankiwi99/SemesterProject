theorem goal : Devout Walter := by
  -- From faces challenges, Walter seeks guidance
  have hseeks_and_looks : SeeksGuidance Walter ∧ LooksForAnswers Walter :=
    faces_imp_seeks_and_looks Walter walter_faces
  have hseeks : SeeksGuidance Walter := And.left hseeks_and_looks
  -- Thus Walter finds inner peace
  have hinter : FindsInnerPeace Walter :=
    walter_believes_or_seeks_imp_innerpeace (Or.inr hseeks)
  -- Inner peace implies (sincere or compassionate), but Walter is neither
  have hsinc_or_comp : LivesSincerely Walter ∨ ActsWithCompassion Walter :=
    innerpeace_imp_sincere_or_compassion Walter hinter
  have contra : False := by
    cases hsinc_or_comp with
    | inl hsinc =>
        exact walter_not_sincere hsinc
    | inr hcomp =>
        exact walter_not_compassion hcomp
  exact False.elim contra
