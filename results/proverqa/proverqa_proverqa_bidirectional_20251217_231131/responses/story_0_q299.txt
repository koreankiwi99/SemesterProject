Premises: Faith has good energy or a joyful personality. Everyone who dedicates themselves to their craft either has a means of creative expression or inspires others (or both). Everyone who is passionate about music has a creative expression or a joyful personality (or both). Faith values community. Faith appreciates beauty. For all people, if they volunteer frequently, then they value community and have strong connections. Faith does not have an uplifting influence. Faith has an outgoing nature. If Faith has good energy, then she is an active listener and cares for others. Anyone with a joyful personality who cares for others will spread happiness. If Faith has a vision, then she is inspirational and encourages others. If Faith is either passionate about art or passionate about music (but not both), then she is inspirational. Everyone who has an outgoing nature either volunteers frequently or cares for others (or both). Anyone who appreciates beauty has good taste. If Faith either helps animals or donates to charity (but not both), then she volunteers frequently. If someone is inspirational, then they have an uplifting influence. Faith does not spread happiness. Faith is imaginative. If Faith has creative expression, then she either has good taste or is imaginative (but not both). Faith does not have strong connections. Anyone who values aesthetics or appreciates design has good taste.
Conclusion: Faith is not passionate about art.

open Classical

axiom Person : Type
axiom Faith : Person

-- Predicates
axiom GoodEnergy : Person → Prop
axiom JoyfulPersonality : Person → Prop

axiom PassionateMusic : Person → Prop

axiom Imaginative : Person → Prop

-- Premises
axiom A1 : GoodEnergy Faith ∨ JoyfulPersonality Faith
axiom A2 : ∀ x, DedicatesToCraft x → (CreativeExpression x ∨ InspiresOthers x)
axiom A3 : ∀ x, PassionateMusic x → (CreativeExpression x ∨ JoyfulPersonality x)
axiom A4 : ValuesCommunity Faith
axiom A5 : AppreciatesBeauty Faith
axiom A6 : ∀ x, VolunteerFrequently x → (ValuesCommunity x ∧ StrongConnections x)
axiom A7 : ¬ UpliftingInfluence Faith

axiom A10 : ∀ x, (JoyfulPersonality x ∧ CaresForOthers x) → SpreadHappiness x
axiom A11 : Vision Faith → (Inspirational Faith ∧ EncouragesOthers Faith)
axiom A12 : ((PassionateArt Faith ∨ PassionateMusic Faith) ∧ ¬(PassionateArt Faith ∧ PassionateMusic Faith)) → Inspirational Faith
axiom A13 : ∀ x, OutgoingNature x → (VolunteerFrequently x ∨ CaresForOthers x)
axiom A14 : ∀ x, AppreciatesBeauty x → GoodTaste x
axiom A15 : ((HelpsAnimals Faith ∨ DonatesToCharity Faith) ∧ ¬(HelpsAnimals Faith ∧ DonatesToCharity Faith)) → VolunteerFrequently Faith
axiom A16 : ∀ x, Inspirational x → UpliftingInfluence x
axiom A17 : ¬ SpreadHappiness Faith
axiom A18 : Imaginative Faith
axiom A19 : CreativeExpression Faith → ((GoodTaste Faith ∨ Imaginative Faith) ∧ ¬(GoodTaste Faith ∧ Imaginative Faith))
axiom A20 : ¬ StrongConnections Faith
axiom A21 : ∀ x, (ValuesAesthetics x ∨ AppreciatesDesign x) → GoodTaste x

theorem goal : ¬ PassionateArt Faith := by
  -- From outgoing nature, either volunteers frequently or cares for others
  have h_out : VolunteerFrequently Faith ∨ CaresForOthers Faith :=
    (A13 Faith) A8
  -- Volunteering implies strong connections, but Faith does not have strong connections
  have h_not_vol : ¬ VolunteerFrequently Faith := by
    intro hvol
    have hvc_sc := (A6 Faith) hvol
    have hsc : StrongConnections Faith := hvc_sc.right
    exact A20 hsc
  -- Therefore, Faith cares for others
  have h_cares : CaresForOthers Faith := by
    cases h_out with
    | inl hvol => exact False.elim (h_not_vol hvol)
    | inr hcare => exact hcare
  -- From "joyful ∧ cares → spreads happiness" and "¬ spreads happiness", get ¬ joyful
  have h_not_joy : ¬ JoyfulPersonality Faith := by
    intro hjoy
    have hspread : SpreadHappiness Faith := (A10 Faith) ⟨hjoy, h_cares⟩
    exact A17 hspread
  -- From "good energy ∨ joyful" and ¬ joyful, get good energy
  have h_good : GoodEnergy Faith := by
    cases A1 with
    | inl hge => exact hge
    | inr hjoy => exact False.elim (h_not_joy hjoy)
  -- Also get cares for others from good energy (consistent)
  have _h_cares2 : CaresForOthers Faith := (A9 h_good).right
  -- Faith has good taste (appreciates beauty)
  have h_taste : GoodTaste Faith := (A14 Faith) A5
  -- Faith is imaginative
  have h_imag : Imaginative Faith := A18
  -- Therefore, Faith cannot have creative expression (would force XOR of taste and imaginative)
  have h_not_ce : ¬ CreativeExpression Faith := by
    intro hce
    have hx := A19 hce
    -- hx.right is ¬(GoodTaste ∧ Imaginative), but both hold
    exact hx.right ⟨h_taste, h_imag⟩
  -- Hence Faith is not passionate about music (would imply CE or Joyful, both impossible)
  have h_not_pm : ¬ PassionateMusic Faith := by
    intro hpm
    have disj := (A3 Faith) hpm
    cases disj with
    | inl hce => exact h_not_ce hce
    | inr hjoy => exact h_not_joy hjoy
  -- Now show ¬ passionate about art: assuming it leads to uplifting influence contradiction
  intro hpa
  have hxora : (PassionateArt Faith ∨ PassionateMusic Faith) ∧ ¬(PassionateArt Faith ∧ PassionateMusic Faith) := by
    refine And.intro (Or.inl hpa) ?_
    intro hboth
    exact h_not_pm hboth.right
  have hins : Inspirational Faith := A12 hxora
  have h_up : UpliftingInfluence Faith := (A16 Faith) hins
  exact A7 h_up

Verification Success: True
