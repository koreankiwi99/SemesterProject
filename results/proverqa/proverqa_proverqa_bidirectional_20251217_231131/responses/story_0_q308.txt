axiom A20 : PrefersAnonymity Lawson ∨ CravesSpotlight Lawson

-- Conclusion: Lawson is not a talented writer.
theorem conclusion : ¬ TalentedWriter Lawson := by
  -- From A14 and A15, XOR in A1 cannot hold
  have hW : DesiresWealth Lawson := A14
  have hV : ValuesPrivacy Lawson := A15
  have hnotXOR : ¬ ((DesiresWealth Lawson ∧ ¬ ValuesPrivacy Lawson) ∨ (ValuesPrivacy Lawson ∧ ¬ DesiresWealth Lawson)) := by
    intro h
    cases h with
    | inl h1 =>
      exact h1.right hV
    | inr h2 =>
      exact h2.right hW
  -- Hence Lawson does not seek recognition
  have notSR : ¬ SeeksRecognition Lawson := by
    intro hSR
    exact hnotXOR (A1 hSR)
  -- From XOR between seeking recognition and eschewing fame (A10), deduce eschews fame
  have hEF : EschewsFame Lawson := by
    cases A10 with
    | inl h =>
      exact False.elim (notSR h.left)
    | inr h =>
      exact h.left
  -- Anyone who ghostwrites or eschews fame works behind the scenes (A11)
  have hWBS : WorksBehindScenes Lawson := by
    have : GhostwritesBooks Lawson ∨ EschewsFame Lawson := Or.inr hEF
    exact A11 Lawson this
  -- If Lawson were a talented writer and works behind the scenes, he would be influential (A9), contradicting A12
  intro hTW
  have hInf : Influential Lawson := A9 ⟨hTW, hWBS⟩
  exact A12 hInf
