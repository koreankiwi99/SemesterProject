theorem goal : EnjoysCool Dimples := by
  -- Dimples feels hot because she exercises outside
  have hHot : FeelsHot Dimples := by
    have : ExercisesOutside Dimples ∨ LivesDesert Dimples := Or.inl p14
    exact p9 _ this
  -- From ¬StaysCool and (AvoidsSunlight ∧ FeelsHot) → StaysCool, we get ¬AvoidsSunlight
  have hNotAvoid : ¬ AvoidsSunlight Dimples := by
    intro hA
    have hCool : StaysCool Dimples := p2 _ ⟨hA, hHot⟩
    exact p6 hCool
  -- From SeeksShade → NeedsRelaxation and ¬NeedsRelaxation, we get ¬SeeksShade
  have hNotSeek : ¬ SeeksShade Dimples := by
    intro hSeek
    have hNeed : NeedsRelaxation Dimples := (p15 _ hSeek).right
    exact p8 hNeed
  -- If Dimples preferred warm, given ¬AvoidsSunlight, she'd seek shade (by the XOR rule), contradiction
  have hNotPref : ¬ PrefersWarm Dimples := by
    intro hPref
    have hx : ((AvoidsSunlight Dimples ∨ PrefersWarm Dimples) ∧ ¬(AvoidsSunlight Dimples ∧ PrefersWarm Dimples)) := by
      have hor : AvoidsSunlight Dimples ∨ PrefersWarm Dimples := Or.inr hPref
      have hnotand : ¬(AvoidsSunlight Dimples ∧ PrefersWarm Dimples) := by
        intro hBoth
        exact hNotAvoid hBoth.left
      exact And.intro hor hnotand
    have hSeek : SeeksShade Dimples := p12 _ hx
    exact hNotSeek hSeek
  -- From the XOR of EnjoysCool vs PrefersWarm for Dimples and ¬PrefersWarm, conclude EnjoysCool
  cases p7a with
  | inl hEnjoy => exact hEnjoy
  | inr hPref  => exact False.elim (hNotPref hPref)
