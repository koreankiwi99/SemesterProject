-- 4. If Araya has a creative passion or a stable job (or both), then she leads a balanced life.
axiom P4 : (CreativePassion Araya ∨ StableJob Araya) → LeadsBalancedLife Araya

-- 5. Everyone makes plans or sets goals.

-- 6. Araya does not make plans.

-- 7. Anyone who has ambition either works hard or sets goals (or both).

-- 8. Araya follows her intuition.

-- 9. Araya either values practicality or embraces spontaneity, but not both.

-- 10. If Araya is motivated, then she is either pursuing hobbies or has a stable job, but not both.
axiom P10 : Motivated Araya → ((PursuingHobbies Araya ∨ StableJob Araya) ∧ ¬ (PursuingHobbies Araya ∧ StableJob Araya))

-- 11. Anyone who follows their intuition and enjoys exploration can embrace spontaneity.

-- 12. Araya is pursuing hobbies.
axiom P12 : PursuingHobbies Araya

-- 13. Araya enjoys exploration.
axiom P13 : EnjoysExploration Araya

-- From the premises, we can actually derive that Araya leads a balanced life.
theorem balanced : LeadsBalancedLife Araya := by
  -- From P5 and P6, Araya must set goals.
  have h_or := P5 Araya
  have h_sets : SetsGoals Araya := by
    cases h_or with
    | inl h_mp =>
      exact False.elim (P6 h_mp)
    | inr h_sg =>
      exact h_sg
  -- Hence Araya is motivated.
  have h_mot : Motivated Araya := P1 Araya h_sets
  -- From P10 and P12, Araya cannot have a stable job.

  have h_no_sj : ¬ StableJob Araya := by

    exact h_not_both ⟨P12, h_sj⟩
  -- From P11, P8, and P13, Araya embraces spontaneity.

  -- From P9_not_both and h_es, Araya does not value practicality.
  have h_not_vp : ¬ ValuesPracticality Araya := by

    | inl h_vp =>
      exact False.elim (h_not_vp h_vp)
    | inr h_cp =>
      exact h_cp
  -- Therefore, by P4, Araya leads a balanced life.
