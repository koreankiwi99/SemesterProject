Premises: Dean does not mentor students. Dean contributes to society. If Dean makes an impact and has expertise, then he receives awards. Dean is not an active volunteer. If Dean is either an expert ornithologist or a passionate conservationist (but not both), then he is a respected scientist. Dean either holds degrees or receives awards. If Dean either mentors students or inspires others (but not both), then he makes an impact. If Dean contributes to society, then he either receives awards or gains recognition (but not both). Dean either has expertise or works hard. Dean gains recognition. Dean does not publish research. Anyone who publishes research or is a respected scientist can inspire others. Dean does not work hard. If Dean is a passionate conservationist, then he is both an active volunteer and public speaker.
Conclusion: Dean is not an expert ornithologist.

axiom Person : Type

axiom Contributes : Person → Prop

axiom Awards : Person → Prop
axiom Active : Person → Prop
axiom Ornithologist : Person → Prop
axiom Conservationist : Person → Prop
axiom Respected : Person → Prop
axiom Degrees : Person → Prop
axiom Inspires : Person → Prop
axiom Recognition : Person → Prop
axiom Publishes : Person → Prop
axiom WorksHard : Person → Prop
axiom PublicSpeaker : Person → Prop

axiom axNotMentor : ¬ Mentors Dean
axiom axContributes : Contributes Dean
axiom axImpactExpertiseAwards : (Impact Dean ∧ Expertise Dean) → Awards Dean
axiom axNotActive : ¬ Active Dean
axiom axXorOrnConsRespected :
  ((Ornithologist Dean ∨ Conservationist Dean) ∧ ¬ (Ornithologist Dean ∧ Conservationist Dean)) → Respected Dean
axiom axDegreesOrAwards : Degrees Dean ∨ Awards Dean
axiom axXorMentorInspImpact :
  ((Mentors Dean ∨ Inspires Dean) ∧ ¬ (Mentors Dean ∧ Inspires Dean)) → Impact Dean
axiom axContribXorAwardsRecognition :
  Contributes Dean → ((Awards Dean ∨ Recognition Dean) ∧ ¬ (Awards Dean ∧ Recognition Dean))
axiom axExpertiseOrWork : Expertise Dean ∨ WorksHard Dean
axiom axRecognition : Recognition Dean
axiom axNotPublish : ¬ Publishes Dean
axiom axPublishOrRespectedInspireAll :
  ∀ x : Person, (Publishes x ∨ Respected x) → Inspires x
axiom axNotWorksHard : ¬ WorksHard Dean
axiom axConsActivePublic :
  Conservationist Dean → (Active Dean ∧ PublicSpeaker Dean)

theorem goal : ¬ Ornithologist Dean := by
  intro hOrn
  -- Deduce ¬ Conservationist Dean
  have hNotCons : ¬ Conservationist Dean := by
    intro hCons
    have hApr := axConsActivePublic hCons
    have hActive : Active Dean := And.left hApr
    exact axNotActive hActive
  -- From Orn and ¬Cons, get respected via XOR antecedent
  have hRespected : Respected Dean := by
    have hxor : ((Ornithologist Dean ∨ Conservationist Dean) ∧ ¬ (Ornithologist Dean ∧ Conservationist Dean)) := by
      constructor
      · exact Or.inl hOrn
      · intro hAnd
        have hCons : Conservationist Dean := And.right hAnd
        exact hNotCons hCons
    exact axXorOrnConsRespected hxor
  -- From respected, get inspires
  have hInspires : Inspires Dean := by
    have hImp := axPublishOrRespectedInspireAll Dean
    have hDisj : Publishes Dean ∨ Respected Dean := Or.inr hRespected
    exact hImp hDisj
  -- From ¬Mentor and inspires, get XOR condition and then impact
  have hXorMentInsp : ((Mentors Dean ∨ Inspires Dean) ∧ ¬ (Mentors Dean ∧ Inspires Dean)) := by
    constructor
    · exact Or.inr hInspires
    · intro hBoth
      have hMent : Mentors Dean := And.left hBoth
      exact axNotMentor hMent
  have hImpact : Impact Dean := axXorMentorInspImpact hXorMentInsp
  -- From expertise or workhard and not workhard, get expertise
  have hExpertise : Expertise Dean := by
    cases axExpertiseOrWork with
    | inl hE => exact hE
    | inr hW =>
        exact False.elim (axNotWorksHard hW)
  -- Hence awards
  have hAwards : Awards Dean := axImpactExpertiseAwards ⟨hImpact, hExpertise⟩
  -- But contributes implies awards xor recognition, and recognition holds, so not awards
  have hXorA := axContribXorAwardsRecognition axContributes
  have hNotBoth : ¬ (Awards Dean ∧ Recognition Dean) := And.right hXorA
  have hNoAwards : ¬ Awards Dean := by
    intro hA
    exact hNotBoth ⟨hA, axRecognition⟩
  exact hNoAwards hAwards
