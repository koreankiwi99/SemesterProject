-- Conclusion: Rebekah does not study butterflies.
theorem goal : ¬ StudiesButterflies Rebekah := by
  -- First, show Rebekah does not study moths.
  have notMoths : ¬ StudiesMoths Rebekah := by
    intro hM
    have h := ax17 hM
    have bothBW : SpecializesInBees Rebekah ∧ StudiesWasps Rebekah := And.intro ax3 ax9
    exact h.right bothBW
  -- Assume she studies butterflies and derive a contradiction.
  intro hB
  -- From (butterflies XOR moths), deduce success.
  have xorBM :
      (StudiesButterflies Rebekah ∨ StudiesMoths Rebekah) ∧
      ¬(StudiesButterflies Rebekah ∧ StudiesMoths Rebekah) := by
    refine And.intro (Or.inl hB) ?_
    intro hBoth
    exact notMoths hBoth.right
  have hSucc : Successful Rebekah := ax23 xorBM
  -- Success implies breakthroughs.
  have hBr : MakesBreakthroughs Rebekah := ax4 (Or.inl hSucc)
  -- Breakthroughs XOR not publishing gives advances field.
  have xorBP :
      (MakesBreakthroughs Rebekah ∨ PublishesPapers Rebekah) ∧
      ¬(MakesBreakthroughs Rebekah ∧ PublishesPapers Rebekah) := by
    refine And.intro (Or.inl hBr) ?_
    intro hBoth
    exact ax12 hBoth.right
  have hAdv : AdvancesField Rebekah := ax10 xorBP
  -- From advancing the field and being an entomologist, deduce dedicated.
  have hDedInsp : DedicatedResearcher Rebekah ∧ InspiresOthers Rebekah :=
    ax11 Rebekah (And.intro ax24 hAdv)
  have hDed : DedicatedResearcher Rebekah := hDedInsp.left
  -- From the XOR of (dedicated, pursues), get not pursuing independent projects.
  have notPursues : ¬ PursuesIndependentProjects Rebekah := by
    intro hP
    exact ax1.right (And.intro hDed hP)
  -- From investigating ecosystems and not pursuing, get the XOR and thus innovative.
  have xorIP :
      (InvestigatesEcosystems Rebekah ∨ PursuesIndependentProjects Rebekah) ∧
      ¬(InvestigatesEcosystems Rebekah ∧ PursuesIndependentProjects Rebekah) := by
    refine And.intro (Or.inl ax14) ?_
    intro hBoth
    exact notPursues hBoth.right
  have hInv : Innovative Rebekah := ax8 xorIP
  -- Innovative entomologist must explore or develop, contradicting given negations.
  have hExplOrDev : ExploresRemoteAreas Rebekah ∨ DevelopsNewTheories Rebekah :=
    ax6 Rebekah (And.intro ax24 hInv)
  cases hExplOrDev with
  | inl hE => exact ax7 hE
  | inr hD => exact ax20 hD
