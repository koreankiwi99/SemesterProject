    exact hNotBoth hBoth

    have hOr := And.left hSG_MD
    cases hOr with
    | inl hsg => exact False.elim (hNotSG hsg)
    | inr hmd => exact hmd

  -- Using ¬Motivated and the implication, deduce FacesChallenges (classically)

      ¬((MeetsDeadlines Isabela ∨ FacesChallenges Isabela) ∧
        ¬(MeetsDeadlines Isabela ∧ FacesChallenges Isabela)) := by
    intro h
    have hMot := md_fc_xor_implies_motivated h
    exact not_motivated_isabela hMot
  have hFC : FacesChallenges Isabela := by
    by_contra hNotFC
    have hOr : MeetsDeadlines Isabela ∨ FacesChallenges Isabela := Or.inl hMD
    have hNotBoth : ¬(MeetsDeadlines Isabela ∧ FacesChallenges Isabela) := by
      intro hf
      exact hNotFC (And.right hf)
    have hConj :
        (MeetsDeadlines Isabela ∨ FacesChallenges Isabela) ∧
        ¬(MeetsDeadlines Isabela ∧ FacesChallenges Isabela) := And.intro hOr hNotBoth
    exact hNotXor_md_fc hConj

  -- From EncountersProblems, get (HO ∨ SF) ∧ ¬(HO ∧ SF), and separately show ¬SF, hence HO

  have h := faces_and_helps_implies_brave Isabela
