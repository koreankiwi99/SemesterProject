-- Premises

-- If Addison either publishes research or leads workshops (but not both), then she is influential.
axiom prem_addison_xor_influential :
  ((PublishesResearch Addison ∧ ¬ LeadsWorkshops Addison) ∨
   (LeadsWorkshops Addison ∧ ¬ PublishesResearch Addison)) → Influential Addison

-- If Faye either studied fungi or discovered a new species (but not both), then she is renowned.
axiom prem_faye_xor_study_discover_renowned :
  ((StudiedFungi Faye ∧ ¬ DiscoveredNewSpecies Faye) ∨
   (DiscoveredNewSpecies Faye ∧ ¬ StudiedFungi Faye)) → Renowned Faye

-- If Faye is influential, then she collaborates with others and mentors students.
axiom prem_faye_influential_imp :
  Influential Faye → (Collaborates Faye ∧ MentorsStudents Faye)

-- If Muffinhead either studied fungi or discovered a new species (but not both), then she is renowned.
axiom prem_muffin_xor_study_discover_renowned :
  ((StudiedFungi Muffinhead ∧ ¬ DiscoveredNewSpecies Muffinhead) ∨
   (DiscoveredNewSpecies Muffinhead ∧ ¬ StudiedFungi Muffinhead)) → Renowned Muffinhead

-- If a mycologist receives awards, then they are respected.
axiom prem_mycologist_awards_respected :
  ∀ x : Person, (Mycologist x ∧ ReceivesAwards x) → Respected x

-- Faye leads workshops.
axiom prem_faye_leads : LeadsWorkshops Faye

-- Faye is not respected.
axiom prem_faye_not_respected : ¬ Respected Faye

-- Addison leads workshops.
axiom prem_addison_leads : LeadsWorkshops Addison

-- All mycologists who collaborate with others develop new techniques and share knowledge.
axiom prem_mycologist_collab_dev_share :
  ∀ x : Person, (Mycologist x ∧ Collaborates x) → (DevelopsNewTechniques x ∧ SharesKnowledge x)

-- Faye shares knowledge.
axiom prem_faye_shares : SharesKnowledge Faye

-- Every mycologist who contributes to the field is either influential or advances scientific knowledge (or both).
axiom prem_mycologist_contributes_imp :
  ∀ x : Person, (Mycologist x ∧ ContributesToField x) → (Influential x ∨ AdvancesScientificKnowledge x)

-- If Faye has passion, then she is renowned and makes an impact.
axiom prem_faye_passion_imp : HasPassion Faye → (Renowned Faye ∧ MakesImpact Faye)

-- Muffinhead is not respected.
axiom prem_muffin_not_respected : ¬ Respected Muffinhead

-- Faye did not discover a new species.
axiom prem_faye_not_discovered : ¬ DiscoveredNewSpecies Faye

-- Addison shares knowledge.
axiom prem_addison_shares : SharesKnowledge Addison

-- Faye does not develop new techniques.
axiom prem_faye_not_develops : ¬ DevelopsNewTechniques Faye

-- If Faye either publishes research or leads workshops (but not both), then she is influential.
axiom prem_faye_xor_publish_lead_influential :
  ((PublishesResearch Faye ∧ ¬ LeadsWorkshops Faye) ∨
   (LeadsWorkshops Faye ∧ ¬ PublishesResearch Faye)) → Influential Faye

-- If a mycologist publishes research and is renowned, then they receive awards.
axiom prem_mycologist_publish_renowned_awards :
  ∀ x : Person, (Mycologist x ∧ PublishesResearch x ∧ Renowned x) → ReceivesAwards x

-- Conclusion: Faye did not study fungi.

  sorry
