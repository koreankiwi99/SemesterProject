theorem goal :
  (StingsWhenThreatened Sweetsie ∧ ¬ NeverStings Sweetsie) ∨
  (NeverStings Sweetsie ∧ ¬ StingsWhenThreatened Sweetsie) := by

  have hProtect : ProtectsHive Sweetsie := by
    exact p22 Sweetsie ⟨p14, p1⟩

    exact p13 Sweetsie hProtect
  -- Construct the XOR: choose the left side (stings when threatened and not never-stings)
  apply Or.inl
  apply And.intro
  · exact hSting
  · unfold NeverStings
    intro hnever
    exact hnever hSting
