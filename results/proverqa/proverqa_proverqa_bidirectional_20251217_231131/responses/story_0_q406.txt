-- Goal: Alana loves adventure.
theorem goal : LovesAdventure Alana := by
  -- From exploring new places to taking calculated risks
  have h_risk : TakesCalculatedRisks Alana := premise_explores_to_risk premise_explores
  -- From making own decisions and taking risks to valuing independence
  have h_vi : ValuesIndependence Alana :=
    premise_mod_risk_to_vi Alana ⟨premise_makes_own, h_risk⟩
  -- From creative nature and not guided by reason to trusting intuition
  have h_ti_or_gr : GuidedByReason Alana ∨ TrustsIntuition Alana :=
    premise_creative_xor_reason_ti_disj premise_creative
  have h_ti : TrustsIntuition Alana := by
    cases h_ti_or_gr with
    | inl h_gr =>
        exact False.elim (premise_not_guided h_gr)
    | inr h_ti => exact h_ti
  -- From spontaneous and trusting intuition to embracing uncertainty
  have h_eu : EmbracesUncertainty Alana :=
    premise_s_ti_to_eu Alana ⟨premise_spontaneous, h_ti⟩
  -- From valuing independence and embracing uncertainty to being free-spirited
  have h_fs : FreeSpirited Alana := premise_vi_eu_to_fs Alana ⟨h_vi, h_eu⟩
  -- Free-spirited implies loves adventure or pursues creative passions
  have h_la_or_pcp : LovesAdventure Alana ∨ PursuesCreativePassions Alana :=
    premise_fs_to_la_or_pcp Alana h_fs
  -- From exclusive-or with AppreciatesSimpleThings true, get ¬PCP
  have h_ast_and_not_pcp : AppreciatesSimpleThings Alana ∧ ¬ PursuesCreativePassions Alana := by
    cases premise_pcp_xor_ast with
    | inl h =>
        -- h: PCP ∧ ¬AST, contradicts AST true
        exact False.elim (h.right premise_appreciates)
    | inr h =>
        -- h: AST ∧ ¬PCP
        exact h
  have h_not_pcp : ¬ PursuesCreativePassions Alana := h_ast_and_not_pcp.right
  -- Disjunctive elimination to conclude LovesAdventure
  cases h_la_or_pcp with
  | inl h_la => exact h_la
  | inr h_pcp => exact False.elim (h_not_pcp h_pcp)
