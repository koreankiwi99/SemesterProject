axiom Person : Type

axiom ExaminesCropPathogens : Person → Prop
axiom CollaboratesWithPeers : Person → Prop
axiom ConductsLabResearch : Person → Prop
axiom Innovative : Person → Prop

-- Premises
-- 1. Exclusive or between SpecializesMicrobiology and EnvironmentallyConscious for Elianna
axiom p1 :
  (SpecializesMicrobiology Elianna ∧ ¬EnvironmentallyConscious Elianna) ∨
  (EnvironmentallyConscious Elianna ∧ ¬SpecializesMicrobiology Elianna)

-- 2. If Rosalie conducts lab research, then she has a PhD and collaborates with peers.
axiom p2 : ConductsLabResearch Rosalie → (HasPhD Rosalie ∧ CollaboratesWithPeers Rosalie)

-- 3. Elianna either cares about nature or is environmentally conscious.
axiom p3 : CaresAboutNature Elianna ∨ EnvironmentallyConscious Elianna

-- 4. If Elianna investigates microorganisms or analyzes plant samples, then she can examine fungal behavior.
axiom p4 :
  (InvestigatesMicroorganisms Elianna ∨ AnalyzesPlantSamples Elianna) →
  ExaminesFungalBehavior Elianna

-- 5. If Elianna has a PhD, then she either publishes academic papers or develops new treatments, but not both.
axiom p5 :
  HasPhD Elianna →
  ((PublishesAcademicPapers Elianna ∧ ¬DevelopsNewTreatments Elianna) ∨
   (DevelopsNewTreatments Elianna ∧ ¬PublishesAcademicPapers Elianna))

-- 6. If Elianna researches plant diseases, then she either examines fungal behavior or studies water molds (but not both).
axiom p6 :
  ResearchesPlantDiseases Elianna →
  ((ExaminesFungalBehavior Elianna ∧ ¬StudiesWaterMolds Elianna) ∨
   (StudiesWaterMolds Elianna ∧ ¬ExaminesFungalBehavior Elianna))

-- 7. Elianna does not publish academic papers.
axiom p7 : ¬PublishesAcademicPapers Elianna

-- 8. If Elianna studies water molds, then she specializes in microbiology.
axiom p8 : StudiesWaterMolds Elianna → SpecializesMicrobiology Elianna

-- 9. If Elianna investigates downy mildew or examines crop pathogens, then she studies water molds.
axiom p9 :
  (InvestigatesDownyMildew Elianna ∨ ExaminesCropPathogens Elianna) →
  StudiesWaterMolds Elianna

-- 10. If Rosalie has a PhD, then she either publishes academic papers or develops new treatments, but not both.
axiom p10 :
  HasPhD Rosalie →
  ((PublishesAcademicPapers Rosalie ∧ ¬DevelopsNewTreatments Rosalie) ∨
   (DevelopsNewTreatments Rosalie ∧ ¬PublishesAcademicPapers Rosalie))

-- 11. If Elianna conducts lab research, then she has a PhD and collaborates with peers.
axiom p11 : ConductsLabResearch Elianna → (HasPhD Elianna ∧ CollaboratesWithPeers Elianna)

-- 12. Elianna conducts lab research.
axiom p12 : ConductsLabResearch Elianna

-- 13. Elianna either develops new treatments or examines fungal behavior, but not both.
axiom p13 :
  (DevelopsNewTreatments Elianna ∧ ¬ExaminesFungalBehavior Elianna) ∨
  (ExaminesFungalBehavior Elianna ∧ ¬DevelopsNewTreatments Elianna)

-- 14. Elianna is either environmentally conscious or innovative (or both).
axiom p14 : EnvironmentallyConscious Elianna ∨ Innovative Elianna

-- 15. Elianna researches plant diseases.
axiom p15 : ResearchesPlantDiseases Elianna

-- Conclusion: Elianna is innovative.
theorem conclusion : Innovative Elianna := by
  -- From lab research, Elianna has a PhD.
  have hPhdCollab := p11 p12
  have hPhd : HasPhD Elianna := hPhdCollab.left

  -- From PhD and not publishing, deduce she develops new treatments.
  have hDev : DevelopsNewTreatments Elianna := by
    have hx := p5 hPhd
    cases hx with
    | inl h =>
        have hPub : PublishesAcademicPapers Elianna := h.left
        exact False.elim (p7 hPub)
    | inr h =>
        exact h.left

  -- From xor between Dev and EFB, and Dev true, get ¬EFB.
  have hNotEFB : ¬ExaminesFungalBehavior Elianna := by
    cases p13 with
    | inl h => exact h.right
    | inr h =>
        have contra : False := h.right hDev
        exact fun _ => False.elim contra

  -- From plant diseases xor (EFB, SWM) and ¬EFB, get SWM.
  have hSWM : StudiesWaterMolds Elianna := by
    have hx := p6 p15
    cases hx with
    | inl h =>
        have : False := hNotEFB h.left
        exact False.elim this
    | inr h => exact h.left

  -- From SWM, get specializes in microbiology.
  have hMS : SpecializesMicrobiology Elianna := p8 hSWM

  -- From xor between MS and EC and MS true, get ¬EC.
  have hNotEC : ¬EnvironmentallyConscious Elianna := by
    cases p1 with
    | inl h => exact h.right
    | inr h =>
        have contra : False := h.right hMS
        exact fun _ => False.elim contra

  -- From EC ∨ Innov and ¬EC, conclude Innov.
  cases p14 with
  | inl hEC => exact False.elim (hNotEC hEC)
  | inr hInnov => exact hInnov
