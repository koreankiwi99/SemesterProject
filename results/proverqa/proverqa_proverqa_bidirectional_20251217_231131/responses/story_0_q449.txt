Premises: If Kevin has a PhD, then he either collects data alongside rivers or writes books (but not both). For all humans, if they study marine life or love birds (or both), then they care about conservation. Kevin writes books. If someone either conducts field research or collects data alongside rivers (but not both), then they work outdoors. Kevin conducts field research. If Kevin works outdoors, then he either studies marine life or has scientific interests (or both). Kevin either pursues artistic passions or has scientific interests, but not both. Kevin pursues artistic passions. Kevin has a PhD.
Conclusion: Kevin cares about conservation.

Lean Code:
axiom Person : Type
axiom Kevin : Person

axiom HasPhD : Person → Prop
axiom CollectsDataRivers : Person → Prop
axiom WritesBooks : Person → Prop
axiom ConductsFieldResearch : Person → Prop
axiom WorksOutdoors : Person → Prop
axiom StudiesMarineLife : Person → Prop
axiom ScientificInterests : Person → Prop
axiom PursuesArtisticPassions : Person → Prop
axiom LovesBirds : Person → Prop
axiom CaresConservation : Person → Prop

-- If Kevin has a PhD, then he either collects data alongside rivers or writes books (but not both).
axiom phd_xor_river_or_books :
  HasPhD Kevin →
  ((CollectsDataRivers Kevin ∧ ¬ WritesBooks Kevin) ∨ (WritesBooks Kevin ∧ ¬ CollectsDataRivers Kevin))

-- For all persons (humans), if they study marine life or love birds (or both), then they care about conservation.
axiom studies_or_birds_care :
  ∀ p : Person, (StudiesMarineLife p ∨ LovesBirds p) → CaresConservation p

-- Kevin writes books.
axiom kevin_writes : WritesBooks Kevin

-- If someone either conducts field research or collects data alongside rivers (but not both), then they work outdoors.
axiom xor_research_river_outdoors :
  ∀ p : Person,
    (((ConductsFieldResearch p ∧ ¬ CollectsDataRivers p) ∨ (CollectsDataRivers p ∧ ¬ ConductsFieldResearch p)) → WorksOutdoors p)

-- Kevin conducts field research.
axiom kevin_conducts_field : ConductsFieldResearch Kevin

-- If Kevin works outdoors, then he either studies marine life or has scientific interests (or both).
axiom outdoors_implies_sml_or_si :
  WorksOutdoors Kevin → (StudiesMarineLife Kevin ∨ ScientificInterests Kevin)

-- Kevin either pursues artistic passions or has scientific interests, but not both.
axiom arts_or_sci_xor :
  (PursuesArtisticPassions Kevin ∧ ¬ ScientificInterests Kevin) ∨ (ScientificInterests Kevin ∧ ¬ PursuesArtisticPassions Kevin)

-- Kevin pursues artistic passions.
axiom kevin_pursues_art : PursuesArtisticPassions Kevin

-- Kevin has a PhD.
axiom kevin_has_phd : HasPhD Kevin

-- Conclusion: Kevin cares about conservation.
theorem goal : CaresConservation Kevin := by
  -- From PhD XOR (rivers vs books) and "Kevin writes books", deduce ¬ collects data alongside rivers.
  have hx : ((CollectsDataRivers Kevin ∧ ¬ WritesBooks Kevin) ∨ (WritesBooks Kevin ∧ ¬ CollectsDataRivers Kevin)) :=
    phd_xor_river_or_books kevin_has_phd
  have hnotCDR : ¬ CollectsDataRivers Kevin := by
    cases hx with
    | inl h =>
      -- h.right : ¬ WritesBooks Kevin contradicts kevin_writes; from False, derive ¬ CollectsDataRivers Kevin.
      exact fun _ => False.elim (h.right kevin_writes)
    | inr h =>
      exact h.right

  -- Use XOR (conducts vs rivers) with conducts true and ¬ rivers to deduce works outdoors.
  have hxor2 :
    ((ConductsFieldResearch Kevin ∧ ¬ CollectsDataRivers Kevin) ∨ (CollectsDataRivers Kevin ∧ ¬ ConductsFieldResearch Kevin)) :=
    Or.inl ⟨kevin_conducts_field, hnotCDR⟩
  have hWO : WorksOutdoors Kevin := xor_research_river_outdoors Kevin hxor2

  -- Outdoors implies studies marine life or scientific interests.
  have hSmlOrSi : (StudiesMarineLife Kevin ∨ ScientificInterests Kevin) :=
    outdoors_implies_sml_or_si hWO

  -- From arts XOR sci and "Kevin pursues art", deduce ¬ scientific interests.
  have hnotSI : ¬ ScientificInterests Kevin := by
    cases arts_or_sci_xor with
    | inl h => exact h.right
    | inr h => exact fun _ => h.right kevin_pursues_art

  -- From (studies marine life ∨ scientific interests) and ¬ scientific interests, deduce studies marine life.
  have hSML : StudiesMarineLife Kevin := by
    cases hSmlOrSi with
    | inl hsml => exact hsml
    | inr hsi => exact False.elim (hnotSI hsi)

  -- Universal rule yields cares about conservation from studies marine life.
  exact studies_or_birds_care Kevin (Or.inl hSML)
