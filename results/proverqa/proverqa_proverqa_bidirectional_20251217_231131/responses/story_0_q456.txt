Premises: If someone has charisma and seeks influence, then they can pursue political power. Axel has charisma. Axel promotes justice. Axel either pursues political power or endorses capitalism, but not both. Oaklee promotes justice. For all humans, if they promote justice and value human rights, then they advocate for equality. If Axel hates injustice, then he sympathizes with workers and fights for freedom. If Axel fights injustice or seeks equality, then he supports the proletariat. Axel works in a factory. Axel either endorses capitalism or believes in socialism, but not both. Anyone who works in a factory or lives in poverty can sympathize with workers. Axel seeks influence. Axel lives in poverty. If Axel believes in socialism and supports the proletariat, then he joins the Bolshevik party. Axel values human rights. Anyone who sympathizes with workers and advocates for equality supports the proletariat.
Conclusion: Axel joins the Bolshevik party.

axiom Person : Type
axiom Axel : Person
axiom Oaklee : Person

axiom HasCharisma : Person → Prop
axiom SeeksInfluence : Person → Prop

axiom JoinsBolshevikParty : Person → Prop

-- Premises
axiom charisma_influence_to_power :
  ∀ x : Person, (HasCharisma x ∧ SeeksInfluence x) → PursuesPoliticalPower x
axiom axel_has_charisma : HasCharisma Axel
axiom axel_promotes_justice : PromotesJustice Axel
axiom xor_power_or_capitalism :
  (PursuesPoliticalPower Axel ∨ EndorsesCapitalism Axel) ∧ ¬(PursuesPoliticalPower Axel ∧ EndorsesCapitalism Axel)
axiom oaklee_promotes_justice : PromotesJustice Oaklee
axiom justice_and_rights_to_equality :
  ∀ x : Person, (PromotesJustice x ∧ ValuesHumanRights x) → AdvocatesEquality x
axiom axel_hates_injustice_imp :
  HatesInjustice Axel → (SympathizesWorkers Axel ∧ FightsForFreedom Axel)
axiom axel_fights_or_seeks_to_supports :
  (FightsInjustice Axel ∨ AdvocatesEquality Axel) → SupportsProletariat Axel
axiom axel_works_factory : WorksFactory Axel
axiom xor_capitalism_or_socialism :
  (EndorsesCapitalism Axel ∨ BelievesSocialism Axel) ∧ ¬(EndorsesCapitalism Axel ∧ BelievesSocialism Axel)
axiom factory_or_poverty_to_sympathize :
  ∀ x : Person, (WorksFactory x ∨ LivesInPoverty x) → SympathizesWorkers x
axiom axel_seeks_influence : SeeksInfluence Axel
axiom axel_lives_in_poverty : LivesInPoverty Axel
axiom axel_socialism_and_supports_to_joins :
  (BelievesSocialism Axel ∧ SupportsProletariat Axel) → JoinsBolshevikParty Axel
axiom axel_values_human_rights : ValuesHumanRights Axel
axiom sympathize_and_advocate_to_support :
  ∀ x : Person, (SympathizesWorkers x ∧ AdvocatesEquality x) → SupportsProletariat x

-- Conclusion: Axel joins the Bolshevik party.
theorem goal : JoinsBolshevikParty Axel := by
  -- Axel pursues political power from charisma and seeking influence
  have h_pp : PursuesPoliticalPower Axel :=
    charisma_influence_to_power Axel ⟨axel_has_charisma, axel_seeks_influence⟩
  -- From XOR (capitalism or socialism) and the other XOR, derive that Axel believes in socialism
  have h_cap_or_soc : EndorsesCapitalism Axel ∨ BelievesSocialism Axel :=
    And.left xor_capitalism_or_socialism
  have h_not_pp_and_cap : ¬(PursuesPoliticalPower Axel ∧ EndorsesCapitalism Axel) :=
    And.right xor_power_or_capitalism
  have h_soc : BelievesSocialism Axel := by
    cases h_cap_or_soc with
    | inl h_cap =>
        have h_pp_cap : PursuesPoliticalPower Axel ∧ EndorsesCapitalism Axel := ⟨h_pp, h_cap⟩
        exact False.elim (h_not_pp_and_cap h_pp_cap)
    | inr h_soc => exact h_soc
  -- Axel advocates equality from promoting justice and valuing human rights
  have h_eq : AdvocatesEquality Axel :=
    justice_and_rights_to_equality Axel ⟨axel_promotes_justice, axel_values_human_rights⟩
  -- Axel sympathizes with workers from working in a factory (or living in poverty)
  have h_symp : SympathizesWorkers Axel :=
    factory_or_poverty_to_sympathize Axel (Or.inl axel_works_factory)
  -- Hence Axel supports the proletariat
  have h_supports : SupportsProletariat Axel :=
    sympathize_and_advocate_to_support Axel ⟨h_symp, h_eq⟩
  -- Therefore, Axel joins the Bolshevik party
  exact axel_socialism_and_supports_to_joins ⟨h_soc, h_supports⟩

Verification Success: True
