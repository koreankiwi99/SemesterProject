axiom prem3_nand : ¬ (PursuesPersonalInterests Braden ∧ DiligentlyContributes Braden)

axiom prem4_nand : ¬ (SpecializesEntomology Lilly ∧ StudiesOtherFields Lilly)

axiom prem6_nand : ¬ (SpecializesEntomology Braden ∧ StudiesOtherFields Braden)

axiom prem13_nand : ¬ (PursuesPersonalInterests Rio ∧ DiligentlyContributes Rio)

-- Conclusion: Braden specializes in entomology.

    intro hsof
    exact prem9_nand (And.intro hsof hdeg)
  -- From XOR between SpecializesEntomology and StudiesOtherFields, and not StudiesOtherFields, deduce SpecializesEntomology.
