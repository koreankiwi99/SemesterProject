Premises: Sky does not have natural talent. If Sky is an artist, then she either tells a story or evokes emotion, but not both. Sky evokes emotion. If Sky either expresses emotion or connects with the audience (but not both), then she builds a connection. Sky does not achieve success. If someone performs with elegance and connects with the audience, then they inspire others. If Sky has passion, then she tells a story and speaks from her heart. If Sky inspires others, then she will achieve success. Sky cannot touch the hearts of others. For all humans, if they are trained in ballet or have natural talent, then they perform with elegance. Sky is an artist. Anyone who either tells a story or builds a connection can touch the hearts of others. Sky expresses emotion.
Conclusion: Sky is not trained in ballet.

axiom Person : Type
axiom Sky : Person

axiom NaturalTalent : Person → Prop
axiom Artist : Person → Prop
axiom TellsStory : Person → Prop
axiom EvokesEmotion : Person → Prop
axiom ExpressesEmotion : Person → Prop
axiom ConnectsAudience : Person → Prop
axiom BuildsConnection : Person → Prop
axiom AchieveSuccess : Person → Prop
axiom PerformsElegance : Person → Prop
axiom InspiresOthers : Person → Prop
axiom Passion : Person → Prop
axiom SpeaksFromHeart : Person → Prop
axiom TouchesHearts : Person → Prop
axiom TrainedInBallet : Person → Prop

-- Premises
axiom a1_noTalent : ¬ NaturalTalent Sky
axiom a2_artist_xor :
  Artist Sky → ((TellsStory Sky ∨ EvokesEmotion Sky) ∧ ¬ (TellsStory Sky ∧ EvokesEmotion Sky))
axiom a3_evokes : EvokesEmotion Sky
axiom a4_exprOrConn_xor_builds :
  ((ExpressesEmotion Sky ∨ ConnectsAudience Sky) ∧ ¬ (ExpressesEmotion Sky ∧ ConnectsAudience Sky)) → BuildsConnection Sky
axiom a5_noSuccess : ¬ AchieveSuccess Sky
axiom a6_elegConn_inspires : ∀ x : Person, (PerformsElegance x ∧ ConnectsAudience x) → InspiresOthers x
axiom a7_passion_story_heart : Passion Sky → (TellsStory Sky ∧ SpeaksFromHeart Sky)
axiom a8_inspires_to_success : InspiresOthers Sky → AchieveSuccess Sky
axiom a9_noTouch : ¬ TouchesHearts Sky
axiom a10_train_or_talent_elegance : ∀ x : Person, (TrainedInBallet x ∨ NaturalTalent x) → PerformsElegance x
axiom a11_artist : Artist Sky
axiom a12_tells_or_builds_touch : ∀ x : Person, (TellsStory x ∨ BuildsConnection x) → TouchesHearts x
axiom a13_expresses : ExpressesEmotion Sky

-- Conclusion: Sky is not trained in ballet.
theorem not_trained_Sky : ¬ TrainedInBallet Sky := by
  intro hTrained
  -- From "anyone who tells a story or builds a connection can touch hearts" and "Sky cannot touch hearts"
  -- we get ¬(TellsStory Sky ∨ BuildsConnection Sky), hence ¬BuildsConnection Sky.
  have hNoTSorBC : ¬ (TellsStory Sky ∨ BuildsConnection Sky) := by
    intro h
    have hHearts : TouchesHearts Sky := a12_tells_or_builds_touch Sky h
    exact a9_noTouch hHearts
  have hNoBuild : ¬ BuildsConnection Sky := by
    intro hB
    exact hNoTSorBC (Or.inr hB)

  -- From training (assumed), get elegance.
  have hPerf : PerformsElegance Sky := by
    have hDisj : TrainedInBallet Sky ∨ NaturalTalent Sky := Or.inl hTrained
    exact a10_train_or_talent_elegance Sky hDisj

  -- (Elegance ∧ Connects) → Success (via Inspire)
  have hImpToSuccess : (PerformsElegance Sky ∧ ConnectsAudience Sky) → AchieveSuccess Sky := by
    intro hPC
    have hInsp : InspiresOthers Sky := a6_elegConn_inspires Sky hPC
    exact a8_inspires_to_success hInsp

  -- Since ¬Success, we have ¬(Elegance ∧ Connects), hence from Elegance we get ¬Connects.
  have hNotBoth : ¬ (PerformsElegance Sky ∧ ConnectsAudience Sky) := by
    intro hPC
    have hSucc : AchieveSuccess Sky := hImpToSuccess hPC
    exact a5_noSuccess hSucc
  have hNotConnect : ¬ ConnectsAudience Sky := by
    intro hC
    exact hNotBoth ⟨hPerf, hC⟩

  -- Using ExpressesEmotion and ¬Connects, the XOR condition holds, yielding BuildsConnection,
  -- contradicting ¬BuildsConnection.
  have hOrEC : ExpressesEmotion Sky ∨ ConnectsAudience Sky := Or.inl a13_expresses
  have hNotAndEC : ¬ (ExpressesEmotion Sky ∧ ConnectsAudience Sky) := by
    intro hAnd
    exact hNotConnect hAnd.right
  have hXorEC : (ExpressesEmotion Sky ∨ ConnectsAudience Sky) ∧ ¬ (ExpressesEmotion Sky ∧ ConnectsAudience Sky) :=
    And.intro hOrEC hNotAndEC
  have hBuild : BuildsConnection Sky := a4_exprOrConn_xor_builds hXorEC
  exact hNoBuild hBuild

Verification Success: True
