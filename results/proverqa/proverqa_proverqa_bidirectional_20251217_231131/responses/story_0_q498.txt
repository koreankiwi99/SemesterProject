  have hRM_imp_AG : RequiresMedicalAttention Saul → AchieveGoals Saul := by
    intro hrm
    have hconj : RequiresMedicalAttention Saul ∧ HasStrongWill Saul := And.intro hrm p13
    have : AchieveGoals Saul := p7 Saul hconj
    exact this
  have hNotRM : ¬RequiresMedicalAttention Saul := by
    intro hrm
    have hAG : AchieveGoals Saul := hRM_imp_AG hrm
    exact p1 hAG
  -- Uses assistive devices and not requiring medical attention triggers disabled

    | inl h =>
        exact False.elim (p2 h.left)

        exact h.left

        exact h.left

      | inl h =>
          exact False.elim (h.right p16)

          exact h.right

    exact p11 hOA
  -- Conclude losing motivation from the xor and ¬(finds new strength)
  cases hLMorFNS with
  | inl h =>
      exact h.left
  | inr h =>
      exact False.elim (hNotFNS h.left)
