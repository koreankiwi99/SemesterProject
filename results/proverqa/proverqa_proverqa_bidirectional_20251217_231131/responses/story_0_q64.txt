Premises: Alden listens actively. Anyone who has influence or commands respect holds power. Alden possesses experience. Anyone who makes key decisions or influences others has authority. Alden faces consequences. Alden has strong opinions. Alden either listens actively or speaks persuasively, but not both. If someone has authority, then they hold power. Anyone who takes leadership or earns respect has authority. If Alden holds power, then he either upholds justice or faces consequences, but not both. Superman does not hold a high office. For anyone who possesses experience and has strong opinions, they can make key decisions.
Conclusion: Alden does not uphold justice.

axiom p3 : PossessesExperience Alden
axiom p4 : ∀ x : Person, (MakesKeyDecisions x ∨ InfluencesOthers x) → HasAuthority x
axiom p5 : FacesConsequences Alden
axiom p6 : HasStrongOpinions Alden
axiom p7 : (ListensActively Alden ∧ ¬ SpeaksPersuasively Alden) ∨ (¬ ListensActively Alden ∧ SpeaksPersuasively Alden)
axiom p8 : ∀ x : Person, HasAuthority x → HoldsPower x
axiom p9 : ∀ x : Person, (TakesLeadership x ∨ EarnsRespect x) → HasAuthority x
axiom p10 : HoldsPower Alden → ((UpholdsJustice Alden ∧ ¬ FacesConsequences Alden) ∨ (¬ UpholdsJustice Alden ∧ FacesConsequences Alden))
axiom p11 : ¬ HoldsHighOffice Superman
axiom p12 : ∀ x : Person, (PossessesExperience x ∧ HasStrongOpinions x) → MakesKeyDecisions x

-- Conclusion: Alden does not uphold justice.
theorem conclusion : ¬ UpholdsJustice Alden := by
  -- From experience and strong opinions, Alden makes key decisions
  have hmkd : MakesKeyDecisions Alden := by
    apply p12
    exact And.intro p3 p6
  -- From making key decisions, Alden has authority
  have hauth : HasAuthority Alden := by
    apply p4 Alden
    exact Or.inl hmkd
  -- Authority implies holding power
  have hpow : HoldsPower Alden := by
    exact p8 Alden hauth
  -- Holding power gives an exclusive-or between justice and consequences
  have hx : (UpholdsJustice Alden ∧ ¬ FacesConsequences Alden) ∨ (¬ UpholdsJustice Alden ∧ FacesConsequences Alden) := p10 hpow
  -- Using that Alden faces consequences, deduce he does not uphold justice
  cases hx with
  | inl h1 =>
      have nf : ¬ FacesConsequences Alden := h1.right
      exact False.elim (nf p5)
  | inr h2 =>
      exact h2.left
