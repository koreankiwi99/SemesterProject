Premises: For all camels, if they have a dry mouth and feel weak, then they are thirsty. If Goobers carries a heavy load, then he either stumbles frequently or feels exhausted (or both). Goobers does not reach water. Goobers does not have strong legs. If a camel stumbles frequently, then it loses balance and gets injured. For every camel, it is either thirsty or well-hydrated, but not both. If Goobers feels exhausted, then he either rests in the shade or walks briskly, but not both. Every camel that crosses the desert is either thirsty or finds an oasis. Goobers does not lose balance. Goobers gets injured. Goobers carries a heavy load. Goobers either rests in the shade or traverses dunes, but not both at the same time. All camels that have owners are able to carry loads and traverse dunes. If Goobers traverses dunes, then he either crosses the desert or reaches water. Goobers walks briskly. All camels that find an oasis have strong legs and endurance. Blossom does not lose balance.
Conclusion: Goobers is well-hydrated.

-- Goal: negate the conclusion "Goobers is well-hydrated"
theorem goal : ¬ WellHydrated Goobers := by
  -- From heavy load, Goobers either stumbles or is exhausted
  have hSE : Stumbles Goobers ∨ Exhausted Goobers := premise2 premise11
  -- Show Goobers does not stumble (would contradict no loss of balance)
  have hNotStumbles : ¬ Stumbles Goobers := by
    intro hS
    have hLB_and_inj := premise5 Goobers hS
    have hLB : LoseBalance Goobers := And.left hLB_and_inj
    exact premise9 hLB
  -- Therefore, Goobers is exhausted
  have hEx : Exhausted Goobers := by
    cases hSE with
    | inl hS => exact False.elim (hNotStumbles hS)
    | inr hE => exact hE
  -- From exhaustion, (rest or brisk) and not both; combined with brisk -> not rest
  have h7 := premise7 hEx
  have hNotBothRW : ¬ (RestShade Goobers ∧ WalksBriskly Goobers) := And.right h7
  have hNotRest : ¬ RestShade Goobers := by
    intro hR
    exact hNotBothRW (And.intro hR premise15)
  -- From (rest or traverse) and not rest, deduce traverse
  have h12 := premise12
  have hTrav : TraversesDunes Goobers := by
    cases (And.left h12) with
    | inl hR => exact False.elim (hNotRest hR)
    | inr hT => exact hT
  -- From traverse -> (cross or reach water), but not reach water, so cross
  have hCross : CrossDesert Goobers := by
    cases premise14 hTrav with
    | inl hC => exact hC
    | inr hW => exact False.elim (premise3 hW)
  -- From crossing desert -> (thirsty or finds oasis)
  have hTO : Thirsty Goobers ∨ FindsOasis Goobers := premise8 Goobers hCross
  -- Can't find oasis (would give strong legs, which is false)
  have hNotOasis : ¬ FindsOasis Goobers := by
    intro hO
    have hSL_and_End := premise16 Goobers hO
    have hSL : StrongLegs Goobers := And.left hSL_and_End
    exact premise4 hSL
  -- Therefore, Goobers is thirsty
  have hThirst : Thirsty Goobers := by
    cases hTO with
    | inl hT => exact hT
    | inr hO => exact False.elim (hNotOasis hO)
  -- Use exclusivity to rule out well-hydrated
  have hNotBoth : ¬ (Thirsty Goobers ∧ WellHydrated Goobers) := And.right (premise6 Goobers)
  intro hWH
  exact hNotBoth (And.intro hThirst hWH)
