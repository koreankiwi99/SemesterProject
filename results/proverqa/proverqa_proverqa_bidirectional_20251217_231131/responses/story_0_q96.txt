Premises: Hope either pursues alternative careers or is a dedicated scientist, but not both. Anyone who conducts research or publishes papers is a dedicated scientist. Hope does not work independently. If Hope is a dedicated scientist, then she either makes a groundbreaking discovery or inspires others (or both). Hope does not inspire others. If Hope either excels academically or pursues alternative careers (but not both), then she has innovative ideas. Hope either works independently or collaborates with others, but not both. Hope does not excel academically. If someone has innovative ideas, then they will make valuable contributions. If Hope makes valuable contributions, then she thinks critically and collaborates with others. Hope does not think critically.
Conclusion: Hope makes a groundbreaking discovery.

-- Propositional atoms
axiom PursuesAlt : Prop        -- Hope pursues alternative careers (A)
axiom Dedicated : Prop         -- Hope is a dedicated scientist (D)
axiom Research : Prop          -- Hope conducts research (R)
axiom Publishes : Prop         -- Hope publishes papers (P)
axiom WorksInd : Prop          -- Hope works independently (W)
axiom Groundbreaking : Prop    -- Hope makes a groundbreaking discovery (G)
axiom Inspires : Prop          -- Hope inspires others (I)
axiom Excels : Prop            -- Hope excels academically (E)
axiom Innovative : Prop        -- Hope has innovative ideas (N)
axiom Valuable : Prop          -- Hope makes valuable contributions (V)
axiom ThinksCritically : Prop  -- Hope thinks critically (T)
axiom Collaborates : Prop      -- Hope collaborates with others (C)

-- Premises
axiom xor_A_or_D : (PursuesAlt ∨ Dedicated) ∧ ¬(PursuesAlt ∧ Dedicated)
axiom researchOrPublish_imp_D : (Research ∨ Publishes) → Dedicated
axiom notWorksInd : ¬WorksInd
axiom D_imp_GorI : Dedicated → (Groundbreaking ∨ Inspires)
axiom notInspires : ¬Inspires
axiom xor_E_or_A_imp_Innov : ((Excels ∨ PursuesAlt) ∧ ¬(Excels ∧ PursuesAlt)) → Innovative
axiom xor_W_or_C : (WorksInd ∨ Collaborates) ∧ ¬(WorksInd ∧ Collaborates)
axiom notExcels : ¬Excels
axiom Innov_imp_Val : Innovative → Valuable
axiom Val_imp_TandC : Valuable → (ThinksCritically ∧ Collaborates)
axiom notThinks : ¬ThinksCritically

-- Conclusion: Hope makes a groundbreaking discovery.
theorem goal : Groundbreaking := by
  -- From D we can get G using ¬I
  have D_to_G : Dedicated → Groundbreaking := by
    intro hD
    have h := D_imp_GorI hD
    cases h with
    | inl hG => exact hG
    | inr hI => exact False.elim (notInspires hI)
  -- Split on A ∨ D
  have hAorD : PursuesAlt ∨ Dedicated := (And.left xor_A_or_D)
  cases hAorD with
  | inl hA =>
      -- From A and ¬E, build the XOR (E ∨ A) ∧ ¬(E ∧ A), derive Innovative, then V, then T, contradiction
      have hEorA : Excels ∨ PursuesAlt := Or.inr hA
      have hNotEA : ¬(Excels ∧ PursuesAlt) := by
        intro hEA
        exact notExcels hEA.left
      have hX : (Excels ∨ PursuesAlt) ∧ ¬(Excels ∧ PursuesAlt) := And.intro hEorA hNotEA
      have hN : Innovative := xor_E_or_A_imp_Innov hX
      have hV : Valuable := Innov_imp_Val hN
      have hTandC : ThinksCritically ∧ Collaborates := Val_imp_TandC hV
      have hT : ThinksCritically := hTandC.left
      exact False.elim (notThinks hT)
  | inr hD =>
      exact D_to_G hD
