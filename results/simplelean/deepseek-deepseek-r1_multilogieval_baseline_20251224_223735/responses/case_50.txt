<lean>
axiom obj : Type
axiom Jesse : obj
axiom Sprints : obj → Prop
axiom Stretching : obj → Prop
axiom RunningImproved : obj → Prop
axiom FlexibleAndLessInjury : obj → Prop
axiom A1 : ∀ x, Sprints x → RunningImproved x
axiom A3 : ∀ x, Sprints x ∨ Stretching x
axiom A4 : ∀ x, Stretching x → FlexibleAndLessInjury x
axiom notRunning : ¬ RunningImproved Jesse
theorem flex : FlexibleAndLessInjury Jesse := by
  have noSprints : ¬ Sprints Jesse :=
    fun hSprints => notRunning (A1 Jesse hSprints)
  have stretch : Stretching Jesse :=
    Or.resolve_left (A3 Jesse) noSprints
  exact A4 Jesse stretch
</lean>

ANSWER: Yes
