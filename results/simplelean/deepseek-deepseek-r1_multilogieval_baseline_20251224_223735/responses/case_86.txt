<lean>
axiom Person : Type
axiom Snowstorm : Prop
axiom SchoolCanceled : Prop
axiom StudiesHard : Person → Prop
axiom GoodGrades : Person → Prop
axiom ProudParents : Person → Prop
axiom IceCream : Person → Prop
axiom Priya : Person

axiom A1 : ∀ (p : Person), StudiesHard p → GoodGrades p
axiom A2 : Snowstorm → SchoolCanceled
axiom A3 : StudiesHard Priya ∨ ¬SchoolCanceled
axiom A4 : ∀ (p : Person), GoodGrades p → ProudParents p
axiom A5 : ∀ (p : Person), ProudParents p → IceCream p

theorem result : Snowstorm → IceCream Priya := by
  intro hSnow
  have hCancel : SchoolCanceled := A2 hSnow
  have hStudy : StudiesHard Priya := by
    cases A3 with
    | inl h => exact h
    | inr nh => contradiction
  have hGrades : GoodGrades Priya := A1 Priya hStudy
  have hProud : ProudParents Priya := A4 Priya hGrades
  exact A5 Priya hProud

</lean>

ANSWER: Yes
