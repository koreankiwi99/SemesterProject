<lean>
axiom Athlete : Type
axiom Lebron : Athlete
axiom TrainsHard : Athlete → Prop
axiom RemainsFocused : Athlete → Prop
axiom PerformsBest : Athlete → Prop

axiom coaching : ∀ x : Athlete, TrainsHard x → (RemainsFocused x → PerformsBest x)
axiom lebron_focused : RemainsFocused Lebron
axiom lebron_not_performed : ¬ PerformsBest Lebron

theorem lebron_not_trained : ¬ TrainsHard Lebron :=
  fun h_train : TrainsHard Lebron =>
    have h_implication : RemainsFocused Lebron → PerformsBest Lebron := coaching Lebron h_train
    have h_performed : PerformsBest Lebron := h_implication lebron_focused
    lebron_not_performed h_performed
</lean>

ANSWER: No
