<lean>
axiom Person : Type
axiom Alex : Person
axiom StudiedHard : Person → Prop
axiom GoodGrades : Person → Prop
axiom Participated : Person → Prop
axiom MadeFriends : Person → Prop
axiom Invited : Person → Prop

-- If studied hard, get good grades
axiom A1 : ∀ p : Person, StudiedHard p → GoodGrades p
-- If participated, make new friends
axiom A2 : ∀ p : Person, Participated p → MadeFriends p
-- Either studied hard or participated
axiom A3 : StudiedHard Alex ∨ Participated Alex
-- If make new friends, get invited
axiom A4 : ∀ p : Person, MadeFriends p → Invited p

-- Given that Alex did not get good grades
axiom NotGoodGrades : ¬ GoodGrades Alex

theorem alex_invited : Invited Alex := by
  -- Since ¬GoodGrades Alex and A1, contrapositive gives ¬StudiedHard Alex
  have no_study : ¬ StudiedHard Alex := by
    intro H
    apply NotGoodGrades
    exact A1 Alex H
  -- Now use disjunction A3: StudiedHard Alex ∨ Participated Alex
  cases A3 with
  | inl study => contradiction  -- contradicts no_study
  | inr participate =>
    -- So Alex participated, then by A2: MadeFriends Alex
    have friends : MadeFriends Alex := A2 Alex participate
    -- Then by A4: Invited Alex
    exact A4 Alex friends

</lean>

ANSWER: True
