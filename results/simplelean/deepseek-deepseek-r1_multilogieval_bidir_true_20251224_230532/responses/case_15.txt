<lean>
axiom obj : Type
axiom Student : obj
axiom Study : obj → Prop
axiom Excel : obj → Prop
axiom Participate : obj → Prop
axiom Leader : obj → Prop
axiom Prefect : obj → Prop

axiom premise1 : Study Student → Excel Student
axiom premise2 : Participate Student → Leader Student
axiom premise3 : Study Student ∨ Participate Student
axiom premise4 : ∀ x, Leader x → Prefect x

axiom given : ¬ Excel Student

theorem chosen : Prefect Student := by
  have not_study : ¬ Study Student := fun hStudy => given (premise1 hStudy)
  have participate : Participate Student := by
    cases premise3 with
    | inl study => contradiction
    | inr part => exact part
  have leader : Leader Student := premise2 participate
  exact premise4 Student leader
</lean>

ANSWER: True
