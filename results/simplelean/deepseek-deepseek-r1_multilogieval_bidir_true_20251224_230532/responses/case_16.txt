<lean>
axiom obj : Type
axiom Sarah : obj
axiom Practices : obj → Prop
axiom TechniqueImproves : obj → Prop
axiom LearnsFaster : obj → Prop
axiom ListensAttentively : obj → Prop

-- If practiced regularly, technique improves
axiom A1 : ∀ x : obj, Practices x → TechniqueImproves x

-- If technique improves, learns faster
axiom A2 : ∀ x : obj, TechniqueImproves x → LearnsFaster x

-- Either practices regularly or listens attentively (or both)
axiom A3 : ∀ x : obj, Practices x ∨ ListensAttentively x

-- Given premise: Sarah is not able to learn songs faster
axiom T1 : ¬ LearnsFaster Sarah

-- Prove: Sarah listens attentively to music
theorem sarah_listens : ListensAttentively Sarah := by
  have no_improve : ¬ TechniqueImproves Sarah := 
    fun h_tech : TechniqueImproves Sarah => T1 (A2 Sarah h_tech)
  have no_practice : ¬ Practices Sarah := 
    fun h_prac : Practices Sarah => no_improve (A1 Sarah h_prac)
  cases (A3 Sarah) with
  | inl h_prac => exact False.elim (no_practice h_prac)
  | inr h_listen => exact h_listen
</lean>

ANSWER: True
