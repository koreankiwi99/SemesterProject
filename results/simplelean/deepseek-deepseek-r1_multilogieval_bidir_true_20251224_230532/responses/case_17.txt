<lean>
axiom Person : Type
axiom James : Person

axiom PracticesThoroughly : Person → Prop
axiom FeelsPrepared : Person → Prop
axiom SpeaksConfidently : Person → Prop
axiom JoinsClub : Person → Prop

-- If practices thoroughly, then feels prepared
axiom A1 : ∀ x : Person, PracticesThoroughly x → FeelsPrepared x

-- If feels prepared, then speaks confidently
axiom A2 : ∀ x : Person, FeelsPrepared x → SpeaksConfidently x

-- Either practices thoroughly or joins club (or both)
axiom A3 : ∀ x : Person, PracticesThoroughly x ∨ JoinsClub x

-- James cannot speak confidently
axiom James_not_confident : ¬ SpeaksConfidently James

theorem James_joins_club : JoinsClub James := by
  have no_prep : ¬ FeelsPrepared James := fun h_prep => 
    James_not_confident (A2 James h_prep)
  have no_practice : ¬ PracticesThoroughly James := fun h_practice => 
    no_prep (A1 James h_practice)
  cases (A3 James) with
  | inl h_practice => contradiction
  | inr h_joins => exact h_joins
</lean>

ANSWER: True
