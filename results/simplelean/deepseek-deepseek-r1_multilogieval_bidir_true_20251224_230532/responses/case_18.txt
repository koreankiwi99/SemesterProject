<lean>
axiom obj : Type
axiom Sam : obj
axiom Impressed : obj → Prop
axiom Promotion : obj → Prop
axiom ExtraWork : obj → Prop
axiom Experience : obj → Prop
axiom LeadTeam : obj → Prop

axiom impresses_implies_promotion : ∀ x, Impressed x → Promotion x
axiom extra_work_implies_experience : ∀ x, ExtraWork x → Experience x
axiom employee_choice : Impressed Sam ∨ ExtraWork Sam
axiom experience_implies_lead : ∀ x, Experience x → LeadTeam x

theorem sam_leads_team (no_promotion : ¬ Promotion Sam) : LeadTeam Sam := by
  have not_impressed : ¬ Impressed Sam := fun h => no_promotion (impresses_implies_promotion Sam h)
  cases employee_choice with
  | inl impressed => contradiction
  | inr extra => 
      have experience : Experience Sam := extra_work_implies_experience Sam extra
      exact experience_implies_lead Sam experience
</lean>

ANSWER: True
