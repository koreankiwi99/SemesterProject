<lean>
axiom Rains : Prop
axiom Watered : Prop
axiom Hot : Prop
axiom Play : Prop

axiom rain_watered : Rains → Watered
axiom hot_play : Hot → Play
axiom rain_or_no_play : Rains ∨ ¬Play

theorem watered_naturally (hHot : Hot) : Watered := by
  have hPlay : Play := hot_play hHot
  cases rain_or_no_play
  case inl hRain => 
      exact rain_watered hRain
  case inr hNoPlay => 
      contradiction
</lean>

ANSWER: True
