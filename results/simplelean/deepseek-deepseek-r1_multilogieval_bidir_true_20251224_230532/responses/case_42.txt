<lean>
axiom Sunny : Prop
axiom Flu : Prop
axiom Sick : Prop
axiom Play : Prop
axiom SleepWell : Prop
axiom sunny_implies_play : Sunny → Play
axiom flu_implies_sick : Flu → Sick
axiom disjunction : Sunny ∨ ¬Sick ∨ (Sunny ∧ ¬Sick)
axiom play_implies_sleep : Play → SleepWell
theorem children_sleep_well (flu_occurred : Flu) : SleepWell := by
  have sick_occurred : Sick := flu_implies_sick flu_occurred
  rcases disjunction with hSunny | hNotSick | ⟨hSunny, hNotSick⟩
  . exact play_implies_sleep (sunny_implies_play hSunny)
  . contradiction
  . contradiction
</lean>

ANSWER: True
