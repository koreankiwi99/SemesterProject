<lean>
axiom obj : Type
axiom Matt : obj
axiom WatchedVideos : obj → Prop
axiom GainSkills : obj → Prop
axiom RedoLivingRoom : obj → Prop
axiom ReadBlogs : obj → Prop
axiom GetNewIdeas : obj → Prop

-- If watched DIY videos, gain decorating skills
axiom A1 : ∀ x, WatchedVideos x → GainSkills x
-- If gain skills, redo living room
axiom A2 : ∀ x, GainSkills x → RedoLivingRoom x
-- Did not redo living room (given for Matt)
axiom A3 : ¬ RedoLivingRoom Matt
-- Either watched videos or read blogs (or both)
axiom A4 : ∀ x, WatchedVideos x ∨ ReadBlogs x
-- If read blogs, get new decorating ideas
axiom A5 : ∀ x, ReadBlogs x → GetNewIdeas x

theorem MattGetsIdeas : GetNewIdeas Matt := by
  have no_skills : ¬ GainSkills Matt := fun h_skills => 
    A3 (A2 Matt h_skills)
  have no_videos : ¬ WatchedVideos Matt := fun h_videos => 
    no_skills (A1 Matt h_videos)
  have : WatchedVideos Matt ∨ ReadBlogs Matt := A4 Matt
  cases this with
  | inl h_videos => contradiction
  | inr h_blogs => exact A5 Matt h_blogs
</lean>

ANSWER: True
