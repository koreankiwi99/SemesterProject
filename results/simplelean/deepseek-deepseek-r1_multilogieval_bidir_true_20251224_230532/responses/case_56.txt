<lean>
axiom obj : Type
axiom Emma : obj
axiom SpeaksDaily : obj → Prop
axiom FluencyImproves : obj → Prop
axiom CanHaveRealConvo : obj → Prop
axiom ListensPodcasts : obj → Prop
axiom GainsListening : obj → Prop

axiom A1 : ∀ x, SpeaksDaily x → FluencyImproves x
axiom A2 : ∀ x, FluencyImproves x → CanHaveRealConvo x
axiom A3 : ∀ x, SpeaksDaily x ∨ ListensPodcasts x
axiom A4 : ∀ x, ListensPodcasts x → GainsListening x

theorem answer : ¬ CanHaveRealConvo Emma → GainsListening Emma := by
  intro noConversation
  have h : SpeaksDaily Emma ∨ ListensPodcasts Emma := A3 Emma
  cases h with
  | inl speaks =>
      have fluency : FluencyImproves Emma := A1 Emma speaks
      have conversation : CanHaveRealConvo Emma := A2 Emma fluency
      contradiction
  | inr podcasts =>
      exact A4 Emma podcasts
</lean>

ANSWER: True
