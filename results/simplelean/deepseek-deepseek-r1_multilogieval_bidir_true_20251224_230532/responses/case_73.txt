<lean>
axiom obj : Type
axiom Simone : obj
axiom TrainsWithDedication : obj → Prop
axiom AdequateRest : obj → Prop
axiom PerformAtPeak : obj → Prop

axiom R1 : ∀ x : obj, TrainsWithDedication x → AdequateRest x → PerformAtPeak x
axiom T1 : AdequateRest Simone
axiom T2 : ¬ PerformAtPeak Simone

theorem not_trained : ¬ TrainsWithDedication Simone := by
  intro h
  have peak : PerformAtPeak Simone := R1 Simone h T1
  exact T2 peak
</lean>

ANSWER: True
