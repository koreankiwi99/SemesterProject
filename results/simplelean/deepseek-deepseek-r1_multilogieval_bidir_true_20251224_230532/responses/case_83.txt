<lean>
axiom P : Prop -- team practices every day
axiom S : Prop -- it is sunny outside
axiom Play : Prop -- children play at the park
axiom Prepared : Prop -- team is prepared
axiom Confident : Prop -- team feels confident
axiom HoldHeadHigh : Prop -- team holds head high

-- If team practices daily, they will be prepared
axiom A1 : P → Prepared

-- If sunny, children play at park
axiom A2 : S → Play

-- Either team practiced daily or kids did not go to park
axiom A3 : P ∨ ¬Play

-- If prepared, then confident
axiom A4 : Prepared → Confident

-- If confident, then hold head high
axiom A5 : Confident → HoldHeadHigh

-- Prove: If sunny, then team holds head high
theorem sun_implies_high : S → HoldHeadHigh :=
  fun hS : S =>
  have hPlay : Play := A2 hS
  have hP_or_not : P ∨ ¬Play := A3
  have hP : P := by
    cases hP_or_not with
    | inl hp => exact hp
    | inr not_play => contradiction  -- hPlay contradicts ¬Play
  have hPrepared : Prepared := A1 hP
  have hConfident : Confident := A4 hPrepared
  show HoldHeadHigh from A5 hConfident
</lean>

ANSWER: True
