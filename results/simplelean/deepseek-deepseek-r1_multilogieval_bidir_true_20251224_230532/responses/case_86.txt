<lean>
axiom obj : Type
axiom Priya : obj
axiom Snowstorm : Prop
axiom SchoolCanceled : Prop
axiom StudiesHard : obj → Prop
axiom GoodGrades : obj → Prop
axiom ProudParents : obj → Prop
axiom IceCream : obj → Prop

axiom R1 : ∀ x : obj, StudiesHard x → GoodGrades x
axiom R2 : Snowstorm → SchoolCanceled
axiom T1 : StudiesHard Priya ∨ ¬ SchoolCanceled
axiom R3 : ∀ x : obj, GoodGrades x → ProudParents x
axiom R4 : ∀ x : obj, ProudParents x → IceCream x

theorem snowstorm_icecream : Snowstorm → IceCream Priya := by
  intro hSnow
  have hCancel : SchoolCanceled := R2 hSnow
  cases T1
  case inl hStudy => 
    have hGrades := R1 Priya hStudy
    have hProud := R3 Priya hGrades
    exact R4 Priya hProud
  case inr notCancel => 
    contradiction
</lean>

ANSWER: True
