<lean>
axiom obj : Type
axiom Party : obj
axiom Sam : obj
axiom BringsDish : obj → Prop
axiom EnoughFood : obj → Prop
axiom BirthdayParty : obj → Prop
axiom Cake : obj → Prop
axiom IceCream : obj → Prop

axiom A1 : ∀ x : obj, BringsDish x → EnoughFood x
axiom A2 : ∀ x : obj, BirthdayParty x → Cake x ∧ IceCream x
axiom A3 : ∀ x : obj, BringsDish x ∨ ¬Cake x

theorem sam_has_enough_food : BirthdayParty Sam → EnoughFood Sam := by
  intro h
  -- From the fact it's Sam's birthday party, we get cake and ice cream.
  have h2 := A2 Sam h
  -- Extract the cake part from the conjunction.
  have h_cake := h2.1
  -- Apply the third premise to Sam's scenario.
  have h3 := A3 Sam
  -- Since there is cake, the second disjunct (¬Cake Sam) is false, so BringsDish Sam must be true.
  cases h3 with
  | inl h_brings =>
    -- If someone brought a dish, apply the first axiom to get enough food.
    exact A1 Sam h_brings
  | inr h_no_cake =>
    -- Contradiction: we have Cake Sam from A2, but h_no_cake says ¬Cake Sam.
    contradiction
</lean>

ANSWER: True
