<lean>
axiom obj : Type
axiom Alex : obj
axiom AdvertiseCleverly : obj → Prop
axiom OfferDiscounts : obj → Prop
axiom AttractCustomers : obj → Prop
axiom IncreaseSales : obj → Prop
axiom ExpandBusiness : obj → Prop

axiom P1 : ∀ x : obj, AdvertiseCleverly x → AttractCustomers x
axiom P2 : ∀ x : obj, OfferDiscounts x → IncreaseSales x
axiom P3 : ∀ x : obj, AdvertiseCleverly x ∨ OfferDiscounts x
axiom P4 : ∀ x : obj, IncreaseSales x → ExpandBusiness x

theorem alex_expanded : ¬ AttractCustomers Alex → ExpandBusiness Alex := by
  intro h
  -- From ¬ AttractCustomers Alex, use contrapositive on P1 to get ¬ AdvertiseCleverly Alex
  have h1 : ¬ AdvertiseCleverly Alex := fun a => h (P1 Alex a)
  -- From P3, since ¬ AdvertiseCleverly Alex, it must be OfferDiscounts Alex
  have h2 : OfferDiscounts Alex := by
    cases P3 Alex with
    | inl a => contradiction -- If AdvertiseCleverly Alex, contradicts h1
    | inr b => exact b
  -- From P2, OfferDiscounts Alex implies IncreaseSales Alex
  have h3 : IncreaseSales Alex := P2 Alex h2
  -- From P4, IncreaseSales Alex implies ExpandBusiness Alex
  exact P4 Alex h3
</lean>

ANSWER: Yes
