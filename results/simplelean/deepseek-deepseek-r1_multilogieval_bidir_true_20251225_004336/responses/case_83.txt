<lean>
axiom Practice : Prop
axiom Sunny : Prop
axiom Play : Prop
axiom Prepared : Prop
axiom Confident : Prop
axiom HoldHeadHigh : Prop

axiom A1 : Practice → Prepared
axiom A2 : Sunny → Play
axiom A3 : Practice ∨ ¬ Play
axiom A4 : Prepared → Confident
axiom A5 : Confident → HoldHeadHigh

theorem sunny_implies_head_high : Sunny → HoldHeadHigh := by
  intro hSunny
  have hPlay : Play := A2 hSunny
  cases A3
  case inl hPractice =>
    exact A5 (A4 (A1 hPractice))
  case inr notPlay =>
    contradiction
</lean>

ANSWER: Yes
