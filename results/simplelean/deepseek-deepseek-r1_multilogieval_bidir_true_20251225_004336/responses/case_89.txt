<lean>
axiom monsoon : Prop
axiom farmer_waters : Prop
axiom heavy_rains : Prop
axiom crops_grow_healthy : Prop
axiom bountiful_harvest : Prop
axiom good_profit : Prop

-- If farmer waters, crops grow healthy
axiom A1 : farmer_waters → crops_grow_healthy

-- If monsoon, then heavy rains
axiom A2 : monsoon → heavy_rains

-- Either farmer waters daily OR no heavy rains (exclusive-or not needed)
axiom A3 : farmer_waters ∨ ¬heavy_rains

-- If crops grow healthy, harvest is bountiful
axiom A4 : crops_grow_healthy → bountiful_harvest

-- If harvest bountiful, then good profit
axiom A5 : bountiful_harvest → good_profit

-- Prove: monsoon → good_profit
theorem monsoon_profit : monsoon → good_profit :=
  fun h_monsoon : monsoon =>
  have h_rains : heavy_rains := A2 h_monsoon
  have h_water_case : farmer_waters ∨ ¬heavy_rains := A3
  have h_water : farmer_waters := by 
    cases h_water_case with
    | inl h => exact h
    | inr nh => contradiction  -- h_rains contradicts nh
  have h_grow : crops_grow_healthy := A1 h_water
  have h_harvest : bountiful_harvest := A4 h_grow
  show good_profit from A5 h_harvest
</lean>

ANSWER: Yes
