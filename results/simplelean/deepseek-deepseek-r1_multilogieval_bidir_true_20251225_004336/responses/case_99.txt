<lean>
axiom obj : Type
axiom Rain : Prop
axiom Saturated : Prop
axiom Watered : Prop
axiom Bloom : Prop
axiom Fragrance : Prop
axiom Butterflies : Prop

axiom P1 : Watered → Bloom
axiom P2 : Rain → Saturated
axiom P3 : Watered ∨ ¬Saturated
axiom P4 : Bloom → Fragrance
axiom P5 : Fragrance → Butterflies

theorem attract_butterflies : Rain → Butterflies := by
  intro h_rain
  have h_saturated : Saturated := P2 h_rain
  have h_watered : Watered := by
    -- From P3 (Watered ∨ ¬Saturated) and knowing Saturated (h_saturated), ¬Saturated is false.
    cases P3 with
    | inl h => exact h
    | inr h => contradiction  -- h contradicts h_saturated
  have h_bloom : Bloom := P1 h_watered
  have h_fragrance : Fragrance := P4 h_bloom
  exact P5 h_fragrance
</lean>

ANSWER: Yes
