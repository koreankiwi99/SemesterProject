axiom James : Student
axiom WorksInLibrary : Student → Prop
axiom FromCSDept : Student → Prop
axiom HasUniPartTimeJob : Student → Prop

axiom TakesClassWithDavid : Student → Prop

axiom A2 : ∀ s, HasUniPartTimeJob s → WorksInLibrary s
axiom A3 : ∀ s, TakesDatabase s → FromCSDept s

axiom A7 : ¬ WorksInLab James ∨ ¬ HasUniPartTimeJob James

theorem goal : TakesDatabase James ∨ HasUniPartTimeJob James := by
