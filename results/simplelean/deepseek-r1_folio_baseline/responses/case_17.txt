<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom SpottedNearCampus : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

axiom all_spotted_rabbits_cute : ∀ x, Rabbit x ∧ SpottedNearCampus x → Cute x
axiom some_turtle_spotted : ∃ x, Turtle x ∧ SpottedNearCampus x
axiom only_rabbits_squirrels_spotted : ∀ x, SpottedNearCampus x → Rabbit x ∨ Squirrel x
axiom skittish_not_calm : ∀ x, Skittish x → ¬ Calm x
axiom spotted_squirrels_skittish : ∀ x, Squirrel x ∧ SpottedNearCampus x → Skittish x
axiom Rockie : obj
axiom Rockie_spotted : SpottedNearCampus Rockie
axiom Rockie_calm : Calm Rockie

-- Prove Rockie is not a squirrel
theorem Rockie_not_squirrel : ¬ Squirrel Rockie := by
  intro h
  have h1 : Squirrel Rockie ∧ SpottedNearCampus Rockie := ⟨h, Rockie_spotted⟩
  have h2 : Skittish Rockie := spotted_squirrels_skittish Rockie h1
  have h3 : ¬ Calm Rockie := skittish_not_calm Rockie h2
  exact h3 Rockie_calm

-- Prove Rockie is a rabbit
theorem Rockie_rabbit : Rabbit Rockie := by
  have h : Rabbit Rockie ∨ Squirrel Rockie := only_rabbits_squirrels_spotted Rockie Rockie_spotted
  cases h with
  | inl rabbit => exact rabbit
  | inr squirrel => exact absurd squirrel Rockie_not_squirrel

-- We cannot prove Rockie is not a turtle because:
-- 1. Rockie is a rabbit (proven above) and calm, but we have no axiom relating rabbits to turtles.
-- 2. There exists a spotted turtle (some_turtle_spotted), which must be either rabbit or squirrel by only_rabbits_squirrels_spotted.
--    This implies turtles might be rabbits (if categories overlap), so Rockie could potentially be a turtle.
-- 3. No disjointness axioms between Rabbit, Squirrel, Turtle are provided.

-- Therefore, ¬Turtle Rockie is unprovable, and Turtle Rockie is also unprovable.
