The context establishes conditions for ETS financial aid: applicants must prove economic hardship, which is satisfied by living in a single-parent family or having few resources. Tom meets both conditions and is applying for the GRE, so financial aid must be provided.

<lean>
axiom Person : Type
axiom Tom : Person
axiom AppliesForGRE : Person → Prop
axiom ProvidesFinancialAid : Person → Prop
axiom ProvesEconomicHardship : Person → Prop
axiom LivesInSingleParentFamily : Person → Prop
axiom HasFewResources : Person → Prop

axiom financial_aid_rule : ∀ p : Person, AppliesForGRE p → ProvesEconomicHardship p → ProvidesFinancialAid p
axiom hardship_criteria : ∀ p : Person, (LivesInSingleParentFamily p ∨ HasFewResources p) → ProvesEconomicHardship p
axiom toms_family : LivesInSingleParentFamily Tom
axiom toms_resources : HasFewResources Tom
axiom toms_application : AppliesForGRE Tom

theorem tom_gets_aid : ProvidesFinancialAid Tom := by
  apply financial_aid_rule Tom toms_application
  apply hardship_criteria Tom
  left  -- Uses LivesInSingleParentFamily condition (disjunction left branch)
  exact toms_family
</lean>

ANSWER: True
