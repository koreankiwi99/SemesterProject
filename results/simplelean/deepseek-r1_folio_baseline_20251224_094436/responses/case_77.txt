axiom Person : Type
axiom hannah : Person

axiom WorksOnCampus : Person → Prop
axiom NeedsMoney : Person → Prop
axiom OrdersTakeout : Person → Prop
axiom EnjoysDiningHall : Person → Prop
axiom NotPickyEater : Person → Prop
axiom SpendsTimeWithFriends : Person → Prop

axiom A1 : ∀ (p : Person), WorksOnCampus p → NeedsMoney p
axiom A2 : ∀ (p : Person), OrdersTakeout p → WorksOnCampus p
axiom A3 : ∀ (p : Person), OrdersTakeout p ∨ EnjoysDiningHall p
axiom A4 : ∀ (p : Person), EnjoysDiningHall p → NotPickyEater p
axiom A5 : ∀ (p : Person), EnjoysDiningHall p → SpendsTimeWithFriends p

axiom H1 : WorksOnCampus hannah
axiom H2 : NeedsMoney hannah → (NotPickyEater hannah ∧ ¬ NeedsMoney hannah)

theorem target : 
    NotPickyEater hannah ∨ (¬ NotPickyEater hannah → SpendsTimeWithFriends hannah) := 
by 
  have NM : NeedsMoney hannah := A1 hannah H1
  have H2_result : NotPickyEater hannah ∧ ¬ NeedsMoney hannah := H2 NM
  exact False.elim (H2_result.2 NM)

1. `A1` implies `NeedsMoney hannah` from `H1`
2. `H2` then produces both `¬ NeedsMoney hannah` and `NeedsMoney hannah`
3. This contradiction allows the goal to be proven via *ex falso quodlibet*
