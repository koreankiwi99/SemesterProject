axiom obj : Type
axiom DriedThaiChilies : obj
axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop

axiom T2 : ∀ x, Cupcake x → BakedSweet x
axiom T3 : ∀ x, MalaHotpot x → Spicy x
axiom T4 : ∀ x, ProductOfBakedByMelissa x → Cupcake x
axiom T5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

theorem disprove_statement : ¬ (Cupcake DriedThaiChilies ∧ ProductOfBakedByMelissa DriedThaiChilies) := by
  intro ⟨hCupcake, hMelissa⟩
  have hBaked : BakedSweet DriedThaiChilies := T2 DriedThaiChilies hCupcake
  have not_spicy : ¬ Spicy DriedThaiChilies := T1 DriedThaiChilies hBaked
  cases T5 with
  | inl spicy => exact not_spicy spicy
  | inr hOr =>
      cases hOr with
      | inl mala => exact not_spicy (T3 DriedThaiChilies mala)
      | inr not_baked => exact not_baked hBaked
