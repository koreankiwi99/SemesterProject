<lean>
axiom Person : Type
axiom Grumpy : Person → Prop
axiom MeanToAnimals : Person → Prop
axiom NiceToAnimals : Person → Prop
axiom AnimalLover : Person → Prop
axiom PetOwner : Person → Prop
axiom Tom : Person

-- Premises
axiom A1 : ∀ p : Person, NiceToAnimals p → ¬ MeanToAnimals p
axiom A2 : ∃ p : Person, Grumpy p ∧ MeanToAnimals p
axiom A3 : ∀ p : Person, AnimalLover p → NiceToAnimals p
axiom A4 : ∀ p : Person, PetOwner p → AnimalLover p
axiom A5 : PetOwner Tom

-- Prove Tom is not both grumpy and mean to animals
theorem tom_not_grumpy_and_mean : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) :=
  fun h : Grumpy Tom ∧ MeanToAnimals Tom =>
  have pet_tom : PetOwner Tom := A5
  have lover_tom : AnimalLover Tom := A4 Tom pet_tom
  have nice_tom : NiceToAnimals Tom := A3 Tom lover_tom
  have not_mean_tom : ¬ MeanToAnimals Tom := A1 Tom nice_tom
  not_mean_tom h.2

</lean>

ANSWER: True
