<lean>
axiom Person : Type
axiom Work : Type
axiom City : Type
axiom daniShapiro : Person
axiom familyHistory : Work
axiom familySecrets : Work
axiom boston : City
axiom americanCity : City  -- Specific city named "American City"

axiom BornIn : Person → City → Prop
axiom American : Person → Prop
axiom IsNovel : Work → Prop
axiom IsPodcast : Work → Prop
axiom WrittenBy : Work → Person → Prop
axiom CreatedBy : Work → Person → Prop
axiom IsNovelWriter : Person → Prop
axiom AmericanCity : City → Prop  -- Property of being an American city

-- A podcast is not a novel
axiom A1 : ∀ w : Work, IsPodcast w → ¬ IsNovel w

-- If born in American City (specific city), then American
axiom R1 : ∀ p : Person, BornIn p americanCity → American p

-- If a novel is written by a person, they're a novel writer
axiom R2 : ∀ w : Work, ∀ p : Person, IsNovel w → WrittenBy w p → IsNovelWriter p

-- Given facts
axiom F1 : American daniShapiro
axiom F2 : WrittenBy familyHistory daniShapiro
axiom F3 : IsNovel familyHistory
axiom F4 : CreatedBy familySecrets daniShapiro
axiom F5 : IsPodcast familySecrets
axiom F6 : AmericanCity boston

-- Prove Dani Shapiro is a novel writer
theorem dani_is_novel_writer : IsNovelWriter daniShapiro := 
  R2 familyHistory daniShapiro F3 F2
