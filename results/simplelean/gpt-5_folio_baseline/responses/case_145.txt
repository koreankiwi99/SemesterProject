-- Facts

  have h := R3 Vladimir
  cases h with
  | inl hu => exact hu

  R2 Vladimir vladimir_US

theorem vladimir_can_participate : CanParticipate2024 Vladimir :=
  R1 Vladimir vladimir_can_register
