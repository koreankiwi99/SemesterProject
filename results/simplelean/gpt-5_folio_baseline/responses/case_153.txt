axiom A5 : ∀ x : obj, ChemComp x → ¬ SingleElement x
-- The mixture either both is a chemical compound and contains only one element, or neither.
axiom A6 : (ChemComp Mix ∧ SingleElement Mix) ∨ (¬ ChemComp Mix ∧ ¬ SingleElement Mix)

  intro hC
  cases A6 with
  | inl hBoth =>

      have hNotS : ¬ SingleElement Mix := A5 Mix hC
      exact hNotS hS

      have hNotC : ¬ ChemComp Mix := hNeither.left
      exact hNotC hC

    intro hC
    have hNotS : ¬ SingleElement Mix := A5 Mix hC

    | inl hBoth =>
        exact hNotS hBoth.right
