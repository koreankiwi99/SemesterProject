axiom obj : Type
axiom Human : obj → Prop
axiom Mortal : obj → Prop
axiom Greek : obj → Prop

axiom all_humans_mortal : ∀ x : obj, Human x → Mortal x
axiom all_greeks_human : ∀ x : obj, Greek x → Human x

-- From the premises we can derive that all Greeks are mortal
lemma greeks_mortal : ∀ x : obj, Greek x → Mortal x :=
  fun x gx => all_humans_mortal x (all_greeks_human x gx)

-- However, we cannot derive the existence of any Greek,
-- so we cannot prove ∃ x, Greek x ∧ Mortal x, nor its negation.
-- Hence the truth of "Some Greeks are mortal" is undetermined from the given axioms.
