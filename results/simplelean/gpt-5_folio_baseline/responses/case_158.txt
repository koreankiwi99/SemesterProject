<lean>
axiom obj : Type
axiom James : obj

-- Predicates
axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom HasUnivPartTimeJob : obj → Prop
axiom TakesDBCourse : obj → Prop
axiom TakesClassWithDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- Axioms from the context
axiom A1 : ∀ x : obj, WorksInLibrary x → ¬ FromCS x
axiom A2 : ∀ x : obj, HasUnivPartTimeJob x → WorksInLibrary x
axiom A3 : ∀ x : obj, TakesDBCourse x → FromCS x
axiom A4 : ∀ x : obj, TakesClassWithDavid x → TakesDBCourse x
axiom A5 : ∀ x : obj, WorksInLab x → TakesClassWithDavid x
axiom J1 : WorksInLab James
axiom J2 : ¬ WorksInLab James
axiom J3 : ¬ HasUnivPartTimeJob James

-- Claim: James either takes the database course or has a university part-time job
theorem james_db_or_pt : TakesDBCourse James ∨ HasUnivPartTimeJob James := by
  have hDavid : TakesClassWithDavid James := A5 James J1
  have hDB : TakesDBCourse James := A4 James hDavid
  exact Or.inl hDB
</lean>

ANSWER: True
