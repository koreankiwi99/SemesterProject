<lean>
axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

-- A hawk never swims.
axiom H1 : ∀ x : obj, Hawk x → ¬ Swims x
-- Some birds are hawks.
axiom H2 : ∃ x : obj, Bird x ∧ Hawk x

-- Therefore, it is not the case that all birds swim.
theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swims x) := by
  intro h
  cases H2 with
  | intro w hBwHw =>
    have Bw : Bird w := And.left hBwHw
    have Hw' : Hawk w := And.right hBwHw
    have ns : ¬ Swims w := H1 w Hw'
    have sw : Swims w := h w Bw
    exact ns sw
</lean>

ANSWER: False
