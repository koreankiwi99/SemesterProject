<lean>
axiom obj : Type
axiom Heinrich : obj
axiom Politician : obj → Prop
axiom MemberPrussianStateParliament : obj → Prop
axiom InNaziReichstag : obj → Prop

axiom H1 : Politician Heinrich
axiom H2 : MemberPrussianStateParliament Heinrich
axiom H3 : InNaziReichstag Heinrich

-- Statement: No politicians are part of the Nazi Reichstag.
def NoPoliticiansInNaziReichstag : Prop := ∀ x : obj, Politician x → ¬ InNaziReichstag x

-- We can prove the negation by exhibiting Heinrich as a counterexample.
theorem neg_NoPoliticiansInNaziReichstag : ¬ NoPoliticiansInNaziReichstag :=
by
  intro h
  have hn : ¬ InNaziReichstag Heinrich := h Heinrich H1
  exact hn H3

theorem exists_politician_in_nazi_reichstag : ∃ x : obj, Politician x ∧ InNaziReichstag x :=
by
  exact Exists.intro Heinrich (And.intro H1 H3)
</lean>

ANSWER: False
