<lean>
axiom Person : Type
axiom Taylor : Person

axiom EatSalads : Person → Prop
axiom Conscious : Person → Prop
axiom HealthyChildhood : Person → Prop
axiom MeetsIntake : Person → Prop
axiom Disregard : Person → Prop
axiom GymDaily : Person → Prop

-- All people who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : Person, EatSalads x → Conscious x
-- All people who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : Person, HealthyChildhood x → EatSalads x
-- All people who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : Person, MeetsIntake x → HealthyChildhood x
-- All people who disregard their physical well-being are not very conscious about their health and eating habits.
axiom A4 : ∀ x : Person, Disregard x → ¬ Conscious x
-- If people visit the gym at least once a day, then they always fulfill their daily nutritional intakes.
axiom A5 : ∀ x : Person, GymDaily x → MeetsIntake x
-- Taylor either grew up in a health-conscious childhood home and disregards her physical well-being, or she did neither.
axiom A6 : (HealthyChildhood Taylor ∧ Disregard Taylor) ∨ (¬HealthyChildhood Taylor ∧ ¬Disregard Taylor)

theorem taylor_not_gym : ¬ GymDaily Taylor := by
  -- First, show Taylor did neither (the left disjunct leads to a contradiction)
  have nh_and_nd : ¬HealthyChildhood Taylor ∧ ¬Disregard Taylor := by
    cases A6 with
    | inl h =>
      have es : EatSalads Taylor := A2 Taylor h.left
      have c  : Conscious Taylor := A1 Taylor es
      have nc : ¬Conscious Taylor := A4 Taylor h.right
      exact False.elim (nc c)
    | inr h =>
      exact h
  have nh : ¬HealthyChildhood Taylor := nh_and_nd.left
  intro g
  have mi : MeetsIntake Taylor := A5 Taylor g
  have hh : HealthyChildhood Taylor := A3 Taylor mi
  exact nh hh
</lean>

ANSWER: False
