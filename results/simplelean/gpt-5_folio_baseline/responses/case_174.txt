<lean>
axiom Person : Type
axiom Taylor : Person

axiom EatSaladsRegularly : Person → Prop
axiom VeryConscious : Person → Prop
axiom GrewUpHCH : Person → Prop
axiom FulfillDailyNutrition : Person → Prop
axiom DisregardWellbeing : Person → Prop
axiom VisitGymDaily : Person → Prop

-- Rules
axiom R1 : ∀ x : Person, EatSaladsRegularly x → VeryConscious x
axiom R2 : ∀ x : Person, GrewUpHCH x → EatSaladsRegularly x
axiom R3 : ∀ x : Person, FulfillDailyNutrition x → GrewUpHCH x
axiom R4 : ∀ x : Person, DisregardWellbeing x → ¬ VeryConscious x
axiom R5 : ∀ x : Person, VisitGymDaily x → FulfillDailyNutrition x

-- Taylor's specific information
axiom T :
  (GrewUpHCH Taylor ∧ DisregardWellbeing Taylor) ∨
  (¬GrewUpHCH Taylor ∧ ¬DisregardWellbeing Taylor)

theorem taylor_neither_hch_nor_gym_daily :
  ¬GrewUpHCH Taylor ∧ ¬VisitGymDaily Taylor := by
  -- First, show the first disjunct is impossible
  have not_left : ¬(GrewUpHCH Taylor ∧ DisregardWellbeing Taylor) := by
    intro h
    have h_hch : GrewUpHCH Taylor := h.left
    have h_dis : DisregardWellbeing Taylor := h.right
    have h_eat : EatSaladsRegularly Taylor := R2 Taylor h_hch
    have h_vc  : VeryConscious Taylor := R1 Taylor h_eat
    have h_not_vc : ¬ VeryConscious Taylor := R4 Taylor h_dis
    exact h_not_vc h_vc
  -- From the exclusive-style info, deduce the second disjunct
  have not_hch_and_not_dis : ¬GrewUpHCH Taylor ∧ ¬DisregardWellbeing Taylor := by
    cases T with
    | inl h =>
      have contra : False := not_left h
      exact False.elim contra
    | inr h => exact h
  have not_hch : ¬GrewUpHCH Taylor := not_hch_and_not_dis.left
  -- Gym daily would imply HCH, contradicting ¬HCH
  have not_gym : ¬VisitGymDaily Taylor := by
    intro hGym
    have h_fulfill : FulfillDailyNutrition Taylor := R5 Taylor hGym
    have h_hch : GrewUpHCH Taylor := R3 Taylor h_fulfill
    exact not_hch h_hch
  exact And.intro not_hch not_gym
</lean>

ANSWER: True
