<lean>
axiom obj : Type
axiom Human : obj → Prop
axiom Horse : obj → Prop
axiom Hooves : obj → Prop
axiom all_horses_hooves : ∀ x : obj, Horse x → Hooves x
axiom no_humans_hooves : ∀ x : obj, Human x → ¬ Hooves x

theorem no_human_is_horse : ¬ (∃ x : obj, Human x ∧ Horse x) := by
  intro h
  cases h with
  | intro x hx =>
    have hh : Hooves x := all_horses_hooves x hx.right
    have nhh : ¬ Hooves x := no_humans_hooves x hx.left
    exact nhh hh
</lean>

ANSWER: False
