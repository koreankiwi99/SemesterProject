<lean>
axiom obj : Type
axiom Tom : obj

axiom Nice : obj → Prop
axiom Mean : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop

-- Context axioms
-- No one nice to animals is also mean to animals.
axiom A1 : ∀ x : obj, Nice x → ¬ Mean x
-- Some grumpy people are mean to animals.
axiom A2 : ∃ x : obj, Grumpy x ∧ Mean x
-- All animal lovers are nice to animals.
axiom A3 : ∀ x : obj, AnimalLover x → Nice x
-- All pet owners love animals.
axiom A4 : ∀ x : obj, PetOwner x → AnimalLover x
-- Tom is a pet owner.
axiom A5 : PetOwner Tom

-- Derived facts about Tom
theorem tom_animalLover : AnimalLover Tom := A4 Tom A5
theorem tom_nice : Nice Tom := A3 Tom tom_animalLover
theorem tom_not_mean : ¬ Mean Tom := A1 Tom tom_nice

-- Target statement:
-- If Tom is not both grumpy and mean, then Tom is neither grumpy nor an animal lover.
def claim : Prop := (¬ (Grumpy Tom ∧ Mean Tom)) → (¬ Grumpy Tom ∧ ¬ AnimalLover Tom)

-- We show the claim is false (i.e., we prove its negation).
theorem not_claim : ¬ claim := by
  intro h
  -- Antecedent holds because Tom is not mean
  have hAnte : ¬ (Grumpy Tom ∧ Mean Tom) := by
    intro hgm
    exact tom_not_mean hgm.right
  -- Apply the claimed implication to get the consequent
  have hConseq : (¬ Grumpy Tom ∧ ¬ AnimalLover Tom) := h hAnte
  -- But Tom is an animal lover, contradiction
  have hNotAL : ¬ AnimalLover Tom := hConseq.right
  exact hNotAL tom_animalLover
</lean>

ANSWER: False
