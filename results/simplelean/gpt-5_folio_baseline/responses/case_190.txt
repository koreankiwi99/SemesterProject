-- We prove the negation of the queried implication:

    intro hLe
    have hAct : Activates Yuri := (leaps_activates Yuri) hLe
    exact yuri_not_activates hAct
  -- Next, derive ¬Pro Yuri from the chain Pro → CanJump → Leaps → Activates

    have hLe : Leaps Yuri := (canjump_leaps Yuri) hCan

  -- Therefore, ¬(American Yuri ∧ Pro Yuri)
