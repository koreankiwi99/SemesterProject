<lean>
axiom Event : Type
axiom Happy : Event → Prop
axiom Sad : Event → Prop

-- Events are either happy or sad (exclusively)
axiom happy_or_sad_exclusive : ∀ e : Event, (Happy e ∧ ¬ Sad e) ∨ (Sad e ∧ ¬ Happy e)

-- At least one event is happy (witnessed)
axiom E0 : Event
axiom E0_happy : Happy E0

-- The statement "All events are sad" is false.
theorem not_all_events_sad : ¬ (∀ e : Event, Sad e) := by
  intro hall
  have hs : Sad E0 := hall E0
  have disj := happy_or_sad_exclusive E0
  cases disj with
  | inl h =>
    -- h : Happy E0 ∧ ¬ Sad E0
    exact h.right hs
  | inr h =>
    -- h : Sad E0 ∧ ¬ Happy E0
    exact h.right E0_happy
</lean>

ANSWER: False
