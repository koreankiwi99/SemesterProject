<lean>
axiom obj : Type
axiom AiltonSilva : obj
axiom Brazilian : obj → Prop
axiom PlaysFor : obj → obj → Prop
axiom Club : obj → Prop
axiom Nautico : obj
axiom Braga : obj
axiom Fluminense : obj

-- Context facts
axiom C1 : Club Nautico
axiom C2 : Club Braga
axiom C3 : Club Fluminense
axiom F1 : PlaysFor AiltonSilva Nautico
axiom F2 : Brazilian AiltonSilva

-- The claim in question:
def NoOnePlayingForNauticoIsBrazilian : Prop :=
  ∀ x : obj, PlaysFor x Nautico → ¬ Brazilian x

-- We can prove the negation by giving a counterexample (AiltonSilva).
theorem exists_brazilian_playing_for_nautico : ∃ x : obj, PlaysFor x Nautico ∧ Brazilian x :=
  Exists.intro AiltonSilva (And.intro F1 F2)

theorem not_no_one_playing_for_nautico_is_brazilian : ¬ NoOnePlayingForNauticoIsBrazilian :=
  by
    intro h
    have hb : ¬ Brazilian AiltonSilva := h AiltonSilva F1
    exact hb F2
</lean>
