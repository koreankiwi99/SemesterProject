axiom ChaperonesDances : obj → Prop
axiom StudentAttendsSchool : obj → Prop

axiom A6 :
  Club Bonnie ∧
  ((AttendsEvents Bonnie ∧ StudentAttendsSchool Bonnie) ∨
   ¬ (AttendsEvents Bonnie ∧ ¬ StudentAttendsSchool Bonnie))

    intro hNotChap
