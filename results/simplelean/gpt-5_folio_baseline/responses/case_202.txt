<lean>
axiom obj : Type
axiom AiltonSilva : obj
axiom Ailton : obj
axiom Braga : obj
axiom Nautico : obj
axiom Fluminense : obj

axiom FootballClub : obj → Prop
axiom LoanedOutTo : obj → obj → Prop

-- Ailton Silva is commonly known as Ailton (same person)
axiom alias : Ailton = AiltonSilva

-- Ailton was loaned out to Braga
axiom loanAiltonBraga : LoanedOutTo Ailton Braga

-- Náutico and Braga (and Fluminense) are football clubs
axiom fcNautico : FootballClub Nautico
axiom fcBraga : FootballClub Braga
axiom fcFluminense : FootballClub Fluminense

theorem ailton_silva_loaned_to_fc : ∃ c, FootballClub c ∧ LoanedOutTo AiltonSilva c := by
  have l2 : LoanedOutTo AiltonSilva Braga := by
    simpa [alias] using loanAiltonBraga
  exact ⟨Braga, And.intro fcBraga l2⟩
</lean>

ANSWER: True
