<lean>
axiom obj : Type
axiom StrangerThings : obj
axiom BlackMirror : obj

axiom NetflixShow : obj → Prop
axiom Popular : obj → Prop
axiom BingeKaren : obj → Prop
axiom DownloadKaren : obj → Prop
axiom ShareWithLisa : obj → Prop

-- Context facts
axiom ST_popular : Popular StrangerThings
axiom ST_netflix : NetflixShow StrangerThings
axiom BM_netflix : NetflixShow BlackMirror
axiom BM_not_download : ¬ DownloadKaren BlackMirror

-- Rules
axiom popular_implies_binge :
  ∀ x : obj, (NetflixShow x ∧ Popular x) → BingeKaren x

axiom binge_iff_download_on_netflix :
  ∀ x : obj, NetflixShow x → (BingeKaren x ↔ DownloadKaren x)

axiom binge_netflix_implies_share :
  ∀ x : obj, (NetflixShow x ∧ BingeKaren x) → ShareWithLisa x

theorem share_stranger_things_with_lisa : ShareWithLisa StrangerThings := by
  have hbinge : BingeKaren StrangerThings :=
    popular_implies_binge StrangerThings (And.intro ST_netflix ST_popular)
  exact binge_netflix_implies_share StrangerThings (And.intro ST_netflix hbinge)
</lean>

ANSWER: True
