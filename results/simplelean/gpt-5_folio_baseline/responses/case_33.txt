<lean>
axiom obj : Type
axiom DiamondMine : obj
axiom RoderickStrong : obj
axiom CreedBrothers : obj
axiom IvyNile : obj
axiom Imperium : obj

axiom ProWrestlingStable : obj → Prop
axiom Leads : obj → obj → Prop
axiom Includes : obj → obj → Prop
axiom Feud : obj → obj → Prop

axiom DM_is_stable : ProWrestlingStable DiamondMine
axiom RS_leads_DM : Leads RoderickStrong DiamondMine
axiom DM_includes_CB : Includes DiamondMine CreedBrothers
axiom DM_includes_IN : Includes DiamondMine IvyNile
axiom Imperium_feuds_DM : Feud Imperium DiamondMine

theorem roderick_strong_leads_a_pro_wrestling_stable :
  ∃ s : obj, ProWrestlingStable s ∧ Leads RoderickStrong s :=
by
  refine ⟨DiamondMine, And.intro DM_is_stable RS_leads_DM⟩
</lean>

ANSWER: True
