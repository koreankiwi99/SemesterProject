<lean>
axiom obj : Type

-- Entities
axiom SymphonyNo9 : obj
axiom Beethoven : obj
axiom ViennaMusicSociety : obj

-- Predicates and relations
axiom MusicPiece : obj → Prop
axiom Composer : obj → Prop
axiom Wrote : obj → obj → Prop
axiom Premiered : obj → obj → Prop
axiom Orchestra : obj → Prop
axiom Leads : obj → obj → Prop
axiom Conductor : obj → Prop

-- Given facts and rules
axiom T1 : MusicPiece SymphonyNo9
axiom R1 : ∀ x : obj, Composer x → ∀ y : obj, MusicPiece y → Wrote x y
axiom T2 : Wrote Beethoven SymphonyNo9
axiom T3 : Premiered ViennaMusicSociety SymphonyNo9
axiom T4 : Orchestra ViennaMusicSociety
axiom T5 : Leads Beethoven ViennaMusicSociety
axiom R2 : ∀ x : obj, Orchestra x → ∀ y : obj, Leads y x → Conductor y

-- Goal: Some orchestras premiered music pieces.
theorem some_orchestra_premiered_a_music_piece :
  ∃ o : obj, Orchestra o ∧ ∃ p : obj, MusicPiece p ∧ Premiered o p :=
by
  refine Exists.intro ViennaMusicSociety ?_
  refine And.intro T4 ?_
  refine Exists.intro SymphonyNo9 ?_
  exact And.intro T1 T3
</lean>

ANSWER: True
