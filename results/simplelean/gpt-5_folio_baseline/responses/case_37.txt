<lean>
axiom obj : Type

axiom MusicPiece : obj → Prop
axiom Composer : obj → Prop
axiom Conductor : obj → Prop
axiom Orchestra : obj → Prop

axiom Wrote : obj → obj → Prop
axiom Premiered : obj → obj → Prop
axiom Leads : obj → obj → Prop

axiom Beethoven : obj
axiom Symphony9 : obj
axiom ViennaMusicSociety : obj

-- Facts from context
axiom T1 : MusicPiece Symphony9
-- Interpreting "Composers write music pieces" in a safe direction:
axiom T2 : ∀ x y, Wrote x y → MusicPiece y
axiom T3 : Wrote Beethoven Symphony9
axiom T4 : Premiered ViennaMusicSociety Symphony9
axiom T5 : Orchestra ViennaMusicSociety
axiom T6 : Leads Beethoven ViennaMusicSociety
axiom T7 : ∀ o, Orchestra o → ∃ c, Conductor c ∧ Leads c o

theorem some_orchestra_premiered_music_pieces :
  ∃ o, Orchestra o ∧ ∃ p, MusicPiece p ∧ Premiered o p :=
by
  refine Exists.intro ViennaMusicSociety ?_
  refine And.intro T5 ?_
  refine Exists.intro Symphony9 ?_
  exact And.intro T1 T4
</lean>

ANSWER: True
