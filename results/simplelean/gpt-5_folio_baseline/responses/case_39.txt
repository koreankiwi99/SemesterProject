-- Axioms from the context
axiom A1 : ∀ x : obj, Zaha x ∧ Adores x → Interesting x
axiom A2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ Interesting x
axiom A3 : ∀ x : obj, Adores x → Zaha x ∨ Kelly x
axiom A4 : ∀ x : obj, Kelly x ∧ Adores x → Evocative x
axiom A5 : ∀ x : obj, Kelly x ∧ Adores x → Dreamy x
axiom A6 : ∀ x : obj, DesignByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- A useful derived fact: there cannot be a design by Max that he adores which is interesting
theorem no_adored_interesting_design_by_max :
  ∀ x : obj, DesignByMax x → Adores x → Interesting x → False := by
  intro x Dx Ax Ix
  have h : DesignByMax x ∧ Adores x ∧ Interesting x := And.intro Dx (And.intro Ax Ix)
  have hBE : Brutalist x ∧ Evocative x := A6 x h
  have hB : Brutalist x := And.left hBE
  have hNotI : ¬ Interesting x := A2 x (And.intro hB Ax)
  exact hNotI Ix
