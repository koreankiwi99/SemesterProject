axiom obj : Type
axiom DesignByMax : obj → Prop
axiom Adores : obj → Prop
axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Brutalist : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom H1 : ∀ x : obj, Zaha x ∧ Adores x → Interesting x
-- No brutalist buildings that Max adores have interesting geometries.
axiom H2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ Interesting x
-- Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom H3 : ∀ x : obj, Adores x → Zaha x ∨ Kelly x
-- All of Kelly Wearstler's design styles that Max adores are evocative.
axiom H4 : ∀ x : obj, Kelly x ∧ Adores x → Evocative x
-- All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom H5 : ∀ x : obj, Kelly x ∧ Adores x → Dreamy x
-- If a design by Max that he adores has interesting geometries, then it is a brutalist building and evocative.
axiom H6 : ∀ x : obj, DesignByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- Useful derived fact: any design by Max that he adores is evocative.
theorem adored_design_by_max_evocative : ∀ x : obj, DesignByMax x ∧ Adores x → Evocative x := by
  intro x
  intro h
  cases h with
  | intro hdm ha =>
    have hz_or_k : Zaha x ∨ Kelly x := H3 x ha
    cases hz_or_k with
    | inl hz =>
      have hint : Interesting x := H1 x (And.intro hz ha)
      have hbe : Brutalist x ∧ Evocative x := H6 x (And.intro hdm (And.intro ha hint))
      exact hbe.right
    | inr hk =>
      exact H4 x (And.intro hk ha)
