<lean>
axiom obj : Type

axiom Animal : obj → Prop
axiom Mammal : obj → Prop
axiom Human : obj → Prop

axiom OccursVirus : obj → Prop
axiom CanOccurVirus : obj → Prop
axiom MayGetMonkeypox : obj → Prop
axiom HasMonkeypox : obj → Prop

axiom Fever : obj → Prop
axiom Headache : obj → Prop
axiom MusclePains : obj → Prop
axiom Tired : obj → Prop

axiom Person : obj → Prop
axiom Flu : obj → Prop

-- When the Monkeypox virus occurs in a being, it may get Monkeypox.
axiom R_occurs_may : ∀ x : obj, OccursVirus x → MayGetMonkeypox x

-- Monkeypox virus can occur in certain animals. (There exists at least one animal it can occur in.)
axiom Ex_animal_can_occur : ∃ x : obj, Animal x ∧ CanOccurVirus x

-- Humans are mammals.
axiom R_human_mammal : ∀ x : obj, Human x → Mammal x

-- Mammals are animals.
axiom R_mammal_animal : ∀ x : obj, Mammal x → Animal x

-- Symptoms of Monkeypox include fever, headache, muscle pains, and tiredness.
axiom R_symptoms : ∀ x : obj, HasMonkeypox x → Fever x ∧ Headache x ∧ MusclePains x ∧ Tired x

-- People feel tired when they get the flu.
axiom R_flu_tired : ∀ x : obj, Person x ∧ Flu x → Tired x

theorem there_is_an_animal : ∃ x : obj, Animal x :=
by
  obtain ⟨x, hx⟩ := Ex_animal_can_occur
  exact Exists.intro x hx.left
</lean>

ANSWER: True
