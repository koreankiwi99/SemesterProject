<lean>
axiom obj : Type
axiom Mike : obj

axiom InternationalStudent : obj → Prop
axiom InUS : obj → Prop
axiom HasF1 : obj → Prop
axiom HasJ1 : obj → Prop
axiom WantsWorkUS : obj → Prop
axiom NeedsApplyCPT : obj → Prop
axiom NeedsApplyOPT : obj → Prop

-- International students in the US have either an F1 visa or a J1 visa.
axiom Rule_visas :
  ∀ x : obj, InternationalStudent x ∧ InUS x → HasF1 x ∨ HasJ1 x

-- An international student in the US with an F1 visa needs to apply for CPT or OPT
-- if the student wants to work in the US.
axiom Rule_work_auth :
  ∀ x : obj, InternationalStudent x ∧ InUS x ∧ HasF1 x ∧ WantsWorkUS x →
              NeedsApplyCPT x ∨ NeedsApplyOPT x

-- Mike is an international student.
axiom Fact_MikeIntl : InternationalStudent Mike

-- Mike needs to apply for CPT if he wants to work in the US.
axiom Fact_MikeCPT_if_work : WantsWorkUS Mike → NeedsApplyCPT Mike

-- From these premises, we cannot derive HasF1 Mike nor its negation.
-- We lack that Mike is in the US, and even with it, Rule_visas only yields HasF1 Mike ∨ HasJ1 Mike.
-- Hence the statement "Mike has an F1 visa" is underdetermined.
