<lean>
axiom Series : Type
axiom Person : Type
axiom Title : Type

axiom Badults : Series
axiom AndrewCollins : Person
axiom TheSecretDudeSociety : Title

axiom WorkingTitle : Series → Title → Prop
axiom ScriptEditor : Person → Series → Prop

-- From the context
axiom WT_Badults : WorkingTitle Badults TheSecretDudeSociety
axiom SE_Badults : ScriptEditor AndrewCollins Badults

theorem collins_script_editor_for_series_with_secret_dude_title :
  ∃ s : Series, WorkingTitle s TheSecretDudeSociety ∧ ScriptEditor AndrewCollins s :=
by
  refine Exists.intro Badults ?h
  exact And.intro WT_Badults SE_Badults
</lean>

ANSWER: True
