<lean>
axiom obj : Type

axiom SADC : obj
axiom SirDigby : obj
axiom NemesisOf : obj → obj

axiom SuperheroMovie : obj → Prop
axiom NamedAfter : obj → obj → Prop
axiom InMovie : obj → obj → Prop
axiom GoodGuy : obj → Prop
axiom BadGuy : obj → Prop
axiom Fights : obj → obj → Prop
axiom Wins : obj → Prop

-- In superhero movies, the good guys always win.
axiom R1 : ∀ m x : obj, SuperheroMovie m → InMovie x m → GoodGuy x → Wins x
-- The Surprising Adventures of Sir Digby Chicken Caesar is a superhero movie.
axiom T1 : SuperheroMovie SADC
-- The Surprising Adventures of Sir Digby Chicken Caesar is named after Sir Digby.
axiom T2 : NamedAfter SADC SirDigby
-- If a superhero movie is named after a character, that character is a good guy.
axiom R2 : ∀ m x : obj, SuperheroMovie m → NamedAfter m x → GoodGuy x
-- If a superhero movie is named after a character, that character is in the movie.
axiom R3 : ∀ m x : obj, SuperheroMovie m → NamedAfter m x → InMovie x m
-- Good guys fight bad guys and vice versa.
axiom R4a : ∀ x y : obj, GoodGuy x ∧ BadGuy y → Fights x y
axiom R4b : ∀ x y : obj, BadGuy x ∧ GoodGuy y → Fights x y
-- Sir Digby fights his nemesis.
axiom T3 : Fights SirDigby (NemesisOf SirDigby)
-- If somebody wins a fight, the person they are fighting does not win.
axiom R5 : ∀ x y : obj, Fights x y → Wins x → ¬ Wins y

theorem nemesis_not_win : ¬ Wins (NemesisOf SirDigby) :=
by
  have hGood : GoodGuy SirDigby := R2 SADC SirDigby T1 T2
  have hIn : InMovie SirDigby SADC := R3 SADC SirDigby T1 T2
  have hWin : Wins SirDigby := R1 SADC SirDigby T1 hIn hGood
  have hFight : Fights SirDigby (NemesisOf SirDigby) := T3
  exact R5 SirDigby (NemesisOf SirDigby) hFight hWin
</lean>

ANSWER: True
