<lean>
axiom obj : Type
axiom MachineTranslation : obj

axiom NLPTask : obj → Prop
axiom LanguageGeneration : obj → Prop
axiom LanguageUnderstanding : obj → Prop
axiom OutputTextSequence : obj → Prop

-- Categorization as exclusive: every NLP task is either generation or understanding, but not both
axiom categorization :
  ∀ x : obj,
    NLPTask x →
    (LanguageGeneration x ∧ ¬ LanguageUnderstanding x) ∨
    (LanguageUnderstanding x ∧ ¬ LanguageGeneration x)

-- If the output is a text sequence, then it is a language generation task
axiom gen_of_text_output : ∀ x : obj, OutputTextSequence x → LanguageGeneration x

-- Machine translation facts
axiom MT_is_task : NLPTask MachineTranslation
axiom MT_outputs_text : OutputTextSequence MachineTranslation

-- From the above, machine translation is not a language understanding task
theorem mt_not_understanding : ¬ LanguageUnderstanding MachineTranslation := by
  have genMT : LanguageGeneration MachineTranslation :=
    gen_of_text_output MachineTranslation MT_outputs_text
  have cat := categorization MachineTranslation MT_is_task
  cases cat with
  | inl h =>
      -- h : LanguageGeneration MT ∧ ¬ LanguageUnderstanding MT
      exact h.right
  | inr h =>
      -- h : LanguageUnderstanding MT ∧ ¬ LanguageGeneration MT, but we have genMT
      intro _
      exact h.right genMT
</lean>

ANSWER: False
