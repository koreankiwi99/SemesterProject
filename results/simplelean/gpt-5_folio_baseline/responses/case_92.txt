axiom Person : Type
axiom Animal : Type
axiom Building : Type
axiom Money : Type

-- Constants
axiom Tom : Person
axiom Fluffy : Animal
axiom OliveGarden : Building
axiom M2000 : Money
axiom M1500 : Money

-- Predicates and relations
axiom Managed : Building → Prop
axiom AllowsPets : Building → Prop
axiom Pet : Animal → Prop
axiom Cat : Animal → Prop
axiom OwnedBy : Animal → Person → Prop
axiom AllowedMoveInWithPet : Person → Animal → Building → Prop
axiom MonthlyRent : Building → Money → Prop
axiom SecurityDeposit : Building → Money → Prop
axiom WillRentIn : Person → Building → Prop

-- Order relations on Money
axiom GT : Money → Money → Prop
axiom LE : Money → Money → Prop
axiom gt_not_le : ∀ x y, GT x y → ¬ LE x y
axiom gt_trans : ∀ a b c, GT a b → GT b c → GT a c

-- Facts from the context
axiom managed_og : Managed OliveGarden
axiom monthly_og : MonthlyRent OliveGarden M2000
axiom gt_2000_1500 : GT M2000 M1500

-- A deposit is required to rent an apartment in a managed building.
axiom deposit_required : ∀ b, Managed b → ∃ d, SecurityDeposit b d

-- The security deposit is either equal to the monthly rent of that building or more.
axiom deposit_policy :
  ∀ b r d, Managed b → MonthlyRent b r → SecurityDeposit b d → (d = r ∨ GT d r)

-- Pets are allowed in some managed buildings.
axiom pets_allowed_some : ∃ b, Managed b ∧ AllowsPets b

-- If a managed building allows pets, people are allowed to move in with a pet.
axiom pet_move_rule :
  ∀ b, Managed b → AllowsPets b → ∀ p a, Pet a → AllowedMoveInWithPet p a b

-- Cats are pets.
axiom cats_are_pets : ∀ a, Cat a → Pet a

-- Fluffy is Tom's cat.
axiom fluffy_cat : Cat Fluffy
axiom fluffy_owned_by_tom : OwnedBy Fluffy Tom

-- Tom will rent an apartment in a managed building if he is allowed to move in with Fluffy,
-- and the security deposit is no more than $1500.
axiom tom_rents_if :
  ∀ b d, Managed b →
         AllowedMoveInWithPet Tom Fluffy b →
         SecurityDeposit b d →
         LE d M1500 →
         WillRentIn Tom b

-- Lemma: Any security deposit at The Olive Garden is not ≤ $1500.
theorem og_deposit_not_le : ∀ d, SecurityDeposit OliveGarden d → ¬ LE d M1500 := by
  intro d hdep
  have h := deposit_policy OliveGarden M2000 d managed_og monthly_og hdep
  cases h with
  | inl heq =>
      have hgt : GT d M1500 := by simpa [heq] using gt_2000_1500
      exact gt_not_le d M1500 hgt
  | inr hgt_d_2000 =>
      have hgt : GT d M1500 := gt_trans d M2000 M1500 hgt_d_2000 gt_2000_1500
      exact gt_not_le d M1500 hgt

-- Target statement: WillRentIn Tom OliveGarden.
-- From the given axioms, we cannot prove WillRentIn Tom OliveGarden,
-- nor can we prove its negation, since "tom_rents_if" is only a sufficient
-- condition and we have no axioms giving necessary conditions for renting.
-- Hence the truth value of WillRentIn Tom OliveGarden is undecidable from the context.
</lean>
