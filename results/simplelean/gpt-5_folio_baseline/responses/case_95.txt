axiom BusinessOrganization : obj → Prop
axiom LegalEntity : obj → Prop
axiom Company : obj → Prop
axiom PrivateCompany : obj → Prop
axiom CreatedUnderLaw : obj → Prop
axiom HasLegalObligations : obj → Prop

-- All business organizations are legal entities.
axiom BO_to_LE : ∀ x : obj, BusinessOrganization x → LegalEntity x
-- All companies are business organizations.
axiom C_to_BO : ∀ x : obj, Company x → BusinessOrganization x
-- All private companies are companies.
axiom PC_to_C : ∀ x : obj, PrivateCompany x → Company x
-- All legal entities are created under law.
axiom LE_to_CUL : ∀ x : obj, LegalEntity x → CreatedUnderLaw x
-- All legal entities have legal obligations.
axiom LE_to_HO : ∀ x : obj, LegalEntity x → HasLegalObligations x
-- If H is created under law, then it is not a private company.
axiom H_CUL_implies_not_PC : CreatedUnderLaw H → ¬ PrivateCompany H

-- We can derive that H is not a private company.
theorem H_not_private : ¬ PrivateCompany H := by
  intro hpc
  have hc : Company H := PC_to_C H hpc
  have hb : BusinessOrganization H := C_to_BO H hc
  have hl : LegalEntity H := BO_to_LE H hb
  have hcul : CreatedUnderLaw H := LE_to_CUL H hl
  have hnp : ¬ PrivateCompany H := H_CUL_implies_not_PC hcul
  exact hnp hpc
