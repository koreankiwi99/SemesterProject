axiom Person : Type
axiom Luke : Person

axiom Baby : Person → Prop
axiom MultipleBirth : Person → Prop
axiom SpendTimeWithSiblings : Person → Prop
axiom HaveSiblingsBornTogether : Person → Prop
axiom ComplainOftenAboutSiblings : Person → Prop
axiom LiveAtHome : Person → Prop
axiom LiveWithStrangers : Person → Prop

-- Rules from the context
axiom R1 : ∀ x : Person, MultipleBirth x → SpendTimeWithSiblings x
axiom R2 : ∀ x : Person, HaveSiblingsBornTogether x → MultipleBirth x
axiom R3 : ∀ x : Person, ComplainOftenAboutSiblings x → HaveSiblingsBornTogether x
axiom R4 : ∀ x : Person, LiveAtHome x → ¬ LiveWithStrangers x
axiom R5 : ∀ x : Person, SpendTimeWithSiblings x → LiveAtHome x

-- Disjunction about Luke
axiom Luke_disj :
  (Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨
  (¬ Baby Luke ∧ ¬ MultipleBirth Luke ∧ ¬ LiveWithStrangers Luke)

-- From the rules we can show that no one can both be born in a multiple birth and live with strangers
theorem not_MB_and_LWS (x : Person) : ¬ (MultipleBirth x ∧ LiveWithStrangers x) := by
  intro h
  have hmb : MultipleBirth x := h.left
  have hlws : LiveWithStrangers x := h.right
  have hsp : SpendTimeWithSiblings x := R1 x hmb
  have hhome : LiveAtHome x := R5 x hsp
  have hnstr : ¬ LiveWithStrangers x := R4 x hhome
  exact hnstr hlws

-- Hence the first disjunct about Luke is impossible
theorem not_first :
  ¬ (Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) := by
  intro h
  have hmb : MultipleBirth Luke := h.right.left
  have hlws : LiveWithStrangers Luke := h.right.right
  exact (not_MB_and_LWS Luke) ⟨hmb, hlws⟩

-- So the second disjunct must hold: Luke is not a baby, not born in a multiple birth, and does not live with strangers
theorem second_holds :
  (¬ Baby Luke ∧ ¬ MultipleBirth Luke ∧ ¬ LiveWithStrangers Luke) := by
  cases Luke_disj with
  | inl h1 => exact False.elim (not_first h1)
  | inr h2 => exact h2
