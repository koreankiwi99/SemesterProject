<lean>
axiom obj : Type
axiom Luke : obj

axiom MultipleBirth : obj → Prop
axiom SpendTimeWithSiblings : obj → Prop
axiom HaveSiblingsBornTogether : obj → Prop
axiom ComplainsOftenAboutAnnoyingSiblings : obj → Prop
axiom LiveAtHome : obj → Prop
axiom LiveWithStrangers : obj → Prop

-- People born in a multiple birth spend a lot of time with their siblings
axiom A1 : ∀ x : obj, MultipleBirth x → SpendTimeWithSiblings x
-- If people have siblings who were born together, then they were born in a multiple birth
axiom A2 : ∀ x : obj, HaveSiblingsBornTogether x → MultipleBirth x
-- If people complain often about annoying siblings, then they have siblings who were born together
axiom A3 : ∀ x : obj, ComplainsOftenAboutAnnoyingSiblings x → HaveSiblingsBornTogether x
-- If people live at home, then they do not live with strangers
axiom A4 : ∀ x : obj, LiveAtHome x → ¬ LiveWithStrangers x
-- If people spend a lot of time with their siblings, then they often live at home
axiom A5 : ∀ x : obj, SpendTimeWithSiblings x → LiveAtHome x

-- Luke either is born in a multiple birth and lives with strangers,
-- or is not born in a multiple birth and does not live with strangers
axiom A6 : (MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨ (¬ MultipleBirth Luke ∧ ¬ LiveWithStrangers Luke)

theorem luke_neither : ¬ MultipleBirth Luke ∧ ¬ ComplainsOftenAboutAnnoyingSiblings Luke := by
  have contra_first : ¬ (MultipleBirth Luke ∧ LiveWithStrangers Luke) := by
    intro h
    have hMB : MultipleBirth Luke := h.left
    have hSpend : SpendTimeWithSiblings Luke := A1 Luke hMB
    have hHome : LiveAtHome Luke := A5 Luke hSpend
    have hNotStrangers : ¬ LiveWithStrangers Luke := A4 Luke hHome
    exact hNotStrangers h.right
  have hSecond : (¬ MultipleBirth Luke ∧ ¬ LiveWithStrangers Luke) := by
    cases A6 with
    | inl h1 => exact False.elim (contra_first h1)
    | inr h2 => exact h2
  have hNotMB : ¬ MultipleBirth Luke := hSecond.left
  have notCompl : ¬ ComplainsOftenAboutAnnoyingSiblings Luke := by
    intro hC
    have hHas : HaveSiblingsBornTogether Luke := A3 Luke hC
    have hMB2 : MultipleBirth Luke := A2 Luke hHas
    exact hNotMB hMB2
  exact And.intro hNotMB notCompl
</lean>

ANSWER: True
