<lean>
axiom obj : Type
axiom Tom : obj

axiom GREApplicant : obj → Prop
axiom ProvedHardship : obj → Prop
axiom FinancialAid : obj → Prop
axiom SingleParentFamily : obj → Prop
axiom FewResources : obj → Prop

-- ETS provides financial aid to GRE applicants who prove economic hardship
axiom R1 : ∀ x : obj, GREApplicant x → ProvedHardship x → FinancialAid x
-- Living in a single-parent family or having few resources implies one can prove economic hardship
axiom R2 : ∀ x : obj, Or (SingleParentFamily x) (FewResources x) → ProvedHardship x

-- Facts about Tom
axiom T1 : SingleParentFamily Tom
axiom T2 : FewResources Tom
axiom T3 : GREApplicant Tom

theorem ets_provides_financial_aid_to_tom : FinancialAid Tom :=
by
  have hhard : ProvedHardship Tom := R2 Tom (Or.inl T1)
  exact R1 Tom T3 hhard
</lean>

ANSWER: True
