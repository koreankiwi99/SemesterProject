<lean>
-- Entities
axiom person : Type
axiom Tom : person

-- Predicates
axiom SingleParentFamily : person → Prop
axiom FewResources : person → Prop
axiom ApplyingGRE : person → Prop
axiom ProvesEconomicHardship : person → Prop
axiom ReceivesETSAid : person → Prop

-- Rules from context
axiom R_hardship : ∀ x : person, (SingleParentFamily x ∨ FewResources x) → ProvesEconomicHardship x
axiom R_aid : ∀ x : person, (ApplyingGRE x ∧ ProvesEconomicHardship x) → ReceivesETSAid x

-- Facts about Tom
axiom T_single : SingleParentFamily Tom
axiom T_resources : FewResources Tom
axiom T_applying : ApplyingGRE Tom

-- From the first sentence about cost
axiom CheaperThan300 : Prop  -- the GRE test cost is cheaper than $300
axiom Below300 : Prop        -- the GRE test cost is below $300
axiom R_cheaper_implies_below : CheaperThan300 → Below300
axiom T_cheaper : CheaperThan300

-- Irrelevant but derivable aid fact (not required for the question)
theorem tom_gets_aid : ReceivesETSAid Tom := by
  have hhard : ProvesEconomicHardship Tom := by
    apply R_hardship Tom
    exact Or.inl T_single
  apply R_aid Tom
  exact And.intro T_applying hhard

-- Target theorem
theorem gre_cost_below_300 : Below300 := by
  apply R_cheaper_implies_below
  exact T_cheaper
</lean>

ANSWER: True
