axiom obj : Type
axiom Fir : obj → Prop
axiom Evergreen : obj → Prop
axiom Worship : obj → Prop

-- All fir trees are evergreens.
axiom A1 : ∀ x : obj, Fir x → Evergreen x
-- Some objects of worship are fir trees.
axiom A2 : ∃ x : obj, Worship x ∧ Fir x

-- From the given information, we can derive that some objects of worship are evergreens.
theorem some_worship_are_evergreens : ∃ x : obj, Evergreen x ∧ Worship x := by
  apply Exists.elim A2
  intro x hx
  have hxE : Evergreen x := A1 x hx.right
  exact Exists.intro x ⟨hxE, hx.left⟩

-- ∃ x : obj, Evergreen x ∧ ¬ Worship x
