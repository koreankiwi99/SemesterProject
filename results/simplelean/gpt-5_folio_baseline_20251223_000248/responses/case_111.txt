<lean>
axiom obj : Type
axiom PicurisMountains : obj
axiom JuanDeOnate : obj
axiom HardingPegmatiteMine : obj

axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

-- Picuris Mountains are a mountain range, and are in New Mexico or Texas
axiom A_MR : MountainRange PicurisMountains
axiom A_inNM_or_TX : InNewMexico PicurisMountains ∨ InTexas PicurisMountains

-- Juan de Onate visited the Picuris Mountains
axiom A_Visited : Visited JuanDeOnate PicurisMountains

-- The Harding Pegmatite Mine is a mine located in the Picuris Mountains and was donated
axiom A_Mine : Mine HardingPegmatiteMine
axiom A_Located : LocatedIn HardingPegmatiteMine PicurisMountains
axiom A_Donated : Donated HardingPegmatiteMine

-- No Texas mountain range has a mine that has been donated
axiom A_noTX :
  ∀ r : obj, MountainRange r → InTexas r → ¬ (∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m)

theorem onate_visited_mr_in_nm :
  ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ Visited JuanDeOnate r := by
  have existsMine : ∃ m : obj, Mine m ∧ LocatedIn m PicurisMountains ∧ Donated m := by
    exact Exists.intro HardingPegmatiteMine (And.intro A_Mine (And.intro A_Located A_Donated))
  have notTX : ¬ InTexas PicurisMountains := by
    intro htx
    have h := A_noTX PicurisMountains A_MR htx
    exact h existsMine
  have inNM : InNewMexico PicurisMountains := by
    cases A_inNM_or_TX with
    | inl hnm => exact hnm
    | inr htx => exact False.elim (notTX htx)
  exact Exists.intro PicurisMountains (And.intro A_MR (And.intro inNM A_Visited))
</lean>

ANSWER: True
