-- No Texas mountain range has a donated mine located in it
axiom no_TX_range_with_donated_mine :
  ∀ x : obj, MountainRange x → InTX x → ¬ (∃ m : obj, Mine m ∧ LocatedIn m x ∧ Donated m)

-- From the donated mine in Picuris, deduce Picuris is not in Texas
theorem not_inTX_Picuris : ¬ InTX Picuris :=
  fun htx =>
    have hx : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m :=
      Exists.intro Harding (And.intro Harding_mine (And.intro Harding_loc Harding_don))
    have hcontra := no_TX_range_with_donated_mine Picuris PM_mr htx
    hcontra hx

-- Using the disjunction and ¬ InTX, we can deduce Picuris is in New Mexico
theorem inNM_Picuris : InNM Picuris :=
  Or.elim PM_inNM_or_TX
    (fun h => h)
    (fun htx => False.elim (not_inTX_Picuris htx))

-- We know Juan visited Picuris and Picuris is not in Texas
