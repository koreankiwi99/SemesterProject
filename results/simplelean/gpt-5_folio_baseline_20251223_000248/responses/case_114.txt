  ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Michael can block any shooting from a person as long as the person does not jump when shooting.

  ∀ x : obj, ¬ JumpsWhenShooting x → CanBlockShooting Michael x

  ∀ x : obj, Shooter x → JumpsWhenShooting x → GreatShooter x

  intro y hy hneq
