<lean>
axiom Person : Type
axiom Michael Peter Windy : Person
axiom Man : Person → Prop
axiom Taller : Person → Person → Prop
axiom Block : Person → Person → Prop
axiom JumpsWhenShooting : Person → Prop
axiom Shooter : Person → Prop
axiom GreatShooter : Person → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R_block_taller : ∀ x y : Person, Man x → Man y → Taller x y → Block x y

-- Michael is a man who is taller than everyone else in his class.
axiom Man_Michael : Man Michael
axiom InMichaelsClass : Person → Prop
axiom Michael_taller_classmates : ∀ y : Person, InMichaelsClass y ∧ y ≠ Michael → Taller Michael y

-- Transitivity of taller
axiom Taller_trans : ∀ x y z : Person, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom Man_Peter : Man Peter
axiom Peter_taller_Michael : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom Michael_blocks_if_not_jump : ∀ p : Person, ¬ JumpsWhenShooting p → Block Michael p

-- Michael cannot block Windy's shooting.
axiom Not_block_Michael_Windy : ¬ Block Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom Great_if_shooter_and_jump : ∀ p : Person, Shooter p → JumpsWhenShooting p → GreatShooter p

-- From mentioning Windy's shooting, we take Windy to be a shooter.
axiom Shooter_Windy : Shooter Windy

open classical

theorem windy_great : GreatShooter Windy :=
  -- First, deduce that Windy jumps when shooting:
  have h1 : ¬ JumpsWhenShooting Windy → Block Michael Windy := Michael_blocks_if_not_jump Windy
  have hJump : JumpsWhenShooting Windy := by
    by_contra hNotJump
    exact Not_block_Michael_Windy (h1 hNotJump)
  -- Apply the rule that a shooter who can jump when shooting is great:
  Great_if_shooter_and_jump Windy Shooter_Windy hJump
</lean>

ANSWER: True

<lean>
axiom Person : Type
axiom Michael : Person
axiom Peter : Person
axiom Windy : Person
axiom Man : Person → Prop
axiom Taller : Person → Person → Prop
axiom Block : Person → Person → Prop
axiom JumpsWhenShooting : Person → Prop
axiom Shooter : Person → Prop
axiom GreatShooter : Person → Prop

-- Law of excluded middle (classical logic)
axiom em : ∀ p : Prop, p ∨ ¬ p

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R_block_taller : ∀ x y : Person, Man x → Man y → Taller x y → Block x y

-- Michael is a man who is taller than everyone else in his class.
axiom Man_Michael : Man Michael
axiom InMichaelsClass : Person → Prop
axiom Michael_taller_classmates : ∀ y : Person, InMichaelsClass y ∧ y ≠ Michael → Taller Michael y

-- Transitivity of taller
axiom Taller_trans : ∀ x y z : Person, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom Man_Peter : Man Peter
axiom Peter_taller_Michael : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom Michael_blocks_if_not_jump : ∀ p : Person, ¬ JumpsWhenShooting p → Block Michael p

-- Michael cannot block Windy's shooting.
axiom Not_block_Michael_Windy : ¬ Block Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom Great_if_shooter_and_jump : ∀ p : Person, Shooter p → JumpsWhenShooting p → GreatShooter p

-- Windy is a shooter (since we talk about Windy's shooting).
axiom Shooter_Windy : Shooter Windy

theorem windy_great : GreatShooter Windy :=
  let h1 : ¬ JumpsWhenShooting Windy → Block Michael Windy := Michael_blocks_if_not_jump Windy
  let hJump : JumpsWhenShooting Windy :=
    match em (JumpsWhenShooting Windy) with
    | Or.inl hj => hj
    | Or.inr hnot => False.elim (Not_block_Michael_Windy (h1 hnot))
  Great_if_shooter_and_jump Windy Shooter_Windy hJump
</lean>

ANSWER: True
