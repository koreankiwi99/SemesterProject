axiom Block : obj → obj → Prop
axiom InClassOfMichael : obj → Prop
axiom JumpsWhenShooting : obj → Prop
axiom GreatShooter : obj → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R_block_if_taller_men : ∀ x y : obj, Man x → Man y → Taller x y → Block x y

-- If x taller than y and y taller than z, then x taller than z.
axiom R_taller_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Michael is a man who is taller than everyone else in his class.
axiom T_michael_man : Man Michael
axiom T_michael_taller_class : ∀ y : obj, InClassOfMichael y → Taller Michael y

-- Peter is a man who is taller than Michael.
axiom T_peter_man : Man Peter
axiom T_peter_taller_michael : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom R_michael_blocks_nonjumpers : ∀ p : obj, ¬ JumpsWhenShooting p → Block Michael p

-- Michael cannot block Windy's shooting.
axiom T_michael_cannot_block_windy : ¬ Block Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom R_jump_implies_great : ∀ s : obj, JumpsWhenShooting s → GreatShooter s

-- Some derivable facts:

-- Peter can block Michael (since Peter is a man, Michael is a man, and Peter is taller than Michael).

  · exact T_peter_man
  · exact T_michael_man
  · exact T_peter_taller_michael
