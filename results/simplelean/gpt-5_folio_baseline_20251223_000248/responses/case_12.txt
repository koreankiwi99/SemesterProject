<lean>
axiom Game : Type
axiom Company : Type

axiom Zelda : Game

axiom Japanese : Company → Prop
axiom Created : Company → Game → Prop
axiom Top10 : Game → Prop
axiom MillionSales : Game → Prop

-- A Japanese game company created The Legend of Zelda.
axiom H1 : ∃ c : Company, Japanese c ∧ Created c Zelda
-- All games on the Top 10 list are made by Japanese game companies.
axiom H2 : ∀ g : Game, Top10 g → ∃ c : Company, Japanese c ∧ Created c g
-- If a game sells more than one million copies, then it will be included in the Top 10 list.
axiom H3 : ∀ g : Game, MillionSales g → Top10 g
-- The Legend of Zelda sold more than one million copies.
axiom H4 : MillionSales Zelda

theorem zelda_on_top10 : Top10 Zelda := by
  exact H3 Zelda H4
</lean>

ANSWER: True
