axiom Person : Type
axiom John : Person

axiom MakesPlans : Person → Prop
axiom Engaged : Person → Prop
axiom InvitesOthers : Person → Prop
axiom InvitesFriends : Person → Prop
axiom WellAttended : Person → Prop
axiom LargerFamily : Person → Prop

-- 1. All people who make wedding plans are engaged.
axiom A1 : ∀ x : Person, MakesPlans x → Engaged x
-- 2. All people who invite others to their ceremony make wedding plans.
axiom A2 : ∀ x : Person, InvitesOthers x → MakesPlans x
-- 3. Anyone who has a well-attended wedding invites others to their ceremony.
axiom A3 : ∀ x : Person, WellAttended x → InvitesOthers x
-- 4. Some people that have well-attended weddings have larger families.
axiom A4 : ∃ x : Person, WellAttended x ∧ LargerFamily x
-- 5. If John is engaged, then not (John invites friends to their ceremony and makes wedding plans).
axiom A5 : Engaged John → ¬ (InvitesFriends John ∧ MakesPlans John)
-- 6. If John has a larger family, then either John has a well-attended wedding or invites friends to their ceremony.
axiom A6 : LargerFamily John → (WellAttended John ∨ InvitesFriends John)

lemma well_attended_implies_makes_plans (x : Person) :
  WellAttended x → MakesPlans x :=
  fun hw => A2 x (A3 x hw)

lemma well_attended_implies_engaged (x : Person) :
  WellAttended x → Engaged x :=
  fun hw => A1 x (well_attended_implies_makes_plans x hw)

lemma john_not_invites_friends_if_well_attended :
  WellAttended John → ¬ InvitesFriends John :=
  fun hw =>
    let m : MakesPlans John := well_attended_implies_makes_plans John hw
    let e : Engaged John := A1 John m
    let nf : ¬ (InvitesFriends John ∧ MakesPlans John) := A5 e
    show ¬ InvitesFriends John from
      fun hf => nf ⟨hf, m⟩

-- We cannot derive either LargerFamily John or its negation from the given axioms.
