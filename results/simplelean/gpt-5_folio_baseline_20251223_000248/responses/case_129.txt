axiom Engaged : Person → Prop

axiom LargerFamily : Person → Prop
axiom InviteFriends : Person → Prop

-- 1. All people who make wedding plans are engaged.
axiom A1 : ∀ x : Person, MakePlans x → Engaged x
-- 2. All people who invite others to come to their ceremony make wedding plans.
axiom A2 : ∀ x : Person, InviteOthers x → MakePlans x
-- 3. Anyone who has a well-attended wedding invites others to come to their ceremony.
axiom A3 : ∀ x : Person, WellAttended x → InviteOthers x
-- 4. Some people that have well-attended weddings have larger families.
axiom A4 : ∃ x : Person, WellAttended x ∧ LargerFamily x
-- 5. If John is engaged, then John does not invite friends to their ceremony and makes wedding plans.
axiom A5 : Engaged John → (¬ InviteFriends John ∧ MakePlans John)
-- 6. If John has a larger family, then either John has a well-attended wedding or invites friends to their ceremony.
axiom A6 : LargerFamily John → (WellAttended John ∨ InviteFriends John)

-- Useful derived facts:
lemma wellAttended_implies_engaged : ∀ x : Person, WellAttended x → Engaged x :=
  fun x hx => A1 x (A2 x (A3 x hx))

lemma largerFamilyJohn_implies_engaged_or_inviteFriends :
  LargerFamily John → (Engaged John ∨ InviteFriends John) :=
  fun h => Or.elim (A6 h)
    (fun hw => Or.inl (wellAttended_implies_engaged John hw))
    (fun hf => Or.inr hf)

  | inl hw =>
      exact Or.inl (wellAttended_implies_engaged John hw)
