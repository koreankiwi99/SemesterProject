-- A Japanese game company created the game the Legend of Zelda.
axiom Zelda_made_by_japanese_exists : ∃ c : Company, Japanese c ∧ MadeBy Zelda c

-- All games on the Top 10 list are made by Japanese game companies
-- (formalized as: each Top10 game has some Japanese company that made it).
axiom Top10_made_by_japanese_exists :
  ∀ g : Game, Top10 g → ∃ c : Company, Japanese c ∧ MadeBy g c

-- If a game sells more than one million copies, then it is included in the Top 10 list.
axiom sales_imp_top10 : ∀ g : Game, SoldOver1M g → Top10 g

-- The Legend of Zelda sold more than one million copies.
axiom zelda_sold_over_1m : SoldOver1M Zelda

-- Deductions about Zelda (not directly needed for the question but derivable):

  ∃ c : Company, Japanese c ∧ MadeBy Zelda c :=
  Top10_made_by_japanese_exists Zelda zelda_top10
