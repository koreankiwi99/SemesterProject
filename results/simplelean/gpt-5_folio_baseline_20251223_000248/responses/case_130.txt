<lean>
axiom Person : Type
axiom John : Person

axiom MakePlans : Person → Prop
axiom Engaged : Person → Prop
axiom InviteOthers : Person → Prop
axiom InviteFriends : Person → Prop
axiom WellAttended : Person → Prop
axiom LargerFamily : Person → Prop

-- All people who make wedding plans are engaged.
axiom A1 : ∀ x : Person, MakePlans x → Engaged x
-- All people who invite others to come to their ceremony make wedding plans.
axiom A2 : ∀ x : Person, InviteOthers x → MakePlans x
-- Anyone who has a well-attended wedding invites others to come to their ceremony.
axiom A3 : ∀ x : Person, WellAttended x → InviteOthers x
-- Some people that have well-attended weddings have larger families.
axiom A4 : ∃ x : Person, WellAttended x ∧ LargerFamily x
-- If John is engaged, then John does not invite friends to their ceremony and does not make wedding plans.
axiom A5 : Engaged John → (¬ InviteFriends John ∧ ¬ MakePlans John)
-- If John has a larger family, then either John has a well-attended wedding or invites friends to their ceremony.
axiom A6 : LargerFamily John → (WellAttended John ∨ InviteFriends John)

-- From A1 and A5 we can show John cannot make wedding plans.
theorem not_makeplans_John : ¬ MakePlans John := by
  intro hmp
  have heng : Engaged John := A1 John hmp
  have hnm : ¬ MakePlans John := (A5 heng).right
  exact hnm hmp

-- Hence John does not invite others to his ceremony.
theorem not_inviteothers_John : ¬ InviteOthers John := by
  intro hio
  have hmp : MakePlans John := A2 John hio
  exact not_makeplans_John hmp

-- Consequently, John does not have a well-attended wedding.
theorem not_wellattended_John : ¬ WellAttended John := by
  intro hwa
  have hio : InviteOthers John := A3 John hwa
  exact not_inviteothers_John hio

-- From LargerFamily John, only InviteFriends John can follow (since WellAttended is impossible),
-- but this does not yield a contradiction, so we cannot refute LargerFamily John.
theorem large_implies_invitefriends_John : LargerFamily John → InviteFriends John := by
  intro hlf
  have hw_or_if : WellAttended John ∨ InviteFriends John := A6 hlf
  have hnowa : ¬ WellAttended John := not_wellattended_John
  cases hw_or_if with
  | inl hwa => exact False.elim (hnowa hwa)
  | inr hf  => exact hf
