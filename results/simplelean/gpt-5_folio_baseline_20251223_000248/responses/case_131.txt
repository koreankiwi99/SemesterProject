<lean>
axiom obj : Type

axiom Mammal : obj → Prop
axiom Platypus : obj → Prop
axiom Human : obj → Prop
axiom Teeth : obj → Prop

-- Some mammals have teeth.
axiom E1 : ∃ x : obj, Mammal x ∧ Teeth x

-- Platypuses have no teeth.
axiom R1 : ∀ x : obj, Platypus x → ¬ Teeth x

-- Platypuses are mammals.
axiom R2 : ∀ x : obj, Platypus x → Mammal x

-- Humans have teeth.
axiom R3 : ∀ x : obj, Human x → Teeth x

theorem platypuses_mammals_no_teeth : ∀ x : obj, Platypus x → (Mammal x ∧ ¬ Teeth x) :=
by
  intro x hx
  apply And.intro
  · exact R2 x hx
  · exact R1 x hx
</lean>

ANSWER: True
