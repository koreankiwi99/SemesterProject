<lean>
axiom obj : Type
axiom Space : obj
axiom Duster : obj

axiom Plunger : obj → Prop
axiom Vampire : obj → Prop
axiom Vacuum : obj → Prop
axiom HouseholdAppliance : obj → Prop
axiom Sucks : obj → Prop

-- Plungers suck.
axiom R1 : ∀ x : obj, Plunger x → Sucks x
-- Vacuums suck.
axiom R2 : ∀ x : obj, Vacuum x → Sucks x
-- Vampires suck.
axiom R3 : ∀ x : obj, Vampire x → Sucks x

-- Space is a vacuum.
axiom T1 : Vacuum Space

-- A duster is a household appliance that doesn't suck.
axiom T2 : HouseholdAppliance Duster
axiom T3 : ¬ Sucks Duster

theorem space_sucks : Sucks Space := by
  have hv : Vacuum Space := T1
  exact R2 Space hv
</lean>

ANSWER: True
