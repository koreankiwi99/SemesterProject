<lean>
axiom obj : Type
axiom Plunger : obj → Prop
axiom Vacuum : obj → Prop
axiom Vampire : obj → Prop
axiom Space : obj
axiom Duster : obj
axiom HouseholdAppliance : obj → Prop
axiom Suck : obj → Prop

-- Plungers suck. Vacuums suck. Vampires suck.
axiom A1 : ∀ x : obj, Plunger x → Suck x
axiom A2 : ∀ x : obj, Vacuum x → Suck x
axiom A3 : ∀ x : obj, Vampire x → Suck x

-- Space is a vacuum.
axiom A4 : Vacuum Space

-- A duster is a household appliance that doesn't suck.
axiom A5 : HouseholdAppliance Duster
axiom A6 : ¬ Suck Duster

-- The statement "If something is a household appliance, it sucks" is false,
-- since Duster is a counterexample.
theorem not_all_household_appliance_suck : ¬ (∀ x : obj, HouseholdAppliance x → Suck x) := by
  intro h
  have : Suck Duster := h Duster A5
  exact A6 this
</lean>

ANSWER: False
