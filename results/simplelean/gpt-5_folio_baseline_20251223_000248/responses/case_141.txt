-- Axioms from the context
axiom A1 : ∀ x : obj, AppropriateAllAges x → CanWatchWithoutParents x
axiom A2 : ∀ x : obj, ExcessiveEroticViolentContent x → ¬ CanWatchWithoutParents x
axiom A3 : ∀ x : obj, RatedGeneralAudience x → AppropriateAllAges x
axiom A4 : ∀ x : obj, FamilyFriendlyAnimated x → RatedGeneralAudience x
axiom A5 : ∀ x : obj, FrozenSeries x → FamilyFriendlyAnimated x

axiom TFilmH : Film Hachi
axiom Hachi_FFA_or_App : FamilyFriendlyAnimated Hachi ∨ AppropriateAllAges Hachi

-- From the disjunction, Hachi is watchable by children without parents
theorem Hachi_can_watch : CanWatchWithoutParents Hachi := by
  cases Hachi_FFA_or_App with
  | inl hffa =>
      have hga : RatedGeneralAudience Hachi := A4 Hachi hffa
      have happ : AppropriateAllAges Hachi := A3 Hachi hga
      exact A1 Hachi happ
  | inr happ =>
      exact A1 Hachi happ

-- Therefore, Hachi does not contain excessive erotic and violent content

  intro hex
  have hw : CanWatchWithoutParents Hachi := Hachi_can_watch
  have hnot : ¬ CanWatchWithoutParents Hachi := A2 Hachi hex
  exact hnot hw
