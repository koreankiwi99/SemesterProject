<lean>
axiom obj : Type

-- Predicates for entities and properties
axiom Animal : obj → Prop
axiom HeckCattle : obj → Prop
axiom Aurochs : obj → Prop
axiom ExtinctAnimal : obj → Prop
axiom Resembles : obj → obj → Prop
axiom ToBeBredBack : obj → Prop
axiom BredBack : obj → Prop
axiom ArtificiallySelected : obj → Prop

-- From "Breeding back is a form of artificial selection"
axiom R_bredback_imp_artificial : ∀ x : obj, BredBack x → ArtificiallySelected x

-- Heck cattle and aurochs are animals
axiom T_heck_animals : ∀ x : obj, HeckCattle x → Animal x
axiom T_aurochs_animals : ∀ x : obj, Aurochs x → Animal x

-- Some animals to be bred back resemble extinct animals
axiom T_some_animals_to_be_bred_back_resemble_extinct :
  ∃ x : obj, Animal x ∧ ToBeBredBack x ∧ (∃ y : obj, ExtinctAnimal y ∧ Resembles x y)

-- Heck cattle were bred back (modeled as: any Heck cattle were bred back)
axiom T_heck_bredback : ∀ x : obj, HeckCattle x → BredBack x

-- Existence of some Heck cattle
axiom T_exists_heck : ∃ x : obj, HeckCattle x

-- Goal: Some Heck cattle are artificially selected
theorem some_heck_cattle_artificially_selected :
  ∃ x : obj, HeckCattle x ∧ ArtificiallySelected x :=
by
  obtain ⟨x, hxHC⟩ := T_exists_heck
  have hxBB : BredBack x := T_heck_bredback x hxHC
  have hxAS : ArtificiallySelected x := R_bredback_imp_artificial x hxBB
  exact ⟨x, And.intro hxHC hxAS⟩
</lean>

ANSWER: True
