-- All organic compounds contain carbon.

-- No chemical compounds contain only one element.
axiom A5 : ∀ x : obj, ChemicalCompound x → ¬ OnlyOneElement x
-- The mixture either both is a chemical compound and contains only one element, or neither.
axiom A6 : (ChemicalCompound Mixture ∧ OnlyOneElement Mixture) ∨ (¬ ChemicalCompound Mixture ∧ ¬ OnlyOneElement Mixture)

-- From A5 and A6, the first disjunct in A6 is impossible.
lemma mixture_not_both : ¬ (ChemicalCompound Mixture ∧ OnlyOneElement Mixture) := by
  intro h
  have hno : ¬ OnlyOneElement Mixture := A5 Mixture h.left
  exact hno h.right

-- Therefore the mixture is neither a chemical compound nor does it contain only one element.
lemma mixture_neither : ¬ ChemicalCompound Mixture ∧ ¬ OnlyOneElement Mixture := by
  cases A6 with
  | inl hboth =>
      exact False.elim (mixture_not_both hboth)
  | inr hneither =>
      exact hneither

  exact mixture_neither.left

theorem mixture_not_one_element : ¬ OnlyOneElement Mixture := by
  exact mixture_neither.right
