axiom obj : Type
axiom Mixture : obj
axiom Hydrocarbon : obj → Prop
axiom Organic : obj → Prop
axiom Alkane : obj → Prop
axiom ChemicalCompound : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom OnlyOneElement : obj → Prop

axiom all_hc_org : ∀ x : obj, Hydrocarbon x → Organic x
axiom all_alk_hc : ∀ x : obj, Alkane x → Hydrocarbon x
axiom all_org_cc : ∀ x : obj, Organic x → ChemicalCompound x
axiom all_org_contain_carbon : ∀ x : obj, Organic x → ContainsCarbon x
axiom no_cc_only_one : ∀ x : obj, ChemicalCompound x → ¬ OnlyOneElement x
axiom mixture_both_or_neither :
  (ChemicalCompound Mixture ∧ OnlyOneElement Mixture) ∨
  (¬ ChemicalCompound Mixture ∧ ¬ OnlyOneElement Mixture)

lemma not_one_mixture : ¬ OnlyOneElement Mixture := by
  intro hone
  have h := mixture_both_or_neither
  cases h with
  | inl hboth =>
      have hcc : ChemicalCompound Mixture := hboth.left
      have hnotone : ¬ OnlyOneElement Mixture := no_cc_only_one Mixture hcc
      exact hnotone hone
  | inr hneither =>
      have hnotone : ¬ OnlyOneElement Mixture := hneither.right
      exact hnotone hone

lemma not_cc_mixture : ¬ ChemicalCompound Mixture := by
  intro hcc
  have h := mixture_both_or_neither
  cases h with
  | inl hboth =>
      have hone : OnlyOneElement Mixture := hboth.right
      have hnotone : ¬ OnlyOneElement Mixture := no_cc_only_one Mixture hcc
      exact hnotone hone
  | inr hneither =>
      have hnotcc : ¬ ChemicalCompound Mixture := hneither.left
      exact hnotcc hcc

lemma not_alkane_mixture : ¬ Alkane Mixture := by
  intro halk
  have hhc : Hydrocarbon Mixture := all_alk_hc Mixture halk
  have horg : Organic Mixture := all_hc_org Mixture hhc
  have hcc : ChemicalCompound Mixture := all_org_cc Mixture horg
  exact not_cc_mixture hcc

theorem mixture_statement :
  (OnlyOneElement Mixture ∨ ContainsCarbon Mixture) →
  (¬ ChemicalCompound Mixture ∧ ¬ Alkane Mixture) := by
  intro h
  apply And.intro
  · exact not_cc_mixture
  · exact not_alkane_mixture
</lean>

axiom obj : Type

axiom Hydrocarbon : obj → Prop
axiom Organic : obj → Prop
axiom Alkane : obj → Prop
axiom ChemicalCompound : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom OnlyOneElement : obj → Prop

axiom all_alk_hc : ∀ x : obj, Alkane x → Hydrocarbon x
axiom all_org_cc : ∀ x : obj, Organic x → ChemicalCompound x
axiom all_org_contain_carbon : ∀ x : obj, Organic x → ContainsCarbon x
axiom no_cc_only_one : ∀ x : obj, ChemicalCompound x → ¬ OnlyOneElement x
axiom mixture_both_or_neither :
  (ChemicalCompound Mixture ∧ OnlyOneElement Mixture) ∨
  (¬ ChemicalCompound Mixture ∧ ¬ OnlyOneElement Mixture)

lemma neither_mixture : (¬ ChemicalCompound Mixture ∧ ¬ OnlyOneElement Mixture) := by
  have h := mixture_both_or_neither
  cases h
  case inl hboth =>
    have hcc : ChemicalCompound Mixture := hboth.left
    have hone : OnlyOneElement Mixture := hboth.right
    have f : False := (no_cc_only_one Mixture hcc) hone
    exact False.elim f
  case inr hneither =>
    exact hneither

lemma not_one_mixture : ¬ OnlyOneElement Mixture :=
  neither_mixture.right

lemma not_cc_mixture : ¬ ChemicalCompound Mixture :=
  neither_mixture.left

lemma not_alkane_mixture : ¬ Alkane Mixture := by
  intro halk
  have hhc : Hydrocarbon Mixture := all_alk_hc Mixture halk
  have horg : Organic Mixture := all_hc_org Mixture hhc
  have hcc : ChemicalCompound Mixture := all_org_cc Mixture horg
  exact not_cc_mixture hcc

theorem mixture_statement :
  (OnlyOneElement Mixture ∨ ContainsCarbon Mixture) →
  (¬ ChemicalCompound Mixture ∧ ¬ Alkane Mixture) := by
  intro _
  exact And.intro not_cc_mixture not_alkane_mixture
