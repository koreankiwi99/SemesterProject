<lean>
axiom obj : Type
axiom Student : obj → Prop
axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom HasUniPTJob : obj → Prop
axiom TakesDB : obj → Prop
axiom TakesWithDavid : obj → Prop
axiom WorksLab : obj → Prop
axiom James : obj

-- None of the students who currently work in the library are from the CS department.
axiom A1 : ∀ x : obj, Student x ∧ WorksLibrary x → ¬ FromCS x
-- All students with part-time jobs offered by the university currently work in the library.
axiom A2 : ∀ x : obj, Student x ∧ HasUniPTJob x → WorksLibrary x
-- All students who are taking the database course are from the CS department.
axiom A3 : ∀ x : obj, Student x ∧ TakesDB x → FromCS x
-- All students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, Student x ∧ TakesWithDavid x → TakesDB x
-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, Student x ∧ WorksLab x → TakesWithDavid x

-- James is a student working in the lab.
axiom T1 : Student James
axiom T2 : WorksLab James
-- James doesn't work in the lab or have a part-time job offered by the university.
axiom T3 : ¬ (WorksLab James ∨ HasUniPTJob James)

-- Target: Is "James has a part-time job offered by the university" true?
-- We prove its negation.
theorem james_no_uni_pt_job : ¬ HasUniPTJob James := by
  intro hjob
  have sj : Student James := T1
  have wlab : WorksLab James := T2
  have hDavid : TakesWithDavid James := A5 James (And.intro sj wlab)
  have hDB : TakesDB James := A4 James (And.intro sj hDavid)
  have hCS : FromCS James := A3 James (And.intro sj hDB)
  have wlib : WorksLibrary James := A2 James (And.intro sj hjob)
  have notCS : ¬ FromCS James := A1 James (And.intro sj wlib)
  exact notCS hCS
</lean>

ANSWER: False
