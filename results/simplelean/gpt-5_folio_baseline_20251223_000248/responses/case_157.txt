<lean>
axiom Person : Type
axiom James : Person

axiom Student : Person → Prop
axiom WorksLibrary : Person → Prop
axiom FromCS : Person → Prop
axiom HasUniPTJob : Person → Prop
axiom TakesDB : Person → Prop
axiom TakesWithDavid : Person → Prop
axiom WorksLab : Person → Prop

-- 1. None of the students who currently work in the library are from the CS department.
axiom A1 : ∀ x : Person, Student x ∧ WorksLibrary x → ¬ FromCS x
-- 2. All students with university-offered part-time jobs work in the library.
axiom A2 : ∀ x : Person, Student x ∧ HasUniPTJob x → WorksLibrary x
-- 3. All students taking the database course are from the CS department.
axiom A3 : ∀ x : Person, Student x ∧ TakesDB x → FromCS x
-- 4. All students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : Person, Student x ∧ TakesWithDavid x → TakesDB x
-- 5. All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : Person, Student x ∧ WorksLab x → TakesWithDavid x

-- 6. James is a student working in the lab.
axiom SJ : Student James
axiom LJ : WorksLab James

-- 7. James doesn't work in the lab or have a university-offered part-time job.
-- We formalize the natural language as a disjunction of negatives (not needed for the proof).
axiom Amb : ¬ WorksLab James ∨ ¬ HasUniPTJob James

theorem james_no_unijob : ¬ HasUniPTJob James :=
by
  intro hjob
  have hstudent : Student James := SJ
  have hlab : WorksLab James := LJ
  have hdavid : TakesWithDavid James := A5 James ⟨hstudent, hlab⟩
  have hdb : TakesDB James := A4 James ⟨hstudent, hdavid⟩
  have hcs : FromCS James := A3 James ⟨hstudent, hdb⟩
  have hlib : WorksLibrary James := A2 James ⟨hstudent, hjob⟩
  have hnotcs : ¬ FromCS James := A1 James ⟨hstudent, hlib⟩
  exact hnotcs hcs
</lean>

ANSWER: True
