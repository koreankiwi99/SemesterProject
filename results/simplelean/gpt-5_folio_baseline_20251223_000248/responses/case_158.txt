<lean>
axiom obj : Type
axiom James : obj

axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom UniPartTimeJob : obj → Prop
axiom TakesDBCourse : obj → Prop
axiom TakesClassWithDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- None who work in the library are from CS
axiom A1 : ∀ x : obj, WorksInLibrary x → ¬ FromCS x
-- All with university part-time jobs work in the library
axiom A2 : ∀ x : obj, UniPartTimeJob x → WorksInLibrary x
-- All taking the database course are from CS
axiom A3 : ∀ x : obj, TakesDBCourse x → FromCS x
-- All taking a class with Professor David are taking the database course
axiom A4 : ∀ x : obj, TakesClassWithDavid x → TakesDBCourse x
-- All working in the lab are taking a class with Professor David
axiom A5 : ∀ x : obj, WorksInLab x → TakesClassWithDavid x
-- James works in the lab
axiom A6 : WorksInLab James
-- James doesn't work in the lab nor have a university part-time job
axiom A7a : ¬ WorksInLab James
axiom A7b : ¬ UniPartTimeJob James

theorem james_db_or_uniPT : TakesDBCourse James ∨ UniPartTimeJob James :=
by
  have hD : TakesClassWithDavid James := A5 James A6
  have hDB : TakesDBCourse James := A4 James hD
  exact Or.inl hDB
</lean>

ANSWER: True
