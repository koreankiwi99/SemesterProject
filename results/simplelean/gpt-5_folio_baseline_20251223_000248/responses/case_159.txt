<lean>
axiom obj : Type
axiom James : obj

axiom WorkLib : obj → Prop
axiom FromCS : obj → Prop
axiom PTU : obj → Prop
axiom DB : obj → Prop
axiom WithDavid : obj → Prop
axiom WorkLab : obj → Prop

-- None of the students who currently work in the library are from the CS department.
axiom A1 : ∀ x : obj, WorkLib x → ¬ FromCS x
-- All students with part-time university jobs currently work in the library.
axiom A2 : ∀ x : obj, PTU x → WorkLib x
-- All students taking the database course are from the CS department.
axiom A3 : ∀ x : obj, DB x → FromCS x
-- All students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, WithDavid x → DB x
-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, WorkLab x → WithDavid x

-- James is a student working in the lab.
axiom T1 : WorkLab James
-- James doesn't work in the lab or have a part-time job offered by the university.
-- Interpreted as: not working in the lab and not having a PTU job.
axiom T2 : ¬ WorkLab James ∧ ¬ PTU James

theorem james_takes_database : DB James :=
by
  have hwd : WithDavid James := A5 James T1
  exact A4 James hwd
</lean>

ANSWER: True
