axiom WantsCompose : obj → Prop
axiom HasAccessProgram : obj → Prop
axiom CanCompose : obj → Prop
axiom WillCompose : obj → Prop

-- If George likes music, he wants to compose.
axiom A1 : LikesMusic George → WantsCompose George
-- If George has access to a program, he can compose.
axiom A2 : HasAccessProgram George → CanCompose George
-- If George can compose and wants to compose, he will compose.
axiom A3 : CanCompose George ∧ WantsCompose George → WillCompose George

-- From A3, we can derive: if George will not compose, then it's not the case that he both can and wants to compose.
theorem not_will_implies_not_can_and_not_want :
  (¬ WillCompose George) → ¬ (CanCompose George ∧ WantsCompose George) := by
  intro hNotWill
  intro hCanAndWant
  have hWill : WillCompose George := A3 hCanAndWant
  exact hNotWill hWill
