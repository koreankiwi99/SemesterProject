<lean>
axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

-- A hawk never swims
axiom A1 : ∀ x : obj, Hawk x → ¬ Swims x
-- Some birds are hawks
axiom A2 : ∃ x : obj, Bird x ∧ Hawk x

-- Therefore, it is not the case that all birds swim
theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swims x) := by
  intro h
  cases A2 with
  | intro w hw =>
    have bw : Bird w := And.left hw
    have hk : Hawk w := And.right hw
    have hs : Swims w := h w bw
    have hns : ¬ Swims w := A1 w hk
    exact hns hs
</lean>

ANSWER: False
