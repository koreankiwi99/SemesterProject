<lean>
axiom obj : Type
axiom HeinrichSchmidt : obj
axiom GermanPolitician : obj → Prop
axiom MemberPrussianStateParliament : obj → Prop
axiom MemberNaziReichstag : obj → Prop

axiom HS_gp : GermanPolitician HeinrichSchmidt
axiom HS_psp : MemberPrussianStateParliament HeinrichSchmidt
axiom HS_nr : MemberNaziReichstag HeinrichSchmidt

theorem exists_german_politician_in_both :
  ∃ x : obj, GermanPolitician x ∧ MemberPrussianStateParliament x ∧ MemberNaziReichstag x := by
  exists HeinrichSchmidt
  constructor
  · exact HS_gp
  · constructor
    · exact HS_psp
    · exact HS_nr
</lean>

ANSWER: True
