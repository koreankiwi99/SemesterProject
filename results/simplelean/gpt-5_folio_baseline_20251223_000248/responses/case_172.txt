<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatsSaladsRegularly : obj → Prop
axiom VeryConscious : obj → Prop
axiom GrewUpHealthConsciousHome : obj → Prop
axiom FulfillsDailyIntakes : obj → Prop
axiom DisregardsPhysicalWellBeing : obj → Prop
axiom VisitsGymDaily : obj → Prop

-- All people who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : obj, EatsSaladsRegularly x → VeryConscious x
-- All people who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : obj, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
-- All people who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : obj, FulfillsDailyIntakes x → GrewUpHealthConsciousHome x
-- All people who disregard their physical well-being are not very conscious about their health and eating habits.
axiom A4 : ∀ x : obj, DisregardsPhysicalWellBeing x → ¬ VeryConscious x
-- If people visit the gym at least once a day, then they always fulfill their daily nutritional intakes.
axiom A5 : ∀ x : obj, VisitsGymDaily x → FulfillsDailyIntakes x

-- Taylor either (grew up health-conscious and disregards) or (neither).
axiom Taylor_disj :
  (GrewUpHealthConsciousHome Taylor ∧ DisregardsPhysicalWellBeing Taylor) ∨
  (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardsPhysicalWellBeing Taylor)

-- No one can both have grown up in a health-conscious home and disregard physical well-being.
theorem no_G_and_D : ∀ x : obj, ¬ (GrewUpHealthConsciousHome x ∧ DisregardsPhysicalWellBeing x) :=
fun x h =>
  have hxG : GrewUpHealthConsciousHome x := And.left h
  have hxD : DisregardsPhysicalWellBeing x := And.right h
  have hxEat : EatsSaladsRegularly x := A2 x hxG
  have hxVC : VeryConscious x := A1 x hxEat
  have hxNotVC : ¬ VeryConscious x := A4 x hxD
  show False from hxNotVC hxVC

-- Therefore, Taylor must be in the "neither" case.
theorem Taylor_neither : ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardsPhysicalWellBeing Taylor :=
Or.elim Taylor_disj
  (fun hGD => False.elim (no_G_and_D Taylor hGD))
  (fun hneither => hneither)
