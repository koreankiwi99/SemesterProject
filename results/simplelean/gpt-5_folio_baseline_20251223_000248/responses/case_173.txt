<lean>
axiom Person : Type
axiom Taylor : Person

axiom EatsSaladsRegularly : Person → Prop
axiom VeryConscious : Person → Prop
axiom GrewUpHealthConsciousHome : Person → Prop
axiom DisregardWellbeing : Person → Prop
axiom FulfillsDailyNutrition : Person → Prop
axiom VisitsGymDaily : Person → Prop

-- All people who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : Person, EatsSaladsRegularly x → VeryConscious x
-- All people who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : Person, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
-- All people who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : Person, FulfillsDailyNutrition x → GrewUpHealthConsciousHome x
-- All people who disregard their physical well-being are not very conscious about their health and eating habits.
axiom A4 : ∀ x : Person, DisregardWellbeing x → ¬ VeryConscious x
-- If people visit the gym at least once a day, then they always fulfill their daily nutritional intakes.
axiom A5 : ∀ x : Person, VisitsGymDaily x → FulfillsDailyNutrition x

-- Taylor either (grew up in a health-conscious childhood home and disregards her physical well-being) or she did neither.
axiom Taylor_case : (GrewUpHealthConsciousHome Taylor ∧ DisregardWellbeing Taylor) ∨ (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardWellbeing Taylor)

theorem taylor_not_gym : ¬ VisitsGymDaily Taylor := by
  -- From growing up in a health-conscious home, one is very conscious (via salads)
  have Hchain : ∀ x : Person, GrewUpHealthConsciousHome x → VeryConscious x := by
    intro x hx
    exact A1 x (A2 x hx)
  -- Do case analysis on Taylor's disjunction
  cases Taylor_case with
  | inl h =>
    -- h : GrewUp ... ∧ Disregard ...
    have hG : GrewUpHealthConsciousHome Taylor := And.left h
    have hD : DisregardWellbeing Taylor := And.right h
    have hC : VeryConscious Taylor := Hchain Taylor hG
    have hNotC : ¬ VeryConscious Taylor := A4 Taylor hD
    have hFalse : False := hNotC hC
    intro _
    exact hFalse
  | inr h =>
    -- h : ¬ GrewUp ... ∧ ¬ Disregard ...
    have hNotG : ¬ GrewUpHealthConsciousHome Taylor := And.left h
    intro hGym
    have hFul : FulfillsDailyNutrition Taylor := A5 Taylor hGym
    have hG : GrewUpHealthConsciousHome Taylor := A3 Taylor hFul
    exact hNotG hG
</lean>

ANSWER: False
