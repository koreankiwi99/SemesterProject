<lean>
axiom Person : Type
axiom Taylor : Person

axiom EatsSaladsRegularly : Person → Prop
axiom VeryConscious : Person → Prop
axiom GrewUpHC : Person → Prop
axiom FulfillDaily : Person → Prop
axiom DisregardWellBeing : Person → Prop
axiom VisitGymDaily : Person → Prop

axiom A1 : ∀ x : Person, EatsSaladsRegularly x → VeryConscious x
axiom A2 : ∀ x : Person, GrewUpHC x → EatsSaladsRegularly x
axiom A3 : ∀ x : Person, FulfillDaily x → GrewUpHC x
axiom A4 : ∀ x : Person, DisregardWellBeing x → ¬ VeryConscious x
axiom A5 : ∀ x : Person, VisitGymDaily x → FulfillDaily x
axiom A6 : (GrewUpHC Taylor ∧ DisregardWellBeing Taylor) ∨ (¬ GrewUpHC Taylor ∧ ¬ DisregardWellBeing Taylor)

theorem taylor_neither_grew_nor_visit : (¬ GrewUpHC Taylor) ∧ (¬ VisitGymDaily Taylor) := by
  -- Show (GrewUpHC ∧ Disregard) leads to contradiction (conscious and not conscious)
  have h_not_G_and_D : ¬ (GrewUpHC Taylor ∧ DisregardWellBeing Taylor) := by
    intro h
    have hg : GrewUpHC Taylor := h.left
    have hd : DisregardWellBeing Taylor := h.right
    have hc : VeryConscious Taylor := A1 Taylor (A2 Taylor hg)
    have hnc : ¬ VeryConscious Taylor := A4 Taylor hd
    exact hnc hc
  -- From the exclusive-style premise, deduce ¬GrewUpHC ∧ ¬Disregard
  have h_notG_notD : (¬ GrewUpHC Taylor) ∧ (¬ DisregardWellBeing Taylor) := by
    cases A6 with
    | inl hGD =>
        exact False.elim (h_not_G_and_D hGD)
    | inr hNN =>
        exact hNN
  have h_notG : ¬ GrewUpHC Taylor := h_notG_notD.left
  -- VisitGymDaily → FulfillDaily → GrewUpHC, so with ¬GrewUpHC we get ¬VisitGymDaily
  have v_to_g : VisitGymDaily Taylor → GrewUpHC Taylor := fun hv => A3 Taylor (A5 Taylor hv)
  have h_notV : ¬ VisitGymDaily Taylor := by
    intro hv
    apply h_notG
    exact v_to_g hv
  exact And.intro h_notG h_notV
</lean>

ANSWER: True
