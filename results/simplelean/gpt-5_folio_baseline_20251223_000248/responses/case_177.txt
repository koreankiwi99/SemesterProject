<lean>
axiom obj : Type
axiom LastSummerOlympics : obj
axiom TokyoOlympics : obj
axiom UnitedStates : obj

axiom WonMostMedalsIn : obj → obj → Prop

-- The last summer Olympic games was in Tokyo.
axiom last_is_tokyo : LastSummerOlympics = TokyoOlympics
-- The United States won the most medals in Tokyo.
axiom us_most_tokyo : WonMostMedalsIn UnitedStates TokyoOlympics

theorem us_most_last : WonMostMedalsIn UnitedStates LastSummerOlympics := by
  simpa [last_is_tokyo] using us_most_tokyo
</lean>

ANSWER: True
