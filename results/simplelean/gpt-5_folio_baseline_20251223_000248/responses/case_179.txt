axiom obj : Type
axiom Susan : obj
axiom John : obj
axiom LGA : obj

axiom FliesFrom : obj → obj → Prop
axiom FliesTo : obj → obj → Prop

-- Susan flies to LGA
axiom T1 : FliesTo Susan LGA
-- The departure and arrival cannot be at the same airport
axiom R1 : ∀ p a, FliesTo p a → ¬ FliesFrom p a
-- John flies from LGA (irrelevant to the question)
axiom T2 : FliesFrom John LGA

-- The statement in question would be: FliesFrom Susan LGA
-- We can prove its negation using the constraint.
theorem not_susan_from_LGA : ¬ FliesFrom Susan LGA := by
  exact R1 Susan LGA T1
