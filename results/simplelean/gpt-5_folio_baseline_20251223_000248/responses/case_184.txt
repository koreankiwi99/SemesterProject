<lean>
axiom obj : Type

-- Entities
axiom DaniShapiro : obj
axiom FamilyHistory : obj
axiom FamilySecrets : obj
axiom Boston : obj

-- Predicates
axiom Podcast : obj → Prop
axiom Novel : obj → Prop
axiom Book : obj → Prop
axiom Person : obj → Prop
axiom Writer : obj → Prop
axiom NovelWriter : obj → Prop
axiom American : obj → Prop
axiom AmericanCity : obj → Prop
axiom BornIn : obj → obj → Prop
axiom WrittenBy : obj → obj → Prop

-- Rules
axiom podcast_not_novel : ∀ x : obj, Podcast x → ¬ Novel x
axiom born_in_american_implies_american :
  ∀ p c : obj, Person p → BornIn p c → AmericanCity c → American p
axiom novel_book_written_by_person_implies_novel_writer :
  ∀ b p : obj, Book b → Novel b → WrittenBy b p → Person p → NovelWriter p

-- Facts
axiom ds_american : American DaniShapiro
axiom ds_writer : Writer DaniShapiro
axiom ds_person : Person DaniShapiro

axiom fh_written_by_ds : WrittenBy FamilyHistory DaniShapiro
axiom fh_is_novel : Novel FamilyHistory
axiom fh_is_book : Book FamilyHistory

axiom familysecrets_is_podcast : Podcast FamilySecrets
axiom boston_is_american_city : AmericanCity Boston

theorem dani_shapiro_is_novel_writer : NovelWriter DaniShapiro := by
  exact novel_book_written_by_person_implies_novel_writer
    FamilyHistory DaniShapiro fh_is_book fh_is_novel fh_written_by_ds ds_person
