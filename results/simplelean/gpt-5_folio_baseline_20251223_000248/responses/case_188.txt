<lean>
axiom obj : Type
axiom Yuri : obj

axiom ProBB : obj → Prop            -- Professional basketball player
axiom American : obj → Prop         -- American national
axiom CanJump : obj → Prop          -- Can do jump shots
axiom LeapStraight : obj → Prop     -- Leaps straight into the air
axiom ActivateLeg : obj → Prop      -- Activates leg muscles

-- Context axioms
axiom some_pro_not_american : ∃ x : obj, ProBB x ∧ ¬ American x
axiom all_pro_can_jump : ∀ x : obj, ProBB x → CanJump x
axiom jump_implies_leap : ∀ x : obj, CanJump x → LeapStraight x
axiom leap_implies_activate : ∀ x : obj, LeapStraight x → ActivateLeg x
axiom Yuri_not_activate : ¬ ActivateLeg Yuri

theorem yuri_not_american_pro : ¬ (American Yuri ∧ ProBB Yuri) := by
  intro h
  have hyPro : ProBB Yuri := h.right
  have hyCan : CanJump Yuri := all_pro_can_jump Yuri hyPro
  have hyleap : LeapStraight Yuri := jump_implies_leap Yuri hyCan
  have hyact : ActivateLeg Yuri := leap_implies_activate Yuri hyleap
  exact Yuri_not_activate hyact
</lean>

ANSWER: True
