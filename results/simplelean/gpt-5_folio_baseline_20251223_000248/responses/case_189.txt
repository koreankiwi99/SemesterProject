<lean>
axiom obj : Type
axiom Yuri : obj

axiom ProfessionalBasketballPlayer : obj → Prop
axiom American : obj → Prop
axiom CanJumpShots : obj → Prop
axiom LeapsStraightUp : obj → Prop
axiom ActivatesLegMuscles : obj → Prop

axiom SomeProNotAmerican : ∃ x : obj, ProfessionalBasketballPlayer x ∧ ¬ American x
axiom AllProsJump : ∀ x : obj, ProfessionalBasketballPlayer x → CanJumpShots x
axiom JumpImpliesLeap : ∀ x : obj, CanJumpShots x → LeapsStraightUp x
axiom LeapImpliesActivate : ∀ x : obj, LeapsStraightUp x → ActivatesLegMuscles x
axiom YuriNotActivate : ¬ ActivatesLegMuscles Yuri

theorem not_Pro_Yuri : ¬ ProfessionalBasketballPlayer Yuri := by
  intro hy
  have hjump : CanJumpShots Yuri := AllProsJump Yuri hy
  have hleap : LeapsStraightUp Yuri := JumpImpliesLeap Yuri hjump
  have hact : ActivatesLegMuscles Yuri := LeapImpliesActivate Yuri hleap
  exact YuriNotActivate hact

theorem query_neg : ¬ (American Yuri ∧ ProfessionalBasketballPlayer Yuri) := by
  intro h
  exact not_Pro_Yuri h.right
</lean>
