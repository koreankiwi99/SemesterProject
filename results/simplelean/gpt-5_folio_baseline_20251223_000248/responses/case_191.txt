<lean>
axiom obj : Type
axiom Yuri : obj

axiom American : obj → Prop
axiom Pro : obj → Prop -- professional basketball player
axiom JumpShots : obj → Prop
axiom Leap : obj → Prop
axiom Activate : obj → Prop

-- Context axioms
axiom some_pro_not_american : ∃ x : obj, Pro x ∧ ¬ American x
axiom all_pro_jump : ∀ x : obj, Pro x → JumpShots x
axiom jump_leap : ∀ x : obj, JumpShots x → Leap x
axiom leap_activate : ∀ x : obj, Leap x → Activate x
axiom yuri_not_activate : ¬ Activate Yuri

-- From the chain of implications, any pro player activates leg muscles
theorem pro_implies_activate : ∀ x : obj, Pro x → Activate x := by
  intro x
  intro hx
  have hj : JumpShots x := all_pro_jump x hx
  have hl : Leap x := jump_leap x hj
  exact leap_activate x hl

-- Hence, Yuri is not a professional basketball player
theorem not_pro_yuri : ¬ Pro Yuri := by
  intro hypro
  have ha : Activate Yuri := pro_implies_activate Yuri hypro
  exact yuri_not_activate ha

-- The statement "If Yuri is not an American professional basketball player, then Yuri is a professional basketball player" is false
theorem statement_false : ¬ ((¬ (American Yuri ∧ Pro Yuri)) → Pro Yuri) := by
  intro h
  -- From ¬ Pro Yuri, we get ¬ (American Yuri ∧ Pro Yuri)
  have hnot : ¬ (American Yuri ∧ Pro Yuri) := by
    intro hAnd
    exact not_pro_yuri hAnd.right
  -- The implication yields Pro Yuri, contradicting ¬ Pro Yuri
  have hyPro : Pro Yuri := h hnot
  exact not_pro_yuri hyPro
