axiom AiltonSilva : obj

axiom Braga : obj
axiom Nautico : obj
axiom Fluminense : obj

axiom PlaysFor : obj → obj → Prop

axiom Club_Braga : FootballClub Braga
axiom Club_Nautico : FootballClub Nautico
axiom Club_Fluminense : FootballClub Fluminense

axiom PlaysFor_Nautico : PlaysFor AiltonSilva Nautico

-- The negation of the claim "Ailton Silva does not play for a football club"
-- is that there exists a football club c such that Ailton Silva plays for c.
-- This is provable using Náutico.
theorem plays_for_some_club : ∃ c : obj, FootballClub c ∧ PlaysFor AiltonSilva c := by
  refine Exists.intro Nautico ?h
  exact And.intro Club_Nautico PlaysFor_Nautico
</lean>
