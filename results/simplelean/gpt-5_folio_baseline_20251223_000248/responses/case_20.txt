<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom Near : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop
axiom Rockie : obj

-- Axioms from the context
axiom A1 : ∀ x : obj, Rabbit x ∧ Near x → Cute x
axiom A2 : ∃ x : obj, Turtle x ∧ Near x
axiom A3 : ∀ x : obj, Near x → (Rabbit x ∨ Squirrel x)
axiom A4 : ∀ x : obj, Skittish x → ¬ Calm x
axiom A5 : ∀ x : obj, Squirrel x ∧ Near x → Skittish x
axiom A6 : Near Rockie ∧ Calm Rockie

-- Useful derived facts
theorem not_skittish_Rockie : ¬ Skittish Rockie := by
  have hCalm : Calm Rockie := And.right A6
  intro hSk
  have hNotCalm : ¬ Calm Rockie := A4 Rockie hSk
  exact hNotCalm hCalm

theorem not_squirrel_Rockie : ¬ Squirrel Rockie := by
  have hNear : Near Rockie := And.left A6
  intro hSq
  have hSk : Skittish Rockie := A5 Rockie (And.intro hSq hNear)
  exact not_skittish_Rockie hSk

theorem rabbit_Rockie : Rabbit Rockie := by
  have hNear : Near Rockie := And.left A6
  have disj : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie hNear
  have hNotSq : ¬ Squirrel Rockie := not_squirrel_Rockie
  cases disj with
  | inl hr => exact hr
  | inr hs => exact False.elim (hNotSq hs)

theorem cute_Rockie : Cute Rockie := by
  have hNear : Near Rockie := And.left A6
  have hRab : Rabbit Rockie := rabbit_Rockie
  exact A1 Rockie (And.intro hRab hNear)

-- Target: If Rockie is cute and calm, then Rockie is a skittish turtle.
-- We prove the negation of the statement.
theorem neg_target :
  ¬ ((Cute Rockie ∧ Calm Rockie) → (Skittish Rockie ∧ Turtle Rockie)) := by
  intro h
  have hCalm : Calm Rockie := And.right A6
  have hCute : Cute Rockie := cute_Rockie
  have hQ : Skittish Rockie ∧ Turtle Rockie := h (And.intro hCute hCalm)
  exact not_skittish_Rockie hQ.left
</lean>

ANSWER: False
