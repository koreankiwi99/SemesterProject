axiom obj : Type

axiom Ailton : obj
axiom Braga : obj
axiom Nautico : obj
axiom Fluminense : obj

axiom FootballPlayer : obj → Prop
axiom BrazilianFootballer : obj → Prop
axiom FootballClub : obj → Prop
axiom LoanedOutTo : obj → obj → Prop
axiom PlaysFor : obj → obj → Prop

axiom T1 : FootballPlayer Ailton
axiom T2 : LoanedOutTo Ailton Braga
axiom T3 : BrazilianFootballer Ailton
axiom T4 : PlaysFor Ailton Nautico
axiom T5 : FootballClub Nautico
axiom T6 : FootballClub Braga
axiom T7 : FootballClub Fluminense

-- The statement to evaluate: "Ailton was not loaned out to a football club."
-- This would be: ¬ ∃ c, FootballClub c ∧ LoanedOutTo Ailton c
-- We prove its negation by exhibiting Braga as such a club.

theorem exists_club_loaned : ∃ c, FootballClub c ∧ LoanedOutTo Ailton c := by
  refine Exists.intro Braga ?h
  apply And.intro
  · exact T6
  · exact T2

theorem negate_statement : ¬ (¬ ∃ c, FootballClub c ∧ LoanedOutTo Ailton c) := by
  intro h
  exact h exists_club_loaned

-- Alternatively, derive a direct contradiction from the statement itself:
theorem statement_false : (¬ ∃ c, FootballClub c ∧ LoanedOutTo Ailton c) → False := by
  intro h
  exact h exists_club_loaned
