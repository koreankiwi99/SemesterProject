<lean>
axiom Show : Type
axiom ST : Show
axiom BM : Show

axiom NetflixShow : Show → Prop
axiom Popular : Show → Prop
axiom Binge : Show → Prop        -- Karen binge-watches the show
axiom Download : Show → Prop     -- Karen downloads the show
axiom ShareWithLisa : Show → Prop -- Karen shares the show with Lisa

-- "Stranger Things" is a popular Netflix show.
axiom ST_Netflix : NetflixShow ST
axiom ST_Popular : Popular ST

-- If a Netflix show is popular, Karen will binge-watch it.
axiom R1 : ∀ s : Show, NetflixShow s → Popular s → Binge s

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom R2 : ∀ s : Show, NetflixShow s → (Binge s ↔ Download s)

-- "Black Mirror" is a Netflix show and Karen does not download it.
axiom BM_Netflix : NetflixShow BM
axiom BM_not_download : ¬ Download BM

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom R3 : ∀ s : Show, NetflixShow s → Binge s → ShareWithLisa s

theorem share_ST : ShareWithLisa ST := by
  have hB : Binge ST := R1 ST ST_Netflix ST_Popular
  exact R3 ST ST_Netflix hB
</lean>

ANSWER: True
