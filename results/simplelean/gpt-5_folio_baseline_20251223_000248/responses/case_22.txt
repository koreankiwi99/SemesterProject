<lean>
axiom Show : Type
axiom StrangerThings : Show
axiom BlackMirror : Show

axiom NetflixShow : Show → Prop
axiom Popular : Show → Prop
axiom Binge : Show → Prop
axiom Download : Show → Prop
axiom ShareWithLisa : Show → Prop

-- Facts about Stranger Things
axiom ST_netflix : NetflixShow StrangerThings
axiom ST_popular : Popular StrangerThings

-- Rules
axiom popular_implies_binge : ∀ s : Show, NetflixShow s → Popular s → Binge s
axiom iff_binge_download : ∀ s : Show, NetflixShow s → (Download s ↔ Binge s)
axiom binge_share : ∀ s : Show, NetflixShow s → Binge s → ShareWithLisa s

-- Facts about Black Mirror
axiom BM_netflix : NetflixShow BlackMirror
axiom not_download_BM : ¬ Download BlackMirror

-- Conclusion: Black Mirror is not popular
theorem black_mirror_not_popular : ¬ Popular BlackMirror := by
  intro hpop
  have hbinge : Binge BlackMirror := popular_implies_binge BlackMirror BM_netflix hpop
  have hiff : Download BlackMirror ↔ Binge BlackMirror := iff_binge_download BlackMirror BM_netflix
  have hBtoD : Binge BlackMirror → Download BlackMirror := Iff.mpr hiff
  have hdl : Download BlackMirror := hBtoD hbinge
  exact not_download_BM hdl
</lean>

ANSWER: False
