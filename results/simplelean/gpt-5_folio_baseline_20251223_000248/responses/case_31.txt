<lean>
axiom obj : Type
axiom Djokovic : obj

axiom Mixer : obj → Prop
axiom GrandSlam : obj → Prop
axiom OscarNom : obj → Prop
axiom ProfTennis : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom TaxHaven : obj → Prop

-- Context axioms
axiom A1 : ∀ x : obj, Mixer x → (GrandSlam x ∨ OscarNom x)
axiom A2 : ∀ x : obj, Mixer x → GrandSlam x → ProfTennis x
axiom A3 : ∀ x : obj, Mixer x → OscarNom x → Celebrity x
axiom A4 : ∀ x : obj, Mixer x → ProfTennis x → Athlete x
axiom A5 : ∀ x : obj, Mixer x → Celebrity x → WellPaid x
axiom A6 : ∀ x : obj, Mixer x → Athlete x → Famous x
axiom A7 : ∀ x : obj, Mixer x → WellPaid x → TaxHaven x

-- Djokovic-specific facts
axiom A8 : Mixer Djokovic
axiom A9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

theorem djokovic_taxhaven : TaxHaven Djokovic := by
  have hM : Mixer Djokovic := A8
  have hdisj : GrandSlam Djokovic ∨ OscarNom Djokovic := A1 Djokovic hM
  cases hdisj with
  | inl hGS =>
      have hPT : ProfTennis Djokovic := A2 Djokovic hM hGS
      have hAth : Athlete Djokovic := A4 Djokovic hM hPT
      have hFam : Famous Djokovic := A6 Djokovic hM hAth
      have hWP : WellPaid Djokovic := A9 (And.intro hFam hAth)
      exact A7 Djokovic hM hWP
  | inr hON =>
      have hCel : Celebrity Djokovic := A3 Djokovic hM hON
      have hWP : WellPaid Djokovic := A5 Djokovic hM hCel
      exact A7 Djokovic hM hWP
</lean>

ANSWER: True
