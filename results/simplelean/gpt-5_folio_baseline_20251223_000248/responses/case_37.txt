<lean>
axiom obj : Type

axiom Symphony9 : obj
axiom Beethoven : obj
axiom ViennaMusicSociety : obj

axiom MusicPiece : obj → Prop
axiom Composer : obj → Prop
axiom Orchestra : obj → Prop
axiom Wrote : obj → obj → Prop
axiom Premiered : obj → obj → Prop
axiom Leads : obj → obj → Prop
axiom Conductor : obj → Prop

axiom H1 : MusicPiece Symphony9
axiom H2 : ∀ p m, Composer p → Wrote p m → MusicPiece m
axiom H3 : Wrote Beethoven Symphony9
axiom H4 : Premiered ViennaMusicSociety Symphony9
axiom H5 : Orchestra ViennaMusicSociety
axiom H6 : Leads Beethoven ViennaMusicSociety
axiom H7 : ∀ p o, Leads p o → Orchestra o → Conductor p

theorem some_orchestras_premiered_music_pieces :
  ∃ o, Orchestra o ∧ ∃ m, MusicPiece m ∧ Premiered o m := by
  apply Exists.intro ViennaMusicSociety
  apply And.intro
  · exact H5
  · apply Exists.intro Symphony9
    apply And.intro
    · exact H1
    · exact H4
</lean>

ANSWER: True
