<lean>
axiom obj : Type

axiom DesignByMax : obj → Prop
axiom Adored : obj → Prop
axiom ZahaStyle : obj → Prop
axiom KellyStyle : obj → Prop
axiom Style : obj → Prop
axiom Brutalist : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom A1 : ∀ x : obj, ZahaStyle x ∧ Adored x → Interesting x

-- No brutalist buildings that Max adores have interesting geometries.
axiom A2 : ∀ x : obj, Brutalist x ∧ Adored x → ¬ Interesting x

-- Every style that Max adores is either Zaha Hadid's style or Kelly Wearstler's style.
axiom A3 : ∀ x : obj, Style x ∧ Adored x → (ZahaStyle x ∨ KellyStyle x)

-- All of Kelly Wearstler's design styles that Max adores are evocative.
axiom A4 : ∀ x : obj, KellyStyle x ∧ Adored x → Evocative x

-- All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom A5 : ∀ x : obj, KellyStyle x ∧ Adored x → Dreamy x

-- If a design by Max that he adores has interesting geometries, then it is a brutalist building and evocative.
axiom A6 : ∀ x : obj, DesignByMax x ∧ Adored x ∧ Interesting x → Brutalist x ∧ Evocative x

-- A useful consequence: no design by Max that he adores can have interesting geometries.
theorem no_interesting_adored_design :
  ∀ x : obj, DesignByMax x → Adored x → ¬ Interesting x :=
by
  intro x hxDesign hxAdored hxInteresting
  have hBE : Brutalist x ∧ Evocative x :=
    A6 x (And.intro hxDesign (And.intro hxAdored hxInteresting))
  have hBrut : Brutalist x := And.left hBE
  have hNotInt : ¬ Interesting x := A2 x (And.intro hBrut hxAdored)
  exact hNotInt hxInteresting
