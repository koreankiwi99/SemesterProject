<lean>
axiom obj : Type
axiom Coco : obj

axiom RankedHighlyWTA : obj → Prop
axiom MostActiveGS : obj → Prop
axiom LostToIgaRG2022 : obj → Prop
axiom LostToRafaRG2022 : obj → Prop
axiom PlayerRG2022 : obj → Prop
axiom FemaleRG2022 : obj → Prop
axiom MaleRG2022 : obj → Prop

-- If someone is ranked highly by the WTA, then they are among the most active Grand-Slam players.
axiom R1 : ∀ x : obj, RankedHighlyWTA x → MostActiveGS x
-- Everyone who lost to Iga Świątek at Roland Garros 2022 is ranked highly by the WTA.
axiom R2 : ∀ x : obj, LostToIgaRG2022 x → RankedHighlyWTA x
-- All female tennis players at Roland Garros 2022 lost to Iga Świątek.
axiom R3 : ∀ x : obj, FemaleRG2022 x → LostToIgaRG2022 x
-- All tennis players at Roland Garros 2022 are either female or male.
axiom R4 : ∀ x : obj, PlayerRG2022 x → (FemaleRG2022 x ∨ MaleRG2022 x)
-- All male tennis players at Roland Garros 2022 lost to Rafael Nadal.
axiom R5 : ∀ x : obj, MaleRG2022 x → LostToRafaRG2022 x
-- If Coco is ranked highly by the WTA or lost to Rafael Nadal, then Coco is not a male tennis player at Roland Garros 2022.
axiom R6 : (RankedHighlyWTA Coco ∨ LostToRafaRG2022 Coco) → ¬ MaleRG2022 Coco
-- Coco Gauff is at Roland Garros 2022.
axiom T1 : PlayerRG2022 Coco

theorem coco_most_active : MostActiveGS Coco := by
  have hdisj : FemaleRG2022 Coco ∨ MaleRG2022 Coco := R4 Coco T1
  cases hdisj with
  | inl hf =>
      have hIga : LostToIgaRG2022 Coco := R3 Coco hf
      have hWTA : RankedHighlyWTA Coco := R2 Coco hIga
      exact R1 Coco hWTA
  | inr hm =>
      have hRafa : LostToRafaRG2022 Coco := R5 Coco hm
      have hNotMale : ¬ MaleRG2022 Coco := R6 (Or.inr hRafa)
      exact False.elim (hNotMale hm)
</lean>

ANSWER: True
