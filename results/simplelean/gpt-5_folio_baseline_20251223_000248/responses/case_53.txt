<lean>
axiom obj : Type

axiom French : obj
axiom Spanish : obj
axiom German : obj
axiom Basque : obj

axiom Romance : obj → Prop
axiom IndoEuropean : obj → Prop
axiom Related : obj → obj → Prop

-- All Romance languages are Indo-European languages.
axiom A1 : ∀ x : obj, Romance x → IndoEuropean x

-- Romance languages are a language family (captured via relatedness within the family).
axiom A2 : ∀ x y : obj, Romance x → Romance y → Related x y

-- French and Spanish are Romance languages.
axiom A3 : Romance French
axiom A4 : Romance Spanish

-- German is related to Spanish.
axiom A5 : Related German Spanish

-- Basque is not related to any other language.
axiom A6 : ∀ x : obj, x ≠ Basque → ¬ Related Basque x
axiom A7 : ∀ x : obj, x ≠ Basque → ¬ Related x Basque

theorem french_is_indo_european : IndoEuropean French := by
  apply A1
  exact A3
</lean>

ANSWER: True
