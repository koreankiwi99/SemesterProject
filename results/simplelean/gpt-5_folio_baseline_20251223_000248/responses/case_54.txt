<lean>
axiom obj : Type
axiom RobertLewandowski : obj
axiom BayernMunchen : obj

axiom Striker : obj → Prop
axiom SoccerPlayer : obj → Prop
axiom Player : obj → Prop
axiom Leaves : obj → obj → Prop
axiom PlaysFor : obj → obj → Prop

axiom A1 : Striker RobertLewandowski
axiom A2 : ∀ x : obj, Striker x → SoccerPlayer x
axiom A3 : Leaves RobertLewandowski BayernMunchen
axiom A4 : ∀ x t : obj, Player x → Leaves x t → ¬ PlaysFor x t

theorem RL_is_soccer_player : SoccerPlayer RobertLewandowski := by
  apply A2 RobertLewandowski
  exact A1
</lean>

ANSWER: True
