<lean>
axiom obj : Type

-- Entities
axiom Billings : obj
axiom Montana : obj
axiom Butte : obj
axiom Helena : obj
axiom Missoula : obj
axiom White_Sulphur_Springs : obj
axiom St_Pierre : obj
axiom Bristol : obj
axiom Texarkana : obj
axiom Texhoma : obj
axiom Union_City : obj

-- Predicates and relations
axiom City : obj → Prop
axiom StateUS : obj → Prop
axiom In : obj → obj → Prop

-- Facts from the context
axiom T_Billings_city : City Billings
axiom T_Montana_state : StateUS Montana
axiom T_Billings_in_Montana : In Billings Montana

axiom T_Butte_city : City Butte
axiom T_Helena_city : City Helena
axiom T_Missoula_city : City Missoula
axiom T_Butte_in_Montana : In Butte Montana
axiom T_Helena_in_Montana : In Helena Montana
axiom T_Missoula_in_Montana : In Missoula Montana

axiom T_WSS_city : City White_Sulphur_Springs
axiom T_same_state_WSS_Butte : ∃ s : obj, StateUS s ∧ In White_Sulphur_Springs s ∧ In Butte s

axiom T_StPierre_city : City St_Pierre
axiom T_StPierre_not_in_Montana : ¬ In St_Pierre Montana

axiom T_not_in_both_Butte_StPierre : ∀ x : obj, In x Butte → ¬ In x St_Pierre

axiom T_unique_state_except_exceptions :
  ∀ x s1 s2,
    City x → StateUS s1 → StateUS s2 →
    In x s1 → In x s2 →
    x ≠ Bristol → x ≠ Texarkana → x ≠ Texhoma → x ≠ Union_City →
    s1 = s2

-- Claim: Montana is home to the city of Missoula, i.e., Missoula is in Montana.
theorem montana_home_missoula : In Missoula Montana :=
  T_Missoula_in_Montana
</lean>

ANSWER: True
