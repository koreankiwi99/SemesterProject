<lean>
axiom obj : Type

-- Entities
axiom Badults : obj
axiom AndrewCollins : obj
axiom TheSecretDudeSociety : obj

-- Predicates
axiom WorkingTitle : obj → obj → Prop  -- WorkingTitle series title
axiom ScriptEditor : obj → obj → Prop   -- ScriptEditor person series

-- Given facts
axiom WT_Badults : WorkingTitle Badults TheSecretDudeSociety
axiom SE_AC_Badults : ScriptEditor AndrewCollins Badults

-- Claim: There exists a series with working title "The Secret Dude Society" for which Andrew Collins was the script editor.
theorem collins_editor_of_secret_dude_series :
  ∃ s : obj, WorkingTitle s TheSecretDudeSociety ∧ ScriptEditor AndrewCollins s :=
by
  exact Exists.intro Badults (And.intro WT_Badults SE_AC_Badults)
</lean>

ANSWER: True
