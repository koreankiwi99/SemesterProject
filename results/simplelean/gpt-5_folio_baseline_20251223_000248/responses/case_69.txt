<lean>
axiom Person : Type
axiom Show : Type

axiom MemberOfPappys : Person → Prop
axiom StarsIn : Person → Show → Prop

axiom BBCTwo : Prop  -- not directly used as a type; use predicates instead
axiom BBCThree : Prop -- same as above

axiom PilotedOnBBCTwo : Show → Prop
axiom PilotedOnBBCThree : Show → Prop

axiom Badults : Show

-- Context assumptions:
-- "Badults is a British Sitcom series starring members of Pappy's."
axiom stars_members_of_pappys_in_badults : ∃ p : Person, MemberOfPappys p ∧ StarsIn p Badults

-- "Badults was piloted in July 2013 on BBC Three."
axiom badults_piloted_on_bbc_three : PilotedOnBBCThree Badults

-- The statement to evaluate:
-- "No members of Pappy's have starred in a show piloting on BBC Two or BBC Three."
-- Formalized as: ∀ p, MemberOfPappys p → ∀ s, StarsIn p s → ¬ (PilotedOnBBCTwo s ∨ PilotedOnBBCThree s)

-- We prove the negation (i.e., the statement is false) by providing a counterexample from the context.
theorem statement_is_false :
  ¬ (∀ p : Person, MemberOfPappys p → ∀ s : Show, StarsIn p s → ¬ (PilotedOnBBCTwo s ∨ PilotedOnBBCThree s)) :=
by
  intro H
  cases stars_members_of_pappys_in_badults with
  | intro p hp =>
    cases hp with
    | intro hpMem hpStar =>
      have hneg := H p hpMem Badults hpStar
      have hdisj : PilotedOnBBCTwo Badults ∨ PilotedOnBBCThree Badults :=
        Or.inr badults_piloted_on_bbc_three
      exact hneg hdisj
</lean>

ANSWER: False
