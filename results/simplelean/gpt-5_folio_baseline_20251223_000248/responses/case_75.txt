<lean>
axiom obj : Type
axiom AtSchool : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom NotPicky : obj → Prop
axiom SpendLotDining : obj → Prop

axiom A1 : ∀ x : obj, AtSchool x → Work x → Need x
axiom A2 : ∀ x : obj, AtSchool x → OrderTakeout x → Work x
axiom A3 : ∀ x : obj, AtSchool x → (OrderTakeout x ∨ EnjoyDining x)
axiom A4 : ∀ x : obj, AtSchool x → EnjoyDining x → NotPicky x
axiom A5 : ∀ x : obj, AtSchool x → EnjoyDining x → SpendLotDining x

axiom Hannah : obj
axiom H_at : AtSchool Hannah
axiom H_work : Work Hannah
axiom H_cond : Need Hannah → (NotPicky Hannah ∧ ¬ Need Hannah)

theorem hannah_needs_money : Need Hannah := by
  apply A1
  · exact H_at
  · exact H_work

-- Also derivable: ¬ Need Hannah, from the conditional
theorem hannah_not_need_money : ¬ Need Hannah := by
  intro h
  exact (And.right (H_cond h)) h
</lean>

ANSWER: True
