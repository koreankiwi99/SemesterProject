<lean>
axiom obj : Type
axiom Hannah : obj

axiom AtSchool : obj → Prop
axiom WorksOnCampus : obj → Prop
axiom NeedsMoneyForTuition : obj → Prop
axiom OrdersTakeoutFrequently : obj → Prop
axiom EnjoysDiningHallMeals : obj → Prop
axiom NotPickyEater : obj → Prop
axiom SpendsLotsOfTimeInDiningHalls : obj → Prop

-- Rules
axiom R1 : ∀ x : obj, AtSchool x → WorksOnCampus x → NeedsMoneyForTuition x
axiom R2 : ∀ x : obj, AtSchool x → OrdersTakeoutFrequently x → WorksOnCampus x
axiom R3 : ∀ x : obj, AtSchool x → (OrdersTakeoutFrequently x ∨ EnjoysDiningHallMeals x)
axiom R4 : ∀ x : obj, AtSchool x → EnjoysDiningHallMeals x → NotPickyEater x
axiom R5 : ∀ x : obj, AtSchool x → EnjoysDiningHallMeals x → SpendsLotsOfTimeInDiningHalls x

-- Facts about Hannah
axiom H_at : AtSchool Hannah
axiom H_works : WorksOnCampus Hannah
axiom H_cond : NeedsMoneyForTuition Hannah → (NotPickyEater Hannah ∧ ¬ NeedsMoneyForTuition Hannah)

theorem hannah_not_picky_and_spends_time :
  AtSchool Hannah ∧ NotPickyEater Hannah ∧ SpendsLotsOfTimeInDiningHalls Hannah := by
  have hAt : AtSchool Hannah := H_at
  have hDisj : OrdersTakeoutFrequently Hannah ∨ EnjoysDiningHallMeals Hannah := R3 Hannah hAt
  have hNP_ST : NotPickyEater Hannah ∧ SpendsLotsOfTimeInDiningHalls Hannah := by
    cases hDisj with
    | inl hOrders =>
        have hWorks : WorksOnCampus Hannah := R2 Hannah hAt hOrders
        have hNeeds : NeedsMoneyForTuition Hannah := R1 Hannah hAt hWorks
        have hNP_and_notNeeds : NotPickyEater Hannah ∧ ¬ NeedsMoneyForTuition Hannah := H_cond hNeeds
        have hNP : NotPickyEater Hannah := hNP_and_notNeeds.left
        have hFalse : False := hNP_and_notNeeds.right hNeeds
        have hST : SpendsLotsOfTimeInDiningHalls Hannah := False.elim hFalse
        exact And.intro hNP hST
    | inr hEnjoys =>
        have hNP : NotPickyEater Hannah := R4 Hannah hAt hEnjoys
        have hST : SpendsLotsOfTimeInDiningHalls Hannah := R5 Hannah hAt hEnjoys
        exact And.intro hNP hST
  exact And.intro hAt (And.intro hNP_ST.left hNP_ST.right)
</lean>

ANSWER: True
