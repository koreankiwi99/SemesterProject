<lean>
axiom obj : Type
axiom At : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom Picky : obj → Prop
axiom Spend : obj → Prop
axiom Hannah : obj

-- Rules about people at Mary's school
axiom R1 : ∀ x : obj, At x → Work x → Need x
axiom R2 : ∀ x : obj, At x → Order x → Work x
axiom R3 : ∀ x : obj, At x → (Order x ∨ Enjoy x)
axiom R4 : ∀ x : obj, At x → Enjoy x → ¬ Picky x
axiom R5 : ∀ x : obj, At x → Enjoy x → Spend x

-- Facts about Hannah
axiom H_at : At Hannah
axiom H_work : Work Hannah
axiom H_cond : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah)

theorem hannah_statement :
  At Hannah ∧ (¬ Picky Hannah ∨ (Picky Hannah → Spend Hannah)) := by
  have hAt : At Hannah := H_at
  have hWork : Work Hannah := H_work
  have hNeed : Need Hannah := R1 Hannah hAt hWork
  have hNotPicky : ¬ Picky Hannah := (H_cond hNeed).left
  exact And.intro hAt (Or.inl hNotPicky)
</lean>

ANSWER: True
