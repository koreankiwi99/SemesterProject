<lean>
axiom obj : Type

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

axiom DTC : obj  -- Dried Thai chilies

-- Axioms from the context
axiom A1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
axiom A2 : ∀ x : obj, Cupcake x → BakedSweet x
axiom A3 : ∀ x : obj, MalaHotpot x → Spicy x
axiom A4 : ∀ x : obj, FromBBM x → Cupcake x
axiom A5 : Spicy DTC ∨ MalaHotpot DTC ∨ ¬ BakedSweet DTC

theorem not_BBM_DTC : ¬ FromBBM DTC :=
by
  intro hBBM
  have hCup : Cupcake DTC := A4 DTC hBBM
  have hBaked : BakedSweet DTC := A2 DTC hCup
  have hNotSpicy : ¬ Spicy DTC := A1 DTC hBaked
  have hNotMala : ¬ MalaHotpot DTC := fun hM => hNotSpicy (A3 DTC hM)
  have h := A5
  cases h with
  | inl hSpicy =>
      exact hNotSpicy hSpicy
  | inr h2 =>
      cases h2 with
      | inl hMala =>
          exact hNotMala hMala
      | inr hNotBaked =>
          exact hNotBaked hBaked
</lean>
