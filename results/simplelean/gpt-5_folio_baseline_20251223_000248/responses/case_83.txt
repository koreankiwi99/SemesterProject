<lean>
axiom obj : Type

axiom Listed : obj → Prop
axiom ManyNeg : obj → Prop
axiom RatingGT4 : obj → Prop
axiom ProvideTakeout : obj → Prop
axiom Popular : obj → Prop

axiom HPS : obj  -- Hamden Plaza Subway store

-- If a restaurant is listed in Yelp’s recommendations, then it has not received many negative reviews.
axiom A1 : ∀ x : obj, Listed x → ¬ ManyNeg x
-- All restaurants with a rating greater than four are listed in Yelp’s recommendations.
axiom A2 : ∀ x : obj, RatingGT4 x → Listed x
-- Some restaurants that do not provide take-out service receive many negative reviews.
axiom A3 : ∃ x : obj, ¬ ProvideTakeout x ∧ ManyNeg x
-- All restaurants that are popular among local residents have ratings greater than four.
axiom A4 : ∀ x : obj, Popular x → RatingGT4 x
-- The Hamden Plaza Subway store has a rating greater than four, or it is popular among local residents.
axiom A5 : RatingGT4 HPS ∨ Popular HPS

theorem hamden_implication :
  (ProvideTakeout HPS ∧ ManyNeg HPS) → (RatingGT4 HPS ∧ ¬ ProvideTakeout HPS) :=
by
  intro h
  -- Derive RatingGT4 HPS from A5 and A4
  have hr : RatingGT4 HPS := Or.elim A5 (fun hrg => hrg) (fun hpop => A4 HPS hpop)
  -- From rating > 4, get listed; from listed, get not many negative reviews
  have hl : Listed HPS := A2 HPS hr
  have hnotmn : ¬ ManyNeg HPS := A1 HPS hl
  -- Contradiction from ManyNeg HPS
  have mn : ManyNeg HPS := h.right
  have contra : False := hnotmn mn
  -- From contradiction, derive ¬ ProvideTakeout HPS
  have hnpto : ¬ ProvideTakeout HPS := False.elim contra
  exact And.intro hr hnpto
</lean>

ANSWER: True
