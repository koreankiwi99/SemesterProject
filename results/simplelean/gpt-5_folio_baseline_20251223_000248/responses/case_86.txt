<lean>
axiom obj : Type

axiom SMovie : obj
axiom SirDigby : obj
axiom Nemesis : obj

axiom SuperheroMovie : obj → Prop
axiom NamedAfter : obj → obj → Prop
axiom InMovie : obj → obj → Prop
axiom Good : obj → Prop
axiom Bad : obj → Prop
axiom Fights : obj → obj → Prop
axiom WinsAgainst : obj → obj → Prop

-- Facts
axiom T_movie_super : SuperheroMovie SMovie
axiom T_named : NamedAfter SMovie SirDigby
axiom T_fight : Fights SirDigby Nemesis

-- Rules
axiom R_named_good : ∀ m x, SuperheroMovie m → NamedAfter m x → Good x
axiom R_named_in : ∀ m x, SuperheroMovie m → NamedAfter m x → InMovie x m
axiom R_good_fight_bad : ∀ x y, Good x → Fights x y → Bad y
axiom R_bad_fight_good : ∀ x y, Bad x → Fights x y → Good y
axiom R_super_win : ∀ m x y, SuperheroMovie m → InMovie x m → Good x → Fights x y → WinsAgainst x y
axiom R_win_antisymm : ∀ x y, WinsAgainst x y → ¬ WinsAgainst y x

theorem nemesis_does_not_win : ¬ WinsAgainst Nemesis SirDigby := by
  have hg : Good SirDigby := R_named_good SMovie SirDigby T_movie_super T_named
  have hin : InMovie SirDigby SMovie := R_named_in SMovie SirDigby T_movie_super T_named
  have hw : WinsAgainst SirDigby Nemesis := R_super_win SMovie SirDigby Nemesis T_movie_super hin hg T_fight
  exact R_win_antisymm SirDigby Nemesis hw
</lean>

ANSWER: True
