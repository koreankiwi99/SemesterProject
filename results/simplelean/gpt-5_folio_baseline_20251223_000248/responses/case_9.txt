<lean>
axiom obj : Type
axiom Tom : obj

axiom WildTurkey : obj → Prop
axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Goulds : obj → Prop
axiom Merriams : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop

-- Context assumptions
axiom T_wt : WildTurkey Tom
axiom T_not_east : ¬ Eastern Tom
axiom T_not_osceola : ¬ Osceola Tom
axiom T_not_goulds : ¬ Goulds Tom
axiom T_not_merriams : ¬ Merriams Tom
axiom T_not_riogrande : ¬ RioGrande Tom

-- Exhaustiveness of the six types for wild turkeys
axiom exhaustive :
  ∀ x : obj,
    WildTurkey x →
    (Eastern x ∨ Osceola x ∨ Goulds x ∨ Merriams x ∨ RioGrande x ∨ Ocellated x)

theorem tom_ocellated : Ocellated Tom :=
by
  have h1 := exhaustive Tom T_wt
  have h2 := Or.resolve_left h1 T_not_east
  have h3 := Or.resolve_left h2 T_not_osceola
  have h4 := Or.resolve_left h3 T_not_goulds
  have h5 := Or.resolve_left h4 T_not_merriams
  have h6 := Or.resolve_left h5 T_not_riogrande
  exact h6
</lean>

ANSWER: True
