<lean>
axiom obj : Type
axiom MachineTranslation : obj

axiom NLPTask : obj → Prop
axiom LanguageGeneration : obj → Prop
axiom LanguageUnderstanding : obj → Prop
axiom OutputTextSequence : obj → Prop

-- A natural language processing task can be categorized as either LG or LU
axiom categorize : ∀ x : obj, NLPTask x → (LanguageGeneration x ∨ LanguageUnderstanding x)

-- If the output of a NLP task is a text sequence, then it is a language generation task
axiom gen_rule : ∀ x : obj, NLPTask x → OutputTextSequence x → LanguageGeneration x

-- Machine translation is an NLP task
axiom MT_is_NLP : NLPTask MachineTranslation

-- The output of machine translation is a text sequence
axiom MT_output_text : OutputTextSequence MachineTranslation

theorem MT_is_generation : LanguageGeneration MachineTranslation :=
  gen_rule MachineTranslation MT_is_NLP MT_output_text
</lean>

ANSWER: True
