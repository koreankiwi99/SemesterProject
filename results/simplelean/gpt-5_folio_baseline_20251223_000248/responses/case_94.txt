<lean>
axiom Building : Type
axiom OliveGarden : Building
axiom Managed : Building → Prop
axiom monthly_rent : Building → Nat
axiom security_deposit : Building → Nat

axiom managed_og : Managed OliveGarden
axiom rent_og : monthly_rent OliveGarden = 2000
axiom deposit_ge_rent : ∀ b : Building, Managed b → security_deposit b ≥ monthly_rent b

theorem og_deposit_ge_2000 : security_deposit OliveGarden ≥ 2000 := by
  have h := deposit_ge_rent OliveGarden managed_og
  simpa [rent_og] using h
</lean>

ANSWER: True
