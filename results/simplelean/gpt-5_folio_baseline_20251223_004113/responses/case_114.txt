axiom obj : Type
axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

axiom Man : obj → Prop
axiom ClassOfMichael : obj → Prop
axiom Taller : obj → obj → Prop
axiom CanBlock : obj → obj → Prop
axiom JumpWhenShooting : obj → Prop
axiom Shooter : obj → Prop
axiom GreatShooter : obj → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R_block_by_height : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom Michael_man : Man Michael
axiom Michael_taller_than_class : ∀ y : obj, ClassOfMichael y → Taller Michael y

-- Transitivity of taller-than.
axiom taller_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom Peter_man : Man Peter
axiom Peter_taller_Michael : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom Michael_blocks_if_no_jump : ∀ p : obj, ¬ JumpWhenShooting p → CanBlock Michael p

-- Michael cannot block Windy's shooting.
axiom not_block_Windy : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom jump_shooter_great : ∀ s : obj, Shooter s → JumpWhenShooting s → GreatShooter s

-- Useful consequence: Peter is taller than every member of Michael's class.
theorem Peter_taller_than_all_in_Michael_class :
  ∀ c : obj, ClassOfMichael c → Taller Peter c := by
  intro c hc
  apply taller_trans Peter Michael c
  · exact Peter_taller_Michael
  · exact Michael_taller_than_class c hc

-- From blocking axiom and the fact Michael cannot block Windy, we can derive ¬¬(Windy jumps when shooting).
theorem not_not_Windy_jumps : ¬¬ JumpWhenShooting Windy := by
  intro hno
  have hblock : CanBlock Michael Windy := Michael_blocks_if_no_jump Windy hno
  exact not_block_Windy hblock
