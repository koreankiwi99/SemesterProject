<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom WorksLibrary : obj → Prop
axiom WorksLab : obj → Prop
axiom PartTimeUni : obj → Prop
axiom InCS : obj → Prop
axiom TakesDatabase : obj → Prop
axiom TakesWithDavid : obj → Prop

-- None of the students who currently work in the library are from the CS department.
axiom A1 : ∀ x : obj, WorksLibrary x → ¬ InCS x
-- All students with part-time university jobs work in the library.
axiom A2 : ∀ x : obj, PartTimeUni x → WorksLibrary x
-- All students taking the database course are from the CS department.
axiom A3 : ∀ x : obj, TakesDatabase x → InCS x
-- All students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, TakesWithDavid x → TakesDatabase x
-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, WorksLab x → TakesWithDavid x
-- James is a student working in the lab.
axiom A6 : Student James
axiom A7 : WorksLab James
-- James doesn't work in the lab or have a part-time job offered by the university.
axiom A8 : ¬ (WorksLab James ∨ PartTimeUni James)

theorem james_takes_database : TakesDatabase James := by
  have h_david : TakesWithDavid James := A5 James A7
  exact A4 James h_david
</lean>

ANSWER: True
