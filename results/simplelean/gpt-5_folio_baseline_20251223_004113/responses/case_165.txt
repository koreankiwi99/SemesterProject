<lean>
axiom obj : Type
axiom Show : obj → Prop
axiom Monday8pm : obj → Prop
axiom GivesRosesOnTV : obj → Prop
axiom BachelorEpisode : obj → Prop
axiom PortraysRealLives : obj → Prop
axiom RealityTV : obj → Prop
axiom BreakingBad : obj

axiom E1 : ∃ s : obj, Show s ∧ Monday8pm s ∧ GivesRosesOnTV s
axiom R1 : ∀ x : obj, GivesRosesOnTV x → BachelorEpisode x
axiom R2 : ∀ x : obj, BachelorEpisode x → PortraysRealLives x
axiom R3 : ∀ x : obj, PortraysRealLives x → RealityTV x
axiom F1 : Show BreakingBad
axiom F2 : ¬RealityTV BreakingBad

theorem breakingbad_roses_implies_monday8pm : GivesRosesOnTV BreakingBad → Monday8pm BreakingBad :=
by
  intro hroses
  have hEpisode : BachelorEpisode BreakingBad := R1 BreakingBad hroses
  have hPortray : PortraysRealLives BreakingBad := R2 BreakingBad hEpisode
  have hReality : RealityTV BreakingBad := R3 BreakingBad hPortray
  have contra : False := F2 hReality
  exact False.elim contra
</lean>

ANSWER: True
