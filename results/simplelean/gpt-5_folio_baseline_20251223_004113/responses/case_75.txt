<lean>
axiom obj : Type
axiom At : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom Picky : obj → Prop
axiom SpendTime : obj → Prop
axiom Hannah : obj

-- Rules about people at Mary's school
axiom R1 : ∀ x : obj, At x ∧ Work x → Need x
axiom R2 : ∀ x : obj, At x ∧ Order x → Work x
axiom R3 : ∀ x : obj, At x → Order x ∨ Enjoy x
axiom R4 : ∀ x : obj, At x ∧ Enjoy x → ¬ Picky x
axiom R5 : ∀ x : obj, At x ∧ Enjoy x → SpendTime x

-- Facts about Hannah
axiom T1 : At Hannah
axiom T2 : Work Hannah
axiom T3 : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah)

theorem hannah_needs_money : Need Hannah := by
  have h : At Hannah ∧ Work Hannah := And.intro T1 T2
  exact R1 Hannah h
</lean>

ANSWER: True
