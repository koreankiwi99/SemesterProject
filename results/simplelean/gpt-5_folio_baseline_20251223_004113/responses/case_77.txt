<lean>
axiom obj : Type
axiom At : obj → Prop
axiom Work : obj → Prop
axiom Needs : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom Picky : obj → Prop
axiom Spend : obj → Prop
axiom Hannah : obj

-- If people at Mary's school work in student jobs on campus, then they need money.
axiom R1 : ∀ x : obj, At x ∧ Work x → Needs x
-- If people at Mary's school order takeout frequently in college, then they work in student jobs on campus.
axiom R2 : ∀ x : obj, At x ∧ Order x → Work x
-- People at Mary's school order takeout frequently in college or enjoy the dining hall meals and recipes.
axiom R3 : ∀ x : obj, At x → (Order x ∨ Enjoy x)
-- If people at Mary's school enjoy the dining hall meals and recipes, then they are not picky eaters.
axiom R4 : ∀ x : obj, At x ∧ Enjoy x → ¬ Picky x
-- If people at Mary's school enjoy the dining hall meals and recipes, then they spend a lot of their time in the campus dining halls.
axiom R5 : ∀ x : obj, At x ∧ Enjoy x → Spend x

-- Hannah is at Mary's school.
axiom T6 : At Hannah
-- Hannah works; and if she needs money, then she is neither picky nor needs money.
axiom T7a : Work Hannah
axiom T7b : Needs Hannah → (¬ Picky Hannah ∧ ¬ Needs Hannah)

theorem hannah_statement :
  At Hannah ∧ (¬ Picky Hannah ∨ (Picky Hannah → Spend Hannah)) :=
by
  have hNeeds : Needs Hannah := R1 Hannah ⟨T6, T7a⟩
  have hNotPicky : ¬ Picky Hannah := (T7b hNeeds).left
  exact And.intro T6 (Or.inl hNotPicky)
</lean>

ANSWER: True
