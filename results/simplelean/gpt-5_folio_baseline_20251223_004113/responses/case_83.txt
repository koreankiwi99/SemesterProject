<lean>
axiom obj : Type
axiom Listed : obj → Prop
axiom ManyNeg : obj → Prop
axiom RatingGT4 : obj → Prop
axiom Popular : obj → Prop
axiom TakeOut : obj → Prop

-- If listed in Yelp’s recommendations, then not many negative reviews
axiom A1 : ∀ x : obj, Listed x → ¬ ManyNeg x
-- All with rating > 4 are listed
axiom A2 : ∀ x : obj, RatingGT4 x → Listed x
-- Some restaurants without take-out receive many negative reviews
axiom A3 : ∃ x : obj, ¬ TakeOut x ∧ ManyNeg x
-- Popular among locals implies rating > 4
axiom A4 : ∀ x : obj, Popular x → RatingGT4 x

-- The Hamden Plaza Subway store
axiom hps : obj
-- It has rating > 4 or is popular among locals
axiom A5 : RatingGT4 hps ∨ Popular hps

theorem query :
  (TakeOut hps ∧ ManyNeg hps) → (RatingGT4 hps ∧ ¬ TakeOut hps) := by
  intro h
  have ratingH : RatingGT4 hps := by
    cases A5 with
    | inl hr => exact hr
    | inr hp => exact A4 hps hp
  have notNegH : ¬ ManyNeg hps := by
    have listedH : Listed hps := A2 hps ratingH
    exact A1 hps listedH
  have contra : False := notNegH h.right
  apply And.intro
  · exact ratingH
  · intro ht
    exact False.elim contra
</lean>

ANSWER: True
