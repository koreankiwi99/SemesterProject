axiom Vertebrate : obj → Prop

axiom Bee : obj → Prop

-- Animals are either invertebrates or vertebrates.
axiom A1 : ∀ x : obj, Animal x → (Invertebrate x ∨ Vertebrate x)
-- All animals with backbones reproduce by male-and-female mating.
axiom A2 : ∀ x : obj, Animal x → Backbone x → MateMF x
-- All vertebrate animals have a backbone.
axiom A3 : ∀ x : obj, Animal x → Vertebrate x → Backbone x
-- All bees do not reproduce by male-and-female mating.
axiom A4 : ∀ x : obj, Bee x → ¬ MateMF x
-- All queen bees are bees.
axiom A5 : ∀ x : obj, QueenBee x → Bee x
-- Harry is a bee.
axiom A6 : Bee Harry

-- Some derived consequences (but they still require Animal Harry, which is not given):
theorem not_mf_Harry : ¬ MateMF Harry := by
  exact A4 Harry A6

theorem animal_not_vertebrate_Harry : Animal Harry → ¬ Vertebrate Harry := by
  intro hA hV
  have hB : Backbone Harry := A3 Harry hA hV
  have hMF : MateMF Harry := A2 Harry hA hB
  exact (A4 Harry A6) hMF

theorem animal_invertebrate_Harry : Animal Harry → Invertebrate Harry := by
  intro hA
  have h := A1 Harry hA
  cases h with
  | inl hI => exact hI
  | inr hV =>
      have hNV : ¬ Vertebrate Harry := animal_not_vertebrate_Harry hA
      have : False := hNV hV
      exact False.elim this
