-- From being in a multiple birth, we can derive not living with strangers
theorem mb_not_strangers (x : Person) : MultipleBirth x → ¬ LiveWithStrangers x := by
  intro hmb
  have hst : SpendTimeWithSiblings x := R1 x hmb
  have hhome : LiveAtHome x := R5 x hst
  exact R4 x hhome

theorem luke_not_first_branch : ¬ (MultipleBirth Luke ∧ LiveWithStrangers Luke) := by
  intro h
  have hnot : ¬ LiveWithStrangers Luke := mb_not_strangers Luke h.left
  exact hnot h.right

theorem luke_second_branch : ¬ MultipleBirth Luke ∧ ¬ LiveWithStrangers Luke := by

  have hnf := luke_not_first_branch
  cases disj with
