<lean>
axiom obj : Type
axiom Luke : obj

axiom MultipleBirth : obj → Prop
axiom HangWithSibs : obj → Prop
axiom HaveSibsBornTogether : obj → Prop
axiom Complain : obj → Prop
axiom LiveAtHome : obj → Prop
axiom LiveWithStrangers : obj → Prop
axiom Baby : obj → Prop

-- People born in a multiple birth spend a lot of time with their siblings
axiom R1 : ∀ x : obj, MultipleBirth x → HangWithSibs x
-- If people have siblings who were born together, then they were born in a multiple birth
axiom R2 : ∀ x : obj, HaveSibsBornTogether x → MultipleBirth x
-- If people complain often about annoying siblings, then they have siblings who were born together
axiom R3 : ∀ x : obj, Complain x → HaveSibsBornTogether x
-- If people live at home, then they do not live with strangers
axiom R4 : ∀ x : obj, LiveAtHome x → ¬ LiveWithStrangers x
-- If people spend a lot of time with their siblings, then they live at home
axiom R5 : ∀ x : obj, HangWithSibs x → LiveAtHome x

-- Luke either is a baby born in a multiple birth and lives with strangers,
-- or is not a baby born in a multiple birth and does not live with strangers
axiom T1 :
  ((Baby Luke ∧ MultipleBirth Luke) ∧ LiveWithStrangers Luke) ∨
  (¬(Baby Luke ∧ MultipleBirth Luke) ∧ ¬ LiveWithStrangers Luke)

-- Derived: Multiple birth implies not living with strangers
theorem mb_not_strangers : ∀ x : obj, MultipleBirth x → ¬ LiveWithStrangers x := by
  intro x hmb
  have hhang : HangWithSibs x := R1 x hmb
  have hhome : LiveAtHome x := R5 x hhang
  exact R4 x hhome

-- The left disjunct of T1 is impossible given the other axioms
theorem not_left :
  ¬ (((Baby Luke ∧ MultipleBirth Luke) ∧ LiveWithStrangers Luke)) := by
  intro h
  have hbmb : Baby Luke ∧ MultipleBirth Luke := h.left
  have hmb : MultipleBirth Luke := hbmb.right
  have hstr : LiveWithStrangers Luke := h.right
  have hnstr : ¬ LiveWithStrangers Luke := mb_not_strangers Luke hmb
  exact hnstr hstr

-- Therefore, the right disjunct must hold
theorem right_holds :
  (¬(Baby Luke ∧ MultipleBirth Luke) ∧ ¬ LiveWithStrangers Luke) := by
  cases T1 with
  | inl hleft => exact False.elim (not_left hleft)
  | inr hright => exact hright

-- Useful chain: complaining implies not living with strangers
theorem complain_not_strangers : ∀ x : obj, Complain x → ¬ LiveWithStrangers x := by
  intro x hc
  have hmb : MultipleBirth x := R2 x (R3 x hc)
  exact mb_not_strangers x hmb
