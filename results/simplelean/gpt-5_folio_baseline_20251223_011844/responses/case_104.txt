<lean>
axiom obj : Type
axiom Tom : obj

-- Predicates
axiom GreApplicant : obj → Prop
axiom EconomicHardship : obj → Prop
axiom FinancialAidETS : obj → Prop
axiom LivesSingleParent : obj → Prop
axiom FewResources : obj → Prop

-- Rules
-- ETS provides financial aid to GRE applicants who prove economic hardship
axiom R1 : ∀ x : obj, GreApplicant x → EconomicHardship x → FinancialAidETS x
-- Those living in single-parent families or having few resources can prove economic hardship
axiom R2 : ∀ x : obj, LivesSingleParent x ∨ FewResources x → EconomicHardship x

-- Facts about Tom
axiom TomApplicant : GreApplicant Tom
axiom TomSingleParent : LivesSingleParent Tom
axiom TomFewResources : FewResources Tom

theorem ets_provides_financial_aid_to_Tom : FinancialAidETS Tom := by
  apply R1 Tom
  · exact TomApplicant
  ·
    have h : LivesSingleParent Tom ∨ FewResources Tom := Or.inl TomSingleParent
    exact R2 Tom h
</lean>

ANSWER: True
