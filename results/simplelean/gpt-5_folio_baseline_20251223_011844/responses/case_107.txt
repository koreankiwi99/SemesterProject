axiom obj : Type
axiom Peter : obj

axiom Spill : obj → Prop                -- spills a lot of food on clothing
axiom NotablyTidy : obj → Prop
axiom ClumsyFoodie : obj → Prop
axiom GoesOutFrequently : obj → Prop
axiom Cleanly : obj → Prop
axiom ValuesOrderSpotlessness : obj → Prop
axiom FamilyPrioritizesOrderSpotlessness : obj → Prop

-- If people spill a lot of food on their clothing, then they are not notably tidy.
axiom R1 : ∀ x : obj, Spill x → ¬ NotablyTidy x

-- All clumsy foodies who go out frequently are people who spill a lot of food on their clothing.
axiom R2 : ∀ x : obj, ClumsyFoodie x ∧ GoesOutFrequently x → Spill x

-- If people are cleanly, then they are notably tidy.
axiom R3 : ∀ x : obj, Cleanly x → NotablyTidy x

-- If people value order and spotlessness, then they are cleanly.
axiom R4 : ∀ x : obj, ValuesOrderSpotlessness x → Cleanly x

-- All people whose families prioritize order and spotlessness themselves value order and spotlessness.
axiom R5 : ∀ x : obj, FamilyPrioritizesOrderSpotlessness x → ValuesOrderSpotlessness x

-- Peter is either (spills and is cleanly) or (does not spill and is not cleanly).
axiom Peter_disj : (Spill Peter ∧ Cleanly Peter) ∨ (¬ Spill Peter ∧ ¬ Cleanly Peter)

-- Useful derived facts
theorem tidy_of_cleanly (x : obj) : Cleanly x → NotablyTidy x := R3 x
theorem not_tidy_of_spill (x : obj) : Spill x → ¬ NotablyTidy x := R1 x

-- The left branch of Peter's disjunction is contradictory (it yields both tidy and not tidy).
theorem contradiction_if_left_branch : (Spill Peter ∧ Cleanly Peter) → False :=
  fun h =>
    let t : NotablyTidy Peter := R3 Peter h.right
    let nt : ¬ NotablyTidy Peter := R1 Peter h.left
    nt t
