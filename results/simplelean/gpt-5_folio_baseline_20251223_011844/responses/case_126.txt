<lean>
axiom obj : Type
axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom Fly : obj → Prop
axiom Wings : obj → Prop
axiom Breathe : obj → Prop

axiom A1 : ∀ x : obj, Pet x ∧ Mammal x → Animal x
axiom A2 : ∀ x : obj, Monkey x → Mammal x
axiom A3 : ∀ x : obj, Pet x → Monkey x ∨ Bird x
axiom A4 : ∀ x : obj, Pet x ∧ Bird x → Fly x
axiom A5 : ∀ x : obj, Animal x ∧ Pet x → Breathe x
axiom A6 : ∀ x : obj, Pet x ∧ Fly x → Wings x

axiom Rock : obj
axiom Rpet : Pet Rock
axiom Rdisj : Fly Rock ∨ Bird Rock ∨ ¬Breathe Rock

theorem rock_has_wings : Wings Rock := by
  have pr : Pet Rock := Rpet
  have d1 : Monkey Rock ∨ Bird Rock := A3 Rock pr
  have d2 : Fly Rock ∨ Bird Rock ∨ ¬Breathe Rock := Rdisj
  cases d2 with
  | inl hf =>
      exact A6 Rock ⟨pr, hf⟩
  | inr rest =>
      cases rest with
      | inl hb =>
          have hf : Fly Rock := A4 Rock ⟨pr, hb⟩
          exact A6 Rock ⟨pr, hf⟩
      | inr hnb =>
          cases d1 with
          | inl hmonkey =>
              have hm : Mammal Rock := A2 Rock hmonkey
              have ha : Animal Rock := A1 Rock ⟨pr, hm⟩
              have hb : Breathe Rock := A5 Rock ⟨ha, pr⟩
              have : False := hnb hb
              exact False.elim this
          | inr hbird =>
              have hf : Fly Rock := A4 Rock ⟨pr, hbird⟩
              exact A6 Rock ⟨pr, hf⟩
</lean>
