<lean>
axiom obj : Type

-- Predicates
axiom Animal : obj → Prop
axiom HeckCattle : obj → Prop
axiom Aurochs : obj → Prop
axiom ExtinctAnimal : obj → Prop
axiom Resemble : obj → obj → Prop
axiom BredBack : obj → Prop
axiom ArtificiallySelected : obj → Prop
axiom ToBeBredBack : obj → Prop

-- Context axioms
-- Breeding back is a form of artificial selection: anything bred back is artificially selected
axiom R_bb_as : ∀ x : obj, BredBack x → ArtificiallySelected x

-- Heck cattle were bred back (existential reading of the statement)
axiom H_bred : ∃ x : obj, HeckCattle x ∧ BredBack x

-- Heck cattle are animals
axiom HeckCattleAnimals : ∀ x : obj, HeckCattle x → Animal x

-- Aurochs are animals
axiom AurochsAnimals : ∀ x : obj, Aurochs x → Animal x

-- Some animals to be bred back resemble extinct animals
axiom SomeToBeBredBackResembleExtinct :
  ∃ x : obj, Animal x ∧ ToBeBredBack x ∧ (∃ y : obj, ExtinctAnimal y ∧ Resemble x y)

-- Goal: Some Heck cattle are artificially selected
theorem some_heck_cattle_artificially_selected : ∃ x : obj, HeckCattle x ∧ ArtificiallySelected x := by
  rcases H_bred with ⟨h, hh, hb⟩
  exact ⟨h, And.intro hh (R_bb_as h hb)⟩
</lean>

ANSWER: True
