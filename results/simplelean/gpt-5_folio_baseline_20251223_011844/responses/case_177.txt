<lean>
axiom Game : Type
axiom Country : Type

axiom LastSummer : Game
axiom Tokyo : Game

axiom SportingEvent : Game → Prop
-- The summer Olympic games is a sporting event (applied to the last one for our context)
axiom last_is_sport : SportingEvent LastSummer

axiom UnitedStates : Country
axiom WonMostMedals : Country → Game → Prop

-- The last summer Olympic games was in Tokyo.
axiom last_in_tokyo : LastSummer = Tokyo
-- The United States won the most medals in Tokyo.
axiom us_won_tokyo : WonMostMedals UnitedStates Tokyo

-- Therefore, the United States won the most medals in the last summer Olympic games.
theorem us_won_last : WonMostMedals UnitedStates LastSummer :=
  by
    simpa [last_in_tokyo] using us_won_tokyo
</lean>

ANSWER: True
