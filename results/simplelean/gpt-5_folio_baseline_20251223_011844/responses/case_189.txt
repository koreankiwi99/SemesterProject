axiom obj : Type
axiom Yuri : obj

axiom Professional : obj → Prop
axiom American : obj → Prop
axiom JumpShot : obj → Prop
axiom Leap : obj → Prop
axiom ActivateLegs : obj → Prop

-- Some professional basketball players are not American nationals.
axiom some_non_american_pro : ∃ x : obj, Professional x ∧ ¬ American x

-- All professional basketball players can do jump shots.
axiom pro_implies_jump : ∀ x : obj, Professional x → JumpShot x

-- If someone can jump shots, they leap straight into the air.
axiom jump_implies_leap : ∀ x : obj, JumpShot x → Leap x

-- If someone leaps straight into the air, they activate their leg muscles.
axiom leap_implies_activate : ∀ x : obj, Leap x → ActivateLegs x

-- Yuri does not activate his leg muscles.
axiom yuri_not_activate : ¬ ActivateLegs Yuri

theorem yuri_not_pro : ¬ Professional Yuri := by
  intro hPro
  have hJump : JumpShot Yuri := pro_implies_jump Yuri hPro
  have hLeap : Leap Yuri := jump_implies_leap Yuri hJump
  have hAct : ActivateLegs Yuri := leap_implies_activate Yuri hLeap
  exact yuri_not_activate hAct

theorem not_american_pro_yuri : ¬ (American Yuri ∧ Professional Yuri) := by
  intro h
  apply yuri_not_pro
  exact h.right
