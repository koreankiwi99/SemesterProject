<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom NearCampus : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

axiom Rockie : obj

-- All rabbits that can be spotted near the campus are cute.
axiom R1 : ∀ x : obj, Rabbit x ∧ NearCampus x → Cute x
-- Some turtles can be spotted near the campus.
axiom E1 : ∃ x : obj, Turtle x ∧ NearCampus x
-- The only animals that can be spotted near the campus are rabbits and squirrels.
axiom R2 : ∀ x : obj, NearCampus x → Rabbit x ∨ Squirrel x
-- If something is skittish, then it is not calm.
axiom R3 : ∀ x : obj, Skittish x → ¬Calm x
-- All the squirrels that can be spotted near the campus are skittish.
axiom R4 : ∀ x : obj, Squirrel x ∧ NearCampus x → Skittish x
-- Rockie can be spotted near the campus, and it is calm.
axiom T_Rockie_near : NearCampus Rockie
axiom T_Rockie_calm : Calm Rockie

-- From Rockie being calm and near campus, Rockie cannot be a squirrel.
theorem not_squirrel_rockie : ¬ Squirrel Rockie :=
fun hs =>
  let hsNear : Squirrel Rockie ∧ NearCampus Rockie := And.intro hs T_Rockie_near
  let hSk : Skittish Rockie := R4 Rockie hsNear
  let hNotCalm : ¬ Calm Rockie := R3 Rockie hSk
  hNotCalm T_Rockie_calm

-- From the "only animals near campus are rabbits or squirrels" and not-squirrel, Rockie is a rabbit.
theorem rabbit_rockie : Rabbit Rockie :=
by
  have h := R2 Rockie T_Rockie_near
  exact h.elim (fun hr => hr) (fun hs => False.elim (not_squirrel_rockie hs))

-- Hence Rockie is cute (as a rabbit near campus).
theorem cute_rockie : Cute Rockie :=
R1 Rockie (And.intro rabbit_rockie T_Rockie_near)

-- Target statement to evaluate: (Cute Rockie ∧ Calm Rockie) → (Skittish Rockie ∧ Turtle Rockie)
-- We prove its negation.
theorem not_target :
  ¬ ((Cute Rockie ∧ Calm Rockie) → (Skittish Rockie ∧ Turtle Rockie)) :=
by
  intro hImp
  have hAnte : Cute Rockie ∧ Calm Rockie := And.intro cute_rockie T_Rockie_calm
  have hCons : Skittish Rockie ∧ Turtle Rockie := hImp hAnte
  have hSk : Skittish Rockie := hCons.left
  have hNotCalm : ¬ Calm Rockie := R3 Rockie hSk
  exact hNotCalm T_Rockie_calm
</lean>
