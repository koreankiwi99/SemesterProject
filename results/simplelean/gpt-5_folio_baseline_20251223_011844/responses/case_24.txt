<lean>
axiom obj : Type
axiom event : Type

axiom Beijing : obj
axiom Hosted : obj → event → Prop

axiom SummerOlympics2008 : event
axiom WinterOlympics : event

-- From the context:
-- "Beijing hosted the 2008 Summer Olympics and 2008 Summer Paralympics Games."
axiom H_summer_2008 : Hosted Beijing SummerOlympics2008
-- "Beijing has hosted the Summer and Winter Olympics and the Summer and Winter Paralympics."
axiom H_winter : Hosted Beijing WinterOlympics

theorem beijing_hosted_both : Hosted Beijing SummerOlympics2008 ∧ Hosted Beijing WinterOlympics := by
  constructor
  · exact H_summer_2008
  · exact H_winter
</lean>

ANSWER: True
