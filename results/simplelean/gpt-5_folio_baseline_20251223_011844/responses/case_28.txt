<lean>
axiom obj : Type
axiom Marvin : obj

axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromMars : obj → Prop
axiom FromEarth : obj → Prop
axiom HighlyIntelligent : obj → Prop

-- All aliens are extraterrestrials.
axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x
-- If someone is from Mars, then they are an alien.
axiom A2 : ∀ x : obj, FromMars x → Alien x
-- No extraterrestrials are human.
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬ Human x
-- All highly intelligent beings from Earth are humans.
axiom A4 : ∀ x : obj, HighlyIntelligent x ∧ FromEarth x → Human x
-- Marvin is a highly intelligent being.
axiom A5 : HighlyIntelligent Marvin
-- Marvin is either from Earth and from Mars, or he is from neither.
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬FromEarth Marvin ∧ ¬FromMars Marvin)
-- If Marvin is not from Earth, then Marvin is an extraterrestrial.
axiom A7 : ¬ FromEarth Marvin → Extraterrestrial Marvin

theorem marvin_neither_human_nor_from_mars : ¬ Human Marvin ∧ ¬ FromMars Marvin := by
  -- First, show ¬ FromMars Marvin
  have not_mars : ¬ FromMars Marvin := by
    intro mMars
    -- From the disjunction, deduce FromEarth Marvin ∧ FromMars Marvin (right disjunct contradicts mMars)
    have hEM : FromEarth Marvin ∧ FromMars Marvin := by
      cases A6 with
      | inl h1 => exact h1
      | inr h2 =>
          have nm : ¬ FromMars Marvin := h2.right
          exact False.elim (nm mMars)
    have e : FromEarth Marvin := hEM.left
    have human : Human Marvin := A4 Marvin ⟨A5, e⟩
    have extr : Extraterrestrial Marvin := A1 Marvin (A2 Marvin mMars)
    have not_human : ¬ Human Marvin := A3 Marvin extr
    exact not_human human
  -- Next, show ¬ Human Marvin
  have not_earth : ¬ FromEarth Marvin := by
    intro e
    cases A6 with
    | inl h1 =>
        have m : FromMars Marvin := h1.right
        exact not_mars m
    | inr h2 =>
        have ne : ¬ FromEarth Marvin := h2.left
        exact ne e
  have extr : Extraterrestrial Marvin := A7 not_earth
  have not_human : ¬ Human Marvin := A3 Marvin extr
  exact And.intro not_human not_mars
</lean>

ANSWER: True
