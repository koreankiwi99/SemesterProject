axiom GrandSlam : obj → Prop
axiom OscarNom : obj → Prop

axiom TaxHaven : obj → Prop

-- Everyone at the mixer is a Grand Slam champion or an Oscar-nominated actor.
axiom R1 : ∀ x : obj, AtMixer x → (GrandSlam x ∨ OscarNom x)
-- Every Grand Slam champion at the mixer is a professional tennis player.
axiom R2 : ∀ x : obj, AtMixer x → GrandSlam x → ProTennis x
-- All Oscar-nominated actors at the mixer are celebrities.
axiom R3 : ∀ x : obj, AtMixer x → OscarNom x → Celebrity x
-- All professional tennis players at the mixer are athletes.
axiom R4 : ∀ x : obj, AtMixer x → ProTennis x → Athlete x
-- If a person at the mixer is a celebrity, then they are well paid.
axiom R5 : ∀ x : obj, AtMixer x → Celebrity x → WellPaid x
-- If a person at the mixer is an athlete, then they are famous.
axiom R6 : ∀ x : obj, AtMixer x → Athlete x → Famous x
-- All well-paid people at the mixer live in tax havens.
axiom R7 : ∀ x : obj, AtMixer x → WellPaid x → TaxHaven x

-- Djokovic is at the mixer
axiom DjAt : AtMixer Djokovic
-- If Djokovic is a famous athlete, then Djokovic is well-paid.
axiom DjImp : Famous Djokovic → Athlete Djokovic → WellPaid Djokovic

-- We can derive that Djokovic is well-paid (but not that he is a Grand Slam champion).
theorem Djokovic_well_paid : WellPaid Djokovic := by
  have h : GrandSlam Djokovic ∨ OscarNom Djokovic := R1 Djokovic DjAt
  cases h with
  | inl gs =>
      have pt : ProTennis Djokovic := R2 Djokovic DjAt gs
      have ath : Athlete Djokovic := R4 Djokovic DjAt pt
      have fam : Famous Djokovic := R6 Djokovic DjAt ath
      exact DjImp fam ath
  | inr osc =>
      have cel : Celebrity Djokovic := R3 Djokovic DjAt osc
      exact R5 Djokovic DjAt cel
