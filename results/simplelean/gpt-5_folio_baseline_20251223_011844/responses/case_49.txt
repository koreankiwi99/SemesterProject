axiom NineteenEightyFour : obj

axiom Digital : obj → Prop
axiom Analog : obj → Prop
axiom PrintedText : obj → Prop
axiom StreamingService : obj → Prop
axiom HardcoverBook : obj → Prop

-- No digital media are analog.
axiom no_digital_are_analog : ∀ x : obj, Digital x → ¬ Analog x
-- Every printed text is analog media.
axiom printed_are_analog : ∀ x : obj, PrintedText x → Analog x
-- All streaming services are digital media.
axiom streaming_are_digital : ∀ x : obj, StreamingService x → Digital x
-- If an object is a hardcover book, then it is printed text.
axiom hardcover_implies_printed : ∀ x : obj, HardcoverBook x → PrintedText x
-- If 1984 is a streaming service, then 1984 is a hardcover book.
axiom if_1984_streaming_then_hardcover : StreamingService NineteenEightyFour → HardcoverBook NineteenEightyFour

-- From the given axioms, 1984 cannot be a streaming service.
theorem not_streaming_1984 : ¬ StreamingService NineteenEightyFour := by
  intro hS
  have hD : Digital NineteenEightyFour := streaming_are_digital NineteenEightyFour hS
  have hNotA : ¬ Analog NineteenEightyFour := no_digital_are_analog NineteenEightyFour hD
  have hHC : HardcoverBook NineteenEightyFour := if_1984_streaming_then_hardcover hS
  have hPT : PrintedText NineteenEightyFour := hardcover_implies_printed NineteenEightyFour hHC
  have hA : Analog NineteenEightyFour := printed_are_analog NineteenEightyFour hPT
  exact hNotA hA
