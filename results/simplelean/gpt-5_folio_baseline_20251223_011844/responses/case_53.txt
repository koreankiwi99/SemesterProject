<lean>
axiom obj : Type
axiom French : obj
axiom Spanish : obj
axiom German : obj
axiom Basque : obj

axiom Romance : obj → Prop
axiom IndoEuropean : obj → Prop
axiom Related : obj → obj → Prop

axiom Family : (obj → Prop) → Prop

-- All Romance languages are Indo-European languages.
axiom A1 : ∀ x : obj, Romance x → IndoEuropean x
-- Romance languages are a language family.
axiom A2 : Family Romance
-- All languages within a language family are related to each other.
axiom A3 : ∀ F : obj → Prop, Family F → ∀ x y : obj, F x → F y → Related x y
-- French and Spanish are both Romance languages.
axiom A4 : Romance French
axiom A5 : Romance Spanish
-- German is related to Spanish.
axiom A6 : Related German Spanish
-- Basque is not related to any other language.
axiom A7a : ∀ x : obj, ¬ Related Basque x
axiom A7b : ∀ x : obj, ¬ Related x Basque

theorem french_is_indo_european : IndoEuropean French := by
  exact A1 French A4
</lean>

ANSWER: True
