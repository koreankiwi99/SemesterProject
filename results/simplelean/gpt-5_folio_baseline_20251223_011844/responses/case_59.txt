<lean>
axiom City : Type
axiom State : Type

axiom Billings : City
axiom Butte : City
axiom Helena : City
axiom Missoula : City
axiom WhiteSulphurSprings : City
axiom StPierre : City
axiom Bristol : City
axiom Texarkana : City
axiom Texhoma : City
axiom UnionCity : City

axiom Montana : State

axiom InUSState : City → State → Prop

-- Context facts
axiom billings_in_montana : InUSState Billings Montana
axiom butte_in_montana : InUSState Butte Montana
axiom helena_in_montana : InUSState Helena Montana
axiom missoula_in_montana : InUSState Missoula Montana

axiom stpierre_not_in_montana : ¬ InUSState StPierre Montana

-- Cities in the same state: White Sulphur Springs and Butte
axiom wss_same_as_butte_left : ∀ s : State, InUSState WhiteSulphurSprings s → InUSState Butte s
axiom wss_same_as_butte_right : ∀ s : State, InUSState Butte s → InUSState WhiteSulphurSprings s

-- Uniqueness of state membership except for some cities
axiom unique_state_except :
  ∀ (c : City) (s1 s2 : State),
    InUSState c s1 → InUSState c s2 →
    s1 = s2 ∨ c = Bristol ∨ c = Texarkana ∨ c = Texhoma ∨ c = UnionCity

-- Claim: Montana is home to the city of Missoula
theorem montana_home_missoula : InUSState Missoula Montana := missoula_in_montana
</lean>

ANSWER: True
