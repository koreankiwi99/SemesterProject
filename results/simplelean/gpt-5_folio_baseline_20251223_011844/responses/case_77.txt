<lean>
axiom obj : Type

axiom AtMS : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom Picky : obj → Prop
axiom SpendDining : obj → Prop

-- Rules about people at Mary's school
axiom R1 : ∀ x : obj, AtMS x ∧ Work x → Need x
axiom R2 : ∀ x : obj, AtMS x ∧ OrderTakeout x → Work x
axiom R3 : ∀ x : obj, AtMS x → (OrderTakeout x ∨ EnjoyDining x)
axiom R4 : ∀ x : obj, AtMS x ∧ EnjoyDining x → ¬ Picky x
axiom R5 : ∀ x : obj, AtMS x ∧ EnjoyDining x → SpendDining x

-- Hannah-specific facts
axiom Hannah : obj
axiom H_at : AtMS Hannah
axiom H_work : Work Hannah
axiom H_need_imp : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah)

-- Goal: Hannah is at Mary's school and either she is not picky
-- or, if she is, then she spends a lot of her time in the dining halls.
theorem hannah_statement :
  AtMS Hannah ∧ (¬ Picky Hannah ∨ (Picky Hannah → SpendDining Hannah)) := by
  have hNeed : Need Hannah := R1 Hannah (And.intro H_at H_work)
  have hNotPicky : ¬ Picky Hannah := (H_need_imp hNeed).left
  exact And.intro H_at (Or.inl hNotPicky)
</lean>

ANSWER: True
