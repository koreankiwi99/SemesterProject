<lean>
axiom obj : Type
axiom HPS : obj

axiom Listed : obj → Prop
axiom ManyNeg : obj → Prop
axiom RatingGT4 : obj → Prop
axiom Takeout : obj → Prop
axiom Popular : obj → Prop

axiom A1 : ∀ x : obj, Listed x → ¬ ManyNeg x
axiom A2 : ∀ x : obj, RatingGT4 x → Listed x
axiom A3 : ∃ x : obj, ¬ Takeout x ∧ ManyNeg x
axiom A4 : ∀ x : obj, Popular x → RatingGT4 x
axiom A5 : RatingGT4 HPS ∨ Popular HPS

theorem hps_claim :
  (Takeout HPS ∧ ManyNeg HPS) → (RatingGT4 HPS ∧ ¬ Takeout HPS) :=
by
  intro h
  have hT : Takeout HPS := And.left h
  have hM : ManyNeg HPS := And.right h
  have hRG : RatingGT4 HPS := by
    cases A5 with
    | inl rg => exact rg
    | inr pop => exact A4 HPS pop
  have hListed : Listed HPS := A2 HPS hRG
  have hNotM : ¬ ManyNeg HPS := A1 HPS hListed
  have contra : False := hNotM hM
  exact And.intro hRG (fun _ => False.elim contra)
</lean>

ANSWER: True
