<lean>
axiom obj : Type
axiom HPS : obj

axiom Listed : obj → Prop
axiom ManyNeg : obj → Prop
axiom TakeOut : obj → Prop
axiom RatingGT4 : obj → Prop
axiom Popular : obj → Prop

-- If a restaurant is listed in Yelp’s recommendations, then the restaurant has not received many negative reviews.
axiom A1 : ∀ x : obj, Listed x → ¬ ManyNeg x
-- All restaurants with a rating greater than four are listed in Yelp’s recommendations.
axiom A2 : ∀ x : obj, RatingGT4 x → Listed x
-- Some restaurants that do not provide take-out service receive many negative reviews.
axiom A3 : ∃ x : obj, ¬ TakeOut x ∧ ManyNeg x
-- All restaurants that are popular among local residents have ratings greater than four.
axiom A4 : ∀ x : obj, Popular x → RatingGT4 x
-- The Hamden Plaza Subway store has a rating greater than four, or it is popular among local residents.
axiom A5 : RatingGT4 HPS ∨ Popular HPS

theorem hps_takeout_or_not_manyneg : TakeOut HPS ∨ ¬ ManyNeg HPS :=
by
  right
  have rem : RatingGT4 HPS ∨ Popular HPS := A5
  cases rem with
  | inl hgt =>
      have hl : Listed HPS := A2 HPS hgt
      exact A1 HPS hl
  | inr hpop =>
      have hgt : RatingGT4 HPS := A4 HPS hpop
      have hl : Listed HPS := A2 HPS hgt
      exact A1 HPS hl
</lean>

ANSWER: True
