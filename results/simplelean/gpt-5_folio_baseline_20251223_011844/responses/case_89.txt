<lean>
axiom obj : Type
axiom Person : obj → Prop
axiom Book : obj → Prop
axiom Reads : obj → obj → Prop
axiom GainsKnowledge : obj → Prop
axiom Smarter : obj → Prop
axiom ContainsKnowledge : obj → Prop

axiom Harry : obj
axiom Walden : obj

-- Books contain tons of knowledge.
axiom K1 : ∀ b : obj, Book b → ContainsKnowledge b
-- When a person reads a book, that person gains knowledge.
axiom R1 : ∀ p b : obj, Person p → Book b → Reads p b → GainsKnowledge p
-- If a person gains knowledge, they become smarter.
axiom R2 : ∀ p : obj, GainsKnowledge p → Smarter p
-- Harry read the book “Walden” by Henry Thoreau.
axiom H1 : Person Harry
axiom H2 : Book Walden
axiom H3 : Reads Harry Walden

-- Statement: A smarter person has gained knowledge (interpreted existentially).
theorem smarter_person_has_gained_knowledge : ∃ p : obj, Person p ∧ Smarter p ∧ GainsKnowledge p :=
by
  have hk : GainsKnowledge Harry := R1 Harry Walden H1 H2 H3
  have hs : Smarter Harry := R2 Harry hk
  exact Exists.intro Harry (And.intro H1 (And.intro hs hk))
</lean>

ANSWER: True
