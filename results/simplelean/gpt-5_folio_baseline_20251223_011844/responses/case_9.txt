<lean>
axiom obj : Type
axiom Tom : obj

axiom WildTurkey : obj → Prop
axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Goulds : obj → Prop
axiom Merriam : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop

-- Every wild turkey is of one of the six listed types
axiom six_types :
  ∀ x : obj, WildTurkey x →
    (Eastern x ∨ Osceola x ∨ Goulds x ∨ Merriam x ∨ RioGrande x ∨ Ocellated x)

-- Given facts about Tom
axiom notEastern : ¬ Eastern Tom
axiom notOsceola : ¬ Osceola Tom
axiom notGoulds : ¬ Goulds Tom
axiom notMerriam : ¬ Merriam Tom
axiom notRioGrande : ¬ RioGrande Tom
axiom tomWild : WildTurkey Tom

theorem tom_is_ocellated : Ocellated Tom := by
  have h := six_types Tom tomWild
  cases h with
  | inl hE => exact False.elim (notEastern hE)
  | inr h1 =>
    cases h1 with
    | inl hO => exact False.elim (notOsceola hO)
    | inr h2 =>
      cases h2 with
      | inl hG => exact False.elim (notGoulds hG)
      | inr h3 =>
        cases h3 with
        | inl hM => exact False.elim (notMerriam hM)
        | inr h4 =>
          cases h4 with
          | inl hR => exact False.elim (notRioGrande hR)
          | inr hOc => exact hOc
</lean>

ANSWER: True
