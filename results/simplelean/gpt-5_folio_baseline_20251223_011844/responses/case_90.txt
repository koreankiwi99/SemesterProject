<lean>
axiom obj : Type
axiom MT : obj
axiom NLP : obj → Prop
axiom LG : obj → Prop
axiom LU : obj → Prop
axiom OutputText : obj → Prop

-- Every NLP task can be categorized as either generation or understanding
axiom categorize : ∀ x : obj, NLP x → (LG x ∨ LU x)
-- If the output of an NLP task is a text sequence, then it is a generation task
axiom output_rule : ∀ x : obj, NLP x → OutputText x → LG x

-- Facts about machine translation
axiom MT_is_NLP : NLP MT
axiom MT_output_text : OutputText MT

theorem MT_is_LG : LG MT := by
  exact output_rule MT MT_is_NLP MT_output_text
</lean>

ANSWER: True
