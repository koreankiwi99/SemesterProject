<lean>
axiom obj : Type

-- Predicates and functions
axiom Managed : obj → Prop
axiom MonthlyRent : obj → Nat
axiom SecurityDeposit : obj → Nat

-- Specific building
axiom OliveGarden : obj
axiom OG_managed : Managed OliveGarden
axiom OG_rent_2000 : MonthlyRent OliveGarden = 2000

-- Rule: at any managed building, the security deposit is at least the monthly rent
axiom deposit_ge_rent : ∀ b : obj, Managed b → SecurityDeposit b ≥ MonthlyRent b

-- Target: The security deposit at the Olive Garden is either $2000 or more (i.e., ≥ 2000)
theorem og_deposit_ge_2000 : SecurityDeposit OliveGarden ≥ 2000 := by
  have h := deposit_ge_rent OliveGarden OG_managed
  simpa [OG_rent_2000] using h
</lean>

ANSWER: True
