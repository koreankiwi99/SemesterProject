<lean>
axiom Alg : Type
axiom alg : Alg

axiom ML : Alg → Prop
axiom Supervised : Alg → Prop
axiom Unsupervised : Alg → Prop
axiom Reinforcement : Alg → Prop
axiom RequiresLabeled : Alg → Prop

-- Categorization: any ML algorithm is supervised, unsupervised, or reinforcement
axiom categorize : ∀ a : Alg, ML a → (Supervised a ∨ Unsupervised a ∨ Reinforcement a)
-- Unsupervised algorithms do not require labeled data
axiom unsup_no_labels : ∀ a : Alg, Unsupervised a → ¬ RequiresLabeled a
-- The SOTA text summarization model is trained with an ML algorithm 'alg'
axiom trained_with_ML : ML alg
-- Reinforcement learning is not used to train the SOTA model
axiom not_rl_used : ¬ Reinforcement alg
-- The ML algorithm for training text summarization models requires labeled data
axiom requires_labels : RequiresLabeled alg

theorem supervised_used : Supervised alg := by
  have hdisj := categorize alg trained_with_ML
  have hnotU : ¬ Unsupervised alg := fun hU => by
    have hnotReq := unsup_no_labels alg hU
    exact hnotReq requires_labels
  have hnotR : ¬ Reinforcement alg := not_rl_used
  cases hdisj with
  | inl hS =>
    exact hS
  | inr hUR =>
    cases hUR with
    | inl hU =>
      exact False.elim (hnotU hU)
    | inr hR =>
      exact False.elim (hnotR hR)
</lean>
