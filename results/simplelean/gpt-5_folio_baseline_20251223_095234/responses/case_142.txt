<lean>
axiom obj : Type

-- Predicates
axiom Animal : obj → Prop
axiom HeckCattle : obj → Prop
axiom Aurochs : obj → Prop
axiom BredBack : obj → Prop
axiom ArtificiallySelected : obj → Prop

-- Optional predicates from context (not needed for the proof)
axiom ToBeBredBack : obj → Prop
axiom Extinct : obj → Prop
axiom Resembles : obj → obj → Prop

-- Axioms from the textual context
-- Breeding back is a form of artificial selection
axiom R1 : ∀ x : obj, BredBack x → ArtificiallySelected x
-- Heck cattle were bred back
axiom T1 : ∃ x : obj, HeckCattle x ∧ BredBack x
-- Heck cattle are animals
axiom R2 : ∀ x : obj, HeckCattle x → Animal x
-- Aurochs are animals
axiom R3 : ∀ x : obj, Aurochs x → Animal x
-- Some animals to be bred back resemble extinct animals (not used)
axiom T2 : ∃ x y : obj, Animal x ∧ ToBeBredBack x ∧ Extinct y ∧ Resembles x y

-- Target: Some Heck cattle are artificially selected
theorem some_heck_cattle_are_artificially_selected : ∃ x : obj, HeckCattle x ∧ ArtificiallySelected x := by
  have ⟨x, hx⟩ := T1
  have hxHC : HeckCattle x := And.left hx
  have hxBB : BredBack x := And.right hx
  have hxAS : ArtificiallySelected x := R1 x hxBB
  exact Exists.intro x (And.intro hxHC hxAS)
</lean>

ANSWER: True
