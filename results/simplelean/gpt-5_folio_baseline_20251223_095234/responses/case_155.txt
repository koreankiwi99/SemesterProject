axiom Mortal : obj → Prop

axiom AllHumanMortal : ∀ x : obj, Human x → Mortal x
axiom AllGreekHuman : ∀ x : obj, Greek x → Human x

-- Target: ∃ x, Greek x ∧ Mortal x
-- Cannot be proven without assuming the existence of a Greek.

-- From the existence of a Greek, we can derive the existence of a mortal Greek.
theorem exists_mortal_greek_of_exists_greek : (∃ x : obj, Greek x) → ∃ x : obj, Greek x ∧ Mortal x :=
by
  intro h
  cases h with
  | intro g gG =>
    have gH : Human g := AllGreekHuman g gG
    have gM : Mortal g := AllHumanMortal g gH
    exact ⟨g, And.intro gG gM⟩
