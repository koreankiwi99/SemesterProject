<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom WorkLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom PartTime : obj → Prop
axiom Database : obj → Prop
axiom ClassWithDavid : obj → Prop
axiom Lab : obj → Prop

-- None of the students who currently work in the library are from CS
axiom A1 : ∀ x : obj, Student x → WorkLibrary x → ¬ FromCS x
-- All students with part-time university jobs work in the library
axiom A2 : ∀ x : obj, Student x → PartTime x → WorkLibrary x
-- All students taking the database course are from CS
axiom A3 : ∀ x : obj, Student x → Database x → FromCS x
-- All students taking a class with Prof David are taking the database course
axiom A4 : ∀ x : obj, Student x → ClassWithDavid x → Database x
-- All students working in the lab are taking a class with Prof David
axiom A5 : ∀ x : obj, Student x → Lab x → ClassWithDavid x

-- James is a student working in the lab
axiom T1 : Student James
axiom T2 : Lab James
-- James doesn't work in the lab or have a part-time job (i.e., not (Lab James or PartTime James))
axiom T3 : ¬ (Lab James ∨ PartTime James)

-- Prove: James has a part-time job offered by the university
theorem james_parttime : PartTime James := by
  have notLab : ¬ Lab James := fun h => T3 (Or.inl h)
  have f : False := notLab T2
  exact False.elim f
</lean>

ANSWER: True
