<lean>
axiom obj : Type
axiom James : obj

axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom PartTimeUni : obj → Prop
axiom TakingDB : obj → Prop
axiom WithDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- 1. None who work in the library are from CS
axiom A1 : ∀ x : obj, WorksInLibrary x → ¬ FromCS x
-- 2. All with university part-time jobs work in the library
axiom A2 : ∀ x : obj, PartTimeUni x → WorksInLibrary x
-- 3. All taking the database course are from CS
axiom A3 : ∀ x : obj, TakingDB x → FromCS x
-- 4. All taking a class with Professor David are taking the database course
axiom A4 : ∀ x : obj, WithDavid x → TakingDB x
-- 5. All working in the lab are taking a class with Professor David
axiom A5 : ∀ x : obj, WorksInLab x → WithDavid x
-- 6. James works in the lab
axiom A6 : WorksInLab James
-- 7. James doesn't work in the lab or have a university part-time job (interpreted as a disjunction)
axiom A7 : ¬ WorksInLab James ∨ ¬ PartTimeUni James

theorem james_not_parttime_uni : ¬ PartTimeUni James := by
  intro hpt
  have wl : WorksInLibrary James := A2 James hpt
  have ncs : ¬ FromCS James := A1 James wl
  have wd : WithDavid James := A5 James A6
  have db : TakingDB James := A4 James wd
  have cs : FromCS James := A3 James db
  exact ncs cs
</lean>

ANSWER: True
