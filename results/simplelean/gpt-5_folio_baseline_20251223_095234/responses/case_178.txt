<lean>
axiom obj : Type
axiom Horse : obj → Prop
axiom Human : obj → Prop
axiom HasHooves : obj → Prop

axiom allHorseHooves : ∀ x : obj, Horse x → HasHooves x
axiom noHumanHooves : ∀ x : obj, Human x → ¬ HasHooves x

theorem no_human_is_horse : ¬ (∃ x : obj, Human x ∧ Horse x) := by
  intro hex
  cases hex with
  | intro x hx =>
    cases hx with
    | intro hxHuman hxHorse =>
      have hh : HasHooves x := allHorseHooves x hxHorse
      have nh : ¬ HasHooves x := noHumanHooves x hxHuman
      exact nh hh
</lean>

ANSWER: False
