axiom LeapsStraightUp : obj → Prop

axiom pros_can_jump : ∀ x : obj, ProfessionalBasketballPlayer x → CanJumpShots x
axiom jump_implies_leap : ∀ x : obj, CanJumpShots x → LeapsStraightUp x
axiom leap_implies_activate : ∀ x : obj, LeapsStraightUp x → ActivatesLegMuscles x
axiom yuri_not_activate : ¬ ActivatesLegMuscles Yuri

  have hact : ActivatesLegMuscles Yuri := leap_implies_activate Yuri hleap
  exact yuri_not_activate hact
