<lean>
axiom obj : Type
axiom Beijing : obj

axiom Hosted2008SummerOlympics : obj → Prop
axiom HostedWinterOlympics : obj → Prop

-- From the context:
axiom H2008 : Hosted2008SummerOlympics Beijing
axiom HWinter : HostedWinterOlympics Beijing

theorem beijing_hosted_both :
  Hosted2008SummerOlympics Beijing ∧ HostedWinterOlympics Beijing := by
  constructor
  · exact H2008
  · exact HWinter
</lean>

ANSWER: True
