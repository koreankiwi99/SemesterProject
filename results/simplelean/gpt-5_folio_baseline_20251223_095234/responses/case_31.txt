<lean>
axiom obj : Type

axiom Mixer : obj → Prop
axiom GrandSlam : obj → Prop
axiom OscarNom : obj → Prop
axiom ProTennis : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom TaxHaven : obj → Prop

axiom Djokovic : obj

-- Everyone at the mixer is a Grand Slam champion or an Oscar-nominated actor.
axiom A1 : ∀ x : obj, Mixer x → (GrandSlam x ∨ OscarNom x)
-- Every Grand Slam champion at the mixer is a professional tennis player.
axiom A2 : ∀ x : obj, Mixer x → GrandSlam x → ProTennis x
-- All Oscar-nominated actors at the mixer are celebrities.
axiom A3 : ∀ x : obj, Mixer x → OscarNom x → Celebrity x
-- All professional tennis players at the mixer are athletes.
axiom A4 : ∀ x : obj, Mixer x → ProTennis x → Athlete x
-- If a person at the mixer is a celebrity, then they are well paid.
axiom A5 : ∀ x : obj, Mixer x → Celebrity x → WellPaid x
-- If a person at the mixer is an athlete, then they are famous.
axiom A6 : ∀ x : obj, Mixer x → Athlete x → Famous x
-- All well-paid people at the mixer live in tax havens.
axiom A7 : ∀ x : obj, Mixer x → WellPaid x → TaxHaven x

-- Djokovic is at the mixer
axiom A8 : Mixer Djokovic
-- If Djokovic is a famous athlete, then Djokovic is well-paid.
axiom A9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

theorem djokovic_taxhaven : TaxHaven Djokovic := by
  have djm : Mixer Djokovic := A8
  have disj : GrandSlam Djokovic ∨ OscarNom Djokovic := A1 Djokovic djm
  cases disj with
  | inl gs =>
      have pro : ProTennis Djokovic := A2 Djokovic djm gs
      have ath : Athlete Djokovic := A4 Djokovic djm pro
      have fam : Famous Djokovic := A6 Djokovic djm ath
      have wp : WellPaid Djokovic := A9 ⟨fam, ath⟩
      exact A7 Djokovic djm wp
  | inr osc =>
      have cel : Celebrity Djokovic := A3 Djokovic djm osc
      have wp : WellPaid Djokovic := A5 Djokovic djm cel
      exact A7 Djokovic djm wp
</lean>

ANSWER: True
