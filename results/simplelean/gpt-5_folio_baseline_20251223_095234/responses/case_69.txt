<lean>
axiom obj : Type
axiom Badults : obj

axiom MemberOfPappys : obj → Prop
axiom StarredIn : obj → obj → Prop
axiom PilotedOnBBCTwo : obj → Prop
axiom PilotedOnBBCThree : obj → Prop

-- Context axioms:
-- "Badults is a British Sitcom series starring members of Pappy's."
axiom A1 : ∃ p : obj, MemberOfPappys p ∧ StarredIn p Badults
-- "Badults was piloted in July 2013 on BBC Three."
axiom A2 : PilotedOnBBCThree Badults

-- Negation of the claim ("No members of Pappy's have starred in a show piloting on BBC Two or BBC Three")
-- is the existence of a Pappy member starring in a show piloted on BBC Two or BBC Three.
theorem exists_pappy_member_in_bbc2_or_bbc3_pilot :
  ∃ p s : obj, MemberOfPappys p ∧ StarredIn p s ∧ (PilotedOnBBCTwo s ∨ PilotedOnBBCThree s) := by
  rcases A1 with ⟨p, hp_mem, hp_star⟩
  refine ⟨p, Badults, ?_⟩
  exact And.intro hp_mem (And.intro hp_star (Or.inr A2))
</lean>

ANSWER: False
