<lean>
axiom obj : Type

axiom Person : obj → Prop
axiom Book : obj → Prop
axiom Reads : obj → obj → Prop
axiom GainsKnowledge : obj → Prop
axiom Smarter : obj → Prop

axiom Harry : obj
axiom Walden : obj

axiom HP : Person Harry
axiom BW : Book Walden
axiom Hread : Reads Harry Walden

axiom R_read_gain : ∀ x y, Person x → Book y → Reads x y → GainsKnowledge x
axiom R_gain_smarter : ∀ x, GainsKnowledge x → Smarter x

theorem smarter_person_has_gained : ∃ x, Person x ∧ Smarter x ∧ GainsKnowledge x := by
  have g : GainsKnowledge Harry := R_read_gain Harry Walden HP BW Hread
  have s : Smarter Harry := R_gain_smarter Harry g
  exact Exists.intro Harry (And.intro HP (And.intro s g))
</lean>

ANSWER: True
