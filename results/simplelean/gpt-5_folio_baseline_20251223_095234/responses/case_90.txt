<lean>
axiom obj : Type
axiom MachineTranslation : obj
axiom NLPTask : obj → Prop
axiom LangGen : obj → Prop
axiom LangUnd : obj → Prop
axiom OutputTextSeq : obj → Prop

-- A natural language processing task can be categorized as either a generation or understanding task
axiom A1 : ∀ x : obj, NLPTask x → (LangGen x ∨ LangUnd x)
-- If the output is a text sequence, then it is a language generation task
axiom A2 : ∀ x : obj, OutputTextSeq x → LangGen x
-- Machine translation is a natural language processing task
axiom A3 : NLPTask MachineTranslation
-- The output of machine translation is a text sequence
axiom A4 : OutputTextSeq MachineTranslation

theorem mt_is_generation : LangGen MachineTranslation := by
  exact A2 MachineTranslation A4
</lean>

ANSWER: True
