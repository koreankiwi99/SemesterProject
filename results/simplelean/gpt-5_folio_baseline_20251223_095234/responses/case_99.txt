<lean>
axiom obj : Type

axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom ReproMF : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop

axiom Harry : obj

-- Axioms from the context
axiom A1 : ∀ x : obj, Invertebrate x ∨ Vertebrate x
axiom A2 : ∀ x : obj, Backbone x → ReproMF x
axiom A3 : ∀ x : obj, Vertebrate x → Backbone x
axiom A4 : ∀ x : obj, Bee x → ¬ ReproMF x
axiom A5 : ∀ x : obj, QueenBee x → Bee x
axiom A6 : Bee Harry

theorem harry_invertebrate_or_queenbee : Invertebrate Harry ∨ QueenBee Harry := by
  have h_not_repro : ¬ ReproMF Harry := A4 Harry A6
  have h_not_vert : ¬ Vertebrate Harry := fun hv => h_not_repro (A2 Harry (A3 Harry hv))
  have h_or := A1 Harry
  have h_inv : Invertebrate Harry := by
    cases h_or with
    | inl hinv => exact hinv
    | inr hvert =>
        have hrep : ReproMF Harry := A2 Harry (A3 Harry hvert)
        exact False.elim (h_not_repro hrep)
  exact Or.inl h_inv
</lean>

ANSWER: True
