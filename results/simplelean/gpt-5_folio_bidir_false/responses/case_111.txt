axiom Picuris : obj
axiom Juan : obj
axiom HardingMine : obj

-- Predicates and relations
axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

-- Context axioms
axiom T1 : MountainRange Picuris
axiom T2 : InNewMexico Picuris ∨ InTexas Picuris
axiom T3 : Visited Juan Picuris
axiom T4 : Mine HardingMine
axiom T5 : LocatedIn HardingMine Picuris
axiom T6 : Donated HardingMine

-- No Texas mountain range has a donated mine located in it
axiom R1 : ∀ r m : obj, MountainRange r → InTexas r → Mine m → LocatedIn m r → Donated m → False

-- From the donated mine located in Picuris, Picuris cannot be in Texas
theorem not_in_texas_picuris : ¬InTexas Picuris := by
  intro htx
  exact R1 Picuris HardingMine T1 htx T4 T5 T6

-- Therefore Picuris is in New Mexico
theorem picuris_in_nm : InNewMexico Picuris := by
  exact Or.resolve_right T2 not_in_texas_picuris

-- Hence there exists a mountain range in New Mexico that Juan visited (namely, Picuris)
theorem exists_nm_range_visited : ∃ r, MountainRange r ∧ InNewMexico r ∧ Visited Juan r := by
  refine Exists.intro Picuris ?_
  exact And.intro T1 (And.intro picuris_in_nm T3)
