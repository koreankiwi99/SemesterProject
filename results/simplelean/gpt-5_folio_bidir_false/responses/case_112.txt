axiom InTexas : obj → Prop

-- Context axioms
axiom A1 : MountainRange Picuris

axiom A4 : Mine HardingPegmatiteMine

axiom A7 : ∀ x : obj, MountainRange x → InTexas x → ¬ ∃ y : obj, Mine y ∧ LocatedIn y x ∧ Donated y
-- "There are no mountain ranges in Texas that have mines that have been donated."

-- From A4, A5, A6 we get that Picuris has a donated mine located in it.
theorem picuris_has_donated_mine : ∃ y : obj, Mine y ∧ LocatedIn y Picuris ∧ Donated y := by
  refine Exists.intro HardingPegmatiteMine ?_
  exact And.intro A4 (And.intro A5 A6)

-- Therefore Picuris is not in Texas (otherwise it would contradict A7).
theorem picuris_not_in_texas : ¬ InTexas Picuris := by
  intro hTX
  have ex := picuris_has_donated_mine
  have noex : ¬ ∃ y : obj, Mine y ∧ LocatedIn y Picuris ∧ Donated y := A7 Picuris A1 hTX
  exact noex ex
