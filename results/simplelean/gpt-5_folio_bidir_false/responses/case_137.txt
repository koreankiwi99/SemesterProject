axiom Supervised : Algorithm → Prop
axiom Unsupervised : Algorithm → Prop
axiom Reinforcement : Algorithm → Prop
axiom RequiresLabeled : Algorithm → Prop
axiom UsedToTrain : Algorithm → Model → Prop

-- Context formalization
axiom T_used : UsedToTrain AlgTS SOTA
axiom T_ml : ML AlgTS
axiom R_uns_no_labels : ∀ a : Algorithm, Unsupervised a → ¬RequiresLabeled a
axiom T_not_rl : ¬Reinforcement AlgTS
axiom T_requires : RequiresLabeled AlgTS
axiom R_categorized : ∀ a : Algorithm, ML a → (Supervised a ∨ Unsupervised a ∨ Reinforcement a)

-- From "unsupervised ⇒ does not require labels" and "AlgTS requires labels", derive ¬unsupervised AlgTS
theorem not_unsupervised_AlgTS : ¬Unsupervised AlgTS := by
  intro hU
  have hNotReq : ¬RequiresLabeled AlgTS := R_uns_no_labels AlgTS hU
  exact hNotReq T_requires

-- From categorization and the two negations, deduce that AlgTS is supervised
theorem supervised_used : Supervised AlgTS := by
  have h := R_categorized AlgTS T_ml
  cases h with
  | inl hS => exact hS
  | inr h' =>
      cases h' with
      | inl hU =>
          have f : False := not_unsupervised_AlgTS hU
          exact False.elim f
      | inr hR =>
          have f : False := T_not_rl hR
          exact False.elim f
