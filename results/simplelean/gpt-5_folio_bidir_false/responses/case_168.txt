-- Statement: No politicians are part of the Nazi Reichstag.
def NoPoliticiansInNaziReichstag : Prop := ∀ x : obj, Politician x → ¬NaziReichstag x

by
  intro H
  have hnot : ¬NaziReichstag HeinrichSchmidt := H HeinrichSchmidt HS_politician
  exact hnot HS_reichstag

-- Equivalent explicit counterexample
theorem counterexample_exists : ∃ x : obj, Politician x ∧ NaziReichstag x :=
by
  exact Exists.intro HeinrichSchmidt (And.intro HS_politician HS_reichstag)
