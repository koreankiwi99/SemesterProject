<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatsSaladsRegularly : obj → Prop
axiom VeryConscious : obj → Prop
axiom GrewUpHealthConsciousHome : obj → Prop
axiom FulfillDailyNutrition : obj → Prop
axiom DisregardPhysicalWellBeing : obj → Prop
axiom VisitsGymDaily : obj → Prop

-- All who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : obj, EatsSaladsRegularly x → VeryConscious x
-- All who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : obj, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
-- All who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : obj, FulfillDailyNutrition x → GrewUpHealthConsciousHome x
-- All who disregard their physical well-being are not very conscious about their health and eating habits.
axiom A4 : ∀ x : obj, DisregardPhysicalWellBeing x → ¬ VeryConscious x
-- If people visit the gym at least once a day, then they fulfill their daily nutritional intakes.
axiom A5 : ∀ x : obj, VisitsGymDaily x → FulfillDailyNutrition x
-- Taylor either grew up in a health-conscious childhood home and disregards physical well-being, or she did neither.
axiom A6 : (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellBeing Taylor) ∨ (¬GrewUpHealthConsciousHome Taylor ∧ ¬DisregardPhysicalWellBeing Taylor)

theorem taylor_not_gym : ¬VisitsGymDaily Taylor := by
  intro hVGD
  have hFDN : FulfillDailyNutrition Taylor := A5 Taylor hVGD
  have hGHC : GrewUpHealthConsciousHome Taylor := A3 Taylor hFDN
  have hESR : EatsSaladsRegularly Taylor := A2 Taylor hGHC
  have hVC  : VeryConscious Taylor := A1 Taylor hESR
  cases A6 with
  | inl h1 =>
      have hDPW : DisregardPhysicalWellBeing Taylor := h1.right
      have hnotVC : ¬ VeryConscious Taylor := A4 Taylor hDPW
      exact hnotVC hVC
  | inr h2 =>
      have hnotGHC : ¬ GrewUpHealthConsciousHome Taylor := h2.left
      exact hnotGHC hGHC
</lean>

ANSWER: False
