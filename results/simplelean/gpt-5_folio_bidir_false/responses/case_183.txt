<lean>
axiom obj : Type
axiom Tom : obj

axiom Nice : obj → Prop
axiom Mean : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop

-- No one nice to animals is also mean to animals.
axiom A1 : ∀ x : obj, Nice x → ¬ Mean x
-- Some grumpy people are mean to animals.
axiom A2 : ∃ x : obj, Grumpy x ∧ Mean x
-- All animal lovers are nice to animals.
axiom A3 : ∀ x : obj, AnimalLover x → Nice x
-- All pet owners love animals.
axiom A4 : ∀ x : obj, PetOwner x → AnimalLover x
-- Tom is a pet owner.
axiom A5 : PetOwner Tom

-- Target statement to negate:
-- If Tom is not both grumpy and mean, then Tom is neither grumpy nor an animal lover.
-- We prove the negation of this implication.
theorem neg_target :
  ¬ ( (¬(Grumpy Tom ∧ Mean Tom)) → (¬Grumpy Tom ∧ ¬AnimalLover Tom) ) :=
by
  have aL : AnimalLover Tom := A4 Tom A5
  have nM : ¬Mean Tom :=
    by
      have n : Nice Tom := A3 Tom aL
      exact A1 Tom n
  have ante : ¬(Grumpy Tom ∧ Mean Tom) :=
    by
      intro h
      have m : Mean Tom := And.right h
      exact nM m
  intro hImp
  have hconj : ¬Grumpy Tom ∧ ¬AnimalLover Tom := hImp ante
  have nAL : ¬AnimalLover Tom := And.right hconj
  exact nAL aL
</lean>

ANSWER: False
