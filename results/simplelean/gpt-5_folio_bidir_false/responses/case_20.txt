<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Turtle : obj → Prop
axiom Squirrel : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop
axiom Cute : obj → Prop
axiom Spotted : obj → Prop
axiom Rockie : obj

-- All rabbits that can be spotted near the campus are cute.
axiom A1 : ∀ x : obj, (Rabbit x ∧ Spotted x) → Cute x
-- Some turtles can be spotted near the campus.
axiom A2 : ∃ x : obj, Turtle x ∧ Spotted x
-- The only animals that can be spotted near the campus are rabbits and squirrels.
axiom A3 : ∀ x : obj, Spotted x → (Rabbit x ∨ Squirrel x)
-- If something is skittish, then it is not calm.
axiom A4 : ∀ x : obj, Skittish x → ¬Calm x
-- All the squirrels that can be spotted near the campus are skittish.
axiom A5 : ∀ x : obj, (Squirrel x ∧ Spotted x) → Skittish x
-- Rockie can be spotted near the campus, and it is calm.
axiom A6 : Spotted Rockie ∧ Calm Rockie

-- Statement: If Rockie is cute and calm, then Rockie is a skittish turtle.
-- We prove the negation of this statement.
theorem negate_statement :
  ¬((Cute Rockie ∧ Calm Rockie) → (Skittish Rockie ∧ Turtle Rockie)) := by
  have hS : Spotted Rockie := And.left A6
  have hC : Calm Rockie := And.right A6
  -- ¬(Squirrel Rockie)
  have hNotSq : ¬Squirrel Rockie := by
    intro hSq
    have hSk : Skittish Rockie := by
      have hsqs : Squirrel Rockie ∧ Spotted Rockie := And.intro hSq hS
      exact A5 Rockie hsqs
    have hNotC : ¬Calm Rockie := A4 Rockie hSk
    exact hNotC hC
  -- From Spotted Rockie and A3, get Rabbit Rockie ∨ Squirrel Rockie
  have hRS : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie hS
  -- Deduce Rabbit Rockie
  have hR : Rabbit Rockie := by
    cases hRS with
    | inl hR => exact hR
    | inr hSq =>
      have : False := hNotSq hSq
      exact False.elim this
  -- Hence Cute Rockie by A1
  have hCute : Cute Rockie := by
    have h : Rabbit Rockie ∧ Spotted Rockie := And.intro hR hS
    exact A1 Rockie h
  -- ¬Skittish Rockie from Calm Rockie and A4
  have hNotSk : ¬Skittish Rockie := by
    intro hSk
    have hNotC : ¬Calm Rockie := A4 Rockie hSk
    exact hNotC hC
  -- Antecedent holds
  have hAnte : Cute Rockie ∧ Calm Rockie := And.intro hCute hC
  -- Consequent is false
  have hNotCons : ¬(Skittish Rockie ∧ Turtle Rockie) := by
    intro h
    exact hNotSk h.left
  -- Conclude negation of the implication
  intro hImp
  exact hNotCons (hImp hAnte)
</lean>

ANSWER: False
