-- All employees who schedule a meeting with their customers will go to the company building today.
axiom R1 : ∀ x : obj, (Employee x ∧ SchedulesMeetingsWithCustomers x) → InCompanyToday x
-- Everyone who has lunch in the company building schedules meetings with their customers.
axiom R2 : ∀ x : obj, LunchInCompanyBuilding x → SchedulesMeetingsWithCustomers x
-- Employees have lunch either in the company building or at home.
axiom R3 : ∀ x : obj, Employee x → (LunchInCompanyBuilding x ∨ LunchAtHome x)
-- If an employee has lunch at home, they are working remotely from home.
axiom R4 : ∀ x : obj, (Employee x ∧ LunchAtHome x) → RemoteFromHome x
-- All employees who are in other countries work remotely from home.
axiom R5 : ∀ x : obj, (Employee x ∧ InOtherCountries x) → RemoteFromHome x
-- No managers work remotely from home.
axiom R6 : ∀ x : obj, Manager x → ¬RemoteFromHome x
-- James will appear in the company today iff he is a manager.
axiom R7 : Manager James ↔ InCompanyToday James
