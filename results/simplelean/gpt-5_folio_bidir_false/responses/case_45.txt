axiom AllCatsAreMammals : ∀ x : obj, Cat x → Mammal x
axiom SomePetsNotMammals : ∃ x : obj, Pet x ∧ ¬Mammal x

-- Useful consequence: there exists a pet that is not a cat
theorem some_pet_not_cat : ∃ x : obj, Pet x ∧ ¬Cat x := by
  rcases SomePetsNotMammals with ⟨a, ha⟩
  rcases ha with ⟨hPet, hNotM⟩
  have hNotCat : ¬Cat a := fun hCat => hNotM (AllCatsAreMammals a hCat)
  exact ⟨a, ⟨hPet, hNotCat⟩⟩
