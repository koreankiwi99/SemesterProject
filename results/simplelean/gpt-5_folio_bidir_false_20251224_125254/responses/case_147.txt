axiom Person : Type
axiom Vladimir : Person
axiom Ekaterina : Person

axiom USCitizen : Person → Prop
axiom TWCitizen : Person → Prop
axiom CanRegisterUS : Person → Prop
axiom CanParticipate2024 : Person → Prop
axiom RFOfficial : Person → Prop
axiom GazpromManager : Person → Prop

-- Everyone who can register to vote in the United States can participate in the 2024 United States presidential election.
axiom R1 : ∀ x : Person, CanRegisterUS x → CanParticipate2024 x
-- If someone has United States citizenship, then they can register to vote in the United States.
axiom R2 : ∀ x : Person, USCitizen x → CanRegisterUS x
-- A person either has United States citizenship or Taiwanese citizenship.
axiom R3 : ∀ x : Person, USCitizen x ∨ TWCitizen x
-- No Russian Federation officials hold Taiwanese citizenship.
axiom R4 : ∀ x : Person, RFOfficial x → ¬TWCitizen x
-- Vladimir neither holds Taiwanese citizenship nor is he a manager at Gazprom.
axiom V1 : ¬TWCitizen Vladimir
axiom V2 : ¬GazpromManager Vladimir
-- Ekaterina can register to vote in the United States, or she is a Russian federation official.
axiom E1 : CanRegisterUS Ekaterina ∨ RFOfficial Ekaterina

-- From the premises we can derive that Ekaterina can participate in the 2024 US presidential election.
theorem Ekaterina_participates : CanParticipate2024 Ekaterina := by
  have h : CanRegisterUS Ekaterina ∨ RFOfficial Ekaterina := E1
  cases h with
  | inl hreg =>
      exact R1 Ekaterina hreg
  | inr hrf =>
      have hnotTW : ¬TWCitizen Ekaterina := R4 Ekaterina hrf
      have hcit : USCitizen Ekaterina ∨ TWCitizen Ekaterina := R3 Ekaterina
      have hUSC : USCitizen Ekaterina := by
        cases hcit with
        | inl husc => exact husc
        | inr htw => exact False.elim (hnotTW htw)
      have hreg : CanRegisterUS Ekaterina := R2 Ekaterina hUSC
      exact R1 Ekaterina hreg

-- Therefore, the disjunction "Ekaterina can participate in 2024 US election or Vladimir is a manager at Gazprom" holds.
theorem statement_true : CanParticipate2024 Ekaterina ∨ GazpromManager Vladimir :=
  Or.inl Ekaterina_participates
