<lean>
axiom Person : Type
axiom Taylor : Person

axiom EatSaladsRegularly : Person → Prop
axiom VeryConscious : Person → Prop
axiom GrewUpHealthConsciousHome : Person → Prop
axiom FulfillDailyNutrition : Person → Prop
axiom DisregardPhysicalWellBeing : Person → Prop
axiom VisitGymDaily : Person → Prop

-- All people who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : Person, EatSaladsRegularly x → VeryConscious x
-- All people who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : Person, GrewUpHealthConsciousHome x → EatSaladsRegularly x
-- All people who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : Person, FulfillDailyNutrition x → GrewUpHealthConsciousHome x
-- All people who disregard their physical well-being are not very conscious about their health and eating habits.
axiom A4 : ∀ x : Person, DisregardPhysicalWellBeing x → ¬VeryConscious x
-- If people visit the gym at least once a day, then they always fulfill their daily nutritional intakes.
axiom A5 : ∀ x : Person, VisitGymDaily x → FulfillDailyNutrition x
-- Taylor either grew up in a health-conscious childhood home and disregard her physical well-being, or she did neither.
axiom A6 :
  (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellBeing Taylor) ∨
  (¬GrewUpHealthConsciousHome Taylor ∧ ¬DisregardPhysicalWellBeing Taylor)

-- The statement to evaluate is: ¬GrewUpHealthConsciousHome Taylor ∧ ¬VisitGymDaily Taylor
-- We can actually prove the statement is true from the axioms.

theorem statement_true : ¬GrewUpHealthConsciousHome Taylor ∧ ¬VisitGymDaily Taylor := by
  -- First, show Taylor cannot be both GHC and DPW (it would imply VC and ¬VC).
  have not_both : ¬(GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellBeing Taylor) := by
    intro h
    have hGHC : GrewUpHealthConsciousHome Taylor := h.left
    have hDPW : DisregardPhysicalWellBeing Taylor := h.right
    have hVC : VeryConscious Taylor := A1 Taylor (A2 Taylor hGHC)
    have hNotVC : ¬VeryConscious Taylor := A4 Taylor hDPW
    exact hNotVC hVC
  -- From A6 and not_both, deduce the second disjunct holds: ¬GHC ∧ ¬DPW.
  have hnegGHC_negDPW : ¬GrewUpHealthConsciousHome Taylor ∧ ¬DisregardPhysicalWellBeing Taylor := by
    cases A6 with
    | inl h1 =>
      have : False := not_both h1
      exact False.elim this
    | inr h2 =>
      exact h2
  have hNotGHC : ¬GrewUpHealthConsciousHome Taylor := hnegGHC_negDPW.left
  -- Show that visiting the gym daily would imply GHC.
  have gym_to_ghc : VisitGymDaily Taylor → GrewUpHealthConsciousHome Taylor := by
    intro hGym
    have hFDN : FulfillDailyNutrition Taylor := A5 Taylor hGym
    exact A3 Taylor hFDN
  -- Therefore, ¬GHC implies ¬VisitGymDaily by modus tollens.
  have hNotGym : ¬VisitGymDaily Taylor := by
    intro hGym
    exact hNotGHC (gym_to_ghc hGym)
  exact And.intro hNotGHC hNotGym

-- Therefore, we fail to prove that the conclusion is false.
