<lean>
axiom Person : Type
axiom Airport : Type

axiom Susan : Person
axiom John : Person
axiom LGA : Airport

axiom FliesTo : Person → Airport → Prop
axiom FliesFrom : Person → Airport → Prop

-- Context axioms
axiom T1 : FliesTo Susan LGA
axiom R1 : ∀ p : Person, ∀ a : Airport, ¬(FliesFrom p a ∧ FliesTo p a)
axiom T2 : FliesFrom John LGA

-- We prove the negation of the statement: John flies to LGA airport.
theorem john_not_to_LGA : ¬FliesTo John LGA := by
  intro hTo
  have hNo : ¬(FliesFrom John LGA ∧ FliesTo John LGA) := R1 John LGA
  apply hNo
  exact And.intro T2 hTo
</lean>

ANSWER: False
