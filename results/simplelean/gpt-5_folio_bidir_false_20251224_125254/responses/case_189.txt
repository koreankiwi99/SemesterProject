axiom some_pro_not_american : ∃ x : obj, ProBasketballPlayer x ∧ ¬American x

-- All professional basketball players can do jump shots.
axiom all_pro_can_jump : ∀ x : obj, ProBasketballPlayer x → CanJumpShots x
-- If someone can jump shots, they leap straight into the air.
axiom jump_implies_leap : ∀ x : obj, CanJumpShots x → LeapsStraightUp x
-- If someone leaps straight into the air, they activate their leg muscles.
axiom leap_implies_activate : ∀ x : obj, LeapsStraightUp x → ActivatesLegMuscles x

-- Yuri does not activate his leg muscles.
axiom yuri_not_activate : ¬ActivatesLegMuscles Yuri

-- From the chain of implications, any pro player activates leg muscles.
lemma pro_implies_activate : ∀ x : obj, ProBasketballPlayer x → ActivatesLegMuscles x :=
  fun x hx => leap_implies_activate x (jump_implies_leap x (all_pro_can_jump x hx))

-- Hence, Yuri cannot be a professional basketball player.
theorem yuri_not_pro : ¬ProBasketballPlayer Yuri :=
  by
    intro hPro
    have hAct : ActivatesLegMuscles Yuri := pro_implies_activate Yuri hPro
    exact yuri_not_activate hAct

-- Therefore, Yuri is not an American professional basketball player.
theorem yuri_not_american_pro : ¬(American Yuri ∧ ProBasketballPlayer Yuri) :=
  by
    intro h
    exact yuri_not_pro h.right
