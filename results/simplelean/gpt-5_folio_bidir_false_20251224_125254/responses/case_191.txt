<lean>
axiom obj : Type
axiom Yuri : obj

axiom Professional : obj → Prop
axiom American : obj → Prop
axiom JumpShot : obj → Prop
axiom LeapStraight : obj → Prop
axiom ActivateLeg : obj → Prop

-- Some professional basketball players are not American nationals.
axiom A1 : ∃ x : obj, Professional x ∧ ¬American x
-- All professional basketball players can do jump shots.
axiom A2 : ∀ x : obj, Professional x → JumpShot x
-- If someone can jump shots, they leap straight into the air.
axiom A3 : ∀ x : obj, JumpShot x → LeapStraight x
-- If someone leaps straight into the air, they activate their leg muscles.
axiom A4 : ∀ x : obj, LeapStraight x → ActivateLeg x
-- Yuri does not activate his leg muscles.
axiom A5 : ¬ActivateLeg Yuri

-- Prove the negation of: If Yuri is not an American professional basketball player, then Yuri is a professional basketball player.
theorem negate_statement :
  ¬( (¬(American Yuri ∧ Professional Yuri)) → Professional Yuri ) := by
  -- Chain: Professional → JumpShot → LeapStraight → ActivateLeg
  have hchain : ∀ x : obj, Professional x → ActivateLeg x := by
    intro x
    intro hpro
    have hjump : JumpShot x := A2 x hpro
    have hleap : LeapStraight x := A3 x hjump
    have hact : ActivateLeg x := A4 x hleap
    exact hact
  -- From ¬ActivateLeg Yuri and the chain, get ¬Professional Yuri
  have npro : ¬Professional Yuri := by
    intro hpro
    have hact : ActivateLeg Yuri := hchain Yuri hpro
    exact A5 hact
  -- Hence, ¬(American Yuri ∧ Professional Yuri)
  have ante : ¬(American Yuri ∧ Professional Yuri) := by
    intro h
    exact npro h.right
  -- Use A ∧ ¬B to refute (A → B)
  intro himp
  have hypro : Professional Yuri := himp ante
  exact npro hypro
</lean>

ANSWER: False
