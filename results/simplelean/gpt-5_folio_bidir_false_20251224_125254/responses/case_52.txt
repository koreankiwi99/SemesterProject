axiom obj : Type
axiom French : obj
axiom Spanish : obj
axiom German : obj
axiom Basque : obj

axiom Romance : obj → Prop
axiom IndoEuropean : obj → Prop
axiom Related : obj → obj → Prop

-- All Romance languages are Indo-European.
axiom A1 : ∀ x : obj, Romance x → IndoEuropean x

-- Romance languages form a family: any two Romance languages are related.
axiom A2 : ∀ x y : obj, Romance x → Romance y → Related x y

-- French and Spanish are Romance languages.
axiom A3 : Romance French
axiom A4 : Romance Spanish

-- German is related to Spanish.
axiom A5 : Related German Spanish

-- Basque is not related to any language.
axiom A6 : ∀ y : obj, ¬Related Basque y

-- A derivable fact: Basque is not a Romance language.
theorem not_romance_basque : ¬Romance Basque := by
  intro h
  have rel : Related Basque Spanish := A2 Basque Spanish h A4
  have nb : ¬Related Basque Spanish := A6 Spanish
  exact nb rel
