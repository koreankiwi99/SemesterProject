def SameState (c1 c2 : City) : Prop := ∃ s : State, InState c1 s ∧ InState c2 s

-- Facts from the context
axiom T_Billings_Montana : InState Billings Montana
axiom T_Butte_Montana : InState Butte Montana

axiom Ex_Texarkana : Exception Texarkana
axiom Ex_Texhoma : Exception Texhoma
axiom Ex_UnionCity : Exception UnionCity
axiom NotEx_Butte : ¬Exception Butte

axiom unique_state : ∀ c s1 s2, ¬Exception c → InState c s1 → InState c s2 → s1 = s2

-- Goal: Butte and St Pierre are not in the same state
theorem not_same_butte_stp : ¬SameState Butte StPierre := by
  intro h
  cases h with
  | intro s hs =>
    cases hs with
    | intro hB hP =>
      have h_eq : s = Montana := unique_state Butte s Montana NotEx_Butte hB T_Butte_Montana
      have hPmont : InState StPierre Montana := by simpa [h_eq] using hP
      exact T_StPierre_not_Montana hPmont
