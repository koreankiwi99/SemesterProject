<lean>
axiom obj : Type

axiom Pet : obj → Prop
axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop

axiom Ted : obj

-- Premises
axiom P1 : ∀ x : obj, BrownSwissCattle x → Cow x
axiom P2 : ∃ x : obj, Pet x ∧ BrownSwissCattle x
axiom P3 : ∀ x : obj, Cow x → Domesticated x
axiom P4 : ∀ x : obj, Alligator x → ¬Domesticated x
axiom P5 : Alligator Ted

theorem BrownSwissCattle_dom : ∀ x : obj, BrownSwissCattle x → Domesticated x := by
  intro x h
  have hxCow : Cow x := P1 x h
  exact P3 x hxCow

theorem Ted_not_BrownSwissCattle : ¬BrownSwissCattle Ted := by
  intro hB
  have hDom : Domesticated Ted := BrownSwissCattle_dom Ted hB
  have hNotDom : ¬Domesticated Ted := P4 Ted P5
  exact hNotDom hDom

theorem not_Ted_pet_and_BrownSwissCattle : ¬(Pet Ted ∧ BrownSwissCattle Ted) := by
  intro h
  exact Ted_not_BrownSwissCattle h.right
</lean>

ANSWER: False
