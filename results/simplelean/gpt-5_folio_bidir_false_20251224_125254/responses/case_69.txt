<lean>
axiom Person : Type
axiom Show : Type
axiom Channel : Type

axiom BBCTwo : Channel
axiom BBCThree : Channel
axiom Badults : Show

axiom MemberOfPappys : Person → Prop
axiom StarredIn : Person → Show → Prop
axiom PilotedOn : Show → Channel → Prop

-- Context:
-- Badults is a series starring members of Pappy's.
axiom A1 : ∃ p : Person, MemberOfPappys p ∧ StarredIn p Badults
-- Badults was piloted on BBC Three.
axiom A2 : PilotedOn Badults BBCThree

-- Negation of the statement "No members of Pappy's have starred in a show piloting on BBC Two or BBC Three":
-- There exists a member of Pappy's who starred in a show that piloted on BBC Two or BBC Three.
theorem neg_statement :
  ∃ p : Person, ∃ s : Show, MemberOfPappys p ∧ StarredIn p s ∧ (PilotedOn s BBCTwo ∨ PilotedOn s BBCThree) :=
by
  rcases A1 with ⟨p, hpMem, hpStar⟩
  refine ⟨p, Badults, ?_⟩
  exact And.intro hpMem (And.intro hpStar (Or.inr A2))
</lean>

ANSWER: False
