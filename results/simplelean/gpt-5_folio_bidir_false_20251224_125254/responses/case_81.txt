axiom Spicy : obj → Prop

axiom MalaHotpot : obj → Prop
axiom BBMProduct : obj → Prop

-- No baked sweets are spicy.
axiom A1 : ∀ x : obj, BakedSweet x → ¬Spicy x
-- All cupcakes are baked sweets.
axiom A2 : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom A3 : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom A4 : ∀ x : obj, BBMProduct x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom A5 : Spicy DTC ∨ MalaHotpot DTC ∨ ¬BakedSweet DTC

-- From "No baked sweets are spicy", we get "Spicy → ¬BakedSweet".
theorem spicy_implies_not_baked : ∀ x : obj, Spicy x → ¬BakedSweet x := by
  intro x
  intro hs
  intro hb
  have hns : ¬Spicy x := A1 x hb
  exact hns hs

-- Dried Thai chilies are not baked sweets (by case analysis on the disjunction).
theorem dtc_not_baked : ¬BakedSweet DTC := by
  have h := A5
  cases h with
  | inl hs =>
    exact (spicy_implies_not_baked DTC) hs
  | inr h2 =>
    cases h2 with
    | inl hm =>
      have hs : Spicy DTC := A3 DTC hm
      exact (spicy_implies_not_baked DTC) hs
    | inr hnb => exact hnb

-- Hence they are not products of Baked by Melissa (since those are cupcakes, hence baked sweets).
theorem dtc_not_bbm : ¬BBMProduct DTC := by
  intro hbbm
  have cup : Cupcake DTC := A4 DTC hbbm
  have bs : BakedSweet DTC := A2 DTC cup
  exact dtc_not_baked bs
