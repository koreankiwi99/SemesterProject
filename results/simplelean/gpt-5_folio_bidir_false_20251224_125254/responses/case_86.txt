<lean>
axiom obj : Type
axiom movie : Type

axiom SirDigby : obj
axiom Nemesis : obj
axiom M : movie

axiom SuperheroMovie : movie → Prop
axiom NamedAfter : movie → obj → Prop
axiom InMovie : obj → movie → Prop
axiom GoodGuy : obj → Prop
axiom BadGuy : obj → Prop
axiom Fights : obj → obj → Prop
axiom WinsFight : obj → obj → Prop

-- In superhero movies, the good guys always win (their fights)
axiom AlwaysWin : ∀ m : movie, SuperheroMovie m → ∀ p : obj, InMovie p m → GoodGuy p → ∀ q : obj, Fights p q → WinsFight p q

-- Good guys fight bad guys and vice versa
axiom GG_fight_implies_BG : ∀ x y : obj, GoodGuy x → Fights x y → BadGuy y
axiom BG_fight_implies_GG : ∀ x y : obj, BadGuy x → Fights x y → GoodGuy y

-- Sir Digby fights his nemesis
axiom SD_fights_nemesis : Fights SirDigby Nemesis

-- If a superhero movie is named after a character, that character is a good guy
axiom NamedAfter_implies_GG : ∀ m : movie, ∀ p : obj, SuperheroMovie m → NamedAfter m p → GoodGuy p

-- The movie is a superhero movie and is named after Sir Digby
axiom M_is_superhero : SuperheroMovie M
axiom M_named_after_SD : NamedAfter M SirDigby

-- If somebody wins a fight, the person they are fighting does not win
axiom win_antisym : ∀ x y : obj, WinsFight x y → ¬WinsFight y x

-- If a superhero movie is named after a character, that character is in the movie
axiom NamedAfter_implies_inMovie : ∀ m : movie, ∀ p : obj, SuperheroMovie m → NamedAfter m p → InMovie p m

-- Derived facts
theorem SD_is_good : GoodGuy SirDigby :=
  by
    exact NamedAfter_implies_GG M SirDigby M_is_superhero M_named_after_SD

theorem SD_in_movie : InMovie SirDigby M :=
  by
    exact NamedAfter_implies_inMovie M SirDigby M_is_superhero M_named_after_SD

theorem SD_wins_over_nemesis : WinsFight SirDigby Nemesis :=
  by
    exact AlwaysWin M M_is_superhero SirDigby SD_in_movie SD_is_good Nemesis SD_fights_nemesis

theorem nemesis_does_not_win : ¬WinsFight Nemesis SirDigby :=
  by
    exact win_antisym SirDigby Nemesis SD_wins_over_nemesis
