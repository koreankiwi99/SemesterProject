<lean>
axiom obj : Type
axiom Luke : obj

axiom Born : obj → Prop
axiom SpendTimeWithSiblings : obj → Prop
axiom LiveAtHome : obj → Prop
axiom LiveWithStrangers : obj → Prop
axiom HaveSiblingsBornTogether : obj → Prop
axiom ComplainOften : obj → Prop

-- People who are born in a multiple birth spend time with their siblings
axiom A1 : ∀ x : obj, Born x → SpendTimeWithSiblings x
-- If people have siblings who were born together, then they were born in a multiple birth
axiom A2 : ∀ x : obj, HaveSiblingsBornTogether x → Born x
-- If people complain often about annoying siblings, then they have siblings who were born together
axiom A3 : ∀ x : obj, ComplainOften x → HaveSiblingsBornTogether x
-- If people live at home, then they do not live with strangers
axiom A4 : ∀ x : obj, LiveAtHome x → ¬ LiveWithStrangers x
-- If people spend a lot of time with their siblings, then they often live at home
axiom A5 : ∀ x : obj, SpendTimeWithSiblings x → LiveAtHome x
-- Luke either (born in multiple birth ∧ live with strangers) or (not born in multiple birth ∧ not live with strangers)
axiom A6 : (Born Luke ∧ LiveWithStrangers Luke) ∨ (¬ Born Luke ∧ ¬ LiveWithStrangers Luke)

theorem luke_neither_born_nor_complains : ¬ Born Luke ∧ ¬ ComplainOften Luke := by
  -- First, rule out the first disjunct of A6 using the chain Born -> Spend -> Home -> ¬Strangers
  have h_second : (¬ Born Luke ∧ ¬ LiveWithStrangers Luke) := by
    cases A6 with
    | inl h1 =>
        have hb : Born Luke := h1.left
        have hs : SpendTimeWithSiblings Luke := A1 Luke hb
        have hh : LiveAtHome Luke := A5 Luke hs
        have hnlws : ¬ LiveWithStrangers Luke := A4 Luke hh
        have contra : False := hnlws h1.right
        exact False.elim contra
    | inr h2 => exact h2
  have h_not_born : ¬ Born Luke := h_second.left
  have h_not_complain : ¬ ComplainOften Luke := by
    intro hc
    have hsibs : HaveSiblingsBornTogether Luke := A3 Luke hc
    have hborn : Born Luke := A2 Luke hsibs
    exact h_not_born hborn
  exact And.intro h_not_born h_not_complain
</lean>

ANSWER: True
