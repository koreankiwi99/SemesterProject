<lean>
axiom obj : Type
axiom Tom : obj

axiom GREApplicant : obj → Prop
axiom ProvesHardship : obj → Prop
axiom ProvidesAid : obj → Prop
axiom SingleParentFamily : obj → Prop
axiom FewResources : obj → Prop
axiom OutOfWorkDad : obj → Prop

-- ETS provides financial aid to those GRE applicants who prove economic hardship.
axiom R1 : ∀ x : obj, GREApplicant x → ProvesHardship x → ProvidesAid x

-- Those living in single-parent families or having few resources can prove economic hardship.
axiom R2 : ∀ x : obj, (SingleParentFamily x ∨ FewResources x) → ProvesHardship x

-- Facts about Tom
axiom T1 : SingleParentFamily Tom
axiom T2 : FewResources Tom
axiom T3 : GREApplicant Tom
axiom T4 : OutOfWorkDad Tom

theorem ets_provides_financial_aid_to_tom : ProvidesAid Tom :=
by
  have hph : ProvesHardship Tom := R2 Tom (Or.inl T1)
  exact R1 Tom T3 hph
</lean>

ANSWER: True
