<lean>
axiom obj : Type
axiom Picuris : obj
axiom Juan : obj
axiom HPeg : obj

axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

-- Picuris Mountains are a mountain range in New Mexico or Texas
axiom A1 : MountainRange Picuris
axiom A2 : InNewMexico Picuris ∨ InTexas Picuris

-- Juan de Onate visited the Picuris Mountains
axiom A3 : Visited Juan Picuris

-- The Harding Pegmatite Mine, located in the Picuris Mountains, was donated
axiom A4 : Mine HPeg ∧ (LocatedIn HPeg Picuris ∧ Donated HPeg)

-- There are no mountain ranges in Texas that have mines that have been donated
axiom A5 : ∀ r : obj, MountainRange r ∧ InTexas r → ¬ ∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m

-- From the above, we can show the Picuris Mountains are not in Texas
theorem picuris_not_in_texas : ¬ InTexas Picuris := by
  intro hTX
  have hnone : ¬ ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m :=
    A5 Picuris (And.intro A1 hTX)
  have hexists : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m := by
    refine Exists.intro HPeg ?_
    have hm : Mine HPeg := And.left A4
    have hloc_don : LocatedIn HPeg Picuris ∧ Donated HPeg := And.right A4
    have hloc : LocatedIn HPeg Picuris := And.left hloc_don
    have hdon : Donated HPeg := And.right hloc_don
    exact And.intro hm (And.intro hloc hdon)
  exact hnone hexists
