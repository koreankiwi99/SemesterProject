-- From the context, Picuris cannot be in Texas
theorem not_texas_picuris : ¬InTexas Picuris := by
  intro htx
  have hno : ¬∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m := A6 Picuris A1 htx
  have hex : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m := by
    refine Exists.intro Harding ?_
    exact And.intro A3 (And.intro A4 A5)
  exact hno hex

-- Therefore Picuris is in New Mexico

  | inr htx => exact False.elim (hnotx htx)
