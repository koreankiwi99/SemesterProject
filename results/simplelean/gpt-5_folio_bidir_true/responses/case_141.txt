<lean>
axiom obj : Type
axiom Hachi : obj

axiom Film : obj → Prop
axiom AppropriateForAllAges : obj → Prop
axiom ChildrenCanWatchNoGuidance : obj → Prop
axiom ExcessiveEroticViolentContent : obj → Prop
axiom RatedGA : obj → Prop
axiom FamilyFriendlyAnimated : obj → Prop
axiom FrozenSeries : obj → Prop

-- If a film is appropriate for people of all ages, then children can watch it without guidance.
axiom R1 : ∀ x : obj, AppropriateForAllAges x → ChildrenCanWatchNoGuidance x
-- If a film contains excessive erotic and violent content, children cannot watch it without guidance.
axiom R2 : ∀ x : obj, ExcessiveEroticViolentContent x → ¬ ChildrenCanWatchNoGuidance x
-- If a film is rated General Audience, then it is appropriate for people of all ages.
axiom R3 : ∀ x : obj, RatedGA x → AppropriateForAllAges x
-- All family-friendly animated films are rated General Audience.
axiom R4 : ∀ x : obj, FamilyFriendlyAnimated x → RatedGA x
-- All movies in the Frozen series are family-friendly animated films.
axiom R5 : ∀ x : obj, FrozenSeries x → FamilyFriendlyAnimated x

-- “Hachi: A dog’s Tale” is a film.
axiom T1 : Film Hachi
-- “Hachi: A dog’s Tale” is either a family-friendly animated film or is appropriate for people of all ages.
axiom T2 : FamilyFriendlyAnimated Hachi ∨ AppropriateForAllAges Hachi

-- Derive that Hachi is appropriate for all ages
theorem Hachi_appropriate : AppropriateForAllAges Hachi :=
by
  cases T2 with
  | inl hFF =>
    have hGA : RatedGA Hachi := R4 Hachi hFF
    exact R3 Hachi hGA
  | inr hA =>
    exact hA

-- Thus children can watch Hachi without guidance
theorem Hachi_watchable : ChildrenCanWatchNoGuidance Hachi :=
by
  exact R1 Hachi Hachi_appropriate

-- Hence Hachi does not contain excessive erotic and violent content
theorem Hachi_not_evc : ¬ ExcessiveEroticViolentContent Hachi :=
by
  intro hE
  have hNotCW : ¬ ChildrenCanWatchNoGuidance Hachi := R2 Hachi hE
  exact hNotCW Hachi_watchable
