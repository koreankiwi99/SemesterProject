<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom UniPartTime : obj → Prop
axiom TakingDatabase : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- None of the students who currently work in the library are from CS
axiom A1 : ∀ x : obj, Student x ∧ WorksInLibrary x → ¬ FromCS x
-- All of the students with part-time jobs offered by the university currently work in the library
axiom A2 : ∀ x : obj, Student x ∧ UniPartTime x → WorksInLibrary x
-- All of the students who are taking the database course are from CS
axiom A3 : ∀ x : obj, Student x ∧ TakingDatabase x → FromCS x
-- All of the students taking a class with Professor David are taking the database course
axiom A4 : ∀ x : obj, Student x ∧ TakingWithDavid x → TakingDatabase x
-- All students working in the lab are taking a class with Professor David
axiom A5 : ∀ x : obj, Student x ∧ WorksInLab x → TakingWithDavid x

-- James is a student working in the lab
axiom T1 : Student James
axiom T2 : WorksInLab James

-- James doesn't work in the lab or have a part-time job offered by the university
axiom T3 : ¬ WorksInLab James ∨ ¬ UniPartTime James

-- Statement to evaluate: James does not have a part-time job offered by the university.
theorem james_no_uniparttime : ¬ UniPartTime James := by
  cases T3 with
  | inl hNotLab =>
      have hFalse : False := hNotLab T2
      intro _
      exact hFalse.elim
  | inr hNotPT =>
      exact hNotPT
</lean>

ANSWER: True
