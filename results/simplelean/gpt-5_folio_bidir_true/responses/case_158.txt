<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom UniPTJob : obj → Prop
axiom TakingDB : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorkingLab : obj → Prop

-- None of the students who currently work in the library are from CS
axiom A1 : ∀ x : obj, Student x ∧ WorksLibrary x → ¬ FromCS x
-- All students with university part-time jobs work in the library
axiom A2 : ∀ x : obj, Student x ∧ UniPTJob x → WorksLibrary x
-- All students taking the database course are from CS
axiom A3 : ∀ x : obj, Student x ∧ TakingDB x → FromCS x
-- All students taking a class with Professor David are taking the database course
axiom A4 : ∀ x : obj, Student x ∧ TakingWithDavid x → TakingDB x
-- All students working in the lab are taking a class with Professor David
axiom A5 : ∀ x : obj, Student x ∧ WorkingLab x → TakingWithDavid x
-- James is a student working in the lab
axiom A6 : Student James ∧ WorkingLab James
-- James doesn't work in the lab and doesn't have a university part-time job
axiom A7 : ¬ WorkingLab James ∧ ¬ UniPTJob James

theorem james_db_or_pt : TakingDB James ∨ UniPTJob James := by
  have hSW : Student James ∧ WorkingLab James := A6
  have hS : Student James := And.left hSW
  have hLab : WorkingLab James := And.right hSW
  have hDavid : TakingWithDavid James := A5 James (And.intro hS hLab)
  have hDB : TakingDB James := A4 James (And.intro hS hDavid)
  exact Or.inl hDB
</lean>

ANSWER: True
