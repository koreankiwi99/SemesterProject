axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swim : obj → Prop

-- Premises
axiom H1 : ∀ x : obj, Hawk x → ¬ Swim x          -- A hawk never swims.
axiom H2 : ∃ x : obj, Bird x ∧ Hawk x             -- Some birds are hawks.

-- The statement to evaluate: ∀ x, Bird x → Swim x
-- This cannot be derived; in fact, it is refuted by the premises:
theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swim x) := by
  intro hAll
  cases H2 with
  | intro w hw =>
    have hBw : Bird w := And.left hw
    have hHw : Hawk w := And.right hw
    have hSw : Swim w := hAll w hBw
    have hNotSw : ¬ Swim w := H1 w hHw
    exact hNotSw hSw
