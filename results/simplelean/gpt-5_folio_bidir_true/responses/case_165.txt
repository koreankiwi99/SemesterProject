<lean>
axiom obj : Type

axiom BreakingBad : obj

axiom Show : obj → Prop
axiom Monday8 : obj → Prop
axiom GivesRoses : obj → Prop
axiom Bachelor : obj → Prop
axiom PortraysReal : obj → Prop
axiom RealityTV : obj → Prop

-- Some show airing at 8 pm on Monday gives out roses on TV.
axiom E1 : ∃ s : obj, Monday8 s ∧ GivesRoses s

-- If a show gives out roses on TV, then the show is an episode of The Bachelor.
axiom R1 : ∀ s : obj, GivesRoses s → Bachelor s

-- The Bachelor portrays the lives of real people (episodes inherit this property).
axiom R2 : ∀ s : obj, Bachelor s → PortraysReal s

-- All shows portraying the lives of real people are reality TV shows.
axiom R3 : ∀ s : obj, PortraysReal s → RealityTV s

-- Breaking Bad is a show.
axiom F1 : Show BreakingBad

-- Breaking Bad is not a reality TV show.
axiom F2 : ¬ RealityTV BreakingBad

-- Statement: If roses are given out during Breaking Bad, then it is on Monday at 8 pm.
theorem bb_roses_implies_mon8 : GivesRoses BreakingBad → Monday8 BreakingBad := by
  intro hR
  have hBach : Bachelor BreakingBad := R1 BreakingBad hR
  have hReal : PortraysReal BreakingBad := R2 BreakingBad hBach
  have hRTV : RealityTV BreakingBad := R3 BreakingBad hReal
  have : False := F2 hRTV
  exact False.elim this
</lean>

ANSWER: True
