<lean>
axiom obj : Type
axiom HeinrichSchmidt : obj
axiom GermanPolitician : obj → Prop
axiom MemberPrussianParliament : obj → Prop
axiom MemberNaziReichstag : obj → Prop

axiom H1 : GermanPolitician HeinrichSchmidt
axiom H2 : MemberPrussianParliament HeinrichSchmidt
axiom H3 : MemberNaziReichstag HeinrichSchmidt

theorem exists_german_politician_in_both :
  ∃ x : obj, GermanPolitician x ∧ MemberPrussianParliament x ∧ MemberNaziReichstag x := by
  refine Exists.intro HeinrichSchmidt ?_
  exact And.intro H1 (And.intro H2 H3)
</lean>

ANSWER: True
