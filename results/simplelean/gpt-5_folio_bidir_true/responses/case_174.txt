<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatsSalads : obj → Prop
axiom Conscious : obj → Prop
axiom GrewHC : obj → Prop
axiom FulfillsIntake : obj → Prop
axiom DisregardWellbeing : obj → Prop
axiom VisitsGymDaily : obj → Prop

axiom A1 : ∀ x : obj, EatsSalads x → Conscious x
axiom A2 : ∀ x : obj, GrewHC x → EatsSalads x
axiom A3 : ∀ x : obj, FulfillsIntake x → GrewHC x
axiom A4 : ∀ x : obj, DisregardWellbeing x → ¬ Conscious x
axiom A5 : ∀ x : obj, VisitsGymDaily x → FulfillsIntake x
axiom A6 : (GrewHC Taylor ∧ DisregardWellbeing Taylor) ∨ (¬ GrewHC Taylor ∧ ¬ DisregardWellbeing Taylor)

theorem taylor_neither_grew_nor_visits : ¬ GrewHC Taylor ∧ ¬ VisitsGymDaily Taylor := by
  have hNotGD : ¬ (GrewHC Taylor ∧ DisregardWellbeing Taylor) := by
    intro h
    have hG : GrewHC Taylor := h.left
    have hD : DisregardWellbeing Taylor := h.right
    have hC : Conscious Taylor := A1 Taylor (A2 Taylor hG)
    have hNotC : ¬ Conscious Taylor := A4 Taylor hD
    exact hNotC hC
  have hNotG_and_NotD : (¬ GrewHC Taylor ∧ ¬ DisregardWellbeing Taylor) := by
    cases A6 with
    | inl hGD => exact False.elim (hNotGD hGD)
    | inr h => exact h
  have hNotG : ¬ GrewHC Taylor := hNotG_and_NotD.left
  have hVtoG : VisitsGymDaily Taylor → GrewHC Taylor := by
    intro hV
    have hF : FulfillsIntake Taylor := A5 Taylor hV
    exact A3 Taylor hF
  have hNotV : ¬ VisitsGymDaily Taylor := by
    intro hV
    exact hNotG (hVtoG hV)
  exact And.intro hNotG hNotV
</lean>

ANSWER: True
