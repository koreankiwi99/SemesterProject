<lean>
axiom obj : Type
axiom Tom : obj

axiom NiceToAnimals : obj → Prop
axiom MeanToAnimals : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop

axiom A1 : ∀ x : obj, NiceToAnimals x → ¬ MeanToAnimals x
axiom A2 : ∃ x : obj, Grumpy x ∧ MeanToAnimals x
axiom A3 : ∀ x : obj, AnimalLover x → NiceToAnimals x
axiom A4 : ∀ x : obj, PetOwner x → AnimalLover x
axiom A5 : PetOwner Tom

theorem tom_not_grumpy_and_mean : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) := by
  have hNiceTom : NiceToAnimals Tom := by
    have hALTom : AnimalLover Tom := A4 Tom A5
    exact A3 Tom hALTom
  have hNotMeanTom : ¬ MeanToAnimals Tom := A1 Tom hNiceTom
  intro h
  exact hNotMeanTom (And.right h)
</lean>

ANSWER: True
