<lean>
axiom obj : Type
axiom Djokovic : obj

axiom At : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProfessionalTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom Famous : obj → Prop
axiom WellPaid : obj → Prop
axiom LivesInTaxHaven : obj → Prop

axiom A1 : ∀ x : obj, At x → (GrandSlamChampion x ∨ OscarNominatedActor x)
axiom A2 : ∀ x : obj, At x → GrandSlamChampion x → ProfessionalTennisPlayer x
axiom A3 : ∀ x : obj, At x → OscarNominatedActor x → Celebrity x
axiom A4 : ∀ x : obj, At x → ProfessionalTennisPlayer x → Athlete x
axiom A5 : ∀ x : obj, At x → Celebrity x → WellPaid x
axiom A6 : ∀ x : obj, At x → Athlete x → Famous x
axiom A7 : ∀ x : obj, At x → WellPaid x → LivesInTaxHaven x

axiom A8 : At Djokovic
axiom A9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

theorem djokovic_lives_in_tax_haven : LivesInTaxHaven Djokovic := by
  have djAt : At Djokovic := A8
  have h : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := A1 Djokovic djAt
  cases h with
  | inl djGSC =>
      have djPTP : ProfessionalTennisPlayer Djokovic := A2 Djokovic djAt djGSC
      have djAth : Athlete Djokovic := A4 Djokovic djAt djPTP
      have djFam : Famous Djokovic := A6 Djokovic djAt djAth
      have djWP : WellPaid Djokovic := A9 (And.intro djFam djAth)
      exact A7 Djokovic djAt djWP
  | inr djONA =>
      have djCel : Celebrity Djokovic := A3 Djokovic djAt djONA
      have djWP : WellPaid Djokovic := A5 Djokovic djAt djCel
      exact A7 Djokovic djAt djWP
</lean>

ANSWER: True
