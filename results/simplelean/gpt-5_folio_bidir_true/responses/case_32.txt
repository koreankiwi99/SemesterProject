<lean>
axiom person : Type
axiom Djokovic : person

axiom At : person → Prop
axiom GrandSlamChampion : person → Prop
axiom OscarNominatedActor : person → Prop
axiom ProfessionalTennisPlayer : person → Prop
axiom Celebrity : person → Prop
axiom Athlete : person → Prop
axiom WellPaid : person → Prop
axiom Famous : person → Prop
axiom LivesTaxHaven : person → Prop

-- Premises
axiom A1 : ∀ x : person, At x → (GrandSlamChampion x ∨ OscarNominatedActor x)
axiom A2 : ∀ x : person, At x ∧ GrandSlamChampion x → ProfessionalTennisPlayer x
axiom A3 : ∀ x : person, At x ∧ OscarNominatedActor x → Celebrity x
axiom A4 : ∀ x : person, At x ∧ ProfessionalTennisPlayer x → Athlete x
axiom A5 : ∀ x : person, At x ∧ Celebrity x → WellPaid x
axiom A6 : ∀ x : person, At x ∧ Athlete x → Famous x
axiom A7 : ∀ x : person, At x ∧ WellPaid x → LivesTaxHaven x

axiom A8 : At Djokovic
axiom A9 : Famous Djokovic ∧ Athlete Djokovic → WellPaid Djokovic

-- From the premises we can derive that Djokovic lives in a tax haven.
theorem djokovic_lives_in_tax_haven : LivesTaxHaven Djokovic :=
by
  have hdisj : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := A1 Djokovic A8
  cases hdisj with
  | inl hG =>
    have hPTP : ProfessionalTennisPlayer Djokovic := A2 Djokovic ⟨A8, hG⟩
    have hAth : Athlete Djokovic := A4 Djokovic ⟨A8, hPTP⟩
    have hFam : Famous Djokovic := A6 Djokovic ⟨A8, hAth⟩
    have hWP : WellPaid Djokovic := A9 ⟨hFam, hAth⟩
    exact A7 Djokovic ⟨A8, hWP⟩
  | inr hO =>
    have hCel : Celebrity Djokovic := A3 Djokovic ⟨A8, hO⟩
    have hWP : WellPaid Djokovic := A5 Djokovic ⟨A8, hCel⟩
    exact A7 Djokovic ⟨A8, hWP⟩
