<lean>
axiom obj : Type

axiom AtSchool : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDiningHall : obj → Prop
axiom NotPicky : obj → Prop
axiom SpendTimeDiningHalls : obj → Prop

axiom Hannah : obj

-- If people at Mary's school work in student jobs on campus, then they need to earn money to help pay for their college tuition.
axiom R1 : ∀ x : obj, AtSchool x → (Work x → Need x)
-- If people at Mary's school order takeout frequently in college, then they work in student jobs on campus.
axiom R2 : ∀ x : obj, AtSchool x → (OrderTakeout x → Work x)
-- People at Mary's school order takeout frequently in college or enjoy the dining hall meals and recipes.
axiom R3 : ∀ x : obj, AtSchool x → (OrderTakeout x ∨ EnjoyDiningHall x)
-- If people at Mary's school enjoy the dining hall meals and recipes, then they are not picky eaters.
axiom R4 : ∀ x : obj, AtSchool x → (EnjoyDiningHall x → NotPicky x)
-- If people at Mary's school enjoy the dining hall meals and recipes, then they spend a lot of their time eating and catching up with friends in the campus dining halls.
axiom R5 : ∀ x : obj, AtSchool x → (EnjoyDiningHall x → SpendTimeDiningHalls x)
-- Hannah is at Mary's school.
axiom T1 : AtSchool Hannah
-- Hannah works in student jobs on campus
axiom T2a : Work Hannah
-- and if she needs to earn money ..., then she is neither picky nor needs to earn money ...
axiom T2b : Need Hannah → (NotPicky Hannah ∧ ¬ Need Hannah)

theorem hannah_needs_money : Need Hannah :=
by
  have hs : AtSchool Hannah := T1
  have hw : Work Hannah := T2a
  have himp : Work Hannah → Need Hannah := R1 Hannah hs
  exact himp hw
</lean>

ANSWER: True
