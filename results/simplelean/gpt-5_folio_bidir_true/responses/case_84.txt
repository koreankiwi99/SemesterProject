<lean>
axiom obj : Type
axiom Hamden : obj

axiom ListedInYelp : obj → Prop
axiom ManyNegativeReviews : obj → Prop
axiom RatingGT4 : obj → Prop
axiom ProvidesTakeOut : obj → Prop
axiom PopularLocal : obj → Prop

-- If listed in Yelp’s recommendations, then not many negative reviews.
axiom A1 : ∀ x : obj, ListedInYelp x → ¬ ManyNegativeReviews x
-- Rating > 4 ⇒ listed in Yelp’s recommendations.
axiom A2 : ∀ x : obj, RatingGT4 x → ListedInYelp x
-- Some restaurants do not provide take-out and receive many negative reviews.
axiom A3 : ∃ x : obj, ¬ ProvidesTakeOut x ∧ ManyNegativeReviews x
-- Popular among locals ⇒ rating > 4.
axiom A4 : ∀ x : obj, PopularLocal x → RatingGT4 x
-- Hamden has rating > 4 or is popular among locals.
axiom A5 : RatingGT4 Hamden ∨ PopularLocal Hamden

theorem hamden_takeout_or_not_many_neg :
  ProvidesTakeOut Hamden ∨ ¬ ManyNegativeReviews Hamden := by
  -- From A5 and A4, deduce RatingGT4 Hamden.
  have hR : RatingGT4 Hamden := by
    cases A5 with
    | inl h => exact h
    | inr hp => exact A4 Hamden hp
  -- Then Hamden is listed in Yelp’s recommendations.
  have hL : ListedInYelp Hamden := A2 Hamden hR
  -- Therefore, Hamden does not receive many negative reviews.
  have hNot : ¬ ManyNegativeReviews Hamden := A1 Hamden hL
  -- Conclude the desired disjunction.
  exact Or.inr hNot
</lean>

ANSWER: True
