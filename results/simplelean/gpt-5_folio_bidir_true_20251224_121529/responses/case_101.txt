<lean>
axiom obj : Type
axiom Luke : obj

axiom MultipleBirth : obj → Prop
axiom PlayWithSiblings : obj → Prop
axiom HaveSiblingsBornTogether : obj → Prop
axiom ComplainOften : obj → Prop
axiom LiveAtHome : obj → Prop
axiom LiveWithStrangers : obj → Prop

-- 1. People born in a multiple birth spend a lot of time hanging out/playing with their siblings.
axiom A1 : ∀ x : obj, MultipleBirth x → PlayWithSiblings x
-- 2. If people have siblings who were born together, then they were born in a multiple birth.
axiom A2 : ∀ x : obj, HaveSiblingsBornTogether x → MultipleBirth x
-- 3. If people complain often about annoying siblings, then they have siblings who were born together.
axiom A3 : ∀ x : obj, ComplainOften x → HaveSiblingsBornTogether x
-- 4. If people live at home, then they do not live with strangers.
axiom A4 : ∀ x : obj, LiveAtHome x → ¬ LiveWithStrangers x
-- 5. If people spend a lot of time hanging out/playing with siblings, then they often live at home.
axiom A5 : ∀ x : obj, PlayWithSiblings x → LiveAtHome x
-- 6. Luke either is born in a multiple birth and lives with strangers,
--    or is not born in a multiple birth and does not live with strangers.
axiom A6 : (MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨ (¬ MultipleBirth Luke ∧ ¬ LiveWithStrangers Luke)

-- From (1)+(5)+(4), being born in a multiple birth implies not living with strangers.
theorem mb_implies_not_strangers (x : obj) : MultipleBirth x → ¬ LiveWithStrangers x := by
  intro hmb
  have hplay : PlayWithSiblings x := A1 x hmb
  have hhome : LiveAtHome x := A5 x hplay
  exact A4 x hhome

-- Using A6 and the above lemma, the first disjunct is impossible; thus Luke is not MB and does not live with strangers.
theorem luke_not_mb_and_not_strangers : ¬ MultipleBirth Luke ∧ ¬ LiveWithStrangers Luke := by
  cases A6 with
  | inr h => exact h
  | inl h =>
      have nstr : ¬ LiveWithStrangers Luke := mb_implies_not_strangers Luke h.left
      exact False.elim (nstr h.right)
