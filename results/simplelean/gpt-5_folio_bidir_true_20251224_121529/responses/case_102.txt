-- Premises
axiom P1 : ∀ x : obj, Multiple x → SpendTime x
axiom P2 : ∀ x : obj, HaveSiblingsBornTogether x → Multiple x
axiom P3 : ∀ x : obj, ComplainOften x → HaveSiblingsBornTogether x
axiom P4 : ∀ x : obj, LiveHome x → ¬ LiveWithStrangers x
axiom P5 : ∀ x : obj, SpendTime x → LiveHome x

-- Luke's disjunction (interpreting the "baby" mention as extraneous to the logical chain)
axiom Luke_info : (Multiple Luke ∧ LiveWithStrangers Luke) ∨ (¬ Multiple Luke ∧ ¬ LiveWithStrangers Luke)

-- From the premises, being in a multiple birth implies not living with strangers
theorem multiple_not_strangers : ∀ x : obj, Multiple x → ¬ LiveWithStrangers x := by
  intro x
  intro hM
  have hSpend : SpendTime x := P1 x hM
  have hHome : LiveHome x := P5 x hSpend
  exact P4 x hHome

theorem left_impossible : ¬ (Multiple Luke ∧ LiveWithStrangers Luke) := by
  intro h
  cases h with
  | intro hM hS =>
    have hnS : ¬ LiveWithStrangers Luke := multiple_not_strangers Luke hM
    exact hnS hS

theorem right_branch : ¬ Multiple Luke ∧ ¬ LiveWithStrangers Luke := by
  have h := Luke_info
  have hLeftFalse : ¬ (Multiple Luke ∧ LiveWithStrangers Luke) := left_impossible
  cases h with
  | inl hLeft => exact False.elim (hLeftFalse hLeft)
  | inr hRight => exact hRight

-- Consequently, Luke does NOT complain often about annoying siblings

  have hHas : HaveSiblingsBornTogether Luke := P3 Luke hC
  have hMult : Multiple Luke := P2 Luke hHas
  have hNotMult : ¬ Multiple Luke := And.left right_branch
  exact hNotMult hMult
