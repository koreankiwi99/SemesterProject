<lean>
axiom obj : Type
axiom Tom : obj

-- Predicates
axiom GREApplicant : obj → Prop
axiom ProvesHardship : obj → Prop
axiom SingleParentFamily : obj → Prop
axiom FewResources : obj → Prop
axiom FinancialAidByETS : obj → Prop

-- Rules from the context
-- ETS provides financial aid to those GRE applicants who prove economic hardship.
axiom R1 : ∀ x : obj, GREApplicant x → ProvesHardship x → FinancialAidByETS x
-- Those living in single-parent families or having few resources available can prove economic hardship.
axiom R2 : ∀ x : obj, SingleParentFamily x ∨ FewResources x → ProvesHardship x

-- Facts about Tom
axiom T_single : SingleParentFamily Tom
axiom T_few : FewResources Tom
axiom T_applicant : GREApplicant Tom

theorem ets_provides_financial_aid_to_tom : FinancialAidByETS Tom := by
  have hHard : ProvesHardship Tom := R2 Tom (Or.inl T_single)
  exact R1 Tom T_applicant hHard
</lean>

ANSWER: True
