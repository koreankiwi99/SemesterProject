<lean>
axiom Person : Type
axiom Peter : Person

axiom Spill : Person → Prop
axiom Tidy : Person → Prop            -- notably tidy
axiom Cleanly : Person → Prop
axiom ClumsyFoodieGoOut : Person → Prop
axiom ValuesOrderSpotlessness : Person → Prop
axiom FamilyPrioritizesOrderSpotlessness : Person → Prop

-- If people spill a lot of food on their clothing, then they are not notably tidy.
axiom A1 : ∀ x : Person, Spill x → ¬ Tidy x
-- All clumsy foodies who go out frequently to find new food restaurants to try spill a lot.
axiom A2 : ∀ x : Person, ClumsyFoodieGoOut x → Spill x
-- If people are cleanly, then they are notably tidy.
axiom A3 : ∀ x : Person, Cleanly x → Tidy x
-- If people value order and spotlessness, then they are cleanly.
axiom A4 : ∀ x : Person, ValuesOrderSpotlessness x → Cleanly x
-- All people whose families prioritize order and spotlessness themselves value order and spotlessness.
axiom A5 : ∀ x : Person, FamilyPrioritizesOrderSpotlessness x → ValuesOrderSpotlessness x
-- Peter is either (spills and cleanly) or (not spills and not cleanly).
axiom A6 : (Spill Peter ∧ Cleanly Peter) ∨ (¬ Spill Peter ∧ ¬ Cleanly Peter)

theorem peter_statement :
  (ClumsyFoodieGoOut Peter ∧ FamilyPrioritizesOrderSpotlessness Peter) ∨
  (¬ ClumsyFoodieGoOut Peter ∧ ¬ FamilyPrioritizesOrderSpotlessness Peter) := by
  cases A6 with
  | inl h =>
      have nt : ¬ Tidy Peter := A1 Peter h.left
      have t  : Tidy Peter := A3 Peter h.right
      exact False.elim (nt t)
  | inr h =>
      have ncfg : ¬ ClumsyFoodieGoOut Peter := by
        intro hcfg
        have sp : Spill Peter := A2 Peter hcfg
        exact h.left sp
      have nfp : ¬ FamilyPrioritizesOrderSpotlessness Peter := by
        intro hfp
        have vos : ValuesOrderSpotlessness Peter := A5 Peter hfp
        have cln : Cleanly Peter := A4 Peter vos
        exact h.right cln
      exact Or.inr ⟨ncfg, nfp⟩
</lean>

ANSWER: True
