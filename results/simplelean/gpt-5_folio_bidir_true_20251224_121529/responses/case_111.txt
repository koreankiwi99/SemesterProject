<lean>
axiom obj : Type

-- Entities
axiom Picuris : obj
axiom Onate : obj
axiom Harding : obj

-- Predicates
axiom MountainRange : obj → Prop
axiom InNM : obj → Prop
axiom InTX : obj → Prop
axiom Visited : obj → obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Premises
axiom P1 : MountainRange Picuris
axiom P2 : InNM Picuris ∨ InTX Picuris
axiom P3 : Visited Onate Picuris
axiom P4 : Mine Harding
axiom P5 : LocatedIn Harding Picuris
axiom P6 : Donated Harding
axiom TexasNoDonatedMines :
  ∀ r : obj, MountainRange r → InTX r → ¬ (∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m)

-- Conclusion: Juan de Onate visited a mountain range in New Mexico.
theorem onate_visited_nm_range : ∃ r : obj, MountainRange r ∧ InNM r ∧ Visited Onate r := by
  -- Show Picuris is not in Texas because it has a donated mine
  have exMine : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m :=
    Exists.intro Harding (And.intro P4 (And.intro P5 P6))
  have notInTXPicuris : ¬ InTX Picuris := by
    intro hTX
    have hNo := TexasNoDonatedMines Picuris P1 hTX
    exact hNo exMine
  -- From (InNM Picuris ∨ InTX Picuris) and ¬InTX Picuris, deduce InNM Picuris
  have inNMpicuris : InNM Picuris :=
    Or.elim P2 (fun h => h) (fun htx => False.elim (notInTXPicuris htx))
  -- Build the required existential witness
  exact Exists.intro Picuris (And.intro P1 (And.intro inNMpicuris P3))
</lean>

ANSWER: True
