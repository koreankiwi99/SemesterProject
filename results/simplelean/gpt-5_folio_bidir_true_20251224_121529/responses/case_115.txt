<lean>
axiom Person : Type
axiom Michael : Person
axiom Peter : Person
axiom Windy : Person

axiom Man : Person → Prop
axiom Taller : Person → Person → Prop
axiom Block : Person → Person → Prop
axiom Jump : Person → Prop
axiom Great : Person → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R1 : ∀ x y : Person, Man x → Man y → Taller x y → Block x y

-- Transitivity of taller
axiom TransTaller : ∀ x y z : Person, Taller x y → Taller y z → Taller x z

-- Michael and Peter facts
axiom MichaelIsMan : Man Michael
axiom PeterIsMan : Man Peter
axiom PeterTallerMichael : Taller Peter Michael

-- Michael can block any non-jumping shoot
axiom MichaelBlocksNonJumpers : ∀ y : Person, ¬ Jump y → Block Michael y

-- Michael cannot block Windy's shooting
axiom MichaelNotBlockWindy : ¬ Block Michael Windy

-- Every shooter who can jump when shooting is a great shooter (modeled as: if can jump-when-shooting then great)
axiom JumpImpGreat : ∀ y : Person, Jump y → Great y

-- Law of excluded middle for Jump
axiom EMJump : ∀ y : Person, Jump y ∨ ¬ Jump y

theorem windy_great : Great Windy := by
  have hnb : ¬ Block Michael Windy := MichaelNotBlockWindy
  have hem : Jump Windy ∨ ¬ Jump Windy := EMJump Windy
  cases hem with
  | inl hj =>
      exact JumpImpGreat Windy hj
  | inr hnj =>
      have bm : Block Michael Windy := MichaelBlocksNonJumpers Windy hnj
      exact False.elim (hnb bm)
</lean>

ANSWER: True
