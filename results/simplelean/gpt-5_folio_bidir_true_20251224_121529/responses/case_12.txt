<lean>
axiom Game : Type
axiom Company : Type

axiom Zelda : Game

axiom JapaneseCompany : Company → Prop
axiom MadeBy : Game → Company → Prop
axiom Top10 : Game → Prop
axiom MillionPlus : Game → Prop

-- A Japanese game company created The Legend of Zelda
axiom A1 : ∃ c : Company, JapaneseCompany c ∧ MadeBy Zelda c
-- All games on the Top 10 list are made by Japanese game companies
axiom A2 : ∀ g : Game, Top10 g → ∃ c : Company, JapaneseCompany c ∧ MadeBy g c
-- If a game sells more than one million copies, then it is included in the Top 10 list
axiom A3 : ∀ g : Game, MillionPlus g → Top10 g
-- The Legend of Zelda sold more than one million copies
axiom A4 : MillionPlus Zelda

theorem zelda_top10 : Top10 Zelda := by
  exact A3 Zelda A4
</lean>

ANSWER: True
