<lean>
axiom obj : Type
axiom Rock : obj
axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

axiom R1 : ∀ x : obj, Pet x ∧ Mammal x → Animal x
axiom R2 : ∀ x : obj, Monkey x → Mammal x
axiom R3 : ∀ x : obj, Pet x → Monkey x ∨ Bird x
axiom R4 : ∀ x : obj, Pet x ∧ Bird x → CanFly x
axiom R5 : ∀ x : obj, Pet x ∧ Animal x → CanBreathe x
axiom R6 : ∀ x : obj, Pet x ∧ CanFly x → HasWings x

axiom T1 : Pet Rock
axiom T2 : CanFly Rock ∨ Bird Rock ∨ ¬ CanBreathe Rock

theorem rock_has_wings : HasWings Rock :=
by
  have hPet : Pet Rock := T1
  have h := T2
  cases h with
  | inl hfly =>
      exact R6 Rock (And.intro hPet hfly)
  | inr h2 =>
      cases h2 with
      | inl hbird =>
          have hcf : CanFly Rock := R4 Rock (And.intro hPet hbird)
          exact R6 Rock (And.intro hPet hcf)
      | inr hnotbr =>
          have hmonorbird : Monkey Rock ∨ Bird Rock := R3 Rock hPet
          have hb : Bird Rock :=
            Or.elim hmonorbird
              (fun hmon =>
                have hmamm : Mammal Rock := R2 Rock hmon
                have hanim : Animal Rock := R1 Rock (And.intro hPet hmamm)
                have hbr : CanBreathe Rock := R5 Rock (And.intro hPet hanim)
                have hfalse : False := hnotbr hbr
                False.elim hfalse)
              (fun hbird => hbird)
          have hcf : CanFly Rock := R4 Rock (And.intro hPet hb)
          exact R6 Rock (And.intro hPet hcf)
</lean>

ANSWER: True
