<lean>
axiom obj : Type
axiom Platypus : obj → Prop
axiom Mammal : obj → Prop
axiom Human : obj → Prop
axiom HasTeeth : obj → Prop

-- Some mammals have teeth.
axiom E1 : ∃ x : obj, Mammal x ∧ HasTeeth x
-- Platypuses have no teeth.
axiom A2 : ∀ x : obj, Platypus x → ¬ HasTeeth x
-- Platypuses are mammals.
axiom A3 : ∀ x : obj, Platypus x → Mammal x
-- Humans have teeth.
axiom A4 : ∀ x : obj, Human x → HasTeeth x

theorem platypuses_mammals_no_teeth : ∀ x : obj, Platypus x → Mammal x ∧ ¬ HasTeeth x := by
  intro x hx
  have hm : Mammal x := A3 x hx
  have hnt : ¬ HasTeeth x := A2 x hx
  exact And.intro hm hnt
</lean>

ANSWER: True
