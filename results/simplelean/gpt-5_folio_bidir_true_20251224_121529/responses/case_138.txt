-- Premises
-- Unsupervised learning algorithms do not require labeled data.
axiom unsup_no_labels : ∀ a : Alg, Unsupervised a → ¬ RequiresLabels a
-- The state-of-the-art text summarization model is trained with machine learning algorithms.
axiom some_alg_used : ∃ a : Alg, Uses SOTA a
-- Reinforcement learning is not used to train the state-of-the-art text summarization model.
axiom rl_not_used : ∀ a : Alg, Uses SOTA a → ¬ Reinforcement a
-- The Machine Learning algorithm for training text summarization models requires labeled data.
axiom used_requires_labels : ∀ a : Alg, Uses SOTA a → RequiresLabels a

-- From the premises we can show no unsupervised algorithm is used to train SOTA
theorem no_unsupervised_used : ¬ ∃ a : Alg, Uses SOTA a ∧ Unsupervised a := by
  intro h
  rcases h with ⟨a, hUse, hUnsup⟩
  have hReq : RequiresLabels a := used_requires_labels a hUse
  have hNotReq : ¬ RequiresLabels a := unsup_no_labels a hUnsup
  exact hNotReq hReq
