<lean>
axiom obj : Type

-- Predicates
axiom Animal : obj → Prop
axiom HeckCattle : obj → Prop
axiom Aurochs : obj → Prop
axiom BredBack : obj → Prop
axiom ArtificialSelection : obj → Prop

-- Extra predicates from context (not needed for the proof)
axiom ToBeBredBack : obj → Prop
axiom ResembleExtinctAnimals : obj → Prop

-- Axioms from the context
-- Breeding back is a form of artificial selection
axiom R1 : ∀ x : obj, BredBack x → ArtificialSelection x

-- Heck cattle were bred back (existential reading)
axiom T1 : ∃ x : obj, HeckCattle x ∧ BredBack x

-- Heck cattle are animals
axiom R2 : ∀ x : obj, HeckCattle x → Animal x

-- Aurochs are animals
axiom R3 : ∀ x : obj, Aurochs x → Animal x

-- Some animals to be bred back resemble extinct animals (not needed for the proof)
axiom T3 : ∃ x : obj, Animal x ∧ ToBeBredBack x ∧ ResembleExtinctAnimals x

-- Goal: Some Heck cattle are artificially selected
theorem some_heck_cattle_artificial : ∃ x : obj, HeckCattle x ∧ ArtificialSelection x := by
  obtain ⟨x, hx⟩ := T1
  obtain ⟨hHC, hBB⟩ := hx
  have hAS : ArtificialSelection x := R1 x hBB
  exact ⟨x, And.intro hHC hAS⟩
</lean>

ANSWER: True
