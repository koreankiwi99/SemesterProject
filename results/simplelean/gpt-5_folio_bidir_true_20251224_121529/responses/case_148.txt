axiom CanParticipate2024 : obj → Prop

axiom A1 : ∀ x : obj, CanRegisterUS x → CanParticipate2024 x
-- 2. US citizenship implies ability to register in the US.
axiom A2 : ∀ x : obj, USCitizen x → CanRegisterUS x
-- 3. Every person has either US or Taiwanese citizenship.
axiom A3 : ∀ x : obj, USCitizen x ∨ TWNCitizen x
-- 4. No Russian Federation officials hold Taiwanese citizenship.
axiom A4 : ∀ x : obj, RFOfficial x → ¬ TWNCitizen x
-- 5. Vladimir neither holds Taiwanese citizenship nor is a manager at Gazprom.
axiom A5a : ¬ TWNCitizen Vladimir
axiom A5b : ¬ ManagerGazprom Vladimir
-- 6. Ekaterina can register in the US, or she is a Russian Federation official.
axiom A6 : CanRegisterUS Ekaterina ∨ RFOfficial Ekaterina

theorem Vladimir_USCitizen : USCitizen Vladimir :=
by

  cases h with
  | inl hUS => exact hUS
  | inr hTWN =>

theorem Vladimir_CanParticipate : CanParticipate2024 Vladimir :=
by

      have hdisj := A3 Ekaterina
      cases hdisj with
      | inl hUS =>
          exact A2 Ekaterina hUS
      | inr hTWN =>
          exact False.elim (hnotTWN hTWN)
