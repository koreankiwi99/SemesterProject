<lean>
axiom obj : Type
axiom NVP : obj
axiom NC : obj
axiom PF : obj

axiom PublishedBy : obj → obj → Prop
axiom English : obj → Prop
axiom TranslatedFromItalian : obj → Prop

-- All of New Vessel Press's published books are in English.
axiom A1 : ∀ x : obj, PublishedBy x NVP → English x

-- Neapolitan Chronicles is published by New Vessel Press.
axiom A2 : PublishedBy NC NVP

-- Neapolitan Chronicles was translated from Italian.
axiom A3 : TranslatedFromItalian NC

-- Palace of Flies is published by New Vessel Press.
axiom A4 : PublishedBy PF NVP

theorem nc_english : English NC := by
  apply A1 NC
  exact A2
</lean>

ANSWER: True
