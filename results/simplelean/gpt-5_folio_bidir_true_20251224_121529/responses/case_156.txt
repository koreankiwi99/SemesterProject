<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom UnivPartTimeJob : obj → Prop
axiom TakingDatabase : obj → Prop
axiom WithProfDavid : obj → Prop
axiom WorksLab : obj → Prop

-- None of the students who currently work in the library are from the computer science department.
axiom A1 : ∀ x : obj, WorksLibrary x → ¬ FromCS x
-- All of the students with part-time jobs offered by the university currently work in the library.
axiom A2 : ∀ x : obj, UnivPartTimeJob x → WorksLibrary x
-- All of the students who are taking the database course are from the computer science department.
axiom A3 : ∀ x : obj, TakingDatabase x → FromCS x
-- All of the students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, WithProfDavid x → TakingDatabase x
-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, WorksLab x → WithProfDavid x

-- James is a student working in the lab.
axiom A6 : Student James
axiom A7 : WorksLab James

-- James doesn't work in the lab or have a part-time job offered by the university.
axiom A8 : ¬ WorksLab James ∨ ¬ UnivPartTimeJob James

-- The question asks to prove: UnivPartTimeJob James
-- We cannot prove that; in fact we can derive its negation:
theorem james_no_univ_part_time_job : ¬ UnivPartTimeJob James := by
  intro hjob
  -- From job → library
  have hlib : WorksLibrary James := A2 James hjob
  -- Library → not CS
  have hnotcs : ¬ FromCS James := A1 James hlib
  -- From lab → with David → database → CS
  have hcs : FromCS James := by
    have hwd : WithProfDavid James := A5 James A7
    have hdb : TakingDatabase James := A4 James hwd
    exact A3 James hdb
  -- Contradiction
  exact hnotcs hcs
