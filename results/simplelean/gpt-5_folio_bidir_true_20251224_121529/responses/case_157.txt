<lean>
axiom obj : Type
axiom Student : obj → Prop
axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom PTJobUniv : obj → Prop
axiom TakingDB : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorksLab : obj → Prop
axiom James : obj

-- None of the students who currently work in the library are from the CS department.
axiom A1 : ∀ x : obj, WorksLibrary x → ¬ FromCS x
-- All of the students with part-time jobs offered by the university currently work in the library.
axiom A2 : ∀ x : obj, PTJobUniv x → WorksLibrary x
-- All of the students who are taking the database course are from the CS department.
axiom A3 : ∀ x : obj, TakingDB x → FromCS x
-- All of the students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, TakingWithDavid x → TakingDB x
-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, WorksLab x → TakingWithDavid x
-- James is a student working in the lab.
axiom A6 : Student James ∧ WorksLab James

theorem james_no_univ_pt_job : ¬ PTJobUniv James := by
  intro hpt
  have hlib : WorksLibrary James := A2 James hpt
  have hNotCS : ¬ FromCS James := A1 James hlib
  have hLab : WorksLab James := And.right A6
  have hDavid : TakingWithDavid James := A5 James hLab
  have hDB : TakingDB James := A4 James hDavid
  have hCS : FromCS James := A3 James hDB
  exact hNotCS hCS
</lean>

ANSWER: True
