<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom HasUniPartTimeJob : obj → Prop
axiom TakesDatabase : obj → Prop
axiom TakesWithProfDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- None of the students who currently work in the library are from the CS department.
axiom A1 : ∀ x : obj, Student x ∧ WorksInLibrary x → ¬ FromCS x
-- All of the students with part-time jobs offered by the university currently work in the library.
axiom A2 : ∀ x : obj, Student x ∧ HasUniPartTimeJob x → WorksInLibrary x
-- All of the students who are taking the database course are from the CS department.
axiom A3 : ∀ x : obj, Student x ∧ TakesDatabase x → FromCS x
-- All of the students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, Student x ∧ TakesWithProfDavid x → TakesDatabase x
-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, Student x ∧ WorksInLab x → TakesWithProfDavid x

-- James is a student working in the lab.
axiom J1 : Student James ∧ WorksInLab James
-- James doesn't work in the lab or have a part-time job offered by the university. (interpreted as neither)
axiom J2 : ¬ WorksInLab James ∧ ¬ HasUniPartTimeJob James

theorem james_takes_database : TakesDatabase James := by
  have hS : Student James := And.left J1
  have hLab : WorksInLab James := And.right J1
  have hDavid : TakesWithProfDavid James := A5 James (And.intro hS hLab)
  exact A4 James (And.intro hS hDavid)
</lean>

ANSWER: True
