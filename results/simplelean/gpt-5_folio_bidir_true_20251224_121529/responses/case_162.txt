axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

-- A hawk never swims.
axiom R1 : ∀ x : obj, Hawk x → ¬ Swims x
-- Some birds are hawks.
axiom E1 : ∃ x : obj, Bird x ∧ Hawk x

-- From the premises, we can show it's false that all birds swim.
theorem exists_bird_not_swim : ∃ x : obj, Bird x ∧ ¬ Swims x := by
  apply Exists.elim E1
  intro x hx
  have hxB : Bird x := And.left hx
  have hxH : Hawk x := And.right hx
  have hns : ¬ Swims x := R1 x hxH
  exact Exists.intro x (And.intro hxB hns)

theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swims x) := by
  intro hAll
  apply Exists.elim E1
  intro x hx
  have hxB : Bird x := And.left hx
  have hxH : Hawk x := And.right hx
  have hxns : ¬ Swims x := R1 x hxH
  have hxs : Swims x := hAll x hxB
  exact hxns hxs
