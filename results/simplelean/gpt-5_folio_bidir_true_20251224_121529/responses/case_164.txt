axiom AirsMon8 : obj → Prop
axiom RosesOnTV : obj → Prop
axiom BachelorEpisode : obj → Prop
axiom PortraysReal : obj → Prop
axiom RealityTV : obj → Prop
axiom Show : obj → Prop

-- Some show airing at 8 pm on Monday gives out roses on TV.
axiom E1 : ∃ s : obj, AirsMon8 s ∧ RosesOnTV s

-- If a show gives out roses on TV, then the show is an episode of The Bachelor.
axiom R1 : ∀ x : obj, RosesOnTV x → BachelorEpisode x

-- The Bachelor portrays the lives of real people (modeled as: all its episodes portray real lives).
axiom R2 : ∀ x : obj, BachelorEpisode x → PortraysReal x

-- All shows portraying the lives of real people are reality TV shows.
axiom R3 : ∀ x : obj, PortraysReal x → RealityTV x

-- Breaking Bad is a show.
axiom T1 : Show BreakingBad

-- Breaking Bad is not a reality TV show.
axiom T2 : ¬ RealityTV BreakingBad

-- If a show gives roses on TV, then it's reality TV.
lemma roses_implies_reality (x : obj) : RosesOnTV x → RealityTV x :=
by
  intro hx
  have hB : BachelorEpisode x := R1 x hx
  have hPR : PortraysReal x := R2 x hB
  exact R3 x hPR

-- Therefore, Breaking Bad does not give out roses on TV.
theorem not_roses_on_BB : ¬ RosesOnTV BreakingBad :=
by
  intro h
  have : RealityTV BreakingBad := roses_implies_reality BreakingBad h
  exact T2 this

-- Hence, the claim "Breaking Bad is a TV show in which roses are given out on Monday at 8 pm" is false.
theorem claim_false : ¬ (AirsMon8 BreakingBad ∧ RosesOnTV BreakingBad) :=
by
  intro h
  apply not_roses_on_BB
  exact h.right
