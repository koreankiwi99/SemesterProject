<lean>
axiom obj : Type
axiom Deadly : obj → Prop
axiom LowSurvival : obj → Prop
axiom SevereCancer : obj → Prop
axiom BileDuctCancer : obj → Prop
axiom Cholangiocarcinoma : obj → Prop
axiom MildFlu : obj → Prop
axiom ColorectalCancer : obj

-- If something is deadly, then it has a low survival rate.
axiom A1 : ∀ x : obj, Deadly x → LowSurvival x
-- Severe cancers are deadly.
axiom A2 : ∀ x : obj, SevereCancer x → Deadly x
-- Bile duct cancer is a severe cancer.
axiom A3 : ∀ x : obj, BileDuctCancer x → SevereCancer x
-- All Cholangiocarcinoma is bile duct cancer.
axiom A4 : ∀ x : obj, Cholangiocarcinoma x → BileDuctCancer x
-- Mild flu comes with a low survival rate.
axiom A5 : ∀ x : obj, MildFlu x → LowSurvival x
-- Colorectal cancer is not both a bile duct cancer and with a low survival rate.
axiom A6 : ¬ (BileDuctCancer ColorectalCancer ∧ LowSurvival ColorectalCancer)

-- Derivable fact: Colorectal cancer is not a bile duct cancer.
theorem not_bileduct_colorectal : ¬ BileDuctCancer ColorectalCancer :=
by
  intro hbdc
  have hsev : SevereCancer ColorectalCancer := A3 ColorectalCancer hbdc
  have hdead : Deadly ColorectalCancer := A2 ColorectalCancer hsev
  have hlow : LowSurvival ColorectalCancer := A1 ColorectalCancer hdead
  exact A6 (And.intro hbdc hlow)

-- ∨ (Cholangiocarcinoma ColorectalCancer ∧ MildFlu ColorectalCancer ∧ BileDuctCancer ColorectalCancer)
