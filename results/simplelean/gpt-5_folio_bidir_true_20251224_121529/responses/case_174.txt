<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatSaladsRegularly : obj → Prop
axiom VeryConscious : obj → Prop
axiom GrewUpHealthHome : obj → Prop
axiom FulfillDailyNutritionalIntakes : obj → Prop
axiom DisregardWellbeing : obj → Prop
axiom VisitGymDaily : obj → Prop

axiom R1 : ∀ x : obj, EatSaladsRegularly x → VeryConscious x
axiom R2 : ∀ x : obj, GrewUpHealthHome x → EatSaladsRegularly x
axiom R3 : ∀ x : obj, FulfillDailyNutritionalIntakes x → GrewUpHealthHome x
axiom R4 : ∀ x : obj, DisregardWellbeing x → ¬ VeryConscious x
axiom R5 : ∀ x : obj, VisitGymDaily x → FulfillDailyNutritionalIntakes x

axiom T6 : (GrewUpHealthHome Taylor ∧ DisregardWellbeing Taylor) ∨ (¬GrewUpHealthHome Taylor ∧ ¬DisregardWellbeing Taylor)

theorem taylor_neither_grew_nor_visit : ¬GrewUpHealthHome Taylor ∧ ¬VisitGymDaily Taylor := by
  have hcontra : ¬ (GrewUpHealthHome Taylor ∧ DisregardWellbeing Taylor) := by
    intro h
    have hg : GrewUpHealthHome Taylor := And.left h
    have hd : DisregardWellbeing Taylor := And.right h
    have he : EatSaladsRegularly Taylor := R2 Taylor hg
    have hc : VeryConscious Taylor := R1 Taylor he
    have hnc : ¬ VeryConscious Taylor := R4 Taylor hd
    exact hnc hc
  have hsecond : (¬GrewUpHealthHome Taylor ∧ ¬DisregardWellbeing Taylor) := by
    cases T6 with
    | inl hleft => exact False.elim (hcontra hleft)
    | inr hright => exact hright
  have hNotG : ¬GrewUpHealthHome Taylor := And.left hsecond
  have hNotV : ¬VisitGymDaily Taylor := by
    intro hv
    have hf : FulfillDailyNutritionalIntakes Taylor := R5 Taylor hv
    have hg' : GrewUpHealthHome Taylor := R3 Taylor hf
    exact hNotG hg'
  exact And.intro hNotG hNotV
</lean>

ANSWER: True
