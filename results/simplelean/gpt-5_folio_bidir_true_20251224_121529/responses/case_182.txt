<lean>
axiom obj : Type
axiom Tom : obj

axiom NiceToAnimals : obj → Prop
axiom MeanToAnimals : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, NiceToAnimals x → ¬ MeanToAnimals x
axiom P2 : ∃ x : obj, Grumpy x ∧ MeanToAnimals x
axiom P3 : ∀ x : obj, AnimalLover x → NiceToAnimals x
axiom P4 : ∀ x : obj, PetOwner x → AnimalLover x
axiom P5 : PetOwner Tom

-- Conclusion: Tom is not both grumpy and mean to animals.
theorem tom_not_grumpy_and_mean : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) := by
  have hAL : AnimalLover Tom := P4 Tom P5
  have hNice : NiceToAnimals Tom := P3 Tom hAL
  have hNotMean : ¬ MeanToAnimals Tom := P1 Tom hNice
  intro hGM
  exact hNotMean hGM.right
</lean>

ANSWER: True
