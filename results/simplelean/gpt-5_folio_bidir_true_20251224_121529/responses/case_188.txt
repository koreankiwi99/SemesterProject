<lean>
axiom obj : Type
axiom Yuri : obj

axiom ProfessionalBasketballPlayer : obj → Prop
axiom American : obj → Prop
axiom CanJumpShots : obj → Prop
axiom LeapsStraightUp : obj → Prop
axiom ActivatesLegMuscles : obj → Prop

-- Some professional basketball players are not American nationals.
axiom some_pro_not_american : ∃ x : obj, ProfessionalBasketballPlayer x ∧ ¬ American x
-- All professional basketball players can do jump shots.
axiom pro_can_jump : ∀ x : obj, ProfessionalBasketballPlayer x → CanJumpShots x
-- If someone can jump shots, they leap straight into the air.
axiom jump_implies_leap : ∀ x : obj, CanJumpShots x → LeapsStraightUp x
-- If someone leaps straight into the air, they activate their leg muscles.
axiom leap_implies_activate : ∀ x : obj, LeapsStraightUp x → ActivatesLegMuscles x
-- Yuri does not activate his leg muscles.
axiom yuri_not_activate : ¬ ActivatesLegMuscles Yuri

theorem yuri_not_american_pro : ¬ (American Yuri ∧ ProfessionalBasketballPlayer Yuri) := by
  intro h
  have hPro : ProfessionalBasketballPlayer Yuri := And.right h
  have hAct : ActivatesLegMuscles Yuri := by
    have hJump : CanJumpShots Yuri := pro_can_jump Yuri hPro
    have hLeap : LeapsStraightUp Yuri := jump_implies_leap Yuri hJump
    exact leap_implies_activate Yuri hLeap
  exact yuri_not_activate hAct
</lean>

ANSWER: True
