axiom obj : Type

axiom Jane : obj
axiom KiKi : obj

axiom Animal : obj → Prop
axiom Bark : obj → Prop
axiom Jump : obj → Prop
axiom HasLegs : obj → Prop
axiom Terricolous : obj → Prop
axiom Dog : obj → Prop
axiom Likes : obj → obj → Prop

-- Jane does not like any animals that bark.
axiom H1 : ∀ x : obj, Animal x ∧ Bark x → ¬ Likes Jane x
-- All dogs are animals that bark.
axiom H2 : ∀ x : obj, Dog x → Animal x ∧ Bark x
-- Jane likes all animals that jump.
axiom H3 : ∀ x : obj, Animal x ∧ Jump x → Likes Jane x
-- If an animal has legs, then it jumps.
axiom H4 : ∀ x : obj, Animal x ∧ HasLegs x → Jump x
-- If an animal is terricolous, then it has legs.
axiom H5 : ∀ x : obj, Animal x ∧ Terricolous x → HasLegs x
-- KiKi is an animal.
axiom H6 : Animal KiKi
-- If KiKi neither jumps nor has legs, then KiKi is terricolous or has legs.
axiom H7 : (¬ Jump KiKi ∧ ¬ HasLegs KiKi) → (Terricolous KiKi ∨ HasLegs KiKi)

open Classical

theorem kiki_not_bark_and_not_dog : ¬ Bark KiKi ∧ ¬ Dog KiKi := by
  classical
  -- First, prove KiKi jumps.
  have jumpK : Jump KiKi := by
    by_contra hJ
    have hL : HasLegs KiKi := by
      by_cases hLegs : HasLegs KiKi
      · exact hLegs
      · have hpair : ¬ Jump KiKi ∧ ¬ HasLegs KiKi := And.intro hJ hLegs
        have disj := H7 hpair
        cases disj with
        | inl hT =>
          have legsFromTerr := H5 KiKi (And.intro H6 hT)
          exact legsFromTerr
        | inr hL2 =>
          exact hL2
    have jFromL := H4 KiKi (And.intro H6 hL)
    exact hJ jFromL
  -- From jumping and liking/jane-barking rules, conclude KiKi does not bark.
  have notBark : ¬ Bark KiKi := by
    intro hBark
    have notLike : ¬ Likes Jane KiKi := H1 KiKi (And.intro H6 hBark)
    have like : Likes Jane KiKi := H3 KiKi (And.intro H6 jumpK)
    exact notLike like
  -- From dog -> bark, conclude KiKi is not a dog.
  have notDog : ¬ Dog KiKi := by
    intro hDog
    have barkFromDog : Bark KiKi := (And.right (H2 KiKi hDog))
    exact notBark barkFromDog
  exact And.intro notBark notDog
</lean>

ANSWER: True
