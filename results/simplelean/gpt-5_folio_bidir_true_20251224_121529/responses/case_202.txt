axiom obj : Type

-- Entities
axiom Ailton : obj
axiom Braga : obj
axiom Nautico : obj
axiom Fluminense : obj

-- Predicates/Relations
axiom FootballClub : obj → Prop
axiom LoanedOutTo : obj → obj → Prop

-- Facts from the context
axiom AiltonLoanedToBraga : LoanedOutTo Ailton Braga
axiom BragaIsClub : FootballClub Braga
axiom NauticoIsClub : FootballClub Nautico
axiom FluminenseIsClub : FootballClub Fluminense

theorem ailton_loaned_to_a_club : ∃ c : obj, FootballClub c ∧ LoanedOutTo Ailton c := by
