<lean>
axiom Show : Type
axiom StrangerThings : Show
axiom BlackMirror : Show

axiom Netflix : Show → Prop
axiom Popular : Show → Prop
axiom BingeKaren : Show → Prop
axiom DownloadKaren : Show → Prop
axiom ShareKarenLisa : Show → Prop

-- Facts about Stranger Things
axiom T_ST_popular : Popular StrangerThings
axiom T_ST_netflix : Netflix StrangerThings

-- Rules
axiom R_binge_if_popular_netflix : ∀ x : Show, Netflix x → Popular x → BingeKaren x
axiom R_binge_iff_download_on_netflix : ∀ x : Show, Netflix x → (BingeKaren x ↔ DownloadKaren x)
axiom R_share_if_binge_netflix : ∀ x : Show, Netflix x → BingeKaren x → ShareKarenLisa x

-- Facts about Black Mirror
axiom T_BM_not_download : ¬ DownloadKaren BlackMirror
axiom T_BM_netflix : Netflix BlackMirror

theorem share_stranger_things_with_lisa : ShareKarenLisa StrangerThings := by
  have hBinge : BingeKaren StrangerThings :=
    R_binge_if_popular_netflix StrangerThings T_ST_netflix T_ST_popular
  exact R_share_if_binge_netflix StrangerThings T_ST_netflix hBinge
</lean>

ANSWER: True
