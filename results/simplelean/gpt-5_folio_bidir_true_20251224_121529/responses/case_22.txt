axiom BlackMirror : obj

axiom Popular : obj → Prop
axiom BingeKaren : obj → Prop
axiom DownloadKaren : obj → Prop
axiom ShareWithLisa : obj → Prop

-- "Stranger Things" is a popular Netflix show.
axiom ST_netflix : NetflixShow StrangerThings
axiom ST_pop : Popular StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom R_pop_binge : ∀ x : obj, (NetflixShow x ∧ Popular x) → BingeKaren x

-- If and only if Karen binge-watches a Netflix show, she will download it. (restricted to Netflix shows)
axiom R_binge_download_iff : ∀ x : obj, NetflixShow x → (BingeKaren x ↔ DownloadKaren x)

-- Karen does not download "Black Mirror."
axiom BM_not_download : ¬ DownloadKaren BlackMirror

-- "Black Mirror" is a Netflix show.
axiom BM_netflix : NetflixShow BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom R_binge_share : ∀ x : obj, NetflixShow x → BingeKaren x → ShareWithLisa x

-- From the biconditional and ¬Download, we can derive ¬Binge for Black Mirror.
theorem BM_not_binge : ¬ BingeKaren BlackMirror :=
by
  have hiff : BingeKaren BlackMirror ↔ DownloadKaren BlackMirror := R_binge_download_iff BlackMirror BM_netflix
  have hdir : BingeKaren BlackMirror → DownloadKaren BlackMirror := Iff.mp hiff
  intro hb
  exact BM_not_download (hdir hb)
