axiom BlackMirror : Show

axiom NetflixShow : Show → Prop
axiom Popular : Show → Prop
axiom Binge : Show → Prop      -- Karen binge-watches the show
axiom Download : Show → Prop   -- Karen downloads the show
axiom ShareLisa : Show → Prop  -- Karen shares the show with Lisa

-- "Stranger Things" is a popular Netflix show.
axiom ST_netflix : NetflixShow StrangerThings
axiom ST_popular : Popular StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom R_popular_binge : ∀ s : Show, NetflixShow s ∧ Popular s → Binge s

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom R_binge_iff_download : ∀ s : Show, NetflixShow s → (Binge s ↔ Download s)

-- Karen does not download "Black Mirror."
axiom BM_not_download : ¬ Download BlackMirror

-- "Black Mirror" is a Netflix show.
axiom BM_netflix : NetflixShow BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom R_binge_shareLisa : ∀ s : Show, NetflixShow s → Binge s → ShareLisa s

-- From the iff and non-download, we can derive Karen does not binge-watch "Black Mirror."
theorem BM_not_binge : ¬ Binge BlackMirror := by
  intro hBinge
  have hiff : Binge BlackMirror ↔ Download BlackMirror := R_binge_iff_download BlackMirror BM_netflix
  have hDownload : Download BlackMirror := (Iff.mp hiff) hBinge
  exact BM_not_download hDownload
