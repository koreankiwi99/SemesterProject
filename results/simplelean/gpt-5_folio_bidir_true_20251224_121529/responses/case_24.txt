<lean>
axiom obj : Type
axiom Beijing : obj

axiom Hosted2008SummerOlympics : obj → Prop
axiom HostedWinterOlympics : obj → Prop

-- From the context:
-- "Beijing hosted the 2008 Summer Olympics"
axiom T1 : Hosted2008SummerOlympics Beijing
-- "Beijing has hosted the Summer and Winter Olympics"
axiom T2 : HostedWinterOlympics Beijing

theorem beijing_hosted_both : Hosted2008SummerOlympics Beijing ∧ HostedWinterOlympics Beijing := by
  apply And.intro
  · exact T1
  · exact T2
</lean>

ANSWER: True
