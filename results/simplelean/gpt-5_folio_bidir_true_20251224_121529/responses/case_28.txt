<lean>
axiom obj : Type
axiom Marvin : obj

axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromMars : obj → Prop
axiom FromEarth : obj → Prop
axiom HighlyIntelligent : obj → Prop

axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x
axiom A2 : ∀ x : obj, FromMars x → Alien x
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬ Human x
axiom A4 : ∀ x : obj, HighlyIntelligent x ∧ FromEarth x → Human x
axiom H1 : HighlyIntelligent Marvin
axiom H2 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom A5 : ¬ FromEarth Marvin → Extraterrestrial Marvin

theorem marvin_neither : ¬ Human Marvin ∧ ¬ FromMars Marvin := by
  cases H2
  case inl hboth =>
    have fe : FromEarth Marvin := And.left hboth
    have fm : FromMars Marvin := And.right hboth
    have alien : Alien Marvin := A2 Marvin fm
    have extra : Extraterrestrial Marvin := A1 Marvin alien
    have human : Human Marvin := A4 Marvin (And.intro H1 fe)
    have notHuman : ¬ Human Marvin := A3 Marvin extra
    have falsum : False := notHuman human
    exact False.elim falsum
  case inr hneither =>
    have notFE : ¬ FromEarth Marvin := And.left hneither
    have notFM : ¬ FromMars Marvin := And.right hneither
    have extra : Extraterrestrial Marvin := A5 notFE
    have notHuman : ¬ Human Marvin := A3 Marvin extra
    exact And.intro notHuman notFM
</lean>

ANSWER: True
