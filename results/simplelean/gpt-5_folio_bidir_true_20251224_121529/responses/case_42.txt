<lean>
axiom obj : Type
axiom Coco : obj

axiom RankedHighlyWTA : obj → Prop
axiom MostActiveGS : obj → Prop
axiom LostToIgaRG22 : obj → Prop
axiom LostToNadalRG22 : obj → Prop
axiom FemaleRG22 : obj → Prop
axiom MaleRG22 : obj → Prop
axiom PlayerRG22 : obj → Prop

-- If someone is ranked highly by the WTA, then they are among the most active Grand-Slam players.
axiom A1 : ∀ x : obj, RankedHighlyWTA x → MostActiveGS x
-- Everyone who lost to Iga at RG22 is ranked highly by the WTA.
axiom A2 : ∀ x : obj, LostToIgaRG22 x → RankedHighlyWTA x
-- All female tennis players at RG22 lost to Iga.
axiom A3 : ∀ x : obj, FemaleRG22 x → LostToIgaRG22 x
-- All tennis players at RG22 are either female or male.
axiom A4 : ∀ x : obj, PlayerRG22 x → FemaleRG22 x ∨ MaleRG22 x
-- All male tennis players at RG22 lost to Nadal.
axiom A5 : ∀ x : obj, MaleRG22 x → LostToNadalRG22 x
-- If Coco is ranked highly or lost to Nadal, then Coco is not a male tennis player at RG22.
axiom A6 : (RankedHighlyWTA Coco ∨ LostToNadalRG22 Coco) → ¬ MaleRG22 Coco
-- Coco Gauff is at RG22.
axiom T1 : PlayerRG22 Coco

theorem coco_most_active : MostActiveGS Coco := by
  have hdisj : FemaleRG22 Coco ∨ MaleRG22 Coco := A4 Coco T1
  have hnotmale : ¬ MaleRG22 Coco := by
    intro hmale
    have hlostn : LostToNadalRG22 Coco := A5 Coco hmale
    have hant : RankedHighlyWTA Coco ∨ LostToNadalRG22 Coco := Or.inr hlostn
    have hnm : ¬ MaleRG22 Coco := A6 hant
    exact hnm hmale
  have hfemale : FemaleRG22 Coco := by
    cases hdisj with
    | inl hf => exact hf
    | inr hm =>
      have : False := hnotmale hm
      exact False.elim this
  have hlostiga : LostToIgaRG22 Coco := A3 Coco hfemale
  have hranked : RankedHighlyWTA Coco := A2 Coco hlostiga
  exact A1 Coco hranked
</lean>

ANSWER: True
