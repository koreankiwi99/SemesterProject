axiom obj : Type
axiom Cat : obj → Prop
axiom Mammal : obj → Prop
axiom Pet : obj → Prop

-- All cats are mammals.
axiom A1 : ∀ x : obj, Cat x → Mammal x
-- Some pets are not mammals.
axiom A2 : ∃ x : obj, Pet x ∧ ¬ Mammal x

-- From the premises we can derive that there exists a pet that is not a cat.
theorem exists_pet_not_cat : ∃ x : obj, Pet x ∧ ¬ Cat x :=
by
  cases A2 with
  | intro x hx =>
    cases hx with
    | intro hPet hNotMammal =>
      have hNotCat : ¬ Cat x := by
        intro hCat
        have hMammal : Mammal x := A1 x hCat
        exact hNotMammal hMammal
      exact Exists.intro x (And.intro hPet hNotCat)
