<lean>
axiom obj : Type
axiom N1984 : obj

axiom Digital : obj → Prop
axiom Analog : obj → Prop
axiom Printed : obj → Prop
axiom Streaming : obj → Prop
axiom Hardcover : obj → Prop

-- No digital media are analog.
axiom NoDigitalAnalog : ∀ x : obj, Digital x → ¬ Analog x
-- Every printed text is analog media.
axiom PrintedAnalog : ∀ x : obj, Printed x → Analog x
-- All streaming services are digital media.
axiom StreamingDigital : ∀ x : obj, Streaming x → Digital x
-- If an object is a hardcover book, then it is printed text.
axiom HardcoverPrinted : ∀ x : obj, Hardcover x → Printed x
-- If 1984 is a streaming service, then 1984 is a hardcover book.
axiom Stream1984Hardcover : Streaming N1984 → Hardcover N1984

theorem not_streaming_1984 : ¬ Streaming N1984 := by
  intro hs
  have hd : Digital N1984 := StreamingDigital N1984 hs
  have hb : Hardcover N1984 := Stream1984Hardcover hs
  have hp : Printed N1984 := HardcoverPrinted N1984 hb
  have ha : Analog N1984 := PrintedAnalog N1984 hp
  have hna : ¬ Analog N1984 := NoDigitalAnalog N1984 hd
  exact hna ha
</lean>

ANSWER: True
