<lean>
axiom obj : Type
axiom French : obj
axiom Spanish : obj
axiom German : obj
axiom Basque : obj

axiom Romance : obj → Prop
axiom IndoEuropean : obj → Prop
axiom Related : obj → obj → Prop

-- All Romance languages are Indo-European languages.
axiom A1 : ∀ x : obj, Romance x → IndoEuropean x

-- Romance languages are a language family (modeled via a generic family notion).
axiom IsFamily : (obj → Prop) → Prop
axiom FamilyRelated : ∀ F : obj → Prop, IsFamily F → ∀ x y : obj, F x → F y → Related x y
axiom RomanceFamily : IsFamily Romance

-- French and Spanish are both Romance languages.
axiom A2 : Romance French
axiom A3 : Romance Spanish

-- German is related to Spanish.
axiom A4 : Related German Spanish

-- Basque is not related to any other language.
axiom A5 : ∀ y : obj, ¬ Related Basque y

theorem french_indo_european : IndoEuropean French := by
  apply A1
  exact A2
</lean>

ANSWER: True
