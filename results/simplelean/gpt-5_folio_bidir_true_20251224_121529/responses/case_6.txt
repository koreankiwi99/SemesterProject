<lean>
axiom obj : Type
axiom Animal : obj → Prop
axiom Mammal : obj → Prop
axiom Human : obj → Prop
axiom OccursMonkeypoxVirusIn : obj → Prop

-- Humans are mammals
axiom Hm : ∀ x : obj, Human x → Mammal x
-- Mammals are animals
axiom Ma : ∀ x : obj, Mammal x → Animal x
-- Monkeypox virus can occur in certain animals (implies existence of some animals)
axiom VirusInSomeAnimals : ∃ x : obj, Animal x ∧ OccursMonkeypoxVirusIn x

theorem exists_animal : ∃ x : obj, Animal x := by
  exact Exists.elim VirusInSomeAnimals (fun x hx => Exists.intro x (And.left hx))
</lean>

ANSWER: True
