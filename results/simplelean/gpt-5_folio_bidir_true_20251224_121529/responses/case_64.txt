<lean>
axiom obj : Type
axiom Ted : obj

axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop
axiom Pet : obj → Prop

-- All Brown Swiss cattle are cows.
axiom A1 : ∀ x : obj, BrownSwissCattle x → Cow x
-- Some pets are Brown Swiss cattle.
axiom A2 : ∃ x : obj, Pet x ∧ BrownSwissCattle x
-- All cows are domesticated animals.
axiom A3 : ∀ x : obj, Cow x → Domesticated x
-- Alligators are not domesticated animals.
axiom A4 : ∀ x : obj, Alligator x → ¬ Domesticated x
-- Ted is an alligator.
axiom A5 : Alligator Ted

theorem ted_bsc_implies_not_pet : BrownSwissCattle Ted → ¬ Pet Ted := by
  intro hBSC
  have hCow : Cow Ted := A1 Ted hBSC
  have hDom : Domesticated Ted := A3 Ted hCow
  have hNotDom : ¬ Domesticated Ted := A4 Ted A5
  have hFalse : False := hNotDom hDom
  intro _
  exact False.elim hFalse
</lean>

ANSWER: True
