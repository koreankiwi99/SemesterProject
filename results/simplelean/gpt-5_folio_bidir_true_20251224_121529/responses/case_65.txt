<lean>
axiom obj : Type
axiom YaleUniversity : obj
axiom PrivateIvyLeagueResearchUniversity : obj → Prop
axiom MovedToNewHaven : obj → Prop

-- Context axioms
axiom A1 : PrivateIvyLeagueResearchUniversity YaleUniversity
axiom A2 : MovedToNewHaven YaleUniversity

-- Claim: There exists a private Ivy League research university that moved to New Haven.
theorem exists_private_ivy_league_research_university_moved_to_new_haven :
  ∃ x : obj, PrivateIvyLeagueResearchUniversity x ∧ MovedToNewHaven x :=
by
  apply Exists.intro YaleUniversity
  apply And.intro
  · exact A1
  · exact A2
</lean>

ANSWER: True
