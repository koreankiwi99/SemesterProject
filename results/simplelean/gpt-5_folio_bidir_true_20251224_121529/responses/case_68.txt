<lean>
axiom Series : Type
axiom Badults : Series
axiom Person : Type
axiom AndrewCollins : Person
axiom WorkingTitleSDS : Series → Prop
axiom ScriptEditor : Person → Series → Prop
axiom WT_Badults : WorkingTitleSDS Badults
axiom SE_Badults : ScriptEditor AndrewCollins Badults

theorem collins_editor_of_sds_series :
  ∃ s : Series, WorkingTitleSDS s ∧ ScriptEditor AndrewCollins s := by
  refine Exists.intro Badults ?_
  exact And.intro WT_Badults SE_Badults
</lean>

ANSWER: True
