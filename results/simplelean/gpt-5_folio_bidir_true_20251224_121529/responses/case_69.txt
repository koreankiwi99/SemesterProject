axiom T1 : PilotedOnBBCThree Badults

-- The statement to assess:
def Statement : Prop :=
  ∀ p : obj, MemberOfPappys p → ¬ ∃ s : obj, StarredIn p s ∧ (PilotedOnBBCTwo s ∨ PilotedOnBBCThree s)

-- We derive a counterexample: there exists a Pappy's member who starred in a show piloted on BBC Three.
