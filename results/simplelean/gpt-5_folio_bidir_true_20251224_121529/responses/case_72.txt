<lean>
axiom obj : Type
axiom KO : obj

axiom Stock : obj → Prop
axiom Growth : obj → Prop
axiom Mature : obj → Prop
axiom EarnRapid : obj → Prop
axiom Suitable : obj → Prop
axiom Volatile : obj → Prop

-- All growth stocks are bought to earn profits from rapid price appreciation.
axiom A1 : ∀ x : obj, Growth x → EarnRapid x
-- If a stock is bought to earn profits from rapid price appreciation, then it is not suitable for a retirement fund.
axiom A2 : ∀ x : obj, EarnRapid x → ¬ Suitable x
-- Some stocks are growth stocks.
axiom E1 : ∃ x : obj, Stock x ∧ Growth x
-- All mature stocks are suitable for a retirement fund.
axiom A3 : ∀ x : obj, Mature x → Suitable x
-- KO is a mature stock.
axiom TKO_mature : Mature KO
axiom TKO_stock : Stock KO

theorem KO_neither_stock_nor_volatile :
  (Growth KO ∨ EarnRapid KO) → (¬ Stock KO ∧ ¬ Volatile KO) := by
  intro h
  have h_suitable : Suitable KO := A3 KO TKO_mature
  have h_not_suitable : ¬ Suitable KO := by
    cases h with
    | inl hg =>
      have her : EarnRapid KO := A1 KO hg
      exact A2 KO her
    | inr her =>
      exact A2 KO her
  have hFalse : False := h_not_suitable h_suitable
  have nStock : ¬ Stock KO := by
    intro _
    exact hFalse
  have nVolatile : ¬ Volatile KO := by
    intro _
    exact hFalse
  exact And.intro nStock nVolatile
</lean>

ANSWER: True
