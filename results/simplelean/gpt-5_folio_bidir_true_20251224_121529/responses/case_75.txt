<lean>
axiom obj : Type
axiom Hannah : obj

axiom AtMS : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom Picky : obj → Prop
axiom SpendTimeDining : obj → Prop

-- If people at Mary's school work in student jobs on campus, then they need to earn money.
axiom A1 : ∀ x : obj, AtMS x → Work x → Need x
-- If people at Mary's school order takeout frequently, then they work in student jobs on campus.
axiom A2 : ∀ x : obj, AtMS x → OrderTakeout x → Work x
-- People at Mary's school order takeout frequently or enjoy the dining hall.
axiom A3 : ∀ x : obj, AtMS x → (OrderTakeout x ∨ EnjoyDining x)
-- If people enjoy the dining hall, then they are not picky eaters.
axiom A4 : ∀ x : obj, AtMS x → EnjoyDining x → ¬ Picky x
-- If people enjoy the dining hall, then they spend a lot of their time eating and catching up with friends in the campus dining halls.
axiom A5 : ∀ x : obj, AtMS x → EnjoyDining x → SpendTimeDining x

-- Hannah-specific facts
axiom H_atMS : AtMS Hannah
axiom H_work : Work Hannah
axiom H_imp : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah)

theorem hannah_needs_money : Need Hannah := by
  apply A1 (x := Hannah)
  · exact H_atMS
  · exact H_work
</lean>

ANSWER: True
