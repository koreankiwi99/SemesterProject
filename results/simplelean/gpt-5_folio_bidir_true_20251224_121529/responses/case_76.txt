<lean>
axiom obj : Type
axiom Hannah : obj

axiom At : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom NotPicky : obj → Prop
axiom SpendTime : obj → Prop

-- Rules from the context
axiom R1 : ∀ x : obj, At x ∧ Work x → Need x
axiom R2 : ∀ x : obj, At x ∧ Order x → Work x
axiom R3 : ∀ x : obj, At x → (Order x ∨ Enjoy x)
axiom R4 : ∀ x : obj, At x ∧ Enjoy x → NotPicky x
axiom R5 : ∀ x : obj, At x ∧ Enjoy x → SpendTime x

-- Facts about Hannah
axiom H_at : At Hannah
axiom H_work : Work Hannah
axiom H_cond : Need Hannah → (NotPicky Hannah ∧ ¬ Need Hannah)

theorem hannah_claim : At Hannah ∧ NotPicky Hannah ∧ SpendTime Hannah := by
  have hAt : At Hannah := H_at
  have hdisj : Order Hannah ∨ Enjoy Hannah := R3 Hannah hAt
  cases hdisj with
  | inl hOrder =>
      have hWork : Work Hannah := R2 Hannah ⟨hAt, hOrder⟩
      have hNeed : Need Hannah := R1 Hannah ⟨hAt, hWork⟩
      have hNP_and_notNeed : NotPicky Hannah ∧ ¬ Need Hannah := H_cond hNeed
      have hNotPicky : NotPicky Hannah := And.left hNP_and_notNeed
      have hNotNeed : ¬ Need Hannah := And.right hNP_and_notNeed
      have hFalse : False := hNotNeed hNeed
      have hSpend : SpendTime Hannah := False.elim hFalse
      exact And.intro hAt (And.intro hNotPicky hSpend)
  | inr hEnjoy =>
      have hNotPicky : NotPicky Hannah := R4 Hannah ⟨hAt, hEnjoy⟩
      have hSpend : SpendTime Hannah := R5 Hannah ⟨hAt, hEnjoy⟩
      exact And.intro hAt (And.intro hNotPicky hSpend)
</lean>

ANSWER: True
