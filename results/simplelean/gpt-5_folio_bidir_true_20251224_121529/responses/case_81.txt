<lean>
axiom obj : Type
axiom DTC : obj  -- Dried Thai chilies
axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom MalaHotpot : obj → Prop
axiom BbyMProduct : obj → Prop
axiom Cupcake : obj → Prop

-- No baked sweets are spicy.
axiom A1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom A2 : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom A3 : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom A4 : ∀ x : obj, BbyMProduct x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom A5 : Spicy DTC ∨ (MalaHotpot DTC ∨ ¬ BakedSweet DTC)

theorem dtc_neither : ¬ BbyMProduct DTC ∧ ¬ BakedSweet DTC := by
  -- First show ¬ BakedSweet DTC
  have h_not_bs : ¬ BakedSweet DTC := by
    cases A5 with
    | inl hSpicy =>
        intro hb
        have hnsp : ¬ Spicy DTC := A1 DTC hb
        exact hnsp hSpicy
    | inr hrest =>
        cases hrest with
        | inl hMala =>
            intro hb
            have hSp : Spicy DTC := A3 DTC hMala
            have hnsp : ¬ Spicy DTC := A1 DTC hb
            exact hnsp hSp
        | inr hNotBS =>
            exact hNotBS
  -- Then show ¬ BbyMProduct DTC
  have h_not_bbm : ¬ BbyMProduct DTC := by
    intro hbbm
    have hCup : Cupcake DTC := A4 DTC hbbm
    have hBS : BakedSweet DTC := A2 DTC hCup
    exact h_not_bs hBS
  exact And.intro h_not_bbm h_not_bs
</lean>

ANSWER: True
