axiom obj : Type

-- Predicates
axiom Listed : obj → Prop
axiom ManyNeg : obj → Prop
axiom RatingGT4 : obj → Prop
axiom ProvideTakeOut : obj → Prop
axiom Popular : obj → Prop

-- Constant for the specific store
axiom Hamden : obj

-- Premises
axiom A1 : ∀ x : obj, Listed x → ¬ ManyNeg x
axiom A2 : ∀ x : obj, RatingGT4 x → Listed x
axiom A3 : ∃ x : obj, ¬ ProvideTakeOut x ∧ ManyNeg x
axiom A4 : ∀ x : obj, Popular x → RatingGT4 x
axiom A5 : RatingGT4 Hamden ∨ Popular Hamden

-- We can derive that Hamden has not received many negative reviews
theorem hamden_not_manyneg : ¬ ManyNeg Hamden := by
  have hRG : RatingGT4 Hamden :=
    Or.elim A5 (fun hRG => hRG) (fun hPop => A4 Hamden hPop)
  have hListed : Listed Hamden := A2 Hamden hRG
  exact A1 Hamden hListed
