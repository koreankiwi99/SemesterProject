<lean>
axiom obj : Type
axiom Tom : obj

axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Gould : obj → Prop
axiom Merriam : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop
axiom WildTurkey : obj → Prop

-- Premises about Tom
axiom T1 : ¬ Eastern Tom
axiom T2 : ¬ Osceola Tom
axiom T3 : ¬ Gould Tom
axiom T4 : ¬ Merriam Tom ∧ ¬ RioGrande Tom
axiom T5 : WildTurkey Tom

-- Exhaustiveness: every wild turkey is one of the six types
axiom ExhaustiveTypes :
  ∀ x : obj, WildTurkey x →
    (Eastern x ∨ Osceola x ∨ Gould x ∨ Merriam x ∨ RioGrande x ∨ Ocellated x)

theorem tom_is_ocellated : Ocellated Tom := by
  have h := ExhaustiveTypes Tom T5
  cases h with
  | inl hE =>
    exact False.elim (T1 hE)
  | inr h1 =>
    cases h1 with
    | inl hO =>
      exact False.elim (T2 hO)
    | inr h2 =>
      cases h2 with
      | inl hG =>
        exact False.elim (T3 hG)
      | inr h3 =>
        cases h3 with
        | inl hM =>
          exact False.elim ((And.left T4) hM)
        | inr h4 =>
          cases h4 with
          | inl hR =>
            exact False.elim ((And.right T4) hR)
          | inr hOc =>
            exact hOc
</lean>

ANSWER: True
