<lean>
axiom Building : Type
axiom Amount : Type

axiom OliveGarden : Building

axiom Managed : Building → Prop
axiom MonthlyRent : Building → Amount
axiom SecurityDeposit : Building → Amount
axiom More : Amount → Amount → Prop

axiom A2000 : Amount
axiom A1500 : Amount

-- Facts from the context
axiom OG_managed : Managed OliveGarden
axiom rent_OG_2000 : MonthlyRent OliveGarden = A2000
axiom more_2000_1500 : More A2000 A1500

-- Policy: For any managed building, the security deposit is equal to the monthly rent or more
axiom deposit_policy : ∀ b : Building, Managed b → (SecurityDeposit b = MonthlyRent b ∨ More (SecurityDeposit b) (MonthlyRent b))

theorem og_deposit_eq_2000_or_more : SecurityDeposit OliveGarden = A2000 ∨ More (SecurityDeposit OliveGarden) A2000 := by
  have h := deposit_policy OliveGarden OG_managed
  cases h with
  | inl h_eq =>
      apply Or.inl
      simpa [rent_OG_2000] using h_eq
  | inr h_more =>
      apply Or.inr
      simpa [rent_OG_2000] using h_more
</lean>

ANSWER: True
