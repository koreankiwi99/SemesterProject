axiom Tom : obj
axiom ETS : obj

axiom TakingGRE : obj → Prop
axiom ProvesHardship : obj → Prop
axiom LivesSingleParent : obj → Prop
axiom FewResources : obj → Prop
axiom ProvidesFinancialAid : obj → obj → Prop

-- ETS provides financial aid to GRE applicants who prove hardship
axiom R1 : ∀ x : obj, TakingGRE x ∧ ProvesHardship x → ProvidesFinancialAid ETS x
-- Those living in single-parent families or having few resources can prove hardship
axiom R2 : ∀ x : obj, (LivesSingleParent x ∨ FewResources x) → ProvesHardship x
-- Tom's facts
axiom T1 : LivesSingleParent Tom
axiom T2 : FewResources Tom
axiom T3 : TakingGRE Tom

-- There exists someone taking the GRE who has financial aid from something (ETS).
theorem exists_taker_with_aid : ∃ x : obj, TakingGRE x ∧ ∃ y : obj, ProvidesFinancialAid y x := by
  have hHard : ProvesHardship Tom := R2 Tom (Or.inl T1)
  have hAid : ProvidesFinancialAid ETS Tom := R1 Tom ⟨T3, hHard⟩
  refine ⟨Tom, ?_, ?_⟩
  · exact T3
  · refine ⟨ETS, ?_⟩
    exact hAid
