<lean>
axiom obj : Type

axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom CanBlock : obj → obj → Prop
axiom Jumps : obj → Prop
axiom Shooter : obj → Prop
axiom GreatShooter : obj → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R_block : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man
axiom Michael_man : Man Michael

-- Transitivity of taller
axiom Taller_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom Peter_man : Man Peter
axiom Peter_taller_Michael : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom Michael_blocks_if_not_jump : ∀ p : obj, ¬ Jumps p → CanBlock Michael p

-- Michael cannot block Windy's shooting.
axiom Michael_cannot_block_Windy : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom great_if_shooter_and_jump : ∀ p : obj, Shooter p → Jumps p → GreatShooter p

-- Windy is a shooter (since we talk about Windy's shooting)
axiom Windy_shooter : Shooter Windy

-- Excluded middle to reason by cases on jumping
axiom em : ∀ p : Prop, p ∨ ¬ p

theorem windy_great : GreatShooter Windy := by
  -- Derive that Windy jumps, otherwise Michael could block her
  have jw : Jumps Windy := by
    have h := em (Jumps Windy)
    cases h with
    | inl j => exact j
    | inr nj =>
        have cb : CanBlock Michael Windy := Michael_blocks_if_not_jump Windy nj
        have contra : False := Michael_cannot_block_Windy cb
        exact False.elim contra
  -- Now apply the rule about great shooters
  exact great_if_shooter_and_jump Windy Windy_shooter jw
</lean>

ANSWER: True
