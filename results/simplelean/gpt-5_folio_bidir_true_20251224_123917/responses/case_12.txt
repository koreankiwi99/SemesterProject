<lean>
axiom obj : Type
axiom Zelda : obj
axiom Top10 : obj → Prop
axiom SellsOver1M : obj → Prop
axiom CreatedByJapaneseCompany : obj → Prop

-- A Japanese game company created The Legend of Zelda.
axiom T1 : CreatedByJapaneseCompany Zelda
-- All games on the Top 10 list are made by Japanese game companies.
axiom R1 : ∀ x : obj, Top10 x → CreatedByJapaneseCompany x
-- If a game sells more than one million copies, then it is included in the Top 10 list.
axiom R2 : ∀ x : obj, SellsOver1M x → Top10 x
-- The Legend of Zelda sold more than one million copies.
axiom T2 : SellsOver1M Zelda

theorem zelda_top10 : Top10 Zelda := R2 Zelda T2
</lean>

ANSWER: True
