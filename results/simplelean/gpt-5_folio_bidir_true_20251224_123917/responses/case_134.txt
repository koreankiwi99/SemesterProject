<lean>
axiom obj : Type
axiom Sucks : obj → Prop
axiom Vacuum : obj → Prop
axiom Plunger : obj → Prop
axiom Vampire : obj → Prop
axiom HouseholdAppliance : obj → Prop

axiom Space : obj
axiom Duster : obj

axiom Rvac : ∀ x : obj, Vacuum x → Sucks x
axiom Tspace : Vacuum Space
axiom Rplunger : ∀ x : obj, Plunger x → Sucks x
axiom Rvampire : ∀ x : obj, Vampire x → Sucks x
axiom TdustAppliance : HouseholdAppliance Duster
axiom TdustNotSuck : ¬ Sucks Duster

theorem space_sucks : Sucks Space := by
  apply (Rvac Space)
  exact Tspace
</lean>

ANSWER: True
