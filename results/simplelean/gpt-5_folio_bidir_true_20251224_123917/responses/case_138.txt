axiom Alg : Type
axiom Model : Type
axiom SOTA : Model

axiom Supervised : Alg → Prop
axiom Unsupervised : Alg → Prop
axiom Reinforcement : Alg → Prop
axiom RequiresLabels : Alg → Prop
axiom Used : Alg → Model → Prop

-- ML algorithms are categorized as supervised, unsupervised, or reinforcement
axiom categorization : ∀ a : Alg, Supervised a ∨ Unsupervised a ∨ Reinforcement a

-- Unsupervised learning algorithms do not require labeled data
axiom unsup_no_labels : ∀ a : Alg, Unsupervised a → ¬ RequiresLabels a

-- The SOTA text summarization model is trained with ML algorithms (existence of some used algorithm)
axiom exists_used : ∃ a : Alg, Used a SOTA

-- Reinforcement learning is not used to train the SOTA model
axiom not_rl_used : ¬ ∃ a : Alg, Reinforcement a ∧ Used a SOTA

-- The ML algorithm for training text summarization models requires labeled data
axiom used_requires_labels : ∀ a : Alg, Used a SOTA → RequiresLabels a

-- We cannot prove that an unsupervised algorithm is used; in fact we can show it's impossible.
theorem no_unsup_used : ¬ ∃ a : Alg, Unsupervised a ∧ Used a SOTA := by
  intro h
  apply Exists.elim h
  intro a hpair
  apply And.elim hpair
  intro hU hUsed
  have hReq : RequiresLabels a := used_requires_labels a hUsed
  have hNotReq : ¬ RequiresLabels a := unsup_no_labels a hU
  exact hNotReq hReq
