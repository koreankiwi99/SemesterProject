<lean>
axiom obj : Type
axiom UnitedStates : obj
axiom TokyoOlympics : obj
axiom LastSummerOlympics : obj
axiom WonMostMedals : obj → obj → Prop

-- The last summer Olympic games was in Tokyo (i.e., it was the Tokyo Olympics)
axiom E1 : LastSummerOlympics = TokyoOlympics
-- The United States won the most medals in Tokyo (i.e., in the Tokyo Olympics)
axiom E2 : WonMostMedals UnitedStates TokyoOlympics

theorem us_won_most_medals_last_summer_olympics : WonMostMedals UnitedStates LastSummerOlympics := by
  simpa [E1] using E2
</lean>

ANSWER: True
