<lean>
axiom Show : Type
axiom ST : Show
axiom BM : Show

axiom Netflix : Show → Prop
axiom Popular : Show → Prop
axiom Binge : Show → Prop
axiom Download : Show → Prop
axiom ShareWithLisa : Show → Prop

-- "Stranger Things" is a popular Netflix show.
axiom ST_Netflix : Netflix ST
axiom ST_Popular : Popular ST

-- If a Netflix show is popular, Karen will binge-watch it.
axiom R1 : ∀ s : Show, Netflix s → Popular s → Binge s

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom R2 : ∀ s : Show, Netflix s → (Binge s ↔ Download s)

-- Karen does not download "Black Mirror."
axiom BM_notDownload : ¬ Download BM

-- "Black Mirror" is a Netflix show.
axiom BM_Netflix : Netflix BM

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom R3 : ∀ s : Show, Netflix s → Binge s → ShareWithLisa s

-- Goal: Karen will share "Stranger Things" with Lisa.
theorem share_ST : ShareWithLisa ST := by
  have hBinge : Binge ST := R1 ST ST_Netflix ST_Popular
  exact R3 ST ST_Netflix hBinge
</lean>

ANSWER: True
