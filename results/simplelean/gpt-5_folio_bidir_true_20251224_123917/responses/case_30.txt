axiom GrandSlamChampion : Person → Prop

axiom ProTennisPlayer : Person → Prop

axiom Athlete : Person → Prop

-- Everyone at the mixer is a Grand Slam champion or an Oscar-nominated actor.
axiom A1 : ∀ x : Person, At x → (GrandSlamChampion x ∨ OscarNominatedActor x)
-- Every Grand Slam champion at the mixer is a professional tennis player.
axiom A2 : ∀ x : Person, At x → GrandSlamChampion x → ProTennisPlayer x
-- All Oscar-nominated actors at the mixer are celebrities.
axiom A3 : ∀ x : Person, At x → OscarNominatedActor x → Celebrity x
-- All professional tennis players at the mixer are athletes.
axiom A4 : ∀ x : Person, At x → ProTennisPlayer x → Athlete x
-- If a person at the mixer is a celebrity, then they are well paid.
axiom A5 : ∀ x : Person, At x → Celebrity x → WellPaid x
-- If a person at the mixer is an athlete, then they are famous.
axiom A6 : ∀ x : Person, At x → Athlete x → Famous x
-- All well-paid people at the mixer live in tax havens.
axiom A7 : ∀ x : Person, At x → WellPaid x → LivesInTaxHaven x

-- Djokovic is at the mixer
axiom A8 : At Djokovic
-- If Djokovic is a famous athlete, then Djokovic is well-paid.
axiom A9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

-- From the premises we can derive the disjunction but not which disjunct holds:
theorem djokovic_gsc_or_ona : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := by
  exact A1 Djokovic A8

-- Some consequences (not required for the target claim):

  have hdisj : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := A1 Djokovic A8
  cases hdisj with

      have pt : ProTennisPlayer Djokovic := A2 Djokovic A8 hg
      have ath : Athlete Djokovic := A4 Djokovic A8 pt
      have fam : Famous Djokovic := A6 Djokovic A8 ath
      have hw : WellPaid Djokovic := A9 (And.intro fam ath)
      exact hw
  | inr ho =>
      have cele : Celebrity Djokovic := A3 Djokovic A8 ho
