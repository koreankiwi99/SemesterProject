axiom Mammal : obj → Prop
axiom Pet : obj → Prop

-- All cats are mammals.
axiom R1 : ∀ x : obj, Cat x → Mammal x
-- Some pets are not mammals.
axiom E1 : ∃ x : obj, Pet x ∧ ¬ Mammal x

-- From the premises we can derive that some pet is not a cat.
theorem exists_pet_not_cat : ∃ x : obj, Pet x ∧ ¬ Cat x := by
  cases E1 with
  | intro x hx =>
    cases hx with
    | intro hxPet hxNotMammal =>
      have hNotCat : ¬ Cat x := by
        intro hCat
        have hMammal : Mammal x := R1 x hCat
        exact hxNotMammal hMammal
      exact ⟨x, And.intro hxPet hNotCat⟩
