<lean>
axiom obj : Type
axiom KO : obj

axiom Growth : obj → Prop
axiom Rapid : obj → Prop
axiom Suitable : obj → Prop
axiom Mature : obj → Prop
axiom Stock : obj → Prop
axiom Volatile : obj → Prop

-- All growth stocks are bought to earn profits from rapid price appreciation.
axiom R1 : ∀ x : obj, Growth x → Rapid x
-- If a stock is bought for rapid price appreciation, then it is not suitable for a retirement fund.
axiom R2 : ∀ x : obj, Rapid x → ¬ Suitable x
-- Some stocks are growth stocks.
axiom E1 : ∃ x : obj, Stock x ∧ Growth x
-- All mature stocks are suitable for a retirement fund.
axiom R3 : ∀ x : obj, Mature x → Suitable x
-- KO is a mature stock.
axiom T1 : Mature KO

-- Statement: If KO is a growth stock or bought for rapid price appreciation,
-- then KO is neither a stock nor is its price volatile.
theorem target :
  (Growth KO ∨ Rapid KO) → (¬ Stock KO ∧ ¬ Volatile KO) := by
  intro h
  have hS : Suitable KO := R3 KO T1
  have hFalse : False := by
    cases h with
    | inl hG =>
      have hR : Rapid KO := R1 KO hG
      have hNS : ¬ Suitable KO := R2 KO hR
      exact hNS hS
    | inr hR =>
      have hNS : ¬ Suitable KO := R2 KO hR
      exact hNS hS
  exact False.elim hFalse
</lean>

ANSWER: True
