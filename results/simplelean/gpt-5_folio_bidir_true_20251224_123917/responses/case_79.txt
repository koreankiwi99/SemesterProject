<lean>
axiom obj : Type
axiom DriedThaiChilies : obj
axiom Spicy : obj → Prop
axiom BakedSweet : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

axiom no_baked_sweets_spicy : ∀ x : obj, BakedSweet x → ¬ Spicy x
axiom cupcakes_baked_sweets : ∀ x : obj, Cupcake x → BakedSweet x
axiom mala_hotpots_spicy : ∀ x : obj, MalaHotpot x → Spicy x
axiom bbm_cupcakes : ∀ x : obj, FromBBM x → Cupcake x
axiom dried_thai_chilies_disj : (Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies) ∨ ¬ BakedSweet DriedThaiChilies

theorem dried_thai_chilies_not_bbm : ¬ FromBBM DriedThaiChilies :=
by
  intro hbbm
  have hc : Cupcake DriedThaiChilies := bbm_cupcakes DriedThaiChilies hbbm
  have hbs : BakedSweet DriedThaiChilies := cupcakes_baked_sweets DriedThaiChilies hc
  have hnsp : ¬ Spicy DriedThaiChilies := no_baked_sweets_spicy DriedThaiChilies hbs
  cases dried_thai_chilies_disj with
  | inl h1 =>
      cases h1 with
      | inl hsp =>
          exact hnsp hsp
      | inr hmala =>
          have hsp' : Spicy DriedThaiChilies := mala_hotpots_spicy DriedThaiChilies hmala
          exact hnsp hsp'
  | inr hnotbs =>
      exact hnotbs hbs
</lean>

ANSWER: True
