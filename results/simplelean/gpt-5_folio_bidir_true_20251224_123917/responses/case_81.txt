<lean>
axiom obj : Type
axiom DriedThaiChilies : obj

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- No baked sweets are spicy.
axiom A1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom A2 : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom A3 : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom A4 : ∀ x : obj, FromBBM x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom A5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

theorem dtc_neither_bbm_nor_baked : ¬ FromBBM DriedThaiChilies ∧ ¬ BakedSweet DriedThaiChilies := by
  -- Reduce the given disjunction to (Spicy ∨ ¬ BakedSweet)
  have disj : Spicy DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies := by
    cases A5 with
    | inl hSpicy => exact Or.inl hSpicy
    | inr h =>
      cases h with
      | inl hMH => exact Or.inl (A3 DriedThaiChilies hMH)
      | inr hNotB => exact Or.inr hNotB
  -- Conclude ¬ BakedSweet
  have nb : ¬ BakedSweet DriedThaiChilies := by
    cases disj with
    | inl hSpicy =>
      intro hB
      have hNotSpicy : ¬ Spicy DriedThaiChilies := A1 DriedThaiChilies hB
      exact hNotSpicy hSpicy
    | inr hNotB => exact hNotB
  -- Conclude ¬ FromBBM using the fact that FromBBM → Cupcake → BakedSweet
  have nfrom : ¬ FromBBM DriedThaiChilies := by
    intro hF
    have hC : Cupcake DriedThaiChilies := A4 DriedThaiChilies hF
    have hB : BakedSweet DriedThaiChilies := A2 DriedThaiChilies hC
    exact nb hB
  exact And.intro nfrom nb
</lean>

ANSWER: True
