<lean>
axiom obj : Type
axiom Tom : obj

axiom WildTurkey : obj → Prop
axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Goulds : obj → Prop
axiom Merriams : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop

axiom Partition : ∀ x : obj, WildTurkey x → (Eastern x ∨ Osceola x ∨ Goulds x ∨ Merriams x ∨ RioGrande x ∨ Ocellated x)

axiom T_wild : WildTurkey Tom
axiom T_not_Eastern : ¬ Eastern Tom
axiom T_not_Osceola : ¬ Osceola Tom
axiom T_not_Goulds : ¬ Goulds Tom
axiom T_not_Merriams : ¬ Merriams Tom
axiom T_not_Rio : ¬ RioGrande Tom

theorem tom_is_ocellated : Ocellated Tom := by
  have h := Partition Tom T_wild
  cases h with
  | inl hE =>
      exact False.elim (T_not_Eastern hE)
  | inr h1 =>
      cases h1 with
      | inl hO =>
          exact False.elim (T_not_Osceola hO)
      | inr h2 =>
          cases h2 with
          | inl hG =>
              exact False.elim (T_not_Goulds hG)
          | inr h3 =>
              cases h3 with
              | inl hM =>
                  exact False.elim (T_not_Merriams hM)
              | inr h4 =>
                  cases h4 with
                  | inl hR =>
                      exact False.elim (T_not_Rio hR)
                  | inr hOc =>
                      exact hOc
</lean>

ANSWER: True
