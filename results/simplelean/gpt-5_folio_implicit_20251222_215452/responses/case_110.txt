/-
We formalize:
- U: universe of discourse
- F: fir tree predicate
- E: evergreen predicate
- W: object of worship predicate

Premises:
1) ∀ x, F x → E x        (All fir trees are evergreens)
2) ∃ x, W x ∧ F x        (Some objects of worship are fir trees)

Target claim to test:
T) ∃ x, E x ∧ ¬ W x      (Some evergreens are not objects of worship)
-/

/-- From the premises we can derive that some evergreen is an object of worship. -/
theorem exists_evergreen_object_of_worship
  {U : Type} {E F W : U → Prop}
  (h1 : ∀ x, F x → E x) (h2 : ∃ x, W x ∧ F x) :
  ∃ x, E x ∧ W x :=
by
  cases h2 with
  | intro x hx =>
    have hxF : F x := hx.right
    have hxW : W x := hx.left
    have hEx : E x := h1 x hxF
    exact ⟨x, And.intro hEx hxW⟩

/-
Show the target claim T is not a logical consequence of the premises by providing
a countermodel where the premises hold but T fails.
Let U = Unit, and let all three predicates E, F, W be constantly True.
-/
theorem target_not_valid_in_general :
  ¬ (∀ (U : Type) (E F W : U → Prop),
        (∀ x, F x → E x) →
        (∃ x, W x ∧ F x) →
        (∃ x, E x ∧ ¬ W x)) :=
by
  intro h
  -- Countermodel:
  let U0 := Unit
  let E0 : U0 → Prop := fun _ => True
  let F0 : U0 → Prop := fun _ => True
  let W0 : U0 → Prop := fun _ => True
  -- Premises hold in this model:
  have h1 : ∀ x : U0, F0 x → E0 x := by
    intro _ _; exact True.intro
  have h2 : ∃ x : U0, W0 x ∧ F0 x := by
    exact ⟨(), And.intro True.intro True.intro⟩
  -- Apply the assumed universal validity to get the target, which must fail:
  have hT : ∃ x : U0, E0 x ∧ ¬ W0 x := h U0 E0 F0 W0 h1 h2
  cases hT with
  | intro x hx =>
    -- But W0 x is True by definition, contradiction with ¬ W0 x.
    exact (hx.right True.intro)
