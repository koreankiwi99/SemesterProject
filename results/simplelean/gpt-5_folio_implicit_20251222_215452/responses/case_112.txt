-- We formalize the entities and predicates
universe u

constant Entity : Type u

constant Onate : Entity
constant PM : Entity        -- Picuris Mountains
constant HPM : Entity       -- Harding Pegmatite Mine

constant MountainRange : Entity → Prop
constant InTexas : Entity → Prop
constant InNewMexico : Entity → Prop
constant Mine : Entity → Prop
constant Donated : Entity → Prop
constant LocatedIn : Entity → Entity → Prop
constant Visited : Entity → Entity → Prop

-- Axioms extracted from the textual context
axiom ax1 : MountainRange PM
axiom ax2 : InNewMexico PM ∨ InTexas PM
axiom ax3 : Visited Onate PM
axiom ax4 : Mine HPM
axiom ax5 : LocatedIn HPM PM
axiom ax6 : Donated HPM
axiom ax7 : ∀ r, (MountainRange r ∧ InTexas r) → ¬ ∃ m, Mine m ∧ LocatedIn m r ∧ Donated m
-- "There are no mountain ranges in Texas that have mines that have been donated."

-- From the donation of HPM located in PM and ax7, PM cannot be in Texas.
lemma not_in_texas_PM : ¬ InTexas PM := by
  intro htx
  have ex : ∃ m, Mine m ∧ LocatedIn m PM ∧ Donated m := by
    exact ⟨HPM, And.intro ax4 (And.intro ax5 ax6)⟩
  have hnot := ax7 PM (And.intro ax1 htx)
  exact hnot ex

-- Hence, by the disjunction "in New Mexico or Texas", PM is in New Mexico.
lemma in_new_mexico_PM : InNewMexico PM := by
  have h := ax2
  cases h with
  | inl hnm => exact hnm
  | inr htx => exact False.elim (not_in_texas_PM htx)

-- The queried statement:
-- "Juan de Onate visited a mountain range in Texas."
def Claim : Prop := ∃ r, MountainRange r ∧ InTexas r ∧ Visited Onate r

-- We can show PM is not a witness, but we cannot derive Claim or its negation from the axioms.
lemma PM_not_witnesses_Claim : ¬ (MountainRange PM ∧ InTexas PM ∧ Visited Onate PM) := by
  intro h
  exact not_in_texas_PM h.2.1
