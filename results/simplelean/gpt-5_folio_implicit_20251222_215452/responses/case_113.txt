constant Mine : Entity → Prop

constant HardingPegmatiteMine : Entity

axiom HPM_donated : Donated HardingPegmatiteMine
axiom no_TX_range_with_donated_mine :
  ∀ r : Entity, MountainRange r → InTexas r → ¬ (∃ m : Entity, Mine m ∧ LocatedIn m r ∧ Donated m)

-- Target statement: "The Harding Pegmatite Mine is not located in a mountain range in New Mexico."
def Target : Prop :=
  ¬ (∃ r : Entity, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingPegmatiteMine r)

-- We prove the negation of the Target (i.e., the statement is false)
theorem not_Target : ¬ Target := by
  -- There exists a donated mine located in Picuris Mountains (namely HPM)
  have ex_donated_mine_in_PM :
      ∃ m : Entity, Mine m ∧ LocatedIn m PicurisMountains ∧ Donated m :=
    ⟨HardingPegmatiteMine,
      And.intro HPM_is_mine (And.intro HPM_located_in_PM HPM_donated)⟩
  -- Therefore, Picuris Mountains cannot be in Texas

    intro htx
    have h := no_TX_range_with_donated_mine PicurisMountains PM_is_range htx
    exact h ex_donated_mine_in_PM
  -- From "in NM or in TX" and "not in TX", deduce "in NM"

    PM_in_NM_or_TX.elim (fun hnm => hnm) (fun htx => False.elim (not_in_TX_PM htx))
  -- Exhibit a mountain range in New Mexico that HPM is located in: Picuris Mountains
  have exists_range_in_NM_with_HPM :

    ⟨PicurisMountains, And.intro PM_is_range (And.intro in_NM_PM HPM_located_in_PM)⟩
  -- Conclude the negation of Target
  intro hTarget
  exact hTarget exists_range_in_NM_with_HPM
