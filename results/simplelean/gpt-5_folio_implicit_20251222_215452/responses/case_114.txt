constant Taller : Person → Person → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom A1 : ∀ {x y : Person}, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom A2 : Man Michael ∧ ∀ y : Person, InMichaelClass y ∧ y ≠ Michael → Taller Michael y

-- Transitivity of taller.
axiom A3 : ∀ {x y z : Person}, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom A4 : Man Peter ∧ Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom A5 : ∀ x : Person, ¬ JumpsWhenShooting x → CanBlock Michael x

-- Michael cannot block Windy's shooting.
axiom A6 : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom A7 : ∀ x : Person, Shooter x ∧ JumpsWhenShooting x → GreatShooter x

-- A derived fact: Windy must jump when shooting.
theorem Windy_jumps : JumpsWhenShooting Windy :=
by
  classical
  by_contra h
  have : CanBlock Michael Windy := A5 Windy h
  exact A6 this

-- Target statement: Peter is shorter than a man in Michael's class.
def Target : Prop := ∃ m : Person, Man m ∧ InMichaelClass m ∧ Taller m Peter

-- From the given axioms (which include transitivity but no asymmetry or irreflexivity of Taller),
