<lean>
universe u

constant Person : Type u

-- Predicates
constant Man : Person → Prop
constant Taller : Person → Person → Prop
constant CanBlock : Person → Person → Prop
constant JumpWhenShooting : Person → Prop
constant Shooter : Person → Prop
constant GreatShooter : Person → Prop

-- Individuals
constant Michael : Person
constant Peter : Person
constant Windy : Person

-- Assumptions from the text
axiom Taller_man_blocks :
  ∀ x y : Person, Man x → Man y → Taller x y → CanBlock x y

axiom Michael_is_man : Man Michael

axiom taller_trans :
  ∀ x y z : Person, Taller x y → Taller y z → Taller x z

axiom Peter_is_man : Man Peter
axiom Peter_taller_than_Michael : Taller Peter Michael

axiom Michael_blocks_if_no_jump :
  ∀ p : Person, ¬ JumpWhenShooting p → CanBlock Michael p

axiom Michael_cannot_block_Windy :
  ¬ CanBlock Michael Windy

axiom great_if_can_jump :
  ∀ p : Person, Shooter p → JumpWhenShooting p → GreatShooter p

axiom Windy_is_shooter : Shooter Windy

open Classical

theorem Windy_is_great_shooter : GreatShooter Windy := by
  classical
  have hJump : JumpWhenShooting Windy := by
    by_contra hNotJump
    exact Michael_cannot_block_Windy (Michael_blocks_if_no_jump Windy hNotJump)
  exact great_if_can_jump Windy Windy_is_shooter hJump
</lean>
ANSWER: True

constant Person : Type

-- Predicates
constant Man : Person → Prop
constant Taller : Person → Person → Prop
constant CanBlock : Person → Person → Prop
constant JumpWhenShooting : Person → Prop
constant Shooter : Person → Prop
constant GreatShooter : Person → Prop

-- Individuals
constant Michael : Person
constant Peter : Person
constant Windy : Person

-- Assumptions from the text
axiom Taller_man_blocks :
  ∀ x y : Person, Man x → Man y → Taller x y → CanBlock x y

axiom Michael_is_man : Man Michael

axiom taller_trans :
  ∀ x y z : Person, Taller x y → Taller y z → Taller x z

axiom Peter_is_man : Man Peter
axiom Peter_taller_than_Michael : Taller Peter Michael

axiom Michael_blocks_if_no_jump :
  ∀ p : Person, ¬ JumpWhenShooting p → CanBlock Michael p

axiom Michael_cannot_block_Windy :
  ¬ CanBlock Michael Windy

axiom great_if_can_jump :
  ∀ p : Person, Shooter p → JumpWhenShooting p → GreatShooter p

axiom Windy_is_shooter : Shooter Windy

theorem Windy_is_great_shooter : GreatShooter Windy :=
  let hJump : JumpWhenShooting Windy :=
    Or.elim (Classical.em (JumpWhenShooting Windy))
      (fun hj => hj)
      (fun hNot =>
        False.elim (Michael_cannot_block_Windy (Michael_blocks_if_no_jump Windy hNot)))
  great_if_can_jump Windy Windy_is_shooter hJump
</lean>
ANSWER: True
