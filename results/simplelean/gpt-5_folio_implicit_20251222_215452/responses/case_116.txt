constant Taller : Person → Person → Prop
constant CanBlock : Person → Person → Prop
constant JumpsWhenShooting : Person → Prop
constant GreatShooter : Person → Prop
constant InMichaelsClass : Person → Prop

-- Axioms from the textual context

-- 1. If a man is taller than another man, the taller one can block the other's shooting.
axiom A1 : ∀ {x y : Person}, Man x → Man y → Taller x y → CanBlock x y

-- 2. Michael is a man who is taller than everyone else in his class.
axiom A2_manMichael : Man Michael
axiom A2_tallerClass : ∀ {y : Person}, InMichaelsClass y → Taller Michael y

-- 3. Transitivity of taller: if x>y and y>z then x>z.
axiom A3_transTaller : ∀ {x y z : Person}, Taller x y → Taller y z → Taller x z

-- 4. Peter is a man who is taller than Michael.
axiom A4_manPeter : Man Peter
axiom A4_taller : Taller Peter Michael

-- 5. Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom A5_M_block_nonjump : ∀ {p : Person}, ¬ JumpsWhenShooting p → CanBlock Michael p

-- 6. Michael cannot block Windy's shooting.
axiom A6_M_cannot_block_Windy : ¬ CanBlock Michael Windy

-- 7. Every shooter who can jump when shooting is a great shooter.
axiom A7_jumpers_great : ∀ {s : Person}, JumpsWhenShooting s → GreatShooter s

-- Some derivable facts (do not settle the main question)
