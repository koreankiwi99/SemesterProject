axiom GreenwichIsland_is_island : Island GreenwichIsland
axiom DeceptionIsland_is_island : Island DeceptionIsland

-- Note: As stated in the prompt, we take this as given even if it is geographically unusual.
axiom Antarctica_in_SouthShetlands : LocatedIn Antarctica SouthShetlandIslands

-- Transitivity of location
axiom located_trans : ∀ {A B C : Place}, LocatedIn A B → LocatedIn B C → LocatedIn A C

-- A small derived fact: Barutin Cove is in the South Shetland Islands
theorem BarutinCove_in_SouthShetlands :
  LocatedIn BarutinCove SouthShetlandIslands :=
  located_trans BarutinCove_on_SnowIsland SnowIsland_in_SouthShetlands

-- The target statement:
-- "Barutin Cove is named after all islands in Antarctica."
-- We formalize this as: for every place x, if x is an island and x is located in Antarctica,
-- then Barutin Cove is named after x.

  ∀ x : Place, Island x → LocatedIn x Antarctica → NamedAfter BarutinCove x
