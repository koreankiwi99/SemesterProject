universe u

constant Place : Type u
constant LocatedIn : Place → Place → Prop

-- Entities
constant BarutinCove : Place
constant SnowIsland : Place
constant GreenwichIsland : Place
constant DeceptionIsland : Place
constant SouthShetlandIslands : Place
constant Antarctica : Place

-- Axioms from the text
axiom located_in_trans :
  ∀ {A B C : Place}, LocatedIn A B → LocatedIn B C → LocatedIn A C

axiom Barutin_on_Snow : LocatedIn BarutinCove SnowIsland
axiom Snow_in_SouthShetland : LocatedIn SnowIsland SouthShetlandIslands
axiom Greenwich_in_SouthShetland : LocatedIn GreenwichIsland SouthShetlandIslands
axiom Deception_in_SouthShetland : LocatedIn DeceptionIsland SouthShetlandIslands
axiom Antarctica_in_SouthShetland : LocatedIn Antarctica SouthShetlandIslands

-- Derivable consequence: Barutin Cove is in the South Shetland Islands.
lemma Barutin_in_SouthShetland : LocatedIn BarutinCove SouthShetlandIslands :=
  located_in_trans Barutin_on_Snow Snow_in_SouthShetland

-- The query statement: "Barutin Cove is not located in Antarctica."
def Query : Prop := ¬ LocatedIn BarutinCove Antarctica

From the given axioms we cannot derive either LocatedIn BarutinCove Antarctica
