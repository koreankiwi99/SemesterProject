-- We formalize the predicates and entities.
-- Car is the domain of discourse; john is John's car.
-- Predicates:
--   TransportMultiple: can transport multiple passengers
--   OneSeater: is a one-seater
--   TeslaModel3: is a Tesla Model 3
--   SingleSeatElectricCar: is a single-seat electric car
--   SoloEV: is a Solo EV
namespace Problem

variable (Car : Type u)
variable (TransportMultiple OneSeater TeslaModel3 SingleSeatElectricCar SoloEV : Car → Prop)
variable (john : Car)

-- Axioms from the textual context
variable (ax1 : ∀ x, TransportMultiple x → ¬ OneSeater x)                           -- If can transport multiple, then not one-seater
variable (ax2 : ∀ x, TeslaModel3 x → TransportMultiple x)                           -- All Tesla Model 3 can transport multiple
variable (ax3 : ∀ x, SingleSeatElectricCar x → OneSeater x)                         -- All single-seat electric cars are one-seaters
variable (ax4 : ∀ x, SoloEV x → SingleSeatElectricCar x)                            -- All Solo EV cars are single-seat electric cars
variable (ax5 : (¬ TeslaModel3 john) → ((¬ TeslaModel3 john) ∨ SingleSeatElectricCar john)) -- Given tautological conditional

-- The query statement
def Statement : Prop := TeslaModel3 john

-- Some derived consequences (sound with respect to the axioms)
lemma tesla_implies_not_one (x : Car) : TeslaModel3 x → ¬ OneSeater x := by
  intro hT
  exact ax1 x (ax2 x hT)

lemma no_tesla_and_single (x : Car) : ¬ (TeslaModel3 x ∧ SingleSeatElectricCar x) := by
  intro h
  have hT : TeslaModel3 x := h.left
  have hS : SingleSeatElectricCar x := h.right
  have hNotOne : ¬ OneSeater x := tesla_implies_not_one Car TransportMultiple OneSeater TeslaModel3 SingleSeatElectricCar SoloEV john ax1 ax2 ax3 ax4 ax5 x hT
  have hOne : OneSeater x := ax3 x hS
  exact hNotOne hOne

end Problem
