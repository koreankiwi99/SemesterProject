-- We formalize the domain, predicates, and premises, and derive that Rock has wings.
theorem Rock_has_wings
  (U : Type)
  (Pet Mammal Animal Monkey Bird CanFly HasWings CanBreathe : U → Prop)
  (Rock : U)
  (H1 : ∀ x, Pet x ∧ Mammal x → Animal x)                    -- All of Peter's pets that are mammals are animals.
  (H2 : ∀ x, Monkey x → Mammal x)                             -- All monkeys are mammals.
  (H3 : ∀ x, Pet x → (Monkey x ∨ Bird x))                     -- Peter's pets are all either monkeys or birds.
  (H4 : ∀ x, Pet x ∧ Bird x → CanFly x)                       -- Peter's birds can fly.
  (H5 : ∀ x, Pet x ∧ Animal x → CanBreathe x)                 -- All animals that are Peter's pets can breathe.
  (H6 : ∀ x, Pet x ∧ CanFly x → HasWings x)                   -- If Peter's pet can fly, then it has wings.
  (H7 : Pet Rock)                                             -- Rock is Peter's pet.
  (H8 : CanFly Rock ∨ (Bird Rock) ∨ (¬ CanBreathe Rock))      -- Rock can fly, or Rock is a bird, or Rock cannot breathe.
  : HasWings Rock :=
by
  -- If Rock can fly, then Rock has wings.
  have wingsFromFly : CanFly Rock → HasWings Rock :=
    fun hf => H6 Rock ⟨H7, hf⟩
  -- If Rock is a bird (and a pet), Rock can fly; hence Rock has wings.
  have wingsFromBird : Bird Rock → HasWings Rock :=
    fun hb => wingsFromFly (H4 Rock ⟨H7, hb⟩)
  -- Even if Rock cannot breathe, from being a pet we know Rock is a monkey or a bird.
  -- In the monkey case we derive a contradiction (hence anything, in particular wings);
  -- in the bird case we get wings as above.
  have wingsFromNotBreathe : ¬ CanBreathe Rock → HasWings Rock :=
    fun hnb =>
      by
        have hmb : Monkey Rock ∨ Bird Rock := H3 Rock H7
        cases hmb with
        | inl hmon =>
            have hmamm : Mammal Rock := H2 Rock hmon
            have hanim : Animal Rock := H1 Rock ⟨H7, hmamm⟩
            have hbr : CanBreathe Rock := H5 Rock ⟨H7, hanim⟩
            exact False.elim (hnb hbr)
        | inr hbird =>
            exact wingsFromBird hbird
  -- Case analysis on the given disjunction for Rock.
  exact
    Or.elim H8
      (fun hf => wingsFromFly hf)
      (fun rest =>
        Or.elim rest
          (fun hb => wingsFromBird hb)
          (fun hnb => wingsFromNotBreathe hnb))
</lean>
