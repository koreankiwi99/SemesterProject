-- We formalize the given predicates and entities.

universe u

constant Thing : Type u

constant Hydrocarbon : Thing → Prop
constant Organic : Thing → Prop
constant Alkane : Thing → Prop
constant Chem : Thing → Prop
constant ContainsCarbon : Thing → Prop
constant OnlyOneElement : Thing → Prop

constant mixture : Thing

-- Premises from the textual context
axiom all_hydro_are_organic : ∀ x, Hydrocarbon x → Organic x
axiom all_alkanes_are_hydro : ∀ x, Alkane x → Hydrocarbon x
axiom all_organic_are_chem : ∀ x, Organic x → Chem x
axiom all_organic_contain_carbon : ∀ x, Organic x → ContainsCarbon x
axiom no_chem_is_only_one : ∀ x, Chem x → ¬ OnlyOneElement x
axiom mixture_either_both_or_neither :
  (Chem mixture ∧ OnlyOneElement mixture) ∨ (¬ Chem mixture ∧ ¬ OnlyOneElement mixture)

-- Target statement:
-- If the mixture contains only one element or contains carbon,
-- then the mixture is neither a chemical compound nor an alkane.
theorem mixture_statement :
  (OnlyOneElement mixture ∨ ContainsCarbon mixture) →
  (¬ Chem mixture ∧ ¬ Alkane mixture) := by
  intro h
  -- First, show that the mixture is not a chemical compound
  have h_not_chem : ¬ Chem mixture := by
    intro hchem
    have h_not_one : ¬ OnlyOneElement mixture := no_chem_is_only_one mixture hchem
    cases mixture_either_both_or_neither with
    | inl hboth =>
        exact h_not_one hboth.right
    | inr hneither =>
        exact hneither.left hchem
  -- Next, show that the mixture is not an alkane
  have h_not_alkane : ¬ Alkane mixture := by
    intro halk
    have hhyd : Hydrocarbon mixture := all_alkanes_are_hydro mixture halk
    have horg : Organic mixture := all_hydro_are_organic mixture hhyd
    have hchem : Chem mixture := all_organic_are_chem mixture horg
    exact h_not_chem hchem
  exact And.intro h_not_chem h_not_alkane
</lean>

-- Translate the problem into Lean 4 and prove the target statement.

universe u

constant (Thing : Type u)

constant (Hydrocarbon : Thing → Prop)
constant (Organic : Thing → Prop)
constant (Alkane : Thing → Prop)
constant (Chem : Thing → Prop)
constant (ContainsCarbon : Thing → Prop)
constant (OnlyOneElement : Thing → Prop)

constant (mixture : Thing)

-- Premises
axiom (all_hydro_are_organic : ∀ x, Hydrocarbon x → Organic x)
axiom (all_alkanes_are_hydro : ∀ x, Alkane x → Hydrocarbon x)
axiom (all_organic_are_chem : ∀ x, Organic x → Chem x)
axiom (all_organic_contain_carbon : ∀ x, Organic x → ContainsCarbon x)
axiom (no_chem_is_only_one : ∀ x, Chem x → ¬ OnlyOneElement x)
axiom (mixture_either_both_or_neither :
  (Chem mixture ∧ OnlyOneElement mixture) ∨ (¬ Chem mixture ∧ ¬ OnlyOneElement mixture))

-- From the "either both or neither" premise we can derive an equivalence.
lemma chem_iff_one : Chem mixture ↔ OnlyOneElement mixture := by
  have h := mixture_either_both_or_neither
  cases h with
  | inl hboth =>
      exact Iff.intro
        (fun _ => hboth.right)
        (fun _ => hboth.left)
  | inr hneither =>
      exact Iff.intro
        (fun hchem => False.elim (hneither.left hchem))
        (fun hone  => False.elim (hneither.right hone))

-- Target theorem
theorem mixture_statement :
  (OnlyOneElement mixture ∨ ContainsCarbon mixture) →
  (¬ Chem mixture ∧ ¬ Alkane mixture) := by
  intro _
  -- First, show mixture is not a chemical compound.
  have h_not_chem : ¬ Chem mixture := by
    intro hchem
    have hone : OnlyOneElement mixture := (chem_iff_one).mp hchem
    have hno : ¬ OnlyOneElement mixture := no_chem_is_only_one mixture hchem
    exact hno hone
  -- Then, mixture is not an alkane (since alkanes are hydrocarbons, organic, hence chemical).
  have h_not_alkane : ¬ Alkane mixture := by
    intro halk
    have hhyd : Hydrocarbon mixture := all_alkanes_are_hydro mixture halk
    have horg : Organic mixture := all_hydro_are_organic mixture hhyd
    have hchem : Chem mixture := all_organic_are_chem mixture horg
    exact h_not_chem hchem
  exact And.intro h_not_chem h_not_alkane
</lean>
