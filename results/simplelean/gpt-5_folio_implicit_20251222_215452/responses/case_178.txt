<lean>
universe u

-- Domain of discourse
variable (U : Type u)

-- Predicates
variable (Human Horse HasHooves : U → Prop)

-- The statement under evaluation: "Some humans are horses."
def SomeHumansAreHorses : Prop := ∃ x : U, Human x ∧ Horse x

-- Premises:
-- 1) All horses have hooves.
-- 2) No humans have hooves.
-- Conclusion: From these, it follows that no human is a horse, hence
--             "Some humans are horses" is false.
theorem no_human_is_horse
  (all_horses_have_hooves : ∀ x : U, Horse x → HasHooves x)
  (no_humans_have_hooves : ∀ x : U, Human x → ¬ HasHooves x) :
  ¬ SomeHumansAreHorses U Human Horse := by
  intro h
  -- h : ∃ x, Human x ∧ Horse x
  apply Exists.elim h
  intro x hx
  -- hx : Human x ∧ Horse x
  have hxHuman : Human x := And.left hx
  have hxHorse : Horse x := And.right hx
  have hh : HasHooves x := all_horses_have_hooves x hxHorse
  have nh : ¬ HasHooves x := no_humans_have_hooves x hxHuman
  exact nh hh
</lean>
ANSWER: False
