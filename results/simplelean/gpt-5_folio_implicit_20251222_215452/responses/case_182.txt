<lean>
universe u

constant Person : Type u
constant Nice : Person → Prop
constant Mean : Person → Prop
constant Grumpy : Person → Prop
constant Loves : Person → Prop
constant PetOwner : Person → Prop
constant Tom : Person

-- Premises
axiom h1 : ∀ x : Person, Nice x → ¬ Mean x
axiom h2 : ∃ x : Person, Grumpy x ∧ Mean x
axiom h3 : ∀ x : Person, Loves x → Nice x
axiom h4 : ∀ x : Person, PetOwner x → Loves x
axiom h5 : PetOwner Tom

-- Conclusion: Tom is not both grumpy and mean to animals.
theorem tom_not_grumpy_and_mean : ¬ (Grumpy Tom ∧ Mean Tom) := by
  have hL : Loves Tom := h4 Tom h5
  have hN : Nice Tom := h3 Tom hL
  have hNM : ¬ Mean Tom := h1 Tom hN
  intro h
  exact hNM h.right
</lean>
ANSWER: True
