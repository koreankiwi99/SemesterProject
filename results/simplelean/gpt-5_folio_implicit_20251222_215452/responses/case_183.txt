universe u

constant Person : Type u
constant Nice Mean Grumpy Lover Owner : Person → Prop
constant Tom : Person

-- Axioms from the textual context
axiom H1 : ∀ x : Person, Nice x → ¬ (Mean x)                 -- No one nice to animals is also mean to animals
axiom H2 : ∃ x : Person, Grumpy x ∧ Mean x                   -- Some grumpy people are mean to animals
axiom H3 : ∀ x : Person, Lover x → Nice x                    -- All animal lovers are nice to animals
axiom H4 : ∀ x : Person, Owner x → Lover x                   -- All pet owners love animals
axiom H5 : Owner Tom                                         -- Tom is a pet owner

-- Useful derived facts about Tom
lemma loverTom : Lover Tom :=
  H4 Tom H5

lemma niceTom : Nice Tom :=
  H3 Tom loverTom

lemma notMeanTom : ¬ (Mean Tom) :=
  H1 Tom niceTom

lemma antecedentTrue : ¬ (Grumpy Tom ∧ Mean Tom) := by
  intro h
  exact notMeanTom h.right

lemma notConclusion : ¬ ((¬ (Grumpy Tom)) ∧ (¬ (Lover Tom))) := by
  intro h
  exact h.right loverTom

-- The statement to assess:
def Statement : Prop :=
  (¬ (Grumpy Tom ∧ Mean Tom)) → ((¬ (Grumpy Tom)) ∧ (¬ (Lover Tom)))

-- We prove the negation of the statement
theorem Statement_false : ¬ Statement := by
  intro himp
  have concl : (¬ (Grumpy Tom)) ∧ (¬ (Lover Tom)) := himp antecedentTrue
  exact notConclusion concl

section

constant Person : Type
constant Nice : Person → Prop
constant Mean : Person → Prop
constant Grumpy : Person → Prop
constant Lover : Person → Prop
constant Owner : Person → Prop
constant Tom : Person

-- Axioms from the textual context
axiom H1 : forall x : Person, Nice x → Mean x → False                 -- No one nice to animals is also mean to animals
axiom H2 : Exists (fun x : Person => And (Grumpy x) (Mean x))          -- Some grumpy people are mean to animals
axiom H3 : forall x : Person, Lover x → Nice x                         -- All animal lovers are nice to animals
axiom H4 : forall x : Person, Owner x → Lover x                        -- All pet owners love animals
axiom H5 : Owner Tom                                                   -- Tom is a pet owner

-- Useful derived facts about Tom
lemma loverTom : Lover Tom :=
  H4 Tom H5

lemma niceTom : Nice Tom :=
  H3 Tom loverTom

lemma notMeanTom : Mean Tom → False :=
  H1 Tom niceTom

lemma antecedentTrue : And (Grumpy Tom) (Mean Tom) → False :=
  fun h => notMeanTom h.right

lemma notConclusion : And (Grumpy Tom → False) (Lover Tom → False) → False :=
  fun h => h.right loverTom

-- The statement to assess:
def Statement : Prop :=
  (And (Grumpy Tom) (Mean Tom) → False) → And (Grumpy Tom → False) (Lover Tom → False)

-- We prove the negation of the statement
theorem Statement_false : Statement → False := by
  intro himp
  have concl : And (Grumpy Tom → False) (Lover Tom → False) := himp antecedentTrue
  exact notConclusion concl

end
