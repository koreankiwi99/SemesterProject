universe u
constant Entity : Type u

-- Predicates
constant Podcast : Entity → Prop
constant Novel : Entity → Prop
constant WrittenBy : Entity → Entity → Prop
constant NovelWriter : Entity → Prop
constant American : Entity → Prop
constant Writer : Entity → Prop
constant Created : Entity → Entity → Prop
constant AmericanCity : Entity → Prop
constant BornIn : Entity → Entity → Prop

-- Entities
constant DaniShapiro : Entity
constant FamilyHistory : Entity
constant FamilySecrets : Entity
constant Boston : Entity

-- Axioms from the text
axiom podcast_not_novel : ∀ x, Podcast x → ¬ Novel x
axiom born_in_american_city_implies_american : ∀ p c, (BornIn p c ∧ AmericanCity c) → American p
axiom novel_written_implies_novel_writer : ∀ b p, (Novel b ∧ WrittenBy b p) → NovelWriter p
axiom dani_is_american_writer : American DaniShapiro ∧ Writer DaniShapiro
axiom family_history_written_by_dani : WrittenBy FamilyHistory DaniShapiro
axiom family_history_is_novel : Novel FamilyHistory
axiom dani_created_podcast_family_secrets : Podcast FamilySecrets ∧ Created DaniShapiro FamilySecrets
axiom boston_is_american_city : AmericanCity Boston

-- Goal: Dani Shapiro is a novel writer.
theorem dani_is_novel_writer : NovelWriter DaniShapiro := by
  have hnov : Novel FamilyHistory := family_history_is_novel
  have hwb : WrittenBy FamilyHistory DaniShapiro := family_history_written_by_dani
  exact novel_written_implies_novel_writer FamilyHistory DaniShapiro ⟨hnov, hwb⟩
