-- Predicates

constant AmericanNational : Person → Prop
constant CanDoJumpShots : Person → Prop
constant LeapsStraightIntoTheAir : Person → Prop
constant ActivatesLegMuscles : Person → Prop

-- Premises from the text
axiom some_PBP_not_American :
  ∃ x : Person, ProfessionalBasketballPlayer x ∧ ¬ AmericanNational x

axiom all_PBP_can_jump_shots :
  ∀ x : Person, ProfessionalBasketballPlayer x → CanDoJumpShots x

axiom jump_shots_implies_leap :
  ∀ x : Person, CanDoJumpShots x → LeapsStraightIntoTheAir x

axiom leap_implies_activate_legs :
  ∀ x : Person, LeapsStraightIntoTheAir x → ActivatesLegMuscles x

axiom yuri_not_activate_legs :
  ¬ ActivatesLegMuscles Yuri

-- Deductions
theorem yuri_not_leap : ¬ LeapsStraightIntoTheAir Yuri := by
  intro hleap
  have hact : ActivatesLegMuscles Yuri := leap_implies_activate_legs Yuri hleap
  exact yuri_not_activate_legs hact

theorem yuri_not_can_jump_shots : ¬ CanDoJumpShots Yuri := by
  intro hjump
  have hleap : LeapsStraightIntoTheAir Yuri := jump_shots_implies_leap Yuri hjump
  exact yuri_not_leap hleap

theorem yuri_not_PBP : ¬ ProfessionalBasketballPlayer Yuri := by
  intro hP
  have hJ : CanDoJumpShots Yuri := all_PBP_can_jump_shots Yuri hP
  exact yuri_not_can_jump_shots hJ

-- Note: From the premises, we can derive that Yuri is not a professional basketball player,
-- but we cannot derive whether Yuri is (or is not) an American national.
-- Hence, the status of `AmericanNational Yuri` is undetermined from the given information.
