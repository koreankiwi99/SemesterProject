<lean>
universe u

constant Person : Type u
constant American : Person → Prop
constant ProBasketballPlayer : Person → Prop
constant CanJumpShot : Person → Prop
constant LeapsStraight : Person → Prop
constant ActivatesLegMuscles : Person → Prop
constant Yuri : Person

-- Assumptions from the text
axiom some_nonAmerican_pbp : ∃ x, ProBasketballPlayer x ∧ ¬ American x
axiom all_pbp_can_jump_shot : ∀ x, ProBasketballPlayer x → CanJumpShot x
axiom jump_shot_implies_leap : ∀ x, CanJumpShot x → LeapsStraight x
axiom leap_implies_activate : ∀ x, LeapsStraight x → ActivatesLegMuscles x
axiom yuri_not_activate : ¬ ActivatesLegMuscles Yuri

-- Target: Yuri is not an American professional basketball player
theorem yuri_not_american_pbp : ¬ (American Yuri ∧ ProBasketballPlayer Yuri) := by
  intro h
  have hyPBP : ProBasketballPlayer Yuri := And.right h
  have hyJump : CanJumpShot Yuri := all_pbp_can_jump_shot Yuri hyPBP
  have hyLeap : LeapsStraight Yuri := jump_shot_implies_leap Yuri hyJump
  have hyAct : ActivatesLegMuscles Yuri := leap_implies_activate Yuri hyLeap
  exact yuri_not_activate hyAct
</lean>
ANSWER: True
