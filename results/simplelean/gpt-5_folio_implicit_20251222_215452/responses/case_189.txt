universe u

constant Person : Type u

-- Predicates:
-- P: is a professional basketball player
-- A: is an American national
-- J: can do jump shots
-- L: leaps straight into the air
-- M: activates their leg muscles
constants (P A J L M : Person → Prop)
constant yuri : Person

-- Premises
axiom some_P_not_A : ∃ x : Person, P x ∧ ¬ A x
axiom P_implies_J : ∀ x : Person, P x → J x
axiom J_implies_L : ∀ x : Person, J x → L x
axiom L_implies_M : ∀ x : Person, L x → M x
axiom not_M_yuri : ¬ M yuri

-- Target statement: "Yuri is an American professional basketball player."
def Claim : Prop := A yuri ∧ P yuri

-- From the premises, we can refute the claim.
theorem claim_false : ¬ Claim := by
  -- derive that any professional basketball player activates leg muscles
  have PMP : ∀ x : Person, P x → M x := by
    intro x hxP
    have hxJ : J x := P_implies_J x hxP
    have hxL : L x := J_implies_L x hxJ
    exact L_implies_M x hxL
  -- so Yuri cannot be a professional basketball player
  have not_P_yuri : ¬ P yuri := by
    intro hPy
    have hMy : M yuri := PMP yuri hPy
    exact not_M_yuri hMy
  -- therefore, Yuri is not an American professional basketball player
  intro hClaim
  exact not_P_yuri hClaim.right
