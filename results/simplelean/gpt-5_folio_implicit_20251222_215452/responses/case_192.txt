<lean>
universe u

section

variable (Event : Type u)
variable (Happy Sad : Event → Prop)

/-
Premises:
1) Every event is either happy or sad, but not both (exclusive or).
2) There exists at least one happy event.
-/
variable (either_exclusive : ∀ e : Event, (Happy e ∧ ¬ Sad e) ∨ (Sad e ∧ ¬ Happy e))
variable (exists_happy : ∃ e : Event, Happy e)

/- The statement in question: "All events are sad." -/
def AllEventsAreSad : Prop := ∀ e : Event, Sad e

/- We prove the negation of the statement under the given premises. -/
theorem not_all_events_are_sad : ¬ AllEventsAreSad Event Happy Sad := by
  intro hAllSad
  -- pick a happy event
  have ⟨w, hwHappy⟩ := exists_happy
  -- use exclusivity at w
  have h := either_exclusive w
  cases h with
  | inl h1 =>
    -- h1: Happy w ∧ ¬ Sad w
    have hNotSad : ¬ Sad w := h1.right
    exact hNotSad (hAllSad w)
  | inr h2 =>
    -- h2: Sad w ∧ ¬ Happy w contradicts hwHappy
    exact h2.right hwHappy

end
</lean>
ANSWER: False
