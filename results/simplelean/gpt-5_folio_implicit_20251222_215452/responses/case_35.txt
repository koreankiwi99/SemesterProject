constant Entity : Type u

constant IsStable : Entity → Prop
constant Includes : Entity → Entity → Prop
constant Feud : Entity → Entity → Prop

constant DiamondMine : Entity
constant Imperium : Entity
constant IvyNile : Entity

axiom hStable : IsStable DiamondMine
axiom hIncl : Includes DiamondMine IvyNile
axiom hFeud : Feud Imperium DiamondMine

def Statement : Prop :=
  ¬ ∃ S : Entity, IsStable S ∧ Includes S IvyNile ∧ Feud Imperium S

theorem negStatement : ¬ Statement := by
  unfold Statement
  intro h
  have ex : ∃ S : Entity, IsStable S ∧ Includes S IvyNile ∧ Feud Imperium S := by
    refine ⟨DiamondMine, ?_⟩
    exact And.intro hStable (And.intro hIncl hFeud)
  exact h ex
