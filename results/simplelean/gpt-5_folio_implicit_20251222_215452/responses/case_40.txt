constant Adores : Thing → Prop        -- "Max adores"
constant DesignByMax : Thing → Prop   -- "a design by Max"

-- Axioms from the textual context
axiom h1 : ∀ x : Thing, (ZahaStyle x ∧ Adores x) → Interesting x
axiom h2 : ∀ x : Thing, (Brutalist x ∧ Adores x) → ¬ Interesting x
axiom h3 : ∀ x : Thing, Adores x → (ZahaStyle x ∨ KellyStyle x)
axiom h4 : ∀ x : Thing, (KellyStyle x ∧ Adores x) → Evocative x
axiom h5 : ∀ x : Thing, (KellyStyle x ∧ Adores x) → Dreamy x
axiom h6 : ∀ x : Thing, (DesignByMax x ∧ Adores x ∧ Interesting x) → (Brutalist x ∧ Evocative x)

-- A derived fact: no design by Max that he adores can be interesting (otherwise contradiction with h2 via h6).
theorem no_interesting_adored_design_by_max :
  ∀ x : Thing, (DesignByMax x ∧ Adores x ∧ Interesting x) → False :=
by
  intro x hx
  have hbe : Brutalist x ∧ Evocative x := h6 x hx
  have hb : Brutalist x := hbe.left
  have ha : Adores x := hx.right.left
  have hi : Interesting x := hx.right.right
  have hnot : ¬ Interesting x := h2 x ⟨hb, ha⟩
  exact hnot hi
