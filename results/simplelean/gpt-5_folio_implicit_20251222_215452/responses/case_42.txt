<lean>
namespace RG2022

constant Person : Type
constant Coco : Person

constant RankedHighly : Person → Prop
constant MostActive : Person → Prop
constant LostToIga : Person → Prop
constant LostToRafa : Person → Prop
constant FemaleRG : Person → Prop
constant MaleRG : Person → Prop
constant AtRG : Person → Prop

axiom ranked_highly_implies_most_active : ∀ x, RankedHighly x → MostActive x
axiom lost_to_iga_implies_ranked_highly : ∀ x, LostToIga x → RankedHighly x
axiom femaleRG_lost_to_iga : ∀ x, FemaleRG x → LostToIga x
axiom atRG_female_or_male : ∀ x, AtRG x → (FemaleRG x ∨ MaleRG x)
axiom maleRG_lost_to_rafa : ∀ x, MaleRG x → LostToRafa x
axiom coco_high_or_rafa_implies_not_male : (RankedHighly Coco ∨ LostToRafa Coco) → ¬ (MaleRG Coco)
axiom coco_atRG : AtRG Coco

theorem coco_most_active : MostActive Coco := by
  have hdisj : FemaleRG Coco ∨ MaleRG Coco := atRG_female_or_male Coco coco_atRG
  have hfemale : FemaleRG Coco := by
    cases hdisj with
    | inl hf => exact hf
    | inr hm =>
      have hrafa : LostToRafa Coco := maleRG_lost_to_rafa Coco hm
      have hdh : RankedHighly Coco ∨ LostToRafa Coco := Or.inr hrafa
      have hnotmale : ¬ MaleRG Coco := coco_high_or_rafa_implies_not_male hdh
      exact False.elim (hnotmale hm)
  have hIga : LostToIga Coco := femaleRG_lost_to_iga Coco hfemale
  have hrank : RankedHighly Coco := lost_to_iga_implies_ranked_highly Coco hIga
  exact ranked_highly_implies_most_active Coco hrank

end RG2022
</lean>
