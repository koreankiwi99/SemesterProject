-- We formalize the domain and predicates, and encode the premises.

structure World where
  U : Type
  PrintedText : U → Prop
  Analog : U → Prop
  Digital : U → Prop
  Streaming : U → Prop
  Hardcover : U → Prop
  n1984 : U
  noDigitalAreAnalog : ∀ x, Digital x → ¬ Analog x
  printedIsAnalog : ∀ x, PrintedText x → Analog x
  streamingIsDigital : ∀ x, Streaming x → Digital x
  hardcoverIsPrinted : ∀ x, Hardcover x → PrintedText x
  hyp1984 : Streaming n1984 → Hardcover n1984

-- From the premises we can derive: if 1984 is streaming, then it is printed.
theorem streaming_implies_printed (w : World) :
  w.Streaming w.n1984 → w.PrintedText w.n1984 :=
by
  intro hs
  have hh : w.Hardcover w.n1984 := w.hyp1984 hs
  exact w.hardcoverIsPrinted _ hh
