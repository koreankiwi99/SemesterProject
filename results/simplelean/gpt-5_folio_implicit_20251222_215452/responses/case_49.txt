constant Obj : Type

constant DigitalMedia : Obj → Prop
constant AnalogMedia : Obj → Prop
constant PrintedText : Obj → Prop
constant StreamingService : Obj → Prop
constant HardcoverBook : Obj → Prop

constant N1984 : Obj

axiom no_digital_are_analog : ∀ x : Obj, DigitalMedia x → ¬ AnalogMedia x
axiom printed_are_analog : ∀ x : Obj, PrintedText x → AnalogMedia x
axiom streaming_are_digital : ∀ x : Obj, StreamingService x → DigitalMedia x
axiom hardcover_implies_printed : ∀ x : Obj, HardcoverBook x → PrintedText x
axiom N1984_streaming_implies_hardcover : StreamingService N1984 → HardcoverBook N1984

  have hHB : HardcoverBook N1984 := N1984_streaming_implies_hardcover hS
  have hPT : PrintedText N1984 := hardcover_implies_printed N1984 hHB
  have hA : AnalogMedia N1984 := printed_are_analog N1984 hPT
  have hD : DigitalMedia N1984 := streaming_are_digital N1984 hS
  have hNotA : ¬ AnalogMedia N1984 := no_digital_are_analog N1984 hD
  exact hNotA hA
