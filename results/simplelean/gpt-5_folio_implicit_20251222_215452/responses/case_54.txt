<lean>
namespace Reasoning

constant Entity : Type
constant Team : Type

constant RobertLewandowski : Entity
constant BayernMunchen : Team

constant Striker : Entity → Prop
constant SoccerPlayer : Entity → Prop
constant Left : Entity → Team → Prop
constant PlaysFor : Entity → Team → Prop

axiom h_striker_rl : Striker RobertLewandowski
axiom h_striker_implies_soccer : ∀ x, Striker x → SoccerPlayer x
axiom h_left_rl_bayern : Left RobertLewandowski BayernMunchen
axiom h_left_no_longer_plays : ∀ x t, Left x t → ¬ PlaysFor x t

theorem RL_is_soccer_player : SoccerPlayer RobertLewandowski :=
  h_striker_implies_soccer RobertLewandowski h_striker_rl

end Reasoning
</lean>
