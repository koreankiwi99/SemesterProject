<lean>
constant Person : Type

constant At : Person → Prop
constant Works : Person → Prop
constant Needs : Person → Prop
constant Orders : Person → Prop
constant Enjoys : Person → Prop
constant Picky : Person → Prop
constant Spends : Person → Prop

constant A1 : ∀ x, At x → Works x → Needs x
constant A2 : ∀ x, At x → Orders x → Works x
constant A3 : ∀ x, At x → (Orders x ∨ Enjoys x)
constant A4 : ∀ x, At x → Enjoys x → ¬ Picky x
constant A5 : ∀ x, At x → Enjoys x → Spends x

constant Hannah : Person
constant HAt : At Hannah
constant Hh : Works Hannah ∧ (Needs Hannah → (¬ Picky Hannah ∧ ¬ Needs Hannah))

theorem Hannah_needs_money : Needs Hannah :=
by
  have hWorks : Works Hannah := And.left Hh
  exact A1 Hannah HAt hWorks

theorem Hannah_not_needs_money : ¬ Needs Hannah :=
by
  intro hNeeds
  have hConj := (And.right Hh) hNeeds
  exact hConj.right hNeeds
</lean>
ANSWER: True
