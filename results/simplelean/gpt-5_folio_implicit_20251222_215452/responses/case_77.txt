<lean>
universe u

constant Person : Type u

-- Predicates
constant AtSchool : Person → Prop
constant Work : Person → Prop
constant NeedMoney : Person → Prop
constant OrderTakeout : Person → Prop
constant EnjoyDining : Person → Prop
constant Picky : Person → Prop
constant SpendDining : Person → Prop

-- Individual
constant Hannah : Person

-- Premises
axiom A1 : ∀ x : Person, AtSchool x → Work x → NeedMoney x
axiom A2 : ∀ x : Person, AtSchool x → OrderTakeout x → Work x
axiom A3 : ∀ x : Person, AtSchool x → (OrderTakeout x ∨ EnjoyDining x)
axiom A4 : ∀ x : Person, AtSchool x → EnjoyDining x → ¬ Picky x
axiom A5 : ∀ x : Person, AtSchool x → EnjoyDining x → SpendDining x
axiom A6 : AtSchool Hannah
axiom A7 : Work Hannah ∧ (NeedMoney Hannah → (¬ Picky Hannah ∧ ¬ NeedMoney Hannah))

-- Goal: Hannah is at Mary's school and she either is not a picky eater or,
-- if she is, then she spends a lot of her time eating and catching up with friends in the campus dining halls.
theorem conclusion :
  AtSchool Hannah ∧ (¬ Picky Hannah ∨ (Picky Hannah → SpendDining Hannah)) := by
  have hAt : AtSchool Hannah := A6
  have hWork : Work Hannah := And.left A7
  have hNeed : NeedMoney Hannah := A1 _ hAt hWork
  have hImp : NeedMoney Hannah → (¬ Picky Hannah ∧ ¬ NeedMoney Hannah) := And.right A7
  have hConj : ¬ Picky Hannah ∧ ¬ NeedMoney Hannah := hImp hNeed
  have hNotPicky : ¬ Picky Hannah := And.left hConj
  exact And.intro hAt (Or.inl hNotPicky)
</lean>
ANSWER: True
