<lean>
universe u

constant U : Type u

-- Predicates
constant BakedSweet : U → Prop
constant Spicy : U → Prop
constant Cupcake : U → Prop
constant MalaHotpot : U → Prop
constant FromBBM : U → Prop

-- Entity: Dried Thai chilies
constant DTC : U

-- Axioms from the textual context
-- 1. No baked sweets are spicy.
axiom h1 : ∀ x : U, BakedSweet x → ¬ Spicy x
-- 2. All cupcakes are baked sweets.
axiom h2 : ∀ x : U, Cupcake x → BakedSweet x
-- 3. All mala hotpots are spicy.
axiom h3 : ∀ x : U, MalaHotpot x → Spicy x
-- 4. All products from Baked by Melissa are cupcakes.
axiom h4 : ∀ x : U, FromBBM x → Cupcake x
-- 5. Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom h5 : Spicy DTC ∨ MalaHotpot DTC ∨ ¬ BakedSweet DTC

-- Goal: Dried Thai chilies are not products of Baked by Melissa.
theorem not_fromBBM_DTC : ¬ FromBBM DTC := by
  intro hfbm
  have hc : Cupcake DTC := h4 DTC hfbm
  have hb : BakedSweet DTC := h2 DTC hc
  have hns : ¬ Spicy DTC := h1 DTC hb
  cases h5 with
  | inl hs =>
    exact hns hs
  | inr rest =>
    cases rest with
    | inl hmala =>
      have hs' : Spicy DTC := h3 DTC hmala
      exact hns hs'
    | inr hnotb =>
      exact hnotb hb
</lean>
ANSWER: True
