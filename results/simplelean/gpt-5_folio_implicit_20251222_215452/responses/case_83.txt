<lean>
universe u

theorem hamden_statement_true
  {Restaurant : Type u}
  (Listed Neg RatingHigh TakeOut Popular : Restaurant → Prop)
  (HP : Restaurant)
  (h1 : ∀ r, Listed r → ¬ Neg r)
  (h2 : ∀ r, RatingHigh r → Listed r)
  (h3 : ∃ r, ¬ TakeOut r ∧ Neg r)
  (h4 : ∀ r, Popular r → RatingHigh r)
  (h5 : RatingHigh HP ∨ Popular HP)
  : (TakeOut HP ∧ Neg HP) → (RatingHigh HP ∧ ¬ TakeOut HP) := by
  have notNegHP : ¬ Neg HP := by
    cases h5 with
    | inl hR =>
        exact h1 HP (h2 HP hR)
    | inr hP =>
        exact h1 HP (h2 HP (h4 HP hP))
  intro hTN
  exact False.elim (notNegHP hTN.right)
</lean>
ANSWER: True
