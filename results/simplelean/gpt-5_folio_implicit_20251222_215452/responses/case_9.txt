<lean>
-- We formalize the scenario in Lean 4.

constant Entity : Type

-- Predicates for being a wild turkey and each of the six types
constant WildTurkey : Entity → Prop
constant Eastern : Entity → Prop
constant Osceola : Entity → Prop
constant Gould : Entity → Prop
constant Merriam : Entity → Prop
constant RioGrande : Entity → Prop
constant Ocellated : Entity → Prop

-- Individual Tom
constant Tom : Entity

-- Axiom: Every wild turkey is of one of the six (exhaustiveness)
axiom six_types_exhaustive :
  ∀ x, WildTurkey x →
    (Eastern x ∨ (Osceola x ∨ (Gould x ∨ (Merriam x ∨ (RioGrande x ∨ Ocellated x)))))

-- Facts about Tom from the problem statement
axiom Tom_not_Eastern : ¬ Eastern Tom
axiom Tom_not_Osceola : ¬ Osceola Tom
axiom Tom_not_Gould : ¬ Gould Tom
axiom Tom_not_Merriam : ¬ Merriam Tom
axiom Tom_not_RioGrande : ¬ RioGrande Tom
axiom Tom_is_WildTurkey : WildTurkey Tom

-- Goal: Tom is an Ocellated wild turkey
theorem Tom_is_Ocellated : Ocellated Tom := by
  have h := six_types_exhaustive Tom Tom_is_WildTurkey
  cases h with
  | inl hE =>
      exact False.elim (Tom_not_Eastern hE)
  | inr h1 =>
      cases h1 with
      | inl hO =>
          exact False.elim (Tom_not_Osceola hO)
      | inr h2 =>
          cases h2 with
          | inl hG =>
              exact False.elim (Tom_not_Gould hG)
          | inr h3 =>
              cases h3 with
              | inl hM =>
                  exact False.elim (Tom_not_Merriam hM)
              | inr h4 =>
                  cases h4 with
                  | inl hR =>
                      exact False.elim (Tom_not_RioGrande hR)
                  | inr hOc =>
                      exact hOc
</lean>
ANSWER: True
