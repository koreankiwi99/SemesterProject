axiom FewResources : obj → Prop
axiom ProvideFA : obj → obj → Prop    -- provider → person → Prop

-- ETS provides financial aid to GRE applicants who prove economic hardship
axiom R1 : ∀ x : obj, TakingGRE x ∧ ProvesEH x → ProvideFA ETS x

-- Those in single-parent families or with few resources can prove economic hardship
axiom R2 : ∀ x : obj, SingleParentFamily x ∨ FewResources x → ProvesEH x

-- Facts about Tom
axiom T_spf : SingleParentFamily Tom
axiom T_few : FewResources Tom
axiom T_app : TakingGRE Tom

-- Counterexample to "No one taking the GRE test has financial aid provided to them by something"
theorem exists_gre_with_aid : ∃ x : obj, TakingGRE x ∧ ∃ y : obj, ProvideFA y x := by
  have hEH : ProvesEH Tom := by
    have disj : SingleParentFamily Tom ∨ FewResources Tom := Or.inl T_spf
    exact R2 Tom disj
  have hFA : ProvideFA ETS Tom := by
    have conj : TakingGRE Tom ∧ ProvesEH Tom := And.intro T_app hEH
    exact R1 Tom conj
  exact Exists.intro Tom (And.intro T_app (Exists.intro ETS hFA))
