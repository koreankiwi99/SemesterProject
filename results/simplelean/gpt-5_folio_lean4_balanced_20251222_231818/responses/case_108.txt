lemma not_spill_and_cleanly_all : ∀ x : obj, ¬ (Spill x ∧ Cleanly x) :=
  fun x =>
    fun h =>
      let hs := h.left
      let hc := h.right
      let hnotT := A1 x hs
      let hT := A3 x hc
      hnotT hT

lemma peter_not_spill_and_not_cleanly : ¬ Spill Peter ∧ ¬ Cleanly Peter :=

    (fun h => False.elim ((not_spill_and_cleanly_all Peter) h))
