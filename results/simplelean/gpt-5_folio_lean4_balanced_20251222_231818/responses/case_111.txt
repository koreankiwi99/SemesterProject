<lean>
axiom obj : Type

axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

axiom Picuris : obj
axiom Onate : obj
axiom HardingPegmatiteMine : obj

axiom A1 : MountainRange Picuris
axiom A2 : InNewMexico Picuris ∨ InTexas Picuris
axiom A3 : Visited Onate Picuris
axiom A4 : Mine HardingPegmatiteMine
axiom A5 : LocatedIn HardingPegmatiteMine Picuris
axiom A6 : Donated HardingPegmatiteMine
axiom A7 : ∀ r : obj, MountainRange r ∧ InTexas r → ¬ ∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m

theorem onate_visited_nm_range : ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ Visited Onate r := by
  -- There exists a donated mine located in Picuris
  have exMine : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m :=
    Exists.intro HardingPegmatiteMine (And.intro A4 (And.intro A5 A6))
  -- Show Picuris is not in Texas
  have not_in_TX : ¬ InTexas Picuris := by
    intro htx
    have neg_exists := A7 Picuris (And.intro A1 htx)
    exact neg_exists exMine
  -- From in NM or TX and not in TX, deduce in NM
  have in_NM : InNewMexico Picuris := by
    cases A2 with
    | inl hnm => exact hnm
    | inr htx => exact False.elim (not_in_TX htx)
  -- Conclude with Picuris as witness
  exact Exists.intro Picuris (And.intro A1 (And.intro in_NM A3))
</lean>

ANSWER: True
