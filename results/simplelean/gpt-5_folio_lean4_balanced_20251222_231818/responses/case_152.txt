axiom obj : Type

axiom Hydrocarbon : obj → Prop
axiom OrganicCompound : obj → Prop
axiom Alkane : obj → Prop
axiom ChemicalCompound : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom SingleElement : obj → Prop
axiom Mixture : obj

-- All hydrocarbons are organic compounds.
axiom A1 : ∀ x : obj, Hydrocarbon x → OrganicCompound x
-- All alkanes are hydrocarbons.
axiom A2 : ∀ x : obj, Alkane x → Hydrocarbon x
-- All organic compounds are chemical compounds.
axiom A3 : ∀ x : obj, OrganicCompound x → ChemicalCompound x
-- All organic compounds contain carbon.
axiom A4 : ∀ x : obj, OrganicCompound x → ContainsCarbon x
-- No chemical compounds contain only one element.
axiom A5 : ∀ x : obj, ChemicalCompound x → ¬ SingleElement x
-- The mixture either both is a chemical compound and contains only one element, or neither.
axiom A6 : (ChemicalCompound Mixture ∧ SingleElement Mixture) ∨ (¬ ChemicalCompound Mixture ∧ ¬ SingleElement Mixture)

-- Some derivable consequences:
theorem not_ChemicalCompound_Mixture : ¬ ChemicalCompound Mixture := by
  have h := A6
  cases h with
  | inl h1 =>
      have chem : ChemicalCompound Mixture := h1.left
      have single : SingleElement Mixture := h1.right
      have not_single : ¬ SingleElement Mixture := A5 Mixture chem
      exact (fun _ => False.elim (not_single single))
  | inr h2 =>
      exact h2.left

theorem not_SingleElement_Mixture : ¬ SingleElement Mixture := by
  have h := A6
  cases h with
  | inl h1 =>
      have chem : ChemicalCompound Mixture := h1.left
      have not_single : ¬ SingleElement Mixture := A5 Mixture chem
      exact not_single
  | inr h2 =>
      exact h2.right

theorem not_OrganicCompound_Mixture : ¬ OrganicCompound Mixture := by
  intro org
  have chem : ChemicalCompound Mixture := A3 Mixture org
  exact not_ChemicalCompound_Mixture chem
