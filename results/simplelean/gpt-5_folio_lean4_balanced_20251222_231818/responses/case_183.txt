<lean>
axiom obj : Type
axiom Tom : obj

axiom NiceToAnimals : obj → Prop
axiom MeanToAnimals : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop

-- No one nice to animals is also mean to animals.
axiom A1 : ∀ x : obj, NiceToAnimals x → ¬ MeanToAnimals x
-- Some grumpy people are mean to animals.
axiom A2 : ∃ x : obj, Grumpy x ∧ MeanToAnimals x
-- All animal lovers are nice to animals.
axiom A3 : ∀ x : obj, AnimalLover x → NiceToAnimals x
-- All pet owners love animals.
axiom A4 : ∀ x : obj, PetOwner x → AnimalLover x
-- Tom is a pet owner.
axiom A5 : PetOwner Tom

-- Claim to evaluate:
-- If Tom is not both grumpy and mean to animals, then Tom is neither grumpy nor an animal lover.
theorem negate_claim :
  ¬ ( (¬ (Grumpy Tom ∧ MeanToAnimals Tom)) → (¬ Grumpy Tom ∧ ¬ AnimalLover Tom) ) :=
by
  -- Tom is an animal lover
  have hLover : AnimalLover Tom := A4 Tom A5
  -- Tom is nice to animals
  have hNice : NiceToAnimals Tom := A3 Tom hLover
  -- Thus Tom is not mean to animals
  have hNotMean : ¬ MeanToAnimals Tom := A1 Tom hNice
  -- Hence Tom is not both grumpy and mean
  have hP : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) := fun h => hNotMean h.right
  -- And it's not the case that Tom is neither grumpy nor an animal lover (since he is an animal lover)
  have hNotQ : ¬ (¬ Grumpy Tom ∧ ¬ AnimalLover Tom) := fun h => h.right hLover
  intro hImp
  have q : (¬ Grumpy Tom ∧ ¬ AnimalLover Tom) := hImp hP
  exact hNotQ q
</lean>

ANSWER: False
