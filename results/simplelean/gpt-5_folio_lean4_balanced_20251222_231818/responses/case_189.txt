<lean>
axiom obj : Type
axiom Yuri : obj

axiom ProBBall : obj → Prop
axiom American : obj → Prop
axiom CanJumpShot : obj → Prop
axiom LeapsStraight : obj → Prop
axiom ActivateLegs : obj → Prop

-- Some professional basketball players are not American nationals.
axiom A1 : ∃ x : obj, ProBBall x ∧ ¬ American x
-- All professional basketball players can do jump shots.
axiom A2 : ∀ x : obj, ProBBall x → CanJumpShot x
-- If someone can jump shots, they leap straight into the air.
axiom A3 : ∀ x : obj, CanJumpShot x → LeapsStraight x
-- If someone leaps straight into the air, they activate their leg muscles.
axiom A4 : ∀ x : obj, LeapsStraight x → ActivateLegs x
-- Yuri does not activate his leg muscles.
axiom A5 : ¬ ActivateLegs Yuri

theorem not_american_pro_yuri : ¬ (American Yuri ∧ ProBBall Yuri) := by
  intro h
  have hPro : ProBBall Yuri := h.right
  have hAct : ActivateLegs Yuri := by
    have hJS : CanJumpShot Yuri := A2 Yuri hPro
    have hLeap : LeapsStraight Yuri := A3 Yuri hJS
    exact A4 Yuri hLeap
  exact A5 hAct
</lean>

ANSWER: False
