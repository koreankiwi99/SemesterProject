open Classical

axiom obj : Type
axiom KiKi : obj

axiom Animal : obj → Prop
axiom Bark : obj → Prop
axiom Dog : obj → Prop
axiom Jump : obj → Prop
axiom Legs : obj → Prop
axiom Terricolous : obj → Prop
axiom LikesJane : obj → Prop

-- Jane does not like any animals that bark.
axiom no_like_barkers : ∀ x : obj, Animal x → Bark x → ¬ LikesJane x
-- All dogs are animals that bark.
axiom dogs_bark : ∀ x : obj, Dog x → Animal x ∧ Bark x
-- Jane likes all animals that jump.
axiom likes_animals_that_jump : ∀ x : obj, Animal x → Jump x → LikesJane x
-- If an animal has legs, then it jumps.
axiom legs_imply_jump : ∀ x : obj, Animal x → Legs x → Jump x
-- If an animal is terricolous, then it has legs.
axiom terricolous_implies_legs : ∀ x : obj, Animal x → Terricolous x → Legs x
-- KiKi is an animal.
axiom kiki_animal : Animal KiKi
-- If KiKi neither jumps nor has legs, then KiKi is terricolous or has legs.
axiom kiki_rule : (¬ Jump KiKi ∧ ¬ Legs KiKi) → (Terricolous KiKi ∨ Legs KiKi)

theorem kiki_no_bark_not_dog : ¬ Bark KiKi ∧ ¬ Dog KiKi := by
  -- First, prove Jump KiKi by contradiction
  have hAnimal : Animal KiKi := kiki_animal
  have hJump : Jump KiKi := by
    by_contra hNotJump
    have hNotLegs : ¬ Legs KiKi := fun hLegs => hNotJump (legs_imply_jump KiKi hAnimal hLegs)
    have hAnte : ¬ Jump KiKi ∧ ¬ Legs KiKi := And.intro hNotJump hNotLegs
    have hDisj : Terricolous KiKi ∨ Legs KiKi := kiki_rule hAnte
    cases hDisj with
    | inl hTerr =>
      have hLegsFromTerr : Legs KiKi := terricolous_implies_legs KiKi hAnimal hTerr
      exact hNotLegs hLegsFromTerr
    | inr hLegs =>
      exact hNotLegs hLegs
  -- From Jump and Animal, Jane likes KiKi
  have hLike : LikesJane KiKi := likes_animals_that_jump KiKi hAnimal hJump
  -- Therefore KiKi does not bark
  have hNotBark : ¬ Bark KiKi := by
    intro hBark
    have hNotLike : ¬ LikesJane KiKi := no_like_barkers KiKi hAnimal hBark
    exact hNotLike hLike
  -- Therefore KiKi is not a dog (since dogs bark)
  have hNotDog : ¬ Dog KiKi := by
    intro hDog
    have hAB : Animal KiKi ∧ Bark KiKi := dogs_bark KiKi hDog
    exact hNotBark hAB.right
  exact And.intro hNotBark hNotDog
</lean>

ANSWER: True

theorem kiki_no_bark_not_dog : ¬ Bark KiKi ∧ ¬ Dog KiKi := by
  have hAnimal : Animal KiKi := kiki_animal
  -- Prove Jump KiKi using excluded middle and contradictions
  have hJump : Jump KiKi := by
    have emJ : Jump KiKi ∨ ¬ Jump KiKi := Classical.em (Jump KiKi)
    exact Or.elim emJ
      (fun hJ => hJ)
      (fun hNotJump =>
        let hNotLegs : ¬ Legs KiKi := mt (legs_imply_jump KiKi hAnimal) hNotJump
        let hDisj : Terricolous KiKi ∨ Legs KiKi := kiki_rule ⟨hNotJump, hNotLegs⟩
        Or.elim hDisj
          (fun hTerr =>
            let hLegsFromTerr : Legs KiKi := terricolous_implies_legs KiKi hAnimal hTerr
            False.elim (hNotLegs hLegsFromTerr))
          (fun hLegs => False.elim (hNotLegs hLegs)))
  -- From Jump and Animal, Jane likes KiKi
  have hLike : LikesJane KiKi := likes_animals_that_jump KiKi hAnimal hJump
  -- Therefore KiKi does not bark
  have hNotBark : ¬ Bark KiKi := fun hBark => (no_like_barkers KiKi hAnimal hBark) hLike
  -- Therefore KiKi is not a dog (since dogs bark)
  have hNotDog : ¬ Dog KiKi := fun hDog =>
    let hAB := dogs_bark KiKi hDog
    hNotBark hAB.right
  exact And.intro hNotBark hNotDog
