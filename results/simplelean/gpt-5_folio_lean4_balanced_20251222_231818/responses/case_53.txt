<lean>
axiom obj : Type
axiom fam : Type

axiom InFam : obj → fam → Prop
axiom LanguageFamily : fam → Prop
axiom IndoEuropean : obj → Prop
axiom Related : obj → obj → Prop

axiom French : obj
axiom Spanish : obj
axiom German : obj
axiom Basque : obj

axiom Romance : fam

-- All Romance languages are Indo-European languages.
axiom A1 : ∀ x : obj, InFam x Romance → IndoEuropean x
-- Romance languages are a language family.
axiom A2 : LanguageFamily Romance
-- All languages within a language family are related to each other.
axiom A3 : ∀ F : fam, LanguageFamily F → ∀ x y : obj, InFam x F → InFam y F → Related x y
-- French and Spanish are both Romance languages.
axiom A4 : InFam French Romance
axiom A5 : InFam Spanish Romance
-- German is related to Spanish.
axiom A6 : Related German Spanish
-- Basque is not related to any other language.
axiom A7 : ∀ y : obj, y ≠ Basque → (¬ Related Basque y ∧ ¬ Related y Basque)

theorem french_is_indoeuropean : IndoEuropean French :=
  A1 French A4
</lean>

ANSWER: True
