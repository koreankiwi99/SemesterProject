<lean>
axiom obj : Type
axiom At : obj → Prop
axiom Work : obj → Prop
axiom NeedMoney : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom Picky : obj → Prop
axiom Spend : obj → Prop

axiom A1 : ∀ x : obj, At x → Work x → NeedMoney x
axiom A2 : ∀ x : obj, At x → Order x → Work x
axiom A3 : ∀ x : obj, At x → (Order x ∨ Enjoy x)
axiom A4 : ∀ x : obj, At x → Enjoy x → ¬ Picky x
axiom A5 : ∀ x : obj, At x → Enjoy x → Spend x

axiom Hannah : obj
axiom HAt : At Hannah
axiom HWork : Work Hannah ∧ (NeedMoney Hannah → (¬ Picky Hannah ∧ ¬ NeedMoney Hannah))

theorem hannah_statement :
  At Hannah ∧ (¬ Picky Hannah ∨ (Picky Hannah → Spend Hannah)) := by
  have hAt : At Hannah := HAt
  have hWork : Work Hannah := And.left HWork
  have hCond : NeedMoney Hannah → (¬ Picky Hannah ∧ ¬ NeedMoney Hannah) := And.right HWork
  have hNeed : NeedMoney Hannah := A1 Hannah hAt hWork
  have hNotPicky : ¬ Picky Hannah := And.left (hCond hNeed)
  exact And.intro hAt (Or.inl hNotPicky)
</lean>

ANSWER: True
