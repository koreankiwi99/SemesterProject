<lean>
axiom obj : Type
axiom DriedThaiChilies : obj

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- No baked sweets are spicy.
axiom no_baked_spicy : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom cupcakes_baked : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom hotpot_spicy : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom bbm_cupcake : ∀ x : obj, FromBBM x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom dried_disj :
  Spicy DriedThaiChilies ∨ (MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies)

theorem dried_thai_chilies_neither_bbm_nor_baked :
  ¬ FromBBM DriedThaiChilies ∧ ¬ BakedSweet DriedThaiChilies := by
  -- First show ¬ BakedSweet DriedThaiChilies
  have h_not_baked : ¬ BakedSweet DriedThaiChilies := by
    apply Or.elim dried_disj
    · intro h_spicy
      intro h_baked
      have h_not_spicy := no_baked_spicy DriedThaiChilies h_baked
      exact h_not_spicy h_spicy
    · intro h_right
      apply Or.elim h_right
      · intro h_hotpot
        intro h_baked
        have h_spicy := hotpot_spicy DriedThaiChilies h_hotpot
        have h_not_spicy := no_baked_spicy DriedThaiChilies h_baked
        exact h_not_spicy h_spicy
      · intro h_nb
        exact h_nb
  -- Then show ¬ FromBBM DriedThaiChilies
  have h_not_bbm : ¬ FromBBM DriedThaiChilies := by
    intro h_bbm
    have h_cup := bbm_cupcake DriedThaiChilies h_bbm
    have h_baked := cupcakes_baked DriedThaiChilies h_cup
    exact h_not_baked h_baked
  exact And.intro h_not_bbm h_not_baked
</lean>

ANSWER: True
