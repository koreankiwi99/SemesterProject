<lean>
axiom obj : Type
axiom Tom : obj

axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Goulds : obj → Prop
axiom Merriam : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop
axiom WildTurkey : obj → Prop

-- From the context: every wild turkey is one of the six types listed
axiom classification :
  ∀ x : obj, WildTurkey x →
    (Eastern x ∨ (Osceola x ∨ (Goulds x ∨ (Merriam x ∨ (RioGrande x ∨ Ocellated x)))))

-- Tom-specific facts
axiom Tom_not_Eastern : ¬ Eastern Tom
axiom Tom_not_Osceola : ¬ Osceola Tom
axiom Tom_not_Goulds : ¬ Goulds Tom
axiom Tom_not_Merriam : ¬ Merriam Tom
axiom Tom_not_RioGrande : ¬ RioGrande Tom
axiom Tom_wild : WildTurkey Tom

theorem tom_is_ocellated : Ocellated Tom := by
  have h := classification Tom Tom_wild
  cases h with
  | inl e =>
      exact False.elim (Tom_not_Eastern e)
  | inr h2 =>
      cases h2 with
      | inl o => exact False.elim (Tom_not_Osceola o)
      | inr h3 =>
          cases h3 with
          | inl g => exact False.elim (Tom_not_Goulds g)
          | inr h4 =>
              cases h4 with
              | inl m => exact False.elim (Tom_not_Merriam m)
              | inr h5 =>
                  cases h5 with
                  | inl r => exact False.elim (Tom_not_RioGrande r)
                  | inr oc => exact oc
</lean>

ANSWER: True
