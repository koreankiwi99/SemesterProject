axiom Person : Type
axiom Org : Type

axiom ETS : Org
axiom Tom : Person

-- Predicates
axiom Applicant : Person → Prop
axiom TakingGRE : Person → Prop
axiom ProvesHardship : Person → Prop
axiom SingleParentFamily : Person → Prop
axiom FewResources : Person → Prop
axiom ProvidesAid : Org → Person → Prop

-- Facts from the context
-- ETS provides financial aid to those GRE applicants who prove economic hardship.
axiom ETS_rule : ∀ x : Person, Applicant x → ProvesHardship x → ProvidesAid ETS x
-- Those living in single-parent families or having few resources can prove economic hardship.
axiom Hardship_rule : ∀ x : Person, (SingleParentFamily x ∨ FewResources x) → ProvesHardship x
-- Anyone applying to take the GRE test counts as taking the GRE test (link applicants to the taking context).
axiom Applicant_implies_Taking : ∀ x : Person, Applicant x → TakingGRE x

-- Tom's situation
axiom Tom_single : SingleParentFamily Tom
axiom Tom_few : FewResources Tom
axiom Tom_applicant : Applicant Tom

-- The statement to evaluate:
def Statement := ∀ x : Person, TakingGRE x → ¬ ∃ y : Org, ProvidesAid y x

-- Show the statement is false by giving Tom as a counterexample.
lemma Tom_hardship : ProvesHardship Tom :=
  Hardship_rule Tom (Or.inl Tom_single)

lemma Tom_taking : TakingGRE Tom :=
  Applicant_implies_Taking Tom Tom_applicant

lemma Tom_has_aid_from_ETS : ∃ y : Org, ProvidesAid y Tom :=
  Exists.intro ETS (ETS_rule Tom Tom_applicant Tom_hardship)

theorem statement_false : ¬ Statement := by
  intro h
  have h_not_exists := h Tom Tom_taking
  exact h_not_exists Tom_has_aid_from_ETS
</lean>
