axiom A2 : InNM Picuris ∨ InTX Picuris
axiom A3 : Visited Onate Picuris
axiom A4 : Mine HardingMine
axiom A5 : LocatedIn HardingMine Picuris
axiom A6 : Donated HardingMine
axiom A7 : ∀ r : obj, (MountainRange r ∧ InTX r) → ¬ (∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m)

-- Derived fact: Picuris is not in Texas (since it has a donated mine located in it)
lemma not_inTX_Picuris : ¬ InTX Picuris := by
  intro hTX
  have hno : ¬ ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m := by
    exact A7 Picuris (And.intro A1 hTX)
  have hex : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m := by
    exists HardingMine
    exact And.intro A4 (And.intro A5 A6)
  exact hno hex

-- Consequently, from the disjunction, Picuris is in New Mexico
lemma inNM_Picuris : InNM Picuris := by
  apply Or.elim A2
  · intro hnm; exact hnm
  · intro htx; exact False.elim (not_inTX_Picuris htx)
