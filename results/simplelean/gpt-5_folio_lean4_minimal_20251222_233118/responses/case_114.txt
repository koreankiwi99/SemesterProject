axiom Taller : obj → obj → Prop
axiom Block : obj → obj → Prop

axiom GreatShooter : obj → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom A1 : ∀ x y : obj, Man x → Man y → Taller x y → Block x y

-- Michael is a man who is taller than everyone else in his class.
axiom A2_man : Man Michael
axiom A2_tall : ∀ y : obj, InMClass y → y ≠ Michael → Taller Michael y

-- Transitivity of taller.
axiom A3_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom A4_man : Man Peter
axiom A4_taller : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom A5 : ∀ p : obj, ¬ JumpsWhenShooting p → Block Michael p

-- Michael cannot block Windy's shooting.
axiom A6 : ¬ Block Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom A7 : ∀ s : obj, JumpsWhenShooting s → GreatShooter s

-- A useful derived fact: Peter is taller than any other member of Michael's class (distinct from Michael).
lemma peter_taller_than_any_other_in_Mclass :
  ∀ y : obj, InMClass y → y ≠ Michael → Taller Peter y :=
by
  intro y hy hne
  have hm_y : Taller Michael y := A2_tall y hy hne
  have hp_m : Taller Peter Michael := A4_taller
  exact A3_trans Peter Michael y hp_m hm_y

-- From A5 and A6 we can derive that it's not the case that Windy does not jump (i.e., ¬¬JumpsWhenShooting Windy).
lemma windy_jumps_notnot : ¬¬ JumpsWhenShooting Windy :=
by
  intro hnot
  have : Block Michael Windy := A5 Windy hnot
  exact A6 this
