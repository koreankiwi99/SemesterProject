<lean>
open Classical

axiom obj : Type

axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom CanBlock : obj → obj → Prop
axiom JumpShoot : obj → Prop
axiom GreatShooter : obj → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom A1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man.
axiom A2 : Man Michael

-- Taller is transitive.
axiom A3 : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom A4a : Man Peter
axiom A4b : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom A5 : ∀ p : obj, ¬ JumpShoot p → CanBlock Michael p

-- Michael cannot block Windy's shooting.
axiom A6 : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom A7 : ∀ x : obj, JumpShoot x → GreatShooter x

theorem windy_is_great : GreatShooter Windy :=
by
  classical
  have hJ : JumpShoot Windy :=
    by
      by_contra h
      exact A6 (A5 Windy h)
  exact A7 Windy hJ
</lean>

ANSWER: True

theorem windy_is_great : GreatShooter Windy :=
by
  classical
  by_cases hJ : JumpShoot Windy
  · exact A7 Windy hJ
  ·
    have hcb : CanBlock Michael Windy := A5 Windy hJ
    have hFalse : False := A6 hcb
    exact False.elim hFalse
