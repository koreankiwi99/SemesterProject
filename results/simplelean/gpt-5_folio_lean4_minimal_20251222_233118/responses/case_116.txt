<lean>
axiom Person : Type
axiom Michael : Person
axiom Peter : Person
axiom Windy : Person

axiom Man : Person → Prop
axiom Taller : Person → Person → Prop
axiom CanBlock : Person → Person → Prop
axiom Jump : Person → Prop
axiom Shooter : Person → Prop
axiom GreatShooter : Person → Prop
axiom InMichaelsClass : Person → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom A1 : ∀ x y : Person, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom A2a : Man Michael
axiom A2b : ∀ y : Person, InMichaelsClass y → Taller Michael y

-- Transitivity of "taller than"
axiom A3 : ∀ x y z : Person, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom A4a : Man Peter
axiom A4b : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom A5 : ∀ y : Person, ¬ Jump y → CanBlock Michael y

-- Michael cannot block Windy's shooting.
axiom A6 : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom A7 : ∀ x : Person, Shooter x → Jump x → GreatShooter x

-- A derivable fact: Peter can block Michael (both are men, and Peter is taller than Michael).
theorem peter_blocks_michael : CanBlock Peter Michael := by
  apply A1
  · exact A4a
  · exact A2a
  · exact A4b

-- From "Michael cannot block Windy" and "Michael blocks any non-jumper",
-- we can derive ¬¬ Jump Windy (but not Jump Windy without classical logic).
theorem not_not_jump_windy : ¬¬ Jump Windy := by
  intro hnot
  have hblock : CanBlock Michael Windy := A5 Windy hnot
  exact A6 hblock

-- Cannot prove CanBlock Peter Windy or its negation from the given axioms.
