axiom Alkane : obj → Prop
axiom Organic : obj → Prop
axiom ChemicalCompound : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom OnlyOneElement : obj → Prop

-- Axioms from the context
axiom A1 : ∀ x : obj, Hydrocarbon x → Organic x
axiom A2 : ∀ x : obj, Alkane x → Hydrocarbon x
axiom A3 : ∀ x : obj, Organic x → ChemicalCompound x
axiom A4 : ∀ x : obj, Organic x → ContainsCarbon x
axiom A5 : ∀ x : obj, ChemicalCompound x → ¬ OnlyOneElement x
axiom A6 : (ChemicalCompound Mixture ∧ OnlyOneElement Mixture) ∨ (¬ ChemicalCompound Mixture ∧ ¬ OnlyOneElement Mixture)

-- From A5 and A6, the first disjunct in A6 is impossible, so we get the second disjunct.
lemma not_chem_and_not_one : ¬ ChemicalCompound Mixture ∧ ¬ OnlyOneElement Mixture := by
  have h := A6
  apply Or.elim h
  · intro hboth
    have hc : ChemicalCompound Mixture := hboth.left
    have ho : OnlyOneElement Mixture := hboth.right
    have hno : ¬ OnlyOneElement Mixture := A5 Mixture hc
    exact False.elim (hno ho)
  · intro hneither
    exact hneither

lemma not_chemical : ¬ ChemicalCompound Mixture := not_chem_and_not_one.left
lemma not_only_one_element : ¬ OnlyOneElement Mixture := not_chem_and_not_one.right

-- We can derive ¬ Organic Mixture from ¬ ChemicalCompound Mixture and A3 (modus tollens).
lemma not_organic : ¬ Organic Mixture := by
  intro horg
  exact not_chemical (A3 Mixture horg)

-- Cannot prove ContainsCarbon Mixture or its negation from the given axioms.
-- The only rule introducing ContainsCarbon is A4, which requires Organic Mixture,
-- but we derived ¬ Organic Mixture, and there is no axiom linking non-organic to not containing carbon.
