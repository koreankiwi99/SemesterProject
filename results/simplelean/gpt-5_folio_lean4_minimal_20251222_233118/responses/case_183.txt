<lean>
axiom Person : Type
axiom Tom : Person

axiom NiceToAnimals : Person → Prop
axiom MeanToAnimals : Person → Prop
axiom Grumpy : Person → Prop
axiom AnimalLover : Person → Prop
axiom PetOwner : Person → Prop

-- No one nice to animals is also mean to animals.
axiom H1 : ∀ x : Person, NiceToAnimals x → ¬ MeanToAnimals x
-- Some grumpy people are mean to animals.
axiom H2 : ∃ x : Person, Grumpy x ∧ MeanToAnimals x
-- All animal lovers are nice to animals.
axiom H3 : ∀ x : Person, AnimalLover x → NiceToAnimals x
-- All pet owners love animals.
axiom H4 : ∀ x : Person, PetOwner x → AnimalLover x
-- Tom is a pet owner.
axiom H5 : PetOwner Tom

-- We show the given statement is false by proving its negation.
theorem tom_statement_false :
  ¬ ((¬ (Grumpy Tom ∧ MeanToAnimals Tom)) → (¬ Grumpy Tom ∧ ¬ AnimalLover Tom)) := by
  intro h
  have hlover : AnimalLover Tom := H4 Tom H5
  have hnice : NiceToAnimals Tom := H3 Tom hlover
  have hnotmean : ¬ MeanToAnimals Tom := H1 Tom hnice
  have hant : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) := by
    intro hgm
    exact hnotmean hgm.right
  have hconseq : ¬ Grumpy Tom ∧ ¬ AnimalLover Tom := h hant
  exact hconseq.right hlover
</lean>

ANSWER: False
