<lean>
axiom obj : Type
axiom Yuri : obj

axiom ProBBPlayer : obj → Prop
axiom American : obj → Prop
axiom CanJumpShots : obj → Prop
axiom LeapStraight : obj → Prop
axiom ActivateLegMuscles : obj → Prop

-- Some professional basketball players are not American nationals.
axiom Hsome : ∃ x : obj, ProBBPlayer x ∧ ¬ American x
-- All professional basketball players can do jump shots.
axiom H1 : ∀ x : obj, ProBBPlayer x → CanJumpShots x
-- If someone can jump shots, they leap straight into the air.
axiom H2 : ∀ x : obj, CanJumpShots x → LeapStraight x
-- If someone leaps straight into the air, they activate their leg muscles.
axiom H3 : ∀ x : obj, LeapStraight x → ActivateLegMuscles x
-- Yuri does not activate his leg muscles.
axiom Hy : ¬ ActivateLegMuscles Yuri

theorem not_pro_yuri : ¬ ProBBPlayer Yuri := by
  intro hpro
  have hj : CanJumpShots Yuri := H1 Yuri hpro
  have hl : LeapStraight Yuri := H2 Yuri hj
  have ha : ActivateLegMuscles Yuri := H3 Yuri hl
  exact Hy ha

theorem not_american_pro_yuri : ¬ (American Yuri ∧ ProBBPlayer Yuri) := by
  intro h
  have hpro : ProBBPlayer Yuri := And.right h
  have hj : CanJumpShots Yuri := H1 Yuri hpro
  have hl : LeapStraight Yuri := H2 Yuri hj
  have ha : ActivateLegMuscles Yuri := H3 Yuri hl
  exact Hy ha
</lean>
