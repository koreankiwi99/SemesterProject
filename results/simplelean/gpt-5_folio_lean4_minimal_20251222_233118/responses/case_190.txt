-- Target statement: (¬ Leap Yuri) → (American Yuri ∧ Pro Yuri)
-- We prove its negation.
theorem not_target : ¬ ( (¬ Leap Yuri) → (American Yuri ∧ Pro Yuri) ) := by

  have hNotLeap : ¬ Leap Yuri := by

  have hPro_imp_Act : Pro Yuri → Activate Yuri := by

    have hL : Leap Yuri := canjump_leap Yuri hCJ
    exact leap_activate Yuri hL

  have hNotPro : ¬ Pro Yuri := by
    intro hP
    have hA : Activate Yuri := hPro_imp_Act hP
    exact yuri_not_activate hA
