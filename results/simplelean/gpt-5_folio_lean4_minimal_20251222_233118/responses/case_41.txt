<lean>
axiom obj : Type

-- Predicates on design/style objects
axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Brutalist : obj → Prop
axiom Adores : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop
axiom DesignByMax : obj → Prop

-- All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom T1 : ∀ x : obj, Zaha x ∧ Adores x → Interesting x

-- No brutalist buildings that Max adores have interesting geometries.
axiom T2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ Interesting x

-- Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom T3 : ∀ x : obj, Adores x → Zaha x ∨ Kelly x

-- All of Kelly Wearstler's design styles that Max adores are evocative.
axiom T4 : ∀ x : obj, Kelly x ∧ Adores x → Evocative x

-- All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom T5 : ∀ x : obj, Kelly x ∧ Adores x → Dreamy x

-- If a design by Max that he adores has interesting geometries, then it is a brutalist building and evocative.
axiom T6 : ∀ x : obj, DesignByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- What we can prove: any design by Max that he adores is evocative or dreamy.
theorem adored_designs_evocative_or_dreamy :
  ∀ x : obj, DesignByMax x ∧ Adores x → Evocative x ∨ Dreamy x := by
  intro x h
  have dx : DesignByMax x := And.left h
  have ax : Adores x := And.right h
  have hz_or_hk : Zaha x ∨ Kelly x := T3 x ax
  cases hz_or_hk with
  | inl hz =>
      have hx_int : Interesting x := T1 x (And.intro hz ax)
      have hx_be : Brutalist x ∧ Evocative x := T6 x (And.intro dx (And.intro ax hx_int))
      exact Or.inl (And.right hx_be)
  | inr hk =>
      exact Or.inl (T4 x (And.intro hk ax))

-- Target statement (unprovable from given axioms): every design by Max is evocative or dreamy.
-- theorems of the form: ∀ x, DesignByMax x → Evocative x ∨ Dreamy x
-- From the given axioms, we cannot derive this without assuming Adores x.
-- Nor can we derive its negation (existence of a design by Max that is neither evocative nor dreamy).
-- Hence the truth value is undecidable with the provided information.
