<lean>
axiom obj : Type
axiom Pet : obj → Prop
axiom Cat : obj → Prop
axiom Mammal : obj → Prop

-- All cats are mammals.
axiom A1 : ∀ x : obj, Cat x → Mammal x
-- Some pets are not mammals.
axiom A2 : ∃ x : obj, Pet x ∧ ¬ Mammal x

-- From A1, any non-mammal is not a cat.
theorem notCat_of_notMammal : ∀ x : obj, ¬ Mammal x → ¬ Cat x := by
  intro x hNotMammal hCat
  exact hNotMammal (A1 x hCat)

-- Hence, there exists a pet that is not a cat.
theorem exists_pet_not_cat : ∃ x : obj, Pet x ∧ ¬ Cat x := by
  refine Exists.elim A2 ?_
  intro x hx
  have hxPet : Pet x := And.left hx
  have hxNotMammal : ¬ Mammal x := And.right hx
  have hxNotCat : ¬ Cat x := notCat_of_notMammal x hxNotMammal
  exact Exists.intro x (And.intro hxPet hxNotCat)

-- We cannot derive that no pets are cats (∀ x, Pet x → ¬ Cat x) from the given axioms,
-- nor can we derive its negation (∃ x, Pet x ∧ Cat x).
</lean>
