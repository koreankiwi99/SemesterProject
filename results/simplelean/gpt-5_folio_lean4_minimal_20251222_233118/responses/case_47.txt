-- Define the seasons as exactly four constructors
inductive Season where
  | Spring
  | Summer
  | Fall
  | Winter

-- People
axiom Person : Type
axiom Emma : Person
axiom Mia : Person
axiom James : Person

-- Favorite season as a function

-- Properties: being a student and wanting a long vacation
axiom Student : Person → Prop
axiom WantLongVacation : Person → Prop

-- Rule: All students who want a long vacation have Summer as their favorite season
axiom Rule : ∀ p : Person, Student p → WantLongVacation p → Favorite p = Season.Summer

-- Facts given
axiom EmmaFav : Favorite Emma = Season.Summer
axiom MiaDiff : Favorite Mia ≠ Favorite Emma
axiom JamesWants : WantLongVacation James

-- Consequence: Mia's favorite is not Summer
lemma mia_not_summer : Favorite Mia ≠ Season.Summer := by
  intro h
  apply MiaDiff
  have h2 : Season.Summer = Favorite Emma := EmmaFav.symm
  calc
    Favorite Mia = Season.Summer := h
    _ = Favorite Emma := h2

-- From the given information, we cannot prove Favorite Mia = Season.Spring
-- nor can we prove its negation Favorite Mia ≠ Season.Spring.
-- We only know Favorite Mia ≠ Season.Summer, and Season could be Spring, Fall, or Winter.
</lean>

-- Facts given

theorem mia_not_summer : Not (Favorite Person.Mia = Season.Summer) := by
  intro h
  have hM : Favorite Person.Mia = Season.Summer := h
  have hE : Season.Summer = Favorite Person.Emma := Eq.symm EmmaFav
  have hME : Favorite Person.Mia = Favorite Person.Emma := Eq.trans hM hE
  exact MiaDiff hME

-- nor can we prove its negation Favorite Person.Mia ≠ Season.Spring.
