axiom James : obj

axiom SchedulesMeeting : obj → Prop
axiom GoToBuildingToday : obj → Prop
axiom RemoteFromHome : obj → Prop
axiom InOtherCountries : obj → Prop
axiom AppearInCompanyToday : obj → Prop

-- Axioms from the context
axiom A1 : ∀ x : obj, Employee x ∧ SchedulesMeeting x → GoToBuildingToday x
axiom A2 : ∀ x : obj, LunchInBuilding x → SchedulesMeeting x
axiom A3 : ∀ x : obj, Employee x → (LunchInBuilding x ∨ LunchAtHome x)
axiom A4 : ∀ x : obj, Employee x ∧ LunchAtHome x → RemoteFromHome x
axiom A5 : ∀ x : obj, Employee x ∧ InOtherCountries x → RemoteFromHome x
axiom A6 : ∀ x : obj, Manager x → ¬ RemoteFromHome x
axiom A7 : AppearInCompanyToday James ↔ Manager James

-- Some consequences we can derive
theorem remote_implies_not_manager : ∀ x : obj, RemoteFromHome x → ¬ Manager x :=
by
  intro x
  intro hrem
  intro hman
  have h := A6 x hman
  exact h hrem

theorem employee_lunchAtHome_implies_not_manager : ∀ x : obj, Employee x ∧ LunchAtHome x → ¬ Manager x :=
by
  intro x
  intro h
  have hrem : RemoteFromHome x := A4 x h
  exact remote_implies_not_manager x hrem

theorem employee_inOtherCountries_implies_not_manager : ∀ x : obj, Employee x ∧ InOtherCountries x → ¬ Manager x :=
by
  intro x
  intro h
  have hrem : RemoteFromHome x := A5 x h
  exact remote_implies_not_manager x hrem
