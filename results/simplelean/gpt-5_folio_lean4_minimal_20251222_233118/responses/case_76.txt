axiom obj : Type
axiom Hannah : obj

-- Predicates
axiom AtMS : obj → Prop
axiom Works : obj → Prop
axiom Needs : obj → Prop
axiom OrdersTakeout : obj → Prop
axiom Enjoys : obj → Prop
axiom Picky : obj → Prop
axiom SpendsTime : obj → Prop

-- Rules from the context
axiom R1 : ∀ x : obj, AtMS x ∧ Works x → Needs x
axiom R2 : ∀ x : obj, AtMS x ∧ OrdersTakeout x → Works x
axiom R3 : ∀ x : obj, AtMS x → OrdersTakeout x ∨ Enjoys x
axiom R4 : ∀ x : obj, AtMS x ∧ Enjoys x → ¬ Picky x
axiom R5 : ∀ x : obj, AtMS x ∧ Enjoys x → SpendsTime x

-- Facts about Hannah
axiom T1 : AtMS Hannah
axiom T2 : Works Hannah ∧ (Needs Hannah → (¬ Picky Hannah ∧ ¬ Needs Hannah))

-- Useful consequences
theorem hannah_not_needs : ¬ Needs Hannah := by
  intro hN
  have hImp : Needs Hannah → (¬ Picky Hannah ∧ ¬ Needs Hannah) := And.right T2
  have hNN : ¬ Needs Hannah := And.right (hImp hN)
  exact hNN hN

theorem hannah_not_picky : ¬ Picky Hannah := by
  have hAt : AtMS Hannah := T1
  have hWork : Works Hannah := And.left T2
  have hNeeds : Needs Hannah := R1 Hannah ⟨hAt, hWork⟩
  have hImp : Needs Hannah → (¬ Picky Hannah ∧ ¬ Needs Hannah) := And.right T2
  exact And.left (hImp hNeeds)

theorem hannah_atms_and_not_picky : AtMS Hannah ∧ ¬ Picky Hannah := by
  exact And.intro T1 hannah_not_picky
