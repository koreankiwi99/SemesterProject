<lean>
axiom obj : Type
axiom DTC : obj  -- Dried Thai chilies
axiom Spicy : obj → Prop
axiom BakedSweet : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom BBM : obj → Prop  -- products from Baked by Melissa

axiom A1 : ∀ x : obj, BakedSweet x → ¬ Spicy x                  -- No baked sweets are spicy
axiom A2 : ∀ x : obj, Cupcake x → BakedSweet x                  -- All cupcakes are baked sweets
axiom A3 : ∀ x : obj, MalaHotpot x → Spicy x                    -- All mala hotpots are spicy
axiom A4 : ∀ x : obj, BBM x → Cupcake x                         -- All products from Baked by Melissa are cupcakes
axiom A5 : Spicy DTC ∨ MalaHotpot DTC ∨ ¬ BakedSweet DTC        -- DTC is spicy or mala hotpot or not baked sweet

theorem dtc_not_bbm : ¬ BBM DTC := by
  intro hbbm
  have hbs : BakedSweet DTC := A2 DTC (A4 DTC hbbm)
  cases A5 with
  | inl hSpicy =>
      have hnsp : ¬ Spicy DTC := A1 DTC hbs
      exact hnsp hSpicy
  | inr hrest =>
      cases hrest with
      | inl hmala =>
          have hSp : Spicy DTC := A3 DTC hmala
          have hnsp : ¬ Spicy DTC := A1 DTC hbs
          exact hnsp hSp
      | inr hnotbs =>
          exact hnotbs hbs
</lean>

ANSWER: True
