axiom Tom_not_R : ¬ RioGrande Tom

-- Every wild turkey is one of the six listed types
axiom WT_exhaustive :
  ∀ x : obj, WildTurkey x →
    (Eastern x ∨ (Osceola x ∨ (Goulds x ∨ (Merriams x ∨ (RioGrande x ∨ Ocellated x)))))

theorem tom_is_ocellated : Ocellated Tom := by
  have h := WT_exhaustive Tom TomWild
  cases h with
  | inl hE =>
      exact False.elim (Tom_not_E hE)
  | inr h1 =>
      cases h1 with
      | inl hOs =>
          exact False.elim (Tom_not_Os hOs)
      | inr h2 =>
          cases h2 with
          | inl hG =>
              exact False.elim (Tom_not_G hG)
          | inr h3 =>
              cases h3 with
              | inl hM =>
                  exact False.elim (Tom_not_M hM)
              | inr h4 =>
                  cases h4 with
                  | inl hR =>
                      exact False.elim (Tom_not_R hR)
                  | inr hOc =>
                      exact hOc
