<lean>
axiom Person : Type
axiom Peter : Person

axiom Spill : Person → Prop
axiom Tidy : Person → Prop
axiom Cleanly : Person → Prop
axiom CFG : Person → Prop
axiom Value : Person → Prop
axiom Family : Person → Prop

-- If people spill a lot of food on their clothing, then they are not notably tidy.
axiom A1 : ∀ x : Person, Spill x → ¬ Tidy x
-- All clumsy foodies who go out frequently... spill a lot of food on their clothing.
axiom A2 : ∀ x : Person, CFG x → Spill x
-- If people are cleanly, then they are notably tidy.
axiom A3 : ∀ x : Person, Cleanly x → Tidy x
-- If people value order and spotlessness, then they are cleanly.
axiom A4 : ∀ x : Person, Value x → Cleanly x
-- All people whose families prioritize order and spotlessness themselves value order and spotlessness.
axiom A5 : ∀ x : Person, Family x → Value x

-- Peter is either (spills a lot ∧ cleanly) or (doesn't spill a lot ∧ not cleanly).
axiom Peter_disj : (Spill Peter ∧ Cleanly Peter) ∨ (¬ Spill Peter ∧ ¬ Cleanly Peter)

-- From the rules, Peter cannot both spill a lot and be cleanly.
theorem not_spill_and_cleanly_impossible :
  ¬ (Spill Peter ∧ Cleanly Peter) := by
  intro h
  have ht : Tidy Peter := A3 Peter h.right
  have hnt : ¬ Tidy Peter := A1 Peter h.left
  exact hnt ht

-- Therefore, from Peter_disj we can conclude Peter neither spills a lot nor is cleanly.
theorem not_spill_and_not_cleanly_Peter : ¬ Spill Peter ∧ ¬ Cleanly Peter := by
  have hcontra : ¬ (Spill Peter ∧ Cleanly Peter) := not_spill_and_cleanly_impossible
  cases Peter_disj with
  | inl h =>
    exact False.elim (hcontra h)
  | inr h =>
    exact h
</lean>
