axiom obj : Type

-- Entities
axiom PicurisMountains : obj
axiom HardingPegmatiteMine : obj
axiom JuanDeOnate : obj

-- Predicates
axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

-- Axioms from the text
axiom A1 : MountainRange PicurisMountains
axiom A2 : InNewMexico PicurisMountains ∨ InTexas PicurisMountains
axiom A3 : Visited JuanDeOnate PicurisMountains
axiom A4 : Mine HardingPegmatiteMine
axiom A5 : LocatedIn HardingPegmatiteMine PicurisMountains
axiom A6 : Donated HardingPegmatiteMine
axiom A7 : ∀ r : obj, MountainRange r → InTexas r → ¬(∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m)

-- Picuris Mountains are not in Texas (otherwise it would contradict A7)
theorem not_in_texas_picuris : ¬ InTexas PicurisMountains := by
  intro hTX
  have hRange : MountainRange PicurisMountains := A1
  have exMine : ∃ m : obj, Mine m ∧ LocatedIn m PicurisMountains ∧ Donated m := by
    exact Exists.intro HardingPegmatiteMine (And.intro A4 (And.intro A5 A6))
  have hNo : ¬(∃ m : obj, Mine m ∧ LocatedIn m PicurisMountains ∧ Donated m) := A7 PicurisMountains hRange hTX
  exact hNo exMine

-- Hence Picuris Mountains are in New Mexico
theorem in_new_mexico_picuris : InNewMexico PicurisMountains := by
  have hOr : InNewMexico PicurisMountains ∨ InTexas PicurisMountains := A2
  have hNotTX : ¬ InTexas PicurisMountains := not_in_texas_picuris
  cases hOr with
  | inl hNM => exact hNM
  | inr hTX =>
    have : False := hNotTX hTX
    exact False.elim this

-- Therefore, the Harding Pegmatite Mine is located in a mountain range in New Mexico
theorem hp_located_in_nm_range : ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingPegmatiteMine r := by
  refine Exists.intro PicurisMountains ?_
  exact And.intro A1 (And.intro in_new_mexico_picuris A5)
