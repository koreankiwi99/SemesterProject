axiom obj : Type
axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

axiom Man : obj → Prop
axiom InMClass : obj → Prop
axiom Taller : obj → obj → Prop
axiom CanBlock : obj → obj → Prop
axiom JumpShoot : obj → Prop
axiom GreatShooter : obj → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom A1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom A2 : Man Michael ∧ ∀ z : obj, InMClass z → z ≠ Michael → Taller Michael z

-- Transitivity of taller-than.
axiom A3 : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom A4 : Man Peter ∧ Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom A5 : ∀ x : obj, ¬ JumpShoot x → CanBlock Michael x

-- Michael cannot block Windy's shooting.
axiom A6 : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom A7 : ∀ x : obj, JumpShoot x → GreatShooter x

-- From the given information, Peter is taller than every other member of Michael's class.
theorem peter_taller_than_classmates :
  ∀ z : obj, InMClass z → z ≠ Michael → Taller Peter z :=
by
  intro z hz hzne
  have hmz : Taller Michael z := (And.right A2) z hz hzne
  have hpm : Taller Peter Michael := And.right A4
  exact A3 Peter Michael z hpm hmz

-- Target claim: Peter is shorter than a man in Michael's class.
def peter_shorter_than_a_man_in_michaels_class : Prop :=
  ∃ m : obj, Man m ∧ InMClass m ∧ Taller m Peter
