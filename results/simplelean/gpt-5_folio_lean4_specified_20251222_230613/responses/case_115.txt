<lean>
axiom Person : Type

axiom Michael : Person
axiom Peter : Person
axiom Windy : Person

axiom Man : Person → Prop
axiom InMichaelsClass : Person → Prop
axiom Taller : Person → Person → Prop
axiom Blocks : Person → Person → Prop
axiom JumpsWhenShooting : Person → Prop
axiom GreatShooter : Person → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom A1 : ∀ x y : Person, Man x → Man y → Taller x y → Blocks x y

-- Michael is a man who is taller than everyone else in his class.
axiom A2 : Man Michael
axiom A3 : ∀ p : Person, InMichaelsClass p → Taller Michael p

-- Transitivity of taller
axiom A4 : ∀ x y z : Person, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom A5 : Man Peter
axiom A6 : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom A7 : ∀ p : Person, ¬ JumpsWhenShooting p → Blocks Michael p

-- Michael cannot block Windy's shooting.
axiom A8 : ¬ Blocks Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom A9 : ∀ p : Person, JumpsWhenShooting p → GreatShooter p

-- Some derived facts (not strictly needed for the final question)
theorem peter_blocks_michael : Blocks Peter Michael :=
  A1 Peter Michael A5 A2 A6

-- From A7 and A8 we can derive only double-negation of Windy jumping
theorem not_not_jumps_windy : ¬ ¬ JumpsWhenShooting Windy :=
  fun hnj => A8 (A7 Windy hnj)
