axiom R5 : ¬ TeslaModel3 JohnCar → (¬ TeslaModel3 JohnCar ∨ SingleSeatElectricCar JohnCar)

-- A derived consequence: single-seat electric cars cannot be Tesla Model 3
theorem single_seat_not_tesla : ∀ x : obj, SingleSeatElectricCar x → ¬ TeslaModel3 x := by
  intro x hx htm3
  have hmulti : CanTransportMultiplePassengers x := R2 x htm3
  have hnotone : ¬ OneSeater x := R1 x hmulti
  have hone : OneSeater x := R3 x hx
  exact hnotone hone
