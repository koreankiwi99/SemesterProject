<lean>
axiom Person : Type
axiom Ekaterina : Person
axiom Vladimir : Person

axiom CanRegisterUS : Person → Prop
axiom CanParticipateUS2024 : Person → Prop
axiom USCitizen : Person → Prop
axiom TaiwanCitizen : Person → Prop
axiom RussianOfficial : Person → Prop
axiom GazpromManager : Person → Prop

axiom A1 : ∀ x : Person, CanRegisterUS x → CanParticipateUS2024 x
axiom A2 : ∀ x : Person, USCitizen x → CanRegisterUS x
axiom A3 : ∀ x : Person, USCitizen x ∨ TaiwanCitizen x
axiom A4 : ∀ x : Person, RussianOfficial x → ¬ TaiwanCitizen x
axiom A5 : ¬ TaiwanCitizen Vladimir ∧ ¬ GazpromManager Vladimir
axiom A6 : CanRegisterUS Ekaterina ∨ RussianOfficial Ekaterina

theorem ekaterina_can_register : CanRegisterUS Ekaterina := by
  cases A6 with
  | inl hreg => exact hreg
  | inr hro =>
      have hnotTW : ¬ TaiwanCitizen Ekaterina := A4 Ekaterina hro
      cases A3 Ekaterina with
      | inl hUS => exact A2 Ekaterina hUS
      | inr hTW => exact False.elim (hnotTW hTW)

theorem vladimir_can_participate : CanParticipateUS2024 Vladimir := by
  have hnotTW : ¬ TaiwanCitizen Vladimir := And.left A5
  have hUS : USCitizen Vladimir := by
    cases A3 Vladimir with
    | inl h => exact h
    | inr hTW => exact False.elim (hnotTW hTW)
  have hreg : CanRegisterUS Vladimir := A2 Vladimir hUS
  exact A1 Vladimir hreg

theorem conjunction_true : CanRegisterUS Ekaterina ∧ CanParticipateUS2024 Vladimir :=
  And.intro ekaterina_can_register vladimir_can_participate

theorem neg_statement_false : ¬ (¬ (CanRegisterUS Ekaterina ∧ CanParticipateUS2024 Vladimir)) := by
  intro h
  apply h
  exact conjunction_true
</lean>
