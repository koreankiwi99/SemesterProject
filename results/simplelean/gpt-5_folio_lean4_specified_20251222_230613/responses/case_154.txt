<lean>
axiom obj : Type
axiom mixture : obj

axiom Hydrocarbon : obj → Prop
axiom Organic : obj → Prop
axiom Alkane : obj → Prop
axiom Chemical : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom OneElement : obj → Prop

-- All hydrocarbons are organic compounds.
axiom A1 : ∀ x : obj, Hydrocarbon x → Organic x
-- All alkanes are hydrocarbons.
axiom A2 : ∀ x : obj, Alkane x → Hydrocarbon x
-- All organic compounds are chemical compounds.
axiom A3 : ∀ x : obj, Organic x → Chemical x
-- All organic compounds contain carbon.
axiom A4 : ∀ x : obj, Organic x → ContainsCarbon x
-- No chemical compounds contain only one element.
axiom A5 : ∀ x : obj, Chemical x → ¬ OneElement x
-- The mixture either both is a chemical compound and contains only one element, or neither.
axiom A6 : (Chemical mixture ∧ OneElement mixture) ∨ (¬ Chemical mixture ∧ ¬ OneElement mixture)

theorem mixture_neither_chem_nor_alkane :
  (OneElement mixture ∨ ContainsCarbon mixture) → (¬ Chemical mixture ∧ ¬ Alkane mixture) := by
  -- First, deduce that the mixture is neither a chemical compound nor one element.
  have hneither : (¬ Chemical mixture ∧ ¬ OneElement mixture) := by
    apply Or.elim A6
    · intro hboth
      have hc : Chemical mixture := hboth.left
      have hone : OneElement mixture := hboth.right
      have hnotone : ¬ OneElement mixture := A5 mixture hc
      exact False.elim (hnotone hone)
    · intro h
      exact h
  have hnotchem : ¬ Chemical mixture := hneither.left
  have hnotone : ¬ OneElement mixture := hneither.right
  -- Show that Alkane implies Chemical for the mixture, then use hnotchem to conclude ¬Alkane.
  have alk_to_chem : Alkane mixture → Chemical mixture :=
    fun ha => A3 mixture (A1 mixture (A2 mixture ha))
  have hnotalk : ¬ Alkane mixture := by
    intro ha
    exact hnotchem (alk_to_chem ha)
  intro _
  exact And.intro hnotchem hnotalk
</lean>

ANSWER: True
