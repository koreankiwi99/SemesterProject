<lean>
axiom obj : Type
axiom Tom : obj

axiom NiceToAnimals : obj → Prop
axiom MeanToAnimals : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop

-- No one nice to animals is also mean to animals.
axiom A1 : ∀ x : obj, NiceToAnimals x → ¬ MeanToAnimals x
-- Some grumpy people are mean to animals.
axiom A2 : ∃ x : obj, Grumpy x ∧ MeanToAnimals x
-- All animal lovers are nice to animals.
axiom A3 : ∀ x : obj, AnimalLover x → NiceToAnimals x
-- All pet owners love animals.
axiom A4 : ∀ x : obj, PetOwner x → AnimalLover x
-- Tom is a pet owner.
axiom A5 : PetOwner Tom

-- We show the given statement is false by proving the negation of the implication.
theorem tom_statement_false :
  ¬ ( (¬ (Grumpy Tom ∧ MeanToAnimals Tom)) → (¬ Grumpy Tom ∧ ¬ AnimalLover Tom) ) := by
  have hL : AnimalLover Tom := A4 Tom A5
  have hNM : ¬ MeanToAnimals Tom := A1 Tom (A3 Tom hL)
  have hP : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) := by
    intro h
    exact hNM h.right
  have hNotQ : ¬ (¬ Grumpy Tom ∧ ¬ AnimalLover Tom) := by
    intro h
    exact h.right hL
  intro hImp
  have hQ : (¬ Grumpy Tom ∧ ¬ AnimalLover Tom) := hImp hP
  exact hNotQ hQ
</lean>

ANSWER: False
