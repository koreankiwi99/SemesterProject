axiom obj : Type
axiom American : obj → Prop
axiom ProfBB : obj → Prop
axiom CanJumpShot : obj → Prop
axiom LeapsStraight : obj → Prop
axiom ActivatesLegMuscles : obj → Prop
axiom Yuri : obj

-- Premises
axiom some_prof_not_american : ∃ x : obj, ProfBB x ∧ ¬ American x
axiom all_prof_can_jump : ∀ x : obj, ProfBB x → CanJumpShot x
axiom jump_implies_leap : ∀ x : obj, CanJumpShot x → LeapsStraight x
axiom leap_implies_activate : ∀ x : obj, LeapsStraight x → ActivatesLegMuscles x
axiom yuri_not_activate : ¬ ActivatesLegMuscles Yuri

-- Derived facts
theorem activate_of_prof : ∀ x : obj, ProfBB x → ActivatesLegMuscles x :=
  fun x hx => leap_implies_activate x (jump_implies_leap x (all_prof_can_jump x hx))

theorem yuri_not_prof : ¬ ProfBB Yuri :=
  fun hprof => yuri_not_activate (activate_of_prof Yuri hprof)

theorem not_american_prof_yuri : ¬ (American Yuri ∧ ProfBB Yuri) := by
  intro h
  have hprof : ProfBB Yuri := And.right h
  have hact : ActivatesLegMuscles Yuri := activate_of_prof Yuri hprof
  exact yuri_not_activate hact
