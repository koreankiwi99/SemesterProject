axiom Animal : obj → Prop
axiom Dog : obj → Prop
axiom Bark : obj → Prop
axiom Jump : obj → Prop
axiom HasLegs : obj → Prop
axiom Terricolous : obj → Prop
axiom Like : obj → obj → Prop

-- Jane does not like any animals that bark.
axiom H1 : ∀ x : obj, Animal x → Bark x → ¬ Like Jane x
-- All dogs are animals that bark.
axiom H2 : ∀ x : obj, Dog x → (Animal x ∧ Bark x)
-- Jane likes all animals that jump.
axiom H3 : ∀ x : obj, Animal x → Jump x → Like Jane x
-- If an animal has legs, then it jumps.
axiom H4 : ∀ x : obj, Animal x → HasLegs x → Jump x
-- If an animal is terricolous, then it has legs.
axiom H5 : ∀ x : obj, Animal x → Terricolous x → HasLegs x
-- KiKi is an animal.
axiom H6 : Animal KiKi
-- If KiKi neither jumps nor has legs, then KiKi is terricolous or has legs.
axiom H7 : (¬ Jump KiKi ∧ ¬ HasLegs KiKi) → (Terricolous KiKi ∨ HasLegs KiKi)

-- A derived fact we can prove from the premises:
theorem kiki_not_both_nonjump_nolegs : ¬ (¬ Jump KiKi ∧ ¬ HasLegs KiKi) := by
  intro h
  have hdisj : Terricolous KiKi ∨ HasLegs KiKi := H7 h
  have hNotLegs : ¬ HasLegs KiKi := And.right h
  cases hdisj with
  | inl hTerr =>
      have hLegs : HasLegs KiKi := H5 KiKi H6 hTerr
      exact hNotLegs hLegs
  | inr hLegs =>
      exact hNotLegs hLegs
