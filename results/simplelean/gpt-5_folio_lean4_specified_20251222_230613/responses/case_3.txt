<lean>
axiom obj : Type

-- Entities
axiom James : obj

-- Predicates
axiom Employee : obj → Prop
axiom Manager : obj → Prop
axiom SchedulesWithCustomers : obj → Prop
axiom LunchInCompany : obj → Prop
axiom LunchAtHome : obj → Prop
axiom RemoteFromHome : obj → Prop
axiom InOtherCountry : obj → Prop
axiom InCompanyToday : obj → Prop

-- Premises
-- 1. All employees who schedule a meeting with their customers will go to the company building today.
axiom A1 : ∀ x : obj, Employee x ∧ SchedulesWithCustomers x → InCompanyToday x
-- 2. Everyone who has lunch in the company building schedules meetings with their customers.
axiom A2 : ∀ x : obj, LunchInCompany x → SchedulesWithCustomers x
-- 3. Employees have lunch either in the company building or at home.
axiom A3 : ∀ x : obj, Employee x → (LunchInCompany x ∨ LunchAtHome x)
-- 4. If an employee has lunch at home, they are working remotely from home.
axiom A4 : ∀ x : obj, Employee x ∧ LunchAtHome x → RemoteFromHome x
-- 5. All employees who are in other countries work remotely from home.
axiom A5 : ∀ x : obj, Employee x ∧ InOtherCountry x → RemoteFromHome x
-- 6. No managers work remotely from home.
axiom A6 : ∀ x : obj, Manager x → ¬ RemoteFromHome x
-- 7. James will appear in the company today iff he is a manager.
axiom A7 : InCompanyToday James ↔ Manager James

-- Derived facts (from the premises)

-- If an employee has lunch in the company, they will go to the company today.
theorem employees_lunch_company_go_company :
  ∀ x : obj, Employee x ∧ LunchInCompany x → InCompanyToday x := by
  intro x h
  have hs : SchedulesWithCustomers x := A2 x h.right
  exact A1 x ⟨h.left, hs⟩

-- If James is an employee and has lunch in the company, then he is a manager.
theorem james_lunch_company_employee_implies_manager :
  Employee James ∧ LunchInCompany James → Manager James := by
  intro h
  have inc : InCompanyToday James := employees_lunch_company_go_company James h
  exact (A7.mp) inc

-- If James is an employee and has lunch at home, then he is not a manager.
theorem james_lunch_home_employee_implies_not_manager :
  Employee James ∧ LunchAtHome James → ¬ Manager James := by
  intro h mJ
  have nr : ¬ RemoteFromHome James := A6 James mJ
  have r : RemoteFromHome James := A4 James h
  exact nr r
