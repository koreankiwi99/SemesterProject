<lean>
axiom obj : Type

-- Predicates and entities
axiom ZahaStyle : obj → Prop
axiom KellyStyle : obj → Prop
axiom BrutalistBuilding : obj → Prop
axiom InterestingGeometry : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop
axiom Adores : obj → Prop
axiom Style : obj → Prop
axiom DesignByMax : obj → Prop

-- Premises
-- 1. All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom A1 : ∀ x : obj, ZahaStyle x ∧ Adores x → InterestingGeometry x

-- 2. No brutalist buildings that Max adores have interesting geometries.
axiom A2 : ∀ x : obj, BrutalistBuilding x ∧ Adores x → ¬ InterestingGeometry x

-- 3. Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom A3 : ∀ x : obj, Style x ∧ Adores x → ZahaStyle x ∨ KellyStyle x

-- 4. All of Kelly Wearstler's design styles that Max adores are evocative.
axiom A4 : ∀ x : obj, KellyStyle x ∧ Adores x → Evocative x

-- 5. All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom A5 : ∀ x : obj, KellyStyle x ∧ Adores x → Dreamy x

-- 6. If a design by Max that he adores has interesting geometries,
--    then the design is a brutalist building and evocative.
axiom A6 : ∀ x : obj, DesignByMax x ∧ Adores x ∧ InterestingGeometry x → BrutalistBuilding x ∧ Evocative x

-- The claim in question: "A design by Max is a brutalist building."
-- We formalize this as the existential statement that there exists a design by Max that is a brutalist building.
def Claim : Prop := ∃ x : obj, DesignByMax x ∧ BrutalistBuilding x
