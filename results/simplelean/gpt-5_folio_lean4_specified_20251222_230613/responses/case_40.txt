axiom obj : Type
axiom Adores : obj → Prop
axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Interesting : obj → Prop
axiom Brutalist : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop
axiom DesignByMax : obj → Prop

axiom A1 : ∀ x : obj, Zaha x ∧ Adores x → Interesting x
axiom A2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ Interesting x
axiom A3 : ∀ x : obj, Adores x → (Zaha x ∨ Kelly x)
axiom A4 : ∀ x : obj, Kelly x ∧ Adores x → Evocative x
axiom A5 : ∀ x : obj, Kelly x ∧ Adores x → Dreamy x
axiom A6 : ∀ x : obj, DesignByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

theorem designs_by_max_that_he_adores_are_evocative_and_dreamy :
  ∀ x : obj, DesignByMax x ∧ Adores x → Evocative x ∧ Dreamy x :=
by
  intro x
  intro hx
  have hxD : DesignByMax x := And.left hx
  have hxA : Adores x := And.right hx
  have notZaha : ¬ Zaha x :=
    by
      intro z
      have intx : Interesting x := A1 x (And.intro z hxA)
      have hBE : Brutalist x ∧ Evocative x := A6 x (And.intro hxD (And.intro hxA intx))
      have hBrut : Brutalist x := And.left hBE
      have notInt : ¬ Interesting x := A2 x (And.intro hBrut hxA)
      exact notInt intx
  have disj : Zaha x ∨ Kelly x := A3 x hxA
  have kellyx : Kelly x :=
    by
      cases disj with
      | inl z =>
          exact False.elim (notZaha z)
      | inr k =>
          exact k
  have ev : Evocative x := A4 x (And.intro kellyx hxA)
  have dr : Dreamy x := A5 x (And.intro kellyx hxA)
  exact And.intro ev dr
