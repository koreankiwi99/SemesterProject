<lean>
inductive Season where
| Spring
| Summer
| Fall
| Winter

axiom Person : Type
axiom Emma : Person
axiom Mia : Person
axiom James : Person

axiom Favorite : Person → Season

axiom Student : Person → Prop
axiom WantsLongVacation : Person → Prop

-- All students who want a long vacation have summer as their favorite season.
axiom rule_students_longvac_summer :
  ∀ x : Person, Student x ∧ WantsLongVacation x → Favorite x = Season.Summer

-- Emma's favorite season is summer.
axiom Emma_fav_summer : Favorite Emma = Season.Summer

-- Mia's favorite season is not the same as Emma's.
axiom Mia_fav_not_same_as_Emma : Favorite Mia ≠ Favorite Emma

-- James wants to have a long vacation.
axiom James_wants_long_vacation : WantsLongVacation James

-- A consequence we can prove: Mia's favorite season is not summer.
theorem mia_not_summer : Favorite Mia ≠ Season.Summer := by
  have h := Mia_fav_not_same_as_Emma
  have he := Emma_fav_summer
  intro hEq
  rw [←he] at hEq
  exact h hEq
