axiom James : Person

-- Predicates
axiom Employee : Person → Prop
axiom Manager : Person → Prop
axiom SchedulesMeetingsWithCustomers : Person → Prop
axiom GoToCompanyBuildingToday : Person → Prop
axiom LunchInCompanyBuilding : Person → Prop
axiom LunchAtHome : Person → Prop
axiom WorksRemotelyFromHome : Person → Prop
axiom InOtherCountries : Person → Prop
axiom AppearsInCompanyToday : Person → Prop

-- Premises
-- 1. All employees who schedule a meeting with their customers will go to the company building today.
axiom ax1 : ∀ x : Person, Employee x → SchedulesMeetingsWithCustomers x → GoToCompanyBuildingToday x

-- 2. Everyone who has lunch in the company building schedules meetings with their customers.
axiom ax2 : ∀ x : Person, LunchInCompanyBuilding x → SchedulesMeetingsWithCustomers x

-- 3. Employees have lunch either in the company building or at home.
axiom ax3 : ∀ x : Person, Employee x → (LunchInCompanyBuilding x ∨ LunchAtHome x)

-- 4. If an employee has lunch at home, they are working remotely from home.
axiom ax4 : ∀ x : Person, Employee x → LunchAtHome x → WorksRemotelyFromHome x

-- 5. All employees who are in other countries work remotely from home.
axiom ax5 : ∀ x : Person, Employee x → InOtherCountries x → WorksRemotelyFromHome x

-- 6. No managers work remotely from home.
axiom ax6 : ∀ x : Person, Manager x → ¬ WorksRemotelyFromHome x

-- 7. James will appear in the company today iff he is a manager.
axiom ax7 : AppearsInCompanyToday James ↔ Manager James

-- Target statement (both forms stated)
def TargetStatement : Prop :=
  ((Manager James ∨ InOtherCountries James) → ¬ (LunchAtHome James ∨ WorksRemotelyFromHome James)) ∧
  ((LunchAtHome James ∨ WorksRemotelyFromHome James) → (¬ Manager James ∧ ¬ InOtherCountries James))
