axiom Lang : Type
axiom French : Lang
axiom Spanish : Lang
axiom German : Lang
axiom Basque : Lang

axiom Romance : Lang → Prop
axiom IndoEuropean : Lang → Prop
axiom Related : Lang → Lang → Prop

axiom Family : Type
axiom InFamily : Lang → Family → Prop

-- All Romance languages are Indo-European languages.
axiom A1 : ∀ x : Lang, Romance x → IndoEuropean x

-- All languages within a language family are related to each other.
axiom A2 : ∀ f : Family, ∀ x y : Lang, InFamily x f → InFamily y f → Related x y

-- Romance languages are a language family (i.e., there is a family whose members are exactly the Romance languages).
axiom A3 : ∃ f : Family, ∀ x : Lang, Romance x ↔ InFamily x f

-- French and Spanish are Romance languages.
axiom A4 : Romance French
axiom A5 : Romance Spanish

-- German is related to Spanish.
axiom A6 : Related German Spanish

-- Basque is not related to any other language.
axiom A7 : ∀ y : Lang, y ≠ Basque → (¬ Related Basque y ∧ ¬ Related y Basque)

-- A derivable fact: any two Romance languages are related.
theorem french_spanish_related : Related French Spanish :=
by
  obtain ⟨f, hf⟩ := A3
  have hF : InFamily French f := (hf French).mp A4
  have hS : InFamily Spanish f := (hf Spanish).mp A5
  exact A2 f French Spanish hF hS
