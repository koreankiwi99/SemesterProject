<lean>
axiom obj : Type

axiom At : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom Picky : obj → Prop
axiom Spend : obj → Prop

-- If people at Mary's school work in student jobs on campus, then they need to earn money.
axiom R1 : ∀ x : obj, At x ∧ Work x → Need x
-- If people at Mary's school order takeout frequently in college, then they work in student jobs on campus.
axiom R2 : ∀ x : obj, At x ∧ Order x → Work x
-- People at Mary's school order takeout frequently in college or enjoy the dining hall meals and recipes.
axiom R3 : ∀ x : obj, At x → (Order x ∨ Enjoy x)
-- If people at Mary's school enjoy the dining hall meals and recipes, then they are not picky eaters.
axiom R4 : ∀ x : obj, At x ∧ Enjoy x → ¬ Picky x
-- If people at Mary's school enjoy the dining hall meals and recipes, then they spend a lot of their time eating and catching up with friends in the campus dining halls.
axiom R5 : ∀ x : obj, At x ∧ Enjoy x → Spend x

axiom Hannah : obj
-- Hannah is at Mary's school.
axiom H_at : At Hannah
-- Hannah works in student jobs on campus and if she needs money then she is neither picky nor needs money.
axiom H_work_and_imp : Work Hannah ∧ (Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah))

theorem hannah_at_not_picky_spend :
  At Hannah ∧ ¬ Picky Hannah ∧ Spend Hannah :=
by
  have hA : At Hannah := H_at
  have hW : Work Hannah := And.left H_work_and_imp
  have hNeed : Need Hannah := R1 Hannah (And.intro hA hW)
  have hImp : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah) := And.right H_work_and_imp
  have hNotPicky : ¬ Picky Hannah := And.left (hImp hNeed)
  have hNotNeed : ¬ Need Hannah := And.right (hImp hNeed)
  have hFalse : False := hNotNeed hNeed
  have hSpend : Spend Hannah := False.elim hFalse
  exact And.intro hA (And.intro hNotPicky hSpend)
</lean>

ANSWER: True
