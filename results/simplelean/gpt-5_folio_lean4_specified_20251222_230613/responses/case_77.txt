<lean>
axiom obj : Type
axiom Hannah : obj

axiom AtSchool : obj → Prop
axiom Work : obj → Prop
axiom NeedMoney : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom NotPicky : obj → Prop
axiom SpendTimeDiningHalls : obj → Prop

-- If people at Mary's school work in student jobs on campus, then they need to earn money to help pay for their college tuition.
axiom A1 : ∀ x : obj, AtSchool x ∧ Work x → NeedMoney x
-- If people at Mary's school order takeout frequently in college, then they work in student jobs on campus.
axiom A2 : ∀ x : obj, AtSchool x ∧ OrderTakeout x → Work x
-- People at Mary's school order takeout frequently in college or enjoy the dining hall meals and recipes.
axiom A3 : ∀ x : obj, AtSchool x → (OrderTakeout x ∨ EnjoyDining x)
-- If people at Mary's school enjoy the dining hall meals and recipes, then they are not picky eaters.
axiom A4 : ∀ x : obj, AtSchool x ∧ EnjoyDining x → NotPicky x
-- If people at Mary's school enjoy the dining hall meals and recipes, then they spend a lot of their time eating and catching up with friends in the campus dining halls.
axiom A5 : ∀ x : obj, AtSchool x ∧ EnjoyDining x → SpendTimeDiningHalls x
-- Hannah is at Mary's school.
axiom A6 : AtSchool Hannah
-- Hannah works in student jobs on campus and if she needs to earn money then she is neither picky nor needs to earn money.
axiom A7 : Work Hannah ∧ (NeedMoney Hannah → (NotPicky Hannah ∧ ¬ NeedMoney Hannah))

theorem hannah_statement :
  AtSchool Hannah ∧ (NotPicky Hannah ∨ ((¬ NotPicky Hannah) → SpendTimeDiningHalls Hannah)) := by
  have hAt : AtSchool Hannah := A6
  have hWork : Work Hannah := And.left A7
  have hNeed : NeedMoney Hannah := A1 Hannah (And.intro hAt hWork)
  have hImp : NeedMoney Hannah → (NotPicky Hannah ∧ ¬ NeedMoney Hannah) := And.right A7
  have hNP : NotPicky Hannah := And.left (hImp hNeed)
  exact And.intro hAt (Or.inl hNP)
</lean>

ANSWER: True
