<lean>
axiom obj : Type

axiom Spicy : obj → Prop
axiom BakedSweet : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBakedByMelissa : obj → Prop

axiom DriedThaiChilies : obj

axiom no_baked_sweets_spicy : ∀ x : obj, BakedSweet x → ¬ Spicy x
axiom cupcakes_baked_sweets : ∀ x : obj, Cupcake x → BakedSweet x
axiom mala_spicy : ∀ x : obj, MalaHotpot x → Spicy x
axiom bbm_cupcakes : ∀ x : obj, FromBakedByMelissa x → Cupcake x
axiom dtc_disj : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

theorem dtc_not_bbm : ¬ FromBakedByMelissa DriedThaiChilies := by
  intro hbbm
  have hcup : Cupcake DriedThaiChilies := bbm_cupcakes DriedThaiChilies hbbm
  have hbs : BakedSweet DriedThaiChilies := cupcakes_baked_sweets DriedThaiChilies hcup
  have hnotspicy : ¬ Spicy DriedThaiChilies := no_baked_sweets_spicy DriedThaiChilies hbs
  have hdisj := dtc_disj
  cases hdisj with
  | inl hspicy =>
      exact hnotspicy hspicy
  | inr hrest =>
      cases hrest with
      | inl hmala =>
          have hsp : Spicy DriedThaiChilies := mala_spicy DriedThaiChilies hmala
          exact hnotspicy hsp
      | inr hnotbs =>
          exact hnotbs hbs
</lean>

ANSWER: True
