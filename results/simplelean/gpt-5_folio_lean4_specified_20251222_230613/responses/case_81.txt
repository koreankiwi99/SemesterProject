<lean>
axiom obj : Type
axiom DriedThaiChilies : obj
axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

axiom no_baked_sweets_spicy : ∀ x : obj, BakedSweet x → ¬ Spicy x
axiom cupcakes_baked_sweets : ∀ x : obj, Cupcake x → BakedSweet x
axiom mala_hotpots_spicy : ∀ x : obj, MalaHotpot x → Spicy x
axiom bbm_products_cupcakes : ∀ x : obj, FromBBM x → Cupcake x
axiom DTC_disj : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

theorem dried_thai_chilies_neither_bbm_nor_baked_sweet :
  ¬ FromBBM DriedThaiChilies ∧ ¬ BakedSweet DriedThaiChilies := by
  -- First show Dried Thai chilies are not baked sweets
  have hbns : ¬ BakedSweet DriedThaiChilies := by
    cases DTC_disj with
    | inl hsp =>
        exact fun hbs => (no_baked_sweets_spicy DriedThaiChilies hbs) hsp
    | inr h2 =>
        cases h2 with
        | inl hmh =>
            have hsp := mala_hotpots_spicy DriedThaiChilies hmh
            exact fun hbs => (no_baked_sweets_spicy DriedThaiChilies hbs) hsp
        | inr hnb =>
            exact hnb
  -- Then show they are not products from Baked by Melissa
  have hnot_bbm : ¬ FromBBM DriedThaiChilies := by
    intro hbbm
    have hcup : Cupcake DriedThaiChilies := bbm_products_cupcakes DriedThaiChilies hbbm
    have hbs : BakedSweet DriedThaiChilies := cupcakes_baked_sweets DriedThaiChilies hcup
    exact hbns hbs
  exact And.intro hnot_bbm hbns
</lean>

ANSWER: True
