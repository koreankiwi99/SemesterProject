<lean>
axiom Person : Type
axiom Tom : Person

axiom NiceToAnimals : Person → Prop
axiom MeanToAnimals : Person → Prop
axiom Grumpy : Person → Prop
axiom AnimalLover : Person → Prop
axiom PetOwner : Person → Prop

-- No one nice to animals is mean to animals.
axiom A1 : ∀ x : Person, NiceToAnimals x → ¬ MeanToAnimals x
-- Some grumpy people are mean to animals. (Not needed for the conclusion.)
axiom A2 : ∃ x : Person, Grumpy x ∧ MeanToAnimals x
-- All animal lovers are nice to animals.
axiom A3 : ∀ x : Person, AnimalLover x → NiceToAnimals x
-- All pet owners love animals.
axiom A4 : ∀ x : Person, PetOwner x → AnimalLover x
-- Tom is a pet owner.
axiom A5 : PetOwner Tom

theorem tom_not_both_grumpy_and_mean : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) := by
  have hLover : AnimalLover Tom := A4 Tom A5
  have hNice : NiceToAnimals Tom := A3 Tom hLover
  have hNotMean : ¬ MeanToAnimals Tom := A1 Tom hNice
  intro hGM
  exact hNotMean (And.right hGM)
</lean>

ANSWER: True
