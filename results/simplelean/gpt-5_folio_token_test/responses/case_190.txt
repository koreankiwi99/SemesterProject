axiom obj : Type
axiom Yuri : obj

axiom ProBB : obj → Prop
axiom American : obj → Prop
axiom JumpShots : obj → Prop
axiom Leaps : obj → Prop
axiom Activate : obj → Prop

-- Context axioms
axiom someProNotAmerican : ∃ x : obj, ProBB x ∧ ¬ American x
axiom pro_implies_jump : ∀ x : obj, ProBB x → JumpShots x
axiom jump_implies_leap : ∀ x : obj, JumpShots x → Leaps x
axiom leap_implies_activate : ∀ x : obj, Leaps x → Activate x
axiom yuri_not_activate : ¬ Activate Yuri

-- We prove the negation of the queried implication:
-- ¬(¬Leaps Yuri → American Yuri ∧ ProBB Yuri)
theorem target_neg :
  ¬(¬ Leaps Yuri → American Yuri ∧ ProBB Yuri) := by
  intro h
  -- From Leaps → Activate and ¬Activate, derive ¬Leaps (Yuri)
  have h_not_leap : ¬ Leaps Yuri := by
    intro hleap
    have hact : Activate Yuri := leap_implies_activate Yuri hleap
    exact yuri_not_activate hact
  -- Use the implication to get the consequent
  have hB : American Yuri ∧ ProBB Yuri := h h_not_leap
  -- From ProBB → JumpShots → Leaps, derive Leaps Yuri
  have hleap' : Leaps Yuri := by
    have hpro : ProBB Yuri := hB.right
    have hjump : JumpShots Yuri := pro_implies_jump Yuri hpro
    exact jump_implies_leap Yuri hjump
  -- Contradiction with ¬Leaps Yuri
  exact h_not_leap hleap'
