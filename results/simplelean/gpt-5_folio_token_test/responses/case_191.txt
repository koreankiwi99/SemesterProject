<lean>
axiom Person : Type
axiom Yuri : Person

axiom ProfessionalBasketballPlayer : Person → Prop
axiom American : Person → Prop
axiom CanDoJumpShots : Person → Prop
axiom LeapsStraightUp : Person → Prop
axiom ActivatesLegMuscles : Person → Prop

-- Some professional basketball players are not American nationals.
axiom some_non_american_pro : ∃ x : Person, ProfessionalBasketballPlayer x ∧ ¬ American x

-- All professional basketball players can do jump shots.
axiom pro_can_jump : ∀ x : Person, ProfessionalBasketballPlayer x → CanDoJumpShots x

-- If someone can jump shots, they leap straight into the air.
axiom jump_implies_leap : ∀ x : Person, CanDoJumpShots x → LeapsStraightUp x

-- If someone leaps straight into the air, they activate their leg muscles.
axiom leap_implies_activate : ∀ x : Person, LeapsStraightUp x → ActivatesLegMuscles x

-- Yuri does not activate his leg muscles.
axiom yuri_not_activate : ¬ ActivatesLegMuscles Yuri

-- From being a professional basketball player to activating leg muscles (by chaining).
lemma pro_implies_activate : ∀ x : Person, ProfessionalBasketballPlayer x → ActivatesLegMuscles x := by
  intro x
  intro hx
  have hj : CanDoJumpShots x := pro_can_jump x hx
  have hl : LeapsStraightUp x := jump_implies_leap x hj
  have ha : ActivatesLegMuscles x := leap_implies_activate x hl
  exact ha

-- Therefore, Yuri is not a professional basketball player.
lemma yuri_not_pro : ¬ ProfessionalBasketballPlayer Yuri := by
  intro hpro
  have ha : ActivatesLegMuscles Yuri := pro_implies_activate Yuri hpro
  exact yuri_not_activate ha

-- Target statement: If Yuri is not an American professional basketball player, then Yuri is a professional basketball player.
-- Formalized antecedent as ¬(American Yuri ∧ ProfessionalBasketballPlayer Yuri).
-- We prove the negation of the target implication.
theorem target_false :
  ¬ ( (¬ (American Yuri ∧ ProfessionalBasketballPlayer Yuri)) → ProfessionalBasketballPlayer Yuri ) := by
  intro himp
  have hnp : ¬ ProfessionalBasketballPlayer Yuri := yuri_not_pro
  have hant : ¬ (American Yuri ∧ ProfessionalBasketballPlayer Yuri) := by
    intro hconj
    exact hnp (And.right hconj)
  have hp : ProfessionalBasketballPlayer Yuri := himp hant
  exact hnp hp
</lean>

ANSWER: False
