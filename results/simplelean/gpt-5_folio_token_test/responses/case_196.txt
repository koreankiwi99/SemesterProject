-- A derived fact: it's not the case that KiKi both doesn't jump and doesn't have legs.
theorem not_both_not_jumps_and_not_legs : ¬ (¬ Jumps KiKi ∧ ¬ Legs KiKi) := by
  intro h
  have hdisj : Terricolous KiKi ∨ Legs KiKi := A7 h
  cases hdisj with
  | inl hT =>
      have hL : Legs KiKi := A5 KiKi (And.intro A6 hT)
      exact h.right hL
  | inr hL =>
      exact h.right hL

-- From being a dog, KiKi would bark; but we have no proof KiKi is a dog.

  intro hdog
  have hpair : Animal KiKi ∧ Barks KiKi := A2 KiKi hdog
  exact hpair.right
