<lean>
axiom show : Type
axiom StrangerThings : show
axiom BlackMirror : show

axiom Popular : show → Prop
axiom NetflixShow : show → Prop
axiom BingeKaren : show → Prop
axiom DownloadKaren : show → Prop
axiom ShareWithLisa : show → Prop

-- If a Netflix show is popular, Karen will binge-watch it.
axiom R1 : ∀ s : show, NetflixShow s → Popular s → BingeKaren s
-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom R2 : ∀ s : show, NetflixShow s → (BingeKaren s ↔ DownloadKaren s)
-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom R3 : ∀ s : show, NetflixShow s → BingeKaren s → ShareWithLisa s

-- "Stranger Things" is a popular Netflix show.
axiom T_ST_net : NetflixShow StrangerThings
axiom T_ST_pop : Popular StrangerThings

-- "Black Mirror" is a Netflix show; Karen does not download it.
axiom T_BM_net : NetflixShow BlackMirror
axiom T_BM_not : ¬ DownloadKaren BlackMirror

theorem karen_shares_stranger_things : ShareWithLisa StrangerThings := by
  have hb : BingeKaren StrangerThings := R1 StrangerThings T_ST_net T_ST_pop
  exact R3 StrangerThings T_ST_net hb
</lean>

ANSWER: True
