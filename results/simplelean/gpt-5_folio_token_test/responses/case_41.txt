<lean>
axiom obj : Type

-- Predicates on designs
axiom ByMax : obj → Prop
axiom Adores : obj → Prop
axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Brutalist : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- Axioms from the textual context
-- 1. All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom A1 : ∀ x : obj, Zaha x ∧ Adores x → Interesting x

-- 2. No brutalist buildings that Max adores have interesting geometries.
axiom A2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ Interesting x

-- 3. Every style that Max adores is either Zaha Hadid's or Kelly Wearstler's.
axiom A3 : ∀ x : obj, Adores x → Zaha x ∨ Kelly x

-- 4. All of Kelly Wearstler's design styles that Max adores are evocative.
axiom A4 : ∀ x : obj, Kelly x ∧ Adores x → Evocative x

-- 5. All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom A5 : ∀ x : obj, Kelly x ∧ Adores x → Dreamy x

-- 6. If a design by Max that he adores has interesting geometries, then it is brutalist and evocative.
axiom A6 : ∀ x : obj, ByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- Target: A design by Max (that he adores) is either evocative or dreamy.
theorem max_design_evocative_or_dreamy :
  ∀ x : obj, ByMax x ∧ Adores x → Evocative x ∨ Dreamy x :=
by
  intro x
  intro hx
  have hBy : ByMax x := And.left hx
  have hAd : Adores x := And.right hx
  have hZA : Zaha x ∨ Kelly x := A3 x hAd
  cases hZA with
  | inl hZ =>
      have hInt : Interesting x := A1 x (And.intro hZ hAd)
      have hBE : Brutalist x ∧ Evocative x := A6 x (And.intro hBy (And.intro hAd hInt))
      exact Or.inl hBE.right
  | inr hK =>
      have hEv : Evocative x := A4 x (And.intro hK hAd)
      exact Or.inl hEv
</lean>

ANSWER: True
