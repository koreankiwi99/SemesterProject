<lean>
axiom obj : Type

axiom NineteenEightyFour : obj

axiom DigitalMedia : obj → Prop
axiom AnalogMedia : obj → Prop
axiom PrintedText : obj → Prop
axiom StreamingService : obj → Prop
axiom HardcoverBook : obj → Prop

-- No digital media are analog.
axiom A1 : ∀ x : obj, DigitalMedia x → ¬ AnalogMedia x
-- Every printed text is analog media.
axiom A2 : ∀ x : obj, PrintedText x → AnalogMedia x
-- All streaming services are digital media.
axiom A3 : ∀ x : obj, StreamingService x → DigitalMedia x
-- If an object is a hardcover book, then it is printed text.
axiom A4 : ∀ x : obj, HardcoverBook x → PrintedText x
-- If 1984 is a streaming service, then 1984 is a hardcover book.
axiom A5 : StreamingService NineteenEightyFour → HardcoverBook NineteenEightyFour

theorem not_streaming_1984 : ¬ StreamingService NineteenEightyFour :=
by
  intro hss
  have hhard : HardcoverBook NineteenEightyFour := A5 hss
  have hprint : PrintedText NineteenEightyFour := A4 NineteenEightyFour hhard
  have hanalog : AnalogMedia NineteenEightyFour := A2 NineteenEightyFour hprint
  have hdig : DigitalMedia NineteenEightyFour := A3 NineteenEightyFour hss
  have hnotanalog : ¬ AnalogMedia NineteenEightyFour := A1 NineteenEightyFour hdig
  exact hnotanalog hanalog
</lean>

ANSWER: True
