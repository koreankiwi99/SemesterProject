<lean>
axiom obj : Type

-- Entities
axiom SirDigby : obj
axiom Nemesis : obj
axiom SADC : obj  -- The Surprising Adventures of Sir Digby Chicken Caesar

-- Predicates
axiom SuperheroMovie : obj → Prop
axiom NamedAfter : obj → obj → Prop
axiom InMovie : obj → obj → Prop
axiom GoodGuy : obj → Prop
axiom BadGuy : obj → Prop
axiom Fights : obj → obj → Prop
axiom Wins : obj → Prop

-- Context axioms
-- In superhero movies, the good guys always win (for characters in the movie)
axiom Rwin : ∀ m x, SuperheroMovie m → InMovie x m → GoodGuy x → Wins x
-- If a superhero movie is named after a character, that character is a good guy
axiom Rgood : ∀ m c, SuperheroMovie m → NamedAfter m c → GoodGuy c
-- If a superhero movie is named after a character, that character is in the movie
axiom RinMovie : ∀ m c, SuperheroMovie m → NamedAfter m c → InMovie c m
-- Good guys fight bad guys and vice versa
axiom GoodFightBad : ∀ x y, GoodGuy x → Fights x y → BadGuy y
axiom BadFightGood : ∀ x y, BadGuy x → Fights x y → GoodGuy y
-- If somebody wins a fight, the person they are fighting does not win
axiom NoBothWin : ∀ x y, Fights x y → Wins x → ¬ Wins y

-- Facts about this scenario
axiom T1 : SuperheroMovie SADC
axiom T2 : NamedAfter SADC SirDigby
axiom T3 : Fights SirDigby Nemesis

theorem nemesis_not_win : ¬ Wins Nemesis := by
  have hGG : GoodGuy SirDigby := Rgood SADC SirDigby T1 T2
  have hIM : InMovie SirDigby SADC := RinMovie SADC SirDigby T1 T2
  have hWinSD : Wins SirDigby := Rwin SADC SirDigby T1 hIM hGG
  exact NoBothWin SirDigby Nemesis T3 hWinSD
</lean>

ANSWER: True
