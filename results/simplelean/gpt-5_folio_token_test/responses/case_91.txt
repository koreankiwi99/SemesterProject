<lean>
axiom obj : Type
axiom MT : obj
axiom NLP : obj → Prop
axiom Gen : obj → Prop
axiom Under : obj → Prop
axiom TextOut : obj → Prop

-- A natural language processing task is categorized as either generation or understanding (mutually exclusive)
axiom A1 : ∀ x : obj, NLP x → (Gen x ∨ Under x)
axiom A2 : ∀ x : obj, ¬ (Gen x ∧ Under x)

-- If the output of an NLP task is a text sequence, then it is a generation task
axiom A3 : ∀ x : obj, (NLP x ∧ TextOut x) → Gen x

-- Machine translation is an NLP task, and its output is a text sequence
axiom A4 : NLP MT
axiom A5 : TextOut MT

theorem mt_not_understanding : ¬ Under MT := by
  have hgen : Gen MT := A3 MT ⟨A4, A5⟩
  intro hUnder
  exact A2 MT ⟨hgen, hUnder⟩
</lean>

ANSWER: False
