<lean>
axiom obj : Type
axiom Someone : obj

axiom SavedAggressively : obj → Prop
axiom ImprovedCredit : obj → Prop
axiom AffordDownPayment : obj → Prop
axiom QualifyMortgage : obj → Prop
axiom BuyDreamHome : obj → Prop

axiom R1 : ∀ x : obj, SavedAggressively x → AffordDownPayment x
axiom R2 : ∀ x : obj, ImprovedCredit x → QualifyMortgage x
axiom R3 : SavedAggressively Someone ∨ ImprovedCredit Someone
axiom R4 : ∀ x : obj, QualifyMortgage x → BuyDreamHome x

axiom HnotDP : ¬ AffordDownPayment Someone

theorem bought_dream_home : BuyDreamHome Someone := by
  cases R3 with
  | inl hs =>
      have : False := by
        have adp : AffordDownPayment Someone := R1 Someone hs
        exact HnotDP adp
      exact False.elim this
  | inr ic =>
      have qm : QualifyMortgage Someone := R2 Someone ic
      exact R4 Someone qm
</lean>

ANSWER: Yes
