<lean>
axiom Team : Type
axiom Lakers : Team

axiom StrongPerimeterDefense : Team → Prop
axiom ForceManyTurnovers : Team → Prop
axiom ProtectPaint : Team → Prop
axiom LimitInsideScoring : Team → Prop

axiom A1 : ∀ t : Team, StrongPerimeterDefense t → ForceManyTurnovers t
axiom A2 : ∀ t : Team, ProtectPaint t → LimitInsideScoring t
axiom A3 : StrongPerimeterDefense Lakers ∨ ProtectPaint Lakers

theorem lakers_force_turnovers_of_not_limit :
  ¬ LimitInsideScoring Lakers → ForceManyTurnovers Lakers :=
by
  intro hnot
  apply Or.elim A3
  · intro hspd
    exact A1 Lakers hspd
  · intro hpp
    have his : LimitInsideScoring Lakers := A2 Lakers hpp
    have contra : False := hnot his
    exact False.elim contra
</lean>

ANSWER: Yes
