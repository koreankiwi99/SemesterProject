<lean>
axiom obj : Type
axiom Alex : obj

axiom AdvertisesCleverly : obj → Prop
axiom OffersDiscounts : obj → Prop
axiom AttractsMoreCustomers : obj → Prop
axiom IncreasesSales : obj → Prop
axiom ExpandsBusiness : obj → Prop

axiom R1 : ∀ x : obj, AdvertisesCleverly x → AttractsMoreCustomers x
axiom R2 : ∀ x : obj, OffersDiscounts x → IncreasesSales x
axiom R3 : ∀ x : obj, IncreasesSales x → ExpandsBusiness x

-- Either Alex advertises cleverly or offers discounts
axiom D1 : AdvertisesCleverly Alex ∨ OffersDiscounts Alex
-- Alex did not attract more customers
axiom H1 : ¬ AttractsMoreCustomers Alex

theorem alex_expands : ExpandsBusiness Alex := by
  have hnotAdv : ¬ AdvertisesCleverly Alex := by
    intro hadv
    exact H1 (R1 Alex hadv)
  have hOffer : OffersDiscounts Alex := by
    cases D1 with
    | inl hadv =>
        have : False := hnotAdv hadv
        exact False.elim this
    | inr hoff =>
        exact hoff
  have hInc : IncreasesSales Alex := R2 Alex hOffer
  exact R3 Alex hInc
</lean>

ANSWER: Yes
