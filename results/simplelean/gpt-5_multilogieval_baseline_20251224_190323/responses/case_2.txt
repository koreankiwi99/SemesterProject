<lean>
axiom Rain : Prop
axiom Watered : Prop
axiom Hot : Prop
axiom Sprinklers : Prop

axiom R1 : Rain → Watered
axiom R2 : Hot → Sprinklers
axiom R3 : Rain ∨ ¬ Sprinklers
axiom T1 : Hot

theorem watered_today : Watered := by
  have hs : Sprinklers := R2 T1
  have hr : Rain := by
    cases R3 with
    | inl h => exact h
    | inr hnot =>
      have : False := hnot hs
      exact False.elim this
  exact R1 hr
</lean>

ANSWER: Yes
