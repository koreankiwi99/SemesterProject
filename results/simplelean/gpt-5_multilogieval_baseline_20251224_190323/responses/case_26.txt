<lean>
axiom LostImportantGame : Prop
axiom LostBigMatch : Prop
axiom DisappointedPlayers : Prop
axiom DemoralizedPlayers : Prop

-- When a team loses an important game, the players feel disappointed.
axiom R1 : LostImportantGame → DisappointedPlayers
-- Losing a big match is demoralizing.
axiom R2 : LostBigMatch → DemoralizedPlayers

-- The players are neither disappointed nor demoralized.
axiom H1 : ¬ DisappointedPlayers
axiom H2 : ¬ DemoralizedPlayers

-- Define "losing a significant game" as losing either an important game or a big match.
def LostSignificantGame : Prop := LostImportantGame ∨ LostBigMatch

-- From the players not being disappointed nor demoralized, the team did not lose a significant game.
theorem not_lost_significant : ¬ LostSignificantGame := by
  intro h
  cases h with
  | inl hi =>
      exact H1 (R1 hi)
  | inr hb =>
      exact H2 (R2 hb)
</lean>

ANSWER: No
