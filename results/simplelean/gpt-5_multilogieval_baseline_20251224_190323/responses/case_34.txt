axiom obj : Type
axiom DrLee : obj

axiom Professor : obj → Prop
axiom Publishes : obj → Prop
axiom Expert : obj → Prop
axiom Collaborates : obj → Prop
axiom Recognized : obj → Prop
axiom Curious : obj → Prop
axiom Asks : obj → Prop

-- Publishing research would make them experts
axiom A1 : ∀ x : obj, Publishes x → Expert x
-- Collaborating with top scholars brought recognition
axiom A2 : ∀ x : obj, Collaborates x → Recognized x
-- Professors were neither considered experts nor recognized leaders
axiom A3 : ∀ x : obj, Professor x → (¬ Expert x ∧ ¬ Recognized x)
-- Either research was not published or collaborations did not happen
axiom A4 : (¬ ∃ x : obj, Publishes x) ∨ (¬ ∃ x : obj, Collaborates x)
-- Being curious and asking questions led to publishing original research
axiom A5 : ∀ x : obj, (Curious x ∧ Asks x) → Publishes x
-- Dr. Lee collaborates with top scholars
axiom A6 : Collaborates DrLee

theorem drlee_not_curious_and_asks : ¬ (Curious DrLee ∧ Asks DrLee) := by
  have notPublishes_DrLee : ¬ Publishes DrLee := by
    cases A4 with
    | inl hNoPubl =>
        exact fun hpub => hNoPubl ⟨DrLee, hpub⟩
    | inr hNoCollab =>
        have exCollab : ∃ x : obj, Collaborates x := ⟨DrLee, A6⟩
        have contra : False := hNoCollab exCollab
        exact fun _ => False.elim contra
  exact fun hcurask => notPublishes_DrLee (A5 DrLee hcurask)
