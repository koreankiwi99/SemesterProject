<lean>
axiom WaterDaily : Prop
axiom Drought : Prop
axiom Restrictions : Prop
axiom NotEnoughWater : Prop
axiom GrowWell : Prop
axiom WiltDie : Prop

axiom R1 : WaterDaily → GrowWell
axiom R2 : Drought → Restrictions
axiom R3 : Restrictions → NotEnoughWater
axiom R4 : NotEnoughWater → WiltDie
axiom A1 : WaterDaily ∨ Drought
axiom NGW : ¬GrowWell

theorem plants_wilt_die : WiltDie := by
  cases A1 with
  | inl wd =>
      have gw : GrowWell := R1 wd
      have f : False := NGW gw
      exact False.elim f
  | inr dr =>
      exact R4 (R3 (R2 dr))
</lean>

ANSWER: Yes
