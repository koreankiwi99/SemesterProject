<lean>
axiom Flip : Prop
axiom Gasp : Prop
axiom High : Prop
axiom Clowns : Prop
axiom Laugh : Prop

axiom R1 : Flip → Gasp
axiom R2 : Gasp → High
axiom R3 : Flip ∨ Clowns
axiom R4 : Clowns → Laugh
axiom NotHigh : ¬ High

theorem kids_laughed : Laugh := by
  have flip_imp_high : Flip → High := fun f => R2 (R1 f)
  have not_flip : ¬ Flip := fun f => NotHigh (flip_imp_high f)
  have clowns : Clowns := by
    cases R3 with
    | inl f =>
      exact False.elim (not_flip f)
    | inr c =>
      exact c
  exact R4 clowns
</lean>

ANSWER: Yes
