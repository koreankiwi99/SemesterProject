axiom Wet : Prop              -- The driveway is wet
axiom Rain : Prop             -- It rained recently
axiom Washed : Prop           -- Someone washed their car
axiom WaterDriveway : Prop    -- Water was used on the driveway
axiom WaterOnCar : Prop       -- Water was used on the car
axiom Hose : Prop             -- The hose is uncoiled
axiom Soapy : Prop            -- The sponge is soapy

-- If the driveway is wet, then it rained recently.
axiom A1 : Wet → Rain

-- If someone washed their car, then they used water on the driveway.
axiom A2 : Washed → WaterDriveway

-- Either the driveway is wet, or no water was used on the car, or the driveway is wet even though no water was used.
axiom A3 : (Wet ∨ ¬ WaterOnCar) ∨ (Wet ∧ ¬ WaterOnCar)

-- If the hose is uncoiled, someone was washing their car.
axiom A4 : Hose → Washed

-- Either the hose is uncoiled, or the sponge is soapy (or both).
axiom A5 : Hose ∨ Soapy

-- From "it didn't rain recently" we can derive "the driveway is not wet".
theorem not_wet_of_not_rain (hNR : ¬ Rain) : ¬ Wet := by
  intro hW
  exact hNR (A1 hW)

-- From "it didn't rain recently" and A3 we can derive "no water was used on the car".
theorem not_water_on_car_of_not_rain (hNR : ¬ Rain) : ¬ WaterOnCar := by
  have hNW : ¬ Wet := by
    intro hW
    exact hNR (A1 hW)
  cases A3 with
  | inl h1 =>
      cases h1 with
      | inl hW =>
          have hFalse : False := hNW hW
          intro _
          exact False.elim hFalse
      | inr hNotU =>
          exact hNotU
  | inr hAnd =>
      exact hAnd.right
