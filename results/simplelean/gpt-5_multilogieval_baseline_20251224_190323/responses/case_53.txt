<lean>
axiom obj : Type
axiom Mary : obj

axiom PracticedOften : obj → Prop
axiom TechniqueImproved : obj → Prop
axiom StoodOutHighQuality : obj → Prop
axiom TookArtClass : obj → Prop
axiom LearnedNewMethods : obj → Prop

-- If someone practiced often, their technique improved.
axiom R1 : ∀ x : obj, PracticedOften x → TechniqueImproved x
-- If someone's technique improved, their artwork stood out as high quality.
axiom R2 : ∀ x : obj, TechniqueImproved x → StoodOutHighQuality x
-- Either Mary practiced often or Mary took an art class (or both).
axiom D1 : PracticedOften Mary ∨ TookArtClass Mary
-- If someone takes an art class, they learn new methods and mediums.
axiom R3 : ∀ x : obj, TookArtClass x → LearnedNewMethods x
-- Mary's artwork did not stand out as high quality.
axiom NotStandOutMary : ¬ StoodOutHighQuality Mary

theorem mary_learned_new_methods : LearnedNewMethods Mary := by
  -- From practicing often, Mary's artwork would stand out
  have hPtoS : PracticedOften Mary → StoodOutHighQuality Mary := by
    intro hP
    exact R2 Mary (R1 Mary hP)
  -- Since Mary's artwork did not stand out, she did not practice often
  have hNotP : ¬ PracticedOften Mary := by
    intro hP
    exact NotStandOutMary (hPtoS hP)
  -- From the disjunction and not practicing, conclude she took an art class
  have hClass : TookArtClass Mary := by
    cases D1 with
    | inl hP => exact False.elim (hNotP hP)
    | inr hC => exact hC
  -- Therefore, she learned new methods and mediums
  exact R3 Mary hClass
</lean>

ANSWER: Yes
