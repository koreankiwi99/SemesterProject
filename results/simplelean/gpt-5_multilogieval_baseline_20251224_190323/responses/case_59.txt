<lean>
axiom Person : Type
axiom Emma : Person

axiom LearnedBasicChords : Person → Prop
axiom LearnedStrummingPatterns : Person → Prop
axiom CanPlaySimpleSongs : Person → Prop
axiom CanEntertainFriendsWithGuitar : Person → Prop
axiom TookGuitarLessons : Person → Prop
axiom GainsMusicTheory : Person → Prop

-- If someone learned basic chords and strumming patterns, they can play simple songs.
axiom Rule1 : ∀ x : Person, LearnedBasicChords x ∧ LearnedStrummingPatterns x → CanPlaySimpleSongs x
-- If someone can play simple songs, they can entertain friends with the guitar.
axiom Rule2 : ∀ x : Person, CanPlaySimpleSongs x → CanEntertainFriendsWithGuitar x
-- Either they learned basic chords or they took guitar lessons (or both).
axiom Rule3 : ∀ x : Person, LearnedBasicChords x ∨ TookGuitarLessons x
-- If someone takes guitar lessons, they gain knowledge of music theory.
axiom Rule4 : ∀ x : Person, TookGuitarLessons x → GainsMusicTheory x

-- Emma is unable to entertain friends with the guitar.
axiom Emma_not_entertain : ¬ CanEntertainFriendsWithGuitar Emma

-- Consequences we can derive:
theorem emma_not_play_simple : ¬ CanPlaySimpleSongs Emma := by
  intro h
  have h_ent : CanEntertainFriendsWithGuitar Emma := Rule2 Emma h
  exact Emma_not_entertain h_ent

theorem emma_not_both_learned : ¬ (LearnedBasicChords Emma ∧ LearnedStrummingPatterns Emma) := by
  intro h
  apply emma_not_play_simple
  exact Rule1 Emma h

-- Cannot prove GainsMusicTheory Emma or ¬ GainsMusicTheory Emma from the given axioms.
</lean>
