<lean>
axiom obj : Type
axiom Liz : obj

axiom PreventiveMaintenance : obj → Prop
axiom QualityParts : obj → Prop
axiom LongServiceLife : obj → Prop

-- If preventive maintenance is done, then if quality parts are used, the car will have a long service life.
axiom R1 : ∀ x : obj, PreventiveMaintenance x → (QualityParts x → LongServiceLife x)

-- Facts about Liz's car
axiom QP_Liz : QualityParts Liz
axiom not_Long_Liz : ¬ LongServiceLife Liz

-- From the rule, quality parts, and lack of long service life, conclude no preventive maintenance was done.
theorem no_PM_Liz : ¬ PreventiveMaintenance Liz := by
  intro hPM
  have hImp : QualityParts Liz → LongServiceLife Liz := R1 Liz hPM
  have hLong : LongServiceLife Liz := hImp QP_Liz
  exact not_Long_Liz hLong
</lean>

ANSWER: No
