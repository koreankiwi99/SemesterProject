<lean>
axiom Person : Type
axiom Alex : Person

axiom Relax : Person → Prop
axiom FinishTasks : Person → Prop
axiom ClearMind : Person → Prop
axiom Organized : Person → Prop
axiom ToDoList : Person → Prop

axiom WeekendComing : Prop

-- If it's the weekend coming up, then they relax
axiom R_relax_weekend : ∀ x : Person, WeekendComing → Relax x
-- If someone finishes their tasks, they have a clear mind
axiom R_finish_clear : ∀ x : Person, FinishTasks x → ClearMind x
-- Either the weekend was coming, or Alex did not have a clear mind
axiom R_weekend_or_notclear : WeekendComing ∨ ¬ ClearMind Alex
-- If someone is organized, they finish their tasks
axiom R_org_finish : ∀ x : Person, Organized x → FinishTasks x
-- If they made a to-do list, they are organized
axiom R_list_org : ∀ x : Person, ToDoList x → Organized x

-- Alex does not relax this weekend
axiom H_not_relax : ¬ Relax Alex

theorem alex_not_todo : ¬ ToDoList Alex := by
  intro hlist
  have horg : Organized Alex := R_list_org Alex hlist
  have hfin : FinishTasks Alex := R_org_finish Alex horg
  have hclear : ClearMind Alex := R_finish_clear Alex hfin
  have h_not_weekend : ¬ WeekendComing := by
    intro hw
    have hrelax : Relax Alex := R_relax_weekend Alex hw
    exact H_not_relax hrelax
  have h_not_clear : ¬ ClearMind Alex := by
    refine Or.elim R_weekend_or_notclear ?left ?right
    · intro hw
      intro _
      exact h_not_weekend hw
    · intro hnc
      exact hnc
  exact h_not_clear hclear
</lean>
