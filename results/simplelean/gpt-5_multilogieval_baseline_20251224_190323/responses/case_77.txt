axiom TopYouth : obj → Prop

-- Axioms from the context (interpreted as universal rules)
axiom A1 : ∀ x : obj, PracticedHard x → PlayWell x
axiom A2 : ∀ x : obj, PickedStart x → Talented x
axiom A3 : ∀ x : obj, PracticedHard x ∨ ¬ Talented x
axiom A4 : ∀ x : obj, Natural x → PickedStart x
axiom A5 : ∀ x : obj, TopYouth x → Natural x

-- From the rules, any top youth player will play well
theorem topyouth_playwell : ∀ x : obj, TopYouth x → PlayWell x := by
  intro x hTY
  have hNat : Natural x := A5 x hTY
  have hPick : PickedStart x := A4 x hNat
  have hTal : Talented x := A2 x hPick
  have hDisj : PracticedHard x ∨ ¬ Talented x := A3 x
  have hPrac : PracticedHard x := by
    cases hDisj with
    | inl hP => exact hP
    | inr hNotTal => exact False.elim (hNotTal hTal)
  exact A1 x hPrac

-- Contrapositive for James: if James did not play well, then he was not a top youth player
theorem james_not_playwell_implies_not_top_youth : ¬PlayWell James → ¬TopYouth James := by
  intro hNotPW hTY
  apply hNotPW
  exact topyouth_playwell James hTY
