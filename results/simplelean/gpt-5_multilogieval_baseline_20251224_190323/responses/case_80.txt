<lean>
axiom Rain : Prop
axiom PoolOpen : Prop
axiom KidsSwim : Prop
axiom Game : Prop
axiom Enjoy : Prop
axiom Smiles : Prop

-- If it doesn't rain today, the local softball team will have their game.
axiom H1 : ¬Rain → Game
-- If the swimming pool is open, some kids will go for a swim.
axiom H2 : PoolOpen → KidsSwim
-- Either it won't rain, or the pool will be open (or both).
axiom H3 : ¬Rain ∨ PoolOpen
-- If the softball team has their game, the players will enjoy playing.
axiom H4 : Game → Enjoy
-- If the players enjoy playing, they will have smiles at the end.
axiom H5 : Enjoy → Smiles

theorem smiles_if_no_swim : (¬KidsSwim) → Smiles := by
  intro hNoSwim
  exact Or.elim H3
    (fun hNoRain => H5 (H4 (H1 hNoRain)))
    (fun hPoolOpen =>
      have hSwim : KidsSwim := H2 hPoolOpen
      False.elim (hNoSwim hSwim))
</lean>

ANSWER: Yes
