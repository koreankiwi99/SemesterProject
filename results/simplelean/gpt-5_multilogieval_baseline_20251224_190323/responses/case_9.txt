<lean>
axiom BringDish : Prop
axiom EnoughFood : Prop
axiom Birthday : Prop
axiom Cake : Prop
axiom IceCream : Prop

-- If someone brings a dish to share, there will be enough food for everyone.
axiom R1 : BringDish → EnoughFood
-- If it is a birthday party, there will be cake and ice cream.
axiom R2 : Birthday → Cake ∧ IceCream
-- Either someone brought a dish to share, or there was no birthday cake.
axiom R3 : BringDish ∨ ¬Cake
-- It is Sam's birthday party.
axiom T : Birthday

theorem sam_enough_food : EnoughFood := by
  have hCI : Cake ∧ IceCream := R2 T
  have hCake : Cake := hCI.left
  have hBD : BringDish := by
    cases R3 with
    | inl bd => exact bd
    | inr nCake =>
        exact False.elim (nCake hCake)
  exact R1 hBD
</lean>

ANSWER: Yes
