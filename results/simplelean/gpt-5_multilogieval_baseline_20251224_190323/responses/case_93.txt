<lean>
axiom practiceHard : Prop
axiom prepared : Prop
axiom injured : Prop
axiom canPlay : Prop
axiom wins : Prop
axiom celebrate : Prop

axiom R1 : practiceHard → prepared
axiom R2 : injured → ¬ canPlay
axiom R3 : practiceHard ∨ ¬ injured
axiom R4 : prepared → wins
axiom R5 : wins → celebrate

theorem injured_implies_celebrate : injured → celebrate := by
  intro hi
  have hH : practiceHard := by
    cases R3 with
    | inl hp => exact hp
    | inr hni => exact False.elim (hni hi)
  have hP : prepared := R1 hH
  have hW : wins := R4 hP
  exact R5 hW
</lean>

ANSWER: Yes
