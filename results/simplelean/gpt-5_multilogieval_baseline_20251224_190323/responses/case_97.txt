<lean>
axiom Sub : Prop
axiom Pizza : Prop
axiom Goof : Prop
axiom Hyper : Prop
axiom Learn : Prop
axiom Homework : Prop

axiom A1 : Sub → Goof
axiom A2 : Pizza → Hyper
axiom A3 : Sub ∨ Pizza
axiom A4 : Goof → ¬Learn
axiom A5 : ¬Learn → Homework

theorem not_hyper_implies_homework : ¬Hyper → Homework := by
  intro hNotHyper
  have hNotPizza : ¬Pizza := fun hP => hNotHyper (A2 hP)
  have hSub : Sub := Or.elim A3 (fun hS => hS) (fun hP => False.elim (hNotPizza hP))
  have hGoof : Goof := A1 hSub
  have hNotLearn : ¬Learn := A4 hGoof
  exact A5 hNotLearn
</lean>

ANSWER: Yes
