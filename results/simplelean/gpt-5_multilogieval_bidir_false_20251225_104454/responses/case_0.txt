<lean>
axiom obj : Type
axiom Person : obj

axiom Save : obj → Prop
axiom Improve : obj → Prop
axiom AffordDown : obj → Prop
axiom QualifyMortgage : obj → Prop
axiom BuyDreamHome : obj → Prop

-- If someone saves aggressively, they can afford a down payment.
axiom R1 : ∀ x : obj, Save x → AffordDown x
-- If someone improves their credit, they qualify for a mortgage.
axiom R2 : ∀ x : obj, Improve x → QualifyMortgage x
-- The person either saves aggressively or improves their credit.
axiom R3 : Save Person ∨ Improve Person
-- If someone qualifies for a mortgage, they can buy their dream home.
axiom R4 : ∀ x : obj, QualifyMortgage x → BuyDreamHome x

-- Given: they could not afford a down payment.
axiom Hnot : ¬AffordDown Person

-- From Hnot and R1, we get ¬Save Person
theorem not_save : ¬Save Person := by
  intro hS
  have hA : AffordDown Person := R1 Person hS
  exact Hnot hA

-- From R3 and not_save, we deduce Improve Person
theorem improve_person : Improve Person := by
  cases R3 with
  | inl hS => exact False.elim (not_save hS)
  | inr hI => exact hI

-- Therefore they qualify for a mortgage, and thus can buy their dream home.
theorem buy_person : BuyDreamHome Person := by
  have hQ : QualifyMortgage Person := R2 Person improve_person
  exact R4 Person hQ

-- Hence, we cannot prove ¬BuyDreamHome Person.
