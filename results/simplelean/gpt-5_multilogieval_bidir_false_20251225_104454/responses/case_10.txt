axiom Alex : obj

axiom StudyHard : obj → Prop
axiom GoodGrades : obj → Prop
axiom Participate : obj → Prop
axiom MakeFriends : obj → Prop
axiom Invited : obj → Prop

axiom R1 : ∀ x : obj, StudyHard x → GoodGrades x
axiom R2 : ∀ x : obj, Participate x → MakeFriends x
axiom R3 : ∀ x : obj, MakeFriends x → Invited x

-- The student (Alex) either studied hard or participated
axiom Disj : StudyHard Alex ∨ Participate Alex
-- Given: Alex did not get good grades
axiom NG : ¬GoodGrades Alex

-- From the premises we can actually prove that Alex is invited
theorem invited_Alex : Invited Alex := by
  -- Modus tollens on StudyHard -> GoodGrades and ¬GoodGrades
  have h_not_sh : ¬StudyHard Alex := by
    intro hsh
    have hgg : GoodGrades Alex := R1 Alex hsh
    exact NG hgg
  -- Use the disjunction to get Participate Alex
  have h_part : Participate Alex := by
    cases Disj with
    | inl hsh =>
        have contra : False := h_not_sh hsh
        exact False.elim contra
    | inr hp => exact hp
  -- Chain implications to Invited
  have mf : MakeFriends Alex := R2 Alex h_part
  have inv : Invited Alex := R3 Alex mf
  exact inv
