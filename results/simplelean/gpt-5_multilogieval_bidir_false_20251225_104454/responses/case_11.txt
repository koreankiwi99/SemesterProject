<lean>
axiom obj : Type
axiom Project : obj

axiom NiceDisplay : obj → Prop
axiom JudgesImpressed : obj → Prop
axiom ThoroughResearch : obj → Prop
axiom ExpertsSerious : obj → Prop
axiom Award : obj → Prop

-- If display looks nice, judges will be impressed
axiom R1 : ∀ x : obj, NiceDisplay x → JudgesImpressed x
-- If research is thorough, experts will take them seriously
axiom R2 : ∀ x : obj, ThoroughResearch x → ExpertsSerious x
-- If experts take a project seriously, it will receive an award
axiom R3 : ∀ x : obj, ExpertsSerious x → Award x
-- The project has either a good display or thorough research
axiom R4 : NiceDisplay Project ∨ ThoroughResearch Project
-- Given: judges were not impressed with the display
axiom Tneg : ¬JudgesImpressed Project

-- In fact, the project does receive an award
theorem award_holds : Award Project := by
  have hnotnice : ¬NiceDisplay Project := by
    intro hnice
    exact Tneg (R1 Project hnice)
  have hTR : ThoroughResearch Project := Or.resolve_left R4 hnotnice
  have hExp : ExpertsSerious Project := R2 Project hTR
  exact R3 Project hExp
