axiom OffersDiscounts : obj → Prop
axiom AttractsMoreCustomers : obj → Prop
axiom IncreasesSales : obj → Prop

-- If someone advertises cleverly, they attract more customers.
axiom R1 : ∀ x : obj, AdvertisesCleverly x → AttractsMoreCustomers x
-- If someone offers discounts, they increase sales.
axiom R2 : ∀ x : obj, OffersDiscounts x → IncreasesSales x
-- If someone increases sales, they will expand their business.
axiom R3 : ∀ x : obj, IncreasesSales x → ExpandsBusiness x
-- There exists someone who either advertises cleverly or offers discounts.
axiom ExSome : ∃ x : obj, AdvertisesCleverly x ∨ OffersDiscounts x

-- Given: Alex did not attract more customers.
axiom HAlex_no_attract : ¬ AttractsMoreCustomers Alex

-- From ¬(AttractsMoreCustomers Alex) and (AdvertisesCleverly Alex → AttractsMoreCustomers Alex),
-- we can derive ¬(AdvertisesCleverly Alex).
theorem alex_not_advertises : ¬ AdvertisesCleverly Alex := by
  intro h
  have hA : AttractsMoreCustomers Alex := R1 Alex h
  exact HAlex_no_attract hA
