axiom ReputationGrows : Prop
axiom HostEvents : Prop
axiom MoreCustomers : Prop

axiom R2 : HostEvents → MoreCustomers
-- If customer traffic increases, revenue will improve.
axiom R3 : MoreCustomers → RevenueImproves
-- The owner either provides excellent service or hosts events.
axiom EITHER : GreatService ∨ HostEvents
-- Given: service quality decreased (i.e., not great service).
axiom NotGreat : ¬GreatService

-- In fact, from the premises we can derive that revenue improves.
theorem revenue_improves : RevenueImproves := by
  have host : HostEvents := by
    cases EITHER with
    | inl gs =>
        exact False.elim (NotGreat gs)
    | inr he =>
        exact he
  have mc : MoreCustomers := R2 host
  exact R3 mc
