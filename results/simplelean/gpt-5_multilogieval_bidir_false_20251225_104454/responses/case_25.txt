axiom obj : Type
axiom Emma : obj

axiom Perceptive : obj → Prop
axiom BuildsRapport : obj → Prop
axiom Trusts : obj → Prop
axiom Effective : obj → Prop
axiom Converts : obj → Prop

-- Building rapport leads to trusting relationships
axiom R_rapport_to_trust : ∀ x : obj, BuildsRapport x → Trusts x
-- Effective presentations convert more sales
axiom R_effective_to_convert : ∀ x : obj, Effective x → Converts x
-- Not every salesperson builds trust or converts sales
axiom NotEvery : ∃ x : obj, ¬Trusts x ∧ ¬Converts x
-- If a salesperson is perceptive, they build rapport
axiom R_perceptive_to_rapport : ∀ x : obj, Perceptive x → BuildsRapport x

-- Emma gives effective sales presentations
axiom Emma_effective : Effective Emma

-- Derivable fact: Emma converts more sales
theorem Emma_converts : Converts Emma := by
  have hEff : Effective Emma := Emma_effective
  exact R_effective_to_convert Emma hEff

-- From the given axioms, there is no way to derive ¬Perceptive Emma
-- (no contradiction arises from assuming Perceptive Emma).
-- Hence we cannot prove the negation of the statement "Emma is perceptive".
