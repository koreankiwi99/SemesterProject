<lean>
axiom obj : Type
axiom Team : obj
axiom Players : obj

axiom LostImportant : obj → Prop
axiom LostBig : obj → Prop
axiom Disappointed : obj → Prop
axiom Demoralized : obj → Prop

-- When a team loses an important game, the players feel disappointed.
axiom R1 : LostImportant Team → Disappointed Players
-- Losing a big match is demoralizing.
axiom R2 : LostBig Team → Demoralized Players

-- The players are neither disappointed nor demoralized.
axiom H : ¬Disappointed Players ∧ ¬Demoralized Players

-- Did the team lose a significant game? (i.e., an important game or a big match)
theorem no_significant_loss : ¬(LostImportant Team ∨ LostBig Team) := by
  intro h
  cases h with
  | inl hImp =>
      have d : Disappointed Players := R1 hImp
      exact (And.left H) d
  | inr hBig =>
      have m : Demoralized Players := R2 hBig
      exact (And.right H) m
</lean>

ANSWER: No
