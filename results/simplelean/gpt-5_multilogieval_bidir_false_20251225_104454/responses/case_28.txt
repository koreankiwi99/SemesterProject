<lean>
axiom obj : Type
axiom Jane : obj
axiom Hurt : obj → Prop
axiom SeeDoctor : obj → Prop
axiom Rest : obj → Prop

-- Context: If someone gets injured/hurt, they should see a doctor and rest.
axiom R1 : ∀ x : obj, Hurt x → SeeDoctor x
axiom R2 : ∀ x : obj, Hurt x → Rest x

-- Given statement to evaluate (assumed true for reasoning):
axiom Given : ¬SeeDoctor Jane ∨ ¬Rest Jane

theorem jane_not_hurt : ¬Hurt Jane := by
  intro h
  have sd : SeeDoctor Jane := R1 Jane h
  have rs : Rest Jane := R2 Jane h
  cases Given with
  | inl nsd => exact nsd sd
  | inr nrs => exact nrs rs
</lean>

ANSWER: No
