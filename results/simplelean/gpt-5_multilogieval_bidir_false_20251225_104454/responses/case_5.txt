axiom Understood : obj → Prop
axiom Thrive : obj → Prop

-- If someone understands plant needs, they grow thriving plants
axiom R2 : ∀ x : obj, Understood x → Thrive x
-- Either they learned botany or they sought advice (or both)
axiom R3 : ∀ x : obj, Learned x ∨ Advice x

  have hU : Understood Mary := R1 Mary hL
  have hT : Thrive Mary := R2 Mary hU
  exact T1 hT

  cases h with
  | inl hL =>
    have hnl : ¬Learned Mary := not_learned
    exact False.elim (hnl hL)
  | inr hA =>
    exact hA
