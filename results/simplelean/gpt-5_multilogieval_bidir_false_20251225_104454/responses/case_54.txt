axiom Pumped : Prop
axiom Cheering : Prop
axiom Rival : Prop

axiom R1 : ScoreEarly → Pumped
-- If the crowd gets pumped up, then they start cheering loudly.
axiom R2 : Pumped → Cheering
-- Either the team scores early, or they play their rival team, or both.
axiom R3 : ScoreEarly ∨ Rival
-- If the crowd is pumped up, there will be a lot of booing.
axiom R4 : Pumped → Booing

-- Additional statement to evaluate: The crowd did not start cheering loudly.
axiom TnotC : ¬Cheering

theorem not_score_early : ¬ScoreEarly := fun s => not_pumped (R1 s)

  Or.elim R3
    (fun s => False.elim (not_score_early s))
    (fun r => r)
