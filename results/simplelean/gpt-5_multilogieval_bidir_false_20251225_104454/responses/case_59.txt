axiom obj : Type
axiom Emma : obj

axiom LearnedChords : obj → Prop
axiom LearnedStrum : obj → Prop
axiom PlaySimple : obj → Prop
axiom Entertain : obj → Prop
axiom Lessons : obj → Prop
axiom MusicTheory : obj → Prop

-- If someone learned basic chords and strumming patterns, they could play simple songs.
axiom R1 : ∀ x : obj, LearnedChords x ∧ LearnedStrum x → PlaySimple x
-- If someone could play simple songs, they would be able to entertain friends with the guitar.
axiom R2 : ∀ x : obj, PlaySimple x → Entertain x
-- Either they learned basic chords, or they took guitar lessons, or both.
axiom R3 : ∀ x : obj, LearnedChords x ∨ Lessons x
-- If someone takes guitar lessons, they gain knowledge of music theory.
axiom R4 : ∀ x : obj, Lessons x → MusicTheory x

-- Given statement to evaluate: Emma is unable to entertain friends with the guitar.
axiom Emma_not_entertain : ¬Entertain Emma

-- From R2 (PlaySimple → Entertain), we get its contrapositive: ¬Entertain → ¬PlaySimple
theorem notPlay_of_notEntertain : ∀ x : obj, ¬Entertain x → ¬PlaySimple x := by
  intro x hNotEnt
  intro hPlay
  exact hNotEnt (R2 x hPlay)

-- Therefore, Emma cannot play simple songs.
theorem Emma_not_play : ¬PlaySimple Emma := notPlay_of_notEntertain Emma Emma_not_entertain

-- From R1 (LearnedChords ∧ LearnedStrum → PlaySimple) and Emma_not_play, we get ¬(LearnedChords ∧ LearnedStrum) for Emma.
theorem Emma_not_chords_and_strum : ¬(LearnedChords Emma ∧ LearnedStrum Emma) := by
  intro hBoth
  have hPlay : PlaySimple Emma := R1 Emma hBoth
  exact Emma_not_entertain (R2 Emma hPlay)
