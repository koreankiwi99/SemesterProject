<lean>
axiom obj : Type
axiom Sarah : obj

axiom Practice : obj → Prop
axiom TechniqueImproved : obj → Prop
axiom LearnSongsFaster : obj → Prop
axiom ListenAttentively : obj → Prop

axiom R1 : ∀ x : obj, Practice x → TechniqueImproved x
axiom R2 : ∀ x : obj, TechniqueImproved x → LearnSongsFaster x
axiom disj : Practice Sarah ∨ ListenAttentively Sarah

theorem sarah_listens : (¬ LearnSongsFaster Sarah) → ListenAttentively Sarah := by
  intro hNotFast
  have hP_to_fast : Practice Sarah → LearnSongsFaster Sarah := by
    intro hP
    have hTech : TechniqueImproved Sarah := R1 Sarah hP
    exact R2 Sarah hTech
  have hNotPractice : ¬ Practice Sarah := by
    intro hP
    apply hNotFast
    exact hP_to_fast hP
  cases disj with
  | inl hP =>
      have f : False := hNotPractice hP
      exact False.elim f
  | inr hL =>
      exact hL
</lean>

ANSWER: Yes
