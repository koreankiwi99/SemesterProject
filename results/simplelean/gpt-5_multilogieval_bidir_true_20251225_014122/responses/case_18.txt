<lean>
axiom obj : Type
axiom Sam : obj

axiom Impressed : obj → Prop
axiom ExtraWork : obj → Prop
axiom Promotion : obj → Prop
axiom Experience : obj → Prop
axiom AssignedLead : obj → Prop

-- If someone impressed the manager, they get a promotion
axiom R1 : ∀ x : obj, Impressed x → Promotion x
-- If someone took on extra responsibilities, they gain experience
axiom R2 : ∀ x : obj, ExtraWork x → Experience x
-- If someone gains experience, they will be assigned to lead a new team
axiom R3 : ∀ x : obj, Experience x → AssignedLead x

-- Sam either impressed the manager or took on extra work
axiom Hdisj : Impressed Sam ∨ ExtraWork Sam
-- Sam did not get a promotion
axiom HnotPromo : ¬ Promotion Sam

theorem sam_assigned : AssignedLead Sam := by
  -- From ¬Promotion and (Impressed → Promotion), derive ¬Impressed
  have hNotImp : ¬ Impressed Sam := fun hImp => HnotPromo (R1 Sam hImp)
  -- From (Impressed ∨ ExtraWork) and ¬Impressed, derive ExtraWork
  have hExtra : ExtraWork Sam := by
    cases Hdisj with
    | inl hImp => exact False.elim (hNotImp hImp)
    | inr hEx => exact hEx
  -- Gain experience from extra work
  have hExp : Experience Sam := R2 Sam hExtra
  -- Assigned to lead a new team from experience
  exact R3 Sam hExp
</lean>

ANSWER: Yes
