<lean>
axiom obj : Type
axiom Emma : obj

axiom SpeaksDailyNative : obj → Prop
axiom FluencyImproved : obj → Prop
axiom VocabImproved : obj → Prop
axiom RealConversations : obj → Prop
axiom ListensPodcasts : obj → Prop
axiom ListeningComp : obj → Prop

-- If someone speaks daily with native speakers, their fluency and vocabulary improve
axiom R1 : ∀ x : obj, SpeaksDailyNative x → FluencyImproved x ∧ VocabImproved x
-- If fluency improves, they can have real conversations
axiom R2 : ∀ x : obj, FluencyImproved x → RealConversations x
-- Either they speak daily with native speakers or they listen to educational podcasts (or both)
axiom R3 : ∀ x : obj, SpeaksDailyNative x ∨ ListensPodcasts x
-- If someone listens to educational podcasts, they gain valuable listening comprehension
axiom R4 : ∀ x : obj, ListensPodcasts x → ListeningComp x

theorem emma_listening_from_not_conversations :
  (¬ RealConversations Emma) → ListeningComp Emma := by
  intro hNotRC
  -- From not having real conversations, deduce not speaking daily with natives
  have hNotSD : ¬ SpeaksDailyNative Emma := by
    intro hSD
    have hFIandVI := R1 Emma hSD
    have hFI : FluencyImproved Emma := And.left hFIandVI
    have hRC : RealConversations Emma := R2 Emma hFI
    exact hNotRC hRC
  -- Use the disjunction to conclude listening to podcasts, hence listening comprehension
  have hDisj : SpeaksDailyNative Emma ∨ ListensPodcasts Emma := R3 Emma
  cases hDisj with
  | inl hSD =>
      exact False.elim (hNotSD hSD)
  | inr hLP =>
      exact R4 Emma hLP
</lean>

ANSWER: Yes
