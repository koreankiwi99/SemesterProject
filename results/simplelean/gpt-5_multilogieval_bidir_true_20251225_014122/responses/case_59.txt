<lean>
axiom obj : Type
axiom Emma : obj

axiom LearnedBasicChords : obj → Prop
axiom LearnedStrumming : obj → Prop
axiom CanPlaySimpleSongs : obj → Prop
axiom CanEntertainFriends : obj → Prop
axiom TookLessons : obj → Prop
axiom KnowsMusicTheory : obj → Prop

-- If someone learned basic chords and strumming patterns, they can play simple songs.
axiom R1 : ∀ x : obj, (LearnedBasicChords x ∧ LearnedStrumming x) → CanPlaySimpleSongs x
-- If someone can play simple songs, they can entertain friends with the guitar.
axiom R2 : ∀ x : obj, CanPlaySimpleSongs x → CanEntertainFriends x
-- Either Emma learned basic chords or she took guitar lessons (or both).
axiom R3 : LearnedBasicChords Emma ∨ TookLessons Emma
-- If someone takes guitar lessons, they gain knowledge of music theory.
axiom R4 : ∀ x : obj, TookLessons x → KnowsMusicTheory x
-- Emma is unable to entertain friends with the guitar.
axiom Tneg : ¬ CanEntertainFriends Emma

-- From Tneg and R2 we can derive that Emma cannot play simple songs.
theorem not_play_simple_Emma : ¬ CanPlaySimpleSongs Emma := by
  intro hps
  have h_ent : CanEntertainFriends Emma := R2 Emma hps
  exact Tneg h_ent

-- From not being able to play simple songs and R1, we know Emma did not both learn basic chords and strumming.
theorem not_both_basic_and_strum_Emma : ¬ (LearnedBasicChords Emma ∧ LearnedStrumming Emma) := by
  intro hboth
  have hps : CanPlaySimpleSongs Emma := R1 Emma hboth
  have h_ent : CanEntertainFriends Emma := R2 Emma hps
  exact Tneg h_ent

-- We cannot derive TookLessons Emma (and thus KnowsMusicTheory Emma) from the given premises.
-- R3 only gives LearnedBasicChords Emma ∨ TookLessons Emma, but the left disjunct does not imply lessons or theory.
-- Therefore, KnowsMusicTheory Emma is not provable from the premises.
</lean>
