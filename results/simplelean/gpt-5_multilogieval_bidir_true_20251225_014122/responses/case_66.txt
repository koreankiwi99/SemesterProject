<lean>
axiom Company : Type
axiom Product : Type

axiom Acme : Company
axiom Gadget : Product

axiom Developed : Company → Product → Prop
axiom Innovative : Product → Prop
axiom MarketedEff : Company → Product → Prop
axiom Successful : Product → Prop

-- Rule: If a company develops an innovative product, then if they market it effectively, the product will be commercially successful.
axiom R : ∀ c p, (Developed c p ∧ Innovative p) → (MarketedEff c p → Successful p)

-- Given facts about Acme's gadget
axiom M_acme : MarketedEff Acme Gadget
axiom NS_gadget : ¬ Successful Gadget
axiom D_acme : Developed Acme Gadget

-- From the rule and facts, we can derive that Acme's gadget is not innovative.
theorem not_innovative_gadget : ¬ Innovative Gadget := by
  intro hI
  have hDI : Developed Acme Gadget ∧ Innovative Gadget := And.intro D_acme hI
  have hImp : MarketedEff Acme Gadget → Successful Gadget := R Acme Gadget hDI
  have hS : Successful Gadget := hImp M_acme
  exact NS_gadget hS

-- We cannot prove Innovative Gadget from the given axioms; in fact we proved its negation above.
</lean>
