<lean>
axiom person : Type
axiom exam : Type

axiom Emma : person
axiom BioExam : exam

axiom StudiesHard : person → Prop
axiom EnoughSleepBefore : person → exam → Prop
axiom PerformsWell : person → exam → Prop
axiom Aces : person → exam → Prop

-- Belief/rule: If someone studies hard, then if they get enough sleep before an exam,
-- they will perform well and ace the test (for that exam).
axiom Rule : ∀ x : person, ∀ e : exam, StudiesHard x → EnoughSleepBefore x e → PerformsWell x e ∧ Aces x e

-- Facts about Emma and her biology exam
axiom Sleep_Emma_Bio : EnoughSleepBefore Emma BioExam
axiom NotAce_Emma_Bio : ¬ Aces Emma BioExam

-- From the rule and facts, we can derive that Emma did NOT study hard.
theorem not_study_hard_Emma : ¬ StudiesHard Emma := by
  intro hHard
  have hPair : PerformsWell Emma BioExam ∧ Aces Emma BioExam := Rule Emma BioExam hHard Sleep_Emma_Bio
  have hAce : Aces Emma BioExam := And.right hPair
  exact NotAce_Emma_Bio hAce

-- Therefore, the statement "Emma studied hard" does not follow.
