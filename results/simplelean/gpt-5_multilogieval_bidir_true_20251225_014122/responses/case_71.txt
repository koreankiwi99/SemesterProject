<lean>
axiom obj : Type
axiom Alex : obj

axiom AppliedSunscreen : obj → Prop
axiom Burnt : obj → Prop
axiom Swim : obj → Prop
axiom Exercise : obj → Prop
axiom PoolWarm : Prop
axiom Sunny : Prop

-- If someone applies sunscreen, they do not get burnt
axiom R1 : ∀ x : obj, AppliedSunscreen x → ¬ Burnt x
-- If someone goes swimming, then they get exercise
axiom R2 : ∀ x : obj, Swim x → Exercise x
-- If the pool water is warm, people go swimming
axiom R3 : PoolWarm → ∀ x : obj, Swim x
-- If the weather was sunny, the pool water will be warm
axiom R4 : Sunny → PoolWarm
-- Either they (Alex) applied sunscreen, or they did not get any exercise
axiom R5 : AppliedSunscreen Alex ∨ ¬ Exercise Alex
-- It was a sunny day
axiom T1 : Sunny

-- Statement to evaluate: If Alex got sunburnt, then the weather was sunny
theorem alex_burnt_implies_sunny : Burnt Alex → Sunny := by
  intro _
  exact T1
</lean>

ANSWER: Yes
