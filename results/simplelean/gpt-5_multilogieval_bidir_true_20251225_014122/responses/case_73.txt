<lean>
axiom obj : Type
axiom Simone : obj

axiom TrainsWithDedication : obj → Prop
axiom AdequateRest : obj → Prop
axiom PeakPerformance : obj → Prop

-- If someone trains with dedication, then if they get adequate rest, they will perform at their peak.
axiom CoachesClaim : ∀ x : obj, TrainsWithDedication x → AdequateRest x → PeakPerformance x

-- Simone got plenty of rest.
axiom Rest_Simone : AdequateRest Simone
-- Simone did not play her best.
axiom NotPeak_Simone : ¬ PeakPerformance Simone

-- From the premises we can derive that Simone did not train with dedication.
theorem not_train_Simone : ¬ TrainsWithDedication Simone := by
  intro htrain
  have hpeak : PeakPerformance Simone := CoachesClaim Simone htrain Rest_Simone
  exact NotPeak_Simone hpeak
</lean>
