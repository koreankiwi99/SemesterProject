axiom PracticedHard : Person → Prop
axiom PlayedWell : Person → Prop
axiom PickedStart : Person → Prop
axiom Talented : Person → Prop
axiom NaturalAbility : Person → Prop
axiom TopYouth : Person → Prop

-- If someone practiced hard, they played well.
axiom R1 : ∀ x : Person, PracticedHard x → PlayedWell x
-- If someone is picked to start, they are talented.
axiom R2 : ∀ x : Person, PickedStart x → Talented x
-- Either they practiced hard, or they are not talented. (modeled universally)
axiom R3 : ∀ x : Person, PracticedHard x ∨ ¬ Talented x
-- If someone has natural ability, they get picked to start.
axiom R4 : ∀ x : Person, NaturalAbility x → PickedStart x
-- If someone was a top player in youth leagues, they have natural talent.
axiom R5 : ∀ x : Person, TopYouth x → NaturalAbility x

-- The asked statement ¬PlayedWell James → TopYouth James is not derivable.
-- In fact, from the given premises we can derive the opposite: ¬PlayedWell James → ¬TopYouth James.
theorem not_top_from_not_played : ¬PlayedWell James → ¬TopYouth James := by
  intro hNotPW
  intro hTop
  have hNat : NaturalAbility James := R5 James hTop
  have hPick : PickedStart James := R4 James hNat
  have hTal : Talented James := R2 James hPick
  have hOr : PracticedHard James ∨ ¬ Talented James := R3 James
  have hPH : PracticedHard James := by
    cases hOr with
    | inl hP => exact hP
    | inr hNotTal =>
        exact False.elim (hNotTal hTal)
  have hPW : PlayedWell James := R1 James hPH
  exact hNotPW hPW
