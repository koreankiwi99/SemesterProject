axiom obj : Type
axiom John : obj

axiom WorkedHard : obj → Prop
axiom Sore : obj → Prop
axiom HotShower : obj → Prop
axiom Relaxed : obj → Prop

axiom Cold : Prop
axiom Rain : Prop

axiom A1 : ∀ x : obj, WorkedHard x → Sore x
axiom A2 : ∀ x : obj, HotShower x → Relaxed x
axiom A3 : ∀ x : obj, WorkedHard x ∨ ¬Relaxed x
axiom A4 : Cold → ∀ x : obj, HotShower x
axiom A5 : Rain → Cold

theorem rain_implies_sore_john : Rain → Sore John := by
  intro hr
  have hc : Cold := A5 hr
  have hhs_all : ∀ x : obj, HotShower x := A4 hc
  have hhs : HotShower John := hhs_all John
  have hrel : Relaxed John := A2 John hhs
  have hdisj : WorkedHard John ∨ ¬Relaxed John := A3 John
  have hwh : WorkedHard John := by
    cases hdisj with
    | inl hw => exact hw
    | inr hnrel =>
      exact False.elim (hnrel hrel)
  exact A1 John hwh

theorem not_rain_of_not_sore_john : ¬ Sore John → ¬ Rain := by
  intro hns
  intro hr
  have hs : Sore John := rain_implies_sore_john hr
  exact hns hs
