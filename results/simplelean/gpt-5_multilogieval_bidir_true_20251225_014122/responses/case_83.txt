<lean>
axiom Sunny : Prop
axiom Park : Prop
axiom PracticeDaily : Prop
axiom Prepared : Prop
axiom Confident : Prop
axiom HeadUp : Prop

-- If the team practices every day, they will be prepared for the big game.
axiom R_practice_prepared : PracticeDaily → Prepared
-- If it is sunny outside, children will play at the park.
axiom R_sunny_park : Sunny → Park
-- Either the team practiced daily, or kids did not go to the park today.
axiom A_practice_or_notpark : PracticeDaily ∨ ¬ Park
-- If a team feels prepared, then they will have confidence.
axiom R_prepared_confident : Prepared → Confident
-- When people feel confident, they hold their head up high.
axiom R_confident_headup : Confident → HeadUp

theorem team_heads_high_if_sunny : Sunny → HeadUp := by
  intro s
  have park : Park := R_sunny_park s
  have pd : PracticeDaily := by
    cases A_practice_or_notpark with
    | inl hpd => exact hpd
    | inr hnpark =>
        have f : False := hnpark park
        exact False.elim f
  have prep : Prepared := R_practice_prepared pd
  have conf : Confident := R_prepared_confident prep
  exact R_confident_headup conf
</lean>

ANSWER: Yes
