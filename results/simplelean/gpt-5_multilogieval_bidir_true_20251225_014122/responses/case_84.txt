<lean>
axiom obj : Type
axiom Sarah : obj

-- Predicates
axiom PracticedNewMelodiesDaily : obj → Prop
axiom ExpandedRepertoire : obj → Prop
axiom MoreMaterial : obj → Prop
axiom EnoughSongsForAlbum : obj → Prop
axiom ComposedAlbum : obj → Prop
axiom ReleasedAlbumThisYear : obj → Prop
axiom JammedWithFriends : obj → Prop
axiom Inspired : obj → Prop
axiom CreativityFlourished : obj → Prop

-- Axioms from the context
-- If they practiced new melodies daily, they would expand their repertoire.
axiom A1 : ∀ x : obj, PracticedNewMelodiesDaily x → ExpandedRepertoire x
-- Interpret "had more material" as a consequence of expanded repertoire.
axiom A2 : ∀ x : obj, ExpandedRepertoire x → MoreMaterial x
-- If they had more material, they could produce enough songs for an album.
axiom A3 : ∀ x : obj, MoreMaterial x → EnoughSongsForAlbum x
-- Enough songs yields the ability to compose an album.
axiom A4 : ∀ x : obj, EnoughSongsForAlbum x → ComposedAlbum x
-- So if they practiced new melodies daily, they could compose an album. (derivable)
theorem compose_if_practice_Sarah : PracticedNewMelodiesDaily Sarah → ComposedAlbum Sarah := by
  intro hP
  have hE : ExpandedRepertoire Sarah := A1 Sarah hP
  have hM : MoreMaterial Sarah := A2 Sarah hE
  have hS : EnoughSongsForAlbum Sarah := A3 Sarah hM
  exact A4 Sarah hS

-- Either the musician practiced or jammed (or both). Instantiate to Sarah.
axiom A5 : PracticedNewMelodiesDaily Sarah ∨ JammedWithFriends Sarah

-- If someone jammed with friends, they gained inspiration from collaboration.
axiom A6 : ∀ x : obj, JammedWithFriends x → Inspired x
-- If they felt inspired, their creativity flourished.
axiom A7 : ∀ x : obj, Inspired x → CreativityFlourished x

-- Useful derived lemma: jamming implies creativity flourished (for Sarah)
theorem flourish_if_jam_Sarah : JammedWithFriends Sarah → CreativityFlourished Sarah := by
  intro hJ
  have hI : Inspired Sarah := A6 Sarah hJ
  exact A7 Sarah hI
