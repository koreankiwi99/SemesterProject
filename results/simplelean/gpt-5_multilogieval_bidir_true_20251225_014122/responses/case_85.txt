<lean>
axiom obj : Type
axiom John : obj

axiom Water : obj → Prop
axiom Bloom : obj → Prop
axiom Attract : obj → Prop
axiom Fertilizer : obj → Prop
axiom SoilHighQuality : obj → Prop
axiom HealthyPlants : obj → Prop

axiom A1 : ∀ x : obj, Water x → Bloom x
axiom A2 : ∀ x : obj, Bloom x → Attract x
axiom A3 : ∀ x : obj, Water x ∨ Fertilizer x
axiom A4 : ∀ x : obj, Fertilizer x → SoilHighQuality x
axiom A5 : ∀ x : obj, SoilHighQuality x → HealthyPlants x

theorem john_soil_high_quality : (¬ Attract John) → SoilHighQuality John := by
  intro hNotAttract
  have hWaterOrFert : Water John ∨ Fertilizer John := A3 John
  have hNotWater : ¬ Water John := by
    intro hW
    have hBloom : Bloom John := A1 John hW
    have hAttr : Attract John := A2 John hBloom
    exact hNotAttract hAttr
  exact Or.elim hWaterOrFert
    (fun hW => False.elim (hNotWater hW))
    (fun hF => A4 John hF)
</lean>

ANSWER: Yes
