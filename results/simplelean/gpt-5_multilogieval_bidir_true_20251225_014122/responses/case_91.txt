<lean>
axiom Noisy : Prop
axiom Party : Prop
axiom Drums : Prop
axiom Complain : Prop
axiom Police : Prop

-- If there is a party, then it is noisy.
axiom R1 : Party → Noisy
-- If someone is practicing drums, then it is noisy.
axiom R2 : Drums → Noisy
-- Either there is a party or someone is playing drums (or both).
axiom T1 : Party ∨ Drums
-- If the house gets noisy from a party, neighbors complain (modeled as: party implies complaints).
axiom R3 : Party → Complain
-- If neighbors complain, the police are called.
axiom R4 : Complain → Police

-- Statement to evaluate: If there was no noise level up, were the police called?
theorem no_noise_police : (¬ Noisy) → Police := by
  intro hNo
  -- From the disjunction and the implications, we derive Noisy.
  have hN : Noisy := Or.elim T1 (fun hp => R1 hp) (fun hd => R2 hd)
  -- This contradicts the assumption ¬Noisy.
  have contra : False := hNo hN
  -- From contradiction, derive Police.
  exact False.elim contra
</lean>

ANSWER: Yes
