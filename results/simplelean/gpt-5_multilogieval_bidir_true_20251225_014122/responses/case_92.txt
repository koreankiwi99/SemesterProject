<lean>
axiom EnoughWater : Prop
axiom GrowWell : Prop
axiom Pests : Prop
axiom Damage : Prop
axiom Bountiful : Prop
axiom Profit : Prop

axiom A1 : EnoughWater → GrowWell
axiom A2 : Pests → Damage
axiom A3 : (EnoughWater ∨ ¬Damage) ∨ (EnoughWater ∧ Damage)
axiom A4 : GrowWell → Bountiful
axiom A5 : Bountiful → Profit

theorem pests_profit : Pests → Profit :=
by
  intro p
  have d : Damage := A2 p
  have w_or_notd : EnoughWater ∨ ¬Damage :=
    Or.elim A3
      (fun h => h)
      (fun h => Or.inl h.left)
  have nnD : ¬¬Damage := fun nd => nd d
  have w : EnoughWater := Or.resolve_right w_or_notd nnD
  have g : GrowWell := A1 w
  have b : Bountiful := A4 g
  exact A5 b
</lean>

ANSWER: Yes
