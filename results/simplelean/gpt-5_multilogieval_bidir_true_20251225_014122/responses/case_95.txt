<lean>
axiom Person : Type
axiom Mary : Person

axiom ClearsUp : Prop
axiom GameOn : Prop
axiom FansCheer : Prop
axiom Exciting : Prop

axiom FinishesErrands : Person → Prop
axiom MeetFriends : Person → Prop

-- If the weather clears up, the baseball game will go on.
axiom A1 : ClearsUp → GameOn
-- If someone finishes their errands, they'll meet friends for dinner.
axiom A2 : ∀ x : Person, FinishesErrands x → MeetFriends x
-- Either it will clear up, or Mary's errands will get done (or both).
axiom A3 : ClearsUp ∨ FinishesErrands Mary
-- If the game is on, fans will get to cheer for their team.
axiom A4 : GameOn → FansCheer
-- If fans get to cheer, it will be an exciting night at the ballpark.
axiom A5 : FansCheer → Exciting

theorem exciting_if_mary_did_not_meet :
  (¬ MeetFriends Mary) → Exciting := by
  intro hNotMeet
  -- From ¬(Meet Mary) and (Errands Mary → Meet Mary), get ¬(Errands Mary)
  have hNotErr : ¬ FinishesErrands Mary := by
    intro hErr
    exact hNotMeet (A2 Mary hErr)
  -- From (ClearsUp ∨ Errands Mary) and ¬(Errands Mary), get ClearsUp
  have hClear : ClearsUp := Or.elim A3 (fun h => h) (fun hErr => False.elim (hNotErr hErr))
  -- Chain implications to Exciting
  exact A5 (A4 (A1 hClear))
</lean>

ANSWER: Yes
