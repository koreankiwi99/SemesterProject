-- P: Mark practiced speaking Spanish daily.
-- W: Mark watched Spanish TV shows.
-- C: Mark can converse with native speakers.
-- L: Mark's listening comprehension improved.
-- U: Mark can understand rapid conversational Spanish.

-- Assumptions derived from the context:
-- 1) P ∨ W
-- 2) P → C
-- 3) W → L
-- 4) L → U

-- Goal: ¬C → U

theorem understand_if_not_converse
  {P W C L U : Prop}
  (p_or_w : P ∨ W)
  (p_imp_c : P → C)
  (w_imp_l : W → L)
  (l_imp_u : L → U) :
  (¬ C) → U := by
  intro hNotC
  -- From ¬C and P → C, we get ¬P
  have hNotP : ¬ P := by
    intro hP
    exact hNotC (p_imp_c hP)
  -- From P ∨ W and ¬P, we get W
  have hW : W := Or.resolve_left p_or_w hNotP
  -- From W → L, we get L; then from L → U, we get U
  exact l_imp_u (w_imp_l hW)
