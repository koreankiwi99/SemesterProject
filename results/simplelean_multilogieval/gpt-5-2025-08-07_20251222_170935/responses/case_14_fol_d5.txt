theorem lauren_flexibility
  (S I F A R R M : Prop)
  (h1 : S → I)           -- if she did sprint workouts several times a week, speed improves
  (h2 : I → F)           -- if speed improves, she runs races faster
  (h3 : S ∨ A)           -- either she trained hard at sprinting (S) or she stretched after runs (A)
  (h4 : A → R)           -- if she stretched after runs, she took recovery seriously
  (h5 : R → M)           -- if she took recovery seriously, she focused on flexibility to prevent injury
  : (¬ F) → M := by
  intro notF
  -- From h1 and h2, S implies F
  have hSF : S → F := fun s => h2 (h1 s)
  -- From ¬F and S → F, we get ¬S
  have notS : ¬ S := fun s => notF (hSF s)
  -- From S ∨ A and ¬S, we deduce A
  have a : A := by
    cases h3 with
    | inl s => exact False.elim (notS s)
    | inr a => exact a
  -- Then R and finally M
  exact h5 (h4 a)
