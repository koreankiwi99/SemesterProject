-- Practice: the team practices hard
-- Prepared: the team is prepared for the big game
-- Injured: the star player is injured
-- CanPlay: the star player can play in the game
-- Wins: the team wins the important game
-- Dinner: the team celebrates with a team dinner afterwards

variable (Practice Prepared Injured CanPlay Wins Dinner : Prop)

-- Hypotheses from the context:
-- 1. If the team practices hard, they will be prepared for the big game.        (Practice → Prepared)
-- 2. If the star player is injured, he cannot play in the game.                 (Injured → ¬CanPlay)
-- 3. Either the team practiced hard, or the star player is not injured.         (Practice ∨ ¬Injured)
-- 4. When the team is well-prepared, the team wins the important game.          (Prepared → Wins)
-- 5. If the team wins the important game, they will celebrate with a dinner.    (Wins → Dinner)

theorem dinner_if_injured
  (h1 : Practice -> Prepared)
  (h2 : Injured -> ¬ CanPlay)
  (h3 : Practice ∨ ¬ Injured)
  (h4 : Prepared -> Wins)
  (h5 : Wins -> Dinner)
  : Injured -> Dinner := by
  intro hInj
  -- From Practice ∨ ¬Injured and Injured, derive Practice
  have hPrac : Practice :=
    Or.elim h3
      (fun hp => hp)
      (fun hNotInj => False.elim (hNotInj hInj))
  -- Chain implications to obtain Dinner
  have hPrep : Prepared := h1 hPrac
  have hWin  : Wins := h4 hPrep
  exact h5 hWin
