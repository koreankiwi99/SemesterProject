theorem farmer_profits
  {W D Dry S H P : Prop}
  (h1 : W → S)        -- If the farmer waters daily, crops grow strong and healthy.
  (h2 : D → Dry)      -- If there is a drought, the crops will dry up.
  (h3 : W ∨ ¬ Dry)    -- Either the farmer waters routinely, or the crops did not dry up.
  (h4 : S → H)        -- If the crops grow strong, there will be a bountiful harvest.
  (h5 : H → P)        -- If there is a large harvest, the farmer will make good profits.
  : D → P :=
by
  intro hD
  have hDry : Dry := h2 hD
  have hW : W := Or.elim h3
    (fun hw => hw)
    (fun hNotDry => False.elim (hNotDry hDry))
  have hS : S := h1 hW
  have hH : H := h4 hS
  exact h5 hH
