-- We model the statements as propositional variables.
variable (Practice Prepared Hot Postponed PlaysWell Win : Prop)

-- Hypotheses:
-- 1. If the team practices hard, they will be prepared for the game.
-- 2. If it is very hot outside, the game is postponed.
-- 3. Either the team practiced hard, or the game was not postponed.
-- 4. If a team is prepared, they will play well.
-- 5. When a team plays well, they will win.
-- Goal: If it was very hot outside, the team will win.
theorem hot_implies_win
  (h1 : Practice → Prepared)
  (h2 : Hot → Postponed)
  (h3 : Practice ∨ ¬ Postponed)
  (h4 : Prepared → PlaysWell)
  (h5 : PlaysWell → Win)
  : Hot → Win := by
  intro hHot
  have hPost : Postponed := h2 hHot
  have hPractice : Practice := by
    cases h3 with
    | inl hp => exact hp
    | inr hnp => exact False.elim (hnp hPost)
  have hPrep : Prepared := h1 hPractice
  have hPlay : PlaysWell := h4 hPrep
  exact h5 hPlay
</lean>
