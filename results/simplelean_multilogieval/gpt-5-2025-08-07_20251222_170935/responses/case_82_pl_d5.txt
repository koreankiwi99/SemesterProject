-- We formalize the statements as propositions:
-- Birthday: it's Alex's birthday
-- Bakes: mom bakes a cake
-- Oven375: the oven reaches 375°F
-- Cooked: the cake batter is cooked through properly
-- Fluffy: the cake turns out light and fluffy
-- Enjoy: Alex enjoys the cake

theorem enjoys_if_oven375
  (Birthday Bakes Oven375 Cooked Fluffy Enjoy : Prop)
  (h_bakes : Birthday → Bakes)                 -- If it's Alex's birthday, mom bakes a cake.
  (h_temp : Oven375 → Cooked)                  -- If oven reaches 375°F, the cake is cooked through.
  (h_either : Birthday ∨ ¬ Cooked)             -- Either it's Alex's birthday, or the cake is not cooked through.
  (h_fluffy : Cooked → Fluffy)                 -- If cooked through, then light and fluffy.
  (h_enjoy : Fluffy → Enjoy)                   -- If light and fluffy, Alex enjoys it.
  : Oven375 → Enjoy :=
by
  intro hOven
  have hCooked : Cooked := h_temp hOven
  have hFluffy : Fluffy := h_fluffy hCooked
  exact h_enjoy hFluffy
</lean>
