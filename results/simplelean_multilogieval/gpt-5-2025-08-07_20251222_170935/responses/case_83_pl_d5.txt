-- We model the scenario with propositional variables:
-- win: Alice won the race
-- trophy: Alice was awarded a trophy
-- celebrate: Alice celebrated a victory
-- icecream: Alice went out for ice cream
-- bigMeal: Bob ate a big meal before the race
-- stomachAche: Bob got a stomach ache during the race

theorem alice_icecream_if_bob_big_meal
  (win trophy celebrate icecream bigMeal stomachAche : Prop)
  (h1 : win → trophy)                 -- If Alice wins, she gets a trophy.
  (h2 : bigMeal → stomachAche)        -- If Bob ate a big meal, he gets a stomach ache.
  (h3 : win ∨ ¬ stomachAche)          -- Either Alice won, or Bob did not get a stomach ache.
  (h4 : trophy → celebrate)           -- If Alice was awarded a trophy, she celebrated her win.
  (h5 : celebrate → icecream)         -- Whenever Alice celebrated a victory, she went for ice cream.
  : bigMeal → icecream := by
  intro hMeal
  have hAche : stomachAche := h2 hMeal
  have hWin : win := by
    cases h3 with
    | inl hw => exact hw
    | inr hnot => exact False.elim (hnot hAche)
  have hTrophy : trophy := h1 hWin
  have hCelebrate : celebrate := h4 hTrophy
  exact h5 hCelebrate
</lean>
