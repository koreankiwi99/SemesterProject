-- We formalize the scenario with propositional variables.
-- Summer: it's summer vacation
-- SleepIn: kids sleep in late
-- Visiting: Grandma is visiting
-- Breakfast: Grandma made a big breakfast
-- Busy: the kitchen is busy
-- Beg: the dog is begging for scraps

variable (Summer SleepIn Visiting Breakfast Busy Beg : Prop)

-- Premises encoded as implications/disjunction from the story:
-- 1) If it's summer, the kids sleep in late.
-- 2) If Grandma visits, she makes breakfast.
-- 3) Either it's summer, or Grandma didn't make breakfast.
-- 4) If the kids sleep in late, the kitchen is busy.
-- 5) If the kitchen is busy, the dog begs.

theorem dog_begs_if_grandma_visits
  (h_summer_implies_sleep : Summer → SleepIn)
  (h_visiting_implies_breakfast : Visiting → Breakfast)
  (h_summer_or_not_breakfast : Summer ∨ ¬ Breakfast)
  (h_sleep_implies_busy : SleepIn → Busy)
  (h_busy_implies_beg : Busy → Beg)
  : Visiting → Beg := by
  intro hV
  -- From the disjunction, handle two cases.
  -- Case 1: It's summer. Then SleepIn → Busy → Beg.
  -- Case 2: Not Breakfast. But Visiting → Breakfast, contradiction; from False we can derive Beg.
  refine Or.elim h_summer_or_not_breakfast
    (fun hSummer =>
      h_busy_implies_beg (h_sleep_implies_busy (h_summer_implies_sleep hSummer)))
    (fun hNotBreakfast =>
      False.elim (hNotBreakfast (h_visiting_implies_breakfast hV)))
</lean>
