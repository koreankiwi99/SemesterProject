theorem graduation_implies_thrilled
  (OvenPreheated CakeBakesEvenly JohnsGraduationToday FamilyBakesCake CakeTastesDelicious JohnThrilled : Prop)
  (preheat_implies_even : OvenPreheated → CakeBakesEvenly)                 -- P → E
  (grad_implies_family_bakes : JohnsGraduationToday → FamilyBakesCake)     -- G → F
  (preheated_or_not_family_bakes : OvenPreheated ∨ ¬ FamilyBakesCake)      -- P ∨ ¬F
  (even_implies_tasty : CakeBakesEvenly → CakeTastesDelicious)             -- E → T
  (tasty_implies_thrilled : CakeTastesDelicious → JohnThrilled)            -- T → H
  : JohnsGraduationToday → JohnThrilled := by
  intro g
  have fam : FamilyBakesCake := grad_implies_family_bakes g
  have pre : OvenPreheated :=
    Or.elim preheated_or_not_family_bakes
      (fun hp => hp)
      (fun hnotfam => False.elim (hnotfam fam))
  have even : CakeBakesEvenly := preheat_implies_even pre
  have tasty : CakeTastesDelicious := even_implies_tasty even
  exact tasty_implies_thrilled tasty
