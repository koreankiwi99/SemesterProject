-- We model the statements as propositional variables.
variable (Studied Passed Valentines BoughtRoses Proud Smile : Prop)

-- Premises:
-- 1. If Mark studied hard, he would pass the test: Studied → Passed
-- 2. If it was Valentine's day, Mark would buy roses: Valentines → BoughtRoses
-- 3. Either Mark studied hard, or he did not buy roses: Studied ∨ ¬BoughtRoses
-- 4. If Mark passed his test, he would feel proud: Passed → Proud
-- 5. If Mark felt proud, he would have a big smile: Proud → Smile

theorem mark_smiles
  (h1 : Studied → Passed)
  (h2 : Valentines → BoughtRoses)
  (h3 : Studied ∨ ¬ BoughtRoses)
  (h4 : Passed → Proud)
  (h5 : Proud → Smile)
  : Valentines → Smile :=
by
  intro hv
  have r : BoughtRoses := h2 hv
  have s : Studied := by
    cases h3 with
    | inl hs => exact hs
    | inr hnotr =>
        have : False := hnotr r
        exact False.elim this
  have p : Passed := h1 s
  have pr : Proud := h4 p
  exact h5 pr
</lean>
