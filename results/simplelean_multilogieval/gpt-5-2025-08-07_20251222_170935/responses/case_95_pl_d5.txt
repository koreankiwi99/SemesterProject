theorem romantic_if_no_dinner
  {Anniversary TableAvail Dinner Flowers Loved Romantic : Prop}
  (hAF : Anniversary → Flowers)
  (hTD : TableAvail → Dinner)
  (hOr : Anniversary ∨ TableAvail)
  (hFL : Flowers → Loved)
  (hLR : Loved → Romantic)
  : ¬Dinner → Romantic := by
  intro hNoDinner
  cases hOr with
  | inl hAnn =>
      exact hLR (hFL (hAF hAnn))
  | inr hTable =>
      exact False.elim (hNoDinner (hTD hTable))
